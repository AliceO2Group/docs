308void GPUTPCGMMerger::DebugRefitMergedTrack(const GPUTPCGMMergedTrack& track) const
309{
310 GPUTPCGMMergedTrack trk = track;
311 GPUTrackingRefit refit;
312 ((GPUConstantMem*)GetConstantMem())->ioPtrs.mergedTrackHitStates = ClusterStateExt();
313 ((GPUConstantMem*)GetConstantMem())->ioPtrs.mergedTrackHits = Clusters();
314 refit.SetPtrsFromGPUConstantMem(GetConstantMem());
315 int32_t retval = refit.RefitTrackAsGPU(trk, false, true);
316 if (retval > 0) {
317 GPUTPCGMPropagator prop;
318 prop.SetMaterialTPC();
319 prop.SetPolynomialField(&Param().polynomialField);
320 prop.SetMaxSinPhi(GPUCA_MAX_SIN_PHI);
321 prop.SetPropagateBzOnly(false);
322 prop.SetMatLUT(Param().rec.useMatLUT ? GetConstantMem()->calibObjects.matLUT : nullptr);
323 prop.SetTrack(&trk.Param(), trk.GetAlpha());
324 int32_t err = prop.PropagateToXAlpha(track.GetParam().GetX(), track.GetAlpha(), false);
325 if (err == 0) {
326 printf("REFIT RESULT %d, SnpDiff %f\n", retval, trk.GetParam().GetSinPhi() - track.GetParam().GetSinPhi());
327 if (retval > 20 && fabsf(trk.GetParam().GetSinPhi() - track.GetParam().GetSinPhi()) > 0.01f) {
328 printf("LARGE DIFF\n");
329 }
330 } else {
331 printf("PROPAGATE ERROR\n");
332 }
333 } else {
334 printf("REFIT ERROR\n");
