58 void IntegratingMerger::run(framework::ProcessingContext& ctx)
59 {
60  // we have to avoid mistaking the timer input with data inputs.
61  auto* timerHeader = ctx.inputs().get("timer-publish").header;
62 
63  for (const DataRef& ref : InputRecordWalker(ctx.inputs())) {
64  if (ref.header != timerHeader) {
65  auto other = object_store_helpers::extractObjectFrom(ref);
66  merge(mMergedObjectLastCycle, std::move(other));
67  mDeltasMerged++;
68  }
69  }
70 
71  if (ctx.inputs().isValid("timer-publish")) {
72  finishCycle(ctx.outputs());
73  }
74 }

76 void IntegratingMerger::finishCycle(DataAllocator& outputs)
77 {
78  mCyclesSinceReset++;
79 
80  if (mConfig.publishMovingWindow.value == PublishMovingWindow::Yes) {
81  publishMovingWindow(outputs);
82  }
83 
84  if (!std::holds_alternative<std::monostate>(mMergedObjectLastCycle)) {
85  merge(mMergedObjectIntegral, std::move(mMergedObjectLastCycle));
86  }
87  mMergedObjectLastCycle = std::monostate{};
88  mTotalDeltasMerged += mDeltasMerged;
89 
90  publishIntegral(outputs);
91 
92  if (mConfig.mergedObjectTimespan.value == MergedObjectTimespan::LastDifference ||
93  mConfig.mergedObjectTimespan.value == MergedObjectTimespan::NCycles && mConfig.mergedObjectTimespan.param == mCyclesSinceReset) {
94  clear();
95  }
96 
97  mCollector->send({mTotalDeltasMerged, "total_deltas_merged"}, monitoring::DerivedMetricMode::RATE);
98  mCollector->send({mDeltasMerged, "deltas_merged_since_last_publication"});
99  mCollector->send({mCyclesSinceReset, "cycles_since_reset"});
100  mDeltasMerged = 0;
101 }

103 void IntegratingMerger::merge(ObjectStore& target, ObjectStore&& other)
104 {
105  if (std::holds_alternative<std::monostate>(target)) {
106  LOG(debug) << "Received the first input object in the run or after the last delta reset";
107  target = std::move(other);
108  other = std::monostate{};
109  } else if (std::holds_alternative<TObjectPtr>(target)) {
110  // We expect that if the first object was TObject, then all should.
111  auto targetAsTObject = std::get<TObjectPtr>(target);
112  auto otherAsTObject = std::get<TObjectPtr>(other);
113  algorithm::merge(targetAsTObject.get(), otherAsTObject.get());
114  } else if (std::holds_alternative<MergeInterfacePtr>(target)) {
115  // We expect that if the first object inherited MergeInterface, then all should.
116  auto otherAsMergeInterface = std::get<MergeInterfacePtr>(other);
117  std::get<MergeInterfacePtr>(target)->merge(otherAsMergeInterface.get());
118  } else if (std::holds_alternative<VectorOfTObjectPtrs>(target)) {
119  // We expect that if the first object was Vector of TObjects, then all should.
120  auto targetAsVector = std::get<VectorOfTObjectPtrs>(target);
121  const auto otherAsVector = std::get<VectorOfTObjectPtrs>(other);
122  algorithm::merge(targetAsVector, otherAsVector);
123  } else {
124  LOG(error) << "The target variant has an unrecognized value";
125  }
126 }

143 void IntegratingMerger::publishIntegral(framework::DataAllocator& allocator)
144 {
145  if (std::holds_alternative<std::monostate>(mMergedObjectIntegral)) {
146  LOG(info) << "No objects received since start or reset, nothing to publish";
147  } else if (object_store_helpers::snapshot(allocator, mSubSpec, mMergedObjectIntegral)) {
148  LOG(info) << "Published the merged object with " << mTotalDeltasMerged << " deltas in total,"
149  << " including " << mDeltasMerged << " in the last cycle.";
150  } else {
151  LOG(error) << "mMergedObjectIntegral' variant has an unrecognized value.";
152  }

155 void IntegratingMerger::publishMovingWindow(framework::DataAllocator& allocator)
156 {
157  if (std::holds_alternative<std::monostate>(mMergedObjectLastCycle)) {
158  LOG(info) << "No objects received since the last reset, no moving window to publish";
159  } else if (std::holds_alternative<MergeInterfacePtr>(mMergedObjectLastCycle)) {
160  // if there is MergeInterface, we can publish only the selected moving windows
161  if (auto mw = std::get<MergeInterfacePtr>(mMergedObjectLastCycle)->cloneMovingWindow()) {
162  allocator.snapshot({MergerBuilder::mergerMovingWindowOutputBinding(), 0}, *mw);
163  delete mw;
164  }
165  LOG(info) << "Published a moving window with " << mDeltasMerged << " deltas";
166  } else if (std::holds_alternative<TObjectPtr>(mMergedObjectLastCycle)) {
167  // if there is no MergeInterface, we just publish all deltas
168  allocator.snapshot(framework::OutputRef{MergerBuilder::mergerIntegralOutputBinding(), mSubSpec},
169  *std::get<TObjectPtr>(mMergedObjectLastCycle));
170  LOG(info) << "Published a moving window with " << mDeltasMerged << " deltas.";
171  } else if (std::holds_alternative<VectorOfTObjectPtrs>(mMergedObjectLastCycle)) {
172  const auto& mergedVector = std::get<VectorOfTObjectPtrs>(mMergedObjectLastCycle);
173  const auto vectorToSnapshot = object_store_helpers::toRawObserverPointers(mergedVector);
174  allocator.snapshot(framework::OutputRef{MergerBuilder::mergerIntegralOutputBinding(), mSubSpec}, vectorToSnapshot);
175  } else {
176  LOG(error) << "mMergedObjectIntegral' variant has an unrecognized value.";
177  }
178 }
