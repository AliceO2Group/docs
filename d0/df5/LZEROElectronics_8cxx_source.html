29 // Peak finding algorithm
30 //
31 // It checks if there is a rising trend on four consecutive Timebins
32 // If yes, then it compares the integral to a programmable threshold
33 bool LZEROElectronics::peakFinderOnPatch(TRUElectronics& p, unsigned int patchID)
34 {
35  auto& CurrentPatchTimeSum = p.mTimesum[patchID];
36  auto& TimeSums = std::get<1>(CurrentPatchTimeSum);
37  bool trendOfDigitsInTower = false;
38  if (TimeSums.size() < 4) {
39  return false;
40  // } else if (TimeSums[0] < TimeSums[1] && TimeSums[1] < TimeSums[2] && TimeSums[2] >= TimeSums[3]) {
41  } else if (TimeSums[3] < TimeSums[2] && TimeSums[2] < TimeSums[1] && TimeSums[1] >= TimeSums[0]) {
42  trendOfDigitsInTower = true;
43  } else if (TimeSums[0] > 0. || TimeSums[1] > 0. || TimeSums[2] > 0. || TimeSums[3] > 0.) {
44  }
45  double integralOfADCvalues = 0;
46  for (auto it = TimeSums.begin(); it != TimeSums.end(); it++) {
47  integralOfADCvalues += *it;
48  }
49  bool peakOverThreshold = false;
50  if (integralOfADCvalues > mThreshold && trendOfDigitsInTower) {
51  peakOverThreshold = true;
52  }
53  return peakOverThreshold;
54 }

56 // Peak finding algorithm on all patches
57 // It fills the mPeakFound vector with potential 1s
58 bool LZEROElectronics::peakFinderOnAllPatches(TRUElectronics& p)
59 {
60  bool isFoundGlobal = false;
61  p.mFiredPatches.clear();
62  for (auto& patches : p.mIndexMapPatch) {
63  auto PatchID = std::get<0>(patches);
64  auto isFound = peakFinderOnPatch(p, PatchID);
65  if (isFound) {
66  p.mFiredPatches.push_back(PatchID);
67  isFoundGlobal = true;
68  }
69  }
70  return isFoundGlobal;
71 }

90 void LZEROElectronics::addNoiseDigits(Digit& d1)
91 {
92  auto mSimParam = &(o2::emcal::SimParam::Instance());
93  double amplitude = d1.getAmplitude();
94  double sigma = mSimParam->getPinNoiseTRU();
95 
96  TRandom3 mRandomGenerator(std::chrono::high_resolution_clock::now().time_since_epoch().count());
97  uint16_t noise = std::floor(std::abs(mRandomGenerator.Gaus(0, sigma) / constants::EMCAL_TRU_ADCENERGY)); // ADC
98 
99  Digit d(d1.getTower(), 0., noise, d1.getTimeStamp());
100 
101  d1 += d;
102 }

105 {
106  int counterDigitTimeBin = 0;
107  int sizemDigitMap = -999;
108  TriggerMappingV2 mTriggerMap(mGeometry);
109 
110  for (auto& digitsTimeBin : digitlist) {
111  // Inside the DigitTimebinTRU
112  // Fill the LZEROElectronics with the new ADC value
113  // At the end of the loop run the peak finder
114  // Ship to LONEElectronics in case a peak is found
115  // Entire logic limited to timebin by timebin -> effectively implementing time scan
116 
117  counterDigitTimeBin++;
118 
119  for (auto& [fastor, digitsList] : *digitsTimeBin.mDigitMap) {
120  // Digit loop
121  // The peak finding algorithm is run after getting out of the loop!
122 
123  if (digitsList.size() == 0) {
124  continue;
125  }
126  digitsList.sort();
127 
128  int digIndex = 0;
129  Digit summedDigit;
130  bool first = true;
131  for (auto& ld : digitsList) {
132  if (first) {
133  summedDigit = ld;
134  first = false;
135  } else {
136  // summedDigit += ld;

138  // safety device in case same fastOr
139  // but different towers, i.e. remember
140  // that the += operator fails WITHOUT
141  // feedback if that were to happen
142  Digit digitToSum(summedDigit.getTower(), ld.getAmplitude(), summedDigit.getTimeStamp());
143  summedDigit += digitToSum;
144  }
145  }
146 
147  sizemDigitMap = (*digitsTimeBin.mDigitMap).size();
148  if (mSimulateNoiseDigits) {
149  addNoiseDigits(summedDigit);
150  }
151 
152  auto [whichTRU, whichFastOrTRU] = mTriggerMap.getTRUFromAbsFastORIndex(fastor);
153 
154  auto whichFastOr = std::get<1>(mTriggerMap.convertFastORIndexTRUtoSTU(whichTRU, whichFastOrTRU));
155  auto& patchTRU = patchesFromAllTRUs[whichTRU];
156  auto& fastOrPatchTRU = patchTRU.mFastOrs[whichFastOr];
157  fastOrPatchTRU.updateADC(summedDigit.getAmplitudeADC());
158 
159  digIndex++;
160  }
161 
162  // Evaluate -> peak finder (ALL TRUElectronics in ALL TRUs)
163  // in case peak found:
164  // - Create trigger input (IR of that timebin - delay [typically 8 or 9 samples - rollback (0)])
165  // - Create L1 timesums (trivial - last time integral) -> Collect from all fastOrs in ALL TRUs

168  // Propagating for now all the interaction record
169  // The typical delay is 8 to 9 BCs and the rollback as well
170  // The rollback is taken from EMCALReconstruction/RecoParam.h
171  // The delay is due to the interactions between EMCAL and CTP
172  // It accounts for the difference in times between L0a, L0b, and then there will be a L1 and L1b delay
173  // There is 1BC uncertainty on the trigger readout due to steps in the interaction between CTP and detector simulations
174  bool foundPeak = false;
175  for (auto& patches : patchesFromAllTRUs) {
176  updatePatchesADC(patches);
177  bool foundPeakCurrentTRU = peakFinderOnAllPatches(patches);
178  auto firedPatches = getFiredPatches(patches);
179  if (foundPeakCurrentTRU) {
180  foundPeak = true;
181  }
182  }
183 
184  if (foundPeak == true) {

187  EMCALTriggerInputs TriggerInputsForL1;
188  if (foundPeak) {
