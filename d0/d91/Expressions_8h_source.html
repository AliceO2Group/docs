81 template <typename T>
82 constexpr auto selectArrowType()
83 {
84  if constexpr (std::is_same_v<T, int>) {
85  return atype::INT32;
86  } else if constexpr (std::is_same_v<T, bool>) {
87  return atype::BOOL;
88  } else if constexpr (std::is_same_v<T, float>) {
89  return atype::FLOAT;
90  } else if constexpr (std::is_same_v<T, double>) {
91  return atype::DOUBLE;
92  } else if constexpr (std::is_same_v<T, uint8_t>) {
93  return atype::UINT8;
94  } else if constexpr (std::is_same_v<T, int8_t>) {
95  return atype::INT8;
96  } else if constexpr (std::is_same_v<T, uint16_t>) {
97  return atype::UINT16;
98  } else if constexpr (std::is_same_v<T, int16_t>) {
99  return atype::INT16;
100  } else if constexpr (std::is_same_v<T, int64_t>) {
101  return atype::INT64;
102  } else if constexpr (std::is_same_v<T, uint32_t>) {
103  return atype::UINT32;
104  } else if constexpr (std::is_same_v<T, uint64_t>) {
105  return atype::UINT64;
106  } else {
107  return atype::NA;
108  }
109  O2_BUILTIN_UNREACHABLE();
110 }
