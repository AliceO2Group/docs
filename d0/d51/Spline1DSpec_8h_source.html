228template <typename Spline1DContainer<DataT>::SafetyLevel SafeT>
229GPUdi() int32_t Spline1DContainer<DataT>::getLeftKnotIndexForU(DataT u) const
230{
233 int32_t iu = u < 0 ? 0 : (u > (float)mUmax ? mUmax : (int32_t)u);
234 if (SafeT == SafetyLevel::kSafe) {
235 iu = (iu < 0) ? 0 : (iu > mUmax ? mUmax : iu);
236 }
237 return getUtoKnotMap()[iu];
238}

296 template <SafetyLevel SafeT = SafetyLevel::kSafe>
297 GPUd() void interpolateU(int32_t inpYdim, GPUgeneric() const DataT Parameters[],
298 DataT u, GPUgeneric() DataT S[/*nYdim*/]) const
299 {
300 const auto nYdimTmp = SplineUtil::getNdim<YdimT>(inpYdim);
301 const auto nYdim = nYdimTmp.get();
302 int32_t iknot = TBase::template getLeftKnotIndexForU<SafeT>(u);
303 const DataT* d = Parameters + (2 * nYdim) * iknot;
304 interpolateU(nYdim, getKnots()[iknot], &(d[0]), &(d[nYdim]), &(d[2 * nYdim]), &(d[3 * nYdim]), u, S);
305 }
306
310 template <typename T>
311 GPUd() void interpolateU(int32_t inpYdim, const Knot& knotL,
312 GPUgeneric() const T Sl[/*mYdim*/], GPUgeneric() const T Dl[/*mYdim*/],
313 GPUgeneric() const T Sr[/*mYdim*/], GPUgeneric() const T Dr[/*mYdim*/],
314 DataT u, GPUgeneric() T S[/*mYdim*/]) const
315 {
316 const auto nYdimTmp = SplineUtil::getNdim<YdimT>(inpYdim);
317 const auto nYdim = nYdimTmp.get();
318 T uu = T(u - knotL.u);
319 T li = T(knotL.Li);
320 T v = uu * li; // scaled u
321 for (int32_t dim = 0; dim < nYdim; ++dim) {
322 T df = (Sr[dim] - Sl[dim]) * li;
323 T a = Dl[dim] + Dr[dim] - df - df;
324 T b = df - Dl[dim] - a;
325 S[dim] = ((a * v + b) * v + Dl[dim]) * uu + Sl[dim];
326 }
327 /*
328 another way to calculate f(u):
329 T uu = T(u - knotL.u);
330 T v = uu * T(knotL.Li); // scaled u
331 T vm1 = v-1;
332 T v2 = v * v;
333 float cSr = v2*(3-2*v);
334 float cSl = 1-cSr;
335 float cDl = v*vm1*vm1*knotL.L;
336 float cDr = v2*vm1*knotL.L;
337 return cSl*Sl + cSr*Sr + cDl*Dl + cDr*Dr;
338 */
339 }
340
341 template <typename T>
342 GPUd() static void getUderivatives(const Knot& knotL, DataT u,
343 T& dSl, T& dDl, T& dSr, T& dDr)
344 {
347 u = u - knotL.u;
348 T v = u * T(knotL.Li); // scaled u
349 T vm1 = v - 1.;
350 T a = u * vm1;
351 T v2 = v * v;
352 dSr = v2 * (3. - 2 * v);
353 dSl = 1. - dSr;
354 dDl = vm1 * a;
355 dDr = v * a;
356 // F(u) = dSl * Sl + dSr * Sr + dDl * Dl + dDr * Dr;
357 }
358
359 using TBase::convXtoU;
360 using TBase::getKnot;
361 using TBase::getKnots;
362 using TBase::getNumberOfKnots;
363
364 protected:
365 using TBase::mParameters;
366 using TBase::mYdim;
367 using TBase::TBase; // inherit constructors and hide them
368#ifndef GPUCA_ALIROOT_LIB
369 ClassDefNV(Spline1DSpec, 0);
370#endif
