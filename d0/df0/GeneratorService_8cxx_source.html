19 void GeneratorService::initService(std::string const& genName,
20  std::string const& triggerName,
21  VertexOption const& vxtOption)
22 {
23  auto localSimConfig = o2::conf::SimConfig::make();
24  localSimConfig.getConfigData().mGenerator = genName;
25  localSimConfig.getConfigData().mTrigger = triggerName;
26 
27  o2::eventgen::GeneratorFactory::setPrimaryGenerator(localSimConfig, &mPrimGen);
28 
29  // initialize vertexing based on type
30  if (dynamic_cast<MeanVertexObjectOption const*>(&vxtOption) != nullptr) {
31  auto ccdboption = dynamic_cast<MeanVertexObjectOption const*>(&vxtOption);
32  LOG(info) << "Init prim gen with MeanVertexObject";
33  // assign the mean vertex object
34  if (!ccdboption->meanVertexObject) {
35  LOG(fatal) << "No mean vertex object found - Cannot initialize Event generator";
36  }
37  mPrimGen.setVertexMode(o2::conf::VertexMode::kCCDB, ccdboption->meanVertexObject);
38  } else if (dynamic_cast<NoVertexOption const*>(&vxtOption) != nullptr) {
39  mPrimGen.setVertexMode(o2::conf::VertexMode::kNoVertex);
40  } else if (dynamic_cast<DiamondParamVertexOption const*>(&vxtOption) != nullptr) {
41  mPrimGen.setVertexMode(o2::conf::VertexMode::kDiamondParam);
42  } else {
43  LOG(error) << "Unknown VertexOption passed to Generator initialization";
44  }
45 
46  mStack.setExternalMode(true);
47  mPrimGen.Init();
48 }

50 void GeneratorService::generateEvent_MCTracks(std::vector<MCTrack>& tracks, o2::dataformats::MCEventHeader& header)
51 {
52  mPrimGen.SetEvent(&header);
53  mStack.Reset();
54  mPrimGen.GenerateEvent(&mStack); // this is the usual FairROOT interface going via stack
55 
56  tracks.reserve(mStack.getPrimaries().size());
57  for (auto& tparticle : mStack.getPrimaries()) {
58  tracks.emplace_back(tparticle);
59  }
60 }
