960 bool MatchITSTPCQC::refitV0(const o2::dataformats::V0Index& id, o2::dataformats::V0& v0, o2::globaltracking::RecoContainer& recoData)
961 {
962  if (!recoData.isTrackSourceLoaded(id.getProngID(0).getSource()) || !recoData.isTrackSourceLoaded(id.getProngID(1).getSource())) {
963  return false;
964  }
965  auto seedP = recoData.getTrackParam(id.getProngID(0));
966  auto seedN = recoData.getTrackParam(id.getProngID(1));
967  bool isTPConly = (id.getProngID(0).getSource() == o2::dataformats::GlobalTrackID::TPC) || (id.getProngID(1).getSource() == o2::dataformats::GlobalTrackID::TPC);
968  const auto& svparam = o2::vertexing::SVertexerParams::Instance();
969  if (svparam.mTPCTrackPhotonTune && isTPConly) {
970  mFitterV0.setMaxDZIni(svparam.mTPCTrackMaxDZIni);
971  mFitterV0.setMaxDXYIni(svparam.mTPCTrackMaxDXYIni);
972  mFitterV0.setMaxChi2(svparam.mTPCTrackMaxChi2);
973  mFitterV0.setCollinear(true);
974  }
975  int nCand = mFitterV0.process(seedP, seedN);
976  if (svparam.mTPCTrackPhotonTune && isTPConly) { // restore
977  // Reset immediately to the defaults
978  mFitterV0.setMaxDZIni(svparam.maxDZIni);
979  mFitterV0.setMaxDXYIni(svparam.maxDXYIni);
980  mFitterV0.setMaxChi2(svparam.maxChi2);
981  mFitterV0.setCollinear(false);
982  }
983  if (nCand == 0) { // discard this pair
984  return false;
985  }
986  const int cand = 0;
987  if (!mFitterV0.isPropagateTracksToVertexDone(cand) && !mFitterV0.propagateTracksToVertex(cand)) {
988  return false;
989  }
990  const auto& trPProp = mFitterV0.getTrack(0, cand);
991  const auto& trNProp = mFitterV0.getTrack(1, cand);
992  std::array<float, 3> pP{}, pN{};
993  trPProp.getPxPyPzGlo(pP);
994  trNProp.getPxPyPzGlo(pN);
995  std::array<float, 3> pV0 = {pP[0] + pN[0], pP[1] + pN[1], pP[2] + pN[2]};
996  auto p2V0 = pV0[0] * pV0[0] + pV0[1] * pV0[1] + pV0[2] * pV0[2];
997  const auto& pv = recoData.getPrimaryVertex(id.getVertexID());
998  const auto v0XYZ = mFitterV0.getPCACandidatePos(cand);
999  float dx = v0XYZ[0] - pv.getX(), dy = v0XYZ[1] - pv.getY(), dz = v0XYZ[2] - pv.getZ(), prodXYZv0 = dx * pV0[0] + dy * pV0[1] + dz * pV0[2];
1000  float cosPA = prodXYZv0 / std::sqrt((dx * dx + dy * dy + dz * dz) * p2V0);
1001  new (&v0) o2::dataformats::V0(v0XYZ, pV0, mFitterV0.calcPCACovMatrixFlat(cand), trPProp, trNProp);
1002  v0.setDCA(mFitterV0.getChi2AtPCACandidate(cand));
1003  v0.setCosPA(cosPA);
1004  return true;
1005 }
