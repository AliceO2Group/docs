79 idcDelta = (std::clamp(idcDelta, mCompressionFactor.mFactors.first, mCompressionFactor.mFactors.second) - mCompressionFactor.mFactors.first) * std::numeric_limits<DataT>::max() / (mCompressionFactor.mFactors.second - mCompressionFactor.mFactors.first);

419template <typename DataT>
420class IDCDeltaCompressionHelper
421{
422 public:
423 IDCDeltaCompressionHelper() = default;
424
428 static IDCDelta<DataT> getCompressedIDCs(const IDCDelta<float>& idcDeltaUncompressed)
429 {
430 const auto& paramIDCGroup = ParameterIDCCompression::Instance();
431
432 IDCDelta<DataT> idcCompressed{};
433 compress(idcDeltaUncompressed, idcCompressed, paramIDCGroup.minIDCDeltaValue, paramIDCGroup.maxIDCDeltaValue);
434 return idcCompressed;
435 }
436
440 static SACDelta<DataT> getCompressedSACs(const SACDelta<float>& sacDeltaUncompressed)
441 {
442 const auto& paramSAC = ParameterSAC::Instance();
443 SACDelta<DataT> sacCompressed{};
444 compress(sacDeltaUncompressed.mSACDelta[Side::A], sacCompressed.mSACDelta[Side::A], paramSAC.minSACDeltaValue, paramSAC.maxSACDeltaValue);
445 compress(sacDeltaUncompressed.mSACDelta[Side::C], sacCompressed.mSACDelta[Side::C], paramSAC.minSACDeltaValue, paramSAC.maxSACDeltaValue);
446 return sacCompressed;
447 }
448
449 private:
450 static void compress(const IDCDelta<float>& idcDeltaUncompressed, IDCDelta<DataT>& idcCompressed, const float minValue, const float maxValue)
451 {
452 if (idcDeltaUncompressed.getIDCDelta().empty()) {
453 return;
454 }
455 idcCompressed.getIDCDelta().reserve(idcDeltaUncompressed.getIDCDelta().size());
456 const auto minmaxIDC = std::minmax_element(std::begin(idcDeltaUncompressed.getIDCDelta()), std::end(idcDeltaUncompressed.getIDCDelta()));
457 const auto& paramIDCGroup = ParameterIDCCompression::Instance();
458 const float minIDCDelta = std::clamp(*minmaxIDC.first, minValue, maxValue);
459 const float maxIDCDelta = std::clamp(*minmaxIDC.second, minValue, maxValue);
460 idcCompressed.setCompressionFactor(minIDCDelta, maxIDCDelta);
461 for (auto& idc : idcDeltaUncompressed.getIDCDelta()) {
462 idcCompressed.emplace_back(idc);
463 }
464 }
