76  DataT compressValue(float idcDelta) const
77  {
78  idcDelta = (std::clamp(idcDelta, mCompressionFactor.mFactors.first, mCompressionFactor.mFactors.second) - mCompressionFactor.mFactors.first) * std::numeric_limits<DataT>::max() / (mCompressionFactor.mFactors.second - mCompressionFactor.mFactors.first);
79  return std::nearbyint(idcDelta);
80  }

84  float getValue(const unsigned int index) const { return mCompressionFactor.mFactors.first + (mCompressionFactor.mFactors.second - mCompressionFactor.mFactors.first) * static_cast<float>(mIDCDelta.mIDCDeltaCont[index]) / std::numeric_limits<DataT>::max(); }

390  static SACDelta<DataT> getCompressedSACs(const SACDelta<float>& sacDeltaUncompressed)
391  {
392  const auto& paramSAC = ParameterSAC::Instance();
393  SACDelta<DataT> sacCompressed{};
394  compress(sacDeltaUncompressed.mSACDelta[Side::A], sacCompressed.mSACDelta[Side::A], paramSAC.minSACDeltaValue, paramSAC.maxSACDeltaValue);
395  compress(sacDeltaUncompressed.mSACDelta[Side::C], sacCompressed.mSACDelta[Side::C], paramSAC.minSACDeltaValue, paramSAC.maxSACDeltaValue);
396  return sacCompressed;
397  }
398 
399  private:
400  static void compress(const IDCDelta<float>& idcDeltaUncompressed, IDCDelta<DataT>& idcCompressed, const float minValue, const float maxValue)
401  {
402  idcCompressed.getIDCDelta().reserve(idcDeltaUncompressed.getIDCDelta().size());
403  const auto minmaxIDC = std::minmax_element(std::begin(idcDeltaUncompressed.getIDCDelta()), std::end(idcDeltaUncompressed.getIDCDelta()));
404  const auto& paramIDCGroup = ParameterIDCCompression::Instance();
405  const float minIDCDelta = std::clamp(*minmaxIDC.first, minValue, maxValue);
406  const float maxIDCDelta = std::clamp(*minmaxIDC.second, minValue, maxValue);
407  idcCompressed.setCompressionFactor(minIDCDelta, maxIDCDelta);
408  for (auto& idc : idcDeltaUncompressed.getIDCDelta()) {
409  idcCompressed.emplace_back(idc);
410  }
411  }
412 };
