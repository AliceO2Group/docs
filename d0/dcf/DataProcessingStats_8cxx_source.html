33 void DataProcessingStats::updateStats(CommandSpec cmd)
34 {
35  if (metricSpecs[cmd.id].name.empty()) {
36  throw runtime_error_f("MetricID %d was not registered", (int)cmd.id);
37  }
38  if (metricSpecs[cmd.id].enabled == false) {
39  LOGP(debug, "MetricID {} is disabled", (int)cmd.id);
40  return;
41  }
42  if (cmd.id >= metrics.size()) {
43  throw runtime_error_f("MetricID %d is out of range", (int)cmd.id);
44  }
45  if (cmd.id >= updateInfos.size()) {
46  throw runtime_error_f("MetricID %d is out of range", (int)cmd.id);
47  }
48  pendingCmds++;
49  // Add a static mutex to protect the queue
50  // Get the next available operation in an atomic way.
51  auto idx = nextCmd.fetch_add(1, std::memory_order_relaxed);
52  if (idx == cmds.size()) {
53  // We abort this command
54  pendingCmds--;
55  while (pendingCmds.load(std::memory_order_relaxed) > 0) {
56  // We need to wait for all the pending commands to be processed.
57  // This is needed because we are going to flush the queue.
58  // We cannot flush the queue while there are pending commands
59  // as we might end up with a command being processed twice.
60  std::this_thread::sleep_for(std::chrono::milliseconds(1));
61  }
62  processCommandQueue();
63  insertedCmds.store(0, std::memory_order_relaxed);
64  nextCmd.store(0, std::memory_order_relaxed);
65  pendingCmds++;
66  idx = nextCmd.fetch_add(1, std::memory_order_relaxed);
67  } else if (idx > cmds.size()) {
68  while (cmds.size()) {
69  // We need to wait for the flushing of the queue
70  std::this_thread::sleep_for(std::chrono::milliseconds(1));
71  idx = nextCmd.load(std::memory_order_relaxed);
72  }
73  return updateStats(cmd);
74  }
75  // Save the command.
76  assert(idx < cmds.size());
77  assert(cmd.id < metrics.size());
78  cmds[idx] = Command{cmd.id, cmd.value, getTimestamp(realTimeBase, initialTimeOffset), cmd.op};
79  insertedCmds++;
80  pendingCmds--;
81  // Keep track of the number of commands we have received.
82  updatedMetricsLapse++;
83 }

85 void DataProcessingStats::processCommandQueue()
86 {
87  std::array<char, MAX_CMDS> order;
88  // The range cannot be larger than the number of commands we have.
89  auto range = insertedCmds.load(std::memory_order_relaxed);
90  if (insertedCmds.load(std::memory_order_relaxed) == 0) {
91  return;
92  }
93 
94  std::iota(order.begin(), order.begin() + range, 0);
95  // Shuffle the order in which we will process the commands based
96  // on their timestamp. If two commands are inserted at the same
97  // time, we expect to process them in the order they were inserted.
98  std::stable_sort(order.begin(), order.begin() + range, [this](char a, char b) {
99  return cmds[a].timestamp < cmds[b].timestamp;
100  });
101 
102  // Process the commands in the order we have just computed.
103  for (int i = 0; i < range; ++i) {
104  auto& cmd = cmds[i];
105  assert(cmd.id < updateInfos.size());
106  auto& update = updateInfos[cmd.id];
107  switch (cmd.op) {
108  case Op::Nop:
109  break;
110  case Op::Set:
111  if (cmd.value != metrics[cmd.id] && cmd.timestamp >= update.timestamp) {
112  metrics[cmd.id] = cmd.value;
113  updated[cmd.id] = true;
114  update.timestamp = cmd.timestamp;
115  pushedMetricsLapse++;
116  }
117  break;
118  case Op::Add:
119  if (cmd.value) {
120  metrics[cmd.id] += cmd.value;
121  updated[cmd.id] = true;
122  update.timestamp = cmd.timestamp;
123  pushedMetricsLapse++;
124  }
125  break;
126  case Op::Sub:
127  if (cmd.value) {
128  metrics[cmd.id] -= cmd.value;
129  updated[cmd.id] = true;
130  update.timestamp = cmd.timestamp;
131  pushedMetricsLapse++;
132  }
133  break;
134  case Op::Max:
135  if (cmd.value > metrics[cmd.id]) {
136  metrics[cmd.id] = cmd.value;
137  updated[cmd.id] = true;
138  update.timestamp = cmd.timestamp;
139  pushedMetricsLapse++;
140  }
141  break;
142  case Op::Min:
143  if (cmd.value < metrics[cmd.id]) {
144  metrics[cmd.id] = cmd.value;
145  updated[cmd.id] = true;
146  update.timestamp = cmd.timestamp;
147  pushedMetricsLapse++;
148  }
149  break;
150  case Op::SetIfPositive:
151  if (cmd.value > 0 && cmd.timestamp >= update.timestamp) {
152  metrics[cmd.id] = cmd.value;
153  updated[cmd.id] = true;
154  update.timestamp = cmd.timestamp;
155  pushedMetricsLapse++;
156  }
157  break;
158  case Op::InstantaneousRate: {
159  if (metricSpecs[cmd.id].kind != Kind::Rate) {
160  throw runtime_error_f("MetricID %d is not a rate", (int)cmd.id);
161  }
162  // We keep setting the value to the time average of the previous
163  // update period. so that we can compute the average over time
164  // at the moment of publishing.
165  metrics[cmd.id] = cmd.value;
166  updated[cmd.id] = true;
167  if (update.timestamp == 0) {
168  update.timestamp = cmd.timestamp;
169  }
170  pushedMetricsLapse++;
171  } break;
172  case Op::CumulativeRate: {
173  if (metricSpecs[cmd.id].kind != Kind::Rate) {
174  throw runtime_error_f("MetricID %d is not a rate", (int)cmd.id);
175  }
176  // We keep setting the value to the time average of the previous
177  // update period. so that we can compute the average over time
178  // at the moment of publishing.
179  metrics[cmd.id] += cmd.value;
180  updated[cmd.id] = true;
181  if (update.timestamp == 0) {
182  update.timestamp = cmd.timestamp;
183  }
184  pushedMetricsLapse++;
185  } break;
186  }
187  }
188  // No one should have tried to insert more commands while processing.
189  assert(range == insertedCmds.load(std::memory_order_relaxed));
190  nextCmd.store(0, std::memory_order_relaxed);
191  insertedCmds.store(0, std::memory_order_relaxed);
192 }

194 void DataProcessingStats::flushChangedMetrics(std::function<void(DataProcessingStats::MetricSpec const&, int64_t, int64_t)> const& callback)
195 {
196  publishingInvokedTotal++;
197  bool publish = false;
198  auto currentTimestamp = getTimestamp(realTimeBase, initialTimeOffset);
199  for (size_t ami = 0; ami < availableMetrics.size(); ++ami) {
200  int mi = availableMetrics[ami];
201  auto& update = updateInfos[mi];
202  MetricSpec& spec = metricSpecs[mi];
203  if (spec.name.empty()) {
204  continue;
205  }
206  if (spec.enabled == false) {
207  LOGP(debug, "Metric {} is disabled", spec.name);
208  continue;
209  }
210  if (updated[mi] == false && currentTimestamp - update.timestamp > spec.maxRefreshLatency) {
211  updated[mi] = true;
212  update.timestamp = currentTimestamp;
213  }
214  if (updated[mi] == false) {
215  continue;
216  }
217  if (currentTimestamp - update.lastPublished < spec.minPublishInterval) {
218  continue;
219  }
220  publish = true;
221  if (spec.kind == Kind::Unknown) {
222  LOGP(fatal, "Metric {} has unknown kind", spec.name);
223  }
224  if (spec.kind == Kind::Rate) {
225  if (currentTimestamp - update.timestamp == 0) {
226  callback(spec, update.timestamp, 0);
227  } else {
228  // Timestamp is in milliseconds, we want to convert to seconds.
229  callback(spec, update.timestamp, (1000 * (metrics[mi] - lastPublishedMetrics[mi])) / (currentTimestamp - update.timestamp));
230  }
231  update.timestamp = currentTimestamp; // We reset the timestamp to the current time.
232  } else {
233  callback(spec, update.timestamp, metrics[mi]);
234  }
235  lastPublishedMetrics[mi] = metrics[mi];
236  publishedMetricsLapse++;
237  update.lastPublished = currentTimestamp;
238  updated[mi] = false;
239  }
240  if (publish) {
241  publishingDoneTotal++;
242  }
243  static int64_t startTime = uv_hrtime();
244  int64_t now = uv_hrtime();
245 
246  auto timeDelta = std::max(int64_t(1), now - startTime); // min 1 unit of time to exclude division by 0
247  double averageInvocations = (publishingInvokedTotal * 1000000000) / timeDelta;
248  double averagePublishing = (publishedMetricsLapse * 1000000000) / timeDelta;
249 
250  LOGP(debug, "Publishing invoked {} times / s, {} metrics published / s", (int)averageInvocations, (int)averagePublishing);
251 }

253 void DataProcessingStats::registerMetric(MetricSpec const& spec)
254 {
255  if (spec.name.size() == 0) {
256  throw runtime_error("Metric name cannot be empty.");
257  }
258  if (spec.metricId >= metricSpecs.size()) {
259  throw runtime_error_f("Metric id %d is out of range. Max is %d", spec.metricId, metricSpecs.size());
260  }
261  if (metricSpecs[spec.metricId].name.size() != 0 && spec.name != metricSpecs[spec.metricId].name) {
262  auto currentName = metricSpecs[spec.metricId].name;
263  throw runtime_error_f("Metric %d already registered with name %s", spec.metricId, currentName.data(), spec.name.data());
264  }
265  auto currentMetric = std::find_if(metricSpecs.begin(), metricSpecs.end(), [&spec](MetricSpec const& s) { return s.name == spec.name && s.metricId != spec.metricId; });
266  if (currentMetric != metricSpecs.end()) {
267  throw runtime_error_f("Metric %s already registered with id %d. Cannot reregister with %d.", spec.name.data(), currentMetric->metricId, spec.metricId);
268  }
269  metricSpecs[spec.metricId] = spec;
270  metricsNames[spec.metricId] = spec.name;
271  metrics[spec.metricId] = spec.defaultValue;
272  int64_t currentTime = getTimestamp(realTimeBase, initialTimeOffset);
273  updateInfos[spec.metricId] = UpdateInfo{currentTime, currentTime};
274  updated[spec.metricId] = spec.sendInitialValue;
275  availableMetrics.push_back(spec.metricId);
276 }
