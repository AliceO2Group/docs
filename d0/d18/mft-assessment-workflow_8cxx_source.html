35WorkflowSpec defineDataProcessing(ConfigContext const& configcontext)
36{
37 // Update the (declared) parameters if changed from the command line
38 o2::conf::ConfigurableParam::updateFromString(configcontext.options().get<std::string>("configKeyValues"));
39 // write the configuration used for the workflow
40 o2::conf::ConfigurableParam::writeINI("o2-mft-assessment.ini");
41 auto useMC = !configcontext.options().get<bool>("disable-mc");
42 auto processGen = !configcontext.options().get<bool>("disable-process-gen");
43 auto finalizeAnalysis = configcontext.options().get<bool>("finalize-analysis");
44 auto useGeom = configcontext.options().get<bool>("use-full-geometry");
45
46 LOG(info) << "MFT Assessment: disable-mc = " << configcontext.options().get<std::string>("disable-mc");
47 LOG(info) << "MFT Assessment: disable-process-gen = " << configcontext.options().get<std::string>("disable-process-gen");
48 LOG(info) << "MFT Assessment: finalize-analysis = " << configcontext.options().get<std::string>("finalize-analysis");
49
50 WorkflowSpec specs;
51 specs.emplace_back(o2::mft::getMFTAssessmentSpec(useMC, useGeom, processGen, finalizeAnalysis));
52 return specs;
53}
