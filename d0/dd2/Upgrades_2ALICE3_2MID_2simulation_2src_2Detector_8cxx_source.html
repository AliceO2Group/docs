111void Detector::createGeometry()
112{
113 LOGP(info, "Creating MID geometry");
114
115 TGeoManager* geoManager = gGeoManager;
116 TGeoVolume* vALIC = geoManager->GetVolume("barrel");
117 if (!vALIC) {
118 LOGP(fatal, "Could not find barrel volume while constructing MID geometry");
119 }
120 new TGeoVolumeAssembly(GeometryTGeo::getMIDVolPattern());
121 TGeoVolume* vMID = geoManager->GetVolume(GeometryTGeo::getMIDVolPattern());
122 vALIC->AddNode(vMID, 2, new TGeoTranslation(0, 30., 0));
123
124 char vstrng[100] = "MIDVol";
125 vMID->SetTitle(vstrng);
126
127 // Build the MID
128 mLayers.resize(2);
129 auto& midParam = MIDBaseParam::Instance();
130 const bool standardRadius = (midParam.mLayout == o2::mi3::MIDLayout::StandardRadius);
131
132 if (standardRadius) {
133 mLayers[0] = MIDLayer(0, GeometryTGeo::composeSymNameLayer(0), 301.f, 500.f);
134 mLayers[1] = MIDLayer(1, GeometryTGeo::composeSymNameLayer(1), 311.f, 520.f); // arbitrarily reduced to get multiple of 5.2f
135 } else {
136 mLayers[0] = MIDLayer(0, GeometryTGeo::composeSymNameLayer(0), 266.f, 500.f);
137 mLayers[1] = MIDLayer(1, GeometryTGeo::composeSymNameLayer(1), 276.f, 520.f);
138 }
139
140 for (auto& layer : mLayers) {
141 layer.createLayer(vMID);
142 }

152bool Detector::ProcessHits(FairVolume* vol)
153{
154 // This method is called from the MC stepping
155 if (!(fMC->TrackCharge())) {
156 return false;
157 }
158
159 int lay = vol->getVolumeId();
160 int volID = vol->getMCid();
161
162 // Is it needed to keep a track reference when the outer ITS volume is encountered?
163 auto stack = (o2::data::Stack*)fMC->GetStack();
164 if (fMC->IsTrackExiting() && (lay == 0)) {
165 o2::TrackReference tr(*fMC, GetDetId());
166 tr.setTrackID(stack->GetCurrentTrackNumber());
167 tr.setUserId(lay);
168 stack->addTrackReference(tr);
169 }
170 bool startHit = false, stopHit = false;
171 unsigned char status = 0;
172 if (fMC->IsTrackEntering()) {
173 status |= Hit::kTrackEntering;
174 }
175 if (fMC->IsTrackInside()) {
176 status |= Hit::kTrackInside;
177 }
178 if (fMC->IsTrackExiting()) {
179 status |= Hit::kTrackExiting;
180 }
181 if (fMC->IsTrackOut()) {
182 status |= Hit::kTrackOut;
183 }
184 if (fMC->IsTrackStop()) {
185 status |= Hit::kTrackStopped;
186 }
187 if (fMC->IsTrackAlive()) {
188 status |= Hit::kTrackAlive;
189 }
190
191 // track is entering or created in the volume
192 if ((status & Hit::kTrackEntering) || (status & Hit::kTrackInside && !mTrackData.mHitStarted)) {
193 startHit = true;
194 } else if ((status & (Hit::kTrackExiting | Hit::kTrackOut | Hit::kTrackStopped))) {
195 stopHit = true;
196 }
197
198 // increment energy loss at all steps except entrance
199 if (!startHit) {
200 mTrackData.mEnergyLoss += fMC->Edep();
201 }
202 if (!(startHit | stopHit)) {
203 return false; // do noting
204 }
205
206 if (startHit) {
207 mTrackData.mEnergyLoss = 0.;
208 fMC->TrackMomentum(mTrackData.mMomentumStart);
209 fMC->TrackPosition(mTrackData.mPositionStart);
210 mTrackData.mTrkStatusStart = status;
211 mTrackData.mHitStarted = true;
212 }
213 if (stopHit) {
214 TLorentzVector positionStop;
215 fMC->TrackPosition(positionStop);
216 // Retrieve the indices with the volume path
217 int stave(0), halfstave(0), chipinmodule(0), module;
218 fMC->CurrentVolOffID(1, chipinmodule);
219 fMC->CurrentVolOffID(2, module);
220 fMC->CurrentVolOffID(3, halfstave);
221 fMC->CurrentVolOffID(4, stave);
222
223 Hit* p = addHit(stack->GetCurrentTrackNumber(), lay, mTrackData.mPositionStart.Vect(), positionStop.Vect(),
224 mTrackData.mMomentumStart.Vect(), mTrackData.mMomentumStart.E(), positionStop.T(),
225 mTrackData.mEnergyLoss, mTrackData.mTrkStatusStart, status);
226 // p->SetTotalEnergy(vmc->Etot());
227
228 // RS: not sure this is needed
229 // Increment number of Detector det points in TParticle
230 stack->addHit(GetDetId());
231 }
232
233 return true;
234}
