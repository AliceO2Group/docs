55 void RUDecodeData::fillChipStatistics(int icab, const ChipPixelData* chipData)
56 {
57  cableLinkPtr[icab]->chipStat.nHits += chipData->getData().size();
58  uint32_t action = 0;
59  if (chipData->getErrorFlags()) {
60  cableLinkPtr[icab]->chipStat.addErrors(*chipData, verbosity);
61  auto compid = ChipError::composeID(cableLinkPtr[icab]->feeID, chipData->getChipID());
62  auto& chErr = chipErrorsTF[compid];
63  chErr.first++;
64  chErr.second |= chipData->getErrorFlags();
65  }
66  if (action & ChipStat::ErrActDump) {
67  linkHBFToDump[(uint64_t(cableLinkPtr[icab]->subSpec) << 32) + cableLinkPtr[icab]->hbfEntry] = cableLinkPtr[icab]->irHBF.orbit;
68  }
69 }

73 void RUDecodeData::dumpcabledata(int icab)
74 {
75  const auto* cdat = cableData[icab].data();
76  size_t cdats = cableData[icab].getUnusedSize(), cdi = 0;
77  const auto lll = cableLinkPtr[icab];
78  LOGP(info, "RU#{} Cab{} Nbytes: {} IR: {} | {}", ruSWID, icab, cdats, lll->ir.asString(), lll->describe());
79  std::string dmp = "";
80  while (cdi < cdats) {
81  dmp += fmt::format(" {:#04x}", cdat[cdi++]);
82  if (cdi && (cdi % 9) == 0) {
83  LOGP(info, "wrd#{}: {}", cdi / 9 - 1, dmp);
84  dmp = "";
85  }
86  }
87  if (!dmp.empty()) {
88  LOGP(info, "wrd#{}: {}", cdi / 9 - 1, dmp);
89  }
90 }

94 bool RUDecodeData::checkLinkInSync(int icab, const o2::InteractionRecord ir)
95 {
96  auto* link = cableLinkPtr[icab];
97  if (link->ir == ir) {
98  link->rofJumpWasSeen = false;
99  return true;
100  }
101  if (link->statusInTF == GBTLink::CollectedDataStatus::None || link->statusInTF == GBTLink::CollectedDataStatus::StoppedOnEndOfData) { // link was not seen in this TF or data was over
102  return true;
103  }
104  // apparently there was desynchronization
105  if (link->ir > ir) {
106  link->rofJumpWasSeen = true;
107 #ifdef _RAW_READER_ERROR_CHECKS_
108  link->statistics.errorCounts[GBTLinkDecodingStat::ErrMissingROF]++;
109  linkHBFToDump[(uint64_t(link->subSpec) << 32) + link->hbfEntry] = link->irHBF.orbit;
110  if (link->needToPrintError(link->statistics.errorCounts[GBTLinkDecodingStat::ErrMissingROF])) {
111  LOGP(info, "{} (cable {}) has IR={} > current majority IR={} -> {}", link->describe(),
112  cableHWID[icab], link->ir.asString(), ir.asString(), link->statistics.ErrNames[GBTLinkDecodingStat::ErrMissingROF]);
113  }
114 #endif
115  } else { // link IR is behind the majority IR? In principle, this should never happen
116 #ifdef _RAW_READER_ERROR_CHECKS_
117  link->statistics.errorCounts[GBTLinkDecodingStat::ErrOldROF]++;
118  linkHBFToDump[(uint64_t(link->subSpec) << 32) + link->hbfEntry] = link->irHBF.orbit;
119  if (link->needToPrintError(link->statistics.errorCounts[GBTLinkDecodingStat::ErrOldROF]) && !ROFRampUpStage) {
120  LOGP(error, "{} (cable {}) has IR={} for current majority IR={} -> {}", link->describe(),
121  cableHWID[icab], link->ir.asString(), ir.asString(), link->statistics.ErrNames[GBTLinkDecodingStat::ErrOldROF]);
122  }
123 #endif
124  // clean data of cables of this link
125  for (int i = 0; i < ruInfo->nCables; i++) {
126  if (cableLinkPtr[i] == link) {
127  cableData[i].clear();
128  }
129  }
130  link->rofJumpWasSeen = false;
131  }
132  return false;
133 }
