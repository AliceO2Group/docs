396 } else if (iWay == nWays - 1) {

407 }
408 }

419 return false;
420 }

422 // TODO: we have looping tracks here with 0 accepted clusters in the primary leg. In that case we should refit the track using only the primary leg.
423
424 if (param.par.dodEdx && param.dodEdxDownscaled) {
425 dEdx.computedEdx(merger->OutputTracksdEdx()[iTrk], param);
426 }
427 Alpha = prop.GetAlpha();
428 MoveToReference(prop, param, Alpha);
429 NormalizeAlpha(Alpha);
430
431 return true;

573 return -1e6f;

577#ifndef GPUCA_TEXTURE_FETCH_CONSTRUCTOR

632#endif

641 }
642 }
643 }
