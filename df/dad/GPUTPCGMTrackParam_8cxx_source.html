190 goodRows++;

194 lastSector = cluster.sector;

197 CADEBUG(printf("\t%21sPropaga Alpha %8.3f , X %8.3f - Y %8.3f, Z %8.3f - QPt %7.2f (%7.2f), SP %5.2f (%5.2f) --- Res %8.3f %8.3f --- Cov sY %8.3f sZ %8.3f sSP %8.3f sPt %8.3f - YPt %8.3f - PErr %d", "", prop.GetAlpha(), mX, mP[0], mP[1], mP[4], prop.GetQPt0(), mP[2], prop.GetSinPhi0(), mP[0] - yy, mP[1] - zz, sqrtf(mC[0]), sqrtf(mC[2]), sqrtf(mC[5]), sqrtf(mC[14]), mC[10], retValProp));

277 CADEBUG(if (!CheckCov()) GPUError("INVALID COV AFTER UPDATE!!!"));
278 CADEBUG(printf("\t%21sFit Alpha %8.3f , X %8.3f - Y %8.3f, Z %8.3f - QPt %7.2f (%7.2f), SP %5.2f (%5.2f), DzDs %5.2f %16s --- Cov sY %8.3f sZ %8.3f sSP %8.3f sPt %8.3f - YPt %8.3f - FErr %d %d\n", "", prop.GetAlpha(), mX, mP[0], mP[1], mP[4], prop.GetQPt0(), mP[2], prop.GetSinPhi0(), mP[3], "", sqrtf(mC[0]), sqrtf(mC[2]), sqrtf(mC[5]), sqrtf(mC[14]), mC[10], retValUpd, retValInt));

283 {

285 covYYUpd = mC[0];
286 nMissed = nMissed2 = 0;
287 UnmarkClusters(clusters, ihitMergeFirst, ihit, wayDirection, GPUTPCGMMergedTrackHit::flagHighIncl);

301 const int32_t clusterCount = (ihit - ihitMergeFirst) * wayDirection + 1;

309 qtot /= clusterCount; // TODO: Weighted Average

311 relTime /= clusterCount;

315 }
316 if GPUCA_RTC_CONSTEXPR (GPUCA_GET_CONSTEXPR(param.rec.tpc, dEdxClusterRejectionFlagMask) != GPUCA_GET_CONSTEXPR(param.rec.tpc, dEdxClusterRejectionFlagMaskAlt)) {
317 if (accAlt) {
318 dEdxAlt.fillCluster(qtot, qmax, cluster.row, cluster.sector, mP[2], mP[3], merger->GetConstantMem()->calibObjects, zz, pad, relTime);
319 }
320 }
321 }
322 }
323 }
324 } else if (retValInt || retValUpd >= GPUTPCGMPropagator::updateErrorClusterRejected) { // cluster far away form the track
325 if (retValInt || allowChangeClusters) {
326 MarkClusters(clusters, ihitMergeFirst, ihit, wayDirection, GPUTPCGMMergedTrackHit::flagRejectDistance);
327 } else if (finalFit) {
328 MarkClusters(clusters, ihitMergeFirst, ihit, wayDirection, GPUTPCGMMergedTrackHit::flagRejectErr);
329 }
330 if (!retValInt) {
331 nMissed++;
332 nMissed2++;
333 }
334 } else {
335 break; // bad chi2 for the whole track, stop the fit
336 }
337 }
338 if (finalOutInFit && !(merger->Param().rec.tpc.disableRefitAttachment & 4) && lastRow != 255 && lastSector != 255) {
339 StoreLoopPropagation(merger, lastSector, lastRow, iTrk, lastRow > clusters[(iWay & 1) ? (maxN - 1) : 0].row, prop.GetAlpha());
340 CADEBUG(printf("\t\tSTORING %d lastRow %d row %d out %d\n", iTrk, (int)lastRow, (int)clusters[(iWay & 1) ? (maxN - 1) : 0].row, lastRow > clusters[(iWay & 1) ? (maxN - 1) : 0].row));
341 }
342 if (!(iWay & 1) && !finalFit && !track.CCE() && !track.Looper()) {
343 deltaZ = ShiftZ(clusters, merger, maxN);
344 } else {
345 deltaZ = 0.f;
346 }
347 }
348 ConstrainSinPhi();
349
350 GPUCA_DEBUG_STREAMER_CHECK(if (o2::utils::DebugStreamer::checkStream(o2::utils::StreamFlags::streamUpdateTrack, iTrk)) {
351 o2::utils::DebugStreamer::instance()->getStreamer("debug_accept_track", "UPDATE") << o2::utils::DebugStreamer::instance()->getUniqueTreeName("debug_accept_track").data() << "iTrk=" << iTrk << "outerParam=" << track.OuterParam() << "track=" << this << "ihitStart=" << ihitStart << "\n";
352 })
353
354 if (!(N + NTolerated >= GPUCA_TRACKLET_SELECTOR_MIN_HITS_B5(mP[4] * param.qptB5Scaler) && 2 * NTolerated <= CAMath::Max(10, N) && CheckNumericalQuality(covYYUpd))) {
355 return false; // TODO: NTolerated should never become that large, check what is going wrong!
356 }
357 if (param.rec.tpc.minNClustersFinalTrack != -1 && N + NTolerated < param.rec.tpc.minNClustersFinalTrack) {
358 return false;
359 }
360
361 if (param.par.dodEdx && param.dodEdxEnabled) {
362 dEdx.computedEdx(merger->MergedTracksdEdx()[iTrk], param);
363 if GPUCA_RTC_CONSTEXPR (GPUCA_GET_CONSTEXPR(param.rec.tpc, dEdxClusterRejectionFlagMask) != GPUCA_GET_CONSTEXPR(param.rec.tpc, dEdxClusterRejectionFlagMaskAlt)) {
364 dEdxAlt.computedEdx(merger->MergedTracksdEdxAlt()[iTrk], param);
365 }
366 }
367 Alpha = prop.GetAlpha();
368 MoveToReference(prop, param, Alpha);
369 NormalizeAlpha(Alpha);
370
371 return true;

374GPUdni() void GPUTPCGMTrackParam::MoveToReference(GPUTPCGMPropagator& prop, const GPUParam& param, float& Alpha)
375{
376 static constexpr float kDeg2Rad = M_PI / 180.f;
377 static constexpr float kSectAngle = 2 * M_PI / 18.f;
378
379 if (param.rec.tpc.trackReferenceX <= 500) {
380 GPUTPCGMTrackParam save = *this;
381 float saveAlpha = Alpha;
382 for (int32_t attempt = 0; attempt < 3; attempt++) {
383 float dAngle = CAMath::Round(CAMath::ATan2(mP[0], mX) / kDeg2Rad / 20.f) * kSectAngle;
384 Alpha += dAngle;
385 if (prop.PropagateToXAlpha(param.rec.tpc.trackReferenceX, Alpha, 0)) {
386 break;
387 }
388 ConstrainSinPhi();
389 if (CAMath::Abs(mP[0]) <= mX * CAMath::Tan(kSectAngle / 2.f)) {
390 return;
391 }
392 }
393 *this = save;
394 Alpha = saveAlpha;
395 }
396 if (CAMath::Abs(mP[0]) > mX * CAMath::Tan(kSectAngle / 2.f)) {
397 float dAngle = CAMath::Round(CAMath::ATan2(mP[0], mX) / kDeg2Rad / 20.f) * kSectAngle;
398 Rotate(dAngle);
399 ConstrainSinPhi();
400 Alpha += dAngle;
401 }

404GPUd() void GPUTPCGMTrackParam::MirrorTo(GPUTPCGMPropagator& GPUrestrict() prop, float toY, float toZ, bool inFlyDirection, const GPUParam& param, uint8_t row, uint8_t clusterState, bool mirrorParameters, int8_t sector)
405{
406 if (mirrorParameters) {
407 prop.Mirror(inFlyDirection);
408 }
409 float err2Y, err2Z;
410 prop.GetErr2(err2Y, err2Z, param, toZ, row, clusterState, sector, -1.f, 0.f, 0.f); // Use correct time / avgCharge
411 prop.Model().Y() = mP[0] = toY;

460 }
461 } else {
462 CADEBUG(printf("\t\tMerging hit row %d X %f Y %f Z %f (dy %f, dz %f, chiY %f, chiZ %f)\n", clusters[ihit].row, clx, cly, clz, dy, dz, sqrtf(maxDistY), sqrtf(maxDistZ)));

468 }

473 }

475 CADEBUG(printf("\t\tNo matching cluster in double-row, skipping\n"));
476 return -1;
477 }
478 xx /= count;
479 yy /= count;
480 zz /= count;
481 }
482 return 0;

497GPUd() float GPUTPCGMTrackParam::AttachClusters(const GPUTPCGMMerger* GPUrestrict() Merger, int32_t sector, int32_t iRow, int32_t iTrack, bool goodLeg, float Y, float Z)
498{
499 if (Merger->Param().rec.tpc.disableRefitAttachment & 1) {
500 return -1e6f;
501 }
502 const GPUTPCTracker& GPUrestrict() tracker = *(Merger->GetConstantMem()->tpcTrackers + sector);

504 GPUglobalref() const cahit2* hits = tracker.HitData(row);
505 GPUglobalref() const calink* firsthit = tracker.FirstHitInBin(row);
506 if (row.NHits() == 0) {

513 const float z0 = row.Grid().ZMin() - zOffset; // We can use our own ZOffset, since this is only used temporarily anyway

521 const float tubeY = CAMath::Sqrt(sy2);
522 const float tubeZ = CAMath::Sqrt(sz2);
523 const float sy21 = 1.f / sy2;
524 const float sz21 = 1.f / sz2;

526 Merger->GetConstantMem()->calibObjects.fastTransformHelper->InverseTransformYZtoNominalYZ(sector, iRow, Y, Z, uncorrectedY, uncorrectedZ);
527

530 }
531 row.Grid().GetBinArea(uncorrectedY, uncorrectedZ + zOffset, tubeY, tubeZ, bin, ny, nz);
532

540 }
541 for (int32_t k = 0; k <= nz; k++) {

561 }
562 }
563 }
564 return uncorrectedY;

668{
669 float toX = mX;
670 bool inFlyDirection = (Merger->MergedTracks()[iTrack].Leg() & 1) ^ up;
671
672 static constexpr float kSectAngle = 2 * M_PI / 18.f;
673 const GPUParam& GPUrestrict() param = Merger->Param();
674 bool right = (mP[2] < 0) ^ up;
675 const int32_t sectorSide = sector >= (GPUCA_NSECTORS / 2) ? (GPUCA_NSECTORS / 2) : 0;
676 float lrFactor = right ^ !up ? 1.f : -1.f;
677 // clang-format off
678 CADEBUG(printf("\nCIRCLE Track %d: Sector %d Alpha %f X %f Y %f Z %f SinPhi %f DzDs %f QPt %f - Right %d Up %d lrFactor %f\n", iTrack, sector, prop.GetAlpha(), mX, mP[0], mP[1], mP[2], mP[3], mP[4], (int32_t)right, (int32_t)up, lrFactor));

685 uint32_t maxTries = 100;
686 while (true) {
687 while (CAMath::Abs(mX) <= CAMath::Abs(mP[0]) * CAMath::Tan(kSectAngle / 2.f) + 0.1f) {

689 return;
690 }

693 }

698 return;
699 }
700 int32_t err = prop.PropagateToXAlpha(mX + (up ? 1.f : -1.f), prop.GetAlpha(), inFlyDirection);
701 if (err) {
702 CADEBUG(printf("\t\tpropagation error (%d)\n", err));
703 return;
704 }
705 CADEBUG(printf("\tPropagated to y = %f: X %f Z %f SinPhi %f\n", mX, mP[0], mP[1], mP[2]));
706 for (int32_t j = 0; j < GPUCA_ROW_COUNT; j++) { // TODO: Avoid iterating over all rows
707 float rowX = GPUTPCGeometry::Row2X(j);
708 if (CAMath::Abs(rowX - (-mP[0] * lrFactor)) < 1.5f) {
709 CADEBUG(printf("\t\tAttempt row %d (X %f Y %f Z %f)\n", j, rowX, mX * lrFactor, mP[1]));
710 AttachClusters(Merger, sector, j, iTrack, false, mX * lrFactor, mP[1]);
711 }
712 }
713 }
714 if (maxTries-- == 0) {
715 return;
716 }
717 if (right) {
718 if (++sector >= sectorSide + 18) {
719 sector -= 18;
720 }
721 } else {
722 if (--sector < sectorSide) {
723 sector += 18;
724 }
725 }
726 CADEBUG(printf("\tRotating to sector %d: %f --> %f\n", sector, prop.GetAlpha(), param.Alpha(sector) + (CAMath::Pi() / 2.f) * lrFactor));
727 int32_t err = prop.RotateToAlpha(param.Alpha(sector) + (CAMath::Pi() / 2.f) * lrFactor);
728 if (err) {
729 CADEBUG(printf("Rotation Error %d\n", err));
730 return;
731 }
732 CADEBUG(printf("\tAfter Rotating Alpha %f Position X %f Y %f Z %f SinPhi %f\n", prop.GetAlpha(), mP[0], mX, mP[1], mP[2]));
733 }

736GPUdi() void GPUTPCGMTrackParam::AttachClustersLooper(const GPUTPCGMMerger* GPUrestrict() Merger, int32_t sector, int32_t iRow, int32_t iTrack, bool outwards, GPUTPCGMPropagator& GPUrestrict() prop)
737{
738 static constexpr float kSectAngle = 2 * M_PI / 18.f;
739 // Note that the coordinate system is rotated by 90 degree swapping X and Y!
740 float X = mP[2] > 0 ? mP[0] : -mP[0];
741 float Y = mP[2] > 0 ? -mX : mX;
742 float Z = mP[1];
743 float SinPhi = CAMath::Sqrt(1 - mP[2] * mP[2]) * (mP[2] > 0 ? -1 : 1);
744 float b = prop.GetBz(prop.GetAlpha(), mX, mP[0], mP[1]);
745
746 float dx = outwards ? 1.f : -1.f;
747 const float myRowX = GPUTPCGeometry::Row2X(iRow);
748 // printf("\nAttachMirror sector %d row %d outwards %d\n", (int)sector, (int)iRow, (int)outwards);
749 // printf("X %f Y %f Z %f SinPhi %f -->\n", mX, mP[0], mP[1], mP[2]);
750 // printf("X %f Y %f Z %f SinPhi %f, dx %f\n", X, Y, Z, SinPhi, dx);
751 uint32_t maxTries = 100;
752 while (maxTries--) {
753 float ex = CAMath::Sqrt(1 - SinPhi * SinPhi);
754 float exi = 1.f / ex;
755 float dxBzQ = dx * -b * mP[4];
756 float newSinPhi = SinPhi + dxBzQ;
757 if (CAMath::Abs(newSinPhi) > GPUCA_MAX_SIN_PHI_LOW) {
758 // printf("Abort, newSinPhi %f\n", newSinPhi);
759 return;
760 }
761 if (mP[2] > 0 ? (newSinPhi > 0.5) : (newSinPhi < -0.5)) {
762 // printf("Finished, newSinPhi %f\n", newSinPhi);
763 return;
764 }
765 float dS = dx * exi;
766 float h2 = dS * exi * exi;
767 float h4 = .5f * h2 * dxBzQ;
768
769 X += dx;
770 Y += dS * SinPhi + h4;
771 Z += dS * mP[3];
772 SinPhi = newSinPhi;
773 if (CAMath::Abs(X) > CAMath::Abs(Y) * CAMath::Tan(kSectAngle / 2.f)) {
774 // printf("Abort, sector edge\n");
775 return;
776 }
777
778 // printf("count %d: At X %f Y %f Z %f SinPhi %f\n", maxTries, mP[2] > 0 ? -Y : Y, mP[2] > 0 ? X : -X, Z, SinPhi);
779 float paramX = mP[2] > 0 ? -Y : Y;
780 int32_t step = outwards ? 1 : -1;
781 int32_t found = 0;
782 for (int32_t j = iRow; j >= 0 && j < GPUCA_ROW_COUNT && found < 3; j += step) {
783 float rowX = mX + GPUTPCGeometry::Row2X(j) - myRowX;
784 if (CAMath::Abs(rowX - paramX) < 1.5f) {
785 // printf("Attempt row %d at y %f\n", j, X);
786 AttachClusters(Merger, sector, j, iTrack, false, mP[2] > 0 ? X : -X, Z);
787 }
788 }
789 }

803GPUd() float GPUTPCGMTrackParam::ShiftZ(const GPUTPCGMMerger* GPUrestrict() merger, int32_t sector, float cltmax, float cltmin, float clx)
804{
805 if (!merger->Param().par.continuousTracking) {
806 return 0.f;
807 }
808 float deltaZ = 0.f;
809 bool beamlineReached = false;
810 const float r1 = CAMath::Max(0.0001f, CAMath::Abs(mP[4] * merger->Param().polynomialField.GetNominalBz()));
811 if (r1 < 0.01501) { // 100 MeV @ 0.5T ~ 0.66m cutof
812 const float dist2 = mX * mX + mP[0] * mP[0];
813 const float dist1r2 = dist2 * r1 * r1;
814 if (dist1r2 < 4) {
815 const float alpha = CAMath::ACos(1 - 0.5f * dist1r2); // Angle of a circle, such that |(cosa, sina) - (1,0)| == dist
816 const float beta = CAMath::ATan2(mP[0], mX);
817 const int32_t comp = mP[2] > CAMath::Sin(beta);
818 const float sinab = CAMath::Sin((comp ? 0.5f : -0.5f) * alpha + beta); // Angle of circle through origin and track position, to be compared to Snp
819 const float res = CAMath::Abs(sinab - mP[2]);
820
821 if (res < 0.2) {
822 const float r = 1.f / r1;
823 const float dS = alpha * r;
824 float z0 = dS * mP[3];
825 if (CAMath::Abs(z0) > GPUTPCGeometry::TPCLength()) {
826 z0 = z0 > 0 ? GPUTPCGeometry::TPCLength() : -GPUTPCGeometry::TPCLength();
827 }
828 deltaZ = mP[1] - z0;
829 beamlineReached = true;
830
831 // printf("X %9.3f Y %9.3f QPt %9.3f R %9.3f --> Alpha %9.3f Snp %9.3f Snab %9.3f Res %9.3f dS %9.3f z0 %9.3f\n", mX, mP[0], mP[4], r, alpha / 3.1415 * 180, mP[2], sinab, res, dS, z0);
832 }
833 }
834 }
835
836 if (!beamlineReached) {
837 float refZ = ((sector < GPUCA_NSECTORS / 2) ? merger->Param().rec.tpc.defaultZOffsetOverR : -merger->Param().rec.tpc.defaultZOffsetOverR) * clx;
838 float basez;
839 merger->GetConstantMem()->calibObjects.fastTransformHelper->getCorrMap()->TransformIdealZ(sector, cltmax, basez, mTOffset);
840 deltaZ = basez - refZ;
