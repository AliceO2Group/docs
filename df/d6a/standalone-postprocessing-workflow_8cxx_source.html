59 WorkflowSpec defineDataProcessing(ConfigContext const& configcontext)
60 {
61  WorkflowSpec specs;
62  GID::mask_t srcTrc, srcCls;
63 
64  o2::conf::ConfigurableParam::updateFromString(configcontext.options().get<std::string>("configKeyValues"));
65  auto useMC = !configcontext.options().get<bool>("disable-mc");
66 
67  std::shared_ptr<o2::steer::MCKinematicsReader> mcKinematicsReader;
68  if (useMC) {
69  mcKinematicsReader = std::make_shared<o2::steer::MCKinematicsReader>("collisioncontext.root");
70  }
71  bool anyStudy{false};
72  // Declare specs related to studies hereafter
73  if (configcontext.options().get<bool>("impact-parameter-study")) {
74  anyStudy = true;
75  srcTrc = GID::getSourcesMask(configcontext.options().get<std::string>("track-sources"));
76  srcCls = GID::getSourcesMask(configcontext.options().get<std::string>("cluster-sources"));
77  o2::globaltracking::InputHelper::addInputSpecs(configcontext, specs, srcCls, srcTrc, srcTrc, useMC, srcCls, srcTrc);
78  // o2::globaltracking::InputHelper::addInputSpecsPVertex(configcontext, specs, useMC);
79  // o2::globaltracking::InputHelper::addInputSpecsSVertex(configcontext, specs);
80  specs.emplace_back(o2::its::study::getImpactParameterStudy(srcTrc, srcCls, useMC));
81  }
82  if (configcontext.options().get<bool>("cluster-size-study")) {
83  anyStudy = true;
84  srcTrc = GID::getSourcesMask(configcontext.options().get<std::string>("track-sources"));
85  srcCls = GID::getSourcesMask(configcontext.options().get<std::string>("cluster-sources"));
86  o2::globaltracking::InputHelper::addInputSpecs(configcontext, specs, srcCls, srcTrc, srcTrc, useMC, srcCls, srcTrc);
87  // o2::globaltracking::InputHelper::addInputSpecsPVertex(configcontext, specs, useMC);
88  // o2::globaltracking::InputHelper::addInputSpecsSVertex(configcontext, specs);
89  specs.emplace_back(o2::its::study::getAvgClusSizeStudy(srcTrc, srcCls, useMC, mcKinematicsReader));
90  }
91  if (configcontext.options().get<bool>("pid-study")) {
92  anyStudy = true;
93  srcTrc = GID::getSourcesMask(configcontext.options().get<std::string>("track-sources"));
94  srcCls = GID::getSourcesMask(configcontext.options().get<std::string>("cluster-sources"));
95  o2::globaltracking::InputHelper::addInputSpecs(configcontext, specs, srcCls, srcTrc, srcTrc, useMC, srcCls, srcTrc);
96  // o2::globaltracking::InputHelper::addInputSpecsPVertex(configcontext, specs, useMC);
97  // o2::globaltracking::InputHelper::addInputSpecsSVertex(configcontext, specs);
98  specs.emplace_back(o2::its::study::getPIDStudy(srcTrc, srcCls, useMC, mcKinematicsReader));
99  }
100  if (configcontext.options().get<bool>("track-study")) {
101  anyStudy = true;
102  srcTrc = GID::getSourcesMask(configcontext.options().get<std::string>("track-sources"));
103  srcCls = GID::getSourcesMask(configcontext.options().get<std::string>("cluster-sources"));
104  if (!configcontext.options().get<bool>("input-from-upstream")) {
105  o2::globaltracking::InputHelper::addInputSpecs(configcontext, specs, srcCls, srcTrc, srcTrc, useMC, srcCls, srcTrc);
106  }
107  specs.emplace_back(o2::its::study::getTrackCheckStudy(GID::getSourcesMask("ITS"), GID::getSourcesMask("ITS"), useMC, mcKinematicsReader));
108  // o2::globaltracking::InputHelper::addInputSpecsPVertex(configcontext, specs, useMC);
109  // o2::globaltracking::InputHelper::addInputSpecsSVertex(configcontext, specs);
110  }
111  if (configcontext.options().get<bool>("anomaly-study")) {
112  anyStudy = true;
113  // srcTrc = GID::getSourcesMask(configcontext.options().get<std::string>("track-sources"));
114  srcCls = GID::getSourcesMask(configcontext.options().get<std::string>("cluster-sources"));
115  if (!configcontext.options().get<bool>("input-from-upstream")) {
116  o2::globaltracking::InputHelper::addInputSpecs(configcontext, specs, srcCls, srcTrc, srcTrc, useMC, srcCls, srcTrc);
117  }
118  // o2::globaltracking::InputHelper::addInputSpecsPVertex(configcontext, specs, useMC);
119  // o2::globaltracking::InputHelper::addInputSpecsSVertex(configcontext, specs);
120  specs.emplace_back(o2::its::study::getAnomalyStudy(srcCls, useMC));
121  }
122  if (!anyStudy) {
123  LOGP(info, "No study selected, dryrunning");
124  }
125 
126  o2::raw::HBFUtilsInitializer hbfIni(configcontext, specs);
127  // write the configuration used for the studies workflow
128  o2::conf::ConfigurableParam::writeINI("o2_its_standalone_configuration.ini");
129 
130  return std::move(specs);
131 }
