129WorkflowSpec concat(T&& t, ARGS&&... args)
130{
131 if constexpr (sizeof...(args) == 0) {
132 return t;
133 } else {
134 auto rest = concat(std::forward<ARGS>(args)...);
135 // insert rest at the end of t
136 t.insert(t.end(), rest.begin(), rest.end());
137 return t;
138 }
