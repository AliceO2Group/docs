20 GeneratorHybrid::GeneratorHybrid(const std::string& inputgens)
21 {
22  if (!parseJSON(inputgens)) {
23  LOG(fatal) << "Failed to parse JSON configuration from input generators";
24  exit(1);
25  }
26  mRandomize = GeneratorHybridParam::Instance().randomize;
27  if (mConfigs.size() != mInputGens.size()) {
28  LOG(fatal) << "Number of configurations does not match the number of generators";
29  exit(1);
30  }
31  if (mConfigs.size() == 0) {
32  for (auto gen : mInputGens) {
33  mConfigs.push_back("");
34  }
35  }
36  int index = 0;
37  if (!mRandomize) {
38  if (mFractions.size() != mInputGens.size()) {
39  LOG(fatal) << "Number of fractions does not match the number of generators";
40  return;
41  }
42  // Check if all elements of mFractions are 0
43  if (std::all_of(mFractions.begin(), mFractions.end(), [](int i) { return i == 0; })) {
44  LOG(fatal) << "All fractions provided are 0, no simulation will be performed";
45  return;
46  }
47  }
48  for (auto gen : mInputGens) {
49  // Search if the generator name is inside generatorNames (which is a vector of strings)
50  LOG(info) << "Checking if generator " << gen << " is in the list of available generators \n";
51  if (std::find(generatorNames.begin(), generatorNames.end(), gen) != generatorNames.end()) {
52  LOG(info) << "Found generator " << gen << " in the list of available generators \n";
53  if (gen.compare("boxgen") == 0) {
54  if (mConfigs[index].compare("") == 0) {
55  gens.push_back(std::make_unique<o2::eventgen::BoxGenerator>());
56  } else {
57  // Get the index of boxgen configuration
58  int confBoxIndex = std::stoi(mConfigs[index].substr(7));
59  gens.push_back(std::make_unique<o2::eventgen::BoxGenerator>(*mBoxGenConfigs[confBoxIndex]));
60  }
61  mGens.push_back(gen);
62  } else if (gen.compare(0, 7, "pythia8") == 0) {
63  // Check if mConfigs[index] contains pythia8_ and a number
64  if (mConfigs[index].compare("") == 0) {
65  auto pars = Pythia8GenConfig();
66  gens.push_back(std::make_unique<o2::eventgen::GeneratorPythia8>(pars));
67  } else {
68  // Get the index of pythia8 configuration
69  int confPythia8Index = std::stoi(mConfigs[index].substr(8));
70  gens.push_back(std::make_unique<o2::eventgen::GeneratorPythia8>(*mPythia8GenConfigs[confPythia8Index]));
71  }
72  mConfsPythia8.push_back(mConfigs[index]);
73  mGens.push_back(gen);
74  } else if (gen.compare("extkinO2") == 0) {
75  int confO2KineIndex = std::stoi(mConfigs[index].substr(9));
76  gens.push_back(std::make_unique<o2::eventgen::GeneratorFromO2Kine>(*mO2KineGenConfigs[confO2KineIndex]));
77  mGens.push_back(gen);
78  } else if (gen.compare("evtpool") == 0) {
79  int confEvtPoolIndex = std::stoi(mConfigs[index].substr(8));
80  gens.push_back(std::make_unique<o2::eventgen::GeneratorFromEventPool>(mEventPoolConfigs[confEvtPoolIndex]));
81  mGens.push_back(gen);
82  } else if (gen.compare("external") == 0) {
83  int confextIndex = std::stoi(mConfigs[index].substr(9));
84  auto& extgen_filename = mExternalGenConfigs[confextIndex]->fileName;
85  auto& extgen_func = mExternalGenConfigs[confextIndex]->funcName;
86  auto extGen = std::unique_ptr<o2::eventgen::Generator>(o2::conf::GetFromMacro<o2::eventgen::Generator*>(extgen_filename, extgen_func, "FairGenerator*", "extgen"));
87  if (!extGen) {
88  LOG(fatal) << "Failed to load external generator from " << extgen_filename << " with function " << extgen_func;
89  exit(1);
90  }
91  gens.push_back(std::move(extGen));
92  mGens.push_back(gen);
93  } else if (gen.compare("hepmc") == 0) {
94  int confHepMCIndex = std::stoi(mConfigs[index].substr(6));
95  gens.push_back(std::make_unique<o2::eventgen::GeneratorHepMC>());
96  auto& globalConfig = o2::conf::SimConfig::Instance();
97  dynamic_cast<o2::eventgen::GeneratorHepMC*>(gens.back().get())->setup(*mFileOrCmdGenConfigs[confHepMCIndex], *mHepMCGenConfigs[confHepMCIndex], globalConfig);
98  mGens.push_back(gen);
99  }
100  } else {
101  LOG(fatal) << "Generator " << gen << " not found in the list of available generators \n";
102  exit(1);
103  }
104  index++;
105  }
106 }

108 Bool_t GeneratorHybrid::Init()
109 {
110  // init all sub-gens
111  int count = 0;
112  for (auto& gen : mGens) {
113  if (gen == "pythia8pp") {
114  auto config = std::string(std::getenv("O2_ROOT")) + "/share/Generators/egconfig/pythia8_inel.cfg";
115  LOG(info) << "Setting \'Pythia8\' base configuration: " << config << std::endl;
116  dynamic_cast<o2::eventgen::GeneratorPythia8*>(gens[count].get())->setConfig(config);
117  } else if (gen == "pythia8hf") {
118  auto config = std::string(std::getenv("O2_ROOT")) + "/share/Generators/egconfig/pythia8_hf.cfg";
119  LOG(info) << "Setting \'Pythia8\' base configuration: " << config << std::endl;
120  dynamic_cast<o2::eventgen::GeneratorPythia8*>(gens[count].get())->setConfig(config);
121  } else if (gen == "pythia8hi") {
122  auto config = std::string(std::getenv("O2_ROOT")) + "/share/Generators/egconfig/pythia8_hi.cfg";
123  LOG(info) << "Setting \'Pythia8\' base configuration: " << config << std::endl;
124  dynamic_cast<o2::eventgen::GeneratorPythia8*>(gens[count].get())->setConfig(config);
125  } else if (gen == "pythia8powheg") {
126  auto config = std::string(std::getenv("O2_ROOT")) + "/share/Generators/egconfig/pythia8_powheg.cfg";
127  LOG(info) << "Setting \'Pythia8\' base configuration: " << config << std::endl;
128  dynamic_cast<o2::eventgen::GeneratorPythia8*>(gens[count].get())->setConfig(config);
129  }
130  gens[count]->Init();
131  addSubGenerator(count, gen);
132  count++;
133  }
134  if (mRandomize) {
135  if (std::all_of(mFractions.begin(), mFractions.end(), [](int i) { return i == 1; })) {
136  LOG(info) << "Full randomisation of generators order";
137  } else {
138  LOG(info) << "Randomisation based on fractions";
139  int allfracs = 0;
140  for (auto& f : mFractions) {
141  allfracs += f;
142  }
143  // Assign new rng fractions
144  float sum = 0;
145  float chance = 0;
146  for (int k = 0; k < mFractions.size(); k++) {
147  if (mFractions[k] == 0) {
148  // Generator will not be used if fraction is 0
149  mRngFractions.push_back(-1);
150  LOG(info) << "Generator " << mGens[k] << " will not be used";
151  } else {
152  chance = static_cast<float>(mFractions[k]) / allfracs;
153  sum += chance;
154  mRngFractions.push_back(sum);
155  LOG(info) << "Generator " << (mConfigs[k] == "" ? mGens[k] : mConfigs[k]) << " has a " << chance * 100 << "% chance of being used";
156  }
157  }
158  }
159  } else {
160  LOG(info) << "Generators will be used in sequence, following provided fractions";
161  }
162  return Generator::Init();
163 }

165 Bool_t GeneratorHybrid::generateEvent()
166 {
167  // Order randomisation or sequence of generators
168  // following provided fractions. If not available generators will be used sequentially
169  if (mRandomize) {
170  if (mRngFractions.size() != 0) {
171  // Generate number between 0 and 1
172  float rnum = gRandom->Rndm();
173  // Find generator index
174  for (int k = 0; k < mRngFractions.size(); k++) {
175  if (rnum <= mRngFractions[k]) {
176  mIndex = k;
177  break;
178  }
179  }
180  } else {
181  mIndex = gRandom->Integer(mGens.size());
182  }
183  } else {
184  while (mFractions[mCurrentFraction] == 0 || mseqCounter == mFractions[mCurrentFraction]) {
185  if (mFractions[mCurrentFraction] != 0) {
186  mseqCounter = 0;
187  }
188  mCurrentFraction = (mCurrentFraction + 1) % mFractions.size();
189  }
190  mIndex = mCurrentFraction;
191  }
192  if (mConfigs[mIndex].compare("") == 0) {
193  LOG(info) << "GeneratorHybrid: generating event with generator " << mGens[mIndex];
194  } else {
195  LOG(info) << "GeneratorHybrid: generating event with generator " << mConfigs[mIndex];
196  }
197  gens[mIndex]->clearParticles(); // clear container of this class
198  gens[mIndex]->generateEvent();
199  // notify the sub event generator
200  notifySubGenerator(mIndex);
201  mseqCounter++;
202  return true;
203 }

216  }

230 Bool_t GeneratorHybrid::parseJSON(const std::string& path)
231 {
232  // Parse JSON file to build map
233  std::ifstream fileStream(path, std::ios::in);
234  if (!fileStream.is_open()) {
235  LOG(error) << "Cannot open " << path;
236  return false;
237  }
238  rapidjson::IStreamWrapper isw(fileStream);
239  rapidjson::Document doc;
240  doc.ParseStream(isw);
241  if (doc.HasParseError()) {
242  LOG(error) << "Error parsing provided json file " << path;
243  LOG(error) << " - Error -> " << rapidjson::GetParseError_En(doc.GetParseError());
244  return false;
245  }
246 
247  // Put the generator names in mInputGens
248  if (doc.HasMember("generators")) {
249  const auto& gens = doc["generators"];
250  for (const auto& gen : gens.GetArray()) {
251  // push in mInputGens the "name" of the generator
252  std::string name = gen["name"].GetString();
253  mInputGens.push_back(name);
254  if (gen.HasMember("config")) {
255  if (name == "boxgen") {
256  const auto& boxconf = gen["config"];
257  auto boxConfig = TBufferJSON::FromJSON<o2::eventgen::BoxGenConfig>(jsonValueToString(boxconf).c_str());
258  mBoxGenConfigs.push_back(std::move(boxConfig));
259  mConfigs.push_back("boxgen_" + std::to_string(mBoxGenConfigs.size() - 1));
260  continue;
261  } else if (name == "pythia8") {
262  const auto& pythia8conf = gen["config"];
263  auto pythia8Config = TBufferJSON::FromJSON<o2::eventgen::Pythia8GenConfig>(jsonValueToString(pythia8conf).c_str());
264  mPythia8GenConfigs.push_back(std::move(pythia8Config));
265  mConfigs.push_back("pythia8_" + std::to_string(mPythia8GenConfigs.size() - 1));
266  continue;
267  } else if (name == "extkinO2") {
268  const auto& o2kineconf = gen["config"];
269  auto o2kineConfig = TBufferJSON::FromJSON<o2::eventgen::O2KineGenConfig>(jsonValueToString(o2kineconf).c_str());
270  mO2KineGenConfigs.push_back(std::move(o2kineConfig));
271  mConfigs.push_back("extkinO2_" + std::to_string(mO2KineGenConfigs.size() - 1));
272  continue;
273  } else if (name == "evtpool") {
274  const auto& o2kineconf = gen["config"];
275  auto poolConfig = TBufferJSON::FromJSON<o2::eventgen::EventPoolGenConfig>(jsonValueToString(o2kineconf).c_str());
276  mEventPoolConfigs.push_back(*poolConfig);
277  mConfigs.push_back("evtpool_" + std::to_string(mEventPoolConfigs.size() - 1));
278  continue;
279  } else if (name == "external") {
280  const auto& extconf = gen["config"];
281  auto extConfig = TBufferJSON::FromJSON<o2::eventgen::ExternalGenConfig>(jsonValueToString(extconf).c_str());
282  mExternalGenConfigs.push_back(std::move(extConfig));
283  mConfigs.push_back("external_" + std::to_string(mExternalGenConfigs.size() - 1));
284  continue;
285  } else if (name == "hepmc") {
286  const auto& genconf = gen["config"];
287  const auto& cmdconf = genconf["configcmd"];
288  const auto& hepmcconf = genconf["confighepmc"];
289  auto cmdConfig = TBufferJSON::FromJSON<o2::eventgen::FileOrCmdGenConfig>(jsonValueToString(cmdconf).c_str());
290  auto hepmcConfig = TBufferJSON::FromJSON<o2::eventgen::HepMCGenConfig>(jsonValueToString(hepmcconf).c_str());
291  mFileOrCmdGenConfigs.push_back(std::move(cmdConfig));
292  mHepMCGenConfigs.push_back(std::move(hepmcConfig));
293  mConfigs.push_back("hepmc_" + std::to_string(mFileOrCmdGenConfigs.size() - 1));
294  continue;
295  } else {
296  mConfigs.push_back("");
297  }
298  } else {
299  if (name == "boxgen" || name == "pythia8" || name == "extkinO2" || name == "external" || name == "hepmc") {
300  LOG(fatal) << "No configuration provided for generator " << name;
301  return false;
302  } else {
303  mConfigs.push_back("");
304  }
305  }
306  }
307  }
308 
309  // Get fractions and put them in mFractions
310  if (doc.HasMember("fractions")) {
311  const auto& fractions = doc["fractions"];
312  for (const auto& frac : fractions.GetArray()) {
313  mFractions.push_back(frac.GetInt());
314  }
315  } else {
316  // Set fractions to unity for all generators in case they are not provided
317  const auto& gens = doc["generators"];
318  for (const auto& gen : gens.GetArray()) {
319  mFractions.push_back(1);
320  }
321  }
322  return true;
323 }
