38 TimebinsPerCacheline = TPCMapMemoryLayout<uint16_t>::Height,
39 EntriesPerCacheline = PadsPerCacheline * TimebinsPerCacheline,
40 NumOfCachedPads = GPUCA_WARP_SIZE / TimebinsPerCacheline,
41 NumCLsPerWarp = GPUCA_WARP_SIZE / EntriesPerCacheline,
42 NumOfCachedTBs = TimebinsPerCacheline,
43 // Threads index shared memory as [iThread / MaxNPadsPerRow][iThread % MaxNPadsPerRow].
