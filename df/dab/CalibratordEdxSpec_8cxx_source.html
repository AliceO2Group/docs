94  void finaliseCCDB(o2::framework::ConcreteDataMatcher& matcher, void* obj) final
95  {
96  if (o2::base::GRPGeomHelper::instance().finaliseCCDB(matcher, obj)) {
97  return;
98  }
99  if (matcher == ConcreteDataMatcher("TPC", "TIMEGAIN", 0)) {
100  mTimeGain = *(o2::tpc::CalibdEdxCorrection*)obj;
101  const auto meanParamTot = mTimeGain.getMeanParams(ChargeType::Tot);
102  LOGP(info, "Updating TimeGain with {} dimensions and mean qTot Params {}", mTimeGain.getDims(), utils::elementsToString(meanParamTot));
103  return;
104  }
105  }

107  void run(ProcessingContext& pc) final
108  {
109  o2::base::TFIDInfoHelper::fillTFIDInfo(pc, mCalibrator->getCurrentTFInfo());
110  o2::base::GRPGeomHelper::instance().checkUpdates(pc);
111  checkUpdates(pc);
112  static bool slotLengthSet = false;
113  if (!slotLengthSet) {
114  // Important, otherwise the call to getSlotForTF below will be wrong, must be called after GRPGeomHelper update to get the correct TF length
115  mCalibrator->checkSlotLength();
116  slotLengthSet = true;
117  }
118 
119  auto& slotTF = mCalibrator->getSlotForTF(mCalibrator->getCurrentTFInfo().tfCounter);
120  auto calib = slotTF.getContainer();
121  const auto stackID = StackID{0, GEMstack::IROCgem};
122  const auto calibInEnties = calib->getCalibrationInput().getEntries(stackID, ChargeType::Tot);
123  calib->setCalibrationInput(mTimeGain);
124  if (calibInEnties != mTimeGain.getEntries(stackID, ChargeType::Tot)) {
125  const auto meanParamTot = calib->getCalibrationInput().getMeanParams(ChargeType::Tot);
126  LOGP(info, "Updating TimeGain with {} dimensions and mean qTot Params {} for slot with TF range {} <=TF<= {}",
127  calib->getCalibrationInput().getDims(), utils::elementsToString(meanParamTot), slotTF.getTFStart(), slotTF.getTFEnd());
128  }
129 
130  const auto tracks = pc.inputs().get<gsl::span<tpc::TrackTPC>>("tracks");
131  LOGP(detail, "Processing TF {} with {} tracks", mCalibrator->getCurrentTFInfo().tfCounter, tracks.size());
132  mRunNumber = mCalibrator->getCurrentTFInfo().runNumber;
133  mCalibrator->process(tracks);
134  sendOutput(pc.outputs());
135 
136  const auto& infoVec = mCalibrator->getTFinterval();
137  LOGP(detail, "Created {} objects for TF {}", infoVec.size(), mCalibrator->getCurrentTFInfo().tfCounter);
138  }

144  sendOutput(eos.outputs());

151  private:
152  void sendOutput(DataAllocator& output)
153  {
154  const auto& calibrations = mCalibrator->getCalibs();
155  const auto& intervals = mCalibrator->getTimeIntervals();
156 
157  assert(calibrations.size() == intervals.size());
158  for (unsigned int i = 0; i < calibrations.size(); i++) {
159  const auto& object = calibrations[i];
160  o2::ccdb::CcdbObjectInfo info(CDBTypeMap.at(CDBType::CalTimeGain), std::string{}, std::string{}, std::map<std::string, std::string>{{"runNumber", std::to_string(mRunNumber)}}, intervals[i].first, intervals[i].second + mCalibIntervalExtensionMS + 1);
161  auto image = o2::ccdb::CcdbApi::createObjectImage(&object, &info);
162  LOGP(info, "Sending object {} / {} of size {} bytes, valid for {} : {} ", info.getPath(), info.getFileName(), image->size(), info.getStartValidityTimestamp(), info.getEndValidityTimestamp());
163  output.snapshot(Output{o2::calibration::Utils::gDataOriginCDBPayload, "TPC_CalibdEdx", i}, *image.get()); // vector<char>
164  output.snapshot(Output{o2::calibration::Utils::gDataOriginCDBWrapper, "TPC_CalibdEdx", i}, info); // root-serialized
165  }
166  mCalibrator->initOutput(); // empty the outputs after they are send
167  }

191  outputs.emplace_back(ConcreteDataTypeMatcher{o2::calibration::Utils::gDataOriginCDBWrapper, "TPC_CalibdEdx"}, Lifetime::Sporadic);
192  std::vector<InputSpec> inputs{{"tracks", "TPC", "MIPS", Lifetime::Sporadic}};
193  inputs.emplace_back("tpctimegain", "TPC", "TIMEGAIN", 0, Lifetime::Condition, ccdbParamSpec(o2::tpc::CDBTypeMap.at(o2::tpc::CDBType::CalTimeGain), {}, 1)); // time-dependent

201  inputs,

203  true);
204  return DataProcessorSpec{
