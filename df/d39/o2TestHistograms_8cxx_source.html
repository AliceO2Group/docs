69  tables = cfgc.options().get<std::vector<std::string>>("aod-metadata-tables");
70  }
71  LOGP(info, "Runtype is {}", runType);
72  bool hasTrackCov = false;
73  bool hasTrackIU = false;
74  for (auto& table : tables) {
75  if (table == "O2trackcov") {
76  hasTrackCov = true;
77  }
78  if (table.starts_with("O2track_iu")) {
79  hasTrackIU = true;
80  }
81  LOGP(info, "- {} present.", table);
82  }
83  // Notice it's important for the tasks to use the same name, otherwise topology generation will be confused.
84  if (runType == "2" || !hasTrackCov) {
85  LOGP(info, "Using only tracks {}", runType);
86  if (hasTrackIU) {
87  return WorkflowSpec{
88  adaptAnalysisTask<EtaAndClsHistogramsIUSimple>(cfgc, TaskName{"simple-histos"}),
89  };
90  }
91  return WorkflowSpec{
92  adaptAnalysisTask<EtaAndClsHistogramsSimple>(cfgc, TaskName{"simple-histos"}),
93  };
94  } else {
95  LOGP(info, "Using tracks extra {}", runType);
96  if (hasTrackIU) {
97  return WorkflowSpec{
98  adaptAnalysisTask<EtaAndClsHistogramsIUSimple>(cfgc, TaskName{"simple-histos"}),
99  };
100  }
101  return WorkflowSpec{
