60  CommonDataProcessors::getGlobalFileSink(std::vector<InputSpec> const& danglingOutputInputs,
61  std::vector<InputSpec>& unmatched)
62 {
63  auto writerFunction = [danglingOutputInputs](InitContext& ic) -> std::function<void(ProcessingContext&)> {
64  auto filename = ic.options().get<std::string>("outfile");
65  auto keepString = ic.options().get<std::string>("keep");
66 
67  if (filename.empty()) {
68  throw runtime_error("output file missing");
69  }
70 
71  bool hasOutputsToWrite = false;
72  auto [variables, outputMatcher] = DataDescriptorQueryBuilder::buildFromKeepConfig(keepString);
73  VariableContext context;
74  for (auto& spec : danglingOutputInputs) {
75  auto concrete = DataSpecUtils::asConcreteDataTypeMatcher(spec);
76  if (outputMatcher->match(concrete, context)) {
77  hasOutputsToWrite = true;
78  }
79  }
80  if (hasOutputsToWrite == false) {
81  return [](ProcessingContext&) mutable -> void {
82  static bool once = false;
83  if (!once) {
84  LOG(debug) << "No dangling output to be dumped.";
85  once = true;
86  }
87  };
88  }
89  auto output = std::make_shared<std::ofstream>(filename.c_str(), std::ios_base::binary);
90  return [output, matcher = outputMatcher](ProcessingContext& pc) mutable -> void {
91  VariableContext matchingContext;
92  LOG(debug) << "processing data set with " << pc.inputs().size() << " entries";
93  for (const auto& entry : pc.inputs()) {
94  LOG(debug) << " " << *(entry.spec);
95  auto header = DataRefUtils::getHeader<header::DataHeader*>(entry);
96  auto dataProcessingHeader = DataRefUtils::getHeader<DataProcessingHeader*>(entry);
97  if (matcher->match(*header, matchingContext) == false) {
98  continue;
99  }
100  output->write(reinterpret_cast<char const*>(header), sizeof(header::DataHeader));
101  output->write(reinterpret_cast<char const*>(dataProcessingHeader), sizeof(DataProcessingHeader));
102  output->write(entry.payload, o2::framework::DataRefUtils::getPayloadSize(entry));
103  LOG(debug) << "wrote data, size " << o2::framework::DataRefUtils::getPayloadSize(entry);
104  }
105  };
106  };
107 
108  std::vector<InputSpec> validBinaryInputs;
109  auto onlyTimeframe = [](InputSpec const& input) {
110  return (DataSpecUtils::partialMatch(input, o2::header::DataOrigin("TFN")) == false) &&
111  input.lifetime == Lifetime::Timeframe;
112  };
113 
114  auto noTimeframe = [](InputSpec const& input) {
115  return (DataSpecUtils::partialMatch(input, o2::header::DataOrigin("TFN")) == true) ||
116  input.lifetime != Lifetime::Timeframe;
117  };
118 
119  std::copy_if(danglingOutputInputs.begin(), danglingOutputInputs.end(),
120  std::back_inserter(validBinaryInputs), onlyTimeframe);
121  std::copy_if(danglingOutputInputs.begin(), danglingOutputInputs.end(),
122  std::back_inserter(unmatched), noTimeframe);
123 
124  DataProcessorSpec spec{
125  "internal-dpl-injected-global-binary-file-sink",
126  validBinaryInputs,
127  Outputs{},
128  AlgorithmSpec(writerFunction),
129  {{"outfile", VariantType::String, "dpl-out.bin", {"Name of the output file"}},
130  {"keep", VariantType::String, "", {"Comma separated list of ORIGIN/DESCRIPTION/SUBSPECIFICATION to save in outfile"}}}};
131 
132  return spec;
133 }

135 DataProcessorSpec CommonDataProcessors::getGlobalFairMQSink(std::vector<InputSpec> const& danglingOutputInputs)
136 {
137 
138  // we build the default channel configuration from the binding of the first input
139  // in order to have more than one we would need to possibility to have support for
140  // vectored options
141  // use the OutputChannelSpec as a tool to create the default configuration for the out-of-band channel
142  OutputChannelSpec externalChannelSpec;
143  externalChannelSpec.name = "downstream";
144  externalChannelSpec.type = ChannelType::Push;
145  externalChannelSpec.method = ChannelMethod::Bind;
146  externalChannelSpec.hostname = "localhost";
147  externalChannelSpec.port = 0;
148  externalChannelSpec.listeners = 0;
149  // in principle, protocol and transport are two different things but fur simplicity
150  // we use ipc when shared memory is selected and the normal tcp url whith zeromq,
151  // this is for building the default configuration which can be simply changed from the
152  // command line
153  externalChannelSpec.protocol = ChannelProtocol::IPC;
154  std::string defaultChannelConfig = formatExternalChannelConfiguration(externalChannelSpec);
155  // at some point the formatting tool might add the transport as well so we have to check
156  return specifyFairMQDeviceOutputProxy("internal-dpl-injected-output-proxy", danglingOutputInputs, defaultChannelConfig.c_str());
157 }

159 void retryMetricCallback(uv_async_t* async)
160 {
161  static size_t lastTimeslice = -1;
162  auto* services = (ServiceRegistryRef*)async->data;
163  auto& timesliceIndex = services->get<TimesliceIndex>();
164  auto* device = services->get<RawDeviceService>().device();
165  auto channel = device->GetChannels().find("metric-feedback");
166  auto oldestPossingTimeslice = timesliceIndex.getOldestPossibleOutput().timeslice.value;
167  if (channel == device->GetChannels().end()) {
168  return;
169  }
170  fair::mq::MessagePtr payload(device->NewMessage());
171  payload->Rebuild(&oldestPossingTimeslice, sizeof(int64_t), nullptr, nullptr);
172  auto consumed = oldestPossingTimeslice;
173 
174  int64_t result = channel->second[0].Send(payload, 100);
175  // If the sending worked, we do not retry.
176  if (result != 0) {
177  // If the sending did not work, we keep trying until it actually works.
178  // This will schedule other tasks in the queue, so the processing of the
179  // data will still happen.
180  uv_async_send(async);
181  } else {
182  lastTimeslice = consumed;
183  }
184 }

186 DataProcessorSpec CommonDataProcessors::getDummySink(std::vector<InputSpec> const& danglingOutputInputs, std::string rateLimitingChannelConfig)
187 {
188  return DataProcessorSpec{
189  .name = "internal-dpl-injected-dummy-sink",
190  .inputs = danglingOutputInputs,
191  .algorithm = AlgorithmSpec{adaptStateful([](CallbackService& callbacks, DeviceState& deviceState, InitContext& ic) {
192  static uv_async_t async;
193  // The callback will only have access to the
194  async.data = new ServiceRegistryRef{ic.services()};
195  uv_async_init(deviceState.loop, &async, retryMetricCallback);
196  auto domainInfoUpdated = [](ServiceRegistryRef services, size_t timeslice, ChannelIndex channelIndex) {
197  LOGP(debug, "Domain info updated with timeslice {}", timeslice);
198  retryMetricCallback(&async);
199  auto& timesliceIndex = services.get<TimesliceIndex>();
200  auto oldestPossingTimeslice = timesliceIndex.getOldestPossibleOutput().timeslice.value;
201  auto& stats = services.get<DataProcessingStats>();
202  stats.updateStats({(int)ProcessingStatsId::CONSUMED_TIMEFRAMES, DataProcessingStats::Op::Set, (int64_t)oldestPossingTimeslice});
203  };
204  callbacks.set<CallbackService::Id::DomainInfoUpdated>(domainInfoUpdated);
205 
206  return adaptStateless([]() {
207  });
208  })},
209  .options = !rateLimitingChannelConfig.empty() ? std::vector<ConfigParamSpec>{{"channel-config", VariantType::String, // raw input channel
210  rateLimitingChannelConfig,
211  {"Out-of-band channel config"}}}
212  : std::vector<ConfigParamSpec>(),
213  .labels = {{"resilient"}}};
214 }

216 AlgorithmSpec CommonDataProcessors::wrapWithRateLimiting(AlgorithmSpec spec)
217 {
218  return PluginManager::wrapAlgorithm(spec, [](AlgorithmSpec::ProcessCallback& original, ProcessingContext& pcx) -> void {
219  auto& raw = pcx.services().get<RawDeviceService>();
220  static RateLimiter limiter;
221  auto limit = std::stoi(raw.device()->fConfig->GetValue<std::string>("timeframes-rate-limit"));
222  LOG(detail) << "Rate limiting to " << limit << " timeframes in flight";
223  limiter.check(pcx, limit, 2000);
224  LOG(detail) << "Rate limiting passed. Invoking old callback";
225  original(pcx);
226  LOG(detail) << "Rate limited callback done";
227  });
228 }
