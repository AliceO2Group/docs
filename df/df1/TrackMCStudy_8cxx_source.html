185 {
186  o2::base::GRPGeomHelper::instance().checkUpdates(pc);
187  mTPCVDriftHelper.extractCCDBInputs(pc);
188  mTPCCorrMapsLoader.extractCCDBInputs(pc);
189  bool updateMaps = false;
190  if (mTPCCorrMapsLoader.isUpdated()) {
191  mTPCCorrMapsLoader.acknowledgeUpdate();
192  updateMaps = true;
193  }
194  if (mTPCVDriftHelper.isUpdated()) {
195  LOGP(info, "Updating TPC fast transform map with new VDrift factor of {} wrt reference {} and DriftTimeOffset correction {} wrt {} from source {}",
196  mTPCVDriftHelper.getVDriftObject().corrFact, mTPCVDriftHelper.getVDriftObject().refVDrift,
197  mTPCVDriftHelper.getVDriftObject().timeOffsetCorr, mTPCVDriftHelper.getVDriftObject().refTimeOffset,

199  mTPCVDriftHelper.acknowledgeUpdate();
200  updateMaps = true;
201  }
202  if (updateMaps) {
203  mTPCCorrMapsLoader.updateVDrift(mTPCVDriftHelper.getVDriftObject().corrFact, mTPCVDriftHelper.getVDriftObject().refVDrift, mTPCVDriftHelper.getVDriftObject().getTimeOffset());
204  }

209  mITSROFrameLengthMUS = o2::base::GRPGeomHelper::instance().getGRPECS()->isDetContinuousReadOut(o2::detectors::DetID::ITS) ? alpParamsITS.roFrameLengthInBC * o2::constants::lhc::LHCBunchSpacingMUS : alpParamsITS.roFrameLengthTrig * 1.e-3;

213  mTPCTBinMUS = elParam.ZbinWidth;
214  }

217 void TrackMCStudy::process(const o2::globaltracking::RecoContainer& recoData)
218 {
219  constexpr float SQRT12Inv = 0.288675f;
220  const auto& params = o2::trackstudy::TrackMCStudyConfig::Instance();
221  auto pvvec = recoData.getPrimaryVertices();
222  auto pvvecLbl = recoData.getPrimaryVertexMCLabels();
223  auto trackIndex = recoData.getPrimaryVertexMatchedTracks(); // Global ID's for associated tracks
224  auto vtxRefs = recoData.getPrimaryVertexMatchedTrackRefs(); // references from vertex to these track IDs
225  auto prop = o2::base::Propagator::Instance();
226  int nv = vtxRefs.size();
227  float vdriftTB = mTPCVDriftHelper.getVDriftObject().getVDrift() * o2::tpc::ParameterElectronics::Instance().ZbinWidth; // VDrift expressed in cm/TimeBin
228  float itsBias = 0.5 * mITSROFrameLengthMUS + o2::itsmft::DPLAlpideParam<o2::detectors::DetID::ITS>::Instance().roFrameBiasInBC * o2::constants::lhc::LHCBunchSpacingMUS; // ITS time is supplied in \mus as beginning of ROF
229 
230  prepareITSData(recoData);
231  loadTPCOccMap(recoData);
232  auto getITSPatt = [&](GTrackID gid, uint8_t& ncl) {
233  int8_t patt = 0;
234  if (gid.getSource() == VTIndex::ITSAB) {
235  const auto& itsTrf = recoData.getITSABRefs()[gid];
236  ncl = itsTrf.getNClusters();
237  for (int il = 0; il < 7; il++) {
238  if (itsTrf.hasHitOnLayer(il)) {
239  patt |= 0x1 << il;
240  }
241  }
242  patt |= 0x1 << 7;
243  } else {
244  const auto& itsTr = recoData.getITSTrack(gid);
245  for (int il = 0; il < 7; il++) {
246  if (itsTr.hasHitOnLayer(il)) {
247  patt |= 0x1 << il;
248  ncl++;
249  }
250  }
251  }
252  return patt;
253  };
254 
255  auto getLowestPadrow = [&recoData](const o2::tpc::TrackTPC& trc) {
256  if (recoData.inputsTPCclusters) {
257  uint8_t clSect = 0, clRow = 0;
258  uint32_t clIdx = 0;
259  const auto clRefs = recoData.getTPCTracksClusterRefs();
260  trc.getClusterReference(clRefs, trc.getNClusterReferences() - 1, clSect, clRow, clIdx);
261  return int(clRow);
262  }
263  return -1;
264  };
265 
266  {
267  const auto* digconst = mcReader.getDigitizationContext();
268  const auto& mcEvRecords = digconst->getEventRecords(false);
269  int ITSTimeBias = o2::itsmft::DPLAlpideParam<o2::detectors::DetID::ITS>::Instance().roFrameBiasInBC;
270  int ITSROFLen = o2::itsmft::DPLAlpideParam<o2::detectors::DetID::ITS>::Instance().roFrameLengthInBC;
271  unsigned int rofCount = 0;
272  const auto ITSClusROFRec = recoData.getITSClustersROFRecords();
273  for (const auto& mcIR : mcEvRecords) {
274  long tbc = mcIR.differenceInBC(recoData.startIR);
275  auto& mcVtx = mMCVtVec.emplace_back();
276  mcVtx.ts = tbc * o2::constants::lhc::LHCBunchSpacingMUS + mcIR.getTimeOffsetWrtBC() * 1e-3;
277  mcVtx.ID = mIntBC.size();
278  mIntBC.push_back(tbc);
279  int occBin = tbc / 8 * mNTPCOccBinLengthInv;
280  mTPCOcc.push_back(occBin < 0 ? mTBinClOcc[0] : (occBin >= mTBinClOcc.size() ? mTBinClOcc.back() : mTBinClOcc[occBin]));
281  // fill ITS occupancy
282  long gbc = mcIR.toLong();
283  while (rofCount < ITSClusROFRec.size()) {
284  long rofbcMin = ITSClusROFRec[rofCount].getBCData().toLong() + ITSTimeBias, rofbcMax = rofbcMin + ITSROFLen;
285  if (gbc < rofbcMin) { // IRs and ROFs are sorted, so this IR is prior of all ROFs
286  mITSOcc.push_back(0);
287  } else if (gbc < rofbcMax) {
288  mITSOcc.push_back(ITSClusROFRec[rofCount].getNEntries());
289  } else {
290  rofCount++; // test next ROF
291  continue;
292  }
293  break;
294  }
295  if (rofCount >= ITSClusROFRec.size()) {
296  mITSOcc.push_back(0); // IR after the last ROF
297  }
298  }
299  }
300  // collect interesting MC particle (tracks and parents)
301  int curSrcMC = 0, curEvMC = 0;
302  for (curSrcMC = 0; curSrcMC < (int)mcReader.getNSources(); curSrcMC++) {
303  if (mVerbose > 1) {
304  LOGP(info, "Source {}", curSrcMC);
305  }
306  int nev = mcReader.getNEvents(curSrcMC);
307  bool okAccVtx = true;
308  if (nev != (int)mMCVtVec.size()) {
309  LOGP(error, "source {} has {} events while {} MC vertices were booked", curSrcMC, nev, mMCVtVec.size());
310  okAccVtx = false;
311  }
312  for (curEvMC = 0; curEvMC < nev; curEvMC++) {
313  if (mVerbose > 1) {
314  LOGP(info, "Event {}", curEvMC);
315  }
316  const auto& mt = mcReader.getTracks(curSrcMC, curEvMC);
317  mCurrMCTracks = gsl::span<const MCTrack>(mt.data(), mt.size());
318  const_cast<o2::dataformats::MCEventHeader&>(mcReader.getMCEventHeader(curSrcMC, curEvMC)).GetVertex(mCurrMCVertex);
319  if (okAccVtx) {
320  auto& pos = mMCVtVec[curEvMC].pos;
321  if (pos[2] < -999) {
322  pos[0] = mCurrMCVertex.X();
323  pos[1] = mCurrMCVertex.Y();
324  pos[2] = mCurrMCVertex.Z();
325  }
326  }
327  for (int itr = 0; itr < mCurrMCTracks.size(); itr++) {
328  processMCParticle(curSrcMC, curEvMC, itr);
329  }
330  }
331  }
332  if (mVerbose > 0) {
333  for (int id = 0; id < mNCheckDecays; id++) {
334  LOGP(info, "Decay PDG={} : {} entries", params.decayPDG[id], mDecaysMaps[id].size());
335  }
336  }
337 
338  // add reconstruction info to MC particles. If MC particle was not selected before but was reconstrected, account MC info
339  for (int iv = 0; iv < nv; iv++) {
340  if (mVerbose > 1) {
341  LOGP(info, "processing PV {} of {}", iv, nv);
342  }
343  o2::MCEventLabel pvLbl;
344  int pvID = -1;
345  if (iv < (int)pvvecLbl.size()) {
346  pvLbl = pvvecLbl[iv];
347  pvID = iv;
348  if (pvLbl.isSet() && pvLbl.getEventID() < mMCVtVec.size()) {
349  mMCVtVec[pvLbl.getEventID()].recVtx.emplace_back(RecPV{pvvec[iv], pvLbl});
350  }
351  }
352  const auto& vtref = vtxRefs[iv];
353  for (int is = GTrackID::NSources; is--;) {
354  DetID::mask_t dm = GTrackID::getSourceDetectorsMask(is);
355  if (!mTracksSrc[is] || !recoData.isTrackSourceLoaded(is) || !(dm[DetID::ITS] || dm[DetID::TPC])) {
356  continue;
357  }
358  int idMin = vtref.getFirstEntryOfSource(is), idMax = idMin + vtref.getEntriesOfSource(is);
359  for (int i = idMin; i < idMax; i++) {
360  auto vid = trackIndex[i];
361  const auto& trc = recoData.getTrackParam(vid);
362  if (trc.getPt() < params.minPt || std::abs(trc.getTgl()) > params.maxTgl) {
363  continue;
364  }
365  auto lbl = recoData.getTrackMCLabel(vid);
366  if (lbl.isValid()) {
367  lbl.setFakeFlag(false);
368  auto entry = mSelMCTracks.find(lbl);
369  if (entry == mSelMCTracks.end()) { // add the track which was not added during MC scan
370  if (lbl.getSourceID() != curSrcMC || lbl.getEventID() != curEvMC) {
371  curSrcMC = lbl.getSourceID();
372  curEvMC = lbl.getEventID();
373  const auto& mt = mcReader.getTracks(curSrcMC, curEvMC);
374  mCurrMCTracks = gsl::span<const MCTrack>(mt.data(), mt.size());
375  const_cast<o2::dataformats::MCEventHeader&>(mcReader.getMCEventHeader(curSrcMC, curEvMC)).GetVertex(mCurrMCVertex);
376  }
377  if (!acceptMCCharged(mCurrMCTracks[lbl.getTrackID()], lbl)) {
378  continue;
379  }
380  entry = mSelMCTracks.find(lbl);
381  }
382  auto& trackFamily = entry->second;
383  if (vid.isAmbiguous()) { // do not repeat ambiguous tracks
384  if (trackFamily.contains(vid)) {
385  continue;
386  }
387  }
388  auto& trf = trackFamily.recTracks.emplace_back();
389  trf.gid = vid; // account(iv, vid);
390  trf.pvID = pvID;
391  trf.pvLabel = pvLbl;
392  while (dm[DetID::ITS] && dm[DetID::TPC]) { // this track should have both ITS and TPC parts, if ITS was mismatched, fill it to its proper MC track slot
393  auto gidSet = recoData.getSingleDetectorRefs(vid);
394  if (!gidSet[GTrackID::ITS].isSourceSet()) {
395  break; // AB track, nothing to check
396  }
397  auto lblITS = recoData.getTrackMCLabel(gidSet[GTrackID::ITS]);
398  if (lblITS == trackFamily.mcTrackInfo.label) {
399  break; // correct match, no need for special treatment
400  }
401  const auto& trcITSF = recoData.getTrackParam(gidSet[GTrackID::ITS]);
402  if (trcITSF.getPt() < params.minPt || std::abs(trcITSF.getTgl()) > params.maxTgl) {
403  break; // ignore this track
404  }
405  auto entryOfFake = mSelMCTracks.find(lblITS);
406  if (entryOfFake == mSelMCTracks.end()) { // this MC track was not selected
407  break;
408  }
409  auto& trackFamilyOfFake = entryOfFake->second;
410  auto& trfOfFake = trackFamilyOfFake.recTracks.emplace_back();
411  trfOfFake.gid = gidSet[GTrackID::ITS]; // account(iv, vid);
412  break;
413  }
414  if (mVerbose > 1) {
415  LOGP(info, "Matched rec track {} to MC track {}", vid.asString(), entry->first.asString());
416  }
417  } else {
418  continue;
419  }
420  }
421  }
422  }
423 
424  // collect ITS/TPC cluster info for selected MC particles
425  if (params.minTPCRefsToExtractClRes > 0) {
426  processTPCTrackRefs();
427  }
428  fillMCClusterInfo(recoData);
429 
430  LOGP(info, "collected {} MC tracks", mSelMCTracks.size());
431  int mcnt = 0;
432  for (auto& entry : mSelMCTracks) {
433  auto& trackFam = entry.second;
434  auto& tracks = trackFam.recTracks;
435  mcnt++;
436  if (tracks.empty()) {
437  continue;
438  }
439  if (mVerbose > 1) {
440  LOGP(info, "Processing MC track#{} {} -> {} reconstructed tracks", mcnt - 1, entry.first.asString(), tracks.size());
441  }
442  // sort according to the gid complexity (in principle, should be already sorted due to the backwards loop over NSources above
443  std::sort(tracks.begin(), tracks.end(), [](const RecTrack& lhs, const RecTrack& rhs) {
444  const auto mskL = lhs.gid.getSourceDetectorsMask();
445  const auto mskR = rhs.gid.getSourceDetectorsMask();
446  bool itstpcL = mskL[DetID::ITS] && mskL[DetID::TPC], itstpcR = mskR[DetID::ITS] && mskR[DetID::TPC];
447  if (itstpcL && !itstpcR) { // to avoid TPC/TRD or TPC/TOF shadowing ITS/TPC
448  return true;
449  }
450  return lhs.gid.getSource() > rhs.gid.getSource();
451  });
452  // fill track params
453  int tcnt = 0;
454  for (auto& tref : tracks) {
455  if (tref.gid.isSourceSet()) {
456  auto gidSet = recoData.getSingleDetectorRefs(tref.gid);
457  tref.track = recoData.getTrackParam(tref.gid);
458  if (recoData.getTrackMCLabel(tref.gid).isFake()) {
459  tref.flags |= RecTrack::FakeGLO;
460  }
461  auto msk = tref.gid.getSourceDetectorsMask();
462  if (msk[DetID::ITS]) {
463  if (gidSet[GTrackID::ITS].isSourceSet()) { // has ITS track rather than AB tracklet
464  tref.pattITS = getITSPatt(gidSet[GTrackID::ITS], tref.nClITS);
465  if (trackFam.entITS < 0) {
466  trackFam.entITS = tcnt;
467  }
468  auto lblITS = recoData.getTrackMCLabel(gidSet[GTrackID::ITS]);
469  if (lblITS.isFake()) {
470  tref.flags |= RecTrack::FakeITS;
471  }
472  if (lblITS == trackFam.mcTrackInfo.label) {
473  trackFam.entITSFound = tcnt;
474  }
475  } else { // AB ITS tracklet
476  tref.pattITS = getITSPatt(gidSet[GTrackID::ITSAB], tref.nClITS);
477  if (recoData.getTrackMCLabel(gidSet[GTrackID::ITSAB]).isFake()) {
478  tref.flags |= RecTrack::FakeITS;
479  }
480  }

748 {

751  const auto& par = o2::globaltracking::MatchTPCITSParams::Instance();

772  }
773  refReached = false;
774  float alp = trcTPC.getAlpha();
775  if (!trcITS.rotate(alp) != 0 || !o2::base::Propagator::Instance()->PropagateToXBxByBz(trcITS, par.XMatchingRef, MaxSnp, 2., par.matCorr)) {

893 bool TrackMCStudy::acceptMCCharged(const MCTrack& tr, const o2::MCCompLabel& lb, int followDecay)
894 {

896  if (tr.GetPt() < params.minPtMC ||
897  std::abs(tr.GetTgl()) > params.maxTglMC ||
898  tr.R2() > params.maxRMC * params.maxRMC) {
899  if (mVerbose > 1 && followDecay > -1) {

901  }

905  float r2 = dx * dx + dy * dy;
906  float posTgl2 = r2 > 1 && std::abs(dz) < 20 ? dz * dz / r2 : 0;
907  if (posTgl2 > params.maxPosTglMC * params.maxPosTglMC) {
908  if (mVerbose > 1 && followDecay > -1) {

910  }
911  return false;
912  }
913  if (params.requireITSorTPCTrackRefs) {
914  auto trspan = mcReader.getTrackRefs(lb.getSourceID(), lb.getEventID(), lb.getTrackID());
915  bool ok = false;
916  for (const auto& trf : trspan) {
917  if (trf.getDetectorId() == DetID::ITS || trf.getDetectorId() == DetID::TPC) {
918  ok = true;

923  return false;

927  if (!pPDG) {
928  LOGP(debug, "Unknown particle {}", tr.GetPdgCode());
929  return false;

932  return false;
933  }
934  return addMCParticle(tr, lb, pPDG);
