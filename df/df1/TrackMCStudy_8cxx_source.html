136 void TrackMCStudy::updateTimeDependentParams(ProcessingContext& pc)
137 {
138  o2::base::GRPGeomHelper::instance().checkUpdates(pc);
139  mTPCVDriftHelper.extractCCDBInputs(pc);
140  if (mTPCVDriftHelper.isUpdated()) {
141  LOGP(info, "Updating TPC fast transform map with new VDrift factor of {} wrt reference {} and DriftTimeOffset correction {} wrt {} from source {}",
142  mTPCVDriftHelper.getVDriftObject().corrFact, mTPCVDriftHelper.getVDriftObject().refVDrift,
143  mTPCVDriftHelper.getVDriftObject().timeOffsetCorr, mTPCVDriftHelper.getVDriftObject().refTimeOffset,
144  mTPCVDriftHelper.getSourceName());
145  mTPCVDriftHelper.acknowledgeUpdate();
146  }
147  static bool initOnceDone = false;
148  if (!initOnceDone) { // this params need to be queried only once
149  initOnceDone = true;
150  const auto& alpParamsITS = o2::itsmft::DPLAlpideParam<o2::detectors::DetID::ITS>::Instance();
151  mITSROFrameLengthMUS = o2::base::GRPGeomHelper::instance().getGRPECS()->isDetContinuousReadOut(o2::detectors::DetID::ITS) ? alpParamsITS.roFrameLengthInBC * o2::constants::lhc::LHCBunchSpacingMUS : alpParamsITS.roFrameLengthTrig * 1.e-3;
152  LOGP(info, "VertexTrackMatcher ITSROFrameLengthMUS:{}", mITSROFrameLengthMUS);
153 
154  auto& elParam = o2::tpc::ParameterElectronics::Instance();
155  mTPCTBinMUS = elParam.ZbinWidth;

159 void TrackMCStudy::process(o2::globaltracking::RecoContainer& recoData)
160 {
161  auto pvvec = recoData.getPrimaryVertices();
162  auto trackIndex = recoData.getPrimaryVertexMatchedTracks(); // Global ID's for associated tracks
163  auto vtxRefs = recoData.getPrimaryVertexMatchedTrackRefs(); // references from vertex to these track IDs
164  auto prop = o2::base::Propagator::Instance();
165  int nv = vtxRefs.size();
166  float vdriftTB = mTPCVDriftHelper.getVDriftObject().getVDrift() * o2::tpc::ParameterElectronics::Instance().ZbinWidth; // VDrift expressed in cm/TimeBin
167  float itsBias = 0.5 * mITSROFrameLengthMUS + o2::itsmft::DPLAlpideParam<o2::detectors::DetID::ITS>::Instance().roFrameBiasInBC * o2::constants::lhc::LHCBunchSpacingNS * 1e-3; // ITS time is supplied in \mus as beginning of ROF
168 
169  if (mCheckMatching) {
170  prepareITSData(recoData);
171  }
172 
173  auto getITSPatt = [&](GTrackID gid, uint8_t& ncl) {
174  int16_t patt = 0;
175  if (gid.getSource() == VTIndex::ITSAB) {
176  const auto& itsTrf = recoData.getITSABRefs()[gid];
177  ncl = itsTrf.getNClusters();
178  for (int il = 0; il < 7; il++) {
179  if (itsTrf.hasHitOnLayer(il)) {
180  patt |= 0x1 << il;
181  }
182  }
183  patt = -patt;
184  } else {
185  const auto& itsTr = recoData.getITSTrack(gid);
186  for (int il = 0; il < 7; il++) {
187  if (itsTr.hasHitOnLayer(il)) {
188  patt |= 0x1 << il;
189  ncl++;
190  }
191  }
192  }
193  return patt;
194  };
195 
196  auto getLowestPadrow = [&recoData](const o2::tpc::TrackTPC& trc) {
197  if (recoData.inputsTPCclusters) {
198  uint8_t clSect = 0, clRow = 0;
199  uint32_t clIdx = 0;
200  const auto clRefs = recoData.getTPCTracksClusterRefs();
201  trc.getClusterReference(clRefs, trc.getNClusterReferences() - 1, clSect, clRow, clIdx);
202  return int(clRow);
203  }
204  return -1;
205  };
206 
207  std::map<o2::MCCompLabel, std::vector<VTIndexV>> MCTRMap;
208 
209  for (int iv = 0; iv < nv; iv++) {
210  if (mVerbose > 0) {
211  LOGP(info, "processing PV {} of {}", iv, nv);
212  }
213  const auto& vtref = vtxRefs[iv];
214  for (int is = GTrackID::NSources; is--;) {
215  DetID::mask_t dm = GTrackID::getSourceDetectorsMask(is);
216  if (!mTracksSrc[is] || !recoData.isTrackSourceLoaded(is) || !(dm[DetID::ITS] || dm[DetID::TPC])) {
217  continue;
218  }
219  int idMin = vtref.getFirstEntryOfSource(is), idMax = idMin + vtref.getEntriesOfSource(is);
220  for (int i = idMin; i < idMax; i++) {
221  auto vid = trackIndex[i];
222  const auto& trc = recoData.getTrackParam(vid);
223  if (trc.getPt() < mMinPt) {
224  continue;
225  }
226  auto lbl = recoData.getTrackMCLabel(vid);
227  if (lbl.isValid()) {
228  lbl.setFakeFlag(false);
229  auto& vvids = MCTRMap[lbl];
230  if (vid.isAmbiguous() || vvids.empty()) { // do not repeat ambiguous tracks
231  bool skip = false;
232  for (const auto& va : vvids) {
233  if (va.second == vid) {
234  skip = true;
235  break;
236  }
237  }
238  if (skip) {
239  continue;
240  }
241  }
242  vvids.emplace_back(iv, vid);
243  }
244  }
245  }
246  }
247  o2::track::TrackParCov dummyTrackParCov;
248  o2::track::TrackPar dummyTrackPar;
249  dummyTrackParCov.invalidate();
250  dummyTrackPar.invalidate();
251 
252  const std::vector<o2::MCTrack>* mcTracks = nullptr;
253  o2::MCCompLabel prevLbl;
254  std::vector<o2::track::TrackParCov> recTracks;
255  std::vector<VTIndex> recGIDs;
256  std::vector<bool> recFakes;
257  std::vector<int16_t> lowestPadrows;
258  std::vector<int16_t> itsPatterns;
259  std::vector<uint8_t> tpcNcls;
260  std::vector<uint8_t> itsNcls;
261  LOGP(info, "Recorded {} reconstructed tracks", MCTRMap.size());
262  size_t count = 0;
263  for (auto ent : MCTRMap) {
264  count++;
265  auto lbl = ent.first;
266  if (lbl.getEventID() != prevLbl.getEventID() || lbl.getSourceID() != prevLbl.getSourceID()) {
267  if (mVerbose > 0) {
268  LOGP(info, "Loading MC Event={} / Src={}", lbl.getEventID(), lbl.getSourceID());
269  }
270  mcTracks = &mcReader.getTracks(lbl.getSourceID(), lbl.getEventID());
271  prevLbl = lbl;
272  }
273  const auto& mcPart = (*mcTracks)[lbl.getTrackID()];
274  int pdg = mcPart.GetPdgCode(), pdgParent = 0;
275  std::array<float, 3> xyz{(float)mcPart.GetStartVertexCoordinatesX(), (float)mcPart.GetStartVertexCoordinatesY(), (float)mcPart.GetStartVertexCoordinatesZ()};
276  std::array<float, 3> pxyz{(float)mcPart.GetStartVertexMomentumX(), (float)mcPart.GetStartVertexMomentumY(), (float)mcPart.GetStartVertexMomentumZ()};
277  TParticlePDG* pPDG = TDatabasePDG::Instance()->GetParticle(pdg);
278  if (!pPDG) {
279  LOGP(error, "Unknown particle {}, skip. Was at {} of {}", pdg, count, MCTRMap.size());
280  continue;
281  }
282  o2::track::TrackPar mctrO2(xyz, pxyz, TMath::Nint(pPDG->Charge() / 3), false);
283 
284  bool primary = mcPart.isPrimary();
285  auto parID = primary ? -1 : mcPart.getMotherTrackId();
286  if (parID >= 0) {
287  const auto& mcPartPar = (*mcTracks)[parID];
288  pdgParent = mcPartPar.GetPdgCode();
289  }
290  auto& vgids = ent.second;
291  // make sure the more global tracks come 1st
292  if (vgids.size() > 1) {
293  std::sort(vgids.begin(), vgids.end(), [](VTIndexV& lhs, VTIndexV& rhs) { return lhs.second.getSource() > rhs.second.getSource(); });
294  }
295  recTracks.clear();
296  recGIDs.clear();
297  recFakes.clear();
298  lowestPadrows.clear();
299  itsPatterns.clear();
300  itsNcls.clear();
301  tpcNcls.clear();
302  if (mVerbose > 1) {
303  LOGP(info, "[{}] Lbl:{} PDG:{:+5d} (par: {:+5d}) | MC: {}", vgids.size(), lbl.asString(), pdg, pdgParent, mctrO2.asString());
304  }
305  int entITS = -1, entTPC = -1, entITSTPC = -1;
306  for (size_t i = 0; i < vgids.size(); i++) {
307  auto vid = vgids[i].second;
308  auto lbl = recoData.getTrackMCLabel(vid);
309  const auto& trc = recoData.getTrackParam(vid);
310  int16_t itsPatt = 0;
311  uint8_t nclITS = 0;
312  uint8_t nclTPC = 0;
313  if (mVerbose > 1) {
314  LOGP(info, " :{} {:22} | [{}] {}", lbl.asString(), vid.asString(), i, ((const o2::track::TrackPar&)trc).asString());
315  }
316  recTracks.push_back(trc);
317  recGIDs.push_back(vid);
318  recFakes.push_back(recoData.getTrackMCLabel(vid).isFake());
319  auto msk = vid.getSourceDetectorsMask();
320  if (mCheckMatching) {
321  lowestPadrows.push_back(-1);
322  }
323  if (msk[DetID::ITS]) {
324  auto gidITS = recoData.getITSContributorGID(vid);
325  itsPatt = getITSPatt(gidITS, nclITS);
326  if (vid.getSource() == VTIndex::ITS) {
327  entITS = i;
328  }
329  if (msk[DetID::TPC]) {
330  entITSTPC = i;
331  }
332  }
333  if (msk[DetID::TPC]) {
334  if (vid.getSource() == VTIndex::TPC) {
335  entTPC = i;
336  }
337  auto gidTPC = recoData.getTPCContributorGID(vid);
338  const auto& trtpc = recoData.getTPCTrack(gidTPC);
339  nclTPC = trtpc.getNClusters();
340  if (mCheckMatching) {
341  auto& lr = lowestPadrows.back();
342  lr = getLowestPadrow(recoData.getTPCTrack(recoData.getTPCContributorGID(vid)));
343  }
344  }
345  tpcNcls.push_back(nclTPC);
346  itsNcls.push_back(nclITS);
347  itsPatterns.push_back(itsPatt);
348  }
349  (*mDBGOut) << "tracks"
350  << "lbl=" << lbl
351  << "mcTr=" << mctrO2
352  << "pdg=" << pdg
353  << "pdgPar=" << pdgParent
354  << "recTr=" << recTracks
355  << "recGID=" << recGIDs
356  << "recFake=" << recFakes
357  << "itsPatt=" << itsPatterns
358  << "nClITS=" << itsNcls
359  << "nClTPC=" << tpcNcls;
360  if (mCheckMatching) {
361  (*mDBGOut) << "tracks"
362  << "lowestPadRow=" << lowestPadrows;
363  }
364  (*mDBGOut) << "tracks"
365  << "\n";
366 
367  // special ITS-TPC matching failure output
368  while (mCheckMatching) {
369  if (entITSTPC < 0 && entITS > -1 && entTPC > -1) { // ITS and TPC were found but matching failed
370  auto vidITS = vgids[entITS].second;
371  auto vidTPC = recoData.getTPCContributorGID(vgids[entTPC].second); // might be TPC match to outer detector, extract TPC
372  auto trcTPC = recoData.getTrackParam(vidTPC);
373  auto trcITS = recoData.getTrackParam(vidITS);
374  if (!propagateToRefX(trcTPC, trcITS)) {
375  break;
376  }
377  const auto& trcTPCOrig = recoData.getTPCTrack(vidTPC);
378  const auto& trcITSOrig = recoData.getITSTrack(vidITS);
379  int lowestTPCRow = lowestPadrows[entTPC];
380  float tpcT0 = trcTPCOrig.getTime0(), tF = trcTPCOrig.getDeltaTFwd(), tB = trcTPCOrig.getDeltaTBwd();
381  TBracket tpcBr((tpcT0 - tB) * mTPCTBinMUS, (tpcT0 + tF) * mTPCTBinMUS);
382 
383  (*mDBGOut) << "failMatch"
384  << "mcTr=" << mctrO2
385  << "pdg=" << pdg
386  << "pdgPar=" << pdgParent
387  << "labelITS=" << recoData.getTrackMCLabel(vidITS)
388  << "labelTPC=" << recoData.getTrackMCLabel(vidTPC)
389  << "gidITS=" << vidITS
390  << "gidTPC=" << vidTPC
391  << "itsBracket=" << mITSROFBracket[mITSROF[vidITS.getIndex()]]
392  << "tpcBracket=" << tpcBr
393  << "itsRef=" << trcITS
394  << "tpcRef=" << trcTPC
395  << "itsOrig=" << trcITSOrig
396  << "tpcOrig=" << trcTPCOrig
397  << "itsPatt=" << itsPatterns[entITS]
398  << "tpcLowestRow=" << lowestTPCRow
399  << "\n";
400  } else if (entITSTPC > -1) { // match was found
401  auto contribIDs = recoData.getSingleDetectorRefs(vgids[entITSTPC].second);
402  auto vidMatch = contribIDs[VTIndex::ITSTPC];
403  auto vidTPC = contribIDs[VTIndex::TPC];
404  auto vidITS = contribIDs[VTIndex::ITSAB].isSourceSet() ? contribIDs[VTIndex::ITSAB] : contribIDs[VTIndex::ITS];
405  const auto& trcTPCOrig = recoData.getTPCTrack(vidTPC);
406  o2::MCCompLabel itsLb;
407  int nITScl = 0;
408  if (vidITS.getSource() == VTIndex::ITS) {
409  itsLb = recoData.getTrackMCLabel(vidITS);
410  nITScl = recoData.getITSTrack(vidITS).getNClusters();
411  } else {
412  itsLb = recoData.getITSABMCLabels()[vidITS];
413  nITScl = recoData.getITSABRefs()[vidITS].getNClusters();
414  }
415  int lowestTPCRow = lowestPadrows[entITSTPC];
416  const auto& trackITSTPC = recoData.getTPCITSTrack(vidMatch);
417  float timeTB = trackITSTPC.getTimeMUS().getTimeStamp() / o2::constants::lhc::LHCBunchSpacingMUS / 8; // ITS-TPC time in TPC timebins
418 
419  (*mDBGOut) << "match"
420  << "mcTr=" << mctrO2
421  << "pdg=" << pdg
422  << "pdgPar=" << pdgParent
423  << "labelMatch=" << recoData.getTrackMCLabel(vidMatch)
424  << "labelTPC=" << recoData.getTrackMCLabel(vidTPC)
425  << "labelITS=" << itsLb
426  << "gidTPC=" << vidTPC
427  << "gidITS=" << vidITS
428  << "tpcOrig=" << trcTPCOrig
429  << "nClITS=" << itsNcls[entITSTPC]
430  << "itsPatt=" << itsPatterns[entITSTPC]
431  << "itstpc=" << ((o2::track::TrackParCov&)trackITSTPC)
432  << "matchChi2=" << trackITSTPC.getChi2Match()
433  << "refitChi2=" << trackITSTPC.getChi2Refit()
434  << "timeTB=" << timeTB
435  << "tpcLowestRow=" << lowestTPCRow
436  << "\n";
437  }
438  break;
439  }
440  }
441 }

443 bool TrackMCStudy::propagateToRefX(o2::track::TrackParCov& trcTPC, o2::track::TrackParCov& trcITS)
444 {
445  bool refReached = false;
446  constexpr float TgHalfSector = 0.17632698f;
447  const auto& par = o2::globaltracking::MatchTPCITSParams::Instance();
448  int trialsLeft = 2;
449  while (o2::base::Propagator::Instance()->PropagateToXBxByBz(trcTPC, par.XMatchingRef, MaxSnp, 2., par.matCorr)) {
450  if (refReached) {
451  break;
452  }
453  // make sure the track is indeed within the sector defined by alpha
454  if (fabs(trcTPC.getY()) < par.XMatchingRef * TgHalfSector) {
455  refReached = true;
456  break; // ok, within
457  }
458  if (!trialsLeft--) {
459  break;
460  }
461  auto alphaNew = o2::math_utils::angle2Alpha(trcTPC.getPhiPos());
462  if (!trcTPC.rotate(alphaNew) != 0) {
463  break; // failed (RS: check effect on matching tracks to neighbouring sector)
464  }
465  }
466  if (!refReached) {
467  return false;
468  }
469  refReached = false;
470  float alp = trcTPC.getAlpha();
471  if (!trcITS.rotate(alp) != 0 || !o2::base::Propagator::Instance()->PropagateToXBxByBz(trcITS, par.XMatchingRef, MaxSnp, 2., par.matCorr)) {
472  return false;
473  }
474  return true;
475 }

482 void TrackMCStudy::finaliseCCDB(ConcreteDataMatcher& matcher, void* obj)
483 {
484  if (o2::base::GRPGeomHelper::instance().finaliseCCDB(matcher, obj)) {
485  return;
486  }
487  if (mTPCVDriftHelper.accountCCDBInputs(matcher, obj)) {
488  return;
489  }
490  if (matcher == ConcreteDataMatcher("ITS", "ALPIDEPARAM", 0)) {
491  LOG(info) << "ITS Alpide param updated";
492  const auto& par = o2::itsmft::DPLAlpideParam<o2::detectors::DetID::ITS>::Instance();
