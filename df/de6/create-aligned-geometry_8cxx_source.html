32 /*
33  Simple workflow to create aligned geometry file from the ideal geometry and alignment objects on the CCDB
34  One can profit from the DPL CCDB fetcher --condition-remap functionality to impose difference CCDB
35  servers for different alignment objects.
36  Also, one can use --configKeyValues "HBFUtils.startTime=... functionality to impose the timestamp for the objects to fetch.
37  E.g. to create the geometry using current timestamp and production server, use just
38  o2-create-aligned-geometry-workflow
39 
40  To create the geometry using specific timestamp and ITS alignment from the test CCDB server and MFT alignment
41  from the local snapshot file in the locCCDB directory
42  o2-create-aligned-geometry-workflow --condition-remap http://ccdb-test.cern.ch:8080=ITS/Calib/Align;file://locCCDB=MFT/Calib/Align --configKeyValues "HBFUtils.startTime=1546300800000"
43  */

48 class SeederTask : public Task
49 {
50  public:
51  void run(ProcessingContext& pc) final
52  {
53  const auto& hbfu = o2::raw::HBFUtils::Instance();
54  auto& tinfo = pc.services().get<o2::framework::TimingInfo>();
55  if (hbfu.startTime != 0) {
56  tinfo.creation = hbfu.startTime;
57  } else {
58  tinfo.creation = std::chrono::time_point_cast<std::chrono::milliseconds>(std::chrono::system_clock::now()).time_since_epoch().count();
59  }
60  if (hbfu.orbitFirstSampled != 0) {
61  tinfo.firstTForbit = hbfu.orbitFirstSampled;
62  } else {
63  tinfo.firstTForbit = 0;
64  }
65  auto& stfDist = pc.outputs().make<o2::header::STFHeader>(Output{"FLP", "DISTSUBTIMEFRAME", 0});
66  pc.services().get<ControlService>().endOfStream();
67  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
68  }
69 };

71 class AlignerTask : public Task
72 {
73  public:
74  AlignerTask(DetID::mask_t dets, std::shared_ptr<o2::base::GRPGeomRequest> gr) : mDetsMask(dets), mGGCCDBRequest(gr) {}
75  ~AlignerTask() override = default;

77  void init(InitContext& ic) final
78  {
79  o2::base::GRPGeomHelper::instance().setRequest(mGGCCDBRequest);
80  }
81 
82  void run(ProcessingContext& pc) final
83  {
84  o2::base::GRPGeomHelper::instance().checkUpdates(pc);
85  for (auto id = DetID::First; id <= DetID::Last; id++) {
86  if (mDetsMask[id]) {
87  auto alg = o2::base::GRPGeomHelper::instance().getAlignment(id);
88  if (!alg->empty()) {
89  LOGP(info, "Applying alignment for detector {}", DetID::getName(id));
90  o2::base::GeometryManager::applyAlignment(*alg);
91  } else {
92  LOGP(info, "Alignment for detector {} is empty", DetID::getName(id));
93  }
94  }
95  }
96  gGeoManager->SetName(std::string(o2::base::NameConf::CCDBOBJECT).c_str());
97  auto fnm = o2::base::NameConf::getAlignedGeomFileName();
98  gGeoManager->Export(fnm.c_str());
99  LOG(info) << "Stored to local file " << fnm;
100  pc.services().get<ControlService>().endOfStream();
101  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
102  }
103 
104  void finaliseCCDB(ConcreteDataMatcher& matcher, void* obj) final
105  {
106  o2::base::GRPGeomHelper::instance().finaliseCCDB(matcher, obj);
107  }
108 
109  private:
110  DetID::mask_t mDetsMask{};
111  std::shared_ptr<o2::base::GRPGeomRequest> mGGCCDBRequest;
112 };

116 DataProcessorSpec getAlignerSpec(DetID::mask_t dets)
117 {
118  std::vector<InputSpec> inputs{{"STFDist", "FLP", "DISTSUBTIMEFRAME", 0}}; // just to have some input
119  auto ggRequest = std::make_shared<o2::base::GRPGeomRequest>(false, // orbitResetTime
120  false, // GRPECS
121  false, // GRPLHCIF
122  false, // GRPMagField
123  false, // askMatLUT
124  o2::base::GRPGeomRequest::Alignments, // geometry
125  inputs);
126 
127  return DataProcessorSpec{
128  "geometry-aligned-producer",
129  inputs,
130  {},
131  AlgorithmSpec{adaptFromTask<o2::base::AlignerTask>(dets, ggRequest)},
132  Options{}};
133 }
