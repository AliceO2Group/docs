95 bool DigitizationContext::initSimChains(o2::detectors::DetID detid, std::vector<TChain*>& simchains) const
96 {
97  if (!(simchains.size() == 0)) {
98  // nothing to do ... already setup
99  return false;
100  }
101 
102  simchains.emplace_back(new TChain("o2sim"));
103  // add the main (background) file
104  simchains.back()->AddFile(o2::base::DetectorNameConf::getHitsFileName(detid, mSimPrefixes[0].data()).c_str());
105 
106  for (int source = 1; source < mSimPrefixes.size(); ++source) {
107  simchains.emplace_back(new TChain("o2sim"));
108  // add signal files
109  simchains.back()->AddFile(o2::base::DetectorNameConf::getHitsFileName(detid, mSimPrefixes[source].data()).c_str());
110  }
111 
112  // QED part
113  if (mEventRecordsWithQED.size() > 0) {
114  if (mSimPrefixes.size() >= QEDSOURCEID) {
115  LOG(fatal) << "Too many signal chains; crashes with QED source ID";
116  }
117 
118  // it might be better to use an unordered_map for the simchains but this requires interface changes
119  simchains.resize(QEDSOURCEID + 1, nullptr);
120  simchains[QEDSOURCEID] = new TChain("o2sim");
121  simchains[QEDSOURCEID]->AddFile(o2::base::DetectorNameConf::getHitsFileName(detid, mQEDSimPrefix).c_str());
122  }
123 
124  return true;
125 }

149 bool DigitizationContext::checkVertexCompatibility(bool verbose) const
150 {
151  if (mMaxPartNumber == 1) {
152  return true;
153  }
154 
155  auto checkVertexPair = [](math_utils::Point3D<double> const& p1, math_utils::Point3D<double> const& p2) -> bool {
156  return (p2 - p1).Mag2() < 1E-6;
157  };
158 
159  std::vector<TChain*> kinematicschain;
160  std::vector<TBranch*> headerbranches;
161  std::vector<o2::dataformats::MCEventHeader*> headers;
162  std::vector<math_utils::Point3D<double>> vertices;
163  initSimKinematicsChains(kinematicschain);
164  bool consistent = true;
165  if (kinematicschain.size() > 0) {
166  headerbranches.resize(kinematicschain.size(), nullptr);
167  headers.resize(kinematicschain.size(), nullptr);
168  // loop over all collisions in this context
169  int collisionID = 0;
170  for (auto& collision : getEventParts()) {
171  collisionID++;
172  vertices.clear();
173  for (auto& part : collision) {
174  const auto source = part.sourceID;
175  const auto entry = part.entryID;
176  auto chain = kinematicschain[source];
177  if (!headerbranches[source]) {
178  headerbranches[source] = chain->GetBranch("MCEventHeader.");
179  headerbranches[source]->SetAddress(&headers[source]);
180  }
181  // get the MCEventHeader to read out the vertex
182  headerbranches[source]->GetEntry(entry);
183  auto header = headers[source];
184  vertices.emplace_back(header->GetX(), header->GetY(), header->GetZ());
185  }
186  // analyse vertex matching
187  bool thiscollision = true;
188  const auto& p1 = vertices[0];
189  for (int j = 1; j < vertices.size(); ++j) {
190  const auto& p2 = vertices[j];
191  bool thischeck = checkVertexPair(p1, p2);
192  thiscollision &= thischeck;
193  }
194  if (verbose && !thiscollision) {
195  std::stringstream text;
196  text << "Found inconsistent vertices for digit collision ";
197  text << collisionID << " : ";
198  for (auto& p : vertices) {
199  text << p << " ";
200  }
201  LOG(error) << text.str();
202  }
203  consistent &= thiscollision;
204  }
205  }
206  return consistent;
207 }

219 void DigitizationContext::saveToFile(std::string_view filename) const
220 {
221  // checks if the path content of filename exists ... otherwise it is created before creating the ROOT file
222  auto ensure_path_exists = [](std::string_view filename) {
223  try {
224  // Extract the directory path from the filename
225  std::filesystem::path file_path(filename);
226  std::filesystem::path dir_path = file_path.parent_path();
227 
228  // Check if the directory path is empty (which means filename was just a name without path)
229  if (dir_path.empty()) {
230  // nothing to do
231  return true;
232  }
233 
234  // Create directories if they do not exist
235  if (!std::filesystem::exists(dir_path)) {
236  if (std::filesystem::create_directories(dir_path)) {
237  // std::cout << "Directories created successfully: " << dir_path.string() << std::endl;
238  return true;
239  } else {
240  std::cerr << "Failed to create directories: " << dir_path.string() << std::endl;
241  return false;
242  }
243  }
244  return true;
245  } catch (const std::filesystem::filesystem_error& ex) {
246  std::cerr << "Filesystem error: " << ex.what() << std::endl;
247  return false;
248  } catch (const std::exception& ex) {
249  std::cerr << "General error: " << ex.what() << std::endl;
250  return false;
251  }
252  };
253 
254  if (!ensure_path_exists(filename)) {
255  LOG(error) << "Filename contains path component which could not be created";
256  return;
257  }
258 
259  TFile file(filename.data(), "RECREATE");
260  if (file.IsOpen()) {
261  auto cl = TClass::GetClass(typeid(*this));
262  file.WriteObjectAny(this, cl, "DigitizationContext");
263  file.Close();
264  } else {
265  LOG(error) << "Could not write to file " << filename.data();
266  }
267 }

282 void DigitizationContext::fillQED(std::string_view QEDprefix, int max_events, double qedrate)
283 {
284  o2::steer::InteractionSampler qedInteractionSampler;
285  qedInteractionSampler.setBunchFilling(mBCFilling);
286 
287  // get first and last "hadronic" interaction records and let
288  // QED events range from the first bunch crossing to the last bunch crossing
289  // in this range
290  auto first = mEventRecords.front();
291  auto last = mEventRecords.back();
292  first.bc = 0;
293  last.bc = o2::constants::lhc::LHCMaxBunches;
294  LOG(info) << "QED RATE " << qedrate;
295  qedInteractionSampler.setInteractionRate(qedrate);
296  qedInteractionSampler.setFirstIR(first);
297  qedInteractionSampler.init();
298  qedInteractionSampler.print();
299  std::vector<o2::InteractionTimeRecord> qedinteractionrecords;
300  o2::InteractionTimeRecord t;
301  LOG(info) << "GENERATING COL TIMES";
302  t = qedInteractionSampler.generateCollisionTime();
303  while ((t = qedInteractionSampler.generateCollisionTime()) < last) {
304  qedinteractionrecords.push_back(t);
305  }
306  LOG(info) << "DONE GENERATING COL TIMES";
307  fillQED(QEDprefix, qedinteractionrecords, max_events, false);
308 }

310 void DigitizationContext::fillQED(std::string_view QEDprefix, std::vector<o2::InteractionTimeRecord> const& irecord, int max_events, bool fromKinematics)
311 {
312  mQEDSimPrefix = QEDprefix;
313 
314  std::vector<std::vector<o2::steer::EventPart>> qedEventParts;
315 
316  int numberQEDevents = max_events; // if this is -1 there will be no limitation
317  if (fromKinematics) {
318  // we need to fill the QED parts (using a simple round robin scheme)
319  auto qedKinematicsName = o2::base::NameConf::getMCKinematicsFileName(mQEDSimPrefix);
320  // find out how many events are stored
321  TFile f(qedKinematicsName.c_str(), "OPEN");
322  auto t = (TTree*)f.Get("o2sim");
323  if (!t) {
324  LOG(error) << "No QED kinematics found";
325  throw std::runtime_error("No QED kinematics found");
326  }
327  numberQEDevents = t->GetEntries();
328  }
329 
330  int eventID = 0;
331  for (auto& tmp : irecord) {
332  std::vector<EventPart> qedpart;
333  qedpart.emplace_back(QEDSOURCEID, eventID++);
334  qedEventParts.push_back(qedpart);
335  if (eventID == numberQEDevents) {
336  eventID = 0;
337  }
338  }
339 
340  // we need to do the interleaved event records for detectors consuming both
341  // normal and QED events
342  // --> merge both; sort first according to times and sort second one according to same order
343  auto combinedrecords = mEventRecords;
344  combinedrecords.insert(combinedrecords.end(), irecord.begin(), irecord.end());
345  auto combinedparts = mEventParts;
346  combinedparts.insert(combinedparts.end(), qedEventParts.begin(), qedEventParts.end());
347 
348  // get sorted index vector based on event records
349  std::vector<size_t> idx(combinedrecords.size());
350  std::iota(idx.begin(), idx.end(), 0);
351 
352  std::stable_sort(idx.begin(), idx.end(),
353  [&combinedrecords](size_t i1, size_t i2) { return combinedrecords[i1] < combinedrecords[i2]; });
354 
355  mEventRecordsWithQED.clear();
356  mEventPartsWithQED.clear();
357  for (int i = 0; i < idx.size(); ++i) {
358  mEventRecordsWithQED.push_back(combinedrecords[idx[i]]);
359  mEventPartsWithQED.push_back(combinedparts[idx[i]]);
360  }
361 }

365 // a common helper for timeframe structure
366 std::vector<std::pair<int, int>> getTimeFrameBoundaries(std::vector<o2::InteractionTimeRecord> const& irecords, long startOrbit, long orbitsPerTF)
367 {
368  std::vector<std::pair<int, int>> result;
369 
370  // the goal is to determine timeframe boundaries inside the interaction record vectors
371  // determine if we can do anything
372  if (irecords.size() == 0) {
373  // nothing to do
374  return result;
375  }
376 
377  if (irecords.back().orbit < startOrbit) {
378  LOG(error) << "start orbit larger than last collision entry";
379  return result;
380  }
381 
382  // skip to the first index falling within our constrained
383  int left = 0;
384  while (left < irecords.size() && irecords[left].orbit < startOrbit) {
385  left++;
386  }
387 
388  // now we can start (2 pointer approach)
389  auto right = left;
390  int timeframe_count = 1;
391  while (right < irecords.size()) {
392  if (irecords[right].orbit >= startOrbit + timeframe_count * orbitsPerTF) {
393  // we finished one timeframe
394  result.emplace_back(std::pair<int, int>(left, right - 1));
395  timeframe_count++;
396  left = right;
397  }
398  right++;
399  }
400  // finished last timeframe
401  result.emplace_back(std::pair<int, int>(left, right - 1));
402  return result;
403 }

405 // a common helper for timeframe structure - includes indices for orbits-early (orbits from last timeframe still affecting current one)
406 std::vector<std::tuple<int, int, int>> getTimeFrameBoundaries(std::vector<o2::InteractionTimeRecord> const& irecords,
407  long startOrbit,
408  long orbitsPerTF,
409  float orbitsEarly)
410 {
411  // we could actually use the other method first ... then do another pass to fix the early-index ... or impact index
412  auto true_indices = getTimeFrameBoundaries(irecords, startOrbit, orbitsPerTF);
413 
414  std::vector<std::tuple<int, int, int>> indices_with_early{};
415  for (int ti = 0; ti < true_indices.size(); ++ti) {
416  // for each timeframe we copy the true indices
417  auto& tf_range = true_indices[ti];
418 
419  // init new index without fixing the early index yet
420  indices_with_early.push_back(std::make_tuple(tf_range.first, tf_range.second, -1));
421 
422  // from the second timeframe on we can determine the index in the previous timeframe
423  // which matches our criterion
424  if (orbitsEarly > 0. && ti > 0) {
425  auto& prev_tf_range = true_indices[ti - 1];
426  // in this range search the smallest index which precedes
427  // timeframe ti by not more than "orbitsEarly" orbits
428  // (could probably use binary search, in case optimization becomes necessary)
429  int earlyOrbitIndex = prev_tf_range.second;
430 
431  // this is the orbit of the ti-th timeframe start
432  auto orbit_timeframe_start = startOrbit + ti * orbitsPerTF;
433 
434  auto orbit_timeframe_early_fractional = orbit_timeframe_start - orbitsEarly;
435  auto orbit_timeframe_early_integral = (uint32_t)(orbit_timeframe_early_fractional);
436 
437  auto bc_early = (uint32_t)((orbit_timeframe_early_fractional - orbit_timeframe_early_integral) * o2::constants::lhc::LHCMaxBunches);
438 
439  // this is the interaction record of the ti-th timeframe start
440  o2::InteractionRecord timeframe_start_record(0, orbit_timeframe_early_integral);
441  // this is the interaction record in some previous timeframe after which interactions could still
442  // influence the ti-th timeframe according to orbitsEarly
443  o2::InteractionRecord timeframe_early_record(bc_early, orbit_timeframe_early_integral);
444 
445  auto differenceInBCNS_max = timeframe_start_record.differenceInBCNS(timeframe_early_record);
446 
447  for (int j = prev_tf_range.second; j >= prev_tf_range.first; --j) {
448  // determine difference in timing in NS; compare that with the limit given by orbitsEarly
449  auto timediff_NS = timeframe_start_record.differenceInBCNS(irecords[j]);
450  if (timediff_NS < differenceInBCNS_max) {
451  earlyOrbitIndex = j;
452  } else {
453  break;
454  }
455  }
456  std::get<2>(indices_with_early.back()) = earlyOrbitIndex;
457  }
458  }
459  return indices_with_early;
460 }

464 void DigitizationContext::applyMaxCollisionFilter(std::vector<std::tuple<int, int, int>>& timeframeindices, long startOrbit, long orbitsPerTF, int maxColl, double orbitsEarly)
465 {
466  // the idea is to go through each timeframe and throw away collisions beyond a certain count
467  // then the indices should be condensed
468 
469  std::vector<std::vector<o2::steer::EventPart>> newparts;
470  std::vector<o2::InteractionTimeRecord> newrecords;
471 
472  std::unordered_map<int, int> currMaxId; // the max id encountered for a source
473  std::unordered_map<int, std::unordered_map<int, int>> reIndexMap; // for each source, a map of old to new index for the event parts
474 
475  if (maxColl == -1) {
476  maxColl = mEventRecords.size();
477  }
478 
479  // the actual first actual timeframe
480  int first_timeframe = orbitsEarly > 0. ? 1 : 0;
481 
482  // mapping of old to new indices
483  std::unordered_map<size_t, size_t> indices_old_to_new;
484 
485  // now we can go through the structure timeframe by timeframe
486  for (int tf_id = first_timeframe; tf_id < timeframeindices.size(); ++tf_id) {
487  auto& tf_indices = timeframeindices[tf_id];
488 
489  auto firstindex = std::get<0>(tf_indices); // .first;
490  auto lastindex = std::get<1>(tf_indices); // .second;
491  auto previndex = std::get<2>(tf_indices);
492 
493  LOG(info) << "timeframe indices " << previndex << " : " << firstindex << " : " << lastindex;
494 
495  int collCount = 0; // counting collisions within timeframe
496  // copy to new structure
497  for (int index = previndex >= 0 ? previndex : firstindex; index <= lastindex; ++index) {
498  if (collCount >= maxColl) {
499  continue;
500  }
501 
502  // look if this index was already done?
503  // avoid duplicate entries in transformed records
504  if (indices_old_to_new.find(index) != indices_old_to_new.end()) {
505  continue;
506  }
507 
508  // we put these events under a certain condition
509  bool keep = index < firstindex || collCount < maxColl;
510 
511  if (!keep) {
512  continue;
513  }
514 
515  if (index >= firstindex) {
516  collCount++;
517  }
518 
519  // we must also make sure that we don't duplicate the records
520  // moreover some records are merely put as precoll of tf2 ---> so they shouldn't be part of tf1 in the final
521  // extraction, ouch !
522  // maybe we should combine the filter and individual tf extraction in one step !!
523  indices_old_to_new[index] = newrecords.size();
524  newrecords.push_back(mEventRecords[index]);
525  newparts.push_back(mEventParts[index]);
526 
527  // reindex the event parts to achieve compactification (and initial linear increase)
528  for (auto& part : newparts.back()) {
529  auto source = part.sourceID;
530  auto entry = part.entryID;
531  // have we remapped this entry before?
532  if (reIndexMap.find(source) != reIndexMap.end() && reIndexMap[source].find(entry) != reIndexMap[source].end()) {
533  part.entryID = reIndexMap[source][entry];
534  } else {
535  // assign the next free index
536  if (currMaxId.find(source) == currMaxId.end()) {
537  currMaxId[source] = 0;
538  }
539  part.entryID = currMaxId[source];
540  // cache this assignment
541  reIndexMap[source][entry] = currMaxId[source];
542  currMaxId[source] += 1;
543  }
544  }
545  } // ends one timeframe
546 
547  // correct the timeframe indices
548  if (indices_old_to_new.find(firstindex) != indices_old_to_new.end()) {
549  std::get<0>(tf_indices) = indices_old_to_new[firstindex]; // start
550  }
551  if (indices_old_to_new.find(lastindex) != indices_old_to_new.end()) {
552  std::get<1>(tf_indices) = indices_old_to_new[lastindex]; // end;
553  } else {
554  std::get<1>(tf_indices) = newrecords.size(); // end;
555  }
556  if (indices_old_to_new.find(previndex) != indices_old_to_new.end()) {
557  std::get<2>(tf_indices) = indices_old_to_new[previndex]; // previous or "early" index
558  }
559  }
560  // reassignment
561  mEventRecords = newrecords;
562  mEventParts = newparts;
563 }

612 void DigitizationContext::sampleInteractionVertices(o2::dataformats::MeanVertexObject const& meanv)
613 {
614  // mapping of source x event --> index into mInteractionVertices
615  std::unordered_map<std::pair<int, int>, int, pair_hash> vertex_cache;
616 
617  mInteractionVertices.clear();
618  int collcount = 0;
619 
620  std::unordered_set<int> vset; // used to detect vertex incompatibilities
621  for (auto& coll : mEventParts) {
622  collcount++;
623  vset.clear();
624 
625  // first detect if any of these entries already has an associated vertex
626  for (auto& part : coll) {
627  auto source = part.sourceID;
628  auto event = part.entryID;
629  auto cached_iter = vertex_cache.find(std::pair<int, int>(source, event));
630 
631  if (cached_iter != vertex_cache.end()) {
632  vset.emplace(cached_iter->second);
633  }
634  }
635 
636  // make sure that there is no conflict
637  if (vset.size() > 1) {
638  LOG(fatal) << "Impossible conflict during interaction vertex sampling";
639  }
640 
641  int cacheindex = -1;
642  if (vset.size() == 1) {
643  // all of the parts need to be assigned the same existing vertex
644  cacheindex = *(vset.begin());
645  mInteractionVertices.push_back(mInteractionVertices[cacheindex]);
646  } else {
647  // we need to sample a new point
648  mInteractionVertices.emplace_back(meanv.sample());
649  cacheindex = mInteractionVertices.size() - 1;
650  }
651  // update the cache
652  for (auto& part : coll) {
653  auto source = part.sourceID;
654  auto event = part.entryID;
655  vertex_cache[std::pair<int, int>(source, event)] = cacheindex;
656  }
657  }
658 }

660 DigitizationContext DigitizationContext::extractSingleTimeframe(int timeframeid, std::vector<std::tuple<int, int, int>> const& timeframeindices, std::vector<int> const& sources_to_offset)
661 {
662  DigitizationContext r; // make a return object
663  if (timeframeindices.size() == 0) {
664  LOG(error) << "Timeframe index structure empty; Returning empty object.";
665  return r;
666  }
667  r.mSimPrefixes = mSimPrefixes;
668  r.mMuBC = mMuBC;
669  try {
670  auto tf_ranges = timeframeindices.at(timeframeid);
671 
672  auto startindex = std::get<0>(tf_ranges);
673  auto endindex = std::get<1>(tf_ranges);
674  auto earlyindex = std::get<2>(tf_ranges);
675 
676  if (earlyindex >= 0) {
677  startindex = earlyindex;
678  }
679  std::copy(mEventRecords.begin() + startindex, mEventRecords.begin() + endindex, std::back_inserter(r.mEventRecords));
680  std::copy(mEventParts.begin() + startindex, mEventParts.begin() + endindex, std::back_inserter(r.mEventParts));
681  if (mInteractionVertices.size() > endindex) {
682  std::copy(mInteractionVertices.begin() + startindex, mInteractionVertices.begin() + endindex, std::back_inserter(r.mInteractionVertices));
683  }
684 
685  // let's assume we want to fix the ids for source = source_id
686  // Then we find the first index that has this source_id and take the corresponding number
687  // as offset. Thereafter we subtract this offset from all known event parts.
688  auto perform_offsetting = [&r](int source_id) {
689  auto indices_for_source = r.getCollisionIndicesForSource(source_id);
690  int minvalue = std::numeric_limits<int>::max();
691  for (auto& p : indices_for_source) {
692  if (p.first < minvalue) {
693  minvalue = p.first;
694  }
695  }
696  // now fix them
697  for (auto& p : indices_for_source) {
698  auto index_into_mEventParts = p.second;
699  for (auto& part : r.mEventParts[index_into_mEventParts]) {
700  if (part.sourceID == source_id) {
701  part.entryID -= minvalue;
702  }
703  }
704  }
705  };
706  for (auto source_id : sources_to_offset) {
707  perform_offsetting(source_id);
708  }
709 
710  } catch (std::exception) {
711  LOG(warn) << "No such timeframe id in collision context. Returing empty object";
712  }
713  // fix number of collisions
714  r.setNCollisions(r.mEventRecords.size());
715  return r;
716 }
