22void ZDCTowerParam::setTowerCalib(uint32_t ich, float val, bool ismodified)
23{
24 bool in_list = false;
25 for (int il = 0; il < ChTowerCalib.size(); il++) {
26 if (ich == ChTowerCalib[il]) {
27 in_list = true;
28 break;
