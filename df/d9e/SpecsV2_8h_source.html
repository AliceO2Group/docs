177constexpr unsigned int l2IDStart{l1IDEnd + 1}; //< Start DetID layer 2

180constexpr unsigned int nChips{l2IDEnd + 1}; //< number of Chips (PixelArrays) in IB

182template <typename T = int>
183inline T getDetID2Layer(T detID)
184{
185 if (static_cast<T>(l0IDStart) <= detID && detID <= static_cast<T>(l0IDEnd)) {
186 return 0;
187 } else if (static_cast<T>(l1IDStart) <= detID && detID <= static_cast<T>(l1IDEnd)) {
188 return 1;
189 } else if (static_cast<T>(l2IDStart) <= detID && detID <= static_cast<T>(l2IDEnd)) {
190 return 2;
191 }
192 return -1;
193}
194
195template <typename T = int>
196inline T getSensorID(T detID)
197{
198 auto layer = getDetID2Layer(detID);
199 if (layer == 0) {
200 return ((detID - l0IDStart) < static_cast<T>(l0IDTot) / 2) ? 0 : 1;
201 } else if (layer == 1) {
202 return ((detID - l1IDStart) < static_cast<T>(l1IDTot) / 2) ? 2 : 3;
203 } else if (layer == 2) {
204 return ((detID - l2IDStart) < static_cast<T>(l2IDTot) / 2) ? 4 : 5;
205 }
206 return -1;
207}
208
209template <typename T = int>
210inline bool isDetITS3(T detID)
211{
212 return detID < static_cast<T>(nChips);
213}
