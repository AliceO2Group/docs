334 template <size_t N, std::array<soa::TableRef, N> refs>

337  return [&]<size_t... Is>(std::index_sequence<Is...>) -> std::vector<std::shared_ptr<arrow::Table>> {

342 template <is_spawnable T>
343 struct OutputManager<Spawns<T>> {
344  static bool appendOutput(std::vector<OutputSpec>& outputs, Spawns<T>& what, uint32_t)
345  {
346  outputs.emplace_back(what.spec());
347  return true;
348  }
349 
350  static bool prepare(ProcessingContext& pc, Spawns<T>& what)
351  {
352  using metadata = o2::aod::MetadataTrait<o2::aod::Hash<T::ref.desc_hash>>::metadata;
353  auto originalTable = soa::ArrowHelpers::joinTables(extractOriginals<metadata::sources.size(), metadata::sources>(pc));
354  if (originalTable->schema()->fields().empty() == true) {
355  using base_table_t = typename Spawns<T>::base_table_t::table_t;
356  originalTable = makeEmptyTable<base_table_t>(o2::aod::label<metadata::extension_table_t::ref>());
357  }
358 
359  what.extension = std::make_shared<typename Spawns<T>::extension_t>(o2::framework::spawner<o2::aod::Hash<metadata::extension_table_t::ref.desc_hash>>(originalTable, o2::aod::label<metadata::extension_table_t::ref>()));
360  what.table = std::make_shared<typename T::table_t>(soa::ArrowHelpers::joinTables({what.extension->asArrowTable(), originalTable}));
361  return true;
362  }
363 
364  static bool finalize(ProcessingContext& pc, Spawns<T>& what)
365  {
366  pc.outputs().adopt(what.output(), what.asArrowTable());
367  return true;
368  }
369 
370  static bool postRun(EndOfStreamContext&, Spawns<T>&)
371  {
372  return true;
373  }

648 template <typename T, bool OPT, bool SORTED>
649 struct PresliceManager<PresliceBase<T, OPT, SORTED>> {
650  static bool registerCache(PresliceBase<T, OPT, SORTED>& container, std::vector<StringPair>& bsks, std::vector<StringPair>& bsksU)
651  {
652  if constexpr (OPT) {
653  if (container.binding == "[MISSING]") {
654  return true;
655  }
656  }
657  if constexpr (SORTED) {
658  auto locate = std::find_if(bsks.begin(), bsks.end(), [&](auto const& entry) { return (entry.first == container.bindingKey.first) && (entry.second == container.bindingKey.second); });
659  if (locate == bsks.end()) {
660  bsks.emplace_back(container.getBindingKey());
661  }
662  return true;
663  } else {
664  auto locate = std::find_if(bsksU.begin(), bsksU.end(), [&](auto const& entry) { return (entry.first == container.bindingKey.first) && (entry.second == container.bindingKey.second); });
665  if (locate == bsksU.end()) {
666  bsksU.emplace_back(container.getBindingKey());
667  }
668  return true;
669  }
670  }
671 
672  static bool updateSliceInfo(PresliceBase<T, OPT, SORTED>& container, ArrowTableSlicingCache& cache)
673  {
674  if constexpr (OPT) {
675  if (container.binding == "[MISSING]") {
676  return true;
677  }
678  }
679  if constexpr (SORTED) {
680  container.updateSliceInfo(cache.getCacheFor(container.getBindingKey()));
681  return true;
682  } else {
683  container.updateSliceInfo(cache.getCacheUnsortedFor(container.getBindingKey()));
684  return true;
685  }
686  }
