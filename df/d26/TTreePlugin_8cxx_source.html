248arrow::Result<arrow::RecordBatchGenerator> TTreeFileFormat::ScanBatchesAsync(
249 const std::shared_ptr<arrow::dataset::ScanOptions>& options,
250 const std::shared_ptr<arrow::dataset::FileFragment>& fragment) const
251{
252 // Get the fragment as a TTreeFragment. This might be PART of a TTree.
253 auto treeFragment = std::dynamic_pointer_cast<TTreeFileFragment>(fragment);
254 // This is the schema we want to read
255 auto dataset_schema = options->dataset_schema;
256
257 auto generator = [pool = options->pool, treeFragment, dataset_schema, &totalCompressedSize = mTotCompressedSize,
258 &totalUncompressedSize = mTotUncompressedSize]() -> arrow::Future<std::shared_ptr<arrow::RecordBatch>> {
259 auto schema = treeFragment->format()->Inspect(treeFragment->source());
260
261 std::vector<std::shared_ptr<arrow::Array>> columns;
262 std::vector<std::shared_ptr<arrow::Field>> fields = dataset_schema->fields();
263 auto physical_schema = *treeFragment->ReadPhysicalSchema();
264
265 static TBufferFile buffer{TBuffer::EMode::kWrite, 4 * 1024 * 1024};
266 auto containerFS = std::dynamic_pointer_cast<VirtualRootFileSystemBase>(treeFragment->source().filesystem());
267 auto fs = std::dynamic_pointer_cast<TTreeFileSystem>(containerFS->GetSubFilesystem(treeFragment->source()));
268
269 int64_t rows = -1;
270 auto& tree = fs->GetTree(treeFragment->source());
271 for (auto& field : fields) {
272 // The field actually on disk
273 auto physicalField = physical_schema->GetFieldByName(field->name());
274 TBranch* branch = tree->GetBranch(physicalField->name().c_str());
275 assert(branch);
276 buffer.Reset();
277 auto totalEntries = branch->GetEntries();
278 if (rows == -1) {
279 rows = totalEntries;
280 }
281 if (rows != totalEntries) {
282 throw runtime_error_f("Unmatching number of rows for branch %s", branch->GetName());
283 }
284 arrow::Status status;
285 int readEntries = 0;
286 std::shared_ptr<arrow::Array> array;
287 auto listType = std::dynamic_pointer_cast<arrow::FixedSizeListType>(physicalField->type());
288 if (physicalField->type() == arrow::boolean() ||
289 (listType && physicalField->type()->field(0)->type() == arrow::boolean())) {
290 if (listType) {
291 std::unique_ptr<arrow::ArrayBuilder> builder = nullptr;
292 auto status = arrow::MakeBuilder(pool, physicalField->type()->field(0)->type(), &builder);
293 if (!status.ok()) {
294 throw runtime_error("Cannot create value builder");
295 }
296 auto listBuilder = std::make_unique<arrow::FixedSizeListBuilder>(pool, std::move(builder), listType->list_size());
297 auto valueBuilder = listBuilder.get()->value_builder();
298 // boolean array special case: we need to use builder to create the bitmap
299 status = valueBuilder->Reserve(totalEntries * listType->list_size());
300 status &= listBuilder->Reserve(totalEntries);
301 if (!status.ok()) {
302 throw runtime_error("Failed to reserve memory for array builder");
303 }
304 while (readEntries < totalEntries) {
305 auto readLast = branch->GetBulkRead().GetBulkEntries(readEntries, buffer);
306 readEntries += readLast;
307 status &= static_cast<arrow::BooleanBuilder*>(valueBuilder)->AppendValues(reinterpret_cast<uint8_t const*>(buffer.GetCurrent()), readLast * listType->list_size());
308 }
309 status &= static_cast<arrow::FixedSizeListBuilder*>(listBuilder.get())->AppendValues(readEntries);
310 if (!status.ok()) {
311 throw runtime_error("Failed to append values to array");
312 }
313 status &= listBuilder->Finish(&array);
314 if (!status.ok()) {
315 throw runtime_error("Failed to create array");
316 }
317 } else if (listType == nullptr) {
318 std::unique_ptr<arrow::ArrayBuilder> builder = nullptr;
319 auto status = arrow::MakeBuilder(pool, physicalField->type(), &builder);
320 if (!status.ok()) {
321 throw runtime_error("Cannot create builder");
322 }
323 auto valueBuilder = static_cast<arrow::BooleanBuilder*>(builder.get());
324 // boolean array special case: we need to use builder to create the bitmap
325 status = valueBuilder->Reserve(totalEntries);
326 if (!status.ok()) {
327 throw runtime_error("Failed to reserve memory for array builder");
328 }
329 while (readEntries < totalEntries) {
330 auto readLast = branch->GetBulkRead().GetBulkEntries(readEntries, buffer);
331 readEntries += readLast;
332 status &= valueBuilder->AppendValues(reinterpret_cast<uint8_t const*>(buffer.GetCurrent()), readLast);
333 }
334 if (!status.ok()) {
335 throw runtime_error("Failed to append values to array");
336 }
337 status &= valueBuilder->Finish(&array);
338 if (!status.ok()) {
339 throw runtime_error("Failed to create array");
340 }
341 }
342 } else {
343 // other types: use serialized read to build arrays directly.
344 auto typeSize = physicalField->type()->byte_width();
345 // This is needed for branches which have not been persisted.
346 auto bytes = branch->GetTotBytes();
347 auto branchSize = bytes ? bytes : 1000000;
348 auto&& result = arrow::AllocateResizableBuffer(branchSize, pool);
349 if (!result.ok()) {
350 throw runtime_error("Cannot allocate values buffer");
351 }
352 std::shared_ptr<arrow::Buffer> arrowValuesBuffer = std::move(result).ValueUnsafe();
353 auto ptr = arrowValuesBuffer->mutable_data();
354 if (ptr == nullptr) {
355 throw runtime_error("Invalid buffer");
356 }
357
358 std::unique_ptr<TBufferFile> offsetBuffer = nullptr;
359
360 uint32_t offset = 0;
361 int count = 0;
362 std::shared_ptr<arrow::Buffer> arrowOffsetBuffer;
363 std::span<int> offsets;
364 int size = 0;
365 uint32_t totalSize = 0;
366 TBranch* mSizeBranch = nullptr;
367 int64_t listSize = 1;
368 if (auto fixedSizeList = std::dynamic_pointer_cast<arrow::FixedSizeListType>(physicalField->type())) {
369 listSize = fixedSizeList->list_size();
370 typeSize = fixedSizeList->field(0)->type()->byte_width();
371 } else if (auto vlaListType = std::dynamic_pointer_cast<arrow::ListType>(physicalField->type())) {
372 listSize = -1;
373 typeSize = vlaListType->field(0)->type()->byte_width();
374 }
375 if (listSize == -1) {
376 mSizeBranch = branch->GetTree()->GetBranch((std::string{branch->GetName()} + "_size").c_str());
377 offsetBuffer = std::make_unique<TBufferFile>(TBuffer::EMode::kWrite, 4 * 1024 * 1024);
378 result = arrow::AllocateResizableBuffer((totalEntries + 1) * (int64_t)sizeof(int), pool);
379 if (!result.ok()) {
380 throw runtime_error("Cannot allocate offset buffer");
381 }
382 arrowOffsetBuffer = std::move(result).ValueUnsafe();
383 unsigned char* ptrOffset = arrowOffsetBuffer->mutable_data();
384 auto* tPtrOffset = reinterpret_cast<int*>(ptrOffset);
385 offsets = std::span<int>{tPtrOffset, tPtrOffset + totalEntries + 1};
386
387 // read sizes first
388 while (readEntries < totalEntries) {
389 auto readLast = mSizeBranch->GetBulkRead().GetEntriesSerialized(readEntries, *offsetBuffer);
390 readEntries += readLast;
391 for (auto i = 0; i < readLast; ++i) {
392 offsets[count++] = (int)offset;
393 offset += swap32_(reinterpret_cast<uint32_t*>(offsetBuffer->GetCurrent())[i]);
394 }
395 }
396 offsets[count] = (int)offset;
397 totalSize = offset;
398 readEntries = 0;
399 }
400
401 while (readEntries < totalEntries) {
402 auto readLast = branch->GetBulkRead().GetEntriesSerialized(readEntries, buffer);
403 if (listSize == -1) {
404 size = offsets[readEntries + readLast] - offsets[readEntries];
405 } else {
406 size = readLast * listSize;
407 }
408 readEntries += readLast;
409 swapCopy(ptr, buffer.GetCurrent(), size, typeSize);
410 ptr += (ptrdiff_t)(size * typeSize);
411 }
412 if (listSize >= 1) {
413 totalSize = readEntries * listSize;
414 }
415 std::shared_ptr<arrow::PrimitiveArray> varray;
416 switch (listSize) {
417 case -1:
418 varray = std::make_shared<arrow::PrimitiveArray>(physicalField->type()->field(0)->type(), totalSize, arrowValuesBuffer);
419 array = std::make_shared<arrow::ListArray>(physicalField->type(), readEntries, arrowOffsetBuffer, varray);
420 break;
421 case 1:
422 array = std::make_shared<arrow::PrimitiveArray>(physicalField->type(), readEntries, arrowValuesBuffer);
423 break;
424 default:
425 varray = std::make_shared<arrow::PrimitiveArray>(physicalField->type()->field(0)->type(), totalSize, arrowValuesBuffer);
426 array = std::make_shared<arrow::FixedSizeListArray>(physicalField->type(), readEntries, varray);
427 }
428 }
429
430 branch->SetStatus(false);
431 branch->DropBaskets("all");
432 branch->Reset();
433 branch->GetTransientBuffer(0)->Expand(0);
434
435 columns.push_back(array);
436 }
437 auto batch = arrow::RecordBatch::Make(dataset_schema, rows, columns);
438 totalCompressedSize += tree->GetZipBytes();
439 totalUncompressedSize += tree->GetTotBytes();
440 return batch;
441 };
442 return generator;
443}

496auto arrowTypeFromROOT(EDataType type, int size)
497{
498 auto typeGenerator = [](std::shared_ptr<arrow::DataType> const& type, int size) -> std::shared_ptr<arrow::DataType> {
499 switch (size) {
500 case -1:
501 return arrow::list(type);
502 case 1:
503 return std::move(type);
504 default:
505 return arrow::fixed_size_list(type, size);
506 }
507 };
508
509 switch (type) {
510 case EDataType::kBool_t:
511 return typeGenerator(arrow::boolean(), size);
512 case EDataType::kUChar_t:
513 return typeGenerator(arrow::uint8(), size);
514 case EDataType::kUShort_t:
515 return typeGenerator(arrow::uint16(), size);
516 case EDataType::kUInt_t:
517 return typeGenerator(arrow::uint32(), size);
518 case EDataType::kULong64_t:
519 return typeGenerator(arrow::uint64(), size);
520 case EDataType::kChar_t:
521 return typeGenerator(arrow::int8(), size);
522 case EDataType::kShort_t:
523 return typeGenerator(arrow::int16(), size);
524 case EDataType::kInt_t:
525 return typeGenerator(arrow::int32(), size);
526 case EDataType::kLong64_t:
527 return typeGenerator(arrow::int64(), size);
528 case EDataType::kFloat_t:
529 return typeGenerator(arrow::float32(), size);
530 case EDataType::kDouble_t:
531 return typeGenerator(arrow::float64(), size);
532 default:
533 throw o2::framework::runtime_error_f("Unsupported branch type: %d", static_cast<int>(type));
534 }
535}

537arrow::Result<std::shared_ptr<arrow::Schema>> TTreeFileFormat::Inspect(const arrow::dataset::FileSource& source) const
538{
539 arrow::Schema schema{{}};
540 auto fs = std::dynamic_pointer_cast<VirtualRootFileSystemBase>(source.filesystem());
541 // Actually get the TTree from the ROOT file.
542 auto treeFs = std::dynamic_pointer_cast<TTreeFileSystem>(fs->GetSubFilesystem(source));
543 if (!treeFs.get()) {
544 throw runtime_error_f("Unknown filesystem %s\n", source.filesystem()->type_name().c_str());
545 }
546 auto& tree = treeFs->GetTree(source);
547
548 auto branches = tree->GetListOfBranches();
549 auto n = branches->GetEntries();
550
551 std::vector<BranchInfo> branchInfos;
552 for (auto i = 0; i < n; ++i) {
553 auto branch = static_cast<TBranch*>(branches->At(i));
554 auto name = std::string{branch->GetName()};
555 auto pos = name.find("_size");
556 if (pos != std::string::npos) {
557 name.erase(pos);
558 branchInfos.emplace_back(BranchInfo{name, (TBranch*)nullptr, true});
559 } else {
560 auto lookup = std::find_if(branchInfos.begin(), branchInfos.end(), [&](BranchInfo const& bi) {
561 return bi.name == name;
562 });
563 if (lookup == branchInfos.end()) {
564 branchInfos.emplace_back(BranchInfo{name, branch, false});
565 } else {
566 lookup->ptr = branch;
567 }
568 }
569 }
570
571 std::vector<std::shared_ptr<arrow::Field>> fields;
572 tree->SetCacheSize(25000000);
573 for (auto& bi : branchInfos) {
574 static TClass* cls;
575 EDataType type;
576 bi.ptr->GetExpectedType(cls, type);
577 auto listSize = -1;
578 if (!bi.mVLA) {
579 listSize = static_cast<TLeaf*>(bi.ptr->GetListOfLeaves()->At(0))->GetLenStatic();
580 }
581 auto field = std::make_shared<arrow::Field>(bi.ptr->GetName(), arrowTypeFromROOT(type, listSize));
582 fields.push_back(field);
583
584 tree->AddBranchToCache(bi.ptr);
585 if (strncmp(bi.ptr->GetName(), "fIndexArray", strlen("fIndexArray")) == 0) {
586 std::string sizeBranchName = bi.ptr->GetName();
587 sizeBranchName += "_size";
588 auto* sizeBranch = (TBranch*)tree->GetBranch(sizeBranchName.c_str());
589 if (sizeBranch) {
590 tree->AddBranchToCache(sizeBranch);
591 }
592 }
593 }
594 tree->StopCacheLearningPhase();
595
596 return std::make_shared<arrow::Schema>(fields);
597}

612class TTreeFileWriter : public arrow::dataset::FileWriter
613{
614 std::vector<TBranch*> branches;
615 std::vector<TBranch*> sizesBranches;
616 std::vector<std::shared_ptr<arrow::Array>> valueArrays;
617 std::vector<std::shared_ptr<arrow::Array>> sizeArrays;
618 std::vector<std::shared_ptr<arrow::DataType>> valueTypes;
619
620 std::vector<int64_t> valuesIdealBasketSize;
621 std::vector<int64_t> sizeIdealBasketSize;
622
623 std::vector<int64_t> typeSizes;
624 std::vector<int64_t> listSizes;
625 bool firstBasket = true;
626
627 // This is to create a batsket size according to the first batch.
628 void finaliseBasketSize(std::shared_ptr<arrow::RecordBatch> firstBatch)
629 {
630 O2_SIGNPOST_ID_FROM_POINTER(sid, root_arrow_fs, this);
631 O2_SIGNPOST_START(root_arrow_fs, sid, "finaliseBasketSize", "First batch with %lli rows received and %zu columns",
632 firstBatch->num_rows(), firstBatch->columns().size());
633 for (size_t i = 0; i < branches.size(); i++) {
634 auto* branch = branches[i];
635 auto* sizeBranch = sizesBranches[i];
636
637 int valueSize = valueTypes[i]->byte_width();
638 if (listSizes[i] == 1) {
639 O2_SIGNPOST_EVENT_EMIT(root_arrow_fs, sid, "finaliseBasketSize", "Branch %s exists and uses %d bytes per entry for %lli entries.",
640 branch->GetName(), valueSize, firstBatch->num_rows());
641 assert(sizeBranch == nullptr);
642 branch->SetBasketSize(1024 + firstBatch->num_rows() * valueSize);
643 } else if (listSizes[i] == -1) {
644 O2_SIGNPOST_EVENT_EMIT(root_arrow_fs, sid, "finaliseBasketSize", "Branch %s exists and uses %d bytes per entry.",
645 branch->GetName(), valueSize);
646 // This should probably lookup the
647 auto column = firstBatch->GetColumnByName(schema_->field(i)->name());
648 auto list = std::static_pointer_cast<arrow::ListArray>(column);
649 O2_SIGNPOST_EVENT_EMIT(root_arrow_fs, sid, "finaliseBasketSize", "Branch %s needed. Associated size branch %s and there are %lli entries of size %d in that list.",
650 branch->GetName(), sizeBranch->GetName(), list->length(), valueSize);
651 branch->SetBasketSize(1024 + firstBatch->num_rows() * valueSize * list->length());
652 sizeBranch->SetBasketSize(1024 + firstBatch->num_rows() * 4);
653 } else {
654 O2_SIGNPOST_EVENT_EMIT(root_arrow_fs, sid, "finaliseBasketSize", "Branch %s needed. There are %lli entries per array of size %d in that list.",
655 branch->GetName(), listSizes[i], valueSize);
656 assert(sizeBranch == nullptr);
657 branch->SetBasketSize(1024 + firstBatch->num_rows() * valueSize * listSizes[i]);
658 }
659
660 auto field = firstBatch->schema()->field(i);
661 if (field->name().starts_with("fIndexArray")) {
662 // One int per array to keep track of the size
663 int idealBasketSize = 4 * firstBatch->num_rows() + 1024 + field->type()->byte_width() * firstBatch->num_rows(); // minimal additional size needed, otherwise we get 2 baskets
664 int basketSize = std::max(32000, idealBasketSize); // keep a minimum value
665 sizeBranch->SetBasketSize(basketSize);
666 branch->SetBasketSize(basketSize);
667 }
668 }
669 O2_SIGNPOST_END(root_arrow_fs, sid, "finaliseBasketSize", "Done");
670 }

675 TTreeFileWriter(std::shared_ptr<arrow::Schema> schema, std::shared_ptr<arrow::dataset::FileWriteOptions> options,
676 std::shared_ptr<arrow::io::OutputStream> destination,
677 arrow::fs::FileLocator destination_locator)
678 : FileWriter(schema, options, destination, destination_locator)
679 {
680 // Batches have the same number of entries for each column.
681 auto directoryStream = std::dynamic_pointer_cast<TDirectoryFileOutputStream>(destination_);
682 auto treeStream = std::dynamic_pointer_cast<TTreeOutputStream>(destination_);
683
684 if (directoryStream.get()) {
685 TDirectoryFile* dir = directoryStream->GetDirectory();
686 dir->cd();
687 auto* tree = new TTree(destination_locator_.path.c_str(), "");
688 treeStream = std::make_shared<TTreeOutputStream>(tree, "");
689 } else if (treeStream.get()) {
690 // We already have a tree stream, let's derive a new one
691 // with the destination_locator_.path as prefix for the branches
692 // This way we can multiplex multiple tables in the same tree.
693 auto* tree = treeStream->GetTree();
694 treeStream = std::make_shared<TTreeOutputStream>(tree, destination_locator_.path);
695 } else {
696 // I could simply set a prefix here to merge to an already existing tree.
697 throw std::runtime_error("Unsupported backend.");
698 }
699
700 for (auto i = 0u; i < schema->fields().size(); ++i) {
701 auto& field = schema->field(i);
702 listSizes.push_back(1);
703
704 int valuesIdealBasketSize = 0;
705 // Construct all the needed branches.
706 switch (field->type()->id()) {
707 case arrow::Type::FIXED_SIZE_LIST: {
708 listSizes.back() = std::static_pointer_cast<arrow::FixedSizeListType>(field->type())->list_size();
709 valuesIdealBasketSize = 1024 + valueTypes.back()->byte_width() * listSizes.back();
710 valueTypes.push_back(field->type()->field(0)->type());
711 sizesBranches.push_back(nullptr);
712 std::string leafList = fmt::format("{}[{}]{}", field->name(), listSizes.back(), rootSuffixFromArrow(valueTypes.back()->id()));
713 branches.push_back(treeStream->CreateBranch(field->name().c_str(), leafList.c_str()));
714 } break;
715 case arrow::Type::LIST: {
716 valueTypes.push_back(field->type()->field(0)->type());
717 std::string leafList = fmt::format("{}[{}_size]{}", field->name(), field->name(), rootSuffixFromArrow(valueTypes.back()->id()));
718 listSizes.back() = -1; // VLA, we need to calculate it on the fly;
719 std::string sizeLeafList = field->name() + "_size/I";
720 sizesBranches.push_back(treeStream->CreateBranch((field->name() + "_size").c_str(), sizeLeafList.c_str()));
721 branches.push_back(treeStream->CreateBranch(field->name().c_str(), leafList.c_str()));
722 // Notice that this could be replaced by a better guess of the
723 // average size of the list elements, but this is not trivial.
724 } break;
725 default: {
726 valueTypes.push_back(field->type());
727 std::string leafList = field->name() + rootSuffixFromArrow(valueTypes.back()->id());
728 sizesBranches.push_back(nullptr);
729 branches.push_back(treeStream->CreateBranch(field->name().c_str(), leafList.c_str()));
730 } break;
731 }
732 }
733 // We create the branches from the schema
734 }

736 arrow::Status Write(const std::shared_ptr<arrow::RecordBatch>& batch) override
737 {
738 if (firstBasket) {
739 firstBasket = false;
740 finaliseBasketSize(batch);
741 }
742
743 // Support writing empty tables
744 if (batch->columns().empty() || batch->num_rows() == 0) {
745 return arrow::Status::OK();
746 }
747
748 // Batches have the same number of entries for each column.
749 auto directoryStream = std::dynamic_pointer_cast<TDirectoryFileOutputStream>(destination_);
750 TTree* tree = nullptr;
751 if (directoryStream.get()) {
752 TDirectoryFile* dir = directoryStream->GetDirectory();
753 tree = (TTree*)dir->Get(destination_locator_.path.c_str());
754 }
755 auto treeStream = std::dynamic_pointer_cast<TTreeOutputStream>(destination_);
756
757 if (!tree) {
758 // I could simply set a prefix here to merge to an already existing tree.
759 throw std::runtime_error("Unsupported backend.");
760 }
761
762 for (auto i = 0u; i < batch->columns().size(); ++i) {
763 auto column = batch->column(i);
764 auto& field = batch->schema()->field(i);
765
766 valueArrays.push_back(nullptr);
767
768 switch (field->type()->id()) {
769 case arrow::Type::FIXED_SIZE_LIST: {
770 auto list = std::static_pointer_cast<arrow::FixedSizeListArray>(column);
771 valueArrays.back() = list->values();
772 } break;
773 case arrow::Type::LIST: {
774 auto list = std::static_pointer_cast<arrow::ListArray>(column);
775 valueArrays.back() = list;
776 } break;
777 case arrow::Type::BOOL: {
778 // In case of arrays of booleans, we need to go back to their
779 // char based representation for ROOT to save them.
780 auto boolArray = std::static_pointer_cast<arrow::BooleanArray>(column);
781
782 int64_t length = boolArray->length();
783 arrow::UInt8Builder builder;
784 auto ok = builder.Reserve(length);
785
786 for (int64_t i = 0; i < length; ++i) {
787 if (boolArray->IsValid(i)) {
788 // Expand each boolean value (true/false) to uint8 (1/0)
789 uint8_t value = boolArray->Value(i) ? 1 : 0;
790 auto ok = builder.Append(value);
791 } else {
792 // Append null for invalid entries
793 auto ok = builder.AppendNull();
794 }
795 }
796 valueArrays.back() = *builder.Finish();
797 } break;
798 default:
799 valueArrays.back() = column;
800 }
801 }
802
803 int64_t pos = 0;
804 while (pos < batch->num_rows()) {
805 for (size_t bi = 0; bi < branches.size(); ++bi) {
806 auto* branch = branches[bi];
807 auto* sizeBranch = sizesBranches[bi];
808 auto array = batch->column(bi);
809 auto& field = batch->schema()->field(bi);
810 auto& listSize = listSizes[bi];
811 auto valueType = valueTypes[bi];
812 auto valueArray = valueArrays[bi];
813
814 switch (field->type()->id()) {
815 case arrow::Type::LIST: {
816 auto list = std::static_pointer_cast<arrow::ListArray>(array);
817 listSize = list->value_length(pos);
818 uint8_t const* buffer = std::static_pointer_cast<arrow::PrimitiveArray>(valueArray)->values()->data() + array->offset() + list->value_offset(pos) * valueType->byte_width();
819 branch->SetAddress((void*)buffer);
820 sizeBranch->SetAddress(&listSize);
821 };
822 break;
823 case arrow::Type::FIXED_SIZE_LIST:
824 default: {
825 uint8_t const* buffer = std::static_pointer_cast<arrow::PrimitiveArray>(valueArray)->values()->data() + array->offset() + pos * listSize * valueType->byte_width();
826 branch->SetAddress((void*)buffer);
827 };
828 }
829 }
830 tree->Fill();
831 ++pos;
832 }
833 return arrow::Status::OK();
834 }

837 {
838 auto treeStream = std::dynamic_pointer_cast<TTreeOutputStream>(destination_);
839 auto* tree = treeStream->GetTree();
840 tree->Write("", TObject::kOverwrite);
841 tree->SetDirectory(nullptr);
842
843 return {};
844 };
