216 template <int nLayers>
217 template <Task task>
218 size_t TimeFrameGPU<nLayers>::loadChunkData(const size_t chunk, const size_t offset, const size_t maxRofs) // offset: readout frame to start from, maxRofs: to manage boundaries
219 {
220  size_t nRof{0};
221 
222  mMemChunks[chunk].reset(task, mGpuStreams[chunk]); // Reset chunks memory
223  if constexpr ((bool)task) {
224  nRof = mMemChunks[chunk].loadDataOnDevice(offset, maxRofs, 3, mGpuStreams[chunk]);
225  } else {
226  nRof = mMemChunks[chunk].loadDataOnDevice(offset, maxRofs, nLayers, mGpuStreams[chunk]);
227  }
228  LOGP(debug, "In chunk {}: loaded {} readout frames starting from {}", chunk, nRof, offset);
229  return nRof;
230 }

232 template <int nLayers>
233 inline int TimeFrameGPU<nLayers>::getNClustersInRofSpan(const int rofIdstart, const int rofSpanSize, const int layerId) const
234 {
235  return static_cast<int>(mROFramesClusters[layerId][(rofIdstart + rofSpanSize) < mROFramesClusters.size() ? rofIdstart + rofSpanSize : mROFramesClusters.size() - 1] - mROFramesClusters[layerId][rofIdstart]);
236 }

238 template <int nLayers>
239 inline std::vector<unsigned int> TimeFrameGPU<nLayers>::getClusterSizes()
240 {
241  std::vector<unsigned int> sizes(mUnsortedClusters.size());
242  std::transform(mUnsortedClusters.begin(), mUnsortedClusters.end(), sizes.begin(),
243  [](const auto& v) { return static_cast<unsigned int>(v.size()); });
244  return sizes;
245 }
