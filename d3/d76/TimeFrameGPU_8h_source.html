299 template <int nLayers>
300 template <Task task>
301 size_t TimeFrameGPU<nLayers>::loadChunkData(const size_t chunk, const size_t offset, const size_t maxRofs) // offset: readout frame to start from, maxRofs: to manage boundaries
302 {
303  size_t nRof{0};
304 
305  mMemChunks[chunk].reset(task, mGpuStreams[chunk]); // Reset chunks memory
306  if constexpr ((bool)task) {
307  nRof = mMemChunks[chunk].loadDataOnDevice(offset, maxRofs, 3, mGpuStreams[chunk]);
308  } else {
309  nRof = mMemChunks[chunk].loadDataOnDevice(offset, maxRofs, nLayers, mGpuStreams[chunk]);
310  }
311  LOGP(debug, "In chunk {}: loaded {} readout frames starting from {}", chunk, nRof, offset);
312  return nRof;
313 }
