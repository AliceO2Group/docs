29void ClusterSharingMapSpec::run(ProcessingContext& pc)
30{
31 TStopwatch timer;
32 static int nHBPerTF = 0;
33 const auto tracksTPC = pc.inputs().get<gsl::span<o2::tpc::TrackTPC>>("trackTPC");
34 const auto tracksTPCClRefs = pc.inputs().get<gsl::span<o2::tpc::TPCClRefElem>>("trackTPCClRefs");
35 const auto& clustersTPC = getWorkflowTPCInput(pc);
36 if (pc.services().get<o2::framework::TimingInfo>().globalRunNumberChanged) { // new run is starting
37 auto grp = pc.inputs().get<o2::parameters::GRPECSObject*>("grpecs");
38 nHBPerTF = grp->getNHBFPerTF();
39 LOGP(info, "Will use {} HB per TF from GRPECS", nHBPerTF);
40 }
41
42 std::shared_ptr<o2::gpu::GPUParam> param = o2::gpu::GPUO2InterfaceUtils::getFullParamShared(0.f, nHBPerTF);
43 auto& bufVecSh = pc.outputs().make<std::vector<unsigned char>>(Output{o2::header::gDataOriginTPC, "CLSHAREDMAP", 0}, clustersTPC->clusterIndex.nClustersTotal);
44 size_t occupancyMapSizeBytes = o2::gpu::GPUO2InterfaceRefit::fillOccupancyMapGetSize(nHBPerTF, param.get());
45 auto& bufVecOcc = pc.outputs().make<std::vector<unsigned int>>(Output{o2::header::gDataOriginTPC, "TPCOCCUPANCYMAP", 0}, occupancyMapSizeBytes / sizeof(int));
46 o2::gpu::GPUO2InterfaceRefit::fillSharedClustersAndOccupancyMap(&clustersTPC->clusterIndex, tracksTPC, tracksTPCClRefs.data(), bufVecSh.data(), bufVecOcc.data(), nHBPerTF, param.get());
47
48 timer.Stop();
49 LOGF(info, "Timing for TPC clusters sharing map creation: Cpu: %.3e Real: %.3e s", timer.CpuTime(), timer.RealTime());
50}
