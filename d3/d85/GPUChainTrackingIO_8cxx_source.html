71void GPUChainTracking::DumpData(const char* filename)
72{
73 FILE* fp = fopen(filename, "w+b");
74 if (fp == nullptr) {
75 return;
76 }
77 fwrite(DUMP_HEADER, 1, DUMP_HEADER_SIZE, fp);
78 fwrite(&GPUReconstruction::geometryType, sizeof(GPUReconstruction::geometryType), 1, fp);
79 DumpData(fp, mIOPtrs.clusterData, mIOPtrs.nClusterData, InOutPointerType::CLUSTER_DATA);
80 DumpData(fp, mIOPtrs.rawClusters, mIOPtrs.nRawClusters, InOutPointerType::RAW_CLUSTERS);
81#ifdef GPUCA_HAVE_O2HEADERS
82 if (mIOPtrs.clustersNative) {
83 if (DumpData(fp, &mIOPtrs.clustersNative->clustersLinear, &mIOPtrs.clustersNative->nClustersTotal, InOutPointerType::CLUSTERS_NATIVE)) {
84 fwrite(&mIOPtrs.clustersNative->nClusters[0][0], sizeof(mIOPtrs.clustersNative->nClusters[0][0]), NSLICES * GPUCA_ROW_COUNT, fp);
85 if (mIOPtrs.clustersNative->clustersMCTruth) {
86 const auto& buffer = mIOPtrs.clustersNative->clustersMCTruth->getBuffer();
87 std::pair<const char*, size_t> tmp = {buffer.data(), buffer.size()};
88 DumpData(fp, &tmp.first, &tmp.second, InOutPointerType::CLUSTER_NATIVE_MC);
89 }
90 }
91 }
92 if (mIOPtrs.tpcPackedDigits) {
93 if (DumpData(fp, mIOPtrs.tpcPackedDigits->tpcDigits, mIOPtrs.tpcPackedDigits->nTPCDigits, InOutPointerType::TPC_DIGIT) && mIOPtrs.tpcPackedDigits->tpcDigitsMC) {
94 const char* ptrs[NSLICES];
95 size_t sizes[NSLICES];
96 for (uint32_t i = 0; i < NSLICES; i++) {
97 if (mIOPtrs.tpcPackedDigits->tpcDigitsMC->v[i]) {
98 const auto& buffer = mIOPtrs.tpcPackedDigits->tpcDigitsMC->v[i]->getBuffer();
99 ptrs[i] = buffer.data();
100 sizes[i] = buffer.size();
101 } else {
102 ptrs[i] = nullptr;
103 sizes[i] = 0;
104 }
105 }
106 DumpData(fp, ptrs, sizes, InOutPointerType::TPC_DIGIT_MC);
107 }
108 }
109 if (mIOPtrs.tpcZS) {
110 size_t total = 0;
111 for (int32_t i = 0; i < NSLICES; i++) {
112 for (uint32_t j = 0; j < GPUTrackingInOutZS::NENDPOINTS; j++) {
113 for (uint32_t k = 0; k < mIOPtrs.tpcZS->slice[i].count[j]; k++) {
114 total += mIOPtrs.tpcZS->slice[i].nZSPtr[j][k];
115 }
116 }
117 }
118 std::vector<std::array<char, TPCZSHDR::TPC_ZS_PAGE_SIZE>> pages(total);
119 char* ptr = pages[0].data();
120 GPUTrackingInOutZS::GPUTrackingInOutZSCounts counts;
121 total = 0;
122 for (int32_t i = 0; i < NSLICES; i++) {
123 for (uint32_t j = 0; j < GPUTrackingInOutZS::NENDPOINTS; j++) {
124 for (uint32_t k = 0; k < mIOPtrs.tpcZS->slice[i].count[j]; k++) {
125 memcpy(&ptr[total * TPCZSHDR::TPC_ZS_PAGE_SIZE], mIOPtrs.tpcZS->slice[i].zsPtr[j][k], mIOPtrs.tpcZS->slice[i].nZSPtr[j][k] * TPCZSHDR::TPC_ZS_PAGE_SIZE);
126 counts.count[i][j] += mIOPtrs.tpcZS->slice[i].nZSPtr[j][k];
127 total += mIOPtrs.tpcZS->slice[i].nZSPtr[j][k];
128 }
129 }
130 }
131 total *= TPCZSHDR::TPC_ZS_PAGE_SIZE;
132 if (DumpData(fp, &ptr, &total, InOutPointerType::TPC_ZS)) {
133 fwrite(&counts, sizeof(counts), 1, fp);
134 }
135 }
136 if (mIOPtrs.tpcCompressedClusters) {
137 if (mIOPtrs.tpcCompressedClusters->ptrForward) {
138 throw std::runtime_error("Cannot dump non-flat compressed clusters");
139 }
140 char* ptr = (char*)mIOPtrs.tpcCompressedClusters;
141 size_t size = mIOPtrs.tpcCompressedClusters->totalDataSize;
142 DumpData(fp, &ptr, &size, InOutPointerType::TPC_COMPRESSED_CL);
143 }
144 if (mIOPtrs.settingsTF) {
145 uint32_t n = 1;
146 DumpData(fp, &mIOPtrs.settingsTF, &n, InOutPointerType::TF_SETTINGS);
147 }
148#endif
149 DumpData(fp, mIOPtrs.sliceTracks, mIOPtrs.nSliceTracks, InOutPointerType::SLICE_OUT_TRACK);
150 DumpData(fp, mIOPtrs.sliceClusters, mIOPtrs.nSliceClusters, InOutPointerType::SLICE_OUT_CLUSTER);
151 DumpData(fp, &mIOPtrs.mcLabelsTPC, &mIOPtrs.nMCLabelsTPC, InOutPointerType::MC_LABEL_TPC);
152 DumpData(fp, &mIOPtrs.mcInfosTPC, &mIOPtrs.nMCInfosTPC, InOutPointerType::MC_INFO_TPC);
153 DumpData(fp, &mIOPtrs.mcInfosTPCCol, &mIOPtrs.nMCInfosTPCCol, InOutPointerType::MC_INFO_TPC);
154 DumpData(fp, &mIOPtrs.mergedTracks, &mIOPtrs.nMergedTracks, InOutPointerType::MERGED_TRACK);
155 DumpData(fp, &mIOPtrs.mergedTrackHits, &mIOPtrs.nMergedTrackHits, InOutPointerType::MERGED_TRACK_HIT);
156 DumpData(fp, &mIOPtrs.trdTracks, &mIOPtrs.nTRDTracks, InOutPointerType::TRD_TRACK);
157 DumpData(fp, &mIOPtrs.trdTracklets, &mIOPtrs.nTRDTracklets, InOutPointerType::TRD_TRACKLET);
158 if (mIOPtrs.trdSpacePoints) {
159 DumpData(fp, &mIOPtrs.trdSpacePoints, &mIOPtrs.nTRDTracklets, InOutPointerType::TRD_SPACEPOINT);
160 }
161 DumpData(fp, &mIOPtrs.trdTriggerTimes, &mIOPtrs.nTRDTriggerRecords, InOutPointerType::TRD_TRIGGERRECORDS);
162 DumpData(fp, &mIOPtrs.trdTrackletIdxFirst, &mIOPtrs.nTRDTriggerRecords, InOutPointerType::TRD_TRIGGERRECORDS);
163 DumpData(fp, &mIOPtrs.trdTrigRecMask, &mIOPtrs.nTRDTriggerRecords, InOutPointerType::TRD_TRIGGERRECORDS);
164 fclose(fp);
165}

167int32_t GPUChainTracking::ReadData(const char* filename)
168{
169 ClearIOPointers();
170 FILE* fp = fopen(filename, "rb");
171 if (fp == nullptr) {
172 return (1);
173 }
174
175 char buf[DUMP_HEADER_SIZE + 1] = "";
176 size_t r = fread(buf, 1, DUMP_HEADER_SIZE, fp);
177 if (strncmp(DUMP_HEADER, buf, DUMP_HEADER_SIZE)) {
178 GPUError("Invalid file header");
179 fclose(fp);
180 return -1;
181 }
182 GeometryType geo;
183 r = fread(&geo, sizeof(geo), 1, fp);
184 if (geo != GPUReconstruction::geometryType) {
185 GPUError("File has invalid geometry (%s v.s. %s)", GPUReconstruction::GEOMETRY_TYPE_NAMES[(int32_t)geo], GPUReconstruction::GEOMETRY_TYPE_NAMES[(int32_t)GPUReconstruction::geometryType]);
186 fclose(fp);
187 return 1;
188 }
189 GPUTPCClusterData* ptrClusterData[NSLICES];
190 ReadData(fp, mIOPtrs.clusterData, mIOPtrs.nClusterData, mIOMem.clusterData, InOutPointerType::CLUSTER_DATA, ptrClusterData);
191 AliHLTTPCRawCluster* ptrRawClusters[NSLICES];
192 ReadData(fp, mIOPtrs.rawClusters, mIOPtrs.nRawClusters, mIOMem.rawClusters, InOutPointerType::RAW_CLUSTERS, ptrRawClusters);
193 int32_t nClustersTotal = 0;
194#ifdef GPUCA_HAVE_O2HEADERS
195 mIOMem.clusterNativeAccess.reset(new ClusterNativeAccess);
196 if (ReadData<ClusterNative>(fp, &mIOMem.clusterNativeAccess->clustersLinear, &mIOMem.clusterNativeAccess->nClustersTotal, &mIOMem.clustersNative, InOutPointerType::CLUSTERS_NATIVE)) {
197 r = fread(&mIOMem.clusterNativeAccess->nClusters[0][0], sizeof(mIOMem.clusterNativeAccess->nClusters[0][0]), NSLICES * GPUCA_ROW_COUNT, fp);
198 mIOMem.clusterNativeAccess->setOffsetPtrs();
199 mIOPtrs.clustersNative = mIOMem.clusterNativeAccess.get();
200 std::pair<const char*, size_t> tmp = {nullptr, 0};
201 if (ReadData(fp, &tmp.first, &tmp.second, &mIOMem.clusterNativeMC, InOutPointerType::CLUSTER_NATIVE_MC)) {
202 mIOMem.clusterNativeMCView = std::make_unique<ConstMCLabelContainerView>(gsl::span<const char>(tmp.first, tmp.first + tmp.second));
203 mIOMem.clusterNativeAccess->clustersMCTruth = mIOMem.clusterNativeMCView.get();
204 }
205 }
206 mIOMem.digitMap.reset(new GPUTrackingInOutDigits);
207 if (ReadData(fp, mIOMem.digitMap->tpcDigits, mIOMem.digitMap->nTPCDigits, mIOMem.tpcDigits, InOutPointerType::TPC_DIGIT)) {
208 mIOPtrs.tpcPackedDigits = mIOMem.digitMap.get();
209 const char* ptrs[NSLICES];
210 size_t sizes[NSLICES];
211 if (ReadData(fp, ptrs, sizes, mIOMem.tpcDigitsMC, InOutPointerType::TPC_DIGIT_MC)) {
212 mIOMem.tpcDigitMCMap = std::make_unique<GPUTPCDigitsMCInput>();
213 mIOMem.tpcDigitMCView.reset(new ConstMCLabelContainerView[NSLICES]);
214 for (uint32_t i = 0; i < NSLICES; i++) {
215 if (sizes[i]) {
216 mIOMem.tpcDigitMCView.get()[i] = gsl::span<const char>(ptrs[i], ptrs[i] + sizes[i]);
217 mIOMem.tpcDigitMCMap->v[i] = mIOMem.tpcDigitMCView.get() + i;
218 } else {
219 mIOMem.tpcDigitMCMap->v[i] = nullptr;
220 }
221 }
222 mIOMem.digitMap->tpcDigitsMC = mIOMem.tpcDigitMCMap.get();
223 }
224 }
225 const char* ptr;
226 size_t total;
227 char* ptrZSPages;
228 if (ReadData(fp, &ptr, &total, &mIOMem.tpcZSpagesChar, InOutPointerType::TPC_ZS, &ptrZSPages)) {
229 GPUTrackingInOutZS::GPUTrackingInOutZSCounts counts;
230 r = fread(&counts, sizeof(counts), 1, fp);
231 mIOMem.tpcZSmeta.reset(new GPUTrackingInOutZS);
232 mIOMem.tpcZSmeta2.reset(new GPUTrackingInOutZS::GPUTrackingInOutZSMeta);
233 total = 0;
234 for (int32_t i = 0; i < NSLICES; i++) {
235 for (uint32_t j = 0; j < GPUTrackingInOutZS::NENDPOINTS; j++) {
236 mIOMem.tpcZSmeta2->ptr[i][j] = &ptrZSPages[total * TPCZSHDR::TPC_ZS_PAGE_SIZE];
237 mIOMem.tpcZSmeta->slice[i].zsPtr[j] = &mIOMem.tpcZSmeta2->ptr[i][j];
238 mIOMem.tpcZSmeta2->n[i][j] = counts.count[i][j];
239 mIOMem.tpcZSmeta->slice[i].nZSPtr[j] = &mIOMem.tpcZSmeta2->n[i][j];
240 mIOMem.tpcZSmeta->slice[i].count[j] = 1;
241 total += counts.count[i][j];
242 }
243 }
244 mIOPtrs.tpcZS = mIOMem.tpcZSmeta.get();
245 }
246 if (ReadData(fp, &ptr, &total, &mIOMem.tpcCompressedClusters, InOutPointerType::TPC_COMPRESSED_CL)) {
247 mIOPtrs.tpcCompressedClusters = (const o2::tpc::CompressedClustersFlat*)ptr;
248 }
249 uint32_t n;
250 ReadData(fp, &mIOPtrs.settingsTF, &n, &mIOMem.settingsTF, InOutPointerType::TF_SETTINGS);
251#endif
252 ReadData(fp, mIOPtrs.sliceTracks, mIOPtrs.nSliceTracks, mIOMem.sliceTracks, InOutPointerType::SLICE_OUT_TRACK);
253 ReadData(fp, mIOPtrs.sliceClusters, mIOPtrs.nSliceClusters, mIOMem.sliceClusters, InOutPointerType::SLICE_OUT_CLUSTER);
254 ReadData(fp, &mIOPtrs.mcLabelsTPC, &mIOPtrs.nMCLabelsTPC, &mIOMem.mcLabelsTPC, InOutPointerType::MC_LABEL_TPC);
255 ReadData(fp, &mIOPtrs.mcInfosTPC, &mIOPtrs.nMCInfosTPC, &mIOMem.mcInfosTPC, InOutPointerType::MC_INFO_TPC);
256 ReadData(fp, &mIOPtrs.mcInfosTPCCol, &mIOPtrs.nMCInfosTPCCol, &mIOMem.mcInfosTPCCol, InOutPointerType::MC_INFO_TPC);
257 ReadData(fp, &mIOPtrs.mergedTracks, &mIOPtrs.nMergedTracks, &mIOMem.mergedTracks, InOutPointerType::MERGED_TRACK);
258 ReadData(fp, &mIOPtrs.mergedTrackHits, &mIOPtrs.nMergedTrackHits, &mIOMem.mergedTrackHits, InOutPointerType::MERGED_TRACK_HIT);
259 ReadData(fp, &mIOPtrs.trdTracks, &mIOPtrs.nTRDTracks, &mIOMem.trdTracks, InOutPointerType::TRD_TRACK);
260 ReadData(fp, &mIOPtrs.trdTracklets, &mIOPtrs.nTRDTracklets, &mIOMem.trdTracklets, InOutPointerType::TRD_TRACKLET);
261 uint32_t dummy = 0;
262 ReadData(fp, &mIOPtrs.trdSpacePoints, &dummy, &mIOMem.trdSpacePoints, InOutPointerType::TRD_SPACEPOINT);
263 ReadData(fp, &mIOPtrs.trdTriggerTimes, &mIOPtrs.nTRDTriggerRecords, &mIOMem.trdTriggerTimes, InOutPointerType::TRD_TRIGGERRECORDS);
264 ReadData(fp, &mIOPtrs.trdTrackletIdxFirst, &mIOPtrs.nTRDTriggerRecords, &mIOMem.trdTrackletIdxFirst, InOutPointerType::TRD_TRIGGERRECORDS);
265 ReadData(fp, &mIOPtrs.trdTrigRecMask, &mIOPtrs.nTRDTriggerRecords, &mIOMem.trdTrigRecMask, InOutPointerType::TRD_TRIGGERRECORDS);
266
267 size_t fptr = ftell(fp);
268 fseek(fp, 0, SEEK_END);
269 size_t fend = ftell(fp);
270 fclose(fp);
271 if (fptr != fend) {
272 GPUError("Error reading data file, reading incomplete");
273 return 1;
274 }
275 (void)r;
276 for (uint32_t i = 0; i < NSLICES; i++) {
277 for (uint32_t j = 0; j < mIOPtrs.nClusterData[i]; j++) {
278 ptrClusterData[i][j].id = nClustersTotal++;
279 if ((uint32_t)ptrClusterData[i][j].amp >= 25 * 1024) {
280 GPUError("Invalid cluster charge, truncating (%d >= %d)", (int32_t)ptrClusterData[i][j].amp, 25 * 1024);
281 ptrClusterData[i][j].amp = 25 * 1024 - 1;
282 }
283 }
284 for (uint32_t j = 0; j < mIOPtrs.nRawClusters[i]; j++) {
285 if ((uint32_t)mIOMem.rawClusters[i][j].GetCharge() >= 25 * 1024) {
286 GPUError("Invalid raw cluster charge, truncating (%d >= %d)", (int32_t)ptrRawClusters[i][j].GetCharge(), 25 * 1024);
287 ptrRawClusters[i][j].SetCharge(25 * 1024 - 1);
288 }
289 if ((uint32_t)mIOPtrs.rawClusters[i][j].GetQMax() >= 1024) {
290 GPUError("Invalid raw cluster charge max, truncating (%d >= %d)", (int32_t)ptrRawClusters[i][j].GetQMax(), 1024);
291 ptrRawClusters[i][j].SetQMax(1024 - 1);
292 }
293 }
294 }
295
296 return (0);
297}
