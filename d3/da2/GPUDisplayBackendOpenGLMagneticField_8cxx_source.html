63int32_t GPUDisplayBackendOpenGL::InitMagFieldVisualization()
64{
65#ifndef GPUCA_NO_FMT
66 mMagneticFieldVisualization = std::make_unique<GPUDisplayMagneticField>();
67 mMagneticFieldVisualization->generateSeedPoints(mDisplay->cfgL().bFieldLinesCount);
68
69 CHKERR(mVertexShaderPassthrough = glCreateShader(GL_VERTEX_SHADER));
70 CHKERR(mGeometryShader = glCreateShader(GL_GEOMETRY_SHADER));
71
72 CHKERR(glShaderSource(mVertexShaderPassthrough, 1, &GPUDisplayShaders::vertexShaderPassthrough, nullptr));
73 CHKERR(glCompileShader(mVertexShaderPassthrough));
74
75 const auto constantsExpanded = fmt::format(GPUDisplayShaders::fieldModelShaderConstants,
76 fmt::arg("dimensions", GPUDisplayMagneticField::DIMENSIONS),
77 fmt::arg("solZSegs", GPUDisplayMagneticField::MAX_SOLENOID_Z_SEGMENTS),
78 fmt::arg("solPSegs", GPUDisplayMagneticField::MAX_SOLENOID_P_SEGMENTS),
79 fmt::arg("solRSegs", GPUDisplayMagneticField::MAX_SOLENOID_R_SEGMENTS),
80 fmt::arg("solParams", GPUDisplayMagneticField::MAX_SOLENOID_PARAMETERIZATIONS),
81 fmt::arg("solRows", GPUDisplayMagneticField::MAX_SOLENOID_ROWS),
82 fmt::arg("solColumns", GPUDisplayMagneticField::MAX_SOLENOID_COLUMNS),
83 fmt::arg("solCoeffs", GPUDisplayMagneticField::MAX_SOLENOID_COEFFICIENTS),
84 fmt::arg("dipZSegs", GPUDisplayMagneticField::MAX_DIPOLE_Z_SEGMENTS),
85 fmt::arg("dipYSegs", GPUDisplayMagneticField::MAX_DIPOLE_Y_SEGMENTS),
86 fmt::arg("dipXSegs", GPUDisplayMagneticField::MAX_DIPOLE_X_SEGMENTS),
87 fmt::arg("dipParams", GPUDisplayMagneticField::MAX_DIPOLE_PARAMETERIZATIONS),
88 fmt::arg("dipRows", GPUDisplayMagneticField::MAX_DIPOLE_ROWS),
89 fmt::arg("dipColumns", GPUDisplayMagneticField::MAX_DIPOLE_COLUMNS),
90 fmt::arg("dipCoeffs", GPUDisplayMagneticField::MAX_DIPOLE_COEFFICIENTS),
91 fmt::arg("maxChebOrder", GPUDisplayMagneticField::MAX_CHEBYSHEV_ORDER));
92
93 std::array geomShaderSource = {GPUDisplayShaders::geometryShaderP1, constantsExpanded.c_str(), GPUDisplayShaders::fieldModelShaderCode, GPUDisplayShaders::geometryShaderP2};
94
95 CHKERR(glShaderSource(mGeometryShader, geomShaderSource.size(), geomShaderSource.data(), nullptr));
96 CHKERR(glCompileShader(mGeometryShader));
97
98 if (checkShaderStatus(mVertexShaderPassthrough) || checkShaderStatus(mGeometryShader)) {
99 return 1;
100 }
101
102 CHKERR(mShaderProgramField = glCreateProgram());
103 CHKERR(glAttachShader(mShaderProgramField, mVertexShaderPassthrough));
104 CHKERR(glAttachShader(mShaderProgramField, mGeometryShader));
105 CHKERR(glAttachShader(mShaderProgramField, mFragmentShader));
106 CHKERR(glLinkProgram(mShaderProgramField));
107
108 if (checkProgramStatus(mShaderProgramField)) {
109 return 1;
110 }
111
112 const auto ATTRIB_ZERO = 0;
113 const auto BUFFER_IDX = 0;
114
115 CHKERR(glCreateVertexArrays(1, &VAO_field));
116 CHKERR(glEnableVertexArrayAttrib(VAO_field, ATTRIB_ZERO));
117 CHKERR(glVertexArrayAttribFormat(VAO_field, ATTRIB_ZERO, 3, GL_FLOAT, GL_FALSE, 0));
118
119 CHKERR(glCreateBuffers(1, &VBO_field));
120 CHKERR(glNamedBufferData(VBO_field, mMagneticFieldVisualization->mFieldLineSeedPoints.size() * sizeof(GPUDisplayMagneticField::vtx), mMagneticFieldVisualization->mFieldLineSeedPoints.data(), GL_STATIC_DRAW));
121
122 CHKERR(glVertexArrayVertexBuffer(VAO_field, BUFFER_IDX, VBO_field, 0, sizeof(GPUDisplayMagneticField::vtx)));
123
124 CHKERR(glVertexArrayAttribBinding(VAO_field, ATTRIB_ZERO, BUFFER_IDX));
125
126 CHKERR(glCreateBuffers(1, &mFieldModelViewBuffer));
127 CHKERR(glNamedBufferData(mFieldModelViewBuffer, sizeof(hmm_mat4), nullptr, GL_STREAM_DRAW));
128
129 CHKERR(glCreateBuffers(1, &mFieldModelConstantsBuffer));
130 CHKERR(glNamedBufferData(mFieldModelConstantsBuffer, sizeof(GPUDisplayMagneticField::RenderConstantsUniform), mMagneticFieldVisualization->mRenderConstantsUniform.get(), GL_STREAM_DRAW));
131
132 CHKERR(glCreateBuffers(1, &mSolenoidSegmentsBuffer));
133 CHKERR(glNamedBufferData(mSolenoidSegmentsBuffer, sizeof(GPUDisplayMagneticField::SolenoidSegmentsUniform), mMagneticFieldVisualization->mSolenoidSegments.get(), GL_STREAM_DRAW));
134 CHKERR(glCreateBuffers(1, &mSolenoidParameterizationBuffer));
135 CHKERR(glNamedBufferData(mSolenoidParameterizationBuffer, sizeof(GPUDisplayMagneticField::SolenoidParameterizationUniform), mMagneticFieldVisualization->mSolenoidParameterization.get(), GL_STREAM_DRAW));
136
137 CHKERR(glCreateBuffers(1, &mDipoleSegmentsBuffer));
138 CHKERR(glNamedBufferData(mDipoleSegmentsBuffer, sizeof(GPUDisplayMagneticField::DipoleSegmentsUniform), mMagneticFieldVisualization->mDipoleSegments.get(), GL_STREAM_DRAW));
139 CHKERR(glCreateBuffers(1, &mDipoleParameterizationBuffer));
140 CHKERR(glNamedBufferData(mDipoleParameterizationBuffer, sizeof(GPUDisplayMagneticField::DipoleParameterizationUniform), mMagneticFieldVisualization->mDipoleParameterization.get(), GL_STREAM_DRAW));
141#else
142 throw std::runtime_error("Magnetic field needs fmt");
143#endif
144
145 return 0;
146}

148uint32_t GPUDisplayBackendOpenGL::drawField()
149{
150 if (!mMagneticFieldVisualization) {
151 return InitMagFieldVisualization(); // next frame will fill MVP matrix
152 }
153
154 if (mMagneticFieldVisualization->mFieldLineSeedPoints.size() != (uint32_t)mDisplay->cfgL().bFieldLinesCount) {
155 mMagneticFieldVisualization->generateSeedPoints(mDisplay->cfgL().bFieldLinesCount);
156 CHKERR(glNamedBufferData(VBO_field, mMagneticFieldVisualization->mFieldLineSeedPoints.size() * sizeof(GPUDisplayMagneticField::vtx), mMagneticFieldVisualization->mFieldLineSeedPoints.data(), GL_STATIC_DRAW));
157 }
158
159 mMagneticFieldVisualization->mRenderConstantsUniform->StepSize = mDisplay->cfgL().bFieldStepSize;
160 mMagneticFieldVisualization->mRenderConstantsUniform->StepCount = mDisplay->cfgL().bFieldStepCount;
161 CHKERR(glNamedBufferSubData(mFieldModelConstantsBuffer, 0, sizeof(GPUDisplayMagneticField::RenderConstantsUniform), mMagneticFieldVisualization->mRenderConstantsUniform.get()));
162
163 CHKERR(glBindVertexArray(VAO_field));
164 CHKERR(glUseProgram(mShaderProgramField));
165 CHKERR(glBindBufferBase(GL_UNIFORM_BUFFER, 0, mFieldModelViewBuffer));
166 const std::array<float, 4> drawColor = {1.f, 0.f, 0.f, 1.f};
167 const auto color = glGetUniformLocation(mShaderProgramField, "color");
168 CHKERR(glUniform4fv(color, 1, drawColor.data()));
169
170 CHKERR(glBindBufferBase(GL_SHADER_STORAGE_BUFFER, 0, mFieldModelConstantsBuffer));
171 CHKERR(glBindBufferBase(GL_SHADER_STORAGE_BUFFER, 1, mSolenoidSegmentsBuffer));
172 CHKERR(glBindBufferBase(GL_SHADER_STORAGE_BUFFER, 2, mDipoleSegmentsBuffer));
173 CHKERR(glBindBufferBase(GL_SHADER_STORAGE_BUFFER, 3, mSolenoidParameterizationBuffer));
174 CHKERR(glBindBufferBase(GL_SHADER_STORAGE_BUFFER, 4, mDipoleParameterizationBuffer));
175
176 CHKERR(glDrawArrays(GL_POINTS, 0, mMagneticFieldVisualization->mFieldLineSeedPoints.size()));
177
178 CHKERR(glUseProgram(0));
179 CHKERR(glBindVertexArray(0));
180
181 return 0;
182}

185{
186 CHKERR(glDeleteBuffers(1, &mFieldModelViewBuffer));
187 CHKERR(glDeleteBuffers(1, &mFieldModelConstantsBuffer));

190 CHKERR(glDeleteBuffers(1, &mDipoleSegmentsBuffer));
191 CHKERR(glDeleteBuffers(1, &mDipoleParameterizationBuffer));
192 CHKERR(glDeleteProgram(mShaderProgramField));
193 CHKERR(glDeleteShader(mGeometryShader));
194 CHKERR(glDeleteShader(mVertexShaderPassthrough));
195 CHKERR(glDeleteBuffers(1, &VBO_field));
196 CHKERR(glDeleteVertexArrays(1, &VAO_field));
197}
