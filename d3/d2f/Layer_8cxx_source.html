30Layer::Layer(std::string layerName, float rInn, float rOut, float zLength, float zOffset, float layerX2X0, int layout, int nSegments, float segmentSize, int nSensorsPerSegment, double tiltAngle)
31 : mLayerName(layerName),
32 mInnerRadius(rInn),
33 mOuterRadius(rOut),
34 mZLength(zLength),
35 mZOffset(zOffset),
36 mX2X0(layerX2X0),
37 mLayout(layout),
38 mSegments(nSegments, segmentSize),
39 mSensorsPerSegment(nSensorsPerSegment),
40 mTiltAngle(tiltAngle)
41{
42 float Si_X0 = 9.5f;
43 mChipThickness = mX2X0 * Si_X0;
44 std::string name = "";
45 switch (layout) {
46 case kBarrel:
47 case kBarrelSegmented:
48 name = "barrel";
49 mOuterRadius = mInnerRadius + mChipThickness;
50 break;
51 case kDisk:
52 case kDiskSegmented:
53 name = "forward";
54 mZLength = mChipThickness;
55 break;
56 default:
57 LOG(fatal) << "Invalid layout " << layout;
58 }
59 if (1) { // Sanity checks
60 if (mInnerRadius > mOuterRadius) {
61 LOG(fatal) << "Invalid layer dimensions: rInner " << mInnerRadius << " cm is larger than rOuter " << mOuterRadius << " cm";
62 }
63 if ((mSegments.first != 0 || mSegments.second != 0.0f) && (layout != kBarrelSegmented && layout != kDiskSegmented)) {
64 LOG(fatal) << "Invalid configuration: number of segments " << mSegments.first << " is set for non-segmented layout " << layout;
65 }
66 if ((mSegments.first <= 1 || mSegments.second <= 0.0f) && (layout == kBarrelSegmented || layout == kDiskSegmented)) {
67 LOG(fatal) << "Invalid configuration: number of segments " << mSegments.first << " must be positive for segmented layout " << layout;
68 }
69 if (mSensorsPerSegment <= 0 && (layout == kBarrelSegmented || layout == kDiskSegmented)) {
70 LOG(fatal) << "Invalid configuration: number of sensors per segment " << mSensorsPerSegment << " must be positive for segmented layout " << layout;
71 }
72 if (std::abs(mTiltAngle) > 0.1 && (layout != kBarrelSegmented && layout != kDiskSegmented)) {
73 LOG(fatal) << "Invalid configuration: tilt angle " << mTiltAngle << " is set for non-segmented layout " << layout;
74 }
75 }
76
77 LOGP(info, "TOF: Creating {} layer: rInner: {} (cm) rOuter: {} (cm) zLength: {} (cm) zOffset: {} x2X0: {}", name.c_str(), mInnerRadius, mOuterRadius, mZLength, mZOffset, mX2X0);
78}

81void ITOFLayer::createLayer(TGeoVolume* motherVolume)
82{
83 const std::string chipName = o2::iotof::GeometryTGeo::getITOFChipPattern();
84 const std::string sensName = o2::iotof::GeometryTGeo::getITOFSensorPattern();
85
86 TGeoMedium* medSi = gGeoManager->GetMedium("TF3_SILICON$");
87 TGeoMedium* medAir = gGeoManager->GetMedium("TF3_AIR$");
88 LOGP(info, "Media: {} {}", (void*)medSi, (void*)medAir);
89
90 switch (mLayout) {
91 case kBarrel: {
92 TGeoTube* sensor = new TGeoTube(mInnerRadius, mOuterRadius, mZLength / 2);
93 TGeoTube* chip = new TGeoTube(mInnerRadius, mOuterRadius, mZLength / 2);
94 TGeoTube* layer = new TGeoTube(mInnerRadius, mOuterRadius, mZLength / 2);
95
96 TGeoVolume* sensVol = new TGeoVolume(sensName.c_str(), sensor, medSi);
97 TGeoVolume* chipVol = new TGeoVolume(chipName.c_str(), chip, medSi);
98 TGeoVolume* layerVol = new TGeoVolume(mLayerName.c_str(), layer, medAir);
99 sensVol->SetLineColor(kRed + 3);
100 chipVol->SetLineColor(kRed + 3);
101 layerVol->SetLineColor(kRed + 3);
102
103 LOGP(info, "Inserting Barrel {} in {} ", sensVol->GetName(), chipVol->GetName());
104 ITOFLayer::mRegister.push_back(sensVol->GetName());
105 chipVol->AddNode(sensVol, 1, nullptr);
106
107 LOGP(info, "Inserting Barrel {} in {} ", chipVol->GetName(), layerVol->GetName());
108 layerVol->AddNode(chipVol, 1, nullptr);
109
110 LOGP(info, "Inserting Barrel {} in {} ", layerVol->GetName(), motherVolume->GetName());
111 motherVolume->AddNode(layerVol, 1, nullptr);
112 return;
113 }
114 case kBarrelSegmented: {
115 const double circumference = TMath::TwoPi() * 0.5 * (mInnerRadius + mOuterRadius);
116 const double segmentSize = mSegments.second; // cm circumference / mSegments;
117 const double avgRadius = 0.5 * (mInnerRadius + mOuterRadius);
118 TGeoTube* layer = new TGeoTube(mInnerRadius, mOuterRadius, mZLength / 2);
119 TGeoVolume* layerVol = new TGeoVolume(mLayerName.c_str(), layer, medAir);
120 layerVol->SetLineColor(kRed + 3);
121
122 for (int i = 0; i < mSegments.first; ++i) {
123 LOGP(info, "iTOF: Creating segment {}/{} with size {} and thickness {}cm", i + 1, mSegments.first, segmentSize, (mOuterRadius - mInnerRadius));
124 const double hx = 0.5 * segmentSize;
125 const double hy = 0.5 * (mOuterRadius - mInnerRadius);
126 const double hz = 0.5 * mZLength;
127 TGeoBBox* sensor = new TGeoBBox(hy, hx, hz);
128 TGeoBBox* chip = new TGeoBBox(hy, hx, hz);
129 const std::string segmentTag = Form("segment%d", i + 1);
130 TGeoVolume* sensVol = new TGeoVolume(Form("%s_%s", sensName.c_str(), segmentTag.c_str()), sensor, medSi);
131 TGeoVolume* chipVol = new TGeoVolume(Form("%s_%s", chipName.c_str(), segmentTag.c_str()), chip, medSi);
132 sensVol->SetLineColor(kRed + 3);
133 chipVol->SetLineColor(kRed + 3);
134
135 LOGP(info, " Inserting Barrel {} in {} ", sensVol->GetName(), chipVol->GetName());
136 ITOFLayer::mRegister.push_back(sensVol->GetName());
137 chipVol->AddNode(sensVol, 1, nullptr);
138
139 const double phi = TMath::TwoPi() * i / mSegments.first;
140
141 LOG(info) << " Tilting angle for segment " << i + 1 << ": " << phi * TMath::RadToDeg() << " degrees";
142 const double x = avgRadius * TMath::Cos(phi);
143 const double y = avgRadius * TMath::Sin(phi);
144 auto* rotation = new TGeoRotation(Form("segmentRot%d", i + 1), phi * TMath::RadToDeg() + mTiltAngle, 0, 0);
145 auto* transformation = new TGeoCombiTrans(x, y, 0, rotation);
146
147 LOGP(info, "Inserting Barrel {} in {} ", chipVol->GetName(), layerVol->GetName());
148 layerVol->AddNode(chipVol, 1 + i, transformation);
149 }
150 LOGP(info, "Inserting Barrel {} in {} at r={} cm", layerVol->GetName(), motherVolume->GetName(), avgRadius);
151 motherVolume->AddNode(layerVol, 1, nullptr);
152 return;
153 }
154 default:
155 LOG(fatal) << "Invalid layout " << mLayout;
156 }
