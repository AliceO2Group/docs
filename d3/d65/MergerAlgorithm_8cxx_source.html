36 size_t estimateTreeSize(TTree* tree)
37 {
38  size_t totalSize = 0;
39  auto branchList = tree->GetListOfBranches();
40  for (const auto* branch : *branchList) {
41  totalSize += dynamic_cast<const TBranch*>(branch)->GetTotalSize();
42  }
43  return totalSize;

46 void merge(TObject* const target, TObject* const other)
47 {
48  if (target == nullptr) {
49  throw std::runtime_error("Merging target is nullptr");
50  }
51  if (other == nullptr) {
52  throw std::runtime_error("Object to be merged in is nullptr");
53  }
54  if (other == target) {
55  throw std::runtime_error("Merging target and the other object point to the same address");
56  }
57  // fixme: should we check if names match?
58 
59  // We expect that both objects follow the same structure, but we allow to add missing objects to TCollections.
60  // First we check if an object contains a MergeInterface, as it should overlap default Merge() methods of TObject.
61  if (auto custom = dynamic_cast<MergeInterface*>(target)) {
62 
63  custom->merge(dynamic_cast<MergeInterface* const>(other));
64 
65  } else if (auto targetCollection = dynamic_cast<TCollection*>(target)) {
66 
67  auto otherCollection = dynamic_cast<TCollection*>(other);
68  if (otherCollection == nullptr) {
69  throw std::runtime_error(std::string("The target object '") + target->GetName() +
70  "' is a TCollection, while the other object '" + other->GetName() + "' is not.");
71  }
72 
73  auto otherIterator = otherCollection->MakeIterator();
74  while (auto otherObject = otherIterator->Next()) {
75  TObject* targetObject = targetCollection->FindObject(otherObject->GetName());
76  if (targetObject) {
77  // That might be another collection or a concrete object to be merged, we walk on the collection recursively.
78  merge(targetObject, otherObject);
79  } else {
80  // We prefer to clone instead of passing the pointer in order to simplify deleting the `other`.
81  targetCollection->Add(otherObject->Clone());
82  }
83  }
84  delete otherIterator;
85  } else {
86  Long64_t errorCode = 0;
87  TObjArray otherCollection;
88  otherCollection.SetOwner(false);
89  otherCollection.Add(other);
90 
91  if (target->InheritsFrom(TH1::Class())) {
92  // this includes TH1, TH2, TH3
93  auto targetTH1 = reinterpret_cast<TH1*>(target);
94  if (targetTH1->TestBit(TH1::kIsAverage)) {
95  // Merge() does not support averages, we have to use Add()
96  // this will break if collection.size != 1
97  if (auto otherTH1 = dynamic_cast<TH1*>(otherCollection.First())) {
98  errorCode = targetTH1->Add(otherTH1);
99  }
100  } else {
101  // Add() does not support histograms with labels, thus we resort to Merge() by default
102  errorCode = targetTH1->Merge(&otherCollection);
103  }
104  } else if (target->InheritsFrom(THnBase::Class())) {
105  // this includes THn and THnSparse
106  errorCode = reinterpret_cast<THnBase*>(target)->Merge(&otherCollection);
107  } else if (target->InheritsFrom(TTree::Class())) {
108  auto targetTree = reinterpret_cast<TTree*>(target);
109  auto otherTree = reinterpret_cast<TTree*>(other);
110  auto targetTreeSize = estimateTreeSize(targetTree);
111  auto otherTreeSize = estimateTreeSize(otherTree);
112  if (auto totalSize = targetTreeSize + otherTreeSize; totalSize > 100000000) {
113  LOG(warn) << "The tree '" << targetTree->GetName() << "' would be larger than 100MB (" << totalSize << "B) after merging, skipping to let the system survive";
114  errorCode = 0;
115  } else {
116  errorCode = targetTree->Merge(&otherCollection);
117  }
118  } else if (target->InheritsFrom(TGraph::Class())) {
119  errorCode = reinterpret_cast<TGraph*>(target)->Merge(&otherCollection);
120  } else if (target->InheritsFrom(TEfficiency::Class())) {
121  errorCode = reinterpret_cast<TEfficiency*>(target)->Merge(&otherCollection);
122  } else {
123  LOG(warn) << "Object '" + std::string(target->GetName()) + "' with type '" + std::string(target->ClassName()) + "' is not one of the mergeable types, skipping";
124  }
125  if (errorCode == -1) {
126  LOG(error) << "Failed to merge the input object '" + std::string(other->GetName()) + "' of type '" + std::string(other->ClassName()) //
127  + " and the target object '" + std::string(target->GetName()) + "' of type '" + std::string(target->ClassName()) + "'";
128  }
129  }
130 }

132 void merge(VectorOfTObjectPtrs& targets, const VectorOfTObjectPtrs& others)
133 {
134  for (const auto& other : others) {
135  if (const auto targetSameName = std::find_if(targets.begin(), targets.end(), [&other](const auto& target) {
136  return std::string_view{other->GetName()} == std::string_view{target->GetName()};
137  });
138  targetSameName != targets.end()) {
139  merge(targetSameName->get(), other.get());
140  } else {
141  targets.push_back(std::shared_ptr<TObject>(other->Clone(), deleteTCollections));
142  }
143  }
144 }

146 void deleteRecursive(TCollection* Coll)
147 {
148  // I can iterate a collection
149  Coll->SetOwner(false);
150  auto ITelem = Coll->MakeIterator();
151  while (auto* element = ITelem->Next()) {
152  if (auto* Coll2 = dynamic_cast<TCollection*>(element)) {
153  Coll2->SetOwner(false);
154  deleteRecursive(Coll2);
155  }
156  Coll->Remove(element); // Remove from mother collection
157  delete element; // Delete payload
158  }
159  delete ITelem;
160 }

162 void deleteTCollections(TObject* obj)
163 {
164  if (auto* L = dynamic_cast<TCollection*>(obj)) {
165  deleteRecursive(L);
166  delete L;
167  } else {
168  delete obj;
169  }
