49class TRDGlobalTrackingQC : public Task
50{
51 public:
52 TRDGlobalTrackingQC(std::shared_ptr<DataRequest> dr, std::shared_ptr<o2::base::GRPGeomRequest> gr, bool tpcAvailable, o2::dataformats::GlobalTrackID::mask_t src) : mDataRequest(dr), mGGCCDBRequest(gr), mTPCavailable(tpcAvailable), mTrkMask(src) {}
53 ~TRDGlobalTrackingQC() override = default;
54 void init(InitContext& ic) final
55 {
56 o2::base::GRPGeomHelper::instance().setRequest(mGGCCDBRequest);
57 if (!mTPCavailable) {
58 mQC.disablePID();
59 }
60 if (getenv("ALIEN_JDL_LPMPRODUCTIONTYPE") && std::strcmp(getenv("ALIEN_JDL_LPMPRODUCTIONTYPE"), "MC") == 0) {
61 // apply artificial pad shift in case non-ideal alignment is used to compensate for shift in current alignment from real data
62 mQC.setApplyShift(false);
63 }
64 }
65 void run(ProcessingContext& pc) final
66 {
67 RecoContainer recoData;
68 recoData.collectData(pc, *mDataRequest.get());
69 updateTimeDependentParams(pc); // Make sure this is called after recoData.collectData, which may load some conditions
70 mQC.reset();
71 mQC.setInput(recoData);
72 std::vector<int> triggeredBCFT0;
73 if (mTrkMask[GTrackID::FT0]) { // pile-up tagging was requested
74 auto ft0recPoints = recoData.getFT0RecPoints();
75 uint32_t firstOrbit = 0;
76 for (size_t ft0id = 0; ft0id < ft0recPoints.size(); ft0id++) {
77 const auto& f0rec = ft0recPoints[ft0id];
78 if (ft0id == 0) {
79 firstOrbit = f0rec.getInteractionRecord().orbit;
80 mQC.setFirstOrbit(firstOrbit);
81 }
82 if (o2::ft0::InteractionTag::Instance().isSelected(f0rec)) {
83 triggeredBCFT0.push_back(f0rec.getInteractionRecord().differenceInBC({0, firstOrbit}));
84 }
85 }
86 }
87 mQC.setTriggeredBCFT0(triggeredBCFT0);
88 mQC.run();
89 pc.outputs().snapshot(Output{"TRD", "TRACKINGQC", 0}, mQC.getTrackQC());
90 }
91 void endOfStream(framework::EndOfStreamContext& ec) final {}
92 void finaliseCCDB(framework::ConcreteDataMatcher& matcher, void* obj) final
93 {
94 if (o2::base::GRPGeomHelper::instance().finaliseCCDB(matcher, obj)) {
95 return;
96 }
97 }
98
99 private:
100 void updateTimeDependentParams(framework::ProcessingContext& pc)
101 {
102 o2::base::GRPGeomHelper::instance().checkUpdates(pc);
103 static bool initOnceDone = false;
104 if (!initOnceDone) { // these params need to be queried only once
105 initOnceDone = true;
106
107 mQC.init();
108
109 // Local pad gain calibration from krypton run
110 auto localGain = *(pc.inputs().get<o2::trd::LocalGainFactor*>("localgainfactors"));
111 mQC.setLocalGainFactors(localGain);
112 }
113 }
114
115 o2::dataformats::GlobalTrackID::mask_t mTrkMask;
116 std::shared_ptr<DataRequest> mDataRequest;
117 std::shared_ptr<o2::base::GRPGeomRequest> mGGCCDBRequest;
118 bool mTPCavailable{false};
119 Tracking mQC;
120};

122DataProcessorSpec getTRDGlobalTrackingQCSpec(o2::dataformats::GlobalTrackID::mask_t src)
123{
124 std::vector<OutputSpec> outputs;
125 outputs.emplace_back("TRD", "TRACKINGQC", 0, Lifetime::Timeframe);
126 auto dataRequest = std::make_shared<DataRequest>();
127 bool isTPCavailable = false;
128
129 if (GTrackID::includesSource(GTrackID::Source::ITSTPC, src)) {
130 LOGF(debug, "Found ITS-TPC tracks as input, loading ITS-TPC-TRD");
131 src |= GTrackID::getSourcesMask("ITS-TPC-TRD");
132 }
133 if (GTrackID::includesSource(GTrackID::Source::TPC, src)) {
134 LOGF(debug, "Found TPC tracks as input, loading TPC-TRD");
135 src |= GTrackID::getSourcesMask("TPC-TRD");
136 isTPCavailable = true;
137 }
138 GTrackID::mask_t srcClu = GTrackID::getSourcesMask("TRD"); // we don't need all clusters, only TRD tracklets
139
140 dataRequest->requestTracks(src, false);
141 dataRequest->requestClusters(srcClu, false);
142 dataRequest->inputs.emplace_back("localgainfactors", "TRD", "LOCALGAINFACTORS", 0, Lifetime::Condition, ccdbParamSpec("TRD/Calib/LocalGainFactor"));
143 auto ggRequest = std::make_shared<o2::base::GRPGeomRequest>(false, // orbitResetTime
144 false, // GRPECS=true
145 false, // GRPLHCIF
146 true, // GRPMagField
147 true, // askMatLUT
148 o2::base::GRPGeomRequest::Aligned, // geometry
149 dataRequest->inputs,
150 true);
151 return DataProcessorSpec{
152 "trd-tracking-qc",
153 dataRequest->inputs,
154 outputs,
155 AlgorithmSpec{adaptFromTask<TRDGlobalTrackingQC>(dataRequest, ggRequest, isTPCavailable, src)},
156 Options{}};
157}
