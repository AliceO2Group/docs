141 o2::dataformats::V0Ext SVStudySpec::processV0(int iv, o2::globaltracking::RecoContainer& recoData)
142 {
143  o2::dataformats::V0Ext v0ext;
144  auto invalidate = [&v0ext]() { v0ext.v0ID.setVertexID(-1); };
145 
146  auto v0s = recoData.getV0s();
147  auto v0IDs = recoData.getV0sIdx();
148  static int tfID = 0;
149 
150  const auto& v0id = v0IDs[iv];
151  if (mRefit && !refitV0(v0id, v0ext.v0, recoData)) {
152  invalidate();
153  return v0ext;
154  }
155  const auto& v0sel = mRefit ? v0ext.v0 : v0s[iv];
156  if (mMaxEta < std::abs(v0sel.getEta())) {
157  invalidate();
158  return v0ext;
159  }
160  if (mSelK0 > 0 && std::abs(std::sqrt(v0sel.calcMass2AsK0()) - 0.497) > mSelK0) {
161  invalidate();
162  return v0ext;
163  }
164  if (!mRefit) {
165  v0ext.v0 = v0sel;
166  }
167  v0ext.v0ID = v0id;
168  o2::MCCompLabel lb[2];
169  const o2::MCTrack* mcTrks[2];
170  for (int ip = 0; ip < 2; ip++) {
171  auto& prInfo = v0ext.prInfo[ip];
172  auto gid = v0ext.v0ID.getProngID(ip);
173  auto gidset = recoData.getSingleDetectorRefs(gid);
174  lb[ip] = recoData.getTrackMCLabel(gid);
175  if (lb[ip].isValid()) {
176  prInfo.corrGlo = !lb[ip].isFake();
177  }
178  // get TPC tracks, if any
179  if (gidset[GTrackID::TPC].isSourceSet()) {
180  const auto& tpcTr = recoData.getTPCTrack(gidset[GTrackID::TPC]);
181  prInfo.trackTPC = tpcTr;
182  prInfo.nClTPC = tpcTr.getNClusters();
183  lb[ip] = recoData.getTrackMCLabel(gidset[GTrackID::TPC]);
184  if (lb[ip].isValid()) {
185  prInfo.corrTPC = !lb[ip].isFake();
186  }
187  }
188  // get ITS tracks, if any
189  if (gid.includesDet(DetID::ITS)) {
190  auto gidITS = recoData.getITSContributorGID(gid);
191  if (gidset[GTrackID::ITS].isSourceSet()) {
192  const auto& itsTr = recoData.getITSTrack(gidset[GTrackID::ITS]);
193  prInfo.nClITS = itsTr.getNClusters();
194  lb[ip] = recoData.getTrackMCLabel(gidset[GTrackID::ITS]);
195  if (lb[ip].isValid()) {
196  prInfo.corrITS = !lb[ip].isFake();
197  }
198  for (int il = 0; il < 7; il++) {
199  if (itsTr.hasHitOnLayer(il)) {
200  prInfo.pattITS |= 0x1 << il;
201  }
202  }
203  } else {
204  const auto& itsTrf = recoData.getITSABRefs()[gidset[GTrackID::ITSAB]];
205  prInfo.nClITS = itsTrf.getNClusters();
206  lb[ip] = recoData.getTrackMCLabel(gidset[GTrackID::ITSAB]);
207  if (lb[ip].isValid()) {
208  prInfo.corrITS = !lb[ip].isFake();
209  }
210  for (int il = 0; il < 7; il++) {
211  if (itsTrf.hasHitOnLayer(il)) {
212  prInfo.pattITS |= 0x1 << il;
213  }
214  }
215  prInfo.pattITS |= 0x1 << 31; // flag AB
216  }
217  if (gidset[GTrackID::ITSTPC].isSourceSet()) {
218  auto mtc = recoData.getTPCITSTrack(gidset[GTrackID::ITSTPC]);
219  lb[ip] = recoData.getTrackMCLabel(gidset[GTrackID::ITSTPC]);
220  prInfo.chi2ITSTPC = mtc.getChi2Match();
221  if (lb[ip].isValid()) {
222  prInfo.corrITSTPC = !lb[ip].isFake();
223  }
224  }
225  }
226  if (mUseMC && lb[ip].isValid()) { // temp store of mctrks
227  mcTrks[ip] = mcReader->getTrack(lb[ip]);
228  }
229  }
230  if (mUseMC && (mcTrks[0] != nullptr) && (mcTrks[1] != nullptr)) {
231  // check majority vote on mother particle otherwise leave pdg -1
232  if (lb[0].getSourceID() == lb[1].getSourceID() && lb[0].getEventID() == lb[1].getEventID() &&
233  mcTrks[0]->getMotherTrackId() == mcTrks[1]->getMotherTrackId() && mcTrks[0]->getMotherTrackId() >= 0) {
234  const auto mother = mcReader->getTrack(lb[0].getSourceID(), lb[0].getEventID(), mcTrks[0]->getMotherTrackId());
235  v0ext.mcPID = mother->GetPdgCode();
236  }
237  }
238  return v0ext;
239 }

241 void SVStudySpec::process(o2::globaltracking::RecoContainer& recoData)
242 {
243  auto v0IDs = recoData.getV0sIdx();
244  auto nv0 = v0IDs.size();
245  if (nv0 > recoData.getV0s().size()) {
246  mRefit = true;
247  }
248  std::map<int, std::vector<int>> pv2sv;
249  static int tfID = 0;
250  for (int iv = 0; iv < nv0; iv++) {
251  const auto v0id = v0IDs[iv];
252  pv2sv[v0id.getVertexID()].push_back(iv);
253  }
254  std::vector<o2::dataformats::V0Ext> v0extVec;
255  for (auto it : pv2sv) {
256  int pvID = it.first;
257  auto& vv = it.second;
258  if (pvID < 0 || vv.size() == 0) {
259  continue;
260  }
261  v0extVec.clear();
262  for (int iv0 : vv) {
263  auto v0ext = processV0(iv0, recoData);
264  if (v0ext.v0ID.getVertexID() < 0) {
265  continue;
266  }
267  v0extVec.push_back(v0ext);
268  }
269  if (v0extVec.size()) {
270  const auto& pv = recoData.getPrimaryVertex(pvID);
271  (*mDBGOut) << "v0"
272  << "orbit=" << recoData.startIR.orbit << "tfID=" << tfID
273  << "v0Ext=" << v0extVec
274  << "pv=" << pv
275  << "\n";
276  }
277  tfID++;
278  }
279 }

281 bool SVStudySpec::refitV0(const V0ID& id, o2::dataformats::V0& v0, o2::globaltracking::RecoContainer& recoData)
282 {
283  auto seedP = recoData.getTrackParam(id.getProngID(0));
284  auto seedN = recoData.getTrackParam(id.getProngID(1));
285  bool isTPConly = (id.getProngID(0).getSource() == GTrackID::TPC) || (id.getProngID(1).getSource() == GTrackID::TPC);
286  const auto& svparam = o2::vertexing::SVertexerParams::Instance();
287  if (svparam.mTPCTrackPhotonTune && isTPConly) {
288  mFitterV0.setMaxDZIni(svparam.mTPCTrackMaxDZIni);
289  mFitterV0.setMaxDXYIni(svparam.mTPCTrackMaxDXYIni);
290  mFitterV0.setMaxChi2(svparam.mTPCTrackMaxChi2);
291  mFitterV0.setCollinear(true);
292  }
293  int nCand = mFitterV0.process(seedP, seedN);
294  if (svparam.mTPCTrackPhotonTune && isTPConly) { // restore
295  // Reset immediately to the defaults
296  mFitterV0.setMaxDZIni(svparam.maxDZIni);
297  mFitterV0.setMaxDXYIni(svparam.maxDXYIni);
298  mFitterV0.setMaxChi2(svparam.maxChi2);
299  mFitterV0.setCollinear(false);
300  }
301  if (nCand == 0) { // discard this pair
302  return false;
303  }
304  const int cand = 0;
305  if (!mFitterV0.isPropagateTracksToVertexDone(cand) && !mFitterV0.propagateTracksToVertex(cand)) {
306  return false;
307  }
308  const auto& trPProp = mFitterV0.getTrack(0, cand);
309  const auto& trNProp = mFitterV0.getTrack(1, cand);
310  std::array<float, 3> pP{}, pN{};
311  trPProp.getPxPyPzGlo(pP);
312  trNProp.getPxPyPzGlo(pN);
313  std::array<float, 3> pV0 = {pP[0] + pN[0], pP[1] + pN[1], pP[2] + pN[2]};
314  auto p2V0 = pV0[0] * pV0[0] + pV0[1] * pV0[1] + pV0[2] * pV0[2];
315  const auto& pv = recoData.getPrimaryVertex(id.getVertexID());
316  const auto v0XYZ = mFitterV0.getPCACandidatePos(cand);
317  float dx = v0XYZ[0] - pv.getX(), dy = v0XYZ[1] - pv.getY(), dz = v0XYZ[2] - pv.getZ(), prodXYZv0 = dx * pV0[0] + dy * pV0[1] + dz * pV0[2];
318  float cosPA = prodXYZv0 / std::sqrt((dx * dx + dy * dy + dz * dz) * p2V0);
319  new (&v0) o2::dataformats::V0(v0XYZ, pV0, mFitterV0.calcPCACovMatrixFlat(cand), trPProp, trNProp);
320  v0.setDCA(mFitterV0.getChi2AtPCACandidate(cand));
321  v0.setCosPA(cosPA);
322  return true;
323 }

337 DataProcessorSpec getSVStudySpec(GTrackID::mask_t srcTracks, bool useMC)
338 {
339  std::vector<OutputSpec> outputs;
340  auto dataRequest = std::make_shared<DataRequest>();
341 
342  dataRequest->requestTracks(srcTracks, useMC);
343  dataRequest->requestPrimaryVertices(useMC);
344  dataRequest->requestSecondaryVertices(useMC);
345  dataRequest->inputs.emplace_back("meanvtx", "GLO", "MEANVERTEX", 0, Lifetime::Condition, ccdbParamSpec("GLO/Calib/MeanVertex", {}, 1));
346  auto ggRequest = std::make_shared<o2::base::GRPGeomRequest>(false, // orbitResetTime
347  false, // GRPECS=true
348  false, // GRPLHCIF
349  true, // GRPMagField
350  true, // askMatLUT
351  o2::base::GRPGeomRequest::None, // geometry
352  dataRequest->inputs,
353  true);
354 
355  return DataProcessorSpec{
356  "sv-study",
357  dataRequest->inputs,
358  outputs,
359  AlgorithmSpec{adaptFromTask<SVStudySpec>(dataRequest, ggRequest, srcTracks, useMC)},
360  Options{
361  {"refit", VariantType::Bool, false, {"refit SVertices"}},
362  {"sel-k0", VariantType::Float, -1.f, {"If positive, select K0s with this mass margin"}},
363  {"max-eta", VariantType::Float, 1.2f, {"Cut on track eta"}},
364  }};
365 }
