20void LHCIFfileReader::loadLHCIFfile(const std::string& fileName)
21{
22 // load the LHC IF file into a string
23 if (o2::utils::Str::pathExists(fileName)) {
24 std::string expandedFileName = o2::utils::Str::getFullPath(fileName);
25 std::ifstream ifs(expandedFileName.c_str());
26 if (ifs) {
27 ifs.seekg(0, std::ios::end);
28 const auto size = ifs.tellg();
29 mFileBuffStr.resize(size);
30 ifs.seekg(0);
31 ifs.read(&mFileBuffStr[0], size);
32 ifs.close();
33 }
34 } else {
35 LOG(error) << "File " << fileName << " does not exist!";
36 }
37}
