251 template <bool Always, class T, class S, typename... Args>
252 inline void GPUChain::timeCpy(RecoStep step, int32_t toGPU, S T::*func, Args... args)
253 {
254  if (!Always && step != RecoStep::NoRecoStep && !(GetRecoStepsGPU() & step)) {
255  return;
256  }
257  HighResTimer* timer = nullptr;
258  size_t* bytes = nullptr;
259  if (mRec->mProcessingSettings.debugLevel >= 1 && toGPU >= 0) { // Todo: time special cases toGPU < 0
260  int32_t id = mRec->getRecoStepNum(step, false);
261  if (id != -1) {
262  auto& tmp = mRec->mTimersRecoSteps[id];
263  timer = toGPU ? &tmp.timerToGPU : &tmp.timerToHost;
264  bytes = toGPU ? &tmp.bytesToGPU : &tmp.bytesToHost;
265  (toGPU ? tmp.countToGPU : tmp.countToHost)++;
266  timer->Start();
267  }
268  }
269  size_t n = (mRec->*func)(args...);
270  if (timer) {
271  SynchronizeGPU();
272  timer->Stop();
273  *bytes += n;
274  }
275 }

277 template <class T, class S, typename... Args>
278 bool GPUChain::DoDebugAndDump(GPUChain::RecoStep step, int32_t mask, bool transfer, T& processor, S T::*func, Args&&... args)
279 {
280  if (GetProcessingSettings().keepAllMemory) {
281  if (transfer) {
282  TransferMemoryResourcesToHost(step, &processor, -1, true);
283  }
284  if (GetProcessingSettings().debugLevel >= 6 && (mask == 0 || (GetProcessingSettings().debugMask & mask))) {
285  if (func) {
286  (processor.*func)(args...);
287  }
288  return true;
289  }
290  }
291  return false;
292 }

294 template <class T, class S, typename... Args>
295 int32_t GPUChain::runRecoStep(RecoStep step, S T::*func, Args... args)
296 {
297  if (GetRecoSteps().isSet(step)) {
298  if (GetProcessingSettings().debugLevel >= 1) {
299  mRec->getRecoStepTimer(step).Start();
300  }
301  int32_t retVal = (reinterpret_cast<T*>(this)->*func)(args...);
302  if (GetProcessingSettings().debugLevel >= 1) {
303  mRec->getRecoStepTimer(step).Stop();
304  }
305  return retVal;
306  }
307  return false;
308 }
