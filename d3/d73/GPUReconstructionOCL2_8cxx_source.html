55int32_t GPUReconstructionOCL2Backend::GetOCLPrograms()
56{
57 char platform_version[256] = {};
58 GPUFailedMsg(clGetPlatformInfo(mInternals->platform, CL_PLATFORM_VERSION, sizeof(platform_version), platform_version, nullptr));
59 float ver = 0;
60 sscanf(platform_version, "OpenCL %f", &ver);
61
62 cl_int ocl_error;
63
64 const char* ocl_flags = GPUCA_M_STR(OCL_FLAGS);
65
66#ifdef OPENCL2_ENABLED_SPIRV // clang-format off
67 if (ver >= 2.2f && !GetProcessingSettings().oclCompileFromSources) {
68 GPUInfo("Reading OpenCL program from SPIR-V IL (Platform version %4.2f)", ver);
69 mInternals->program = clCreateProgramWithIL(mInternals->context, _binary_GPUReconstructionOCL2Code_spirv_start, _binary_GPUReconstructionOCL2Code_spirv_len, &ocl_error);
70 ocl_flags = "";
71 } else
72#endif // clang-format on
73 {
74 GPUInfo("Compiling OpenCL program from sources (Platform version %4.2f)", ver);
75 size_t program_sizes[1] = {_binary_GPUReconstructionOCL2Code_src_len};
76 char* programs_sources[1] = {_binary_GPUReconstructionOCL2Code_src_start};
77 mInternals->program = clCreateProgramWithSource(mInternals->context, (cl_uint)1, (const char**)&programs_sources, program_sizes, &ocl_error);
78 }
79
80 if (GPUFailedMsgI(ocl_error)) {
81 GPUError("Error creating OpenCL program from binary");
82 return 1;
83 }
84
85 if (GPUFailedMsgI(clBuildProgram(mInternals->program, 1, &mInternals->device, ocl_flags, nullptr, nullptr))) {
86 cl_build_status status;
87 if (GPUFailedMsgI(clGetProgramBuildInfo(mInternals->program, mInternals->device, CL_PROGRAM_BUILD_STATUS, sizeof(status), &status, nullptr)) == 0 && status == CL_BUILD_ERROR) {
88 size_t log_size;
89 clGetProgramBuildInfo(mInternals->program, mInternals->device, CL_PROGRAM_BUILD_LOG, 0, nullptr, &log_size);
90 std::unique_ptr<char[]> build_log(new char[log_size + 1]);
91 clGetProgramBuildInfo(mInternals->program, mInternals->device, CL_PROGRAM_BUILD_LOG, log_size, build_log.get(), nullptr);
92 build_log[log_size] = 0;
93 GPUError("Build Log:\n\n%s\n", build_log.get());
94 }
95 return 1;
96 }

116bool GPUReconstructionOCL2Backend::CheckPlatform(uint32_t i)
117{
118 char platform_version[64] = {}, platform_vendor[64] = {};
119 clGetPlatformInfo(mInternals->platforms[i], CL_PLATFORM_VERSION, sizeof(platform_version), platform_version, nullptr);
120 clGetPlatformInfo(mInternals->platforms[i], CL_PLATFORM_VENDOR, sizeof(platform_vendor), platform_vendor, nullptr);
121 float ver1 = 0;
122 sscanf(platform_version, "OpenCL %f", &ver1);
123 if (ver1 >= 2.2f) {
124 if (mProcessingSettings.debugLevel >= 2) {
125 GPUInfo("OpenCL 2.2 capable platform found");
126 }
127 return true;
128 }
129
130 if (strcmp(platform_vendor, "Advanced Micro Devices, Inc.") == 0 && ver1 >= 2.0f) {
131 float ver2 = 0;
132 const char* pos = strchr(platform_version, '(');
133 if (pos) {
134 sscanf(pos, "(%f)", &ver2);
135 }
136 if ((ver1 >= 2.f && ver2 >= 2000.f) || ver1 >= 2.1f) {
137 if (mProcessingSettings.debugLevel >= 2) {
138 GPUInfo("AMD ROCm OpenCL Platform found");
139 }
140 return true;
141 }
142 }
143 return false;
144}
