114//___________________________________________________________
115void RawReaderSpecs::processDropTF(const std::string& dropTF)
116{
117 static const std::regex delimDet(";");
118 if (dropTF.empty() || dropTF == "none") {
119 return;
120 }
121 std::sregex_token_iterator iter(dropTF.begin(), dropTF.end(), delimDet, -1), end;
122 for (; iter != end; ++iter) {
123 std::string sdet = iter->str();
124 if (sdet.length() < 5 || sdet[3] != ',') {
125 throw std::runtime_error(fmt::format("Wrong dropTF argument {} in {}", sdet, dropTF));
126 }
127 std::string detName = sdet.substr(0, 3);
128 o2::detectors::DetID det(detName.c_str()); // make sure this is a valid detector
129 std::string sdetArg = sdet.substr(4, sdet.length());
130 int modV = 0, rej = 0, posrej = sdetArg.find(',');
131 if (posrej != std::string::npos) {
132 modV = std::stoi(sdetArg.substr(0, posrej));
133 rej = std::stoi(sdetArg.substr(++posrej, sdetArg.length()));
134 } else {
135 modV = std::stoi(sdetArg);
136 }
137 if (modV < 1 || rej < 0 || rej >= modV) {
138 throw std::runtime_error(fmt::format("Wrong dropTF argument {}, 1st number must be > than 2nd", sdet));
139 }
140 mDropTFMap[detName] = {modV, rej};
141 LOG(info) << " Will drop TF for detector " << detName << " if (TF_ID%" << modV << ")==" << rej;
142 }
143}

146void RawReaderSpecs::init(o2f::InitContext& ic)
147{
148 assert(mReader);
149 mTimer.Start();
150 mTimer.Stop();
151 mVerbosity = ic.options().get<int>("verbosity-level");
152 mRunNumber = ic.options().get<int>("run-number");
153 mReader->setVerbosity(mVerbosity);
154 mReader->init();
155 if (mMaxTFID >= mReader->getNTimeFrames()) {
156 mMaxTFID = mReader->getNTimeFrames() ? mReader->getNTimeFrames() - 1 : 0;
157 }
158 const auto& hbfU = HBFUtils::Instance();
159 if (!hbfU.startTime) {
160 hbfU.setValue("HBFUtils.startTime", std::to_string(std::chrono::duration_cast<std::chrono::milliseconds>(std::chrono::system_clock::now().time_since_epoch()).count()));
161 LOG(warning) << "Run start time is not provided via HBFUtils.startTime, will use now() = " << hbfU.startTime << " ms.";
162 }
163 if (mRunNumber == 0 && hbfU.runNumber > 0) {
164 mRunNumber = hbfU.runNumber;
165 }

168//___________________________________________________________
169void RawReaderSpecs::run(o2f::ProcessingContext& ctx)
170{
171 assert(mReader);
172 auto tTotStart = mTimer.CpuTime();
173 mTimer.Start(false);
174 auto device = ctx.services().get<o2f::RawDeviceService>().device();
175 assert(device);
176 if (mTFRateLimit == -999) {
177 mTFRateLimit = std::stoi(device->fConfig->GetValue<std::string>("timeframes-rate-limit"));
178 }
179 auto findOutputChannel = [&ctx, this](o2h::DataHeader& h) {
180 if (!this->mRawChannelName.empty()) {
181 return std::string{this->mRawChannelName};
182 } else {
183 auto outputRoutes = ctx.services().get<o2f::RawDeviceService>().spec().outputs;
184 for (auto& oroute : outputRoutes) {
185 LOG(debug) << "comparing with matcher to route " << oroute.matcher << " TSlice:" << oroute.timeslice;
186 if (o2f::DataSpecUtils::match(oroute.matcher, h.dataOrigin, h.dataDescription, h.subSpecification) && ((mTFCounter % oroute.maxTimeslices) == oroute.timeslice)) {
187 LOG(debug) << "picking the route:" << o2f::DataSpecUtils::describe(oroute.matcher) << " channel " << oroute.channel;
188 return std::string{oroute.channel};
189 }
190 }
191 }
192 LOGP(error, "Failed to find output channel for {}/{}/{} @ timeslice {}", h.dataOrigin, h.dataDescription, h.subSpecification, h.tfCounter);
193 auto outputRoutes = ctx.services().get<o2f::RawDeviceService>().spec().outputs;
194 for (auto& oroute : outputRoutes) {
195 LOGP(info, "Available output routes: {} channel: {}", o2f::DataSpecUtils::describe(oroute.matcher), oroute.channel);
196 }
197 return std::string{};
198 };
199
200 size_t tfNParts = 0, tfSize = 0;
201 std::unordered_map<std::string, std::unique_ptr<fair::mq::Parts>> messagesPerRoute;
202
203 auto addPart = [&messagesPerRoute, &tfNParts, &tfSize](fair::mq::MessagePtr hd, fair::mq::MessagePtr pl, const std::string& fairMQChannel) {
204 fair::mq::Parts* parts = nullptr;
205 parts = messagesPerRoute[fairMQChannel].get(); // fair::mq::Parts*
206 if (!parts) {
207 messagesPerRoute[fairMQChannel] = std::make_unique<fair::mq::Parts>();
208 parts = messagesPerRoute[fairMQChannel].get();
209 }
210 tfSize += pl->GetSize();
211 tfNParts++;
212 parts->AddPart(std::move(hd));
213 parts->AddPart(std::move(pl));
214 };
215
216 // clean-up before reading next TF
217 auto tfID = mReader->getNextTFToRead();
218 int nlinks = mReader->getNLinks();
219
220 if (tfID < mMinTFID) {
221 tfID = mMinTFID;
222 }
223 mReader->setNextTFToRead(tfID);
224 std::vector<RawFileReader::PartStat> partsSP;
225
226 static o2f::RateLimiter limiter;
227 limiter.check(ctx, mTFRateLimit, mMinSHM);
228
229 // read next time frame
230 LOG(info) << "Reading TF#" << mTFCounter << " (" << tfID << " at iteration " << mLoopsDone << ')';
231 o2::header::Stack dummyStack{o2h::DataHeader{}, o2f::DataProcessingHeader{0}}; // dummy stack to just to get stack size
232 auto hstackSize = dummyStack.size();
233
234 uint32_t firstOrbit = 0;
235 uint64_t creationTime = 0;
236 const auto& hbfU = HBFUtils::Instance();
237
238 for (int il = 0; il < nlinks; il++) {
239 auto& link = mReader->getLink(il);
240
241 if (!mDropTFMap.empty()) { // some TFs should be dropped
242 auto res = mDropTFMap.find(link.origin.str);
243 if (res != mDropTFMap.end() && (mTFCounter % res->second.first) == res->second.second) {
244 LOG(info) << "Dropping " << mTFCounter << " for " << link.origin.str << "/" << link.description.str << "/" << link.subspec;
245 continue; // drop the data
246 }
247 }
248 if (!link.rewindToTF(tfID)) {
249 continue; // this link has no data for wanted TF
250 }
251
252 o2h::DataHeader hdrTmpl(link.description, link.origin, link.subspec); // template with 0 size
253 int nParts = mPartPerSP ? link.getNextTFSuperPagesStat(partsSP) : link.getNHBFinTF();
254 hdrTmpl.payloadSerializationMethod = o2h::gSerializationMethodNone;
255 hdrTmpl.splitPayloadParts = nParts;
256 hdrTmpl.tfCounter = mTFCounter;
257 hdrTmpl.runNumber = mRunNumber;
258 if (mVerbosity > 1) {
259 LOG(info) << link.describe() << " will read " << nParts << " HBFs starting from block " << link.nextBlock2Read;
260 }
261 const auto fmqChannel = findOutputChannel(hdrTmpl);
262 if (fmqChannel.empty()) { // no output channel
263 continue;
264 }
265
266 auto fmqFactory = device->GetChannel(fmqChannel, 0).Transport();
267 while (hdrTmpl.splitPayloadIndex < hdrTmpl.splitPayloadParts) {
268 hdrTmpl.payloadSize = mPartPerSP ? partsSP[hdrTmpl.splitPayloadIndex].size : link.getNextHBFSize();
269 auto hdMessage = fmqFactory->CreateMessage(hstackSize, fair::mq::Alignment{64});
270 auto plMessage = fmqFactory->CreateMessage(hdrTmpl.payloadSize, fair::mq::Alignment{64});
271 auto bread = mPartPerSP ? link.readNextSuperPage(reinterpret_cast<char*>(plMessage->GetData()), &partsSP[hdrTmpl.splitPayloadIndex]) : link.readNextHBF(reinterpret_cast<char*>(plMessage->GetData()));
272 if (bread != hdrTmpl.payloadSize) {
273 LOG(error) << "Link " << il << " read " << bread << " bytes instead of " << hdrTmpl.payloadSize
274 << " expected in TF=" << mTFCounter << " part=" << hdrTmpl.splitPayloadIndex;
275 }
276 // check if the RDH to send corresponds to expected orbit
277 if (hdrTmpl.splitPayloadIndex == 0) {
278 auto ir = o2::raw::RDHUtils::getHeartBeatIR(plMessage->GetData());
279 auto tfid = hbfU.getTF(ir);
280 firstOrbit = hdrTmpl.firstTForbit = (mPreferCalcTF || !link.cruDetector) ? hbfU.getIRTF(tfid).orbit : ir.orbit; // will be picked for the following parts
281 creationTime = hbfU.getTFTimeStamp({0, firstOrbit});
282 }
283 o2::header::Stack headerStack{hdrTmpl, o2f::DataProcessingHeader{mTFCounter, 1, creationTime}};
284 memcpy(hdMessage->GetData(), headerStack.data(), headerStack.size());
285 hdrTmpl.splitPayloadIndex++; // prepare for next
286
287 addPart(std::move(hdMessage), std::move(plMessage), fmqChannel);
288 }
289 LOGF(debug, "Added %d parts for TF#%d(%d in iteration %d) of %s/%s/0x%u", hdrTmpl.splitPayloadParts, mTFCounter, tfID,
290 mLoopsDone, link.origin.as<std::string>(), link.description.as<std::string>(), link.subspec);
291 }
292
293 auto& timingInfo = ctx.services().get<o2f::TimingInfo>();
294 timingInfo.firstTForbit = firstOrbit;
295 timingInfo.creation = creationTime;
296 timingInfo.tfCounter = mTFCounter;
297 timingInfo.runNumber = mRunNumber;
298
299 // send sTF acknowledge message
300 unsigned stfSS[2] = {0, 0xccdb};
301 for (int iss = 0; iss < (mSup0xccdb ? 1 : 2); iss++) {
302 o2::header::STFHeader stfHeader{mTFCounter, firstOrbit, 0};
303 o2::header::DataHeader stfDistDataHeader(o2::header::gDataDescriptionDISTSTF, o2::header::gDataOriginFLP, stfSS[iss], sizeof(o2::header::STFHeader), 0, 1);
304 stfDistDataHeader.runNumber = mRunNumber;
305 stfDistDataHeader.payloadSerializationMethod = o2h::gSerializationMethodNone;
306 stfDistDataHeader.firstTForbit = stfHeader.firstOrbit;
307 stfDistDataHeader.tfCounter = mTFCounter;
308 const auto fmqChannel = findOutputChannel(stfDistDataHeader);
309 if (!fmqChannel.empty()) { // no output channel
310 auto fmqFactory = device->GetChannel(fmqChannel, 0).Transport();
311 o2::header::Stack headerStackSTF{stfDistDataHeader, o2f::DataProcessingHeader{mTFCounter, 1, creationTime}};
312 auto hdMessageSTF = fmqFactory->CreateMessage(hstackSize, fair::mq::Alignment{64});
313 auto plMessageSTF = fmqFactory->CreateMessage(stfDistDataHeader.payloadSize, fair::mq::Alignment{64});
314 memcpy(hdMessageSTF->GetData(), headerStackSTF.data(), headerStackSTF.size());
315 memcpy(plMessageSTF->GetData(), &stfHeader, sizeof(o2::header::STFHeader));
316 addPart(std::move(hdMessageSTF), std::move(plMessageSTF), fmqChannel);
317 }
318 }
319
320 if (mTFCounter) { // delay sending
321 std::this_thread::sleep_for(std::chrono::microseconds((size_t)mDelayUSec));
322 }
323 bool sentSomething = false;
324 for (auto& msgIt : messagesPerRoute) {
325 LOG(info) << "Sending " << msgIt.second->Size() / 2 << " parts to channel " << msgIt.first;
326 device->Send(*msgIt.second.get(), msgIt.first);
327 sentSomething = msgIt.second->Size() > 0;
328 }
329 if (sentSomething) {
330 ctx.services().get<o2f::MessageContext>().fakeDispatch();
331 }
332
333 mTimer.Stop();
334
335 LOGP(info, "Sent payload of {} bytes in {} parts in {} messages for TF#{} firstTForbit={} timeStamp={} | Timing: {}", tfSize, tfNParts,
336 messagesPerRoute.size(), mTFCounter, firstOrbit, creationTime, mTimer.CpuTime() - tTotStart);
337
338 mSentSize += tfSize;
339 mSentMessages += tfNParts;
340 mReader->setNextTFToRead(++tfID);
341 ++mTFCounter;
342 if (tfID > mMaxTFID || mReader->isProcessingStopped()) {
343 if (!mReader->isProcessingStopped() && !mReader->isEmpty() && --mLoop) {
344 mLoopsDone++;
345 mReader->setNextTFToRead(tfID = 0);
346 LOG(info) << "Shall start new loop " << mLoopsDone << " from the beginning of data";
347 } else {
348 if (!mRawChannelName.empty()) { // send endOfStream message to raw channel
349 o2f::SourceInfoHeader exitHdr;
350 exitHdr.state = o2f::InputChannelState::Completed;
351 o2h::DataHeader dh = o2h::DataHeader(o2h::gDataDescriptionInfo, o2h::gDataOriginAny, 0, 0);
352 try {
353 dh.runNumber = strtoul(device->fConfig->GetProperty<std::string>("runNumber", "").c_str(), nullptr, 10);
354 } catch (...) {
355 }
356 const auto exitStack = o2::header::Stack(dh, o2f::DataProcessingHeader(), exitHdr);
357 auto fmqFactory = device->GetChannel(mRawChannelName, 0).Transport();
358 auto hdEOSMessage = fmqFactory->CreateMessage(exitStack.size(), fair::mq::Alignment{64});
359 auto plEOSMessage = fmqFactory->CreateMessage(0, fair::mq::Alignment{64});
360 memcpy(hdEOSMessage->GetData(), exitStack.data(), exitStack.size());
361 fair::mq::Parts eosMsg;
362 eosMsg.AddPart(std::move(hdEOSMessage));
363 eosMsg.AddPart(std::move(plEOSMessage));
364 device->Send(eosMsg, mRawChannelName);
365 LOG(info) << "Sent EoS message to " << mRawChannelName;
366 } else {
367 ctx.services().get<o2f::ControlService>().endOfStream();
368 }
369 ctx.services().get<o2f::ControlService>().readyToQuit(o2f::QuitRequest::Me);
370 mTimer.Stop();
371 LOGP(info, "Finished: payload of {} bytes in {} messages sent for {} TFs, total timing: Real:{:3f}/CPU:{:3f}", mSentSize, mSentMessages, mTFCounter, mTimer.RealTime(), mTimer.CpuTime());
372 }
373 }
374}

377o2f::DataProcessorSpec getReaderSpec(ReaderInp rinp)
378{
379 // check which inputs are present in files to read
380 o2f::DataProcessorSpec spec;
381 spec.name = "raw-file-reader";
382 std::string rawChannelName = "";
383 if (rinp.rawChannelConfig.empty()) {
384 if (!rinp.inifile.empty()) {
385 auto conf = o2::raw::RawFileReader::parseInput(rinp.inifile, rinp.onlyDet);
386 for (const auto& entry : conf) {
387 const auto& ordescard = entry.first;
388 if (!entry.second.empty()) { // origin and decription for files to process
389 spec.outputs.emplace_back(o2f::OutputSpec(o2f::ConcreteDataTypeMatcher{std::get<0>(ordescard), std::get<1>(ordescard)}));
390 }
391 }
392 }
393 // add output for DISTSUBTIMEFRAME
394 spec.outputs.emplace_back(o2f::OutputSpec{{"stfDist"}, o2::header::gDataOriginFLP, o2::header::gDataDescriptionDISTSTF, 0});
395 if (!rinp.sup0xccdb) {
396 spec.outputs.emplace_back(o2f::OutputSpec{{"stfDistCCDB"}, o2::header::gDataOriginFLP, o2::header::gDataDescriptionDISTSTF, 0xccdb}); // will be added automatically
397 }
398 if (!rinp.metricChannel.empty()) {
399 spec.options.emplace_back(o2f::ConfigParamSpec{"channel-config", o2f::VariantType::String, rinp.metricChannel, {"Out-of-band channel config for TF throttling"}});
400 }
401 } else {
402 auto nameStart = rinp.rawChannelConfig.find("name=");
403 if (nameStart == std::string::npos) {
404 throw std::runtime_error("raw channel name is not provided");
405 }
406 nameStart += strlen("name=");
407 auto nameEnd = rinp.rawChannelConfig.find(",", nameStart + 1);
408 if (nameEnd == std::string::npos) {
409 nameEnd = rinp.rawChannelConfig.size();
410 }
411 spec.options.emplace_back(o2f::ConfigParamSpec{"channel-config", o2f::VariantType::String, rinp.rawChannelConfig, {"Out-of-band channel config"}});
412 rinp.rawChannelConfig = rinp.rawChannelConfig.substr(nameStart, nameEnd - nameStart);
413 if (!rinp.metricChannel.empty()) {
414 LOGP(alarm, "Cannot apply TF rate limiting when publishing to raw channel, limiting must be applied on the level of the input raw proxy");
415 LOGP(alarm, R"(To avoid reader filling shm buffer use "--shm-throw-bad-alloc 0 --shm-segment-id 2")");
416 }
417 LOG(info) << "Will send output to non-DPL channel " << rinp.rawChannelConfig;
418 }
419
420 spec.algorithm = o2f::adaptFromTask<RawReaderSpecs>(rinp);
421 spec.options.emplace_back(o2f::ConfigParamSpec{"verbosity-level", o2f::VariantType::Int, 0, {"verbosity level"}});
422 spec.options.emplace_back(o2f::ConfigParamSpec{"run-number", o2f::VariantType::Int, 0, {"impose run number"}});
423
424 return spec;
425}
