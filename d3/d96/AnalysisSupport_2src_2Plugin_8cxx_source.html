33  .summaryHandling = [](ServiceMetricsInfo const& info) {
34  LOGP(info, "## Analysis Run Summary ##");

37  DeviceMetricsInfo &metrics = info.deviceMetricsInfos[mi];
38  for (size_t li = 0; li < metrics.metricLabels.size(); ++li) {
39  MetricLabel const&label = metrics.metricLabels[li];
40  if (strcmp(label.label, "aod-file-read-info") != 0) {
41  continue;
42  }
43  MetricInfo const&metric = metrics.metrics[li];
44  auto &files = metrics.stringMetrics[metric.storeIdx];
45  if (metric.filledMetrics) {
46  LOGP(info, "### Files read stats ###");
47  }
48  for (size_t fi = 0; fi < metric.filledMetrics; ++fi) {
49  LOGP(info, "{}", files[fi % files.size()].data);
50  }
51  }
52  } },
53  .kind = ServiceKind::Serial};
