36 struct RunSummary : o2::framework::ServicePlugin {

38  {
39  return new o2::framework::ServiceSpec{
40  .name = "analysis-run-summary",
41  .init = [](ServiceRegistryRef ref, DeviceState&, fair::mq::ProgOptions&) -> ServiceHandle {
42  return ServiceHandle{TypeIdHelpers::uniqueId<RunSummary>(), nullptr, ServiceKind::Serial, "analysis-run-summary"};
43  },
44  .summaryHandling = [](ServiceMetricsInfo const& info) {
45  LOGP(info, "## Analysis Run Summary ##");
47  for (size_t mi = 0; mi < info.deviceMetricsInfos.size(); ++mi) {
48  DeviceMetricsInfo &metrics = info.deviceMetricsInfos[mi];
49  for (size_t li = 0; li < metrics.metricLabels.size(); ++li) {
50  MetricLabel const&label = metrics.metricLabels[li];
51  if (strcmp(label.label, "aod-file-read-info") != 0) {
52  continue;
53  }
54  MetricInfo const&metric = metrics.metrics[li];
55  auto &files = metrics.stringMetrics[metric.storeIdx];
56  if (metric.filledMetrics) {
57  LOGP(info, "### Files read stats ###");
58  }
59  for (size_t fi = 0; fi < metric.filledMetrics; ++fi) {
60  LOGP(info, "{}", files[fi % files.size()].data);
61  }
62  }
63  } },
64  .kind = ServiceKind::Serial};
65  }

68 std::vector<std::string> getListOfTables(TFile* f)
69 {
70  std::vector<std::string> r;
71  TList* keyList = f->GetListOfKeys();
72 
73  for (auto key : *keyList) {
74  if (!std::string_view(key->GetName()).starts_with("DF_")) {
75  continue;
76  }
77  auto* d = (TDirectory*)f->Get(key->GetName());
78  TList* branchList = d->GetListOfKeys();
79  for (auto b : *branchList) {
80  r.emplace_back(b->GetName());
81  }
82  break;
83  }
84  return r;
85 }

87 struct DiscoverMetadataInAOD : o2::framework::ConfigDiscoveryPlugin {
88  ConfigDiscovery* create() override
89  {
90  return new ConfigDiscovery{
91  .init = []() {},
92  .discover = [](ConfigParamRegistry& registry, int argc, char** argv) -> std::vector<ConfigParamSpec> {
93  auto filename = registry.get<std::string>("aod-file");
94  if (filename.empty()) {
95  return {};
96  }
97  std::vector<ConfigParamSpec> results;
98  TFile* currentFile = nullptr;
99  if (filename.at(0) == '@') {
100  filename.erase(0, 1);
101  // read the text file and set filename to the contents of the first line
102  std::ifstream file(filename);
103  if (!file.is_open()) {
104  LOGP(fatal, "Couldn't open file \"{}\"!", filename);
105  }
106  std::getline(file, filename);
107  file.close();
108  }
109  if (filename.rfind("alien://", 0) == 0) {
110  TGrid::Connect("alien://");
111  }
112  LOGP(info, "Loading metadata from file {} in PID {}", filename, getpid());
113  currentFile = TFile::Open(filename.c_str());
114  if (!currentFile) {
115  LOGP(fatal, "Couldn't open file \"{}\"!", filename);
116  }
117 
118  // Get the metadata, if any
119  auto m = (TMap*)currentFile->Get("metaData");
120  if (!m) {
121  LOGP(info, "No metadata found in file \"{}\"", filename);
122  results.push_back(ConfigParamSpec{"aod-metadata-disable", VariantType::String, "1", {"Metadata not found in AOD"}});
123  return results;
124  }
125  auto it = m->MakeIterator();
126 
127  // Serialise metadata into a ; separated string with : separating key and value
128  bool first = true;
129  while (auto obj = it->Next()) {
130  if (first) {
131  LOGP(info, "Metadata for file \"{}\":", filename);
132  first = false;
133  }
134  auto objString = (TObjString*)m->GetValue(obj);
135  LOGP(info, "- {}: {}", obj->GetName(), objString->String().Data());
136  std::string key = "aod-metadata-" + std::string(obj->GetName());
137  char const* value = strdup(objString->String());
138  results.push_back(ConfigParamSpec{key, VariantType::String, value, {"Metadata in AOD"}});
139  }
140 
141  auto tables = getListOfTables(currentFile);
142  if (tables.empty() == false) {
143  results.push_back(ConfigParamSpec{"aod-metadata-tables", VariantType::ArrayString, tables, {"Tables in first AOD"}});
144  }
145  return results;
146  }};
147  }
148 };
