47 void RawDecoderSpec::endOfStream(framework::EndOfStreamContext& ec)
48 {
49  auto& TFOrbits = mDecoder.getTFOrbits();
50  std::sort(TFOrbits.begin(), TFOrbits.end());
51  size_t l = TFOrbits.size();
52  uint32_t o0 = 0;
53  if (l) {
54  o0 = TFOrbits[0];
55  }
56  int nmiss = 0;
57  int nprt = 0;
58  std::cout << "Missing orbits:";
59  for (int i = 1; i < l; i++) {
60  if ((TFOrbits[i] - o0) > 0x20) {
61  if (nprt < 20) {
62  std::cout << " " << o0 << "-" << TFOrbits[i];
63  }
64  nmiss += (TFOrbits[i] - o0) / 0x20;
65  nprt++;
66  }
67  o0 = TFOrbits[i];
68  }
69  std::cout << std::endl;
70  std::cout << "Number of missing TF:" << nmiss << std::endl;
71  std::cout << "# of IR errors:" << mDecoder.getErrorIR() << " TCR errors:" << mDecoder.getErrorTCR() << std::endl;
72 }
73 void RawDecoderSpec::run(framework::ProcessingContext& ctx)
74 {
75  mOutputDigits.clear();
76  std::map<o2::InteractionRecord, CTPDigit> digits;
77  using InputSpec = o2::framework::InputSpec;
78  using ConcreteDataTypeMatcher = o2::framework::ConcreteDataTypeMatcher;
79  using Lifetime = o2::framework::Lifetime;
80  // setUpDummyLink
81  auto& inputs = ctx.inputs();
82  auto dummyOutput = [&ctx, this]() {
83  if (this->mDoDigits) {
84  ctx.outputs().snapshot(o2::framework::Output{"CTP", "DIGITS", 0}, this->mOutputDigits);
85  }
86  if (this->mDoLumi) {
87  ctx.outputs().snapshot(o2::framework::Output{"CTP", "LUMI", 0}, this->mOutputLumiInfo);
88  }
89  };
90  // if we see requested data type input with 0xDEADBEEF subspec and 0 payload this means that the "delayed message"
91  // mechanism created it in absence of real data from upstream. Processor should send empty output to not block the workflow
92  {
93  static size_t contDeadBeef = 0; // number of times 0xDEADBEEF was seen continuously
94  std::vector<InputSpec> dummy{InputSpec{"dummy", o2::framework::ConcreteDataMatcher{"CTP", "RAWDATA", 0xDEADBEEF}}};
95  for (const auto& ref : o2::framework::InputRecordWalker(inputs, dummy)) {
96  const auto dh = o2::framework::DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
97  auto payloadSize = o2::framework::DataRefUtils::getPayloadSize(ref);
98  if (payloadSize == 0) {
99  auto maxWarn = o2::conf::VerbosityConfig::Instance().maxWarnDeadBeef;
100  if (++contDeadBeef <= maxWarn) {
101  LOGP(alarm, "Found input [{}/{}/{:#x}] TF#{} 1st_orbit:{} Payload {} : assuming no payload for all links in this TF{}",
102  dh->dataOrigin.str, dh->dataDescription.str, dh->subSpecification, dh->tfCounter, dh->firstTForbit, payloadSize,
103  contDeadBeef == maxWarn ? fmt::format(". {} such inputs in row received, stopping reporting", contDeadBeef) : "");
104  }
105  dummyOutput();
106  return;
107  }
108  }
109  contDeadBeef = 0; // if good data, reset the counter
110  }
111  //
112  std::vector<LumiInfo> lumiPointsHBF1;
113  std::vector<InputSpec> filter{InputSpec{"filter", ConcreteDataTypeMatcher{"CTP", "RAWDATA"}, Lifetime::Timeframe}};
114  if (mMaxInputSize > 0) {
115  size_t payloadSize = 0;
116  for (const auto& ref : o2::framework::InputRecordWalker(inputs, filter)) {
117  const auto dh = o2::framework::DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
118  payloadSize += o2::framework::DataRefUtils::getPayloadSize(ref);
119  }
120  if (payloadSize > (size_t)mMaxInputSize) {
121  LOG(fatal) << "Input data size:" << payloadSize;
122  dummyOutput();
123  return;
124  }
125  }
126  int ret = mDecoder.decodeRaw(inputs, filter, mOutputDigits, lumiPointsHBF1);
127  if (ret == 1) {
128  dummyOutput();
129  return;
130  }
131  if (mDoDigits) {
132  LOG(info) << "[CTPRawToDigitConverter - run] Writing " << mOutputDigits.size() << " digits. IR rejected:" << mDecoder.getIRRejected() << " TCR rejected:" << mDecoder.getTCRRejected();
133  ctx.outputs().snapshot(o2::framework::Output{"CTP", "DIGITS", 0}, mOutputDigits);
134  }
135  if (mDoLumi) {
136  uint32_t tfCountsT = 0;
137  uint32_t tfCountsV = 0;
138  for (auto const& lp : lumiPointsHBF1) {
139  tfCountsT += lp.counts;
140  tfCountsV += lp.countsFV0;
141  }
142  // LOG(info) << "Lumi rate:" << tfCounts/(128.*88e-6);
143  // FT0
144  mHistoryT.push_back(tfCountsT);
145  mCountsT += tfCountsT;
146  if (mHistoryT.size() <= mNTFToIntegrate) {
147  mNHBIntegratedT += lumiPointsHBF1.size();
148  } else {
149  mCountsT -= mHistoryT.front();
150  mHistoryT.pop_front();
151  }
152  // FV0
153  mHistoryV.push_back(tfCountsV);
154  mCountsV += tfCountsV;
155  if (mHistoryV.size() <= mNTFToIntegrate) {
156  mNHBIntegratedV += lumiPointsHBF1.size();
157  } else {
158  mCountsV -= mHistoryV.front();
159  mHistoryV.pop_front();
160  }
161  //
162  if (mNHBIntegratedT || mNHBIntegratedV) {
163  mOutputLumiInfo.orbit = lumiPointsHBF1[0].orbit;
164  }
165  mOutputLumiInfo.counts = mCountsT;
166  mOutputLumiInfo.countsFV0 = mCountsV;
167  mOutputLumiInfo.nHBFCounted = mNHBIntegratedT;
168  mOutputLumiInfo.nHBFCountedFV0 = mNHBIntegratedV;
169  if (mVerbose) {
170  mOutputLumiInfo.printInputs();
171  LOGP(info, "Orbit {}: {}/{} counts inp1/inp2 in {}/{} HBFs -> lumi_inp1 = {:.3e}+-{:.3e} lumi_inp2 = {:.3e}+-{:.3e}", mOutputLumiInfo.orbit, mCountsT, mCountsV, mNHBIntegratedT, mNHBIntegratedV, mOutputLumiInfo.getLumi(), mOutputLumiInfo.getLumiError(), mOutputLumiInfo.getLumiFV0(), mOutputLumiInfo.getLumiFV0Error());
172  }
173  ctx.outputs().snapshot(o2::framework::Output{"CTP", "LUMI", 0}, mOutputLumiInfo);
174  }

176 o2::framework::DataProcessorSpec o2::ctp::reco_workflow::getRawDecoderSpec(bool askDISTSTF, bool digits, bool lumi)
177 {
178  if (!digits && !lumi) {
179  throw std::runtime_error("all outputs were disabled");
180  }
181  std::vector<o2::framework::InputSpec> inputs;
182  inputs.emplace_back("TF", o2::framework::ConcreteDataTypeMatcher{"CTP", "RAWDATA"}, o2::framework::Lifetime::Timeframe);
183  if (askDISTSTF) {
184  inputs.emplace_back("stdDist", "FLP", "DISTSUBTIMEFRAME", 0, o2::framework::Lifetime::Timeframe);
185  }
186 
187  std::vector<o2::framework::OutputSpec> outputs;
188  if (digits) {
189  outputs.emplace_back("CTP", "DIGITS", 0, o2::framework::Lifetime::Timeframe);
190  }
191  if (lumi) {
192  outputs.emplace_back("CTP", "LUMI", 0, o2::framework::Lifetime::Timeframe);
193  }
194  return o2::framework::DataProcessorSpec{
195  "ctp-raw-decoder",
196  inputs,
197  outputs,
198  o2::framework::AlgorithmSpec{o2::framework::adaptFromTask<o2::ctp::reco_workflow::RawDecoderSpec>(digits, lumi)},
199  o2::framework::Options{
200  {"ntf-to-average", o2::framework::VariantType::Int, 90, {"Time interval for averaging luminosity in units of TF"}},
201  {"print-errors-num", o2::framework::VariantType::Int, 3, {"Max number of errors to print"}},
202  {"lumi-inp1", o2::framework::VariantType::String, "TVX", {"The first input used for online lumi. Name in capital."}},
203  {"lumi-inp2", o2::framework::VariantType::String, "VBA", {"The second input used for online lumi. Name in capital."}},
204  {"use-verbose-mode", o2::framework::VariantType::Bool, false, {"Verbose logging"}},
205  {"max-input-size", o2::framework::VariantType::Int, 0, {"Do not process input if bigger than max size, 0 - do not check"}},
206  {"ctpinputs-decoding", o2::framework::VariantType::Bool, false, {"Inputs alignment: true - raw decoder - has to be compatible with CTF decoder: allowed options: 10,01,00"}}}};
207 }
