58 Double_t SimParam::getTimeResolution(Double_t energy) const
59 {
60  Double_t res = -1;
61  if (energy > 0) {
62  res = TMath::Sqrt(mTimeResolutionPar0 + mTimeResolutionPar1 / (energy * energy));
63  }
64 
65  // parametrization above is for ns. Convert to seconds:
66  // return res*1e-9;
67  return res;
68 }
