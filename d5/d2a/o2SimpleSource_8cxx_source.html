36 // This is how you can define your processing in a declarative way
37 WorkflowSpec defineDataProcessing(ConfigContext const& ctx)
38 {
39  // Get the dataspec option and creates OutputSpecs from it
40  auto dataspec = ctx.options().get<std::string>("dataspec");
41  auto timer = ctx.options().get<std::string>("timer");
42  std::vector<InputSpec> inputs;
43  std::vector<TimerSpec> timers;
44  if (timer.empty() == false) {
45  // Split timer at every comma, then split each part at every colon
46  // and create a TimerIntervalSpec from it.
47  while (true) {
48  auto comma = timer.find(',');
49  auto colon = timer.find(':');
50  if (colon == std::string::npos) {
51  break;
52  }
53  auto validity = std::stoull(timer.substr(0, colon));
54  auto period = std::stoull(timer.substr(colon + 1, comma - colon - 1));
55  timers.push_back(TimerSpec{.period = period, .validity = validity});
56  if (comma == std::string::npos) {
57  break;
58  }
59  timer = timer.substr(comma + 1);
60  }
61  inputs.emplace_back("timer", "TST", "TIMER", 0, Lifetime::Timer, timerSpecs(timers));
62  }
63 
64  std::vector<InputSpec> matchers = select(dataspec.c_str());
65  std::vector<std::string> outputRefs;
66  std::vector<OutputSpec> outputSpecs;
67  for (auto const& matcher : matchers) {
68  outputRefs.emplace_back(matcher.binding);
69  outputSpecs.emplace_back(DataSpecUtils::asOutputSpec(matcher));
70  }
71 
72  return WorkflowSpec{
73  {.name = ctx.options().get<std::string>("name"),
74  .inputs = inputs,
75  .outputs = outputSpecs,
76  .algorithm = AlgorithmSpec{adaptStateful(
77  [outputSpecs](ConfigParamRegistry const& options) {
78  // the size of the messages is also a workflow option
79  auto dataSize = options.get<int64_t>("data-size");
80  return adaptStateless(
81  [outputSpecs, dataSize](DataAllocator& outputs, ProcessingContext& ctx) {
82  for (auto const& output : outputSpecs) {
83  auto concrete = DataSpecUtils::asConcreteDataMatcher(output);
84  outputs.make<char>(Output{concrete.origin, concrete.description, concrete.subSpec}, dataSize);
85  }
86  });
87  })},
88  .options = {ConfigParamSpec{"data-size", VariantType::Int64, 1LL, {"Size of the created messages"}}}}};
89 }
