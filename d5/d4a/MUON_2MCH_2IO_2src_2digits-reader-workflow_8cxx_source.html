45WorkflowSpec defineDataProcessing(ConfigContext const& cc)
46{
47 bool useMC = !cc.options().get<bool>("disable-mc");
48
49 auto name = fmt::format("mch-digit-reader-{}-{}",
50 cc.options().get<std::string>("mch-output-digits-data-description"),
51 cc.options().get<std::string>("mch-output-digitrofs-data-description"));
52
53 WorkflowSpec specs;
54 specs.emplace_back(o2::mch::getDigitReaderSpec(
55 useMC,
56 name,
57 cc.options().get<std::string>("mch-output-digits-data-description"),
58 cc.options().get<std::string>("mch-output-digitrofs-data-description"),
59 cc.options().get<std::string>("mch-output-digitlabels-data-description")));
60
61 o2::raw::HBFUtilsInitializer hbfIni(cc, specs);
62
63 return specs;
