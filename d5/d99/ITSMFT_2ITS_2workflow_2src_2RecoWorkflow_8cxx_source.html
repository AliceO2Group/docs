53  }
54  if (!upstreamClusters) {
55  specs.emplace_back(o2::its::getClustererSpec(useMC));
56  }
57  if (!disableRootOutput) {
58  specs.emplace_back(o2::its::getClusterWriterSpec(useMC));
59  }
60  if (!trmode.empty()) {
61  if (useCAtracker) {
62  if (useGPUWF) {
63  o2::gpu::GPURecoWorkflowSpec::Config cfg;
64  cfg.runITSTracking = true;
65  cfg.itsTriggerType = useTrig;
66  cfg.itsOverrBeamEst = overrideBeamPosition;
67  cfg.itsTrackingMode = trmode == "sync" ? 0 : (trmode == "async" ? 1 : 2);
68 
69  Inputs ggInputs;
70  auto ggRequest = std::make_shared<o2::base::GRPGeomRequest>(false, true, false, true, true,
71  useGeom ? o2::base::GRPGeomRequest::Aligned : o2::base::GRPGeomRequest::None,
72  ggInputs, true);
73  if (!useGeom) {
74  ggRequest->addInput({"itsTGeo", "ITS", "GEOMTGEO", 0, Lifetime::Condition, framework::ccdbParamSpec("ITS/Config/Geometry")}, ggInputs);
75  }
76 
77  auto task = std::make_shared<o2::gpu::GPURecoWorkflowSpec>(&gPolicyData, cfg, std::vector<int>(), 0, ggRequest);
78  gTask = task;
79  Inputs taskInputs = task->inputs();
80  Options taskOptions = task->options();
81  std::move(ggInputs.begin(), ggInputs.end(), std::back_inserter(taskInputs));
82 
83  specs.emplace_back(DataProcessorSpec{
84  "its-tracker",
85  taskInputs,
86  task->outputs(),
87  AlgorithmSpec{adoptTask<o2::gpu::GPURecoWorkflowSpec>(task)},
88  taskOptions});
89  } else {
90  specs.emplace_back(o2::its::getTrackerSpec(useMC, useGeom, useTrig, trmode, overrideBeamPosition, dtype));
91  }
92  } else {
93  specs.emplace_back(o2::its::getCookedTrackerSpec(useMC, useGeom, useTrig, trmode));
94  }
95  if (!disableRootOutput) {
96  specs.emplace_back(o2::its::getTrackWriterSpec(useMC));
97  specs.emplace_back(o2::globaltracking::getIRFrameWriterSpec("irfr:ITS/IRFRAMES/0", "o2_its_irframe.root", "irframe-writer-its"));
98  }
99  }
100  return specs;
101 }
