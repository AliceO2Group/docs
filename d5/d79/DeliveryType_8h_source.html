184template <>
185inline DeliveryType read(const std::string& str)
186{
187 if (str == "Raw/Int") {
188 return RAW_INT;
189 } else if (str == "Raw/Uint") {
190 return RAW_UINT;
191 } else if (str == "Raw/Float") {
192 return RAW_FLOAT;
193 } else if (str == "Raw/Double") {
194 return RAW_DOUBLE;
195 } else if (str == "Raw/Bool") {
196 return RAW_BOOL;
197 } else if (str == "Raw/Char") {
198 return RAW_CHAR;
199 } else if (str == "Raw/String") {
200 return RAW_STRING;
201 } else if (str == "Raw/Binary") {
202 return RAW_BINARY;
203 } else if (str == "Raw/Time") {
204 return RAW_TIME;
205 } else if (str == "DPVAL/Int") {
206 return DPVAL_INT;
207 } else if (str == "DPVAL/Uint") {
208 return DPVAL_UINT;
209 } else if (str == "DPVAL/Float") {
210 return DPVAL_FLOAT;
211 } else if (str == "DPVAL/Double") {
212 return DPVAL_DOUBLE;
213 } else if (str == "DPVAL/Bool") {
214 return DPVAL_BOOL;
215 } else if (str == "DPVAL/Char") {
216 return DPVAL_CHAR;
217 } else if (str == "DPVAL/String") {
218 return DPVAL_STRING;
219 } else if (str == "DPVAL/Binary") {
220 return DPVAL_BINARY;
221 } else if (str == "DPVAL/Time") {
222 return DPVAL_TIME;
223 } else if (str == "Void") {
224 return VOID;
225 } else {
226 throw std::domain_error("\"" + str +
227 "\" doesn't represent a DeliveryType.");
228 }
229}
