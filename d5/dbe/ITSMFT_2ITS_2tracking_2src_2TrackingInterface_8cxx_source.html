32 void ITSTrackingInterface::initialise()
33 {
34  mRunVertexer = true;
35  mCosmicsProcessing = false;
36  std::vector<VertexingParameters> vertParams;
37  std::vector<TrackingParameters> trackParams;
38  if (mMode == TrackingMode::Unset) {
39  mMode = (TrackingMode)(o2::its::TrackerParamConfig::Instance().trackingMode);
40  LOGP(info, "Tracking mode not set, trying to fetch it from configurable params to: {}", asString(mMode));
41  }
42  if (mMode == TrackingMode::Async) {
43  trackParams.resize(o2::its::TrackerParamConfig::Instance().doUPCIteration ? 4 : 3);
44  vertParams.resize(2); // The number of actual iterations will be set as a configKeyVal to allow for pp/PbPb choice
45  trackParams[1].TrackletMinPt = 0.2f;
46  trackParams[1].CellDeltaTanLambdaSigma *= 2.;
47  trackParams[2].TrackletMinPt = 0.1f;
48  trackParams[2].CellDeltaTanLambdaSigma *= 4.;
49  trackParams[2].MinTrackLength = 4;
50  if (o2::its::TrackerParamConfig::Instance().doUPCIteration) {
51  trackParams[3].TrackletMinPt = 0.1f;
52  trackParams[3].CellDeltaTanLambdaSigma *= 4.;
53  trackParams[3].MinTrackLength = 4;
54  trackParams[3].DeltaROF = 0; // UPC specific setting
55  }
56  for (auto& param : trackParams) {
57  param.ZBins = 64;
58  param.PhiBins = 32;
59  param.CellsPerClusterLimit = 1.e3f;
60  param.TrackletsPerClusterLimit = 1.e3f;
61  }
62  LOGP(info, "Initializing tracker in async. phase reconstruction with {} passes for tracking and {}/{} for vertexing", trackParams.size(), o2::its::VertexerParamConfig::Instance().nIterations, vertParams.size());
63  vertParams[1].phiCut = 0.015f;
64  vertParams[1].tanLambdaCut = 0.015f;
65  vertParams[1].vertPerRofThreshold = 0;
66  vertParams[1].deltaRof = 0;
67  } else if (mMode == TrackingMode::Sync) {
68  trackParams.resize(1);
69  trackParams[0].ZBins = 64;
70  trackParams[0].PhiBins = 32;
71  trackParams[0].MinTrackLength = 4;
72  LOGP(info, "Initializing tracker in sync. phase reconstruction with {} passes", trackParams.size());
73  vertParams.resize(1);
74  } else if (mMode == TrackingMode::Cosmics) {
75  mCosmicsProcessing = true;
76  mRunVertexer = false;
77  trackParams.resize(1);
78  trackParams[0].MinTrackLength = 4;
79  trackParams[0].CellDeltaTanLambdaSigma *= 10;
80  trackParams[0].PhiBins = 4;
81  trackParams[0].ZBins = 16;
82  trackParams[0].PVres = 1.e5f;
83  trackParams[0].MaxChi2ClusterAttachment = 60.;
84  trackParams[0].MaxChi2NDF = 40.;
85  trackParams[0].TrackletsPerClusterLimit = 100.;
86  trackParams[0].CellsPerClusterLimit = 100.;
87  LOGP(info, "Initializing tracker in reconstruction for cosmics with {} passes", trackParams.size());
88 
89  } else {
90  throw std::runtime_error(fmt::format("Unsupported ITS tracking mode {:s} ", asString(mMode)));
91  }
92 
93  for (auto& params : trackParams) {
94  params.CorrType = o2::base::PropagatorImpl<float>::MatCorrType::USEMatCorrLUT;
95  }
96  mTracker->setParameters(trackParams);
97  mVertexer->setParameters(vertParams);
98 }

101 void ITSTrackingInterface::run(framework::ProcessingContext& pc)
102 {
103  auto compClusters = pc.inputs().get<gsl::span<o2::itsmft::CompClusterExt>>("compClusters");
104  gsl::span<const unsigned char> patterns = pc.inputs().get<gsl::span<unsigned char>>("patterns");
105  gsl::span<const o2::itsmft::PhysTrigger> physTriggers;
106  std::vector<o2::itsmft::PhysTrigger> fromTRD;
107  if (mUseTriggers == 2) { // use TRD triggers
108  o2::InteractionRecord ir{0, pc.services().get<o2::framework::TimingInfo>().firstTForbit};
109  auto trdTriggers = pc.inputs().get<gsl::span<o2::trd::TriggerRecord>>("phystrig");
110  for (const auto& trig : trdTriggers) {
111  if (trig.getBCData() >= ir && trig.getNumberOfTracklets()) {
112  ir = trig.getBCData();
113  fromTRD.emplace_back(o2::itsmft::PhysTrigger{ir, 0});
114  }
115  }
116  physTriggers = gsl::span<const o2::itsmft::PhysTrigger>(fromTRD.data(), fromTRD.size());
117  } else if (mUseTriggers == 1) { // use Phys triggers from ITS stream
118  physTriggers = pc.inputs().get<gsl::span<o2::itsmft::PhysTrigger>>("phystrig");
119  }
120 
121  auto rofsinput = pc.inputs().get<gsl::span<o2::itsmft::ROFRecord>>("ROframes");
122  auto& trackROFvec = pc.outputs().make<std::vector<o2::itsmft::ROFRecord>>(Output{"ITS", "ITSTrackROF", 0}, rofsinput.begin(), rofsinput.end());
123  auto& irFrames = pc.outputs().make<std::vector<o2::dataformats::IRFrame>>(Output{"ITS", "IRFRAMES", 0});
124  const auto& alpParams = o2::itsmft::DPLAlpideParam<o2::detectors::DetID::ITS>::Instance(); // RS: this should come from CCDB
125 
126  irFrames.reserve(trackROFvec.size());
127  int nBCPerTF = alpParams.roFrameLengthInBC;
128 
129  LOGP(info, "ITSTracker pulled {} clusters, {} RO frames", compClusters.size(), trackROFvec.size());
130  const dataformats::MCTruthContainer<MCCompLabel>* labels = nullptr;
131  gsl::span<itsmft::MC2ROFRecord const> mc2rofs;
132  if (mIsMC) {
133  labels = pc.inputs().get<const dataformats::MCTruthContainer<MCCompLabel>*>("itsmclabels").release();
134  // get the array as read-only span, a snapshot is sent forward
135  pc.outputs().snapshot(Output{"ITS", "ITSTrackMC2ROF", 0}, pc.inputs().get<gsl::span<itsmft::MC2ROFRecord>>("ITSMC2ROframes"));
136  LOG(info) << labels->getIndexedSize() << " MC label objects , in " << mc2rofs.size() << " MC events";
137  }
138 
139  auto& allClusIdx = pc.outputs().make<std::vector<int>>(Output{"ITS", "TRACKCLSID", 0});
140  auto& allTracks = pc.outputs().make<std::vector<o2::its::TrackITS>>(Output{"ITS", "TRACKS", 0});
141  auto& vertROFvec = pc.outputs().make<std::vector<o2::itsmft::ROFRecord>>(Output{"ITS", "VERTICESROF", 0});
142  auto& vertices = pc.outputs().make<std::vector<Vertex>>(Output{"ITS", "VERTICES", 0});
143 
144  // MC
145  static pmr::vector<o2::MCCompLabel> dummyMCLabTracks, dummyMCLabVerts;
146  static pmr::vector<float> dummyMCPurVerts;
147  auto& allTrackLabels = mIsMC ? pc.outputs().make<std::vector<o2::MCCompLabel>>(Output{"ITS", "TRACKSMCTR", 0}) : dummyMCLabTracks;
148  auto& allVerticesLabels = mIsMC ? pc.outputs().make<std::vector<o2::MCCompLabel>>(Output{"ITS", "VERTICESMCTR", 0}) : dummyMCLabVerts;
149  auto& allVerticesPurities = mIsMC ? pc.outputs().make<std::vector<float>>(Output{"ITS", "VERTICESMCPUR", 0}) : dummyMCPurVerts;
150 
151  std::uint32_t roFrame = 0;
152 
153  bool continuous = o2::base::GRPGeomHelper::instance().getGRPECS()->isDetContinuousReadOut(o2::detectors::DetID::ITS);
154  LOG(info) << "ITSTracker RO: continuous=" << continuous;
155 
156  if (mOverrideBeamEstimation) {
157  mTimeFrame->setBeamPosition(mMeanVertex->getX(),
158  mMeanVertex->getY(),
159  mMeanVertex->getSigmaY2(),
160  mTracker->getParameters()[0].LayerResolution[0],
161  mTracker->getParameters()[0].SystErrorY2[0]);
162  }
163 
164  mTracker->setBz(o2::base::Propagator::Instance()->getNominalBz());
165 
166  gsl::span<const unsigned char>::iterator pattIt = patterns.begin();
167 
168  gsl::span<itsmft::ROFRecord> trackROFspan(trackROFvec);
169  loadROF(trackROFspan, compClusters, pattIt, labels);
170  pattIt = patterns.begin();
171  std::vector<int> savedROF;
172  auto logger = [&](std::string s) { LOG(info) << s; };
173  auto fatalLogger = [&](std::string s) { LOG(fatal) << s; };
174  auto errorLogger = [&](std::string s) { LOG(error) << s; };
175 
176  FastMultEst multEst; // mult estimator
177  std::vector<uint8_t> processingMask, processUPCMask;
178  int cutVertexMult{0}, cutUPCVertex{0}, cutRandomMult = int(trackROFvec.size()) - multEst.selectROFs(trackROFvec, compClusters, physTriggers, processingMask);
179  processUPCMask.resize(processingMask.size(), false);
180  mTimeFrame->setMultiplicityCutMask(processingMask);
181  float vertexerElapsedTime{0.f};
182  if (mRunVertexer) {
183  vertROFvec.reserve(trackROFvec.size());
184  // Run seeding vertexer
185  if constexpr (isGPU) {
186  vertexerElapsedTime = mVertexer->clustersToVerticesHybrid(logger);
187  } else {
188  vertexerElapsedTime = mVertexer->clustersToVertices(logger);
189  }
190  } else { // cosmics
191  mTimeFrame->resetRofPV();
192  }
193  const auto& multEstConf = FastMultEstConfig::Instance(); // parameters for mult estimation and cuts
194  gsl::span<const std::pair<MCCompLabel, float>> vMCRecInfo;
195  for (auto iRof{0}; iRof < trackROFspan.size(); ++iRof) {
196  std::vector<Vertex> vtxVecLoc;
197  auto& vtxROF = vertROFvec.emplace_back(trackROFspan[iRof]);
198  vtxROF.setFirstEntry(vertices.size());
199  if (mRunVertexer) {
200  auto vtxSpan = mTimeFrame->getPrimaryVertices(iRof);
201  if (mIsMC) {
202  vMCRecInfo = mTimeFrame->getPrimaryVerticesMCRecInfo(iRof);
203  }
204  if (o2::its::TrackerParamConfig::Instance().doUPCIteration) {
205  if (vtxSpan.size()) {
206  if (vtxSpan[0].isFlagSet(Vertex::UPCMode) == 1) { // at least one vertex in this ROF and it is from second vertex iteration
207  LOGP(debug, "ROF {} rejected as vertices are from the UPC iteration", iRof);
208  processUPCMask[iRof] = true;
209  cutUPCVertex++;
210  vtxROF.setFlag(o2::itsmft::ROFRecord::VtxUPCMode);
211  } else { // in all cases except if as standard mode vertex was found, the ROF was processed with UPC settings
212  vtxROF.setFlag(o2::itsmft::ROFRecord::VtxStdMode);
213  }
214  } else {
215  vtxROF.setFlag(o2::itsmft::ROFRecord::VtxUPCMode);
216  }
217  } else {
218  vtxROF.setFlag(o2::itsmft::ROFRecord::VtxStdMode);
219  }
220  vtxROF.setNEntries(vtxSpan.size());
221  bool selROF = vtxSpan.size() == 0;
222  for (auto iV{0}; iV < vtxSpan.size(); ++iV) {
223  auto& v = vtxSpan[iV];
224  if (multEstConf.isVtxMultCutRequested() && !multEstConf.isPassingVtxMultCut(v.getNContributors())) {
225  continue; // skip vertex of unwanted multiplicity
226  }
227  selROF = true;
228  vertices.push_back(v);
229  if (mIsMC) {
230  allVerticesLabels.push_back(vMCRecInfo[iV].first);
231  allVerticesPurities.push_back(vMCRecInfo[iV].second);
232  }
233  }
234  if (processingMask[iRof] && !selROF) { // passed selection in clusters and not in vertex multiplicity
235  LOGP(info, "ROF {} rejected by the vertex multiplicity selection [{},{}]", iRof, multEstConf.cutMultVtxLow, multEstConf.cutMultVtxHigh);
236  processingMask[iRof] = selROF;
237  cutVertexMult++;
238  }
239  } else { // cosmics
240  vtxVecLoc.emplace_back(Vertex());
241  vtxVecLoc.back().setNContributors(1);
242  vtxROF.setNEntries(vtxVecLoc.size());
243  for (auto& v : vtxVecLoc) {
244  vertices.push_back(v);
245  }
246  mTimeFrame->addPrimaryVertices(vtxVecLoc, iRof, 0);
247  }
248  }
249  if (mRunVertexer) {
250  LOG(info) << fmt::format(" - Vertex seeding total elapsed time: {} ms for {} ({} + {}) vertices found in {}/{} ROFs",
251  vertexerElapsedTime,
252  mTimeFrame->getPrimaryVerticesNum(),
253  mTimeFrame->getTotVertIteration()[0],
254  o2::its::VertexerParamConfig::Instance().nIterations > 1 ? mTimeFrame->getTotVertIteration()[1] : 0,
255  trackROFspan.size() - mTimeFrame->getNoVertexROF(),
256  trackROFspan.size());
257  LOG(info) << fmt::format("FastMultEst: rejected {}/{} ROFs: random/mult.sel:{} (seed {}), vtx.sel:{}", cutRandomMult + cutVertexMult, trackROFspan.size(), cutRandomMult, multEst.lastRandomSeed, cutVertexMult);
258  }
259  if (mOverrideBeamEstimation) {
260  LOG(info) << fmt::format(" - Beam position set to: {}, {} from meanvertex object", mTimeFrame->getBeamX(), mTimeFrame->getBeamY());
261  } else {
262  LOG(info) << fmt::format(" - Beam position computed for the TF: {}, {}", mTimeFrame->getBeamX(), mTimeFrame->getBeamY());
263  }
264  if (mCosmicsProcessing && compClusters.size() > 1500 * trackROFspan.size()) {
265  LOG(error) << "Cosmics processing was requested with an average detector occupancy exceeding 1.e-7, skipping TF processing.";
266  } else {
267 
268  mTimeFrame->setMultiplicityCutMask(processingMask);
269  mTimeFrame->setROFMask(processUPCMask);
270  // Run CA tracker
271  if constexpr (isGPU) {
272  if (mMode == o2::its::TrackingMode::Async) {
273  mTracker->clustersToTracksHybrid(logger, fatalLogger);
274  } else {
275  mTracker->clustersToTracksHybrid(logger, errorLogger);
276  }
277  } else {
278  if (mMode == o2::its::TrackingMode::Async) {
279  mTracker->clustersToTracks(logger, fatalLogger);
280  } else {
281  mTracker->clustersToTracks(logger, errorLogger);
282  }
283  }
284  size_t totTracks{mTimeFrame->getNumberOfTracks()}, totClusIDs{mTimeFrame->getNumberOfUsedClusters()};
285  allTracks.reserve(totTracks);
286  allClusIdx.reserve(totClusIDs);
287 
288  if (mTimeFrame->hasBogusClusters()) {
289  LOG(warning) << fmt::format(" - The processed timeframe had {} clusters with wild z coordinates, check the dictionaries", mTimeFrame->hasBogusClusters());
290  }
291 
292  for (unsigned int iROF{0}; iROF < trackROFvec.size(); ++iROF) {
293  auto& tracksROF{trackROFvec[iROF]};
294  auto& vtxROF = vertROFvec[iROF];
295  auto& tracks = mTimeFrame->getTracks(iROF);
296  auto number{tracks.size()};
297  auto first{allTracks.size()};
298  int offset = -tracksROF.getFirstEntry(); // cluster entry!!!
299  tracksROF.setFirstEntry(first);
300  tracksROF.setNEntries(number);
301  tracksROF.setFlags(vtxROF.getFlags()); // copies 0xffffffff if cosmics
302  if (processingMask[iROF]) {
303  irFrames.emplace_back(tracksROF.getBCData(), tracksROF.getBCData() + nBCPerTF - 1).info = tracks.size();
304  }
305  allTrackLabels.reserve(mTimeFrame->getTracksLabel(iROF).size()); // should be 0 if not MC
306  std::copy(mTimeFrame->getTracksLabel(iROF).begin(), mTimeFrame->getTracksLabel(iROF).end(), std::back_inserter(allTrackLabels));
307  // Some conversions that needs to be moved in the tracker internals
308  for (unsigned int iTrk{0}; iTrk < tracks.size(); ++iTrk) {
309  auto& trc{tracks[iTrk]};
310  trc.setFirstClusterEntry(allClusIdx.size()); // before adding tracks, create final cluster indices
311  int ncl = trc.getNumberOfClusters(), nclf = 0;
312  for (int ic = TrackITSExt::MaxClusters; ic--;) { // track internally keeps in->out cluster indices, but we want to store the references as out->in!!!
313  auto clid = trc.getClusterIndex(ic);
314  if (clid >= 0) {
315  trc.setClusterSize(ic, mTimeFrame->getClusterSize(clid));
316  allClusIdx.push_back(clid);
317  nclf++;
318  }
319  }
320  assert(ncl == nclf);
321  allTracks.emplace_back(trc);
322  }
323  }
324  LOGP(info, "ITSTracker pushed {} tracks and {} vertices", allTracks.size(), vertices.size());
325  if (mIsMC) {
326  LOGP(info, "ITSTracker pushed {} track labels", allTrackLabels.size());
327  LOGP(info, "ITSTracker pushed {} vertex labels", allVerticesLabels.size());
328  LOGP(info, "ITSTracker pushed {} vertex purities", allVerticesPurities.size());
329  }
330  }

359 void ITSTrackingInterface::finaliseCCDB(ConcreteDataMatcher& matcher, void* obj)
360 {
361  if (o2::base::GRPGeomHelper::instance().finaliseCCDB(matcher, obj)) {
362  return;
363  }
364  if (matcher == ConcreteDataMatcher("ITS", "CLUSDICT", 0)) {
365  LOG(info) << "cluster dictionary updated";
366  setClusterDictionary((const o2::itsmft::TopologyDictionary*)obj);
367  return;
368  }
369  // Note: strictly speaking, for Configurable params we don't need finaliseCCDB check, the singletons are updated at the CCDB fetcher level
370  if (matcher == ConcreteDataMatcher("ITS", "ALPIDEPARAM", 0)) {
371  LOG(info) << "Alpide param updated";
372  const auto& par = o2::itsmft::DPLAlpideParam<o2::detectors::DetID::ITS>::Instance();
373  par.printKeyValues();
374  return;
375  }
376  if (matcher == ConcreteDataMatcher("GLO", "MEANVERTEX", 0)) {
377  LOGP(info, "Mean vertex acquired");
378  setMeanVertex((const o2::dataformats::MeanVertexObject*)obj);
379  return;
380  }
381  if (matcher == ConcreteDataMatcher("ITS", "GEOMTGEO", 0)) {
382  LOG(info) << "ITS GeometryTGeo loaded from ccdb";
383  o2::its::GeometryTGeo::adopt((o2::its::GeometryTGeo*)obj);
384  return;
385  }
