33void ITSTrackingInterface::initialise()
34{
35 mRunVertexer = true;
36 mCosmicsProcessing = false;
37 std::vector<VertexingParameters> vertParams;
38 std::vector<TrackingParameters> trackParams;
39 const auto& trackConf = o2::its::TrackerParamConfig::Instance();
40 float bFactor = std::abs(o2::base::Propagator::Instance()->getNominalBz()) / 5.0066791;
41 float bFactorTracklets = bFactor < 0.01 ? 1. : bFactor; // for tracklets only
42 if (mMode == TrackingMode::Unset) {
43 mMode = (TrackingMode)(trackConf.trackingMode);
44 LOGP(info, "Tracking mode not set, trying to fetch it from configurable params to: {}", asString(mMode));
45 }
46 if (mMode == TrackingMode::Async) {
47 trackParams.resize(trackConf.doUPCIteration ? 4 : 3);
48 vertParams.resize(2); // The number of actual iterations will be set as a configKeyVal to allow for pp/PbPb choice
49 trackParams[1].TrackletMinPt = 0.2f;
50 trackParams[1].CellDeltaTanLambdaSigma *= 2.;
51 trackParams[2].TrackletMinPt = 0.1f;
52 trackParams[2].CellDeltaTanLambdaSigma *= 4.;
53
54 trackParams[0].MinPt[0] = 1.f / 12; // 7cl
55
56 trackParams[1].MinPt[0] = 1.f / 12; // 7cl
57
58 trackParams[2].MinTrackLength = 4;
59 trackParams[2].MinPt[0] = 1.f / 12; // 7cl
60 trackParams[2].MinPt[1] = 1.f / 5; // 6cl
61 trackParams[2].MinPt[2] = 1.f / 1; // 5cl
62 trackParams[2].MinPt[3] = 1.f / 6; // 4cl
63
64 trackParams[2].StartLayerMask = (1 << 6) + (1 << 3);
65 if (o2::its::TrackerParamConfig::Instance().doUPCIteration) {
66 trackParams[3].MinTrackLength = 4;
67 trackParams[3].TrackletMinPt = 0.1f;
68 trackParams[3].CellDeltaTanLambdaSigma *= 4.;
69 trackParams[3].DeltaROF = 0; // UPC specific setting
70 }
71 for (size_t ip = 0; ip < trackParams.size(); ip++) {
72 auto& param = trackParams[ip];
73 param.ZBins = 64;
74 param.PhiBins = 32;
75 param.CellsPerClusterLimit = 1.e3f;
76 param.TrackletsPerClusterLimit = 1.e3f;
77 // check if something was overridden via configurable params
78 if (ip < trackConf.MaxIter) {
79 if (trackConf.startLayerMask[ip] > 0) {
80 trackParams[2].StartLayerMask = trackConf.startLayerMask[ip];
81 }
82 if (trackConf.minTrackLgtIter[ip] > 0) {
83 param.MinTrackLength = trackConf.minTrackLgtIter[ip];
84 }
85 for (int ilg = trackConf.MaxTrackLength; ilg >= trackConf.MinTrackLength; ilg--) {
86 int lslot0 = (trackConf.MaxTrackLength - ilg), lslot = lslot0 + ip * (trackConf.MaxTrackLength - trackConf.MinTrackLength + 1);
87 if (trackConf.minPtIterLgt[lslot] > 0.) {
88 param.MinPt[lslot0] = trackConf.minPtIterLgt[lslot];
89 }
90 }
91 }
92 }
93 LOGP(info, "Initializing tracker in async. phase reconstruction with {} passes for tracking and {}/{} for vertexing", trackParams.size(), o2::its::VertexerParamConfig::Instance().nIterations, vertParams.size());
94 vertParams[1].phiCut = 0.015f;
95 vertParams[1].tanLambdaCut = 0.015f;
96 vertParams[1].vertPerRofThreshold = 0;
97 vertParams[1].deltaRof = 0;
98 } else if (mMode == TrackingMode::Sync) {
99 trackParams.resize(1);
100 trackParams[0].ZBins = 64;
101 trackParams[0].PhiBins = 32;
102 trackParams[0].MinTrackLength = 4;
103 LOGP(info, "Initializing tracker in sync. phase reconstruction with {} passes", trackParams.size());
104 vertParams.resize(1);
105 } else if (mMode == TrackingMode::Cosmics) {
106 mCosmicsProcessing = true;
107 mRunVertexer = false;
108 trackParams.resize(1);
109 trackParams[0].MinTrackLength = 4;
110 trackParams[0].CellDeltaTanLambdaSigma *= 10;
111 trackParams[0].PhiBins = 4;
112 trackParams[0].ZBins = 16;
113 trackParams[0].PVres = 1.e5f;
114 trackParams[0].MaxChi2ClusterAttachment = 60.;
115 trackParams[0].MaxChi2NDF = 40.;
116 trackParams[0].TrackletsPerClusterLimit = 100.;
117 trackParams[0].CellsPerClusterLimit = 100.;
118 LOGP(info, "Initializing tracker in reconstruction for cosmics with {} passes", trackParams.size());
119
120 } else {
121 throw std::runtime_error(fmt::format("Unsupported ITS tracking mode {:s} ", asString(mMode)));
122 }
123
124 for (auto& params : trackParams) {
125 params.CorrType = o2::base::PropagatorImpl<float>::MatCorrType::USEMatCorrLUT;
126 }
127 // adjust pT settings to actual mag. field
128 for (size_t ip = 0; ip < trackParams.size(); ip++) {
129 auto& param = trackParams[ip];
130 param.TrackletMinPt *= bFactorTracklets;
131 for (int ilg = trackConf.MaxTrackLength; ilg >= trackConf.MinTrackLength; ilg--) {
132 int lslot = trackConf.MaxTrackLength - ilg;
133 param.MinPt[lslot] *= bFactor;
134 }
135 }
136 mTracker->setParameters(trackParams);
137 mVertexer->setParameters(vertParams);
138}

140void ITSTrackingInterface::run(framework::ProcessingContext& pc)
141{
142 auto compClusters = pc.inputs().get<gsl::span<o2::itsmft::CompClusterExt>>("compClusters");
143 gsl::span<const unsigned char> patterns = pc.inputs().get<gsl::span<unsigned char>>("patterns");
144 gsl::span<const o2::itsmft::PhysTrigger> physTriggers;
145 std::vector<o2::itsmft::PhysTrigger> fromTRD;
146 if (mUseTriggers == 2) { // use TRD triggers
147 o2::InteractionRecord ir{0, pc.services().get<o2::framework::TimingInfo>().firstTForbit};
148 auto trdTriggers = pc.inputs().get<gsl::span<o2::trd::TriggerRecord>>("phystrig");
149 for (const auto& trig : trdTriggers) {
150 if (trig.getBCData() >= ir && trig.getNumberOfTracklets()) {
151 ir = trig.getBCData();
152 fromTRD.emplace_back(o2::itsmft::PhysTrigger{ir, 0});
153 }
154 }
155 physTriggers = gsl::span<const o2::itsmft::PhysTrigger>(fromTRD.data(), fromTRD.size());
156 } else if (mUseTriggers == 1) { // use Phys triggers from ITS stream
157 physTriggers = pc.inputs().get<gsl::span<o2::itsmft::PhysTrigger>>("phystrig");
158 }
159
160 auto rofsinput = pc.inputs().get<gsl::span<o2::itsmft::ROFRecord>>("ROframes");
161 auto& trackROFvec = pc.outputs().make<std::vector<o2::itsmft::ROFRecord>>(Output{"ITS", "ITSTrackROF", 0}, rofsinput.begin(), rofsinput.end());
162 auto& irFrames = pc.outputs().make<std::vector<o2::dataformats::IRFrame>>(Output{"ITS", "IRFRAMES", 0});
163 const auto& alpParams = o2::itsmft::DPLAlpideParam<o2::detectors::DetID::ITS>::Instance(); // RS: this should come from CCDB
164
165 irFrames.reserve(trackROFvec.size());
166 int nBCPerTF = alpParams.roFrameLengthInBC;
167
168 LOGP(info, "ITSTracker pulled {} clusters, {} RO frames", compClusters.size(), trackROFvec.size());
169 const dataformats::MCTruthContainer<MCCompLabel>* labels = nullptr;
170 gsl::span<itsmft::MC2ROFRecord const> mc2rofs;
171 if (mIsMC) {
172 labels = pc.inputs().get<const dataformats::MCTruthContainer<MCCompLabel>*>("itsmclabels").release();
173 // get the array as read-only span, a snapshot is sent forward
174 pc.outputs().snapshot(Output{"ITS", "ITSTrackMC2ROF", 0}, pc.inputs().get<gsl::span<itsmft::MC2ROFRecord>>("ITSMC2ROframes"));
175 LOG(info) << labels->getIndexedSize() << " MC label objects , in " << mc2rofs.size() << " MC events";
176 }
177
178 auto& allClusIdx = pc.outputs().make<std::vector<int>>(Output{"ITS", "TRACKCLSID", 0});
179 auto& allTracks = pc.outputs().make<std::vector<o2::its::TrackITS>>(Output{"ITS", "TRACKS", 0});
180 auto& vertROFvec = pc.outputs().make<std::vector<o2::itsmft::ROFRecord>>(Output{"ITS", "VERTICESROF", 0});
181 auto& vertices = pc.outputs().make<std::vector<Vertex>>(Output{"ITS", "VERTICES", 0});
182
183 // MC
184 static pmr::vector<o2::MCCompLabel> dummyMCLabTracks, dummyMCLabVerts;
185 static pmr::vector<float> dummyMCPurVerts;
186 auto& allTrackLabels = mIsMC ? pc.outputs().make<std::vector<o2::MCCompLabel>>(Output{"ITS", "TRACKSMCTR", 0}) : dummyMCLabTracks;
187 auto& allVerticesLabels = mIsMC ? pc.outputs().make<std::vector<o2::MCCompLabel>>(Output{"ITS", "VERTICESMCTR", 0}) : dummyMCLabVerts;
188 auto& allVerticesPurities = mIsMC ? pc.outputs().make<std::vector<float>>(Output{"ITS", "VERTICESMCPUR", 0}) : dummyMCPurVerts;
189
190 std::uint32_t roFrame = 0;
191
192 bool continuous = o2::base::GRPGeomHelper::instance().getGRPECS()->isDetContinuousReadOut(o2::detectors::DetID::ITS);
193 LOG(info) << "ITSTracker RO: continuous=" << continuous;
194
195 if (mOverrideBeamEstimation) {
196 mTimeFrame->setBeamPosition(mMeanVertex->getX(),
197 mMeanVertex->getY(),
198 mMeanVertex->getSigmaY2(),
199 mTracker->getParameters()[0].LayerResolution[0],
200 mTracker->getParameters()[0].SystErrorY2[0]);
201 }
202
203 mTracker->setBz(o2::base::Propagator::Instance()->getNominalBz());
204
205 gsl::span<const unsigned char>::iterator pattIt = patterns.begin();
206
207 gsl::span<itsmft::ROFRecord> trackROFspan(trackROFvec);
208 loadROF(trackROFspan, compClusters, pattIt, labels);
209 pattIt = patterns.begin();
210 std::vector<int> savedROF;
211 auto logger = [&](const std::string& s) { LOG(info) << s; };
212 auto fatalLogger = [&](const std::string& s) { LOG(fatal) << s; };
213 auto errorLogger = [&](const std::string& s) { LOG(error) << s; };
214
215 FastMultEst multEst; // mult estimator
216 std::vector<uint8_t> processingMask, processUPCMask;
217 int cutVertexMult{0}, cutUPCVertex{0}, cutRandomMult = int(trackROFvec.size()) - multEst.selectROFs(trackROFvec, compClusters, physTriggers, processingMask);
218 processUPCMask.resize(processingMask.size(), false);
219 mTimeFrame->setMultiplicityCutMask(processingMask);
220 float vertexerElapsedTime{0.f};
221 if (mRunVertexer) {
222 vertROFvec.reserve(trackROFvec.size());
223 // Run seeding vertexer
224 vertexerElapsedTime = mVertexer->clustersToVertices(logger);
225 } else { // cosmics
226 mTimeFrame->resetRofPV();
227 }
228 const auto& multEstConf = FastMultEstConfig::Instance(); // parameters for mult estimation and cuts
229 gsl::span<const std::pair<MCCompLabel, float>> vMCRecInfo;
230 for (auto iRof{0}; iRof < trackROFspan.size(); ++iRof) {
231 std::vector<Vertex> vtxVecLoc;
232 auto& vtxROF = vertROFvec.emplace_back(trackROFspan[iRof]);
233 vtxROF.setFirstEntry(vertices.size());
234 if (mRunVertexer) {
235 auto vtxSpan = mTimeFrame->getPrimaryVertices(iRof);
236 if (mIsMC) {
237 vMCRecInfo = mTimeFrame->getPrimaryVerticesMCRecInfo(iRof);
238 }
239 if (o2::its::TrackerParamConfig::Instance().doUPCIteration) {
240 if (vtxSpan.size()) {
241 if (vtxSpan[0].isFlagSet(Vertex::UPCMode) == 1) { // at least one vertex in this ROF and it is from second vertex iteration
242 LOGP(debug, "ROF {} rejected as vertices are from the UPC iteration", iRof);
243 processUPCMask[iRof] = true;
244 cutUPCVertex++;
245 vtxROF.setFlag(o2::itsmft::ROFRecord::VtxUPCMode);
246 } else { // in all cases except if as standard mode vertex was found, the ROF was processed with UPC settings
247 vtxROF.setFlag(o2::itsmft::ROFRecord::VtxStdMode);
248 }
249 } else {
250 vtxROF.setFlag(o2::itsmft::ROFRecord::VtxUPCMode);
251 }
252 } else {
253 vtxROF.setFlag(o2::itsmft::ROFRecord::VtxStdMode);
254 }
255 vtxROF.setNEntries(vtxSpan.size());
256 bool selROF = vtxSpan.size() == 0;
257 for (auto iV{0}; iV < vtxSpan.size(); ++iV) {
258 auto& v = vtxSpan[iV];
259 if (multEstConf.isVtxMultCutRequested() && !multEstConf.isPassingVtxMultCut(v.getNContributors())) {
260 continue; // skip vertex of unwanted multiplicity
261 }
262 selROF = true;
263 vertices.push_back(v);
264 if (mIsMC) {
265 allVerticesLabels.push_back(vMCRecInfo[iV].first);
266 allVerticesPurities.push_back(vMCRecInfo[iV].second);
267 }
268 }
269 if (processingMask[iRof] && !selROF) { // passed selection in clusters and not in vertex multiplicity
270 LOGP(info, "ROF {} rejected by the vertex multiplicity selection [{},{}]", iRof, multEstConf.cutMultVtxLow, multEstConf.cutMultVtxHigh);
271 processingMask[iRof] = selROF;
272 cutVertexMult++;
273 }
274 } else { // cosmics
275 vtxVecLoc.emplace_back(Vertex());
276 vtxVecLoc.back().setNContributors(1);
277 vtxROF.setNEntries(vtxVecLoc.size());
278 for (auto& v : vtxVecLoc) {
279 vertices.push_back(v);
280 }
281 mTimeFrame->addPrimaryVertices(vtxVecLoc, iRof, 0);
282 }
283 }
284 if (mRunVertexer) {
285 LOG(info) << fmt::format(" - Vertex seeding total elapsed time: {} ms for {} ({} + {}) vertices found in {}/{} ROFs",
286 vertexerElapsedTime,
287 mTimeFrame->getPrimaryVerticesNum(),
288 mTimeFrame->getTotVertIteration()[0],
289 o2::its::VertexerParamConfig::Instance().nIterations > 1 ? mTimeFrame->getTotVertIteration()[1] : 0,
290 trackROFspan.size() - mTimeFrame->getNoVertexROF(),
291 trackROFspan.size());
292 LOG(info) << fmt::format("FastMultEst: rejected {}/{} ROFs: random/mult.sel:{} (seed {}), vtx.sel:{}", cutRandomMult + cutVertexMult, trackROFspan.size(), cutRandomMult, multEst.lastRandomSeed, cutVertexMult);
293 }
294 if (mOverrideBeamEstimation) {
295 LOG(info) << fmt::format(" - Beam position set to: {}, {} from meanvertex object", mTimeFrame->getBeamX(), mTimeFrame->getBeamY());
296 } else {
297 LOG(info) << fmt::format(" - Beam position computed for the TF: {}, {}", mTimeFrame->getBeamX(), mTimeFrame->getBeamY());
298 }
299 if (mCosmicsProcessing && compClusters.size() > 1500 * trackROFspan.size()) {
300 LOG(error) << "Cosmics processing was requested with an average detector occupancy exceeding 1.e-7, skipping TF processing.";
301 } else {
302
303 mTimeFrame->setMultiplicityCutMask(processingMask);
304 mTimeFrame->setROFMask(processUPCMask);
305 // Run CA tracker
306 if (mMode == o2::its::TrackingMode::Async && o2::its::TrackerParamConfig::Instance().fataliseUponFailure) {
307 mTracker->clustersToTracks(logger, fatalLogger);
308 } else {
309 mTracker->clustersToTracks(logger, errorLogger);
310 }
311 size_t totTracks{mTimeFrame->getNumberOfTracks()}, totClusIDs{mTimeFrame->getNumberOfUsedClusters()};
312 allTracks.reserve(totTracks);
313 allClusIdx.reserve(totClusIDs);
314
315 if (mTimeFrame->hasBogusClusters()) {
316 LOG(warning) << fmt::format(" - The processed timeframe had {} clusters with wild z coordinates, check the dictionaries", mTimeFrame->hasBogusClusters());
317 }
318
319 for (unsigned int iROF{0}; iROF < trackROFvec.size(); ++iROF) {
320 auto& tracksROF{trackROFvec[iROF]};
321 auto& vtxROF = vertROFvec[iROF];
322 auto& tracks = mTimeFrame->getTracks(iROF);
323 auto number{tracks.size()};
324 auto first{allTracks.size()};
325 int offset = -tracksROF.getFirstEntry(); // cluster entry!!!
326 tracksROF.setFirstEntry(first);
327 tracksROF.setNEntries(number);
328 tracksROF.setFlags(vtxROF.getFlags()); // copies 0xffffffff if cosmics
329 if (processingMask[iROF]) {
330 irFrames.emplace_back(tracksROF.getBCData(), tracksROF.getBCData() + nBCPerTF - 1).info = tracks.size();
331 }
332 allTrackLabels.reserve(mTimeFrame->getTracksLabel(iROF).size()); // should be 0 if not MC
333 std::copy(mTimeFrame->getTracksLabel(iROF).begin(), mTimeFrame->getTracksLabel(iROF).end(), std::back_inserter(allTrackLabels));
334 // Some conversions that needs to be moved in the tracker internals
335 for (unsigned int iTrk{0}; iTrk < tracks.size(); ++iTrk) {
336 auto& trc{tracks[iTrk]};
337 trc.setFirstClusterEntry(allClusIdx.size()); // before adding tracks, create final cluster indices
338 int ncl = trc.getNumberOfClusters(), nclf = 0;
339 for (int ic = TrackITSExt::MaxClusters; ic--;) { // track internally keeps in->out cluster indices, but we want to store the references as out->in!!!
340 auto clid = trc.getClusterIndex(ic);
341 if (clid >= 0) {
342 trc.setClusterSize(ic, mTimeFrame->getClusterSize(clid));
343 allClusIdx.push_back(clid);
344 nclf++;
345 }
346 }
347 assert(ncl == nclf);
348 allTracks.emplace_back(trc);
349 }
350 }
351 LOGP(info, "ITSTracker pushed {} tracks and {} vertices", allTracks.size(), vertices.size());
352 if (mIsMC) {
353 LOGP(info, "ITSTracker pushed {} track labels", allTrackLabels.size());
354 LOGP(info, "ITSTracker pushed {} vertex labels", allVerticesLabels.size());
355 LOGP(info, "ITSTracker pushed {} vertex purities", allVerticesPurities.size());
356 }
357 }

379 const auto& trParams = mTracker->getParameters();
380 for (size_t it = 0; it < trParams.size(); it++) {
381 const auto& par = trParams[it];
382 LOGP(info, "recoIter#{} : {}", it, par.asString());
383 }
384 }

397void ITSTrackingInterface::finaliseCCDB(ConcreteDataMatcher& matcher, void* obj)
398{
399 if (o2::base::GRPGeomHelper::instance().finaliseCCDB(matcher, obj)) {
400 return;
401 }
402 if (matcher == ConcreteDataMatcher("ITS", "CLUSDICT", 0)) {
403 LOG(info) << "cluster dictionary updated";
404 setClusterDictionary((const o2::itsmft::TopologyDictionary*)obj);
405 return;
406 }
407 // Note: strictly speaking, for Configurable params we don't need finaliseCCDB check, the singletons are updated at the CCDB fetcher level
408 if (matcher == ConcreteDataMatcher("ITS", "ALPIDEPARAM", 0)) {
409 LOG(info) << "Alpide param updated";
410 const auto& par = o2::itsmft::DPLAlpideParam<o2::detectors::DetID::ITS>::Instance();
411 par.printKeyValues();
412 return;
413 }
414 if (matcher == ConcreteDataMatcher("GLO", "MEANVERTEX", 0)) {
415 LOGP(info, "Mean vertex acquired");
416 setMeanVertex((const o2::dataformats::MeanVertexObject*)obj);
417 return;
418 }
419 if (matcher == ConcreteDataMatcher("ITS", "GEOMTGEO", 0)) {
420 LOG(info) << "ITS GeometryTGeo loaded from ccdb";
421 o2::its::GeometryTGeo::adopt((o2::its::GeometryTGeo*)obj);
422 return;
423 }
