107  static arrow::Status appendToList(std::unique_ptr<arrow::FixedSizeListBuilder>& builder, T* data, int size = 1)
108  {
109  using ArrowType = typename detail::ConversionTraits<std::decay_t<T>>::ArrowType;
110  using BuilderType = typename arrow::TypeTraits<ArrowType>::BuilderType;
111  size_t numElements = static_cast<const arrow::FixedSizeListType*>(builder->type().get())->list_size();
112 
113  auto status = builder->AppendValues(size);
114  auto ValueBuilder = static_cast<BuilderType*>(builder->value_builder());
115  status &= ValueBuilder->AppendValues(data, numElements * size, nullptr);
116 
117  return status;
118  }

203  template <typename HolderType, typename PTR>
204  static arrow::Status bulkAppendChunked(HolderType& holder, BulkInfo<PTR> info)
205  {
206  // Appending nullptr is a no-op.
207  if (info.ptr == nullptr) {
208  return arrow::Status::OK();
209  }
210  if constexpr (std::is_same_v<decltype(holder.builder), std::unique_ptr<arrow::FixedSizeListBuilder>>) {
211  if (appendToList<std::remove_pointer_t<decltype(info.ptr)>>(holder.builder, info.ptr, info.size).ok() == false) {
212  throw runtime_error("Unable to append to column");
213  } else {
214  return arrow::Status::OK();
215  }
216  } else {
217  if (holder.builder->AppendValues(info.ptr, info.size, nullptr).ok() == false) {
218  throw runtime_error("Unable to append to column");
219  } else {
220  return arrow::Status::OK();
221  }
222  }
223  }

439 template <typename T>
440 struct CachedInsertion {
441  static constexpr int CHUNK_SIZE = 256;
442 
443  template <typename BUILDER>
444  arrow::Status append(BUILDER& builder, T value)
445  {
446  cache[pos % CHUNK_SIZE] = value;
447  ++pos;
448  if (pos % CHUNK_SIZE == 0) {
449  return builder->AppendValues(cache, CHUNK_SIZE, nullptr);
450  }
451  return arrow::Status::OK();
452  }
453 
454  template <typename BUILDER>
455  arrow::Status flush(BUILDER& builder)
456  {
457  if (pos % CHUNK_SIZE != 0) {
458  return builder->AppendValues(cache, pos % CHUNK_SIZE, nullptr);
459  }
460  return arrow::Status::OK();
461  }
462  T cache[CHUNK_SIZE];
463  int pos = 0;
464 };

580 template <class T>
581 auto constexpr to_tuple(T&& object) noexcept
582 {
583  using type = std::decay_t<T>;
584  if constexpr (is_braces_constructible<type, any_type, any_type, any_type, any_type>{}) {
585  auto&& [p0, p1, p2, p3] = object;
586  return std::make_tuple(p0, p1, p2, p3);
587  } else if constexpr (is_braces_constructible<type, any_type, any_type, any_type>{}) {
588  auto&& [p0, p1, p2] = object;
589  return std::make_tuple(p0, p1, p2);
590  } else if constexpr (is_braces_constructible<type, any_type, any_type>{}) {
591  auto&& [p0, p1] = object;
592  return std::make_tuple(p0, p1);
593  } else if constexpr (is_braces_constructible<type, any_type>{}) {
594  auto&& [p0] = object;
595  return std::make_tuple(p0);
596  } else {
597  return std::make_tuple();
