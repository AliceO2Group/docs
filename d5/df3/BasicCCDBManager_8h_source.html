222 template <typename T>
223 T* CCDBManagerInstance::getForTimeStamp(std::string const& path, long timestamp)
224 {
225  T* ptr = nullptr;
226  mQueries++;
227  auto start = std::chrono::system_clock::now();
228  if (!isCachingEnabled()) {
229  ptr = mCCDBAccessor.retrieveFromTFileAny<T>(path, mMetaData, timestamp, nullptr, "",
230  mCreatedNotAfter ? std::to_string(mCreatedNotAfter) : "",
231  mCreatedNotBefore ? std::to_string(mCreatedNotBefore) : "");
232  if (!ptr) {
233  if (mFatalWhenNull) {
234  reportFatal(std::string("Got nullptr from CCDB for path ") + path + std::string(" and timestamp ") + std::to_string(timestamp));
235  }
236  mFailures++;
237  } else {
238  mFetches++;
239  auto sh = mHeaders.find("fileSize");
240  if (sh != mHeaders.end()) {
241  size_t s = atol(sh->second.c_str());
242  mFetchedSize += s;
243  }
244  }
245  } else {
246  auto& cached = mCache[path];
247  cached.queries++;
248  if ((!isOnline() && cached.isCacheValid(timestamp)) || (mCheckObjValidityEnabled && cached.isValid(timestamp))) {
249  return reinterpret_cast<T*>(cached.noCleanupPtr ? cached.noCleanupPtr : cached.objPtr.get());
250  }
251  ptr = mCCDBAccessor.retrieveFromTFileAny<T>(path, mMetaData, timestamp, &mHeaders, cached.uuid,
252  mCreatedNotAfter ? std::to_string(mCreatedNotAfter) : "",
253  mCreatedNotBefore ? std::to_string(mCreatedNotBefore) : "");
254  if (ptr) { // new object was shipped, old one (if any) is not valid anymore
255  cached.fetches++;
256  mFetches++;
257  if constexpr (std::is_same<TGeoManager, T>::value || std::is_base_of<o2::conf::ConfigurableParam, T>::value) {
258  // some special objects cannot be cached to shared_ptr since root may delete their raw global pointer
259  cached.noCleanupPtr = ptr;
260  } else {
261  cached.objPtr.reset(ptr);
262  }
263  cached.uuid = mHeaders["ETag"];
264 
265  try {
266  if (mHeaders.find("Valid-From") != mHeaders.end()) {
267  cached.startvalidity = std::stol(mHeaders["Valid-From"]);
268  } else {
269  // if meta-information missing assume infinit validity
270  // (should happen only for locally created objects)
271  cached.startvalidity = 0;
272  }
273  if (mHeaders.find("Valid-Until") != mHeaders.end()) {
274  cached.endvalidity = std::stol(mHeaders["Valid-Until"]);
275  } else {
276  cached.endvalidity = std::numeric_limits<long>::max();
277  }
278  cached.cacheValidFrom = timestamp;
279  } catch (std::exception const& e) {
280  reportFatal("Failed to read validity from CCDB response (Valid-From : " + mHeaders["Valid-From"] + std::string(" Valid-Until: ") + mHeaders["Valid-Until"] + std::string(")"));
281  }
282  auto sh = mHeaders.find("fileSize");
283  if (sh != mHeaders.end()) {
284  size_t s = atol(sh->second.c_str());
285  mFetchedSize += s;
286  cached.minSize = std::min(s, cached.minSize);
287  cached.maxSize = std::max(s, cached.minSize);
288  }
289  } else if (mHeaders.count("Error")) { // in case of errors the pointer is 0 and headers["Error"] should be set
290  cached.failures++;
291  cached.clear(); // in case of any error clear cache for this object
292  }
293  // the old object is valid, fetch cache end of validity
294  ptr = reinterpret_cast<T*>(cached.noCleanupPtr ? cached.noCleanupPtr : cached.objPtr.get());
295  if (mHeaders.find("Cache-Valid-Until") != mHeaders.end()) {
296  cached.cacheValidUntil = std::stol(mHeaders["Cache-Valid-Until"]);
297  } else {
298  cached.cacheValidUntil = -1;
299  }
300  mHeaders.clear();
301  mMetaData.clear();
302  if (!ptr) {
303  if (mFatalWhenNull) {
304  reportFatal(std::string("Got nullptr from CCDB for path ") + path + std::string(" and timestamp ") + std::to_string(timestamp));
305  }
306  mFailures++;
307  }
308  }
309  auto end = std::chrono::system_clock::now();
310  mTimerMS += std::chrono::duration_cast<std::chrono::milliseconds>(end - start).count();
311  return ptr;
312 }
