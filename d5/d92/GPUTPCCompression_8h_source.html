98template <class T>
99GPUdi() void GPUTPCCompression::truncateSignificantBits(T& v, uint32_t nBits, uint32_t max)
100{
101 if (nBits == 0) {
102 return;
103 }
104
105 uint32_t val = v;
106 uint32_t ldz = sizeof(uint32_t) * 8 - CAMath::Clz(val);
107 if (val && ldz > nBits) {
108 if (val & (1 << (ldz - nBits - 1))) {
109 val += (1 << (ldz - nBits - 1));
110 ldz = sizeof(uint32_t) * 8 - CAMath::Clz(val);
111 }
112 val &= ((1 << ldz) - 1) ^ ((1 << (ldz - nBits)) - 1);
113 if (val >= max) {
114 val = max - 1;
115 }
116 // GPUInfo("CHANGING X %x --> %x", (uint32_t) v, val);
117 v = val;
118 }
