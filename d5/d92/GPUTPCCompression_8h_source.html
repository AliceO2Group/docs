106template <class T>
107GPUdi() void GPUTPCCompression::truncateSignificantBits(T& v, uint32_t nBits, uint32_t max)
108{
109 if (nBits == 0) {
110 return;
111 }
112
113 uint32_t val = v;
114 uint32_t ldz = sizeof(uint32_t) * 8 - CAMath::Clz(val);
115 if (val && ldz > nBits) {
116 if (val & (1 << (ldz - nBits - 1))) {
117 val += (1 << (ldz - nBits - 1));
118 ldz = sizeof(uint32_t) * 8 - CAMath::Clz(val);
119 }
120 val &= ((1 << ldz) - 1) ^ ((1 << (ldz - nBits)) - 1);
121 if (val >= max) {
122 val = max - 1;
123 }
124 // GPUInfo("CHANGING X %x --> %x", (uint32_t) v, val);
125 v = val;
126 }
