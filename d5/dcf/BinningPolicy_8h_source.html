25 inline void expandConstantBinning(std::vector<double> const& bins, std::vector<double>& expanded)
26 {
27  if (bins[0] != VARIABLE_WIDTH) {
28  int nBins = static_cast<int>(bins[0]);
29  expanded.clear();
30  expanded.resize(nBins + 2);
31  expanded[0] = VARIABLE_WIDTH;
32  for (int i = 0; i <= nBins; i++) {
33  expanded[i + 1] = bins[1] + i * (bins[2] - bins[1]) / nBins;
34  }
35  }
36 }

40 struct BinningPolicyBase {
41  BinningPolicyBase(std::array<std::vector<double>, N> bins, bool ignoreOverflows = true) : mBins(bins), mIgnoreOverflows(ignoreOverflows)
42  {
43  static_assert(N <= 3, "No default binning for more than 3 columns, you need to implement a binning class yourself");
44  for (int i = 0; i < N; i++) {
45  binning_helpers::expandConstantBinning(bins[i], mBins[i]);
46  }

49  template <typename... Ts>
50  int getBin(std::tuple<Ts...> const& data) const
51  {
52  static_assert(sizeof...(Ts) == N, "There must be the same number of binning axes and data values/columns");
53 
54  unsigned int i = 2, j = 2, k = 2;
55  if (this->mIgnoreOverflows) {
56  // underflow
57  if (std::get<0>(data) < this->mBins[0][1]) { // mBins[0][0] is a dummy VARIABLE_WIDTH
58  return -1;
59  }
60  if constexpr (N > 1) {
61  if (std::get<1>(data) < this->mBins[1][1]) { // mBins[1][0] is a dummy VARIABLE_WIDTH
62  return -1;
63  }
64  }
65  if constexpr (N > 2) {
66  if (std::get<2>(data) < this->mBins[2][1]) { // mBins[2][0] is a dummy VARIABLE_WIDTH
67  return -1;
68  }
69  }
70  } else {
71  i = 1;
72  j = 1;
73  k = 1;
74  }
75 
76  for (; i < this->mBins[0].size(); i++) {
77  if (std::get<0>(data) < this->mBins[0][i]) {
78 
79  if constexpr (N > 1) {
80  for (; j < this->mBins[1].size(); j++) {
81  if (std::get<1>(data) < this->mBins[1][j]) {
82 
83  if constexpr (N > 2) {
84  for (; k < this->mBins[2].size(); k++) {
85  if (std::get<2>(data) < this->mBins[2][k]) {
86  return getBinAt(i, j, k);
87  }
88  }
89  if (this->mIgnoreOverflows) {
90  return -1;
91  }
92  }
93 
94  // overflow for mBins[2] only
95  return getBinAt(i, j, k);
96  }
97  }
98 
99  if (this->mIgnoreOverflows) {
100  return -1;
101  }
102 
103  // overflow for mBins[1] only
104  if constexpr (N > 2) {
105  for (k = 2; k < this->mBins[2].size(); k++) {
106  if (std::get<2>(data) < this->mBins[2][k]) {
107  return getBinAt(i, j, k);
108  }
109  }
110  }
111  }
112 
113  // overflow for mBins[2] and mBins[1]
114  return getBinAt(i, j, k);
115  }
116  }
117 
118  if (this->mIgnoreOverflows) {
119  // overflow
120  return -1;
121  }
122 
123  // overflow for mBins[0] only
124  if constexpr (N > 1) {
125  for (j = 2; j < this->mBins[1].size(); j++) {
126  if (std::get<1>(data) < this->mBins[1][j]) {
127 
128  if constexpr (N > 2) {
129  for (k = 2; k < this->mBins[2].size(); k++) {
130  if (std::get<2>(data) < this->mBins[2][k]) {
131  return getBinAt(i, j, k);
132  }
133  }
134  }
135 
136  // overflow for mBins[0] and mBins[2]
137  return getBinAt(i, j, k);
138  }
139  }
140  }
141 
142  // overflow for mBins[0] and mBins[1]
143  if constexpr (N > 2) {
144  for (k = 2; k < this->mBins[2].size(); k++) {
145  if (std::get<2>(data) < this->mBins[2][k]) {
146  return getBinAt(i, j, k);
147  }
148  }
149  }
150 
151  // overflow for all bins
152  return getBinAt(i, j, k);
153  }
154 
155  // Note: Overflow / underflow bin -1 is not included
156  int getXBinsCount() const
157  {
158  return getBinsCount(mBins[0]);
159  }
160 
161  // Note: Overflow / underflow bin -1 is not included
162  int getYBinsCount() const
163  {
164  if constexpr (N == 1) {
165  return 0;
166  }
167  return getBinsCount(mBins[1]);
168  }
169 
170  // Note: Overflow / underflow bin -1 is not included
171  int getZBinsCount() const
172  {
173  if constexpr (N < 3) {
174  return 0;
175  }
176  return getBinsCount(mBins[2]);
177  }
178 
179  // Note: Overflow / underflow bin -1 is not included
180  int getAllBinsCount() const
181  {
182  if constexpr (N == 1) {
183  return getXBinsCount();
184  }
185  if constexpr (N == 2) {
186  return getXBinsCount() * getYBinsCount();
187  }
188  if constexpr (N == 2) {
189  return getXBinsCount() * getYBinsCount() * getZBinsCount();
190  }
191  return -1;
192  }
193 
194  std::array<std::vector<double>, N> mBins;
195  bool mIgnoreOverflows;
196 
197  private:
198  // We substract 1 to account for VARIABLE_WIDTH in the bins vector
199  // We substract second 1 if we omit values below minima (underflow, mapped to -1)
200  // Otherwise we add 1 and we get the number of bins including those below and over the outer edges
201  int getBinAt(unsigned int iRaw, unsigned int jRaw, unsigned int kRaw) const
202  {
203  int shiftBinsWithoutOverflow = getOverflowShift();
204  unsigned int i = iRaw - 1 - shiftBinsWithoutOverflow;
205  unsigned int j = jRaw - 1 - shiftBinsWithoutOverflow;
206  unsigned int k = kRaw - 1 - shiftBinsWithoutOverflow;
207  auto xBinsCount = getXBinsCount();
208  if constexpr (N == 1) {
209  return i;
210  } else if constexpr (N == 2) {
211  return i + j * xBinsCount;
212  } else if constexpr (N == 3) {
213  return i + j * xBinsCount + k * xBinsCount * getYBinsCount();
214  } else {
215  return -1;
216  }
217  }
218 
219  int getOverflowShift() const
220  {
221  return mIgnoreOverflows ? 1 : -1;
222  }
223 
224  // Note: Overflow / underflow bin -1 is not included
225  int getBinsCount(std::vector<double> const& bins) const
226  {
227  return bins.size() - 1 - getOverflowShift();
228  }
229 };
