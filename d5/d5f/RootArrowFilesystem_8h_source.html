82// A registry for all the possible ways of encoding a table in a TFile
83struct RootObjectReadingCapability {
84 // The unique name of this capability
85 std::string name = "unknown";
86 // Convert a logical filename to an actual object to be read
87 // This can be used, e.g. to read an RNTuple stored in
88 // a flat directory structure in a TFile vs a TTree stored inside
89 // a TDirectory (e.g. /DF_1000/o2tracks).
90 std::function<std::string(std::string)> lfn2objectPath;
91 // Given a TFile, return the object which this capability support
92 // Use a void * in order not to expose the kind of object to the
93 // generic reading code. This is also where we load the plugin
94 // which will be used for the actual creation.
95 std::function<void*(TDirectoryFile* file, std::string const& path)> getHandle;
96 // Same as the above, but uses a TBufferFile as storage
97 std::function<void*(TBufferFile*, std::string const&)> getBufferHandle;
98 // This must be implemented to load the actual RootArrowFactory plugin which
99 // implements this capability. This way the detection of the file format
100 // (via get handle) does not need to know about the actual code which performs
101 // the serialization (and might depend on e.g. RNTuple).
102 std::function<RootArrowFactory&()> factory;
103};

109// This acts as registry of all the capabilities (i.e. the ability to

111// the factory (i.e. the serialization plugin)
112struct RootObjectReadingFactory {
113 std::vector<RootObjectReadingCapability> capabilities;
