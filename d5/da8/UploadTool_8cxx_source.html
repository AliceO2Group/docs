58// Remove leading whitespace
59std::string ltrimSpace(std::string src)
60{
61 return src.erase(0, src.find_first_not_of(' '));
62}
63
64// Remove trailing whitespace
65std::string rtrimSpace(std::string src)
66{
67 return src.erase(src.find_last_not_of(' ') + 1);
68}
69
70// Remove leading/trailing whitespace
71std::string trimSpace(std::string const& src)
72{
73 return ltrimSpace(rtrimSpace(src));
74}

76// Split a given string on a delim character, return vector of tokens
77// If trim is true, then also remove leading/trailing whitespace of each token.
78std::vector<std::string> splitString(const std::string& src, char delim, bool trim = false)
79{
80 std::stringstream ss(src);
81 std::string token;
82 std::vector<std::string> tokens;
83
84 while (std::getline(ss, token, delim)) {
85 token = (trim ? trimSpace(token) : token);
86 if (!token.empty()) {
87 tokens.push_back(std::move(token));
88 }
89 }
90
91 return tokens;
92}

96int main(int argc, char* argv[])

102 }

108 std::map<std::string, std::string> filter;

110 if (starttimestamp == -1) {
111 starttimestamp = o2::ccdb::getCurrentTimestamp();
112 }
113 long endtimestamp = vm["endtimestamp"].as<long>();
114 if (endtimestamp == -1) {
115 constexpr long SECONDSPERDAY = 1 * 24 * 60 * 60;
116 endtimestamp = o2::ccdb::getFutureTimestamp(SECONDSPERDAY);
117 }
118
119 auto filename = vm["file"].as<std::string>();
120 auto path = vm["path"].as<std::string>();
121 auto keyname = vm["key"].as<std::string>();
122
123 // Take a vector of strings with elements of form a=b, and
124 // return a vector of pairs with each pair of form <a, b>
125 auto toKeyValPairs = [](std::vector<std::string> const& tokens) {
126 std::vector<std::pair<std::string, std::string>> pairs;
127
128 for (auto& token : tokens) {
129 auto keyval = splitString(token, '=');
130 if (keyval.size() != 2) {
131 // LOG(fatal) << "Illegal command-line key/value string: " << token;
132 continue;
133 }
134
135 std::pair<std::string, std::string> pair = std::make_pair(keyval[0], trimSpace(keyval[1]));
136 pairs.push_back(pair);
137 }
138
139 return pairs;
140 };
141 auto metastring = vm["meta"].as<std::string>();
142 auto keyvalues = toKeyValPairs(splitString(metastring, ';', true));
143
144 // fill meta map
145 std::map<std::string, std::string> meta;
146 for (auto& p : keyvalues) {
147 meta[p.first] = p.second;
148 }
149
150 TFile f(filename.c_str());
151 auto key = f.GetKey(keyname.c_str());
152 if (key) {
153 // get type of key
154 auto classname = key->GetClassName();
155 auto tcl = TClass::GetClass(classname);
156 auto object = f.Get<void>(keyname.c_str());
157 if (tcl->InheritsFrom("TTree")) {
158 auto tree = static_cast<TTree*>(object);
159 tree->LoadBaskets(0x1L << 32); // make tree memory based
160 tree->SetDirectory(nullptr);
161 }
162 // convert classname to typeinfo
163 // typeinfo
164 auto ti = tcl->GetTypeInfo();
165
166 std::cout << " Uploading an object of type " << key->GetClassName()
167 << " to path " << path << " with timestamp validity from " << starttimestamp
168 << " to " << endtimestamp << "\n";
169
170 api.storeAsTFile_impl(object, *ti, path, meta, starttimestamp, endtimestamp);
171 if (!api.isSnapshotMode() && meta.find("adjustableEOV") != meta.end() && meta.find("default") == meta.end()) {
172 o2::ccdb::CcdbObjectInfo oi(path, classname, filename, meta, starttimestamp, endtimestamp);
173 o2::ccdb::adjustOverriddenEOV(api, oi);
174 }
175 } else {
176 std::cerr << "Key " << keyname << " does not exist\n";
177 }
178
179 return 0;
180}
