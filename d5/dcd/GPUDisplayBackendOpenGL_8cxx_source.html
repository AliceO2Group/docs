102void GPUDisplayBackendOpenGL::createFB(GLfb& fb, bool tex, bool withDepth, bool msaa, uint32_t width, uint32_t height)
103{
104 fb.tex = tex;
105 fb.depth = withDepth;
106 fb.msaa = msaa;
107 GLint drawFboId = 0, readFboId = 0;
108 glGetIntegerv(GL_DRAW_FRAMEBUFFER_BINDING, &drawFboId);
109 glGetIntegerv(GL_READ_FRAMEBUFFER_BINDING, &readFboId);
110 CHKERR(glGenFramebuffers(1, &fb.fb_id));
111 CHKERR(glBindFramebuffer(GL_FRAMEBUFFER, fb.fb_id));
112
113 auto createFB_texture = [&](GLuint& id, GLenum storage, GLenum attachment) {
114 GLenum textureType = fb.msaa ? GL_TEXTURE_2D_MULTISAMPLE : GL_TEXTURE_2D;
115 CHKERR(glGenTextures(1, &id));
116 CHKERR(glBindTexture(textureType, id));
117 if (fb.msaa) {
118 CHKERR(glTexImage2DMultisample(GL_TEXTURE_2D_MULTISAMPLE, mDisplay->cfgR().drawQualityMSAA, storage, width, height, false));
119 } else {
120 CHKERR(glTexImage2D(GL_TEXTURE_2D, 0, storage, width, height, 0, storage, GL_UNSIGNED_BYTE, nullptr));
121 CHKERR(glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR));
122 CHKERR(glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR));
123 }
124 CHKERR(glFramebufferTexture2D(GL_FRAMEBUFFER, attachment, textureType, id, 0));
125 };
126
127 auto createFB_renderbuffer = [&](GLuint& id, GLenum storage, GLenum attachment) {
128 CHKERR(glGenRenderbuffers(1, &id));
129 CHKERR(glBindRenderbuffer(GL_RENDERBUFFER, id));
130 if (fb.msaa) {
131 CHKERR(glRenderbufferStorageMultisample(GL_RENDERBUFFER, mDisplay->cfgR().drawQualityMSAA, storage, width, height));
132 } else {
133 CHKERR(glRenderbufferStorage(GL_RENDERBUFFER, storage, width, height));
134 }
135 CHKERR(glFramebufferRenderbuffer(GL_FRAMEBUFFER, attachment, GL_RENDERBUFFER, id));
136 };
137
138 if (tex) {
139 createFB_texture(fb.fbCol_id, GL_RGBA, GL_COLOR_ATTACHMENT0);
140 } else {
141 createFB_renderbuffer(fb.fbCol_id, GL_RGBA, GL_COLOR_ATTACHMENT0);
142 }
143
144 if (withDepth) {
145 if (tex && fb.msaa) {
146 createFB_texture(fb.fbDepth_id, GL_DEPTH_COMPONENT24, GL_DEPTH_ATTACHMENT);
147 } else {
148 createFB_renderbuffer(fb.fbDepth_id, GL_DEPTH_COMPONENT24, GL_DEPTH_ATTACHMENT);
149 }
150 }
151
152 GLenum status = glCheckFramebufferStatus(GL_FRAMEBUFFER);
153 if (status != GL_FRAMEBUFFER_COMPLETE) {
154 GPUError("Error creating framebuffer (tex %d) - incomplete (%d)", (int32_t)tex, status);
155 exit(1);
156 }
157 CHKERR(glBindFramebuffer(GL_DRAW_FRAMEBUFFER, drawFboId));
158 CHKERR(glBindFramebuffer(GL_READ_FRAMEBUFFER, readFboId));
159 fb.created = true;
160}

180uint32_t GPUDisplayBackendOpenGL::drawVertices(const vboList& v, const drawType tt)
181{
182 static constexpr GLenum types[3] = {GL_POINTS, GL_LINES, GL_LINE_STRIP};
183 GLenum t = types[tt];
184 auto first = std::get<0>(v);
185 auto count = std::get<1>(v);
186 auto iSector = std::get<2>(v);
187 if (count == 0) {
188 return 0;
189 }
190
191 if (mDisplay->useMultiVBO()) {
192 if (mDisplay->cfgR().openGLCore) {
193 CHKERR(glBindVertexArray(mVertexArray));
194 }
195 CHKERR(glBindBuffer(GL_ARRAY_BUFFER, mVBOId[iSector]));
196#ifndef GPUCA_DISPLAY_OPENGL_CORE
197 if (!mDisplay->cfgR().openGLCore) {
198 CHKERR(glVertexPointer(3, GL_FLOAT, 0, nullptr));
199 } else
200#endif
201 {
202 CHKERR(glVertexAttribPointer(0, 3, GL_FLOAT, GL_FALSE, 0, nullptr));
203 glEnableVertexAttribArray(0);
204 }
205 }
206
207 if (mDisplay->cfgR().useGLIndirectDraw) {
208 CHKERR(glMultiDrawArraysIndirect(t, (void*)(size_t)((mIndirectSectorOffset[iSector] + first) * sizeof(DrawArraysIndirectCommand)), count, 0));
209 } else if (OPENGL_EMULATE_MULTI_DRAW) {
210 for (uint32_t k = 0; k < count; k++) {
211 CHKERR(glDrawArrays(t, mDisplay->vertexBufferStart()[iSector][first + k], mDisplay->vertexBufferCount()[iSector][first + k]));
212 }
213 } else {
214 static_assert(sizeof(GLsizei) == sizeof(*mDisplay->vertexBufferCount()[iSector].data()), "Invalid counter size does not match GLsizei");
215 CHKERR(glMultiDrawArrays(t, mDisplay->vertexBufferStart()[iSector].data() + first, ((const GLsizei*)mDisplay->vertexBufferCount()[iSector].data()) + first, count));
216 }
217 return count;
218}

301int32_t GPUDisplayBackendOpenGL::InitBackendA()
302{
303 if (mDisplay->GetProcessingSettings().debugLevel >= 2) {
304 auto renderer = glGetString(GL_RENDERER);
305 GPUInfo("Renderer: %s", (const char*)renderer);
306 }
307
308 int32_t glVersion[2] = {0, 0};
309 glGetIntegerv(GL_MAJOR_VERSION, &glVersion[0]);
310 glGetIntegerv(GL_MINOR_VERSION, &glVersion[1]);
311 if (glVersion[0] < GPUDisplayFrontend::GL_MIN_VERSION_MAJOR || (glVersion[0] == GPUDisplayFrontend::GL_MIN_VERSION_MAJOR && glVersion[1] < GPUDisplayFrontend::GL_MIN_VERSION_MINOR)) {
312 GPUError("Unsupported OpenGL runtime %d.%d < %d.%d", glVersion[0], glVersion[1], GPUDisplayFrontend::GL_MIN_VERSION_MAJOR, GPUDisplayFrontend::GL_MIN_VERSION_MINOR);
313 return (1);
314 }
315 mVBOId.resize(GPUCA_NSECTORS);
316 CHKERR(glCreateBuffers(mVBOId.size(), mVBOId.data()));
317 CHKERR(glBindBuffer(GL_ARRAY_BUFFER, mVBOId[0]));
318 CHKERR(glGenBuffers(1, &mIndirectId));
319 CHKERR(glBindBuffer(GL_DRAW_INDIRECT_BUFFER, mIndirectId));
320#ifndef GPUCA_DISPLAY_OPENGL_CORE
321 CHKERR(glShadeModel(GL_SMOOTH)); // Enable Smooth Shading
322#endif
323 setDepthBuffer();
324 setQuality();
325 CHKERR(mVertexShader = glCreateShader(GL_VERTEX_SHADER));
326 CHKERR(mFragmentShader = glCreateShader(GL_FRAGMENT_SHADER));
327 CHKERR(mVertexShaderTexture = glCreateShader(GL_VERTEX_SHADER));
328 CHKERR(mFragmentShaderTexture = glCreateShader(GL_FRAGMENT_SHADER));
329 CHKERR(mFragmentShaderText = glCreateShader(GL_FRAGMENT_SHADER));
330#if defined(GL_VERSION_4_6) && GL_VERSION_4_6 == 1 && defined(GPUCA_BUILD_EVENT_DISPLAY_VULKAN)
331 if (getenv("USE_SPIRV_SHADERS") && atoi(getenv("USE_SPIRV_SHADERS"))) {
332 CHKERR(glShaderBinary(1, &mVertexShader, GL_SHADER_BINARY_FORMAT_SPIR_V_ARB, _binary_shaders_shaders_vertex_vert_spv_start, _binary_shaders_shaders_vertex_vert_spv_len));
333 CHKERR(glSpecializeShader(mVertexShader, "main", 0, nullptr, nullptr));
334 CHKERR(glShaderBinary(1, &mFragmentShader, GL_SHADER_BINARY_FORMAT_SPIR_V_ARB, _binary_shaders_shaders_fragmentUniform_frag_spv_start, _binary_shaders_shaders_fragmentUniform_frag_spv_len));
335 CHKERR(glSpecializeShader(mFragmentShader, "main", 0, nullptr, nullptr));
336 GPUInfo("Using SPIR-V shaders");
337 mSPIRVShaders = true;
338 } else
339#endif
340 {
341 CHKERR(glShaderSource(mVertexShader, 1, &GPUDisplayShaders::vertexShader, nullptr));
342 CHKERR(glCompileShader(mVertexShader));
343 CHKERR(glShaderSource(mFragmentShader, 1, &GPUDisplayShaders::fragmentShader, nullptr));
344 CHKERR(glCompileShader(mFragmentShader));
345 }
346 CHKERR(glShaderSource(mVertexShaderTexture, 1, &GPUDisplayShaders::vertexShaderTexture, nullptr));
347 CHKERR(glCompileShader(mVertexShaderTexture));
348 CHKERR(glShaderSource(mFragmentShaderTexture, 1, &GPUDisplayShaders::fragmentShaderTexture, nullptr));
349 CHKERR(glCompileShader(mFragmentShaderTexture));
350 CHKERR(glShaderSource(mFragmentShaderText, 1, &GPUDisplayShaders::fragmentShaderText, nullptr));
351 CHKERR(glCompileShader(mFragmentShaderText));
352 if (checkShaderStatus(mVertexShader) || checkShaderStatus(mFragmentShader) || checkShaderStatus(mVertexShaderTexture) || checkShaderStatus(mFragmentShaderTexture) || checkShaderStatus(mFragmentShaderText)) {
353 return 1;
354 }
355 CHKERR(mShaderProgram = glCreateProgram());
356 CHKERR(glAttachShader(mShaderProgram, mVertexShader));
357 CHKERR(glAttachShader(mShaderProgram, mFragmentShader));
358 CHKERR(glLinkProgram(mShaderProgram));
359 if (checkProgramStatus(mShaderProgram)) {
360 return 1;
361 }
362 if (mSPIRVShaders) {
363 CHKERR(glGenBuffers(1, &mSPIRVModelViewBuffer));
364 CHKERR(glGenBuffers(1, &mSPIRVColorBuffer));
365 } else {
366 CHKERR(mModelViewProjId = glGetUniformLocation(mShaderProgram, "ModelViewProj"));
367 CHKERR(mColorId = glGetUniformLocation(mShaderProgram, "color"));
368 }
369 CHKERR(mShaderProgramText = glCreateProgram());
370 CHKERR(glAttachShader(mShaderProgramText, mVertexShaderTexture));
371 CHKERR(glAttachShader(mShaderProgramText, mFragmentShaderText));
372 CHKERR(glLinkProgram(mShaderProgramText));
373 if (checkProgramStatus(mShaderProgramText)) {
374 return 1;
375 }
376 CHKERR(mModelViewProjIdText = glGetUniformLocation(mShaderProgramText, "projection"));
377 CHKERR(mColorIdText = glGetUniformLocation(mShaderProgramText, "textColor"));
378 CHKERR(mShaderProgramTexture = glCreateProgram());
379 CHKERR(glAttachShader(mShaderProgramTexture, mVertexShaderTexture));
380 CHKERR(glAttachShader(mShaderProgramTexture, mFragmentShaderTexture));
381 CHKERR(glLinkProgram(mShaderProgramTexture));
382 if (checkProgramStatus(mShaderProgramTexture)) {
383 return 1;
384 }
385 CHKERR(mModelViewProjIdTexture = glGetUniformLocation(mShaderProgramTexture, "projection"));
386 CHKERR(mAlphaIdTexture = glGetUniformLocation(mShaderProgramTexture, "alpha"));
387 CHKERR(glGenVertexArrays(1, &mVertexArray));
388
389 CHKERR(glGenVertexArrays(1, &VAO_texture));
390 CHKERR(glGenBuffers(1, &VBO_texture));
391 CHKERR(glBindVertexArray(VAO_texture));
392 CHKERR(glBindBuffer(GL_ARRAY_BUFFER, VBO_texture));
393 CHKERR(glBufferData(GL_ARRAY_BUFFER, sizeof(float) * 6 * 4, nullptr, GL_DYNAMIC_DRAW));
394 CHKERR(glEnableVertexAttribArray(0));
395 CHKERR(glVertexAttribPointer(0, 4, GL_FLOAT, GL_FALSE, 4 * sizeof(float), nullptr));
396 CHKERR(glBindBuffer(GL_ARRAY_BUFFER, 0));
397
398 CHKERR(glBindVertexArray(0));
399 return 0;
400}

472void GPUDisplayBackendOpenGL::prepareDraw(const hmm_mat4& proj, const hmm_mat4& view, bool requestScreenshot, bool toMixBuffer, float includeMixImage)
473{
474 if (mDisplay->updateRenderPipeline() || mDownsampleFactor != getDownsampleFactor(requestScreenshot)) {
475 updateRenderer(requestScreenshot);
476 }
477 if (mOffscreenBufferMSAA.created) {
478 setFrameBuffer(mOffscreenBufferMSAA.fb_id);
479 } else if (toMixBuffer) {
480 setFrameBuffer(mMixBuffer.fb_id);
481 } else if (mOffscreenBuffer.created) {
482 setFrameBuffer(mOffscreenBuffer.fb_id);
483 } else {
484 setFrameBuffer(0);
485 }
486 clearScreen();
487#ifndef GPUCA_DISPLAY_OPENGL_CORE
488 if (mDisplay->cfgL().smoothPoints && !mDisplay->cfgR().openGLCore) {
489 CHKERR(glEnable(GL_POINT_SMOOTH));
490 } else {
491 CHKERR(glDisable(GL_POINT_SMOOTH));
492 }
493 if (mDisplay->cfgL().smoothLines && !mDisplay->cfgR().openGLCore) {
494 CHKERR(glEnable(GL_LINE_SMOOTH));
495 } else {
496 CHKERR(glDisable(GL_LINE_SMOOTH));
497 }
498#endif
499 CHKERR(glEnable(GL_BLEND));
500 CHKERR(glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA));
501
502 glViewport(0, 0, mRenderWidth, mRenderHeight);
503#ifndef GPUCA_DISPLAY_OPENGL_CORE
504 if (!mDisplay->cfgR().openGLCore) {
505 CHKERR(glEnableClientState(GL_VERTEX_ARRAY));
506 CHKERR(glVertexPointer(3, GL_FLOAT, 0, nullptr));
507 } else
508#endif
509 {
510 CHKERR(glBindVertexArray(mVertexArray));
511 CHKERR(glUseProgram(mShaderProgram));
512 CHKERR(glVertexAttribPointer(0, 3, GL_FLOAT, GL_FALSE, 0, nullptr));
513 CHKERR(glEnableVertexAttribArray(0));
514 }
515#ifndef GPUCA_DISPLAY_OPENGL_CORE
516 if (!mDisplay->cfgR().openGLCore) {
517 CHKERR(glMatrixMode(GL_PROJECTION));
518 CHKERR(glLoadMatrixf(&proj.Elements[0][0]));
519 CHKERR(glMatrixMode(GL_MODELVIEW));
520 CHKERR(glLoadMatrixf(&view.Elements[0][0]));
521 } else
522#endif
523 {
524 const hmm_mat4 modelViewProj = proj * view;
525 if (mSPIRVShaders) {
526 glBindBuffer(GL_UNIFORM_BUFFER, mSPIRVModelViewBuffer);
527 glBufferData(GL_UNIFORM_BUFFER, sizeof(modelViewProj), &modelViewProj, GL_STATIC_DRAW);
528 glBindBufferBase(GL_UNIFORM_BUFFER, 0, mSPIRVModelViewBuffer);
529 } else {
530 CHKERR(glUniformMatrix4fv(mModelViewProjId, 1, GL_FALSE, &modelViewProj.Elements[0][0]));
531 }
532 if (mMagneticFieldVisualization) {
533 CHKERR(glNamedBufferSubData(mFieldModelViewBuffer, 0, sizeof(modelViewProj), &modelViewProj));
534 }
535 }

538void GPUDisplayBackendOpenGL::finishDraw(bool doScreenshot, bool toMixBuffer, float includeMixImage)
539{
540#ifndef GPUCA_DISPLAY_OPENGL_CORE
541 if (!mDisplay->cfgR().openGLCore) {
542 CHKERR(glDisableClientState(GL_VERTEX_ARRAY));
543 } else
544#endif
545 {
546 CHKERR(glBindVertexArray(0));
547 CHKERR(glUseProgram(0));
548 }
549
550 if (mDisplay->cfgR().drawQualityMSAA) {
551 uint32_t dstId = toMixBuffer ? mMixBuffer.fb_id : (mDownsampleFactor != 1 ? mOffscreenBuffer.fb_id : 0);
552 CHKERR(glBlitNamedFramebuffer(mOffscreenBufferMSAA.fb_id, dstId, 0, 0, mRenderWidth, mRenderHeight, 0, 0, mRenderWidth, mRenderHeight, GL_COLOR_BUFFER_BIT, GL_LINEAR));
553 }
554 if (includeMixImage > 0) {
555 setFrameBuffer(mDownsampleFactor != 1 ? mOffscreenBuffer.fb_id : 0);
556 mixImages(includeMixImage);

680void GPUDisplayBackendOpenGL::addFontSymbol(int32_t symbol, int32_t sizex, int32_t sizey, int32_t offsetx, int32_t offsety, int32_t advance, void* data)
681{
682 int32_t oldAlign;
683 CHKERR(glGetIntegerv(GL_UNPACK_ALIGNMENT, &oldAlign));
684 CHKERR(glPixelStorei(GL_UNPACK_ALIGNMENT, 1));
685 if (symbol != (int32_t)mFontSymbols.size()) {
686 throw std::runtime_error("Incorrect symbol ID");
687 }
688 uint32_t texId;
689 glGenTextures(1, &texId);
690 std::vector<uint8_t> tmp;
691 if (!smoothFont()) {
692 tmp.resize(sizex * sizey);
693 uint8_t* src = (uint8_t*)data;
694 for (int32_t i = 0; i < sizex * sizey; i++) {
695 tmp[i] = src[i] > 128 ? 255 : 0;
696 }
697 data = tmp.data();
698 }
699 mFontSymbols.emplace_back(FontSymbolOpenGL{{{sizex, sizey}, {offsetx, offsety}, advance}, texId});
700 if (sizex == 0 || sizey == 0) {
701 return;
702 }
703 glBindTexture(GL_TEXTURE_2D, texId);
704 CHKERR(glTexImage2D(GL_TEXTURE_2D, 0, GL_RED, sizex, sizey, 0, GL_RED, GL_UNSIGNED_BYTE, data));
705 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
706 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
707 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
708 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
709 CHKERR(glPixelStorei(GL_UNPACK_ALIGNMENT, oldAlign));
710}

725void GPUDisplayBackendOpenGL::OpenGLPrint(const char* s, float x, float y, float* color, float scale)
726{
727 if (!mFreetypeInitialized || mDisplay->drawTextInCompatMode()) {
728 return;
729 }
730 CHKERR(glUniform4fv(mColorIdText, 1, color));
731
732 if (smoothFont()) {
733 scale *= 0.25f; // Font size is 48 to have nice bitmap, scale to size 12
734 }
735 for (const char* c = s; *c; c++) {
736 if ((int32_t)*c > (int32_t)mFontSymbols.size()) {
737 GPUError("Trying to draw unsupported symbol: %d > %d\n", (int32_t)*c, (int32_t)mFontSymbols.size());
738 continue;
739 }
740 FontSymbolOpenGL sym = mFontSymbols[*c];
741 if (sym.size[0] && sym.size[1]) {
742 float xpos = x + sym.offset[0] * scale;
743 float ypos = y - (sym.size[1] - sym.offset[1]) * scale;
744 float w = sym.size[0] * scale;
745 float h = sym.size[1] * scale;
746 float vertices[6][4] = {
747 {xpos, ypos + h, 0.0f, 0.0f},
748 {xpos, ypos, 0.0f, 1.0f},
749 {xpos + w, ypos, 1.0f, 1.0f},
750 {xpos, ypos + h, 0.0f, 0.0f},
751 {xpos + w, ypos, 1.0f, 1.0f},
752 {xpos + w, ypos + h, 1.0f, 0.0f}};
753 CHKERR(glBindTexture(GL_TEXTURE_2D, sym.texId));
754 CHKERR(glBindBuffer(GL_ARRAY_BUFFER, VBO_text));
755 CHKERR(glBufferSubData(GL_ARRAY_BUFFER, 0, sizeof(vertices), vertices));
756 CHKERR(glBindBuffer(GL_ARRAY_BUFFER, 0));
757 CHKERR(glDrawArrays(GL_TRIANGLES, 0, 6));
758 }
759 x += (sym.advance >> 6) * scale; // shift is in 1/64th of a pixel
760 }
761 if (!mDisplay->useMultiVBO()) {
762 CHKERR(glBindBuffer(GL_ARRAY_BUFFER, mVBOId[0])); // If we don't use multiple buffers, we keep the default buffer bound
763 }
764}
