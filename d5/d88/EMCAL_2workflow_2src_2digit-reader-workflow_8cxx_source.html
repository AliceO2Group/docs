43WorkflowSpec defineDataProcessing(ConfigContext const& cfgc)
44{
45 bool disableMC = cfgc.options().get<bool>("disable-mc");
46 o2::conf::ConfigurableParam::updateFromString(cfgc.options().get<std::string>("configKeyValues"));
47
48 WorkflowSpec specs;
49 specs.emplace_back(o2::emcal::getPublisherSpec<std::vector<o2::emcal::Digit>>(PublisherConf{
50 "emcal-digit-reader",
51 "o2sim",
52 "emcdigits.root",
53 {"digitbranch", "EMCALDigit", "Digit branch"},
54 {"digittriggerbranch", "EMCALDigitTRGR", "Trigger record branch"},
55 {"mcbranch", "EMCALDigitMCTruth", "MC label branch"},
56 o2::framework::OutputSpec{"EMC", "DIGITS"},
57 o2::framework::OutputSpec{"EMC", "DIGITSTRGR"},
58 o2::framework::OutputSpec{"EMC", "DIGITSMCTR"}},
59 !disableMC));
60
61 // configure dpl timer to inject correct firstTForbit: start from the 1st orbit of TF containing 1st sampled orbit
62 o2::raw::HBFUtilsInitializer hbfIni(cfgc, specs);
63
64 return specs;
