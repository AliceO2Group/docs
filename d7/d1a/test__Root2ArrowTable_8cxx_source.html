44 TEST_CASE("RootTree2Table")
45 {
46  using namespace o2::framework;
48  TTree t1("t1", "a simple Tree with simple variables");
49  Float_t xyz[3];
50  Int_t ij[2];
51  Float_t px, py, pz;
52  Double_t random;
53  Int_t ev;
54  t1.Branch("px", &px, "px/F");
55  t1.Branch("py", &py, "py/F");
56  t1.Branch("pz", &pz, "pz/F");
57  t1.Branch("random", &random, "random/D");
58  t1.Branch("ev", &ev, "ev/I");
59  t1.Branch("xyz", xyz, "xyz[3]/F");
60  t1.Branch("ij", ij, "ij[2]/I");
61  // fill the tree
62  for (Int_t i = 0; i < 1000; i++) {
63  xyz[0] = 1;
64  xyz[1] = 2;
65  xyz[2] = 3;
66  gRandom->Rannor(px, py);
67  pz = px * px + py * py;
68  xyz[2] = i + 1;
69  ij[0] = i;
70  ij[1] = i + 1;
71  random = gRandom->Rndm();
72  ev = i + 1;
73  t1.Fill();
74  }
75 
76  // Create an arrow table from this.
77  TableBuilder builder;
78  TTreeReader reader(&t1);
79  auto&& xyzReader = HolderMaker<float[3]>::make(reader, "xyz");
80  auto&& ijkReader = HolderMaker<int[2]>::make(reader, "ij");
81  auto&& pxReader = HolderMaker<float>::make(reader, "px");
82  auto&& pyReader = HolderMaker<float>::make(reader, "py");
83  auto&& pzReader = HolderMaker<float>::make(reader, "pz");
84  auto&& randomReader = HolderMaker<double>::make(reader, "random");
85  auto&& evReader = HolderMaker<int>::make(reader, "ev");
86 
87  RootTableBuilderHelpers::convertTTree(builder, reader, std::move(xyzReader), std::move(ijkReader), std::move(pxReader), std::move(pyReader), std::move(pzReader), std::move(randomReader), std::move(evReader));
88  auto table = builder.finalize();
89  REQUIRE(table->num_rows() == 1000);
90  REQUIRE(table->num_columns() == 7);
91  REQUIRE(table->schema()->field(0)->type()->id() == arrow::fixed_size_list(arrow::float32(), 3)->id());
92  REQUIRE(table->schema()->field(1)->type()->id() == arrow::fixed_size_list(arrow::int32(), 2)->id());
93  REQUIRE(table->schema()->field(2)->type()->id() == arrow::float32()->id());
94  REQUIRE(table->schema()->field(3)->type()->id() == arrow::float32()->id());
95  REQUIRE(table->schema()->field(4)->type()->id() == arrow::float32()->id());
96  REQUIRE(table->schema()->field(5)->type()->id() == arrow::float64()->id());
97  REQUIRE(table->schema()->field(6)->type()->id() == arrow::int32()->id());
98 
99  {
100  auto chunkToUse = table->column(0)->chunk(0);
101  chunkToUse = std::dynamic_pointer_cast<arrow::FixedSizeListArray>(chunkToUse)->values();
102  auto array = std::static_pointer_cast<arrow::FloatArray>(chunkToUse);
103  // array of 3 floats, time 1000.
104  REQUIRE(array->length() == 3000);
105  const float* c = reinterpret_cast<float const*>(array->values()->data());
106 
107  CHECK(c[0] == 1);
108  CHECK(c[1] == 2);
109  CHECK(c[2] == 1);

113  chunkToUse = std::dynamic_pointer_cast<arrow::FixedSizeListArray>(chunkToUse)->values();
