44 GPUd() bool GPUTPCTrackletConstructor::CheckCov(MEM_LG2(GPUTPCTrackParam) & GPUrestrict() tParam)
45 {
46  bool ok = 1;
47  const float* c = tParam.Cov();
48  for (int32_t i = 0; i < 15; i++) {
49  ok = ok && CAMath::Finite(c[i]);
50  }
51  for (int32_t i = 0; i < 5; i++) {
52  ok = ok && CAMath::Finite(tParam.Par()[i]);
53  }
54  ok = ok && (tParam.X() > 50);
55  if (c[0] <= 0 || c[2] <= 0 || c[5] <= 0 || c[9] <= 0 || c[14] <= 0) {
56  ok = 0;
57  }
58  return (ok);
59 }

62 GPUd() void GPUTPCTrackletConstructor::StoreTracklet(int32_t /*nBlocks*/, int32_t /*nThreads*/, int32_t /*iBlock*/, int32_t /*iThread*/, GPUsharedref() MEM_LOCAL(GPUSharedMemory) & s, GPUTPCThreadMemory& GPUrestrict() r, GPUconstantref() MEM_LG2(GPUTPCTracker) & GPUrestrict() tracker, MEM_LG3(GPUTPCTrackParam) & GPUrestrict() tParam, calink* rowHits)
63 {
64  // reconstruction of tracklets, tracklet store step
65  const uint32_t nHits = r.mLastRow + 1 - r.mFirstRow;
66  if (nHits == 0 || r.mNHits == 0 || (r.mNHits < GPUCA_TRACKLET_SELECTOR_MIN_HITS_B5(tParam.QPt() * tracker.Param().qptB5Scaler) || !CheckCov(tParam) || CAMath::Abs(tParam.GetQPt() * tracker.Param().qptB5Scaler) > tracker.Param().rec.maxTrackQPtB5)) {
67  CADEBUG(printf(" Rejected: nHits %d QPt %f MinHits %d MaxQPt %f CheckCov %d\n", r.mNHits, tParam.QPt(), GPUCA_TRACKLET_SELECTOR_MIN_HITS_B5(tParam.QPt() * tracker.Param().qptB5Scaler), tracker.Param().rec.maxTrackQPtB5, (int32_t)CheckCov(tParam)));
68  return;
69  }
70 
71  /*GPUInfo("Tracklet %d: Hits %3d NDF %3d Chi %8.4f Sign %f Cov: %2.4f %2.4f %2.4f %2.4f %2.4f %2.4f %2.4f %2.4f %2.4f %2.4f %2.4f %2.4f %2.4f %2.4f %2.4f", r.mISH, r.mNHits, tParam.GetNDF(), tParam.GetChi2(), tParam.GetSignCosPhi(),
72  tParam.Cov()[0], tParam.Cov()[1], tParam.Cov()[2], tParam.Cov()[3], tParam.Cov()[4], tParam.Cov()[5], tParam.Cov()[6], tParam.Cov()[7], tParam.Cov()[8], tParam.Cov()[9],
73  tParam.Cov()[10], tParam.Cov()[11], tParam.Cov()[12], tParam.Cov()[13], tParam.Cov()[14]);*/
74 
75  uint32_t hitout = CAMath::AtomicAdd(tracker.NRowHits(), nHits);
76  if (hitout + nHits > tracker.NMaxRowHits()) {
77  tracker.raiseError(GPUErrors::ERROR_TRACKLET_HIT_OVERFLOW, tracker.ISlice(), hitout + nHits, tracker.NMaxRowHits());
78  CAMath::AtomicExch(tracker.NRowHits(), tracker.NMaxRowHits());
79  return;
80  }
81  uint32_t itrout = CAMath::AtomicAdd(tracker.NTracklets(), 1u);
82  if (itrout >= tracker.NMaxTracklets()) {
83  tracker.raiseError(GPUErrors::ERROR_TRACKLET_OVERFLOW, tracker.ISlice(), itrout, tracker.NMaxTracklets());
84  CAMath::AtomicExch(tracker.NTracklets(), tracker.NMaxTracklets());
85  return;
86  }
87 
88  GPUglobalref() MEM_GLOBAL(GPUTPCTracklet) & GPUrestrict() tracklet = tracker.Tracklets()[itrout];
89 
90  CADEBUG(printf(" Storing tracklet: %d rows\n", nHits));
91 
92  tracklet.SetFirstRow(r.mFirstRow);
93  tracklet.SetLastRow(r.mLastRow);
94  tracklet.SetFirstHit(hitout);
95  tracklet.SetParam(tParam.GetParam());
96  int32_t w = tracker.CalculateHitWeight(r.mNHits, tParam.GetChi2());
97  tracklet.SetHitWeight(w);
98 #ifdef __HIPCC__ // Todo: fixme!
99  for (int32_t iRow = r.mFirstRow - 1; ++iRow <= r.mLastRow; /*iRow++*/) {
100 #else
101  for (int32_t iRow = r.mFirstRow; iRow <= r.mLastRow; iRow++) {
102 #endif
103  calink ih = rowHits[iRow];
104  tracker.TrackletRowHits()[hitout + (iRow - r.mFirstRow)] = ih;
105  if (ih != CALINK_INVAL && ih != CALINK_DEAD_CHANNEL) {
106  CA_MAKE_SHARED_REF(GPUTPCRow, row, tracker.Row(iRow), s.mRows[iRow]);
107  tracker.MaximizeHitWeight(row, ih, w);
108  }
109  }
110 }

113 GPUdic(2, 1) void GPUTPCTrackletConstructor::UpdateTracklet(int32_t /*nBlocks*/, int32_t /*nThreads*/, int32_t /*iBlock*/, int32_t /*iThread*/, GPUsharedref() T& s, GPUTPCThreadMemory& GPUrestrict() r, GPUconstantref() MEM_GLOBAL(GPUTPCTracker) & GPUrestrict() tracker, MEM_LG2(GPUTPCTrackParam) & GPUrestrict() tParam, int32_t iRow, calink& rowHit, calink* rowHits)
114 {
115  // reconstruction of tracklets, tracklets update step
116  CA_MAKE_SHARED_REF(GPUTPCRow, row, tracker.Row(iRow), s.mRows[iRow]);
117 
118  float y0 = row.Grid().YMin();
119  float stepY = row.HstepY();
120  float z0 = row.Grid().ZMin() - tParam.ZOffset();
121  float stepZ = row.HstepZ();
122 
123  if (r.mStage == 0) { // fitting part
124  do {
125  if (iRow < r.mStartRow || r.mCurrIH == CALINK_INVAL) {
126  break;
127  }
128  if ((iRow - r.mStartRow) & 1) {
129  rowHit = CALINK_INVAL;
130  break; // SG!!! - jump over the row
131  }
132 
133  cahit2 hh = CA_TEXTURE_FETCH(cahit22, gAliTexRefu2, tracker.HitData(row), r.mCurrIH);
134 
135  int32_t seedIH = r.mCurrIH;
136  r.mCurrIH = CA_TEXTURE_FETCH(calink, gAliTexRefs, tracker.HitLinkUpData(row), r.mCurrIH);
137 
138  float x = row.X();
139  float y = y0 + hh.x * stepY;
140  float z = z0 + hh.y * stepZ;
141  if (iRow != r.mStartRow || !tracker.Param().par.continuousTracking) {
142  tParam.ConstrainZ(z, tracker.ISlice(), z0, r.mLastZ);
143 #if !defined(__OPENCL__) || defined(__OPENCLCPP__)
144  tracker.GetConstantMem()->calibObjects.fastTransformHelper->TransformXYZ(tracker.ISlice(), iRow, x, y, z);
145 #endif
146  }
147  if (iRow == r.mStartRow) {
148  if (tracker.Param().par.continuousTracking) {
149  float refZ = ((z > 0) ? tracker.Param().rec.tpc.defaultZOffsetOverR : -tracker.Param().rec.tpc.defaultZOffsetOverR) * x;
150 #if !defined(__OPENCL__) || defined(__OPENCLCPP__)
151  float zTmp = refZ;
152  tracker.GetConstantMem()->calibObjects.fastTransformHelper->TransformXYZ(tracker.ISlice(), iRow, x, y, zTmp);
153  z += zTmp - refZ; // Add zCorrection (=zTmp - refZ) to z, such that zOffset is set such, that transformed (z - zOffset) becomes refZ
154 #endif
155  tParam.SetZOffset(z - refZ);
156  tParam.SetZ(refZ);
157  r.mLastZ = refZ;
158  } else {
159  tParam.SetZ(z);
160  r.mLastZ = z;
161  tParam.SetZOffset(0.f);
162  }
163  tParam.SetX(x);
164  tParam.SetY(y);
165  r.mLastY = y;
166  CADEBUG(printf("Tracklet %5d: FIT INIT ROW %3d X %8.3f -", r.mISH, iRow, tParam.X()); for (int32_t i = 0; i < 5; i++) { printf(" %8.3f", tParam.Par()[i]); } printf(" -"); for (int32_t i = 0; i < 15; i++) { printf(" %8.3f", tParam.Cov()[i]); } printf("\n"));
167  } else {
168  float dx = x - tParam.X();
169  float dy, dz;
170  if (r.mNHits >= 10) {
171  dy = y - tParam.Y();
172  dz = z - tParam.Z();
173  } else {
174  dy = y - r.mLastY;
175  dz = z - r.mLastZ;
176  }
177  r.mLastY = y;
178  r.mLastZ = z;
179 
180  float ri = 1.f / CAMath::Sqrt(dx * dx + dy * dy);
181  if (iRow == r.mStartRow + 2) {
182  tParam.SetSinPhi(dy * ri);
183  tParam.SetSignCosPhi(dx);
184  tParam.SetDzDs(dz * ri);
185  float err2Y, err2Z;
186  tracker.GetErrors2Seeding(iRow, tParam, -1.f, err2Y, err2Z); // Use correct time
187  tParam.SetCov(0, err2Y);
188  tParam.SetCov(2, err2Z);
189  }
190  float sinPhi, cosPhi;
191  if (r.mNHits >= 10 && CAMath::Abs(tParam.SinPhi()) < GPUCA_MAX_SIN_PHI_LOW) {
192  sinPhi = tParam.SinPhi();
193  cosPhi = CAMath::Sqrt(1 - sinPhi * sinPhi);
194  } else {
195  sinPhi = dy * ri;
196  cosPhi = dx * ri;
197  }
198  CADEBUG(printf("%14s: FIT TRACK ROW %3d X %8.3f -", "", iRow, tParam.X()); for (int32_t i = 0; i < 5; i++) { printf(" %8.3f", tParam.Par()[i]); } printf(" -"); for (int32_t i = 0; i < 15; i++) { printf(" %8.3f", tParam.Cov()[i]); } printf("\n"));
199  if (!tParam.TransportToX(x, sinPhi, cosPhi, tracker.Param().bzCLight, GPUCA_MAX_SIN_PHI)) {
200  rowHit = CALINK_INVAL;
201  break;
202  }
203  CADEBUG(printf("%5s hits %3d: FIT PROP ROW %3d X %8.3f -", "", r.mNHits, iRow, tParam.X()); for (int32_t i = 0; i < 5; i++) { printf(" %8.3f", tParam.Par()[i]); } printf(" -"); for (int32_t i = 0; i < 15; i++) { printf(" %8.3f", tParam.Cov()[i]); } printf("\n"));
204  float err2Y, err2Z;
205  tracker.GetErrors2Seeding(iRow, tParam.GetZ(), sinPhi, tParam.GetDzDs(), -1.f, err2Y, err2Z); // TODO: Use correct time
206 
207  if (r.mNHits >= 10) {
208  const float sErr2 = tracker.Param().GetSystematicClusterErrorIFC2(x, tParam.GetY(), tParam.GetZ(), tracker.ISlice() >= 18);
209  err2Y += sErr2;
210  err2Z += sErr2;
211  const float kFactor = tracker.Param().rec.tpc.hitPickUpFactor * tracker.Param().rec.tpc.hitPickUpFactor * 3.5f * 3.5f;
212  float sy2 = kFactor * (tParam.Err2Y() + err2Y);
213  float sz2 = kFactor * (tParam.Err2Z() + err2Z);
214  if (sy2 > tracker.Param().rec.tpc.hitSearchArea2) {
215  sy2 = tracker.Param().rec.tpc.hitSearchArea2;
216  }
217  if (sz2 > tracker.Param().rec.tpc.hitSearchArea2) {
218  sz2 = tracker.Param().rec.tpc.hitSearchArea2;
219  }
220  dy = y - tParam.Y();
221  dz = z - tParam.Z();
222  if (dy * dy > sy2 || dz * dz > sz2) {
223  if (++r.mNMissed >= tracker.Param().rec.tpc.trackFollowingMaxRowGapSeed) {
224  r.mCurrIH = CALINK_INVAL;
225  }
226  rowHit = CALINK_INVAL;
227  break;
228  }
229  }
230 
231  if (!tParam.Filter(y, z, err2Y, err2Z, GPUCA_MAX_SIN_PHI_LOW)) {
232  rowHit = CALINK_INVAL;
233  break;
234  }
235  CADEBUG(printf("%14s: FIT FILT ROW %3d X %8.3f -", "", iRow, tParam.X()); for (int32_t i = 0; i < 5; i++) { printf(" %8.3f", tParam.Par()[i]); } printf(" -"); for (int32_t i = 0; i < 15; i++) { printf(" %8.3f", tParam.Cov()[i]); } printf("\n"));
236  }
237  rowHit = seedIH;
238  r.mNHitsEndRow = ++r.mNHits;
239  r.mLastRow = iRow;
240  r.mEndRow = iRow;
241  r.mNMissed = 0;
242  } while (0);
243 
244  /*printf("Extrapolate Row %d X %f Y %f Z %f SinPhi %f DzDs %f QPt %f", iRow, tParam.X(), tParam.Y(), tParam.Z(), tParam.SinPhi(), tParam.DzDs(), tParam.QPt());
245  for (int32_t i = 0;i < 15;i++) printf(" C%d=%6.2f", i, tParam.GetCov(i));
246  printf("\n");*/
247 
248  if (r.mCurrIH == CALINK_INVAL) {
249  r.mStage = 1;
250  r.mLastY = tParam.Y(); // Store last spatial position here to start inward following from here
251  r.mLastZ = tParam.Z();
252  if (CAMath::Abs(tParam.SinPhi()) > GPUCA_MAX_SIN_PHI) {
253  r.mGo = 0;
254  }
255  }
256  } else { // forward/backward searching part
257  do {
258  if (r.mStage == 2 && iRow > r.mEndRow) {
259  break;
260  }
261  if (r.mNMissed > tracker.Param().rec.tpc.trackFollowingMaxRowGap) {
262  r.mGo = 0;
263  break;
264  }
265 
266  r.mNMissed++;
267 
268  float x = row.X();
269 #if !defined(__OPENCL__) || defined(__OPENCLCPP__)
270  {
271  float tmpY, tmpZ;
272  if (!tParam.GetPropagatedYZ(tracker.Param().bzCLight, x, tmpY, tmpZ)) {
273  r.mGo = 0;
274  rowHit = CALINK_INVAL;
275  break;
276  }
277  tParam.ConstrainZ(tmpZ, tracker.ISlice(), z0, r.mLastZ);
278  tracker.GetConstantMem()->calibObjects.fastTransformHelper->InverseTransformYZtoX(tracker.ISlice(), iRow, tmpY, tmpZ, x);
279  }
280 #endif
281 
282  CADEBUG(printf("%14s: SEA TRACK ROW %3d X %8.3f -", "", iRow, tParam.X()); for (int32_t i = 0; i < 5; i++) { printf(" %8.3f", tParam.Par()[i]); } printf(" -"); for (int32_t i = 0; i < 15; i++) { printf(" %8.3f", tParam.Cov()[i]); } printf("\n"));
283  if (!tParam.TransportToX(x, tParam.SinPhi(), tParam.GetCosPhi(), tracker.Param().bzCLight, GPUCA_MAX_SIN_PHI_LOW)) {
284  r.mGo = 0;
285  rowHit = CALINK_INVAL;
286  break;
287  }
288  CADEBUG(printf("%14s: SEA PROP ROW %3d X %8.3f -", "", iRow, tParam.X()); for (int32_t i = 0; i < 5; i++) { printf(" %8.3f", tParam.Par()[i]); } printf(" -"); for (int32_t i = 0; i < 15; i++) { printf(" %8.3f", tParam.Cov()[i]); } printf("\n"));
289 
290  bool found = false;
291  float yUncorrected = tParam.GetY(), zUncorrected = tParam.GetZ();
292  do {
293  if (row.NHits() < 1) {
294  rowHit = CALINK_INVAL;
295  break;
296  }
297 
298 #ifndef GPUCA_TEXTURE_FETCH_CONSTRUCTOR
299  GPUglobalref() const cahit2* hits = tracker.HitData(row);
300  GPUglobalref() const calink* firsthit = tracker.FirstHitInBin(row);
301 #endif
302 #if !defined(__OPENCL__) || defined(__OPENCLCPP__)
303  tracker.GetConstantMem()->calibObjects.fastTransformHelper->InverseTransformYZtoNominalYZ(tracker.ISlice(), iRow, yUncorrected, zUncorrected, yUncorrected, zUncorrected);
304 #endif
305 
306  if (tracker.Param().rec.tpc.rejectEdgeClustersInSeeding && tracker.Param().rejectEdgeClusterByY(yUncorrected, iRow, CAMath::Sqrt(tParam.Err2Y()))) {
307  rowHit = CALINK_INVAL;
308  break;
309  }
310  calink best = CALINK_INVAL;
311 
312  float err2Y, err2Z;
313  tracker.GetErrors2Seeding(iRow, *((MEM_LG2(GPUTPCTrackParam)*)&tParam), -1.f, err2Y, err2Z); // TODO: Use correct time
314  if (r.mNHits >= 10) {
315  const float sErr2 = tracker.Param().GetSystematicClusterErrorIFC2(x, tParam.GetY(), tParam.GetZ(), tracker.ISlice() >= 18);
316  err2Y += sErr2;
317  err2Z += sErr2;
318  }
319  if (CAMath::Abs(yUncorrected) < x * MEM_GLOBAL(GPUTPCRow)::getTPCMaxY1X()) { // search for the closest hit
320  const float kFactor = tracker.Param().rec.tpc.hitPickUpFactor * tracker.Param().rec.tpc.hitPickUpFactor * 7.0f * 7.0f;
321  const float maxWindow2 = tracker.Param().rec.tpc.hitSearchArea2;
322  const float sy2 = CAMath::Min(maxWindow2, kFactor * (tParam.Err2Y() + err2Y));
323  const float sz2 = CAMath::Min(maxWindow2, kFactor * (tParam.Err2Z() + err2Z));
324 
325  int32_t bin, ny, nz;
326  row.Grid().GetBinArea(yUncorrected, zUncorrected + tParam.ZOffset(), CAMath::Sqrt(sy2), CAMath::Sqrt(sz2), bin, ny, nz);
327  float ds = 1e6f;
328 
329 #ifdef __HIPCC__ // Todo: fixme!
330  for (int32_t k = -1; ++k <= nz; /*k++*/) {
331 #else
332  for (int32_t k = 0; k <= nz; k++) {
333 #endif
334  int32_t nBinsY = row.Grid().Ny();
335  int32_t mybin = bin + k * nBinsY;
336  uint32_t hitFst = CA_TEXTURE_FETCH(calink, gAliTexRefu, firsthit, mybin);
337  uint32_t hitLst = CA_TEXTURE_FETCH(calink, gAliTexRefu, firsthit, mybin + ny + 1);
338 #ifdef __HIPCC__ // Todo: fixme!
339  for (uint32_t ih = hitFst - 1; ++ih < hitLst; /*ih++*/) {
340 #else
341  for (uint32_t ih = hitFst; ih < hitLst; ih++) {
342 #endif
343  cahit2 hh = CA_TEXTURE_FETCH(cahit2, gAliTexRefu2, hits, ih);
344  float y = y0 + hh.x * stepY;
345  float z = z0 + hh.y * stepZ;
346  float dy = y - yUncorrected;
347  float dz = z - zUncorrected;
348  if (dy * dy < sy2 && dz * dz < sz2) {
349  float dds = tracker.Param().rec.tpc.trackFollowingYFactor * CAMath::Abs(dy) + CAMath::Abs(dz);
350  if (dds < ds) {
351  ds = dds;
352  best = ih;
353  }
354  }
355  }
356  }
357  } // end of search for the closest hit
358 
359  if (best == CALINK_INVAL) {
360  if (r.mNHits == 0 && r.mStage < 3) {
361  if (rowHit == CALINK_INVAL || rowHit == CALINK_DEAD_CHANNEL) {
362  break;
363  }
364  best = rowHit;
365  } else {
366  rowHit = CALINK_INVAL;
367  break;
368  }
369  }
370 
371  cahit2 hh = CA_TEXTURE_FETCH(cahit2, gAliTexRefu2, hits, best);
372  float y = y0 + hh.x * stepY + tParam.GetY() - yUncorrected;
373  float z = z0 + hh.y * stepZ + tParam.GetZ() - zUncorrected;
374 
375  CADEBUG(printf("%14s: SEA Hit %5d (%8.3f %8.3f), Res %f %f\n", "", best, y, z, tParam.Y() - y, tParam.Z() - z));
376 
377  calink oldHit = (r.mStage == 2 && iRow >= r.mStartRow) ? rowHit : CALINK_INVAL;
378  if (oldHit != best && !tParam.Filter(y, z, err2Y, err2Z, GPUCA_MAX_SIN_PHI_LOW, oldHit != CALINK_INVAL) && r.mNHits != 0) {
379  rowHit = CALINK_INVAL;
380  break;
381  }
382  found = true;
383  rowHit = best;
384  r.mNHits++;
385  r.mNMissed = 0;
386  CADEBUG(printf("%5s hits %3d: SEA FILT ROW %3d X %8.3f -", "", r.mNHits, iRow, tParam.X()); for (int32_t i = 0; i < 5; i++) { printf(" %8.3f", tParam.Par()[i]); } printf(" -"); for (int32_t i = 0; i < 15; i++) { printf(" %8.3f", tParam.Cov()[i]); } printf("\n"));
387  if (r.mStage == 1) {
388  r.mLastRow = iRow;
389  } else {
390  r.mFirstRow = iRow;
391  }
392  } while (false);
393  (void)found;
394 #if defined(GPUCA_HAVE_O2HEADERS) && (!defined(__OPENCL__) || defined(__OPENCLCPP__))
395  if (!found && tracker.GetConstantMem()->calibObjects.dEdxCalibContainer) {
396  uint32_t pad = CAMath::Float2UIntRn(tracker.Param().tpcGeometry.LinearY2Pad(tracker.ISlice(), iRow, yUncorrected));
397  if (pad < tracker.Param().tpcGeometry.NPads(iRow) && tracker.GetConstantMem()->calibObjects.dEdxCalibContainer->isDead(tracker.ISlice(), iRow, pad)) {
398  r.mNMissed--;
399  rowHit = CALINK_DEAD_CHANNEL;
400  }
401  }
402 #endif
403  } while (0);
404  }
405  if (r.mNHits == 8 && r.mNMissed == 0 && rowHit != CALINK_INVAL && rowHit != CALINK_DEAD_CHANNEL && rowHits && tracker.Param().par.continuousTracking && rowHits[r.mFirstRow] != CALINK_INVAL && rowHits[r.mFirstRow] != CALINK_DEAD_CHANNEL && rowHits[r.mLastRow] != CALINK_INVAL && rowHits[r.mLastRow] != CALINK_DEAD_CHANNEL) {
406  const GPUglobalref() MEM_GLOBAL(GPUTPCRow) & GPUrestrict() row1 = tracker.Row(r.mFirstRow);
407  const GPUglobalref() MEM_GLOBAL(GPUTPCRow) & GPUrestrict() row2 = tracker.Row(r.mLastRow);
408  GPUglobalref() const cahit2* hits1 = tracker.HitData(row1);
409  GPUglobalref() const cahit2* hits2 = tracker.HitData(row2);
410  const cahit2 hh1 = CA_TEXTURE_FETCH(cahit2, gAliTexRefu2, hits1, rowHits[r.mFirstRow]);
411  const cahit2 hh2 = CA_TEXTURE_FETCH(cahit2, gAliTexRefu2, hits2, rowHits[r.mLastRow]);
