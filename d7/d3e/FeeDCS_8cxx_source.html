25bool FeeDCS::isSMactive(int iSM)
26{
27
28 // assert(iSM>19 && "SM index larger than 19!");
29
30 if (iSM == 10 || iSM == 19) { // SMs 10 and 19 have 1 DDL each
31 return isDDLactive(2 * iSM);
32 } else {
33 return (isDDLactive(2 * iSM) && isDDLactive(2 * iSM + 1));
34 }
35
36 return false;
