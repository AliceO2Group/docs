47int LookUp::groupFinder(int nRow, int nCol)
48{
49 int row_index = nRow / its3::TopologyDictionary::RowClassSpan;
50 if (nRow % TopologyDictionary::RowClassSpan == 0) {
51 row_index--;
52 }
53 int col_index = nCol / its3::TopologyDictionary::ColClassSpan;
54 if (nCol % TopologyDictionary::RowClassSpan == 0) {
55 col_index--;
56 }
57 int grNum = -1;
58 if (row_index > its3::TopologyDictionary::MaxNumberOfRowClasses || col_index > its3::TopologyDictionary::MaxNumberOfColClasses) {
59 grNum = its3::TopologyDictionary::NumberOfRareGroups - 1;
60 } else {
61 grNum = row_index * its3::TopologyDictionary::MaxNumberOfColClasses + col_index;
62 }
63 return grNum;
64}

66int LookUp::findGroupID(int nRow, int nCol, bool IB, const unsigned char patt[itsmft::ClusterPattern::MaxPatternBytes]) const
67{
68 const auto& data = (IB) ? mDictionary.mDataIB : mDictionary.mDataOB;
69 int nBits = nRow * nCol;
70 if (nBits < 9) { // Small unique topology
71 int ID = data.mSmallTopologiesLUT[(nCol - 1) * 255 + (int)patt[0]];
72 if (ID >= 0) {
73 return ID;
74 }
75 } else { // Big unique topology
76 unsigned long hash = itsmft::ClusterTopology::getCompleteHash(nRow, nCol, patt);
77 auto ret = data.mCommonMap.find(hash);
78 if (ret != data.mCommonMap.end()) {
79 return ret->second;
80 }
81 }
82 if (!data.mGroupMap.empty()) { // rare valid topology group
83 int index = groupFinder(nRow, nCol);
84 auto res = data.mGroupMap.find(index);
85 return res == data.mGroupMap.end() ? itsmft::CompCluster::InvalidPatternID : res->second;
86 }
87 return itsmft::CompCluster::InvalidPatternID;
88}
