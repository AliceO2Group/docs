48 void Tracker::clustersToTracks(std::function<void(std::string s)> logger, std::function<void(std::string s)> error)
49 {
50  double total{0};
51  mTraits->UpdateTrackingParameters(mTrkParams);
52  int maxNvertices{-1};
53  if (mTrkParams[0].PerPrimaryVertexProcessing) {
54  for (int iROF{0}; iROF < mTimeFrame->getNrof(); ++iROF) {
55  maxNvertices = std::max(maxNvertices, (int)mTimeFrame->getPrimaryVertices(iROF).size());
56  }
57  }
58 
59  for (int iteration = 0; iteration < (int)mTrkParams.size(); ++iteration) {
60  if (iteration == 3 && mTrkParams[0].DoUPCIteration) {
61  mTimeFrame->swapMasks();
62  }
63  logger(fmt::format("ITS Tracking iteration {} summary:", iteration));
64  double timeTracklets{0.}, timeCells{0.}, timeNeighbours{0.}, timeRoads{0.};
65  int nTracklets{0}, nCells{0}, nNeighbours{0}, nTracks{-static_cast<int>(mTimeFrame->getNumberOfTracks())};
66 
67  total += evaluateTask(&Tracker::initialiseTimeFrame, "Timeframe initialisation", logger, iteration);
68  int nROFsIterations = mTrkParams[iteration].nROFsPerIterations > 0 ? mTimeFrame->getNrof() / mTrkParams[iteration].nROFsPerIterations + bool(mTimeFrame->getNrof() % mTrkParams[iteration].nROFsPerIterations) : 1;
69  int iVertex{std::min(maxNvertices, 0)};
70 
71  do {
72  for (int iROFs{0}; iROFs < nROFsIterations; ++iROFs) {
73  timeTracklets += evaluateTask(
74  &Tracker::computeTracklets, "Tracklet finding", [](std::string) {}, iteration, iROFs, iVertex);
75  nTracklets += mTraits->getTFNumberOfTracklets();
76  if (!mTimeFrame->checkMemory(mTrkParams[iteration].MaxMemory)) {
77  error(fmt::format("Too much memory used during trackleting in iteration {}, check the detector status and/or the selections.", iteration));
78  break;
79  }
80  float trackletsPerCluster = mTraits->getTFNumberOfClusters() > 0 ? float(mTraits->getTFNumberOfTracklets()) / mTraits->getTFNumberOfClusters() : 0.f;
81  if (trackletsPerCluster > mTrkParams[iteration].TrackletsPerClusterLimit) {
82  error(fmt::format("Too many tracklets per cluster ({}) in iteration {}, check the detector status and/or the selections. Current limit is {}", trackletsPerCluster, iteration, mTrkParams[iteration].TrackletsPerClusterLimit));
83  break;
84  }
85 
86  timeCells += evaluateTask(
87  &Tracker::computeCells, "Cell finding", [](std::string) {}, iteration);
88  nCells += mTraits->getTFNumberOfCells();
89  if (!mTimeFrame->checkMemory(mTrkParams[iteration].MaxMemory)) {
90  error(fmt::format("Too much memory used during cell finding in iteration {}, check the detector status and/or the selections.", iteration));
91  break;
92  }
93  float cellsPerCluster = mTraits->getTFNumberOfClusters() > 0 ? float(mTraits->getTFNumberOfCells()) / mTraits->getTFNumberOfClusters() : 0.f;
94  if (cellsPerCluster > mTrkParams[iteration].CellsPerClusterLimit) {
95  error(fmt::format("Too many cells per cluster ({}) in iteration {}, check the detector status and/or the selections. Current limit is {}", cellsPerCluster, iteration, mTrkParams[iteration].CellsPerClusterLimit));
96  break;
97  }
98 
99  timeNeighbours += evaluateTask(
100  &Tracker::findCellsNeighbours, "Neighbour finding", [](std::string) {}, iteration);
101  nNeighbours += mTimeFrame->getNumberOfNeighbours();
102  timeRoads += evaluateTask(
103  &Tracker::findRoads, "Road finding", [](std::string) {}, iteration);
104  }
105  iVertex++;
106  } while (iVertex < maxNvertices);
107  logger(fmt::format(" - Tracklet finding: {} tracklets found in {:.2f} ms", nTracklets, timeTracklets));
108  logger(fmt::format(" - Cell finding: {} cells found in {:.2f} ms", nCells, timeCells));
109  logger(fmt::format(" - Neighbours finding: {} neighbours found in {:.2f} ms", nNeighbours, timeNeighbours));
110  logger(fmt::format(" - Track finding: {} tracks found in {:.2f} ms", nTracks + mTimeFrame->getNumberOfTracks(), timeRoads));
111  total += timeTracklets + timeCells + timeNeighbours + timeRoads;
112  if (mTrkParams[iteration].UseTrackFollower) {
113  int nExtendedTracks{-mTimeFrame->mNExtendedTracks}, nExtendedClusters{-mTimeFrame->mNExtendedUsedClusters};
114  auto timeExtending = evaluateTask(&Tracker::extendTracks, "Extending tracks", [](const std::string&) {}, iteration);
115  total += timeExtending;
116  logger(fmt::format(" - Extending Tracks: {} extended tracks using {} clusters found in {:.2f} ms", nExtendedTracks + mTimeFrame->mNExtendedTracks, nExtendedClusters + mTimeFrame->mNExtendedUsedClusters, timeExtending));
117  }
118  }
119 
120  total += evaluateTask(&Tracker::findShortPrimaries, "Short primaries finding", logger);
121 
122  std::stringstream sstream;
123  if constexpr (constants::DoTimeBenchmarks) {
124  sstream << std::setw(2) << " - "
125  << "Timeframe " << mTimeFrameCounter++ << " processing completed in: " << total << "ms using " << mTraits->getNThreads() << " threads.";
126  }
127  logger(sstream.str());
128 
129  if (mTimeFrame->hasMCinformation()) {
130  computeTracksMClabels();
131  }
132  rectifyClusterIndices();
133  mNumberOfRuns++;
134 }

209  if (mTimeFrame->hasMCinformation()) {
210  computeTracksMClabels();
211  }
212  rectifyClusterIndices();
213  mNumberOfRuns++;

286 void Tracker::computeRoadsMClabels()
287 {
289  if (!mTimeFrame->hasMCinformation()) {
290  return;
291  }
292 
293  mTimeFrame->initialiseRoadLabels();
294 
295  int roadsNum{static_cast<int>(mTimeFrame->getRoads().size())};
296 
297  for (int iRoad{0}; iRoad < roadsNum; ++iRoad) {
298 
299  Road<5>& currentRoad{mTimeFrame->getRoads()[iRoad]};
300  std::vector<std::pair<MCCompLabel, size_t>> occurrences;
301  bool isFakeRoad{false};
302  bool isFirstRoadCell{true};
303 
304  for (int iCell{0}; iCell < mTrkParams[0].CellsPerRoad(); ++iCell) {
305  const int currentCellIndex{currentRoad[iCell]};
306 
307  if (currentCellIndex == constants::its::UnusedIndex) {
308  if (isFirstRoadCell) {
309  continue;
310  } else {
311  break;
312  }
313  }
314 
315  const CellSeed& currentCell{mTimeFrame->getCells()[iCell][currentCellIndex]};
316 
317  if (isFirstRoadCell) {
318 
319  const int cl0index{mTimeFrame->getClusters()[iCell][currentCell.getFirstClusterIndex()].clusterId};
320  auto cl0labs{mTimeFrame->getClusterLabels(iCell, cl0index)};
321  bool found{false};
322  for (size_t iOcc{0}; iOcc < occurrences.size(); ++iOcc) {
323  std::pair<o2::MCCompLabel, size_t>& occurrence = occurrences[iOcc];
324  for (auto& label : cl0labs) {
325  if (label == occurrence.first) {
326  ++occurrence.second;
327  found = true;
328  // break; // uncomment to stop to the first hit
329  }
330  }
331  }
332  if (!found) {
333  for (auto& label : cl0labs) {
334  occurrences.emplace_back(label, 1);
335  }
336  }
337 
338  const int cl1index{mTimeFrame->getClusters()[iCell + 1][currentCell.getSecondClusterIndex()].clusterId};
339 
340  const auto& cl1labs{mTimeFrame->getClusterLabels(iCell + 1, cl1index)};
341  found = false;
342  for (size_t iOcc{0}; iOcc < occurrences.size(); ++iOcc) {
343  std::pair<o2::MCCompLabel, size_t>& occurrence = occurrences[iOcc];
344  for (auto& label : cl1labs) {
345  if (label == occurrence.first) {
346  ++occurrence.second;
347  found = true;
348  // break; // uncomment to stop to the first hit
349  }
350  }
351  }
352  if (!found) {
353  for (auto& label : cl1labs) {
354  occurrences.emplace_back(label, 1);
355  }
356  }
357 
358  isFirstRoadCell = false;
359  }
360 
361  const int cl2index{mTimeFrame->getClusters()[iCell + 2][currentCell.getThirdClusterIndex()].clusterId};
362  const auto& cl2labs{mTimeFrame->getClusterLabels(iCell + 2, cl2index)};
363  bool found{false};
364  for (size_t iOcc{0}; iOcc < occurrences.size(); ++iOcc) {
365  std::pair<o2::MCCompLabel, size_t>& occurrence = occurrences[iOcc];
366  for (auto& label : cl2labs) {
367  if (label == occurrence.first) {
368  ++occurrence.second;
369  found = true;
370  // break; // uncomment to stop to the first hit
371  }
372  }
373  }
374  if (!found) {
375  for (auto& label : cl2labs) {
376  occurrences.emplace_back(label, 1);
377  }
378  }
379  }
380 
381  std::sort(occurrences.begin(), occurrences.end(), [](auto e1, auto e2) {
382  return e1.second > e2.second;
383  });
384 
385  auto maxOccurrencesValue = occurrences[0].first;
386  mTimeFrame->setRoadLabel(iRoad, maxOccurrencesValue.getRawValue(), isFakeRoad);
387  }
388 }

390 void Tracker::computeTracksMClabels()
391 {
392  for (int iROF{0}; iROF < mTimeFrame->getNrof(); ++iROF) {
393  for (auto& track : mTimeFrame->getTracks(iROF)) {
394  std::vector<std::pair<MCCompLabel, size_t>> occurrences;
395  occurrences.clear();
396 
397  for (int iCluster = 0; iCluster < TrackITSExt::MaxClusters; ++iCluster) {
398  const int index = track.getClusterIndex(iCluster);
399  if (index == constants::its::UnusedIndex) {
400  continue;
401  }
402  auto labels = mTimeFrame->getClusterLabels(iCluster, index);
403  bool found{false};
404  for (size_t iOcc{0}; iOcc < occurrences.size(); ++iOcc) {
405  std::pair<o2::MCCompLabel, size_t>& occurrence = occurrences[iOcc];
406  for (auto& label : labels) {
407  if (label == occurrence.first) {
408  ++occurrence.second;
409  found = true;
410  // break; // uncomment to stop to the first hit
411  }
412  }
413  }
414  if (!found) {
415  for (auto& label : labels) {
416  occurrences.emplace_back(label, 1);
417  }
418  }
419  }
420  std::sort(std::begin(occurrences), std::end(occurrences), [](auto e1, auto e2) {
421  return e1.second > e2.second;
422  });
423 
424  auto maxOccurrencesValue = occurrences[0].first;
425  uint32_t pattern = track.getPattern();
426  // set fake clusters pattern
427  for (int ic{TrackITSExt::MaxClusters}; ic--;) {
428  auto clid = track.getClusterIndex(ic);
429  if (clid != constants::its::UnusedIndex) {
430  auto labelsSpan = mTimeFrame->getClusterLabels(ic, clid);
431  for (auto& currentLabel : labelsSpan) {
432  if (currentLabel == maxOccurrencesValue) {
433  pattern |= 0x1 << (16 + ic); // set bit if correct
434  break;
435  }
436  }
437  }
438  }
439  track.setPattern(pattern);
440  if (occurrences[0].second < track.getNumberOfClusters()) {
441  maxOccurrencesValue.setFakeFlag();
442  }
443  mTimeFrame->getTracksLabel(iROF).emplace_back(maxOccurrencesValue);
444  }
445  }
446 }

462 void Tracker::getGlobalConfiguration()
463 {
464  auto& tc = o2::its::TrackerParamConfig::Instance();
465  tc.printKeyValues(true, true);
466  if (tc.useMatCorrTGeo) {
467  mTraits->setCorrType(o2::base::PropagatorImpl<float>::MatCorrType::USEMatCorrTGeo);
468  } else if (tc.useFastMaterial) {
469  mTraits->setCorrType(o2::base::PropagatorImpl<float>::MatCorrType::USEMatCorrNONE);
470  } else {
471  mTraits->setCorrType(o2::base::PropagatorImpl<float>::MatCorrType::USEMatCorrLUT);
472  }
473  setNThreads(tc.nThreads);
474  int nROFsPerIterations = tc.nROFsPerIterations > 0 ? tc.nROFsPerIterations : -1;
475  if (tc.nOrbitsPerIterations > 0) {
477  }
478  for (auto& params : mTrkParams) {
479  if (params.NLayers == 7) {
480  for (int i{0}; i < 7; ++i) {
481  params.SystErrorY2[i] = tc.sysErrY2[i] > 0 ? tc.sysErrY2[i] : params.SystErrorY2[i];
482  params.SystErrorZ2[i] = tc.sysErrZ2[i] > 0 ? tc.sysErrZ2[i] : params.SystErrorZ2[i];
483  }
484  }
485  params.DeltaROF = tc.deltaRof;
486  params.DoUPCIteration = tc.doUPCIteration;
487  params.MaxChi2ClusterAttachment = tc.maxChi2ClusterAttachment > 0 ? tc.maxChi2ClusterAttachment : params.MaxChi2ClusterAttachment;
488  params.MaxChi2NDF = tc.maxChi2NDF > 0 ? tc.maxChi2NDF : params.MaxChi2NDF;
489  params.PhiBins = tc.LUTbinsPhi > 0 ? tc.LUTbinsPhi : params.PhiBins;
490  params.ZBins = tc.LUTbinsZ > 0 ? tc.LUTbinsZ : params.ZBins;
491  params.PVres = tc.pvRes > 0 ? tc.pvRes : params.PVres;
492  params.NSigmaCut *= tc.nSigmaCut > 0 ? tc.nSigmaCut : 1.f;
493  params.CellDeltaTanLambdaSigma *= tc.deltaTanLres > 0 ? tc.deltaTanLres : 1.f;
494  params.TrackletMinPt *= tc.minPt > 0 ? tc.minPt : 1.f;
495  params.nROFsPerIterations = nROFsPerIterations;
496  params.PerPrimaryVertexProcessing = tc.perPrimaryVertexProcessing;
497  params.SaveTimeBenchmarks = tc.saveTimeBenchmarks;
498  for (int iD{0}; iD < 3; ++iD) {
499  params.Diamond[iD] = tc.diamondPos[iD];
500  }
501  params.UseDiamond = tc.useDiamond;
502  if (tc.maxMemory) {
503  params.MaxMemory = tc.maxMemory;
504  }
505  if (tc.useTrackFollower > 0) {
506  params.UseTrackFollower = true;
507  // Bit 0: Allow for mixing of top&bot extension --> implies Bits 1&2 set
508  // Bit 1: Allow for top extension
509  // Bit 2: Allow for bot extension
510  params.UseTrackFollowerMix = ((tc.useTrackFollower & (1 << 0)) != 0);
511  params.UseTrackFollowerTop = ((tc.useTrackFollower & (1 << 1)) != 0);
512  params.UseTrackFollowerBot = ((tc.useTrackFollower & (1 << 2)) != 0);
513  params.TrackFollowerNSigmaCutZ = tc.trackFollowerNSigmaZ;
514  params.TrackFollowerNSigmaCutPhi = tc.trackFollowerNSigmaPhi;
515  }
516  if (tc.cellsPerClusterLimit >= 0) {
517  params.CellsPerClusterLimit = tc.cellsPerClusterLimit;
518  }
519  if (tc.trackletsPerClusterLimit >= 0) {
520  params.TrackletsPerClusterLimit = tc.trackletsPerClusterLimit;
521  }
522  if (tc.findShortTracks >= 0) {
523  params.FindShortTracks = tc.findShortTracks;
524  }
525  }
