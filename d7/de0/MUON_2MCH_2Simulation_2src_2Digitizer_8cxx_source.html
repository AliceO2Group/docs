48 size_t Digitizer::digitize(std::vector<ROFRecord>& rofs,
49  std::vector<Digit>& digits,
50  dataformats::MCLabelContainer& labels)
51 {
52  // digitize every DE and store digits and labels ordered per IR
53  size_t nPileup = 0;
54  std::map<InteractionRecord, DEDigitizer::DigitsAndLabels> irDigitsAndLabels{};
55  for (auto& d : mDEDigitizers) {
56  nPileup += d.second->digitize(irDigitsAndLabels);
57  }
58 
59  // fill the external containers
60  for (const auto& [ir, digitsAndLabels] : irDigitsAndLabels) {
61  rofs.emplace_back(ROFRecord(ir, digits.size(), digitsAndLabels.first.size()));
62  digits.insert(digits.end(), digitsAndLabels.first.begin(), digitsAndLabels.first.end());
63  labels.mergeAtBack(digitsAndLabels.second);
64  }
65 
66  return nPileup;
67 }
