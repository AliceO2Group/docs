56STUDecoderError::ErrorCode_t STUDecoderError::intToErrorCode(int errorcode)
57{
58 static constexpr std::size_t NUMERRORCODES = 8;
59 static constexpr std::array<ErrorCode_t, NUMERRORCODES> errorcodes = {{ErrorCode_t::PAGE_ERROR, ErrorCode_t::WORD_UNEXPECTED, ErrorCode_t::INDEX_UNEXPECTED, ErrorCode_t::ADC_OVERFLOW, ErrorCode_t::FEEID_UNEXPECTED, ErrorCode_t::OLD_PAYLOAD_VERSION, ErrorCode_t::FULL_PAYLOAD_SIZE_UNEXPECTED, ErrorCode_t::SHORT_PAYLOAD_SIZE_UNEXPECTED}};
60 if (errorcode < 0 || errorcode >= NUMERRORCODES) {
61 return ErrorCode_t::UNKNOWN;
62 }
63 return errorcodes[errorcode];
64}
