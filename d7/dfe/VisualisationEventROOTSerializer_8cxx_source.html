189 bool VisualisationEventROOTSerializer::fromFile(VisualisationEvent& event, std::string fileName)
190 {
191  LOGF(info, "VisualisationEventROOTSerializer <- ", fileName);
192  event.mTracks.clear();
193  event.mClusters.clear();
194  event.mCalo.clear();
195 
196  TFile f(fileName.c_str());
197 
198  event.setRunNumber(readInt(f, "runNumber"));
199  event.setRunType(static_cast<parameters::GRPECS::RunType>(readInt(f, "runType")));
200  event.setClMask(readInt(f, "clMask"));
201  event.setTrkMask(readInt(f, "trkMask"));
202  event.setTfCounter(readInt(f, "tfCounter"));
203  event.setFirstTForbit(readInt(f, "firstTForbit"));
204  event.mPrimaryVertex = readInt(f, "primaryVertex");
205 
206  event.setCollisionTime(readString(f, "collisionTime"));
207  event.mEveVersion = readString(f, "eveVersion");
208  event.mWorkflowParameters = readString(f, "workflowParameters");
209 
210  // xyz
211  TNtuple* xyz = (TNtuple*)f.Get("xyz");
212  if (xyz == nullptr) {
213  return false;
214  }
215 
216  // tracks
217  TTree* tracks = (TTree*)f.Get("tracks");
218  if (tracks == nullptr) {
219  delete xyz;
220  return false;
221  }
222 
223  long track_xyz; // first track point
224  int track_points; // number of track points
225  int track_clusters; // number of track clusters
226  int track_source;
227  std::string* track_GID = nullptr;
228  float track_time;
229  int track_charge;
230  float track_theta;
231  float track_phi;
232  float track_eta;
233  int track_PID;
234 
235  tracks->SetBranchAddress("xyz", &track_xyz);
236  tracks->SetBranchAddress("time", &track_time);
237  tracks->SetBranchAddress("charge", &track_charge);
238  tracks->SetBranchAddress("theta", &track_theta);
239  tracks->SetBranchAddress("phi", &track_phi);
240  tracks->SetBranchAddress("eta", &track_eta);
241  tracks->SetBranchAddress("PID", &track_PID);
242  tracks->SetBranchAddress("GID", &track_GID);
243  tracks->SetBranchAddress("source", &track_source);
244  tracks->SetBranchAddress("points", &track_points);
245  tracks->SetBranchAddress("clusters", &track_clusters);
246 
247  Int_t tracksNoOfEntries = (Int_t)tracks->GetEntries();
248  for (Int_t i = 0; i < tracksNoOfEntries; i++) {
249  tracks->GetEntry(i);
250  VisualisationTrack track;
251  track.mTime = track_time;
252  track.mCharge = track_charge;
253  track.mTheta = track_theta;
254  track.mPhi = track_phi;
255  track.mEta = track_eta;
256  track.mPID = track_PID;
257  track.mGID = *track_GID;
258  track.mSource = (o2::dataformats::GlobalTrackID::Source)track_source;
259  xyz->GetEntry(track_xyz);
260  track.addStartCoordinates(xyz->GetArgs());
261  for (size_t i = 0; i < track_points; i++) {
262  xyz->GetEntry(track_xyz + 1 + i);
263  track.addPolyPoint(xyz->GetArgs());
264  }
265  for (size_t i = 0; i < track_clusters; i++) {
266  xyz->GetEntry(track_xyz + 1 + track_points + i);
267  VisualisationCluster cluster(xyz->GetArgs(), track.mTime);
268  cluster.mSource = track.mSource;
269  track.mClusters.emplace_back(cluster);
270  }
271  event.mTracks.emplace_back(track);
272  }
273  if (track_GID != nullptr) {
274  delete track_GID;
275  track_GID = nullptr;
276  }
277 
278  TTree* clusters = (TTree*)f.Get("clusters");
279  if (clusters == nullptr) {
280  delete xyz;
281  delete tracks;
282  return false;
283  }
284 
285  long cluster_xyz;
286  int cluster_source;
287  float cluster_time;
288  clusters->SetBranchAddress("source", &cluster_source);
289  clusters->SetBranchAddress("time", &cluster_time);
290  clusters->SetBranchAddress("xyz", &cluster_xyz);
291  Int_t clustersNoOfEntries = (Int_t)clusters->GetEntries();
292  for (Int_t i = 0; i < clustersNoOfEntries; i++) {
293  clusters->GetEntry(i);
294  xyz->GetEntry(cluster_xyz);
295  VisualisationCluster cluster(xyz->GetArgs(), cluster_time);
296  cluster.mSource = (o2::dataformats::GlobalTrackID::Source)cluster_source;
297  event.mClusters.emplace_back(cluster);
298  }
299 
300  // calorimeters
301  TTree* calo = (TTree*)f.Get("calo");
302  if (calo == nullptr) {
303  delete xyz;
304  delete tracks;
305  delete clusters;
306  return false;
307  }
308  int calo_source;
309  float calo_time;
310  float calo_energy;
311  float calo_eta;
312  float calo_phi;
313  std::string* calo_GID = nullptr;
314  int calo_PID;
315 
316  calo->SetBranchAddress("source", &calo_source);
317  calo->SetBranchAddress("time", &calo_time);
318  calo->SetBranchAddress("energy", &calo_energy);
319  calo->SetBranchAddress("eta", &calo_eta);
320  calo->SetBranchAddress("phi", &calo_phi);
321  calo->SetBranchAddress("GID", &calo_GID);
322  calo->SetBranchAddress("PID", &calo_PID);
323 
324  Int_t nentries = (Int_t)calo->GetEntries();
325  for (Int_t i = 0; i < nentries; i++) {
326  calo->GetEntry(i);
327  VisualisationCalo calorimeter;
328  calorimeter.mSource = (o2::dataformats::GlobalTrackID::Source)calo_source;
329  calorimeter.mTime = calo_time;
330  calorimeter.mEnergy = calo_energy;
331  calorimeter.mEta = calo_eta;
332  calorimeter.mPhi = calo_phi;
333  if (calo_GID) {
334  calorimeter.mGID = *calo_GID;
335  }
336  calorimeter.mPID = calo_PID;
337  event.mCalo.emplace_back(calorimeter);
338  }
339  if (calo_GID != nullptr) {
340  delete calo_GID;
341  calo_GID = nullptr;
342  }
343  delete calo;
344  delete tracks;
345  delete xyz;
346  delete clusters;
347  event.afterLoading();
348  return true;
349 }
