40MEM_CLASS_PRE()
41GPUd() void MEM_LG(GPUTPCGrid)::Create(float yMin, float yMax, float zMin, float zMax, int32_t ny, int32_t nz)
42{
43 //* Create the grid
44 mYMin = yMin;
45 mZMin = zMin;
46
47 float sy = CAMath::Max((yMax + 0.1f - yMin) / ny, GPUCA_MIN_BIN_SIZE);
48 float sz = CAMath::Max((zMax + 0.1f - zMin) / nz, GPUCA_MIN_BIN_SIZE);
49
50 mStepYInv = 1.f / sy;
51 mStepZInv = 1.f / sz;
52
53 mNy = ny;
54 mNz = nz;
55
56 mN = mNy * mNz;
57
58 mYMax = mYMin + mNy * sy;
59 mZMax = mZMin + mNz * sz;
60}

62MEM_CLASS_PRE()
63GPUd() int32_t MEM_LG(GPUTPCGrid)::GetBin(float Y, float Z) const
64{
65 //* get the bin pointer
66 const int32_t yBin = static_cast<int32_t>((Y - mYMin) * mStepYInv);
67 const int32_t zBin = static_cast<int32_t>((Z - mZMin) * mStepZInv);
68 const int32_t bin = zBin * mNy + yBin;
69#ifndef GPUCA_GPUCODE
70 assert(bin >= 0);
71 assert(bin < static_cast<int32_t>(mN));
72#endif
73 return bin;
74}

77GPUd() int32_t MEM_LG(GPUTPCGrid)::GetBinBounded(float Y, float Z) const

118GPUd() void MEM_LG(GPUTPCGrid)::GetBinArea(float Y, float Z, float dy, float dz, int32_t& bin, int32_t& ny, int32_t& nz) const
119{
120 Y -= mYMin;
121 int32_t by = (int32_t)((Y - dy) * mStepYInv);
122 ny = (int32_t)((Y + dy) * mStepYInv) - by;
123 Z -= mZMin;
124 int32_t bz = (int32_t)((Z - dz) * mStepZInv);
125 nz = (int32_t)((Z + dz) * mStepZInv) - bz;
126 if (by >= (int32_t)mNy) {
127 by = mNy - 1;
128 }
129 if (by < 0) {
130 by = 0;
131 }
132 if (bz >= (int32_t)mNz) {
133 bz = mNz - 1;
134 }
135 if (bz < 0) {
136 bz = 0;
137 }
138 if (by + ny >= (int32_t)mNy) {
139 ny = mNy - 1 - by;
140 }
141 if (bz + nz >= (int32_t)mNz) {
142 nz = mNz - 1 - bz;
143 }
144 bin = bz * mNy + by;
145}
