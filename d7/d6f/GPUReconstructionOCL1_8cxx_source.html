87bool GPUReconstructionOCL1Backend::CheckPlatform(uint32_t i)
88{
89 char platform_version[64] = {}, platform_vendor[64] = {};
90 clGetPlatformInfo(mInternals->platforms[i], CL_PLATFORM_VERSION, sizeof(platform_version), platform_version, nullptr);
91 clGetPlatformInfo(mInternals->platforms[i], CL_PLATFORM_VENDOR, sizeof(platform_vendor), platform_vendor, nullptr);
92 if (strcmp(platform_vendor, "Advanced Micro Devices, Inc.") == 0 && strstr(platform_version, "OpenCL 2.0 AMD-APP (") != nullptr) {
93 float ver = 0;
94 sscanf(platform_version, "OpenCL 2.0 AMD-APP (%f)", &ver);
95 if (ver < 2000.f) {
96 if (mProcessingSettings.debugLevel >= 2) {
97 GPUInfo("AMD APP OpenCL Platform found");
98 }
99 return true;
100 }
101 }
102 return false;
103}
