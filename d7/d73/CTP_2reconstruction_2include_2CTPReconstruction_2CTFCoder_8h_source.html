71 o2::ctf::CTFIOSize CTFCoder::encode(VEC& buff, const gsl::span<const CTPDigit>& data, const LumiInfo& lumi)
72 {
73  if (mIRFrameSelector.isSet()) { // preselect data
74  mDataFilt.clear();
75  for (const auto& trig : data) {
76  if (mIRFrameSelector.check(trig.intRecord) >= 0) {
77  mDataFilt.push_back(trig);
78  }
79  }
80  return encode_impl(buff, mDataFilt, lumi);
81  }
82  return encode_impl(buff, data, lumi);

85 template <typename VEC>
86 o2::ctf::CTFIOSize CTFCoder::encode_impl(VEC& buff, const gsl::span<const CTPDigit>& data, const LumiInfo& lumi)
87 {
88  using MD = o2::ctf::Metadata::OptStore;
89  // what to do which each field: see o2::ctd::Metadata explanation
90  constexpr MD optField[CTF::getNBlocks()] = {
91  MD::EENCODE_OR_PACK, // BLC_bcIncTrig
92  MD::EENCODE_OR_PACK, // BLC_orbitIncTrig
93  MD::EENCODE_OR_PACK, // BLC_bytesInput
94  MD::EENCODE_OR_PACK, // BLC_bytesClass
95  };
96 
97  CTFHelper helper(data);
98 
99  // book output size with some margin
100  auto szIni = sizeof(CTFHeader) + helper.getSize() * 2. / 3; // will be autoexpanded if needed
101  buff.resize(szIni);
102 
103  auto ec = CTF::create(buff);
104  using ECB = CTF::base;
105 
106  ec->setHeader(helper.createHeader(lumi));
107  assignDictVersion(static_cast<o2::ctf::CTFDictHeader&>(ec->getHeader()));
108  ec->setANSHeader(mANSVersion);
109  // at every encoding the buffer might be autoexpanded, so we don't work with fixed pointer ec
110  o2::ctf::CTFIOSize iosize;
111 #define ENCODECTP(beg, end, slot, bits) CTF::get(buff.data())->encode(beg, end, int(slot), bits, optField[int(slot)], &buff, mCoders[int(slot)], getMemMarginFactor());
112  // clang-format off
113  iosize += ENCODECTP(helper.begin_bcIncTrig(), helper.end_bcIncTrig(), CTF::BLC_bcIncTrig, 0);
114  iosize += ENCODECTP(helper.begin_orbitIncTrig(), helper.end_orbitIncTrig(), CTF::BLC_orbitIncTrig, 0);
115  iosize += ENCODECTP(helper.begin_bytesInput(), helper.end_bytesInput(), CTF::BLC_bytesInput, 0);
116  iosize += ENCODECTP(helper.begin_bytesClass(), helper.end_bytesClass(), CTF::BLC_bytesClass, 0);
117  // clang-format on
118  CTF::get(buff.data())->print(getPrefix(), mVerbosity);
119  finaliseCTFOutput<CTF>(buff);
120  iosize.rawIn = data.size() * sizeof(CTPDigit);
121  return iosize;
122 }

125 template <typename VTRG>
126 o2::ctf::CTFIOSize CTFCoder::decode(const CTF::base& ec, VTRG& data, LumiInfo& lumi)
127 {
128  auto header = ec.getHeader();
129  checkDictVersion(static_cast<const o2::ctf::CTFDictHeader&>(header));
130  ec.print(getPrefix(), mVerbosity);
131  std::vector<int16_t> bcInc;
132  std::vector<int32_t> orbitInc;
133  std::vector<uint8_t> bytesInput, bytesClass;
134 
135  o2::ctf::CTFIOSize iosize;
136 #define DECODECTP(part, slot) ec.decode(part, int(slot), mCoders[int(slot)])
137  // clang-format off
138  iosize += DECODECTP(bcInc, CTF::BLC_bcIncTrig);
139  iosize += DECODECTP(orbitInc, CTF::BLC_orbitIncTrig);
140  iosize += DECODECTP(bytesInput, CTF::BLC_bytesInput);
141  iosize += DECODECTP(bytesClass, CTF::BLC_bytesClass);
142  // clang-format on
143  //
144  data.clear();
145  std::map<o2::InteractionRecord, CTPDigit> digitsMap;
146  o2::InteractionRecord ir(header.firstBC, header.firstOrbit);
147  lumi.nHBFCounted = header.lumiNHBFs;
148  lumi.nHBFCountedFV0 = header.lumiNHBFsFV0 ? header.lumiNHBFsFV0 : header.lumiNHBFs;
149  lumi.counts = header.lumiCounts;
150  lumi.countsFV0 = header.lumiCountsFV0;
151  lumi.orbit = header.lumiOrbit;
152  lumi.inp1 = int(header.inp1);
153  lumi.inp2 = int(header.inp2);
154  auto itInp = bytesInput.begin();
155  auto itCls = bytesClass.begin();
156  bool checkIROK = (mBCShift == 0); // need to check if CTP offset correction does not make the local time negative ?
157  bool checkIROKInputs = (mBCShiftInputs == 0); // need to check if CTP offset correction does not make the local time negative ?
158  for (uint32_t itrig = 0; itrig < header.nTriggers; itrig++) {
159  // restore TrigRecord
160  if (orbitInc[itrig]) { // non-0 increment => new orbit
161  ir.bc = bcInc[itrig]; // bcInc has absolute meaning
162  ir.orbit += orbitInc[itrig];
163  } else {
164  ir.bc += bcInc[itrig];
165  }
166  if (checkIROKInputs || canApplyBCShiftInputs(ir)) { // correction will be ok
167  auto irs = ir - mBCShiftInputs;
168  uint64_t CTPInputMask = 0;
169  for (int i = 0; i < CTFHelper::CTPInpNBytes; i++) {
170  CTPInputMask |= static_cast<uint64_t>(*itInp++) << (8 * i);
171  }
172  if (CTPInputMask) {
173  if (digitsMap.count(irs)) {
174  if (digitsMap[irs].isInputEmpty()) {
175  digitsMap[irs].CTPInputMask = CTPInputMask;
176  // LOG(info) << "IR1:";
177  // digitsMap[irs].printStream(std::cout);
178  } else {
179  LOG(error) << "CTPInpurMask already exist:" << irs << " dig.CTPInputMask:" << digitsMap[irs].CTPInputMask << " CTPInputMask:" << CTPInputMask;
180  }
181  } else {
182  CTPDigit dig = {irs, CTPInputMask, 0};
183  digitsMap[irs] = dig;
184  // LOG(info) << "IR2:";
185  // digitsMap[irs].printStream(std::cout);
186  }
187  }
188  } else { // correction would make IR prior to mFirstTFOrbit, skip
189  itInp += CTFHelper::CTPInpNBytes;
190  }
191  if (checkIROK || canApplyBCShift(ir)) { // correction will be ok
192  auto irs = ir - mBCShift;
193  uint64_t CTPClassMask = 0;
194  for (int i = 0; i < CTFHelper::CTPClsNBytes; i++) {
195  CTPClassMask |= static_cast<uint64_t>(*itCls++) << (8 * i);
196  }
197  if (CTPClassMask) {
198  if (digitsMap.count(irs)) {
199  if (digitsMap[irs].isClassEmpty()) {
200  digitsMap[irs].CTPClassMask = CTPClassMask;
201  // LOG(info) << "TCM1:";
202  // digitsMap[irs].printStream(std::cout);
203  } else {
204  LOG(error) << "CTPClassMask already exist:" << irs << " dig.CTPClassMask:" << digitsMap[irs].CTPClassMask << " CTPClassMask:" << CTPClassMask;
205  }
206  } else {
207  CTPDigit dig = {irs, 0, CTPClassMask};
208  digitsMap[irs] = dig;
209  // LOG(info) << "TCM2:";
210  // digitsMap[irs].printStream(std::cout);
211  }
212  }
213  } else { // correction would make IR prior to mFirstTFOrbit, skip
214  itCls += CTFHelper::CTPClsNBytes;
215  }
216  }
217  if (mDecodeInps) {
218  o2::pmr::vector<CTPDigit> digits;
219  o2::ctp::RawDataDecoder::shiftInputs(digitsMap, digits, mFirstTFOrbit);
220  for (auto const& dig : digits) {
221  data.emplace_back(dig);
222  }
223  } else {
224  for (auto const& dig : digitsMap) {
225  data.emplace_back(dig.second);
226  }
227  }
228  assert(itInp == bytesInput.end());
229  assert(itCls == bytesClass.end());
230  iosize.rawIn = header.nTriggers * sizeof(CTPDigit);
231  return iosize;
232 }
