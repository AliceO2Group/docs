113 static constexpr bool isSameOrigin(o2::header::DataOrigin origin, ID id)
114 {
115 for (auto i = sizeof(o2::header::DataOrigin); i--;) {
116 if (MAP_DAQtoO2[id].str[i] != origin.str[i]) {
117 return false;
118 }
119 }
120 return true;
121 }

123 static constexpr ID or2daq(o2::header::DataOrigin origin, ID id)
124 {
125 return (id > MAXDAQ) ? (isSameOrigin(origin, UNLOADED) ? UNLOADED : INVALID) : (isSameOrigin(origin, id) ? id : or2daq(origin, id + 1));
