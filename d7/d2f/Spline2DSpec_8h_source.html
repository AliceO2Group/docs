248 template <SafetyLevel SafeT = SafetyLevel::kSafe>
249 GPUd() void interpolateUold(int32_t inpYdim, GPUgeneric() const DataT Parameters[],
250 DataT u1, DataT u2, GPUgeneric() DataT S[/*inpYdim*/]) const
251 {
252
253 const auto nYdimTmp = SplineUtil::getNdim<YdimT>(inpYdim);
254 const int32_t nYdim = nYdimTmp.get();
255
256 const auto maxYdim = SplineUtil::getMaxNdim<YdimT>(inpYdim);
257 const int32_t maxYdim4 = 4 * maxYdim.get();
258
259 const auto nYdim2 = nYdim * 2;
260 const auto nYdim4 = nYdim * 4;
261
262 const DataT& u = u1;
263 const DataT& v = u2;
264 int32_t nu = mGridX1.getNumberOfKnots();
265 int32_t iu = mGridX1.template getLeftKnotIndexForU<SafeT>(u);
266 int32_t iv = mGridX2.template getLeftKnotIndexForU<SafeT>(v);
267
268 const typename TBase::Knot& knotU = mGridX1.template getKnot<SafetyLevel::kNotSafe>(iu);
269 const typename TBase::Knot& knotV = mGridX2.template getKnot<SafetyLevel::kNotSafe>(iv);
270
271 const DataT* par00 = Parameters + (nu * iv + iu) * nYdim4; // values { {Y1,Y2,Y3}, {Y1,Y2,Y3}'v, {Y1,Y2,Y3}'u, {Y1,Y2,Y3}''vu } at {u0, v0}
272 const DataT* par10 = par00 + nYdim4; // values { ... } at {u1, v0}
273 const DataT* par01 = par00 + nYdim4 * nu; // values { ... } at {u0, v1}
274 const DataT* par11 = par01 + nYdim4; // values { ... } at {u1, v1}
275
276 DataT Su0[maxYdim4]; // values { {Y1,Y2,Y3,Y1'v,Y2'v,Y3'v}(v0), {Y1,Y2,Y3,Y1'v,Y2'v,Y3'v}(v1) }, at u0
277 DataT Du0[maxYdim4]; // derivatives {}'_u at u0
278 DataT Su1[maxYdim4]; // values { {Y1,Y2,Y3,Y1'v,Y2'v,Y3'v}(v0), {Y1,Y2,Y3,Y1'v,Y2'v,Y3'v}(v1) }, at u1
279 DataT Du1[maxYdim4]; // derivatives {}'_u at u1
280
281 for (int32_t i = 0; i < nYdim2; i++) {
282 Su0[i] = par00[i];
283 Su0[nYdim2 + i] = par01[i];
284
285 Du0[i] = par00[nYdim2 + i];
286 Du0[nYdim2 + i] = par01[nYdim2 + i];
287
288 Su1[i] = par10[i];
289 Su1[nYdim2 + i] = par11[i];
290
291 Du1[i] = par10[nYdim2 + i];
292 Du1[nYdim2 + i] = par11[nYdim2 + i];
293 }
294
295 DataT parU[maxYdim4]; // interpolated values { {Y1,Y2,Y3,Y1'v,Y2'v,Y3'v}(v0), {Y1,Y2,Y3,Y1'v,Y2'v,Y3'v}(v1) } at u
296
297 typedef Spline1DSpec<DataT, 4 * YdimT, 0> TGridX1;
298 const TGridX1& gridX1 = reinterpret_cast<const TGridX1&>(mGridX1);
299
300 gridX1.interpolateU(nYdim4, knotU, Su0, Du0, Su1, Du1, u, parU);
301
302 const DataT* Sv0 = parU + 0;
303 const DataT* Dv0 = parU + nYdim;
304 const DataT* Sv1 = parU + nYdim2;
305 const DataT* Dv1 = parU + nYdim2 + nYdim;
306
307 typedef Spline1DSpec<DataT, YdimT, 0> TGridX2;
308 const TGridX2& gridX2 = reinterpret_cast<const TGridX2&>(mGridX2);
309 gridX2.interpolateU(nYdim, knotV, Sv0, Dv0, Sv1, Dv1, v, S);
310 }

313 template <SafetyLevel SafeT = SafetyLevel::kSafe>
314 GPUd() void interpolateU(int32_t inpYdim, GPUgeneric() const DataT Parameters[],
315 DataT u1, DataT u2, GPUgeneric() DataT S[/*inpYdim*/]) const
316 {
317
318 const auto nYdimTmp = SplineUtil::getNdim<YdimT>(inpYdim);
319 const int32_t nYdim = nYdimTmp.get();
320
321 // const auto maxYdim = SplineUtil::getMaxNdim<YdimT>(inpYdim);
322 // const int32_t maxYdim4 = 4 * maxYdim.get();
323
324 // const auto nYdim2 = nYdim * 2;
325 const auto nYdim4 = nYdim * 4;
326
327 const DataT& u = u1;
328 const DataT& v = u2;
329 int32_t nu = mGridX1.getNumberOfKnots();
330 int32_t iu = mGridX1.template getLeftKnotIndexForU<SafeT>(u);
331 int32_t iv = mGridX2.template getLeftKnotIndexForU<SafeT>(v);
332
333 const typename TBase::Knot& knotU = mGridX1.template getKnot<SafetyLevel::kNotSafe>(iu);
334 const typename TBase::Knot& knotV = mGridX2.template getKnot<SafetyLevel::kNotSafe>(iv);
335
336 const DataT* A = Parameters + (nu * iv + iu) * nYdim4; // values { {Y1,Y2,Y3}, {Y1,Y2,Y3}'v, {Y1,Y2,Y3}'u, {Y1,Y2,Y3}''vu } at {u0, v0}
337 const DataT* B = A + nYdim4 * nu; // values { ... } at {u0, v1}
338
339 DataT dSl, dDl, dSr, dDr;
340 mGridX1.getUderivatives(knotU, u, dSl, dDl, dSr, dDr);
341 DataT dSd, dDd, dSu, dDu;
342 mGridX2.getUderivatives(knotV, v, dSd, dDd, dSu, dDu);
343
344 // when nYdim == 1:
345 // S = dSl * (dSd * A[0] + dDd * A[1]) + dDl * (dSd * A[2] + dDd * A[3]) +
346 // dSr * (dSd * A[4] + dDd * A[5]) + dDr * (dSd * A[6] + dDd * A[7]) +
347 // dSl * (dSu * B[0] + dDu * B[1]) + dDl * (dSu * B[2] + dDu * B[3]) +
348 // dSr * (dSu * B[4] + dDu * B[5]) + dDr * (dSu * B[6] + dDu * B[7]);
349
350 DataT a[8] = {dSl * dSd, dSl * dDd, dDl * dSd, dDl * dDd,
351 dSr * dSd, dSr * dDd, dDr * dSd, dDr * dDd};
352 DataT b[8] = {dSl * dSu, dSl * dDu, dDl * dSu, dDl * dDu,
353 dSr * dSu, dSr * dDu, dDr * dSu, dDr * dDu};
354
355 // S = sum a[i]*A[i] + b[i]*B[i]
356
357 for (int32_t dim = 0; dim < nYdim; dim++) {
358 S[dim] = 0;
359 for (int32_t i = 0; i < 8; i++) {
360 S[dim] += a[i] * A[nYdim * i + dim] + b[i] * B[nYdim * i + dim];
361 }
362 }
363 }
364
365 protected:
366 using TBase::mGridX1;
367 using TBase::mGridX2;
368 using TBase::mParameters;
369 using TBase::mYdim;
370 using TBase::TBase; // inherit constructors and hide them
371};
