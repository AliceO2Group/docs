56 struct BBox {
57 uint16_t chipID = 0xffff;
58 uint16_t rowMin = 0xffff, colMin = 0xffff;
59 uint16_t rowMax = 0, colMax = 0;
60 explicit BBox(uint16_t c) : chipID(c) {}
61 bool isInside(uint16_t r, uint16_t c) const { return r >= rowMin && r <= rowMax && c >= colMin && c <= colMax; }
62 uint16_t rowSpan() const { return rowMax - rowMin + 1; }
63 uint16_t colSpan() const { return colMax - colMin + 1; }
64 bool isAcceptableSize() const
65 {
66 return rowSpan() <= o2::itsmft::ClusterPattern::MaxRowSpan &&
67 colSpan() <= o2::itsmft::ClusterPattern::MaxColSpan;
68 }
69 void adjust(uint16_t r, uint16_t c)
70 {
71 if (r < rowMin) {
72 rowMin = r;
73 }
74 if (r > rowMax) {
75 rowMax = r;
76 }
77 if (c < colMin) {
78 colMin = c;
79 }
80 if (c > colMax) {
81 colMax = c;
82 }
83 }
84 };
