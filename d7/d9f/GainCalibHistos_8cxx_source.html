37void GainCalibHistos::fill(const std::vector<int>& input)

48void GainCalibHistos::merge(const GainCalibHistos* prev)
49{
50 if (!mInitialized) {
51 init();
52 }
53 for (int i = 0; i < MAXCHAMBER * NBINSGAINCALIB; ++i) {
