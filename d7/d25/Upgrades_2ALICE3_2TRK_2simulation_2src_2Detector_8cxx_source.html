124void Detector::configFromFile(std::string fileName)
125{
126 // Override the default geometry if config file provided
127 std::ifstream confFile(fileName);
128 if (!confFile.good()) {
129 LOGP(fatal, "File {} not found, aborting.", fileName);
130 }
131
132 mLayers.clear();
133
134 LOGP(info, "Overriding geometry of ALICE3 TRK using {} file.", fileName);
135
136 std::string line;
137 std::vector<float> tmpBuff;
138 int layerCount{0};
139 while (std::getline(confFile, line)) {
140 if (line[0] == '/') {
141 continue;
142 }
143 tmpBuff.clear();
144 std::stringstream ss(line);
145 float val;
146 std::string substr;
147 while (getline(ss, substr, '\t')) {
148 tmpBuff.push_back(std::stof(substr));
149 }
150 mLayers.emplace_back(layerCount, std::string{GeometryTGeo::getTRKLayerPattern() + std::to_string(layerCount)}, tmpBuff[0], tmpBuff[1], tmpBuff[2]);
151 ++layerCount;
152 }
153}

212{

278bool Detector::ProcessHits(FairVolume* vol)
279{
280 // This method is called from the MC stepping
281 if (!(fMC->TrackCharge())) {
282 return false;
283 }
284
285 int lay = vol->getVolumeId();
286 int volID = vol->getMCid();
287
288 // Is it needed to keep a track reference when the outer ITS volume is encountered?
289 auto stack = (o2::data::Stack*)fMC->GetStack();
290 if (fMC->IsTrackExiting() && (lay == 0 || lay == mLayers.size() - 1)) {
291 // Keep the track refs for the innermost and outermost layers only
292 o2::TrackReference tr(*fMC, GetDetId());
293 tr.setTrackID(stack->GetCurrentTrackNumber());
294 tr.setUserId(lay);
295 stack->addTrackReference(tr);
296 }
297 bool startHit = false, stopHit = false;
298 unsigned char status = 0;
299 if (fMC->IsTrackEntering()) {
300 status |= Hit::kTrackEntering;
301 }
302 if (fMC->IsTrackInside()) {
303 status |= Hit::kTrackInside;
304 }
305 if (fMC->IsTrackExiting()) {
306 status |= Hit::kTrackExiting;
307 }
308 if (fMC->IsTrackOut()) {
309 status |= Hit::kTrackOut;
310 }
311 if (fMC->IsTrackStop()) {
312 status |= Hit::kTrackStopped;
313 }
314 if (fMC->IsTrackAlive()) {
315 status |= Hit::kTrackAlive;
316 }
317
318 // track is entering or created in the volume
319 if ((status & Hit::kTrackEntering) || (status & Hit::kTrackInside && !mTrackData.mHitStarted)) {
320 startHit = true;
321 } else if ((status & (Hit::kTrackExiting | Hit::kTrackOut | Hit::kTrackStopped))) {
322 stopHit = true;
323 }
324
325 // increment energy loss at all steps except entrance
326 if (!startHit) {
327 mTrackData.mEnergyLoss += fMC->Edep();
328 }
329 if (!(startHit | stopHit)) {
330 return false; // do noting
331 }
332
333 if (startHit) {
334 mTrackData.mEnergyLoss = 0.;
335 fMC->TrackMomentum(mTrackData.mMomentumStart);
336 fMC->TrackPosition(mTrackData.mPositionStart);
337 mTrackData.mTrkStatusStart = status;
338 mTrackData.mHitStarted = true;
339 }
340 if (stopHit) {
341 TLorentzVector positionStop;
342 fMC->TrackPosition(positionStop);
343 // Retrieve the indices with the volume path
344 int stave(0), halfstave(0), chipinmodule(0), module;
345 fMC->CurrentVolOffID(1, chipinmodule);
346 fMC->CurrentVolOffID(2, module);
347 fMC->CurrentVolOffID(3, halfstave);
348 fMC->CurrentVolOffID(4, stave);
349
350 Hit* p = addHit(stack->GetCurrentTrackNumber(), lay, mTrackData.mPositionStart.Vect(), positionStop.Vect(),
351 mTrackData.mMomentumStart.Vect(), mTrackData.mMomentumStart.E(), positionStop.T(),
352 mTrackData.mEnergyLoss, mTrackData.mTrkStatusStart, status);
353 // p->SetTotalEnergy(vmc->Etot());
354
355 // RS: not sure this is needed
356 // Increment number of Detector det points in TParticle
357 stack->addHit(GetDetId());
358 }
359
360 return true;
361}
