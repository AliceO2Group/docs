35class TFDispatcher : public o2::framework::Task
36{
37 public:
38 TFDispatcher(int slot, int ngen, int nlanes, int latency) : mSlot(slot), mNGen(ngen), mNLanes(nlanes), mLatency(latency) {}
39
40 void init(o2::framework::InitContext& ic) final
41 {
42 mMaxTF = ic.options().get<int64_t>("max-timeframes");
43 }
44
45 void run(o2::framework::ProcessingContext& pc) final
46 {
47 for (auto& input : pc.inputs()) {
48 auto tfid = header::get<o2::framework::DataProcessingHeader*>(input.header)->startTime;
49 if (tfid >= mMaxTF - 1) {
50 LOG(info) << "Data generator reached TF " << tfid << ", stopping";
51 pc.services().get<o2::framework::ControlService>().endOfStream();
52 pc.services().get<o2::framework::ControlService>().readyToQuit(o2::framework::QuitRequest::Me);
53 if (!acceptTF(tfid)) {
54 return;
55 }
56 break;
57 }
58 if (!acceptTF(tfid)) {
59 return;
60 }
61 }
62 int size = 100 + gRandom->Integer(100); // push dummy output
63 usleep(mLatency);
64 pc.outputs().snapshot(o2::framework::OutputRef{"output", 0}, size);
65 }
66
67 bool acceptTF(int tfid)
68 {
69
70 // check if the current TF should be processed by this instance of the generator
71
72 int targetSlot = (tfid / mNLanes) % mNGen;
73 if (targetSlot != mSlot) {
74 LOG(info) << "tfid = " << tfid << ", mNLanes = " << mNLanes << ", mNGen = " << mNGen << ", mSlot = " << mSlot << " target slot = " << targetSlot << ": discarded";
75 return false;
76 }
77 LOG(info) << "tfid = " << tfid << ", mNLanes = " << mNLanes << ", mNGen = " << mNGen << ", mSlot = " << mSlot << " target slot = " << targetSlot << ": accepted";
78 return true;
79 }
80
81 private:
82 uint64_t mMaxTF = 1;
83 int mSlot = 1;
84 int mNGen = 1;
85 int mNLanes = 1;
86 int mLatency = 0;
87};

89class TFProcessor : public o2::framework::Task
90{
91 public:
92 TFProcessor(int latency, int latencyRMS) : mMeanLatency(latency), mLatencyRMS(latencyRMS){};
93
94 void init(o2::framework::InitContext& ic) final
95 {
96 mDevCopy = ic.services().get<const o2::framework::DeviceSpec>().inputTimesliceId;
97 gRandom->SetSeed(mDevCopy);
98 mZDCChannelCalib = ic.options().get<bool>("do-ZDC-channel-calib");
99 mZDCChannelCalibInTestMode = ic.options().get<bool>("do-ZDC-channel-calib-in-test-mode");
100 LOG(info) << "TFProcessorCopy: " << mDevCopy << " MeanLatency: " << mMeanLatency << " LatencyRMS: " << mLatencyRMS << " DoZDCChannelCalib: " << mZDCChannelCalib
101 << " DoZDCChannelCalibInTestMode: " << mZDCChannelCalibInTestMode;
102 }
103
104 void run(o2::framework::ProcessingContext& pc) final
105 {
106 auto tfcounter = o2::header::get<o2::framework::DataProcessingHeader*>(pc.inputs().get("input").header)->startTime;
107 // introduceDelay
108 uint32_t delay = std::abs(gRandom->Gaus(mMeanLatency, mLatencyRMS));
109 LOG(info) << "TFProcessorCopy: " << mDevCopy << " Simulate latency of " << delay << " mcs for TF " << tfcounter;
110 usleep(delay);
111
112 // push dummy output
113 auto size = pc.inputs().get<int>("input");
114 auto& output = pc.outputs().make<std::vector<o2::dataformats::CalibInfoZDC>>(o2::framework::OutputRef{"output", 0});
115 output.reserve(size);
116
117 for (int i = size; i--;) {
118 if (!mZDCChannelCalib) {
119 output.emplace_back(gRandom->Integer(32), 0, gRandom->Gaus(clockShift, 100.), 0, 0);
120 } else {
121 int channel = mZDCChannelCalibInTestMode ? gRandom->Integer(100) : gRandom->Integer(32);
122 double tot = gRandom->Gaus(12, 2); // in ns
123 output.emplace_back(channel, 0, value, tot, 0);
124 }
125 }
126 }
127
128 private:
129 int mDevCopy = 0;
130 uint32_t mMeanLatency = 0;
131 uint32_t mLatencyRMS = 1;
132 bool mZDCChannelCalib = false;
133 bool mZDCChannelCalibInTestMode = false;
134};
