30 bool CRUConfig::setValues(std::string_view cruData)
31 {
32  const auto cruDataV = Str::tokenize(cruData.data(), ',');
33  if (cruDataV.size() != CRUConfig::NConfigValues) {
34  LOGP(warning, "Wrong number of CRU config values {}/{} in line {}", cruDataV.size(), NConfigValues, cruData);
35  return false;
36  }
37 
38  linkOn = static_cast<decltype(linkOn)>(std::stol(cruDataV[0]));
39  cmcEnabled = static_cast<decltype(cmcEnabled)>(std::stol(cruDataV[1]));
40  zsOffset = static_cast<decltype(zsOffset)>(std::stol(cruDataV[4]));
41  itCorr0 = std::stof(cruDataV[5]);
42  itfEnabled = static_cast<decltype(itfEnabled)>(std::stol(cruDataV[2]));
43  zsEnabled = static_cast<decltype(zsEnabled)>(std::stol(cruDataV[3]));
44  resyncEnabled = static_cast<decltype(resyncEnabled)>(std::stol(cruDataV[6]));
45 
46  return true;

79 {
80  const auto nEnabled = std::accumulate(cruConfig.begin(), cruConfig.end(), size_t(0), [](size_t b, const auto& c) { return b + (c.cmcEnabled > 0); });

82  LOGP(warning, "CMC not enabled for all CRUs: {} < {}", nEnabled, cruConfig.size());

118 void FEEConfig::setAllLinksOn()
119 {
120  const auto& mapper = Mapper::instance();
121 
122  // ===| Check active link map |===
123  for (int iCRU = 0; iCRU < cruConfig.size(); ++iCRU) {
124  const CRU cru(iCRU);
125  const PartitionInfo& partInfo = mapper.getMapPartitionInfo()[cru.partition()];
126  const int nFECs = partInfo.getNumberOfFECs();
127  const int fecOffset = (nFECs + 1) / 2;
128  cruConfig.at(iCRU).linkOn = 0;
129  for (int iFEC = 0; iFEC < nFECs; ++iFEC) {
130  const int fecTest = (iFEC < fecOffset) ? iFEC : 10 + (iFEC - fecOffset);
131  const int fecBIT = 1 << fecTest;
132  cruConfig.at(iCRU).linkOn |= fecBIT;
133  }
134  }
135 }

137 void FEEConfig::print() const
138 {
139  fmt::print("\n");
140  const auto& mapper = Mapper::instance();
141 
142  auto message = fmt::format("Printing tag {} ({})", int(tag), TagNames.at(tag));
143  const size_t boxWidth = 80;
144  fmt::print(
145  "┌{0:─^{2}}┐\n"
146  "│{1: ^{2}}│\n"
147  "└{0:─^{2}}┘\n",
148  "", message, boxWidth);
149  fmt::print("\n");
150 
151  // ===| CRU summary |=========================================================
152  message = "| CRU summary |";
153  fmt::print("{0:=^{1}}\n", message, boxWidth);
154 
155  if (cruConfig.size() != CRU::MaxCRU) {
156  LOGP(error, "Unexpected size of cru config:{} != {}", cruConfig.size(), (int)CRU::MaxCRU);
157  } else {
158  for (int iCRU = 0; iCRU < cruConfig.size(); ++iCRU) {
159  const auto& c = cruConfig.at(iCRU);
160  const CRU cru(iCRU);
161  const PartitionInfo& partInfo = mapper.getMapPartitionInfo()[cru.partition()];
162  const int nLinks = partInfo.getNumberOfFECs();
163  const auto nLinkOn = std::bitset<32>(c.linkOn).count();
164 
165  fmt::print("CRU {:3d}: linkOn = {:5x} ({:2}/{:2}), cmcEn = {:5x}, zsOffset = {:5x}, itCorr0 = {:4.2f}, itfEn = {:b}, zsEn = {:b}, resyncEn = {:b}\n",
166  iCRU, c.linkOn, nLinkOn, nLinks, c.cmcEnabled, c.zsOffset, c.itCorr0, c.itfEnabled, c.zsEnabled, c.resyncEnabled);
167  }
168  }
169 
170  fmt::print("\n");
171  // ===| CRU summary |=========================================================
172  message = "| Pad maps summary |";
173  fmt::print("{0:=^{1}}\n", message, boxWidth);
174 
175  for ([[maybe_unused]] auto& [key, val] : padMaps) {
176  fmt::print("{}\n", key);
177  }
178 }

186 CalDet<bool> FEEConfig::getDeadChannelMap() const
187 {
188  const auto& mapper = Mapper::instance();
189  CalDet<bool> deadMap("DeadChannelMap");
190 
191  // ===| Check active link map |===
192  for (int iCRU = 0; iCRU < cruConfig.size(); ++iCRU) {
193  const CRU cru(iCRU);
194  const PartitionInfo& partInfo = mapper.getMapPartitionInfo()[cru.partition()];
195  const int nFECs = partInfo.getNumberOfFECs();
196  const int fecOffset = (nFECs + 1) / 2;
197  for (int iFEC = 0; iFEC < nFECs; ++iFEC) {
198  // const int fecInPartition = (iFEC < fecOffset) ? iFEC : 12 + (iFEC % 10);
199  const int fecTest = (iFEC < fecOffset) ? iFEC : 10 + (iFEC - fecOffset);
200  const int fecBIT = 1 << fecTest;
201  if (cruConfig.at(iCRU).linkOn & fecBIT) {
202  continue; // FEC is active
203  }
204 
205  // all channels of the FEC are masked
206  for (int iChannel = 0; iChannel < 80; ++iChannel) {
207  const auto& [sampaOnFEC, channelOnSAMPA] = cru_calib_helpers::getSampaInfo(iChannel, cru);
208  const PadROCPos padROCPos = mapper.padROCPos(cru, iFEC, sampaOnFEC, channelOnSAMPA);
209  deadMap.getCalArray(padROCPos.getROC()).setValue(padROCPos.getRow(), padROCPos.getPad(), true);
210  }
211  }
212  }
213 
214  // ===| Check pedestal values |===============================================
215  const auto& pedestals = padMaps.at(PadConfigNames.at(PadConfig::Pedestals));
216 
217  for (int iROC = 0; iROC < pedestals.getData().size(); ++iROC) {
218  const auto& rocPed = pedestals.getCalArray(iROC);
219  auto& rocDead = deadMap.getCalArray(iROC);
220 
221  for (int iPad = 0; iPad < rocPed.getData().size(); ++iPad) {
222  if (rocPed.getValue(iPad) > 1022) {
223  rocDead.setValue(iPad, true);
224  }
225  }
226  }
227 
228  return deadMap;
229 }
