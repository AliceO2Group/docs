79 template <class Mapping>
80 int RUDecodeData::decodeROF(const Mapping& mp, const o2::InteractionRecord ir, bool verifyDecoder)
81 {
82  nChipsFired = 0;
83  lastChipChecked = 0;
84  int ntot = 0;
85  // poll majority ROF IR to detect desynchronization between the liks
86 
87  std::array<bool, Mapping::getNChips()> doneChips{};
88  auto* chipData = &chipsData[0];
89  seenChipIDs.clear();
90  for (int icab = 0; icab < ruInfo->nCables; icab++) { // cableData is ordered in such a way to have chipIDs in increasing order
91  if (!cableData[icab].getSize()) {
92  continue;
93  }
94  if (!checkLinkInSync(icab, ir)) { // apparently there was desynchronization
95  continue;
96  }
97  auto cabHW = cableHWID[icab];
98  auto chIdGetter = [this, &mp, cabHW](int cid) {
99  // return mp.getGlobalChipID(cid, cabHW, *this->ruInfo);
100  auto chip = mp.getGlobalChipID(cid, cabHW, *this->ruInfo);
101  return chip;
102  };
103  auto localChipIdGetter = [this, &mp, cabHW](int gid) {
104  auto localID = mp.getLocalChipID(gid, cabHW, *this->ruInfo);
105  return localID;
106  };
107 
108  int ret = 0;
109  // dumpcabledata(icab);
110 
111  std::vector<uint16_t>* seenChipIDsPtr = &seenChipIDs;
112  if (verifyDecoder) {
113  seenChipIDsInCable.clear();
114  seenChipIDsPtr = &seenChipIDsInCable;
115  seenChips.clear();
116  }
117  while ((ret = AlpideCoder::decodeChip(*chipData, cableData[icab], *seenChipIDsPtr, chIdGetter)) || chipData->isErrorSet()) { // we register only chips with hits or errors flags set
118  setROFInfo(chipData, cableLinkPtr[icab]);
119  if (verifyDecoder) {
120  auto ID = chipData->getChipID();
121  // Since the errored chips are not recorded, we need to add them
122  // separately
123  if (seenChipIDsInCable.empty() || seenChipIDsInCable.back() != ID) {
124  seenChipIDsInCable.push_back(ID);
125  }
126  seenChips[ID] = chipData;
127  }
128  auto nhits = chipData->getData().size();
129  if (nhits && doneChips[chipData->getChipID()]) {
130  if (chipData->getChipID() == chipsData[nChipsFired - 1].getChipID()) {
131  LOGP(debug, "re-entry into the data of the chip {} after previously detector error", chipData->getChipID());
132  }
133 #ifdef ALPIDE_DECODING_STAT
134  else {
135  chipData->setError(ChipStat::InterleavedChipData);
136  }
137 #endif
138  ret = -1; // discard decoded data
139  nhits = 0;
140  }
141 #ifdef ALPIDE_DECODING_STAT
142  fillChipStatistics(icab, chipData);
143 #endif
144  if (nhits && chipData->getChipID() < Mapping::getNChips()) {
145  doneChips[chipData->getChipID()] = true;
146  ntot += nhits;
147  if (++nChipsFired < chipsData.size()) { // fetch next free chip
148  chipData = &chipsData[nChipsFired];
149  } else {
150  break; // last chip decoded
151  }
152  }
153  if (ret < 0) {
154  break; // negative code was returned by decoder: abandon cable data
155  }
156  }
157  if (verifyDecoder) {
158  AlpideCoder::verifyDecodedCable(seenChips, cableData[icab],
159  seenChipIDsInCable, localChipIdGetter,
160  chIdGetter);
161  seenChipIDs.insert(seenChipIDs.end(), seenChipIDsInCable.begin(), seenChipIDsInCable.end());
162  }
163  cableData[icab].clear();
164  }
165  return ntot;
166 }
