96 void EntropyEncoderSpec::run(ProcessingContext& pc)
97 {
98  if (mSelIR) {
99  GRPGeomHelper::instance().checkUpdates(pc);
100  if (GRPGeomHelper::instance().getGRPECS()->isDetReadOut(o2::detectors::DetID::TPC) && mConfParam->tpcTriggeredMode ^ !GRPGeomHelper::instance().getGRPECS()->isDetContinuousReadOut(o2::detectors::DetID::TPC)) {
101  LOG(fatal) << "configKeyValue tpcTriggeredMode does not match GRP isDetContinuousReadOut(TPC) setting";
102  }
103 
104  mConfig->configGRP.continuousMaxTimeBin = (GRPGeomHelper::instance().getGRPECS()->getNHBFPerTF() * o2::constants::lhc::LHCMaxBunches + 2 * o2::tpc::constants::LHCBCPERTIMEBIN - 2) / o2::tpc::constants::LHCBCPERTIMEBIN;
105  mConfig->configGRP.solenoidBzNominalGPU = GPUO2InterfaceUtils::getNominalGPUBz(*GRPGeomHelper::instance().getGRPMagField());
106  mParam->UpdateSettings(&mConfig->configGRP);
107 
108  mTPCVDriftHelper->extractCCDBInputs(pc);
109  if (mTPCVDriftHelper->isUpdated()) {
110  TPCFastTransformHelperO2::instance()->updateCalibration(*mFastTransform, 0, mTPCVDriftHelper->getVDriftObject().corrFact, mTPCVDriftHelper->getVDriftObject().refVDrift, mTPCVDriftHelper->getVDriftObject().getTimeOffset());
111  }
112  }
113 
114  mCTFCoder.updateTimeDependentParams(pc, true);
115 
116  CompressedClusters clusters;
117  static o2::tpc::detail::TriggerInfo trigComp;
118 
119  if (mFromFile) {
120  auto tmp = pc.inputs().get<CompressedClustersROOT*>("input");
121  if (tmp == nullptr) {
122  LOG(error) << "invalid input";
123  return;
124  }
125  clusters = *tmp;
126  } else {
127  auto tmp = pc.inputs().get<CompressedClustersFlat*>("input");
128  if (tmp == nullptr) {
129  LOG(error) << "invalid input";
130  return;
131  }
132  clusters = *tmp;
133  }
134  auto triggers = pc.inputs().get<gsl::span<o2::tpc::TriggerInfoDLBZS>>("trigger");
135  auto cput = mTimer.CpuTime();
136  mTimer.Start(false);
137  auto& buffer = pc.outputs().make<std::vector<o2::ctf::BufferType>>(Output{"TPC", "CTFDATA", 0});
138  std::vector<bool> rejectHits, rejectTracks, rejectTrackHits, rejectTrackHitsReduced;
139  CompressedClusters clustersFiltered = clusters;
140  std::vector<std::pair<std::vector<unsigned int>, std::vector<unsigned short>>> tmpBuffer(std::max<int>(mNThreads, 1));
141 
142  // prepare trigger info
143  trigComp.clear();
144  for (const auto& trig : triggers) {
145  for (int it = 0; it < o2::tpc::TriggerWordDLBZS::MaxTriggerEntries; it++) {
146  if (trig.triggerWord.isValid(it)) {
147  trigComp.deltaOrbit.push_back(trig.orbit);
148  trigComp.deltaBC.push_back(trig.triggerWord.getTriggerBC(it));
149  trigComp.triggerType.push_back(trig.triggerWord.getTriggerType(it));
150  } else {
151  break;
152  }
153  }
154  }
155 
156  if (mSelIR) {
157  if (clusters.nTracks && clusters.solenoidBz != -1e6f && clusters.solenoidBz != mParam->bzkG) {
158  throw std::runtime_error("Configured solenoid Bz does not match value used for track model encoding");
159  }
160  if (clusters.nTracks && clusters.maxTimeBin != -1e6 && clusters.maxTimeBin != mParam->par.continuousMaxTimeBin) {
161  throw std::runtime_error("Configured max time bin does not match value used for track model encoding");
162  }
163  mCTFCoder.setSelectedIRFrames(pc.inputs().get<gsl::span<o2::dataformats::IRFrame>>("selIRFrames"));
164  rejectHits.resize(clusters.nUnattachedClusters);
165  rejectTracks.resize(clusters.nTracks);
166  rejectTrackHits.resize(clusters.nAttachedClusters);
167  rejectTrackHitsReduced.resize(clusters.nAttachedClustersReduced);
168 
169  const auto& tinfo = pc.services().get<o2::framework::TimingInfo>();
170  const auto firstIR = o2::InteractionRecord(0, tinfo.firstTForbit);
171  const float totalT = std::max(mFastTransform->getMaxDriftTime(0), mFastTransform->getMaxDriftTime(GPUCA_NSLICES / 2));
172 
173  unsigned int offset = 0, lasti = 0;
174  const unsigned int maxTime = (mParam->par.continuousMaxTimeBin + 1) * o2::tpc::ClusterNative::scaleTimePacked - 1;
175 #ifdef WITH_OPENMP
176 #pragma omp parallel for firstprivate(offset, lasti) num_threads(mNThreads)
177 #endif
178  for (unsigned int i = 0; i < clusters.nTracks; i++) {
179  unsigned int tMinP = maxTime, tMaxP = 0;
180  auto checker = [&tMinP, &tMaxP](const o2::tpc::ClusterNative& cl, unsigned int offset) {
181  if (cl.getTimePacked() > tMaxP) {
182  tMaxP = cl.getTimePacked();
183  }
184  if (cl.getTimePacked() < tMinP) {
185  tMinP = cl.getTimePacked();
186  }
187  };
188  if (i < lasti) {
189  offset = lasti = 0; // dynamic OMP scheduling, need to reinitialize offset
190  }
191  while (lasti < i) {
192  offset += clusters.nTrackClusters[lasti++];
193  }
194  lasti++;
195  o2::gpu::TPCClusterDecompressor::decompressTrack(&clusters, *mParam, maxTime, i, offset, checker);
196  const float tMin = o2::tpc::ClusterNative::unpackTime(tMinP), tMax = o2::tpc::ClusterNative::unpackTime(tMaxP);
197  const auto chkVal = firstIR + (tMin * constants::LHCBCPERTIMEBIN);
198  const auto chkExt = totalT > tMax - tMin ? ((totalT - (tMax - tMin)) * constants::LHCBCPERTIMEBIN + 1) : 0;
199  const bool reject = mCTFCoder.getIRFramesSelector().check(o2::dataformats::IRFrame(chkVal, chkVal + 1), chkExt, 0) < 0;
200  if (reject) {
201  for (unsigned int k = offset - clusters.nTrackClusters[i]; k < offset; k++) {
202  rejectTrackHits[k] = true;
203  }
204  for (unsigned int k = offset - clusters.nTrackClusters[i] - i; k < offset - i - 1; k++) {
205  rejectTrackHitsReduced[k] = true;
206  }
207  rejectTracks[i] = true;
208  static std::atomic_flag lock = ATOMIC_FLAG_INIT;
209  while (lock.test_and_set(std::memory_order_acquire)) {
210  }
211  clustersFiltered.nTracks--;
212  clustersFiltered.nAttachedClusters -= clusters.nTrackClusters[i];
213  clustersFiltered.nAttachedClustersReduced -= clusters.nTrackClusters[i] - 1;
214  lock.clear(std::memory_order_release);
215  }
216  }
217  offset = 0;
218  unsigned int offsets[GPUCA_NSLICES][GPUCA_ROW_COUNT];
219  for (unsigned int i = 0; i < GPUCA_NSLICES; i++) {
220  for (unsigned int j = 0; j < GPUCA_ROW_COUNT; j++) {
221  if (i * GPUCA_ROW_COUNT + j >= clusters.nSliceRows) {
222  break;
223  }
224  offsets[i][j] = offset;
225  offset += (i * GPUCA_ROW_COUNT + j >= clusters.nSliceRows) ? 0 : clusters.nSliceRowClusters[i * GPUCA_ROW_COUNT + j];
226  }
227  }
228 
229 #ifdef WITH_OPENMP
230 #pragma omp parallel for num_threads(mNThreads) schedule(static, (GPUCA_NSLICES + mNThreads - 1) / mNThreads) // Static round-robin scheduling with one chunk per thread to ensure correct order of the final vector
231 #endif
232  for (unsigned int ii = 0; ii < clusters.nSliceRows; ii++) {
233  unsigned int i = ii / GPUCA_ROW_COUNT;
234  unsigned int j = ii % GPUCA_ROW_COUNT;
235  o2::tpc::ClusterNative preCl;
236 #ifdef WITH_OPENMP
237  int myThread = omp_get_thread_num();
238 #else
239  int myThread = 0;
240 #endif
241  unsigned int count = 0;
242  const float x = mParam->tpcGeometry.Row2X(j);
243  auto checker = [i, j, firstIR, totalT, x, this, &preCl, &count, &outBuffer = tmpBuffer[myThread], &rejectHits, &clustersFiltered](const o2::tpc::ClusterNative& cl, unsigned int k) {
244  const float y = mParam->tpcGeometry.LinearPad2Y(i, j, cl.getPad());
245  const float r = sqrtf(x * x + y * y);
246  const float maxz = r * mEtaFactor + mMaxZ;
247  const unsigned int deltaBC = std::max<float>(0.f, totalT - mFastTransform->convDeltaZtoDeltaTimeInTimeFrameAbs(maxz)) * constants::LHCBCPERTIMEBIN;
248  const auto chkVal = firstIR + (cl.getTime() * constants::LHCBCPERTIMEBIN) - deltaBC;
249  const auto chkExt = totalT * constants::LHCBCPERTIMEBIN - deltaBC;
250  const bool reject = mCTFCoder.getIRFramesSelector().check(o2::dataformats::IRFrame(chkVal, chkVal + 1), chkExt, 0) < 0;
251  if (reject) {
252  rejectHits[k] = true;
253  clustersFiltered.nSliceRowClusters[i * GPUCA_ROW_COUNT + j]--;
254  static std::atomic_flag lock = ATOMIC_FLAG_INIT;
255  while (lock.test_and_set(std::memory_order_acquire)) {
256  }
257  clustersFiltered.nUnattachedClusters--;
258  lock.clear(std::memory_order_release);
259  } else {
260  outBuffer.first.emplace_back(0);
261  outBuffer.second.emplace_back(0);
262  GPUTPCCompression_EncodeUnattached(clustersFiltered.nComppressionModes, cl, outBuffer.first.back(), outBuffer.second.back(), count++ ? &preCl : nullptr);
263  preCl = cl;
264  }
265  };
266  unsigned int end = offsets[i][j] + clusters.nSliceRowClusters[i * GPUCA_ROW_COUNT + j];
267  o2::gpu::TPCClusterDecompressor::decompressHits(&clusters, offsets[i][j], end, checker);
268  }
269  tmpBuffer[0].first.reserve(clustersFiltered.nUnattachedClusters);
270  tmpBuffer[0].second.reserve(clustersFiltered.nUnattachedClusters);
271  for (int i = 1; i < mNThreads; i++) {
272  tmpBuffer[0].first.insert(tmpBuffer[0].first.end(), tmpBuffer[i].first.begin(), tmpBuffer[i].first.end());
273  tmpBuffer[i].first.clear();
274  tmpBuffer[0].second.insert(tmpBuffer[0].second.end(), tmpBuffer[i].second.begin(), tmpBuffer[i].second.end());
275  tmpBuffer[i].second.clear();
276  }
277  clustersFiltered.timeDiffU = tmpBuffer[0].first.data();
278  clustersFiltered.padDiffU = tmpBuffer[0].second.data();
279  }
280 
281  // transform trigger info to differential form
282  uint32_t prevOrbit = -1;
283  uint16_t prevBC = -1;
284  if (trigComp.triggerType.size()) {
285  prevOrbit = trigComp.firstOrbit = trigComp.deltaOrbit[0];
286  prevBC = trigComp.deltaBC[0];
287  trigComp.deltaOrbit[0] = 0;
288  for (size_t it = 1; it < trigComp.triggerType.size(); it++) {
289  if (trigComp.deltaOrbit[it] == prevOrbit) {
290  auto bc = trigComp.deltaBC[it];
291  trigComp.deltaBC[it] -= prevBC;
292  prevBC = bc;
293  trigComp.deltaOrbit[it] = 0;
294  } else {
295  auto orb = trigComp.deltaOrbit[it];
296  trigComp.deltaOrbit[it] -= prevOrbit;
297  prevOrbit = orb;
298  }
299  }
300  }
301 
302  auto iosize = mCTFCoder.encode(buffer, clusters, clustersFiltered, trigComp, mSelIR ? &rejectHits : nullptr, mSelIR ? &rejectTracks : nullptr, mSelIR ? &rejectTrackHits : nullptr, mSelIR ? &rejectTrackHitsReduced : nullptr);
303  pc.outputs().snapshot({"ctfrep", 0}, iosize);
304  mTimer.Stop();
305  if (mSelIR) {
306  mCTFCoder.getIRFramesSelector().clear();
307  }
308  LOG(info) << iosize.asString() << " in " << mTimer.CpuTime() - cput << " s";
309 }

317 DataProcessorSpec getEntropyEncoderSpec(bool inputFromFile, bool selIR)
318 {
319  std::vector<InputSpec> inputs;
320  header::DataDescription inputType = inputFromFile ? header::DataDescription("COMPCLUSTERS") : header::DataDescription("COMPCLUSTERSFLAT");
321  inputs.emplace_back("input", "TPC", inputType, 0, Lifetime::Timeframe);
322  inputs.emplace_back("trigger", "TPC", "TRIGGERWORDS", 0, Lifetime::Timeframe);
323  inputs.emplace_back("ctfdict", "TPC", "CTFDICT", 0, Lifetime::Condition, ccdbParamSpec("TPC/Calib/CTFDictionaryTree"));
324 
325  std::shared_ptr<o2::base::GRPGeomRequest> ggreq;
326  if (selIR) {
327  inputs.emplace_back("selIRFrames", "CTF", "SELIRFRAMES", 0, Lifetime::Timeframe);
328  ggreq = std::make_shared<o2::base::GRPGeomRequest>(false, true, false, true, false, o2::base::GRPGeomRequest::None, inputs, true);
329  o2::tpc::VDriftHelper::requestCCDBInputs(inputs);
330  }
331  return DataProcessorSpec{
332  "tpc-entropy-encoder", // process id
333  inputs,
334  Outputs{{"TPC", "CTFDATA", 0, Lifetime::Timeframe},
335  {{"ctfrep"}, "TPC", "CTFENCREP", 0, Lifetime::Timeframe}},
336  AlgorithmSpec{adaptFromTask<EntropyEncoderSpec>(inputFromFile, selIR, ggreq)},
337  Options{{"ctf-dict", VariantType::String, "ccdb", {"CTF dictionary: empty or ccdb=CCDB, none=no external dictionary otherwise: local filename"}},
338  {"no-ctf-columns-combining", VariantType::Bool, false, {"Do not combine correlated columns in CTF"}},
339  {"irframe-margin-bwd", VariantType::UInt32, 0u, {"margin in BC to add to the IRFrame lower boundary when selection is requested"}},
340  {"irframe-margin-fwd", VariantType::UInt32, 0u, {"margin in BC to add to the IRFrame upper boundary when selection is requested"}},
341  {"irframe-clusters-maxeta", VariantType::Float, 1.5f, {"Max eta for non-assigned clusters"}},
342  {"irframe-clusters-maxz", VariantType::Float, 25.f, {"Max z for non assigned clusters (combined with maxeta)"}},
343  {"mem-factor", VariantType::Float, 1.f, {"Memory allocation margin factor"}},
344  {"nThreads-tpc-encoder", VariantType::UInt32, 1u, {"number of threads to use for decoding"}},
345  {"ans-version", VariantType::String, {"version of ans entropy coder implementation to use"}}}};
346 }
