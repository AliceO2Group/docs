246inline void CalibdEdxTrackTopologySpline::setSplinesFromFile(TFile& inpf)
247{
248 FlatObject::startConstruction();
249
250 int32_t buffSize = 0;
251 int32_t offsets1[FSplines];
252 int32_t offsets2[FSplines];
253
254 for (uint32_t ireg = 0; ireg < FSplines; ++ireg) {
255 std::string splinename = getSplineName(ireg, ChargeType::Max);
256 SplineType* splineTmpqMax = SplineType::readFromFile(inpf, splinename.data());
257 mCalibSplinesqMax[ireg] = *splineTmpqMax;
258 buffSize = alignSize(buffSize, mCalibSplinesqMax[ireg].getBufferAlignmentBytes());
259 offsets1[ireg] = buffSize;
260 buffSize += mCalibSplinesqMax[ireg].getFlatBufferSize();
261 delete splineTmpqMax;
262 }
263
264 for (uint32_t ireg = 0; ireg < FSplines; ++ireg) {
265 std::string splinename = getSplineName(ireg, ChargeType::Tot);
266 SplineType* splineTmpqTot = SplineType::readFromFile(inpf, splinename.data());
267 mCalibSplinesqTot[ireg] = *splineTmpqTot;
268 buffSize = alignSize(buffSize, mCalibSplinesqTot[ireg].getBufferAlignmentBytes());
269 offsets2[ireg] = buffSize;
270 buffSize += mCalibSplinesqTot[ireg].getFlatBufferSize();
271 delete splineTmpqTot;
272 }
273
274 FlatObject::finishConstruction(buffSize);
275
276 for (uint32_t i = 0; i < FSplines; i++) {
277 mCalibSplinesqMax[i].moveBufferTo(mFlatBufferPtr + offsets1[i]);
278 }
279 for (uint32_t i = 0; i < FSplines; i++) {
280 mCalibSplinesqTot[i].moveBufferTo(mFlatBufferPtr + offsets2[i]);
281 }
282}
