45 {
46 mMinROFs = 1.1 * o2::itsmft::NoiseMap::getMinROFs(prob, relErr);
47 LOGP(info, "Expect at least {} ROFs needed to apply threshold {} with relative error {}", mMinROFs, mProbabilityThreshold, mProbRelErr);
