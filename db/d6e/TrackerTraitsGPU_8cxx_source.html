368 template <int nLayers>
369 void TrackerTraitsGPU<nLayers>::findRoads(const int iteration)
370 {
371  for (int startLevel{mTrkParams[iteration].CellsPerRoad()}; startLevel >= mTrkParams[iteration].CellMinimumLevel(); --startLevel) {
372  const int minimumLayer{startLevel - 1};
373  std::vector<CellSeed> trackSeeds;
374  for (int startLayer{mTrkParams[iteration].CellsPerRoad() - 1}; startLayer >= minimumLayer; --startLayer) {
375  std::vector<int> lastCellId, updatedCellId;
376  std::vector<CellSeed> lastCellSeed, updatedCellSeed;
377 
378  processNeighbours(startLayer, startLevel, mTimeFrame->getCells()[startLayer], lastCellId, updatedCellSeed, updatedCellId);
379 
380  int level = startLevel;
381  for (int iLayer{startLayer - 1}; iLayer > 0 && level > 2; --iLayer) {
382  lastCellSeed.swap(updatedCellSeed);
383  lastCellId.swap(updatedCellId);
384  std::vector<CellSeed>().swap(updatedCellSeed);
385  updatedCellId.clear();
386  processNeighbours(iLayer, --level, lastCellSeed, lastCellId, updatedCellSeed, updatedCellId);
387  }
388  for (auto& seed : updatedCellSeed) {
389  if (seed.getQ2Pt() > 1.e3 || seed.getChi2() > mTrkParams[0].MaxChi2NDF * ((startLevel + 2) * 2 - 5)) {
390  continue;
391  }
392  trackSeeds.push_back(seed);
393  }
394  }
395  if (!trackSeeds.size()) {
396  LOGP(info, "No track seeds found, skipping track finding");
397  continue;
398  }
399  mTimeFrameGPU->createTrackITSExtDevice(trackSeeds);
400  mTimeFrameGPU->loadTrackSeedsDevice(trackSeeds);
401  auto& conf = o2::its::ITSGpuTrackingParamConfig::Instance();
402  trackSeedHandler(
403  mTimeFrameGPU->getDeviceTrackSeeds(), // CellSeed* trackSeeds,
404  mTimeFrameGPU->getDeviceArrayTrackingFrameInfo(), // TrackingFrameInfo** foundTrackingFrameInfo,
405  mTimeFrameGPU->getDeviceTrackITSExt(), // o2::its::TrackITSExt* tracks,
406  trackSeeds.size(), // const size_t nSeeds,
407  mBz, // const float Bz,
408  startLevel, // const int startLevel,
409  mTrkParams[0].MaxChi2ClusterAttachment, // float maxChi2ClusterAttachment,
410  mTrkParams[0].MaxChi2NDF, // float maxChi2NDF,
411  mTimeFrameGPU->getDevicePropagator(), // const o2::base::Propagator* propagator
412  mCorrType, // o2::base::PropagatorImpl<float>::MatCorrType
413  conf.nBlocks,
414  conf.nThreads);
415 
416  mTimeFrameGPU->downloadTrackITSExtDevice(trackSeeds);
417 
418  auto& tracks = mTimeFrameGPU->getTrackITSExt();
419  std::sort(tracks.begin(), tracks.end(), [](const TrackITSExt& a, const TrackITSExt& b) {
420  return a.getChi2() < b.getChi2();
421  });
422 
423  for (auto& track : tracks) {
424  if (!track.getChi2()) {
425  continue; // this is to skip the unset tracks that are put at the beginning of the vector by the sorting. To see if this can be optimised.
426  }
427  int nShared = 0;
428  bool isFirstShared{false};
429  for (int iLayer{0}; iLayer < mTrkParams[0].NLayers; ++iLayer) {
430  if (track.getClusterIndex(iLayer) == UnusedIndex) {
431  continue;
432  }
433  nShared += int(mTimeFrame->isClusterUsed(iLayer, track.getClusterIndex(iLayer)));
434  isFirstShared |= !iLayer && mTimeFrame->isClusterUsed(iLayer, track.getClusterIndex(iLayer));
