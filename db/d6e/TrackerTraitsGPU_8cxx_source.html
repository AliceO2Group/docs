274 template <int nLayers>
275 void TrackerTraitsGPU<nLayers>::findRoads(const int iteration)
276 {
277  for (int startLevel{mTrkParams[iteration].CellsPerRoad()}; startLevel >= mTrkParams[iteration].CellMinimumLevel(); --startLevel) {
278  const int minimumLayer{startLevel - 1};
279  std::vector<CellSeed> trackSeeds;
280  for (int startLayer{mTrkParams[iteration].CellsPerRoad() - 1}; startLayer >= minimumLayer; --startLayer) {
281  std::vector<int> lastCellId, updatedCellId;
282  std::vector<CellSeed> lastCellSeed, updatedCellSeed;
283 
284  processNeighbours(startLayer, startLevel, mTimeFrameGPU->getCells()[startLayer], lastCellId, updatedCellSeed, updatedCellId);
285 
286  int level = startLevel;
287  for (int iLayer{startLayer - 1}; iLayer > 0 && level > 2; --iLayer) {
288  lastCellSeed.swap(updatedCellSeed);
289  lastCellId.swap(updatedCellId);
290  std::vector<CellSeed>().swap(updatedCellSeed);
291  updatedCellId.clear();
292  processNeighbours(iLayer, --level, lastCellSeed, lastCellId, updatedCellSeed, updatedCellId);
293  }
294  for (auto& seed : updatedCellSeed) {
295  if (seed.getQ2Pt() > 1.e3 || seed.getChi2() > mTrkParams[0].MaxChi2NDF * ((startLevel + 2) * 2 - 5)) {
296  continue;
297  }
298  trackSeeds.push_back(seed);
299  }
300  }
301  if (!trackSeeds.size()) {
302  LOGP(info, "No track seeds found, skipping track finding");
303  continue;
304  }
305  mTimeFrameGPU->createTrackITSExtDevice(trackSeeds);
306  mTimeFrameGPU->loadTrackSeedsDevice(trackSeeds);
307  auto& conf = o2::its::ITSGpuTrackingParamConfig::Instance();
308  trackSeedHandler(mTimeFrameGPU->getDeviceTrackSeeds(), // CellSeed* trackSeeds,
309  mTimeFrameGPU->getDeviceArrayTrackingFrameInfo(), // TrackingFrameInfo** foundTrackingFrameInfo,
310  mTimeFrameGPU->getDeviceTrackITSExt(), // o2::its::TrackITSExt* tracks,
311  trackSeeds.size(), // const size_t nSeeds,
312  mBz, // const float Bz,
313  startLevel, // const int startLevel,
314  mTrkParams[0].MaxChi2ClusterAttachment, // float maxChi2ClusterAttachment,
315  mTrkParams[0].MaxChi2NDF, // float maxChi2NDF,
316  mTimeFrameGPU->getDevicePropagator(), // const o2::base::Propagator* propagator
317  mCorrType, // o2::base::PropagatorImpl<float>::MatCorrType
318  conf.nBlocks,
319  conf.nThreads);
320 
321  mTimeFrameGPU->downloadTrackITSExtDevice(trackSeeds);
322 
323  auto& tracks = mTimeFrameGPU->getTrackITSExt();
324  std::sort(tracks.begin(), tracks.end(), [](const TrackITSExt& a, const TrackITSExt& b) {
325  return a.getChi2() < b.getChi2();
326  });
327 
328  for (auto& track : tracks) {
329  if (!track.getChi2()) {
330  continue; // this is to skip the unset tracks that are put at the beginning of the vector by the sorting. To see if this can be optimised.
331  }
332  int nShared = 0;
333  bool isFirstShared{false};
334  for (int iLayer{0}; iLayer < mTrkParams[0].NLayers; ++iLayer) {
335  if (track.getClusterIndex(iLayer) == UnusedIndex) {
336  continue;
337  }
338  nShared += int(mTimeFrameGPU->isClusterUsed(iLayer, track.getClusterIndex(iLayer)));
339  isFirstShared |= !iLayer && mTimeFrameGPU->isClusterUsed(iLayer, track.getClusterIndex(iLayer));
340  }
341 
342  if (nShared > mTrkParams[0].ClusterSharing) {
343  continue;
344  }
345 
346  std::array<int, 3> rofs{INT_MAX, INT_MAX, INT_MAX};
347  for (int iLayer{0}; iLayer < mTrkParams[0].NLayers; ++iLayer) {
348  if (track.getClusterIndex(iLayer) == UnusedIndex) {
349  continue;
350  }
351  mTimeFrameGPU->markUsedCluster(iLayer, track.getClusterIndex(iLayer));
352  int currentROF = mTimeFrameGPU->getClusterROF(iLayer, track.getClusterIndex(iLayer));
353  for (int iR{0}; iR < 3; ++iR) {
354  if (rofs[iR] == INT_MAX) {
355  rofs[iR] = currentROF;
356  }
357  if (rofs[iR] == currentROF) {
358  break;
359  }
360  }
361  }
362  if (rofs[2] != INT_MAX) {
363  continue;
364  }
365  if (rofs[1] != INT_MAX) {
366  track.setNextROFbit();
367  }
368  mTimeFrameGPU->getTracks(std::min(rofs[0], rofs[1])).emplace_back(track);
369  }
370  }
371  mTimeFrameGPU->loadUsedClustersDevice();
372  if (iteration == mTrkParams.size() - 1) {
373  mTimeFrameGPU->unregisterHostMemory(0);
374  }
375 };
