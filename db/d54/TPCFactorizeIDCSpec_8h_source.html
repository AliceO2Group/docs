98  void run(o2::framework::ProcessingContext& pc) final
99  {
100  // store precise timestamp for look up later
101  if (mUsePrecisetimeStamp && pc.inputs().isValid("orbitreset")) {
102  mOrbitResetTime = pc.inputs().get<Long64_t>("orbitreset");
103  if (pc.inputs().countValidInputs() == 1) {
104  return;
105  }
106  }
107 
108  const auto currTF = processing_helpers::getCurrentTF(pc);
109 
110  // set the min range of TFs for first TF
111  if (mProcessedCRUs == 0) {
112  mTFOrbitFirst = processing_helpers::getFirstTForbit(pc);
113  mTFFirst = currTF;
114  for (unsigned int iChunk = 0; iChunk < mIDCFactorization.getNChunks(mSides.front()); ++iChunk) {
115  mTFRangeIDCDelta[iChunk] = getFirstTFDeltaIDC(iChunk);
116  }
117  }
118 
119  // relative TF of current aggreagtion intervall
120  const auto relTF = currTF - mTFFirst;
121 
122  // check if current TF is in range of IDCDelta range
123  findTimeStamp(pc);
124 
125  for (auto& ref : InputRecordWalker(pc.inputs(), mFilter)) {
126  auto const* tpcCRUHeader = o2::framework::DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
127  const int cru = tpcCRUHeader->subSpecification;
128  mIDCFactorization.setIDCs(pc.inputs().get<std::vector<float>>(ref), cru, relTF); // aggregate IDCs
129  ++mProcessedCRUs;
130  }
131 
132  if (!(mProcessedCRUs % mTFsMessaged)) {
133  LOGP(info, "ProcessedTFs: {} currTF: {} relTF: {} OrbitResetTime: {}", mProcessedCRUs / mCRUs.size(), currTF, relTF, mOrbitResetTime);
134  }
135 
136  if (mProcessedCRUs == mCRUs.size() * mIDCFactorization.getNTimeframes()) {
137  mProcessedCRUs = 0; // reset processed TFs for next aggregation interval
138  if constexpr (std::is_same_v<Type, TPCFactorizeIDCSpecGroup>) {
139  mIDCFactorization.factorizeIDCs(true); // calculate DeltaIDC, 0D-IDC, 1D-IDC
140  } else {
141  mIDCFactorization.factorizeIDCs(false); // calculate DeltaIDC, 0D-IDC, 1D-IDC
142  }
143 
144  if (mDebug) {
145  LOGP(info, "dumping aggregated and factorized IDCs to file");
146  mIDCFactorization.dumpToFile(fmt::format("IDCFactorized_{:02}.root", currTF).data());
147  mIDCFactorization.dumpPadFlagMap("padstatusmap.root", "PadStatus");
148  }
149 
150  // storing to CCDB
151  sendOutput(pc.outputs());
152  mTFOrbitFirst = 0;
153  }
154  }

222  bool findTimeStamp(const uint32_t tf, o2::framework::ProcessingContext& pc)
223  {
224  auto it = std::find(mTFRangeIDCDelta.begin(), mTFRangeIDCDelta.end(), tf);
225  if (it != mTFRangeIDCDelta.end()) {
226  const int index = std::distance(mTFRangeIDCDelta.begin(), it);
227  mTimeStampRangeIDCDelta[index] = mUsePrecisetimeStamp ? processing_helpers::getFirstTForbit(pc) : getTimeStamp(pc);
228  // TODO remove found tf?

230  }

394  } else {
395  idcDelta = mIDCFactorization.getIDCDeltaHighCompressed(iChunk, side);
396  }
397  imageIDCDelta = o2::ccdb::CcdbApi::createObjectImage(&idcDelta, &ccdbInfoIDCDelta);
398  break;
399  }
400  case IDCDeltaCompression::NO:
401  IDCDelta<float> idcDelta;
402  if constexpr (std::is_same_v<Type, TPCFactorizeIDCSpecGroup>) {
403  idcDelta = std::move(mIDCStruct).mIDCs.getIDCGroupData();
404  } else {
405  idcDelta = std::move(mIDCFactorization).getIDCDeltaUncompressed(iChunk, side);
406  }
407  imageIDCDelta = o2::ccdb::CcdbApi::createObjectImage(&idcDelta, &ccdbInfoIDCDelta);
408  break;
409  }
410  LOGP(info, "Sending object {} / {} of size {} bytes, valid for {} : {} ", ccdbInfoIDCDelta.getPath(), ccdbInfoIDCDelta.getFileName(), imageIDCDelta->size(), ccdbInfoIDCDelta.getStartValidityTimestamp(), ccdbInfoIDCDelta.getEndValidityTimestamp());
411  output.snapshot(Output{o2::calibration::Utils::gDataOriginCDBPayload, getDataDescriptionCCDBIDCDelta(), iChunk}, *imageIDCDelta.get());
412  output.snapshot(Output{o2::calibration::Utils::gDataOriginCDBWrapper, getDataDescriptionCCDBIDCDelta(), iChunk}, ccdbInfoIDCDelta);
413  auto stopCCDBIDCDelta = timer::now();
414  time = stopCCDBIDCDelta - startCCDBIDCDelta;
415  LOGP(info, "Compression and CCDB object creation time: {}", time.count());
416  }
417 
418  stop = timer::now();
419  time = stop - start;
420  LOGP(info, "IDCDelta CCDB time: {}", time.count());
421  totalTime += time.count();
422  LOGP(info, "CCDB object creation done. Total time: {}", totalTime);
423  }
424  }
425 
426  // reseting aggregated IDCs. This is done for safety, but if all data is received in the next aggregation interval it isnt necessary... remove it?
427  LOGP(info, "Everything done! Clearing memory...");
428  mIDCFactorization.reset();
429  LOGP(info, "Everything cleared. Waiting for new data to arrive.");
430  }
