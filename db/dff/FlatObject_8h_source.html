120template <typename T>
121T* resizeArray(T*& ptr, int32_t oldSize, int32_t newSize, T* newPtr = nullptr)
122{
123 // Resize array pointed by ptr. T must be a POD class.
124 // If the non-null newPtr is provided, use it instead of allocating a new one.
125 // In this case it is up to the user to ensure that it has at least newSize slots allocated.
126 // Return original array pointer, so that the user can manage previously allocate memory
127 if (oldSize < 0) {
128 oldSize = 0;
129 }
130 if (newSize > 0) {
131 if (!newPtr) {
132 newPtr = new T[newSize];
133 }
134 int32_t mcp = std::min(newSize, oldSize);
135 if (mcp) {
136 assert(ptr);
137 std::memmove(newPtr, ptr, mcp * sizeof(T));
138 }
139 if (newSize > oldSize) {
140 std::memset(newPtr + mcp, 0, (newSize - oldSize) * sizeof(T));
141 }
142 }
143 T* oldPtr = ptr;
144 ptr = newPtr;
145 return oldPtr;
146}

375inline void FlatObject::cloneFromObject(const FlatObject& obj, char* newFlatBufferPtr)
376{
380 // that we want to relocate the obj to external buffer)
381
382 assert(obj.isConstructed());
383
384 // providing *this with internal buffer as obj makes sens only if we want to conver it to object with PROVIDED external buffer
385 assert(!(!newFlatBufferPtr && obj.mFlatBufferPtr == mFlatBufferPtr && obj.isBufferInternal()));
386
387 char* oldPtr = resizeArray(mFlatBufferPtr, mFlatBufferSize, obj.mFlatBufferSize, newFlatBufferPtr);
388
389 if (isBufferInternal()) {
390 delete[] oldPtr; // delete old buffer if owned
391 }
392 mFlatBufferSize = obj.mFlatBufferSize;
393 mFlatBufferContainer = newFlatBufferPtr ? nullptr : mFlatBufferPtr; // external buffer is not provided, make object to own the buffer
394 std::memcpy(mFlatBufferPtr, obj.mFlatBufferPtr, obj.mFlatBufferSize);
395 mConstructionMask = (uint32_t)ConstructionState::Constructed;
396}

398inline void FlatObject::moveBufferTo(char* newFlatBufferPtr)
399{
401 if (newFlatBufferPtr == mFlatBufferContainer) {
402 return;
403 }
404 resizeArray(mFlatBufferPtr, mFlatBufferSize, mFlatBufferSize, newFlatBufferPtr);
405 delete[] mFlatBufferContainer;
406 mFlatBufferContainer = nullptr;
407 if (!newFlatBufferPtr) { // resizeArray has created own array
408 mFlatBufferContainer = mFlatBufferPtr;
409 }
410}

412template <class T>
413inline std::string FlatObject::stressTest(T& obj)
414{
418 std::string err;
419
420 if (!obj.isConstructed()) {
421 return "tested object is not constructed!";
422 }
423
424 T tst;
425 tst.cloneFromObject(obj, nullptr);
426 if (!tst.isConstructed() || !tst.isBufferInternal()) {
427 return "error at cloneFromObject()!";
428 }
429
430 obj.destroy();
431
432 char* buf0 = tst.releaseInternalBuffer();
433 char* buf1 = new char[tst.getFlatBufferSize()];
434 char* buf2 = new char[tst.getFlatBufferSize()];
435 std::memcpy(buf1, tst.getFlatBufferPtr(), tst.getFlatBufferSize());
436 tst.setActualBufferAddress(buf1);
437 delete[] buf0;
438
439 tst.setFutureBufferAddress(buf2);
440 std::memcpy(buf2, buf1, tst.getFlatBufferSize());
441 delete[] buf1;
442
443 if (tst.isBufferInternal()) {
444 return err = "error, buffer should be external!";
445 }
446
447 tst.adoptInternalBuffer(buf2);
448 if (!tst.isBufferInternal()) {
449 return err = "error, buffer should be internal!";
450 }
451
452 obj.cloneFromObject(tst, nullptr);
453 if (!obj.isBufferInternal()) {
454 return err = "error, buffer should be internal!";
455 }
456
457 return err;
458}

481template <class T, class TFile>
482inline int32_t FlatObject::writeToFile(T& obj, TFile& outf, const char* name)
483{
485 assert(obj.isConstructed());
486
487 if (outf.IsZombie()) {
488 LOG(error) << "Failed to write to file " << outf.GetName();
489 return -1;
490 }
491
492 bool isBufferExternal = !obj.isBufferInternal();
493 if (isBufferExternal) {
494 obj.adoptInternalBuffer(obj.mFlatBufferPtr);
495 }
496 outf.WriteObjectAny(&obj, T::Class(), name);
497 if (isBufferExternal) {
498 obj.clearInternalBufferPtr();
499 }
500 return 0;
501}

503template <class T, class TFile>
504inline T* FlatObject::readFromFile(TFile& inpf, const char* name)
505{
507
508 if (inpf.IsZombie()) {
509 LOG(error) << "Failed to read from file " << inpf.GetName();
510 return nullptr;
511 }
512 T* pobj = reinterpret_cast<T*>(inpf.GetObjectChecked(name, T::Class()));
513 if (!pobj) {
514 LOG(error) << "Failed to load " << name << " from " << inpf.GetName();
515 return nullptr;
516 }
517 if (pobj->mFlatBufferSize > 0 && pobj->mFlatBufferContainer == nullptr) {
518 LOG(error) << "Failed to load " << name << " from " << inpf.GetName() << ": empty flat buffer container";
519 return nullptr;
520 }
521 pobj->setActualBufferAddress(pobj->mFlatBufferContainer);
522 return pobj;
523}
