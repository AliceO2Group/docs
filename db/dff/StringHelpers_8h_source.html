68consteval uint32_t crc32(char const* str, int length)
69{
70 uint32_t crc = 0xFFFFFFFF;
71 for (auto j = 0; j <= length; ++j) {
72 crc = (crc >> 8) ^ crc_table[(crc ^ static_cast<unsigned int>(str[j])) & 0x000000FF];
73 }
74 return crc;
75}

77template <typename... Ts>
78 requires(std::same_as<Ts, std::string_view> && ...)
79consteval uint32_t crc32(Ts... Vs)
80{
81 uint32_t crc = 0xFFFFFFFF;
82 for (auto& c : std::vector{Vs...} | std::ranges::views::join) {
83 crc = (crc >> 8) ^ crc_table[(crc ^ static_cast<unsigned int>(c)) & 0x000000FF];
84 }

119template <int N>
120consteval uint32_t compile_time_hash_from_literal(const char (&str)[N])
121{
122 return crc32(str, N - 2) ^ 0xFFFFFFFF;
