123GPUd() void GPUTPCGlobalTracking::PerformGlobalTracking(int32_t nBlocks, int32_t nThreads, int32_t iBlock, int32_t iThread, const GPUTPCTracker& tracker, GPUsharedref() MEM_LOCAL(GPUSharedMemory) & smem, GPUTPCTracker& GPUrestrict() sliceTarget, bool right)
124{
125 for (int32_t i = iBlock * nThreads + iThread; i < tracker.CommonMemory()->nLocalTracks; i += nThreads * nBlocks) {
126 {
127 const int32_t tmpHit = tracker.Tracks()[i].FirstHitID();
128 if (tracker.TrackHits()[tmpHit].RowIndex() >= tracker.Param().rec.tpc.globalTrackingMinRows && tracker.TrackHits()[tmpHit].RowIndex() < tracker.Param().rec.tpc.globalTrackingRowRange) {
129 int32_t rowIndex = tracker.TrackHits()[tmpHit].RowIndex();
130 const GPUTPCRow& GPUrestrict() row = tracker.Row(rowIndex);
131 float Y = (float)tracker.Data().HitDataY(row, tracker.TrackHits()[tmpHit].HitIndex()) * row.HstepY() + row.Grid().YMin();
132 if (!right && Y < -row.MaxY() * tracker.Param().rec.tpc.globalTrackingYRangeLower) {
133 // GPUInfo("Track %d, lower row %d, left border (%f of %f)", i, mTrackHits[tmpHit].RowIndex(), Y, -row.MaxY());
134 PerformGlobalTrackingRun(sliceTarget, smem, tracker, i, rowIndex, -tracker.Param().par.dAlpha, -1);
135 }
136 if (right && Y > row.MaxY() * tracker.Param().rec.tpc.globalTrackingYRangeLower) {
137 // GPUInfo("Track %d, lower row %d, right border (%f of %f)", i, mTrackHits[tmpHit].RowIndex(), Y, row.MaxY());

139 }
140 }
141 }
