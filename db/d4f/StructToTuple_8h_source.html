134 template <typename T>
135 consteval auto brace_constructible_size(auto... Members)
136 {
137  if constexpr (requires { T{Members...}; } == false) {
138  static_assert(sizeof...(Members) != 0, "You need to make sure that you have implicit constructors or that you call the explicit constructor correctly.");
139  return sizeof...(Members) - 1;
140  } else {
141  return brace_constructible_size<T>(Members..., UniversalType{});
142  }
143 }

166 template <bool B, typename T>
167 consteval int nested_brace_constructible_size()
168 {
169  using type = std::decay_t<T>;
170  constexpr int nesting = B ? 1 : 0;
171  return brace_constructible_size<type>() - nesting;
172 }
