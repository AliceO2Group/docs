22double AliHLT3DTrackParam::GetDStoPoint(double Bz, const double xyz[3], const double* T0) const
23{
24 //* Get DS = Path/Momentum to a certain space point for Bz field
25
26 double q = fSignQ;
27 if (!T0) {
28 T0 = mParam;
29 } else {
30 q = T0[6];
31 }
32
33 const double kCLight = 0.000299792458;
34 double bq = Bz * q * kCLight;
35 double pt2 = T0[3] * T0[3] + T0[4] * T0[4];
36 if (pt2 < 1.e-4) {
37 return 0;
38 }
39 double dx = xyz[0] - T0[0];
40 double dy = xyz[1] - T0[1];
41 double a = dx * T0[3] + dy * T0[4];
42 double dS = 0;
43 if (TMath::Abs(bq) < 1.e-8) {
44 dS = a / pt2;
45 } else {
46 dS = TMath::ATan2(bq * a, pt2 + bq * (dy * T0[3] - dx * T0[4])) / bq;
47 }
48 return dS;
49}

51void AliHLT3DTrackParam::TransportToDS(double Bz, double DS, double* T0)
52{
53 //* Transport the particle on DS = Path/Momentum, for Bz field
54
55 double tmp[7];
56 if (!T0) {
57 T0 = tmp;
58 T0[0] = mParam[0];
59 T0[1] = mParam[1];
60 T0[2] = mParam[2];
61 T0[3] = mParam[3];
62 T0[4] = mParam[4];
63 T0[5] = mParam[5];
64 T0[6] = fSignQ;
65 }
66 const double kCLight = 0.000299792458;
67 Bz = Bz * T0[6] * kCLight;
68 double bs = Bz * DS;
69 double s = TMath::Sin(bs), c = TMath::Cos(bs);
70 double sB, cB;
71 if (TMath::Abs(bs) > 1.e-10) {
72 sB = s / Bz;
73 cB = (1 - c) / Bz;
74 } else {
75 const Double_t kOvSqr6 = 1. / TMath::Sqrt(6.);
76 sB = (1. - bs * kOvSqr6) * (1. + bs * kOvSqr6) * DS;
77 cB = .5 * sB * bs;
78 }
79
80 double px = T0[3];
81 double py = T0[4];
82 double pz = T0[5];
83
84 double d[6] = {mParam[0] - T0[0], mParam[1] - T0[1], mParam[2] - T0[2], mParam[3] - T0[3], mParam[4] - T0[4], mParam[5] - T0[5]};
85
86 T0[0] = T0[0] + sB * px + cB * py;
87 T0[1] = T0[1] - cB * px + sB * py;
88 T0[2] = T0[2] + DS * pz;
89 T0[3] = c * px + s * py;
90 T0[4] = -s * px + c * py;
91 T0[5] = T0[5];
92
93 // clang-format off
94 double mJ[6][6] = { {1, 0, 0, sB, cB, 0, },
95 {0, 1, 0, -cB, sB, 0, },
96 {0, 0, 1, 0, 0, DS, },
97 {0, 0, 0, c, s, 0, },
98 {0, 0, 0, -s, c, 0, },
99 {0, 0, 0, 0, 0, 1, }};
100 // clang-format on
101
102 for (int32_t i = 0; i < 6; i++) {
103 mParam[i] = T0[i];
104 for (int32_t j = 0; j < 6; j++) {
105 mParam[i] += mJ[i][j] * d[j];
106 }
107 }
108
109 double mA[6][6];
110 for (int32_t k = 0, i = 0; i < 6; i++) {
111 for (int32_t j = 0; j <= i; j++, k++) {
112 mA[i][j] = mA[j][i] = fCov[k];
113 }
114 }
115
116 double mJC[6][6];
117 for (int32_t i = 0; i < 6; i++) {
118 for (int32_t j = 0; j < 6; j++) {
119 mJC[i][j] = 0;
120 for (int32_t k = 0; k < 6; k++) {
121 mJC[i][j] += mJ[i][k] * mA[k][j];
122 }
123 }
124 }
125
126 for (int32_t k = 0, i = 0; i < 6; i++) {
127 for (int32_t j = 0; j <= i; j++, k++) {
128 fCov[k] = 0;
129 for (int32_t l = 0; l < 6; l++) {
130 fCov[k] += mJC[i][l] * mJ[j][l];
131 }
132 }
133 }
134}

152void AliHLT3DTrackParam::GetGlueMatrix(const double xyz[3], double G[6], const double* T0) const
153{
154 //* !
155
156 if (!T0) {
157 T0 = mParam;
158 }
159
160 double dx = xyz[0] - T0[0], dy = xyz[1] - T0[1], dz = xyz[2] - T0[2];
161 double px2 = T0[3] * T0[3], py2 = T0[4] * T0[4], pz2 = T0[5] * T0[5];
162 double s2 = (dx * dx + dy * dy + dz * dz);
163 double p2 = px2 + py2 + pz2;
164 if (p2 > 1.e-4) {
165 s2 /= p2;
166 }
167 double x = T0[3] * s2;
168 double xx = px2 * s2, xy = x * T0[4], xz = x * T0[5], yy = py2 * s2, yz = T0[4] * T0[5] * s2;
169 G[0] = xx;
170 G[1] = xy;
171 G[2] = yy;
172 G[3] = xz;
173 G[4] = yz;
174 G[5] = pz2 * s2;
175}

177void AliHLT3DTrackParam::Filter(const double m[3], const double V[6], const double G[6])
178{
179 //* !
180
181 // clang-format off
182 double
183 c00 = fCov[0],
184 c10 = fCov[1], c11 = fCov[2],
185 c20 = fCov[3], c21 = fCov[4], c22 = fCov[5],
186 c30 = fCov[6], c31 = fCov[7], c32 = fCov[8],
187 c40 = fCov[10], c41 = fCov[11], c42 = fCov[12],
188 c50 = fCov[15], c51 = fCov[16], c52 = fCov[17];
189 // clang-format on
190
191 double z0 = m[0] - mParam[0], z1 = m[1] - mParam[1], z2 = m[2] - mParam[2];
192
193 double mS[6] = {c00 + V[0] + G[0], c10 + V[1] + G[1], c11 + V[2] + G[2], c20 + V[3] + G[3], c21 + V[4] + G[4], c22 + V[5] + G[5]};
194 double mSi[6];
195 mSi[0] = mS[4] * mS[4] - mS[2] * mS[5];
196 mSi[1] = mS[1] * mS[5] - mS[3] * mS[4];
197 mSi[3] = mS[2] * mS[3] - mS[1] * mS[4];
198 double det = 1. / (mS[0] * mSi[0] + mS[1] * mSi[1] + mS[3] * mSi[3]);
199 mSi[0] *= det;
200 mSi[1] *= det;
201 mSi[3] *= det;
202 mSi[2] = (mS[3] * mS[3] - mS[0] * mS[5]) * det;
203 mSi[4] = (mS[0] * mS[4] - mS[1] * mS[3]) * det;
204 mSi[5] = (mS[1] * mS[1] - mS[0] * mS[2]) * det;
205
206 fNDF += 2;
207 fChi2 += (+(mSi[0] * z0 + mSi[1] * z1 + mSi[3] * z2) * z0 + (mSi[1] * z0 + mSi[2] * z1 + mSi[4] * z2) * z1 + (mSi[3] * z0 + mSi[4] * z1 + mSi[5] * z2) * z2);
208
209 double k0, k1, k2; // k = CHtS
210
211 k0 = c00 * mSi[0] + c10 * mSi[1] + c20 * mSi[3];
212 k1 = c00 * mSi[1] + c10 * mSi[2] + c20 * mSi[4];
213 k2 = c00 * mSi[3] + c10 * mSi[4] + c20 * mSi[5];
214
215 mParam[0] += k0 * z0 + k1 * z1 + k2 * z2;
216 fCov[0] -= k0 * c00 + k1 * c10 + k2 * c20;
217
218 k0 = c10 * mSi[0] + c11 * mSi[1] + c21 * mSi[3];
219 k1 = c10 * mSi[1] + c11 * mSi[2] + c21 * mSi[4];
220 k2 = c10 * mSi[3] + c11 * mSi[4] + c21 * mSi[5];
221
222 mParam[1] += k0 * z0 + k1 * z1 + k2 * z2;
223 fCov[1] -= k0 * c00 + k1 * c10 + k2 * c20;
224 fCov[2] -= k0 * c10 + k1 * c11 + k2 * c21;
225
226 k0 = c20 * mSi[0] + c21 * mSi[1] + c22 * mSi[3];
227 k1 = c20 * mSi[1] + c21 * mSi[2] + c22 * mSi[4];
228 k2 = c20 * mSi[3] + c21 * mSi[4] + c22 * mSi[5];
229
230 mParam[2] += k0 * z0 + k1 * z1 + k2 * z2;
231 fCov[3] -= k0 * c00 + k1 * c10 + k2 * c20;
232 fCov[4] -= k0 * c10 + k1 * c11 + k2 * c21;
233 fCov[5] -= k0 * c20 + k1 * c21 + k2 * c22;
234
235 k0 = c30 * mSi[0] + c31 * mSi[1] + c32 * mSi[3];
236 k1 = c30 * mSi[1] + c31 * mSi[2] + c32 * mSi[4];
237 k2 = c30 * mSi[3] + c31 * mSi[4] + c32 * mSi[5];
238
239 mParam[3] += k0 * z0 + k1 * z1 + k2 * z2;
240 fCov[6] -= k0 * c00 + k1 * c10 + k2 * c20;
241 fCov[7] -= k0 * c10 + k1 * c11 + k2 * c21;
242 fCov[8] -= k0 * c20 + k1 * c21 + k2 * c22;
243 fCov[9] -= k0 * c30 + k1 * c31 + k2 * c32;
244
245 k0 = c40 * mSi[0] + c41 * mSi[1] + c42 * mSi[3];
246 k1 = c40 * mSi[1] + c41 * mSi[2] + c42 * mSi[4];
247 k2 = c40 * mSi[3] + c41 * mSi[4] + c42 * mSi[5];
248
249 mParam[4] += k0 * z0 + k1 * z1 + k2 * z2;
250 fCov[10] -= k0 * c00 + k1 * c10 + k2 * c20;
251 fCov[11] -= k0 * c10 + k1 * c11 + k2 * c21;
252 fCov[12] -= k0 * c20 + k1 * c21 + k2 * c22;
253 fCov[13] -= k0 * c30 + k1 * c31 + k2 * c32;
254 fCov[14] -= k0 * c40 + k1 * c41 + k2 * c42;
255
256 k0 = c50 * mSi[0] + c51 * mSi[1] + c52 * mSi[3];
257 k1 = c50 * mSi[1] + c51 * mSi[2] + c52 * mSi[4];
258 k2 = c50 * mSi[3] + c51 * mSi[4] + c52 * mSi[5];
259
260 mParam[5] += k0 * z0 + k1 * z1 + k2 * z2;
261 fCov[15] -= k0 * c00 + k1 * c10 + k2 * c20;
262 fCov[16] -= k0 * c10 + k1 * c11 + k2 * c21;
263 fCov[17] -= k0 * c20 + k1 * c21 + k2 * c22;
264 fCov[18] -= k0 * c30 + k1 * c31 + k2 * c32;
265 fCov[19] -= k0 * c40 + k1 * c41 + k2 * c42;
266 fCov[20] -= k0 * c50 + k1 * c51 + k2 * c52;
267
268 // fit charge
269
270 double px = mParam[3];
271 double py = mParam[4];
272 double pz = mParam[5];
273
274 double p = TMath::Sqrt(px * px + py * py + pz * pz);
275 double pi = 1. / p;
276 double qp = fSignQ * pi;
277 double qp3 = qp * pi * pi;
278 double c60 = qp3 * (c30 + c40 + c50), c61 = qp3 * (c31 + c41 + c51), c62 = qp3 * (c32 + c42 + c52);
279
280 k0 = c60 * mSi[0] + c61 * mSi[1] + c62 * mSi[3];
281 k1 = c60 * mSi[1] + c61 * mSi[2] + c62 * mSi[4];
282 k2 = c60 * mSi[3] + c61 * mSi[4] + c62 * mSi[5];
283
284 qp += k0 * z0 + k1 * z1 + k2 * z2;
285 if (qp > 0) {
286 fSignQ = 1;
287 } else if (qp < 0) {
288 fSignQ = -1;
289 } else {
290 fSignQ = 0;
291 }
292}

320void AliHLT3DTrackParam::RotateCoordinateSystem(double alpha)
321{
322 //* !
323
324 double cA = TMath::Cos(alpha);
325 double sA = TMath::Sin(alpha);
326 double x = mParam[0], y = mParam[1], px = mParam[3], py = mParam[4];
327 mParam[0] = x * cA + y * sA;
328 mParam[1] = -x * sA + y * cA;
329 mParam[2] = mParam[2];
330 mParam[3] = px * cA + py * sA;
331 mParam[4] = -px * sA + py * cA;
332 mParam[5] = mParam[5];
333
334 // clang-format off
335 double mJ[6][6] = { { cA, sA, 0, 0, 0, 0 },
336 { -sA, cA, 0, 0, 0, 0 },
337 { 0, 0, 1, 0, 0, 0 },
338 { 0, 0, 0, cA, sA, 0 },
339 { 0, 0, 0, -sA, cA, 0 },
340 { 0, 0, 0, 0, 0, 1 }};
341 // clang-format on
342
343 double mA[6][6];
344 for (int32_t k = 0, i = 0; i < 6; i++) {
345 for (int32_t j = 0; j <= i; j++, k++) {
346 mA[i][j] = mA[j][i] = fCov[k];
347 }
348 }
349
350 double mJC[6][6];
351 for (int32_t i = 0; i < 6; i++) {
352 for (int32_t j = 0; j < 6; j++) {
353 mJC[i][j] = 0;
354 for (int32_t k = 0; k < 6; k++) {
355 mJC[i][j] += mJ[i][k] * mA[k][j];
356 }
357 }
358 }
359
360 for (int32_t k = 0, i = 0; i < 6; i++) {
361 for (int32_t j = 0; j <= i; j++, k++) {
362 fCov[k] = 0;
363 for (int32_t l = 0; l < 6; l++) {
364 fCov[k] += mJC[i][l] * mJ[j][l];
365 }
366 }
367 }

370void AliHLT3DTrackParam::Get5Parameters(double alpha, double T[6], double C[15]) const
371{
372 //* !
373
374 AliHLT3DTrackParam t = *this;
375 t.RotateCoordinateSystem(alpha);
376 double x = t.mParam[0], y = t.mParam[1], z = t.mParam[2], px = t.mParam[3], py = t.mParam[4], pz = t.mParam[5], q = t.fSignQ;
377
378 double p2 = px * px + py * py + pz * pz;
379 if (p2 < 1.e-8) {
380 p2 = 1;
381 }
382 double n2 = 1. / p2;
383 double n = sqrt(n2);
384
385 T[5] = x;
386 T[0] = y;
387 T[1] = z;
388 T[2] = py / px;
389 T[3] = pz / px;
390 T[4] = q * n;
391
392 // clang-format off
393 double mJ[5][6] = { { -T[2], 1, 0, 0, 0, 0 },
394 { -T[3], 0, 1, 0, 0, 0 },
395 { 0, 0, 0, -T[2] / px, 1. / px, 0 },
396 { 0, 0, 0, -T[3] / px, 0, 1. / px },
397 { 0, 0, 0, -T[4]*n2*px, -T[4]*n2*py, -T[4]*n2*pz}};
398 // clang-format on
399
400 double mA[6][6];
401 for (int32_t k = 0, i = 0; i < 6; i++) {
402 for (int32_t j = 0; j <= i; j++, k++) {
403 mA[i][j] = mA[j][i] = t.fCov[k];
404 }
405 }
406
407 double mJC[5][6];
408 for (int32_t i = 0; i < 5; i++) {
409 for (int32_t j = 0; j < 6; j++) {
410 mJC[i][j] = 0;
411 for (int32_t k = 0; k < 6; k++) {
412 mJC[i][j] += mJ[i][k] * mA[k][j];
413 }
414 }
415 }
416
417 for (int32_t k = 0, i = 0; i < 5; i++) {
418 for (int32_t j = 0; j <= i; j++, k++) {
419 C[k] = 0;
420 for (int32_t l = 0; l < 6; l++) {
421 C[k] += mJC[i][l] * mJ[j][l];
422 }
423 }
424 }
