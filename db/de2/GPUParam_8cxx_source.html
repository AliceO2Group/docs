128 void GPUParam::UpdateSettings(const GPUSettingsGRP* g, const GPUSettingsProcessing* p, const GPURecoStepConfiguration* w, const GPUSettingsRecDynamic* d)
129 {
130  if (g) {
131  UpdateBzOnly(g->solenoidBzNominalGPU);
132  par.assumeConstantBz = g->constBz;
133  par.toyMCEventsFlag = g->homemadeEvents;
134  par.continuousTracking = g->grpContinuousMaxTimeBin != 0;
135  continuousMaxTimeBin = g->grpContinuousMaxTimeBin == -1 ? GPUSettings::TPC_MAX_TF_TIME_BIN : g->grpContinuousMaxTimeBin;
136  tpcCutTimeBin = g->tpcCutTimeBin;
137  }
138  par.earlyTpcTransform = rec.tpc.forceEarlyTransform == -1 ? (!par.continuousTracking) : rec.tpc.forceEarlyTransform;
139  qptB5Scaler = CAMath::Abs(bzkG) > 0.1f ? CAMath::Abs(bzkG) / 5.006680f : 1.f; // Repeat here, since passing in g is optional
140  if (p) {
141  par.debugLevel = p->debugLevel;
142  UpdateRun3ClusterErrors(p->param.tpcErrorParamY, p->param.tpcErrorParamZ);
143  }
144  if (w) {
145  par.dodEdx = dodEdxDownscaled = w->steps.isSet(GPUDataTypes::RecoStep::TPCdEdx);
146  if (par.dodEdx && p && p->tpcDownscaledEdx != 0) {
147  dodEdxDownscaled = (rand() % 100) < p->tpcDownscaledEdx;
148  }
149  }
150  if (d) {
151  rec.dyn = *d;
152  }
153 }

180 void GPUParam::UpdateRun3ClusterErrors(const float* yErrorParam, const float* zErrorParam)
181 {
182 #ifdef GPUCA_TPC_GEOMETRY_O2
183  for (int32_t yz = 0; yz < 2; yz++) {
184  const float* param = yz ? zErrorParam : yErrorParam;
185  for (int32_t rowType = 0; rowType < 4; rowType++) {
186  constexpr int32_t regionMap[4] = {0, 4, 6, 8};
187  ParamErrors[yz][rowType][0] = param[0] * param[0];
188  ParamErrors[yz][rowType][1] = param[1] * param[1] * tpcGeometry.PadHeightByRegion(regionMap[rowType]);
189  ParamErrors[yz][rowType][2] = param[2] * param[2] / tpcGeometry.TPCLength() / tpcGeometry.PadHeightByRegion(regionMap[rowType]);
190  ParamErrors[yz][rowType][3] = param[3] * param[3] * rec.tpc.clusterErrorOccupancyScaler * rec.tpc.clusterErrorOccupancyScaler;
191  }
192  }
193 #endif
194 }
