116void GPUParam::UpdateSettings(const GPUSettingsGRP* g, const GPUSettingsProcessing* p, const GPURecoStepConfiguration* w, const GPUSettingsRecDynamic* d)
117{
118 if (g) {
119 UpdateBzOnly(g->solenoidBzNominalGPU, g->constBz);
120 par.continuousTracking = g->grpContinuousMaxTimeBin != 0;
121 continuousMaxTimeBin = g->grpContinuousMaxTimeBin == -1 ? GPUSettings::TPC_MAX_TF_TIME_BIN : g->grpContinuousMaxTimeBin;
122 tpcCutTimeBin = g->tpcCutTimeBin;
123 }
124 qptB5Scaler = CAMath::Abs(bzkG) > 0.1f ? CAMath::Abs(bzkG) / 5.006680f : 1.f; // Repeat here, since passing in g is optional
125 if (p) {
126 UpdateRun3ClusterErrors(p->param.tpcErrorParamY, p->param.tpcErrorParamZ);
127 }
128 if (w) {
129 par.dodEdx = dodEdxEnabled = w->steps.isSet(GPUDataTypes::RecoStep::TPCdEdx);
130 if (dodEdxEnabled && p && p->tpcDownscaledEdx != 0) {
131 dodEdxEnabled = (rand() % 100) < p->tpcDownscaledEdx;
132 }
133 }
134 if (d) {
135 rec.dyn = *d;
136 }
137}

139void GPUParam::UpdateBzOnly(float newSolenoidBz, bool assumeConstantBz)
140{
141 bzkG = newSolenoidBz;
142 bzCLight = bzkG * o2::gpu::gpu_common_constants::kCLight;
143 polynomialField.Reset();
144 if (assumeConstantBz) {
145 GPUTPCGMPolynomialFieldManager::GetPolynomialField(GPUTPCGMPolynomialFieldManager::kUniform, bzkG, polynomialField);
146 } else {
147 GPUTPCGMPolynomialFieldManager::GetPolynomialField(bzkG, polynomialField);
148 }
149 qptB5Scaler = CAMath::Abs(bzkG) > 0.1f ? CAMath::Abs(bzkG) / 5.006680f : 1.f;
150}

161void GPUParam::UpdateRun3ClusterErrors(const float* yErrorParam, const float* zErrorParam)
162{
163#ifdef GPUCA_TPC_GEOMETRY_O2
164 for (int32_t yz = 0; yz < 2; yz++) {
165 const float* param = yz ? zErrorParam : yErrorParam;
166 for (int32_t rowType = 0; rowType < 4; rowType++) {
167 constexpr int32_t regionMap[4] = {0, 4, 6, 8};
168 ParamErrors[yz][rowType][0] = param[0] * param[0];
169 ParamErrors[yz][rowType][1] = param[1] * param[1] * GPUTPCGeometry::PadHeightByRegion(regionMap[rowType]);
170 ParamErrors[yz][rowType][2] = param[2] * param[2] / GPUTPCGeometry::TPCLength() / GPUTPCGeometry::PadHeightByRegion(regionMap[rowType]);
171 ParamErrors[yz][rowType][3] = param[3] * param[3] * rec.tpc.clusterErrorOccupancyScaler * rec.tpc.clusterErrorOccupancyScaler;
172 }
173 }
174#endif
175}
