117void GPUParam::UpdateSettings(const GPUSettingsGRP* g, const GPUSettingsProcessing* p, const GPURecoStepConfiguration* w, const GPUSettingsRecDynamic* d)
118{
119 if (g) {
120 UpdateBzOnly(g->solenoidBzNominalGPU, g->constBz);
121 par.continuousTracking = g->grpContinuousMaxTimeBin != 0;
122 continuousMaxTimeBin = g->grpContinuousMaxTimeBin == -1 ? GPUSettings::TPC_MAX_TF_TIME_BIN : g->grpContinuousMaxTimeBin;
123 tpcCutTimeBin = g->tpcCutTimeBin;
124 }
125 qptB5Scaler = CAMath::Abs(bzkG) > 0.1f ? CAMath::Abs(bzkG) / 5.006680f : 1.f; // Repeat here, since passing in g is optional
126 if (p) {
127 UpdateRun3ClusterErrors(p->param.tpcErrorParamY, p->param.tpcErrorParamZ);
128 }
129 if (w) {
130 par.dodEdx = dodEdxEnabled = w->steps.isSet(gpudatatypes::RecoStep::TPCdEdx);
131 if (dodEdxEnabled && p && p->tpcDownscaledEdx != 0) {
132 dodEdxEnabled = (rand() % 100) < p->tpcDownscaledEdx;
133 }
134 }
135 if (d) {
136 rec.dyn = *d;
137 }
138}

140void GPUParam::UpdateBzOnly(float newSolenoidBz, bool assumeConstantBz)
141{
142 bzkG = newSolenoidBz;
143 bzCLight = bzkG * o2::gpu::gpu_common_constants::kCLight;
144 polynomialField.Reset();
145 if (assumeConstantBz) {
146 GPUTPCGMPolynomialFieldManager::GetPolynomialField(GPUTPCGMPolynomialFieldManager::kUniform, bzkG, polynomialField);
147 } else {
148 GPUTPCGMPolynomialFieldManager::GetPolynomialField(bzkG, polynomialField);
149 }
150 qptB5Scaler = CAMath::Abs(bzkG) > 0.1f ? CAMath::Abs(bzkG) / 5.006680f : 1.f;
151}

162void GPUParam::UpdateRun3ClusterErrors(const float* yErrorParam, const float* zErrorParam)
163{
164#ifdef GPUCA_TPC_GEOMETRY_O2
165 for (int32_t yz = 0; yz < 2; yz++) {
166 const float* param = yz ? zErrorParam : yErrorParam;
167 for (int32_t rowType = 0; rowType < 4; rowType++) {
168 constexpr int32_t regionMap[4] = {0, 4, 6, 8};
169 ParamErrors[yz][rowType][0] = param[0] * param[0];
170 ParamErrors[yz][rowType][1] = param[1] * param[1] * GPUTPCGeometry::PadHeightByRegion(regionMap[rowType]);
171 ParamErrors[yz][rowType][2] = param[2] * param[2] / GPUTPCGeometry::TPCLength() / GPUTPCGeometry::PadHeightByRegion(regionMap[rowType]);
172 ParamErrors[yz][rowType][3] = param[3] * param[3] * rec.tpc.clusterErrorOccupancyScaler * rec.tpc.clusterErrorOccupancyScaler;
173 }
174 }
175#endif
176}
