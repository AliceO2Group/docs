283 template <typename Kind>
284 struct IndexBuilder {
285  template <typename Key, typename C1, typename... Cs, typename T1, typename... Ts>
286  static auto indexBuilder(const char* label, std::vector<std::shared_ptr<arrow::Table>>&& tables, framework::pack<C1, Cs...>, framework::pack<T1, Ts...>)
287  {
288  auto pool = arrow::default_memory_pool();
289  SelfIndexColumnBuilder self{C1::columnLabel(), pool};
290  std::unique_ptr<ChunkedArrayIterator> keyIndex = nullptr;
291  int64_t counter = 0;
292  if constexpr (!std::is_same_v<T1, Key>) {
293  keyIndex = std::make_unique<ChunkedArrayIterator>(getIndexToKey<T1, Key>(tables[0].get()));
294  }
295 
296  std::array<std::shared_ptr<framework::SelfIndexColumnBuilder>, sizeof...(Cs)> columnBuilders{ColumnTrait<Cs>::template makeColumnBuilder<framework::pack_element_t<framework::has_type_at_v<Cs>(framework::pack<Cs...>{}), framework::pack<Ts...>>, Key>(
297  tables[framework::has_type_at_v<Cs>(framework::pack<Cs...>{}) + 1].get(),
298  pool)...};
299  std::array<bool, sizeof...(Cs)> finds;
300 
301  for (counter = 0; counter < tables[0]->num_rows(); ++counter) {
302  auto idx = -1;
303  if constexpr (std::is_same_v<T1, Key>) {
304  idx = counter;
305  } else {
306  idx = keyIndex->valueAt(counter);
307  }
308  finds = {std::static_pointer_cast<typename Reduction<Key, Cs>::type>(columnBuilders[framework::has_type_at_v<Cs>(framework::pack<Cs...>{})])->template find<Cs>(idx)...};
309  if constexpr (std::is_same_v<Kind, Sparse>) {
310  (std::static_pointer_cast<typename Reduction<Key, Cs>::type>(columnBuilders[framework::has_type_at_v<Cs>(framework::pack<Cs...>{})])->template fill<Cs>(idx), ...);
311  self.fill<C1>(counter);
312  } else if constexpr (std::is_same_v<Kind, Exclusive>) {
313  if (std::none_of(finds.begin(), finds.end(), [](bool const x) { return x == false; })) {
314  (std::static_pointer_cast<typename Reduction<Key, Cs>::type>(columnBuilders[framework::has_type_at_v<Cs>(framework::pack<Cs...>{})])->template fill<Cs>(idx), ...);
315  self.fill<C1>(counter);
316  }
317  }
318  }
319 
320  return makeArrowTable(label,
321  {self.template result<C1>(), std::static_pointer_cast<typename Reduction<Key, Cs>::type>(columnBuilders[framework::has_type_at_v<Cs>(framework::pack<Cs...>{})])->template result<Cs>()...},
322  {self.field(), std::static_pointer_cast<typename Reduction<Key, Cs>::type>(columnBuilders[framework::has_type_at_v<Cs>(framework::pack<Cs...>{})])->field()...});
323  }

325  template <typename IDX, typename Key, typename T1, typename... T>

328  auto t = IDX{indexBuilder(o2::aod::MetadataTrait<IDX>::metadata::tableLabel(),
