43 GPUdii() void GPUTPCDecompressionKernels::decompressTrack(CompressedClusters& cmprClusters, const GPUParam& param, const unsigned int maxTime, const unsigned int trackIndex, unsigned int& clusterOffset, GPUTPCDecompression& decompressor)
44 {
45  float zOffset = 0;
46  unsigned int slice = cmprClusters.sliceA[trackIndex];
47  unsigned int row = cmprClusters.rowA[trackIndex];
48  GPUTPCCompressionTrackModel track;
49  unsigned int clusterIndex;
50  for (clusterIndex = 0; clusterIndex < cmprClusters.nTrackClusters[trackIndex]; clusterIndex++) {
51  unsigned int pad = 0, time = 0;
52  if (clusterIndex != 0) {
53  unsigned char tmpSlice = cmprClusters.sliceLegDiffA[clusterOffset - trackIndex - 1];
54  bool changeLeg = (tmpSlice >= GPUCA_NSLICES);
55  if (changeLeg) {
56  tmpSlice -= GPUCA_NSLICES;
57  }
58  if (cmprClusters.nComppressionModes & GPUSettings::CompressionDifferences) {
59  slice += tmpSlice;
60  if (slice >= GPUCA_NSLICES) {
61  slice -= GPUCA_NSLICES;
62  }
63  row += cmprClusters.rowDiffA[clusterOffset - trackIndex - 1];
64  if (row >= GPUCA_ROW_COUNT) {
65  row -= GPUCA_ROW_COUNT;
66  }
67  } else {
68  slice = tmpSlice;
69  row = cmprClusters.rowDiffA[clusterOffset - trackIndex - 1];
70  }
71  if (changeLeg && track.Mirror()) {
72  break;
73  }
74  if (track.Propagate(param.tpcGeometry.Row2X(row), param.SliceParam[slice].Alpha)) {
75  break;
76  }
77  unsigned int timeTmp = cmprClusters.timeResA[clusterOffset - trackIndex - 1];
78  if (timeTmp & 800000) {
79  timeTmp |= 0xFF000000;
80  }
81  time = timeTmp + ClusterNative::packTime(CAMath::Max(0.f, param.tpcGeometry.LinearZ2Time(slice, track.Z() + zOffset)));
82  float tmpPad = CAMath::Max(0.f, CAMath::Min((float)param.tpcGeometry.NPads(GPUCA_ROW_COUNT - 1), param.tpcGeometry.LinearY2Pad(slice, row, track.Y())));
83  pad = cmprClusters.padResA[clusterOffset - trackIndex - 1] + ClusterNative::packPad(tmpPad);
84  time = time & 0xFFFFFF;
85  pad = (unsigned short)pad;
86  if (pad >= param.tpcGeometry.NPads(row) * ClusterNative::scalePadPacked) {
87  if (pad >= 0xFFFF - 11968) { // Constant 11968 = (2^15 - MAX_PADS(138) * scalePadPacked(64)) / 2
88  pad = 0;
89  } else {
90  pad = param.tpcGeometry.NPads(row) * ClusterNative::scalePadPacked - 1;
91  }
92  }
93  if (param.par.continuousMaxTimeBin > 0 && time >= maxTime) {
94  if (time >= 0xFFFFFF - 544768) { // Constant 544768 = (2^23 - LHCMAXBUNCHES(3564) * MAXORBITS(256) * scaleTimePacked(64) / BCPERTIMEBIN(8)) / 2)
95  time = 0;
96  } else {
97  time = maxTime;
98  }
99  }
100  } else {
101  time = cmprClusters.timeA[trackIndex];
102  pad = cmprClusters.padA[trackIndex];
103  }
104  const auto cluster = decompressTrackStore(cmprClusters, clusterOffset, slice, row, pad, time, decompressor);
105  float y = param.tpcGeometry.LinearPad2Y(slice, row, cluster.getPad());
106  float z = param.tpcGeometry.LinearTime2Z(slice, cluster.getTime());
107  if (clusterIndex == 0) {
108  zOffset = z;
109  track.Init(param.tpcGeometry.Row2X(row), y, z - zOffset, param.SliceParam[slice].Alpha, cmprClusters.qPtA[trackIndex], param);
110  }
111  if (clusterIndex + 1 < cmprClusters.nTrackClusters[trackIndex] && track.Filter(y, z - zOffset, row)) {
112  break;
113  }
114  clusterOffset++;
115  }
116  clusterOffset += cmprClusters.nTrackClusters[trackIndex] - clusterIndex;
117 }
