110int32_t GPUTRDTrackerComponent::ReadConfigurationString(const char* arguments)
111{
112 // Set configuration parameters for the TRD tracker component from the string
113
114 int32_t iResult = 0;
115 if (!arguments) {
116 return iResult;
117 }
118
119 TString allArgs = arguments;
120 TString argument;
121
122 TObjArray* pTokens = allArgs.Tokenize(" ");
123
124 int32_t nArgs = pTokens ? pTokens->GetEntries() : 0;
125
126 for (int32_t i = 0; i < nArgs; i++) {
127 argument = ((TObjString*)pTokens->At(i))->GetString();
128 if (argument.IsNull()) {
129 continue;
130 }
131
132 if (argument.CompareTo("-debugOutput") == 0) {
133 fDebugTrackOutput = true;
134 fVerboseDebugOutput = true;
135 HLTInfo("Tracks are dumped in the GPUTRDTrackGPU format");
136 continue;
137 }
138
139 if (argument.CompareTo("-requireITStrack") == 0) {
140 fRequireITStrack = true;
141 HLTInfo("TRD tracker requires seeds (TPC tracks) to have an ITS match");
142 continue;
143 }
144
145 HLTError("Unknown option \"%s\"", argument.Data());
146 iResult = -EINVAL;
147 }
148 delete pTokens;
149
150 return iResult;
151}

154int32_t GPUTRDTrackerComponent::DoInit(int argc, const char** argv)
155{
156 // see header file for class documentation
157
158 int32_t iResult = 0;
159 if (fTracker) {
160 return -EINPROGRESS;
161 }
162
163 fBenchmark.Reset();
164 fBenchmark.SetTimer(0, "total");
165 fBenchmark.SetTimer(1, "reco");
166
167 if (AliGeomManager::GetGeometry() == nullptr) {
168 AliGeomManager::LoadGeometry();
169 }
170
171 fTrackList = new TList();
172 if (!fTrackList) {
173 return -ENOMEM;
174 }
175 fTrackList->SetOwner(kFALSE);
176
177 TString arguments = "";
178 for (int32_t i = 0; i < argc; i++) {
179 if (!arguments.IsNull()) {
180 arguments += " ";
181 }
182 arguments += argv[i];
183 }
184
185 iResult = ReadConfigurationString(arguments.Data());
186
187 GPUSettingsGRP cfgGRP;
188 cfgGRP.solenoidBzNominalGPU = GetBz();
189 GPUSettingsRec cfgRec;
190 GPUSettingsProcessing cfgDeviceProcessing;
191 GPURecoStepConfiguration cfgRecoStep;
192 cfgRecoStep.steps = GPUDataTypes::RecoStep::NoRecoStep;
193 cfgRecoStep.inputs.clear();
194 cfgRecoStep.outputs.clear();
195 fRec = GPUReconstruction::CreateInstance("CPU", true);
196 fRec->SetSettings(&cfgGRP, &cfgRec, &cfgDeviceProcessing, &cfgRecoStep);
197 fChain = fRec->AddChain<GPUChainTracking>();
198
199 fGeo = new GPUTRDGeometry();
200 if (!fGeo) {
201 return -ENOMEM;
202 }
203 if (!GPUTRDGeometry::CheckGeometryAvailable()) {
204 HLTError("TRD geometry not available");
205 return -EINVAL;
206 }
207 fTracker = new GPUTRDTrackerGPU();
208 if (!fTracker) {
209 return -ENOMEM;
210 }
211 if (fVerboseDebugOutput) {
212 fTracker->EnableDebugOutput();
213 }
214 fRec->RegisterGPUProcessor(fTracker, false);
215 fChain->SetTRDGeometry(reinterpret_cast<o2::trd::GeometryFlat*>(fGeo));
216 if (fRec->Init()) {
217 return -EINVAL;
218 }
219
220 return iResult;
221}

235int32_t GPUTRDTrackerComponent::DoEvent(const AliHLTComponentEventData& evtData, const AliHLTComponentBlockData* blocks, AliHLTComponentTriggerData& /*trigData*/, AliHLTUInt8_t* outputPtr, AliHLTUInt32_t& size, std::vector<AliHLTComponentBlockData>& outputBlocks)
236{
237 // process event
238
239 if (!IsDataEvent()) {
240 return 0;
241 }
242
243 if (evtData.fBlockCnt <= 0) {
244 HLTWarning("no blocks in event");
245 return 0;
246 }
247
248 fBenchmark.StartNewEvent();
249 fBenchmark.Start(0);
250
251 AliHLTUInt32_t maxBufferSize = size;
252 size = 0; // output size
253
254 int32_t iResult = 0;
255
256 if (fTrackList->GetEntries() != 0) {
257 fTrackList->Clear(); // tracks are owned by GPUTRDTrackerGPU
258 }
259
260 int32_t nBlocks = evtData.fBlockCnt;
261
262 const AliHLTTracksData* tpcData = nullptr;
263 const AliHLTTracksData* itsData = nullptr;
264 const AliHLTTrackMCData* tpcDataMC = nullptr;
265
266 std::vector<GPUTRDTrackGPU> tracksTPC;
267 std::vector<int32_t> tracksTPCId;
268
269 bool hasMCtracklets = false;
270
271 int32_t nTrackletsTotal = 0;
272 int32_t nTrackletsTotalMC = 0;
273 const GPUTRDTrackletWord* tracklets = nullptr;
274 const GPUTRDTrackletLabels* trackletsMC = nullptr;
275
276 for (int32_t iBlock = 0; iBlock < nBlocks; iBlock++) {
277 if (blocks[iBlock].fDataType == (kAliHLTDataTypeTrack | kAliHLTDataOriginITS) && fRequireITStrack) {
278 itsData = (const AliHLTTracksData*)blocks[iBlock].fPtr;
279 fBenchmark.AddInput(blocks[iBlock].fSize);
280 } else if (blocks[iBlock].fDataType == (AliHLTTPCDefinitions::TracksOuterDataType() | kAliHLTDataOriginTPC)) {
281 tpcData = (const AliHLTTracksData*)blocks[iBlock].fPtr;
282 fBenchmark.AddInput(blocks[iBlock].fSize);
283 } else if (blocks[iBlock].fDataType == (kAliHLTDataTypeTrackMC | kAliHLTDataOriginTPC)) {
284 tpcDataMC = (const AliHLTTrackMCData*)blocks[iBlock].fPtr;
285 fBenchmark.AddInput(blocks[iBlock].fSize);
286 } else if (blocks[iBlock].fDataType == (AliHLTTRDDefinitions::fgkTRDTrackletDataType)) {
287 tracklets = reinterpret_cast<const GPUTRDTrackletWord*>(blocks[iBlock].fPtr);
288 nTrackletsTotal = blocks[iBlock].fSize / sizeof(GPUTRDTrackletWord);
289 fBenchmark.AddInput(blocks[iBlock].fSize);
290 } else if (blocks[iBlock].fDataType == (AliHLTTRDDefinitions::fgkTRDMCTrackletDataType)) {
291 hasMCtracklets = true;
292 trackletsMC = reinterpret_cast<const GPUTRDTrackletLabels*>(blocks[iBlock].fPtr);
293 nTrackletsTotalMC = blocks[iBlock].fSize / sizeof(GPUTRDTrackletLabels);
294 fBenchmark.AddInput(blocks[iBlock].fSize);
295 }
296 }
297
298 if (tpcData == nullptr) {
299 HLTInfo("did not receive any TPC tracks. Skipping event");
300 return 0;
301 }
302
303 if (nTrackletsTotal == 0) {
304 HLTInfo("did not receive any TRD tracklets. Skipping event");
305 return 0;
306 }
307
308 if (hasMCtracklets && nTrackletsTotal != nTrackletsTotalMC) {
309 HLTError("the numbers of input tracklets does not match the number of input MC labels for them");
310 return -EINVAL;
311 }
312
313 // copy tracklets into temporary vector to allow for sorting them (the input array is const)
314 std::vector<GPUTRDTrackletWord> trackletsTmp(nTrackletsTotal);
315 for (int32_t iTrklt = 0; iTrklt < nTrackletsTotal; ++iTrklt) {
316 trackletsTmp[iTrklt] = tracklets[iTrklt];
317 }
318
319 int32_t nTPCtracks = tpcData->fCount;
320 std::vector<bool> itsAvail(nTPCtracks, false);
321 if (itsData) {
322 // look for ITS tracks with >= 2 hits
323 int32_t nITStracks = itsData->fCount;
324 const AliHLTExternalTrackParam* currITStrack = itsData->fTracklets;
325 for (int32_t iTrkITS = 0; iTrkITS < nITStracks; iTrkITS++) {
326 if (currITStrack->fNPoints >= 2) {
327 itsAvail.at(currITStrack->fTrackID) = true;
328 }
329 uint32_t dSize = sizeof(AliHLTExternalTrackParam) + currITStrack->fNPoints * sizeof(uint32_t);
330 currITStrack = (AliHLTExternalTrackParam*)(((Byte_t*)currITStrack) + dSize);
331 }
332 }
333 std::map<int32_t, int32_t> mcLabels;
334 if (tpcDataMC) {
335 // look for TPC track MC labels
336 int32_t nMCtracks = tpcDataMC->fCount;
337 for (int32_t iMC = 0; iMC < nMCtracks; iMC++) {
338 const AliHLTTrackMCLabel& lab = tpcDataMC->fLabels[iMC];
339 mcLabels[lab.fTrackID] = lab.fMCLabel;
340 }
341 }
342 const AliHLTExternalTrackParam* currOutTrackTPC = tpcData->fTracklets;
343 for (int32_t iTrk = 0; iTrk < nTPCtracks; iTrk++) {
344 // store TPC tracks (if required only the ones with >=2 ITS hits)
345 if (itsData != nullptr && !itsAvail.at(currOutTrackTPC->fTrackID)) {
346 continue;
347 }
348 GPUTRDTrackGPU t(*currOutTrackTPC);
349 int32_t mcLabel = -1;
350 if (tpcDataMC) {
351 if (mcLabels.find(currOutTrackTPC->fTrackID) != mcLabels.end()) {
352 mcLabel = mcLabels[currOutTrackTPC->fTrackID];
353 }
354 }
355 tracksTPC.push_back(t);
356 tracksTPCId.push_back(currOutTrackTPC->fTrackID);
357 uint32_t dSize = sizeof(AliHLTExternalTrackParam) + currOutTrackTPC->fNPoints * sizeof(uint32_t);
358 currOutTrackTPC = (AliHLTExternalTrackParam*)+(((Byte_t*)currOutTrackTPC) + dSize);
359 }
360
361 if (fVerboseDebugOutput) {
362 HLTInfo("TRDTrackerComponent received %i tracklets\n", nTrackletsTotal);
363 }
364
365 fTracker->SetGenerateSpacePoints(true);
366 fTracker->Reset();
367 fChain->mIOPtrs.nMergedTracks = tracksTPC.size();
368 fChain->mIOPtrs.nTRDTracklets = nTrackletsTotal;
369 fChain->mIOPtrs.nTRDTriggerRecords = 1;
370 uint8_t trigRecMaskDummy[1] = {1};
371 fChain->mIOPtrs.trdTrigRecMask = &(trigRecMaskDummy[0]);
372 fRec->PrepareEvent();
373 fRec->SetupGPUProcessor(fTracker, true);
374
375 std::sort(trackletsTmp.begin(), trackletsTmp.end());
376 fChain->mIOPtrs.trdTracklets = &(trackletsTmp[0]);
377
378 // loop over all tracks
379 for (uint32_t iTrack = 0; iTrack < tracksTPC.size(); ++iTrack) {
380 fTracker->LoadTrack(tracksTPC[iTrack], tracksTPCId[iTrack]);
381 }
382
383 fBenchmark.Start(1);
384 fChain->DoTRDGPUTracking<1>(fTracker);
385 fBenchmark.Stop(1);
386
387 GPUTRDTrackGPU* trackArray = fTracker->Tracks();
388 int32_t nTracks = fTracker->NTracks();
389 GPUTRDSpacePoint* spacePoints = fTracker->SpacePoints();
390
391 // TODO delete fTrackList since it only works for TObjects (or use compiler flag after tests with GPU track type)
392 // for (int32_t iTrack=0; iTrack<nTracks; ++iTrack) {
393 // fTrackList->AddLast(&trackArray[iTrack]);
394 //}
395
396 // push back GPUTRDTracks for debugging purposes
397 if (fDebugTrackOutput) {
398 PushBack(fTrackList, (kAliHLTDataTypeTObject | kAliHLTDataOriginTRD), 0x3fffff);
399 }
400 // push back AliHLTExternalTrackParam (default)
401 else {
402
403 AliHLTUInt32_t blockSize = GPUTRDTrackData::GetSize(nTracks);
404 if (size + blockSize > maxBufferSize) {
405 HLTWarning("Output buffer exceeded for tracks");
406 return -ENOSPC;
407 }
408
409 GPUTRDTrackData* outTracks = (GPUTRDTrackData*)(outputPtr);
410 outTracks->fCount = 0;
411 int32_t assignedTracklets = 0;
412
413 for (int32_t iTrk = 0; iTrk < nTracks; ++iTrk) {
414 GPUTRDTrackGPU& t = trackArray[iTrk];
415 if (t.getNtracklets() == 0) {
416 continue;
417 }
418 assignedTracklets += t.getNtracklets();
419 GPUTRDTrackDataRecord& currOutTrack = outTracks->fTracks[outTracks->fCount];
420 t.ConvertTo(currOutTrack);
421 outTracks->fCount++;
422 }
423
424 AliHLTComponentBlockData resultData;
425 FillBlockData(resultData);
426 resultData.fOffset = size;
427 resultData.fSize = blockSize;
428 resultData.fDataType = AliHLTTRDDefinitions::fgkTRDTrackDataType;
429 outputBlocks.push_back(resultData);
430 fBenchmark.AddOutput(resultData.fSize);
431
432 size += blockSize;
433 outputPtr += resultData.fSize;
434
435 blockSize = 0;
436
437 // space points calculated from tracklets
438
439 blockSize = sizeof(GPUTRDTrackPointData) + sizeof(GPUTRDTrackPoint) * nTrackletsTotal;
440
441 if (size + blockSize > maxBufferSize) {
442 HLTWarning("Output buffer exceeded for space points");
443 return -ENOSPC;
444 }
445
446 GPUTRDTrackPointData* outTrackPoints = (GPUTRDTrackPointData*)(outputPtr);
447 outTrackPoints->fCount = nTrackletsTotal;
448
449 { // fill array with 0 for a case..
450 GPUTRDTrackPoint empty;
451 empty.fX[0] = 0;
452 empty.fX[1] = 0;
453 empty.fX[2] = 0;
454 empty.fVolumeId = 0;
455 for (int32_t i = 0; i < nTrackletsTotal; ++i) {
456 outTrackPoints->fPoints[i] = empty;
457 }
458 }
459
460 for (int32_t i = 0; i < nTrackletsTotal; ++i) {
461 const GPUTRDSpacePoint& sp = spacePoints[i];
462 GPUTRDTrackPoint* currOutPoint = &outTrackPoints->fPoints[i];
463 currOutPoint->fX[0] = sp.getX(); // x in sector coordinates
464 currOutPoint->fX[1] = sp.getY(); // y in sector coordinates
465 currOutPoint->fX[2] = sp.getZ(); // z in sector coordinates
466 int32_t detId = trackletsTmp[i].GetDetector();
467 int32_t layer = detId % 6; // TRD layer number for given detector
468 int32_t modId = (detId / 18) * 5 + ((detId % 30) / 6); // global TRD stack number [0..89]
469 int32_t volId = (UShort_t(9 + layer) << 11) | UShort_t(modId); // taken from AliGeomManager::LayerToVolUID(). AliGeomManager::ELayerID(AliGeomManager::kTRD1) == 9
470 currOutPoint->fVolumeId = volId;
471 }
472 AliHLTComponentBlockData resultDataSP;
473 FillBlockData(resultDataSP);
474 resultDataSP.fOffset = size;
475 resultDataSP.fSize = blockSize;
476 resultDataSP.fDataType = AliHLTTRDDefinitions::fgkTRDTrackPointDataType | kAliHLTDataOriginTRD;
477 outputBlocks.push_back(resultDataSP);
478 fBenchmark.AddOutput(resultData.fSize);
479 size += blockSize;
480 outputPtr += resultDataSP.fSize;
481
482 HLTInfo("TRD tracker: output %d tracks (%d assigned tracklets) and %d track points", outTracks->fCount, assignedTracklets, outTrackPoints->fCount);
483 }
484
485 fBenchmark.Stop(0);
486 HLTInfo(fBenchmark.GetStatistics());
487
488 return iResult;
489}
