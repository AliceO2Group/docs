39  Configurable<std::string> vtxModeArg{"vertexMode", "kDiamondParam", "Where the beam-spot vertex should come from. Must be one of kNoVertex, kDiamondParam, kCCDB"};
40  Configurable<int64_t> ttl{"time-limit", -1, "Maximum run time limit in seconds (default no limit)"};
41  Configurable<std::string> outputPrefix{"output", "", "Optional prefix for kinematics files written on disc. If non-empty, files <prefix>_Kine.root + <prefix>_MCHeader.root will be created."};
42  GenCount nEvents = 0;

48  // a pointer because object should only be constructed in the device (not during DPL workflow setup)
49  std::unique_ptr<o2::eventgen::GeneratorService> genservice;
50  TStopwatch timer;

52  void init(o2::framework::InitContext& /*ic*/)
53  {
54  genservice.reset(new o2::eventgen::GeneratorService);
55  o2::utils::RngHelper::setGRandomSeed(seed);
56  nEvents = eventNum;
57  // helper to parse vertex option; returns true if parsing ok, false if failure
58  o2::conf::VertexMode vtxmode;
59  if (!(o2::conf::SimConfig::parseVertexModeString(vtxModeArg, vtxmode))) {
60  LOG(error) << "Could not parse vtxMode";
61  }
62 
63  // update config key params
64  o2::conf::ConfigurableParam::updateFromFile(iniFile);
65  o2::conf::ConfigurableParam::updateFromString((std::string)params);
66  // initialize the service
67  if (vtxmode == o2::conf::VertexMode::kDiamondParam) {
68  genservice->initService(generator, trigger, o2::eventgen::DiamondParamVertexOption());
69  } else if (vtxmode == o2::conf::VertexMode::kNoVertex) {
70  genservice->initService(generator, trigger, o2::eventgen::NoVertexOption());
71  } else if (vtxmode == o2::conf::VertexMode::kCCDB) {
72  LOG(warn) << "Not yet supported. This needs definition of a timestamp and fetching of the MeanVertex CCDB object";
73  }
74  timer.Start();
75 
76  if (outputPrefix->size() > 0 && !outfile.get()) {
77  auto kineoutfilename = o2::base::NameConf::getMCKinematicsFileName(outputPrefix->c_str());
78  outfile.reset(new TFile(kineoutfilename.c_str(), "RECREATE"));
79  outtree.reset(new TTree("o2sim", "o2sim"));
80  }
81  }

83  void run(o2::framework::ProcessingContext& pc)
84  {
85  std::vector<o2::MCTrack> mctracks;
86  o2::dataformats::MCEventHeader mcheader;
87  auto mctrack_ptr = &mctracks;
88  if (outfile.get()) {
89  auto br = o2::base::getOrMakeBranch(*outtree, "MCTrack", &mctrack_ptr);
90  br->SetAddress(&mctrack_ptr);
91  }
92 
93  for (auto i = 0; i < std::min((GenCount)aggregate, nEvents - eventCounter); ++i) {
94  mctracks.clear();
95  genservice->generateEvent_MCTracks(mctracks, mcheader);
96  pc.outputs().snapshot(Output{"MC", "MCHEADER", 0}, mcheader);
97  pc.outputs().snapshot(Output{"MC", "MCTRACKS", 0}, mctracks);
98  ++eventCounter;
99 
100  if (outfile.get() && outtree.get()) {
101  outtree->Fill();
102  }
103  }
104 
105  // report number of TFs injected for the rate limiter to work
106  ++tfCounter;
107  pc.services().get<o2::monitoring::Monitoring>().send(o2::monitoring::Metric{(uint64_t)tfCounter, "df-sent"}.addTag(o2::monitoring::tags::Key::Subsystem, o2::monitoring::tags::Value::DPL));
108  bool time_expired = false;
109  if (ttl > 0) {
110  timer.Stop();
111  time_expired = timer.RealTime() > ttl;
112  timer.Start(false);
113  if (time_expired) {
114  LOG(info) << "TTL expired after " << eventCounter << " events ... sending end-of-stream";
115  }
116  }
117  if (eventCounter >= nEvents || time_expired) {
118  pc.services().get<ControlService>().endOfStream();
119  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
120 
121  // write out data to disc if asked
122  if (outfile.get()) {
123  outtree->SetEntries(eventCounter);
124  outtree->Write();
125  outfile->Close();
126  }
127  }
128  }
129 };
