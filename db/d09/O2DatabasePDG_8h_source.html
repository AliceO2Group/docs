26 // An ALICE specific extension of ROOT's TDatabasePDG
27 //
28 // By using O2DatabasePDG::Instance() in our code instead of TDatabasePDG::Instance(), correct initialization
29 // is guaranteed. Alternatively, a static function is exposed with which particles can be added to TDatabasePDG objects
30 // directly.
31 class O2DatabasePDG
32 {
33  //
34  public:
35  static TDatabasePDG* Instance()
36  {
37  static bool initialized = false; // initialize this --> adds particles to TDatabasePDG;
38  auto db = TDatabasePDG::Instance();
39  if (!initialized) {
40  addALICEParticles(db);
41  if (const char* o2Root = std::getenv("O2_ROOT")) {
42  auto inputExtraPDGs = std::string(o2Root) + "/share/Detectors/gconfig/data/extra_ions_pdg_table.dat";
43  db->ReadPDGTable(inputExtraPDGs.c_str());
44  }
45  initialized = true;
46  }
47  return db;
48  }
49 
50  // adds ALICE particles to a given TDatabasePDG instance
51  static void addALICEParticles(TDatabasePDG* db = TDatabasePDG::Instance());
52 
53  // get particle's (if any) mass
54  static Double_t MassImpl(TParticlePDG* particle, bool& success)
55  {
56  success = false;
57  if (!particle) {
58  return -1.;
59  }
60  success = true;
61  return particle->Mass();
62  }
63 
64  // determine particle to get mass for based on PDG
65  static Double_t Mass(int pdg, bool& success, TDatabasePDG* db = O2DatabasePDG::Instance())
66  {
67  if (pdg < IONBASELOW || pdg > IONBASEHIGH) {
68  // not an ion, return immediately
69  return MassImpl(db->GetParticle(pdg), success);
70  }
71  if (auto particle = db->GetParticle(pdg)) {
72  // see if this ion can be found
73  return MassImpl(particle, success);
74  }
75  // if we are here, try one last time to look for ground state of potential isomere state
76  pdg = pdg / 10 * 10;
77  return MassImpl(db->GetParticle(pdg), success);
78  }
79 
80  // remove default constructor
81  O2DatabasePDG() = delete;
82 
83  private:
84  static constexpr int IONBASELOW{1000000000};
85  static constexpr int IONBASEHIGH{1099999999};
86 };
