139int32_t GPUReconstruction::Init()
140{
141 if (mMaster) {
142 throw std::runtime_error("Must not call init on slave!");
143 }
144 int32_t retVal = InitPhaseBeforeDevice();
145 if (retVal) {
146 return retVal;
147 }
148 for (uint32_t i = 0; i < mSlaves.size(); i++) {
149 retVal = mSlaves[i]->InitPhaseBeforeDevice();
150 if (retVal) {
151 GPUError("Error initialization slave (before deviceinit)");
152 return retVal;
153 }
154 mNStreams = std::max(mNStreams, mSlaves[i]->mNStreams);
155 mHostMemorySize = std::max(mHostMemorySize, mSlaves[i]->mHostMemorySize);
156 mDeviceMemorySize = std::max(mDeviceMemorySize, mSlaves[i]->mDeviceMemorySize);
157 }
158 if (InitDevice()) {
159 return 1;
160 }
161 mHostMemoryPoolEnd = (char*)mHostMemoryBase + mHostMemorySize;
162 mDeviceMemoryPoolEnd = (char*)mDeviceMemoryBase + mDeviceMemorySize;
163 if (InitPhasePermanentMemory()) {
164 return 1;
165 }
166 for (uint32_t i = 0; i < mSlaves.size(); i++) {
167 mSlaves[i]->mDeviceMemoryBase = mDeviceMemoryPermanent;
168 mSlaves[i]->mHostMemoryBase = mHostMemoryPermanent;
169 mSlaves[i]->mDeviceMemorySize = mDeviceMemorySize - ptrDiff(mSlaves[i]->mDeviceMemoryBase, mDeviceMemoryBase);
170 mSlaves[i]->mHostMemorySize = mHostMemorySize - ptrDiff(mSlaves[i]->mHostMemoryBase, mHostMemoryBase);
171 mSlaves[i]->mHostMemoryPoolEnd = mHostMemoryPoolEnd;
172 mSlaves[i]->mDeviceMemoryPoolEnd = mDeviceMemoryPoolEnd;
173 if (mSlaves[i]->InitDevice()) {
174 GPUError("Error initialization slave (deviceinit)");
175 return 1;
176 }
177 if (mSlaves[i]->InitPhasePermanentMemory()) {
178 GPUError("Error initialization slave (permanent memory)");
179 return 1;
180 }
181 mDeviceMemoryPermanent = mSlaves[i]->mDeviceMemoryPermanent;
182 mHostMemoryPermanent = mSlaves[i]->mHostMemoryPermanent;
183 }
184 retVal = InitPhaseAfterDevice();
185 if (retVal) {
186 return retVal;
187 }
188 ClearAllocatedMemory();
189 for (uint32_t i = 0; i < mSlaves.size(); i++) {
190 mSlaves[i]->mDeviceMemoryPermanent = mDeviceMemoryPermanent;
191 mSlaves[i]->mHostMemoryPermanent = mHostMemoryPermanent;
192 retVal = mSlaves[i]->InitPhaseAfterDevice();
193 if (retVal) {
194 GPUError("Error initialization slave (after device init)");
195 return retVal;
196 }
197 mSlaves[i]->ClearAllocatedMemory();
198 }
199 return 0;
200}

452int32_t GPUReconstruction::Exit()
453{
454 if (!mInitialized) {
455 return 1;
456 }
457 for (uint32_t i = 0; i < mSlaves.size(); i++) {
458 if (mSlaves[i]->Exit()) {
459 GPUError("Error exiting slave");
460 }
461 }
462
463 mChains.clear(); // Make sure we destroy a possible ITS GPU tracker before we call the destructors
464 mHostConstantMem.reset(); // Reset these explicitly before the destruction of other members unloads the library
465 if (mProcessingSettings.memoryAllocationStrategy == GPUMemoryResource::ALLOCATION_INDIVIDUAL) {
466 for (uint32_t i = 0; i < mMemoryResources.size(); i++) {
467 if (mMemoryResources[i].mReuse >= 0) {
468 continue;
469 }
470 operator delete(mMemoryResources[i].mPtrDevice GPUCA_OPERATOR_NEW_ALIGNMENT);
471 mMemoryResources[i].mPtr = mMemoryResources[i].mPtrDevice = nullptr;
472 }
473 }
474 mMemoryResources.clear();
475 if (mInitialized) {
476 ExitDevice();
477 }
478 mInitialized = false;
479 return 0;
480}

540size_t GPUReconstruction::AllocateRegisteredMemoryHelper(GPUMemoryResource* res, void*& ptr, void*& memorypool, void* memorybase, size_t memorysize, void* (GPUMemoryResource::*setPtr)(void*), void*& memorypoolend, const char* device)
541{
542 if (res->mReuse >= 0) {
543 ptr = (&ptr == &res->mPtrDevice) ? mMemoryResources[res->mReuse].mPtrDevice : mMemoryResources[res->mReuse].mPtr;
544 if (ptr == nullptr) {
545 GPUError("Invalid reuse ptr (%s)", res->mName);
546 throw std::bad_alloc();
547 }
548 size_t retVal = ptrDiff((res->*setPtr)(ptr), ptr);
549 if (retVal > mMemoryResources[res->mReuse].mSize) {
550 GPUError("Insufficient reuse memory %lu < %lu (%s) (%s)", mMemoryResources[res->mReuse].mSize, retVal, res->mName, device);
551 throw std::bad_alloc();
552 }
553 if (mProcessingSettings.allocDebugLevel >= 2) {
554 std::cout << "Reused (" << device << ") " << res->mName << ": " << retVal << "\n";
555 }
556 return retVal;
557 }
558 if (memorypool == nullptr) {
559 GPUError("Cannot allocate memory from uninitialized pool");
560 throw std::bad_alloc();
561 }
562 size_t retVal;
563 if ((res->mType & GPUMemoryResource::MEMORY_STACK) && memorypoolend) {
564 retVal = ptrDiff((res->*setPtr)((char*)1), (char*)(1));
565 memorypoolend = (void*)((char*)memorypoolend - GPUProcessor::getAlignmentMod<GPUCA_MEMALIGN>(memorypoolend));
566 if (retVal < res->mOverrideSize) {
567 retVal = res->mOverrideSize;
568 }
569 retVal += GPUProcessor::getAlignment<GPUCA_MEMALIGN>(retVal);
570 memorypoolend = (char*)memorypoolend - retVal;
571 ptr = memorypoolend;
572 retVal = std::max<size_t>(ptrDiff((res->*setPtr)(ptr), ptr), res->mOverrideSize);
573 } else {
574 ptr = memorypool;
575 memorypool = (char*)((res->*setPtr)(ptr));
576 retVal = ptrDiff(memorypool, ptr);
577 if (retVal < res->mOverrideSize) {
578 retVal = res->mOverrideSize;
579 memorypool = (char*)ptr + res->mOverrideSize;
580 }
581 memorypool = (void*)((char*)memorypool + GPUProcessor::getAlignment<GPUCA_MEMALIGN>(memorypool));
582 }
583 if (memorypoolend ? (memorypool > memorypoolend) : ((size_t)ptrDiff(memorypool, memorybase) > memorysize)) {
584 std::cerr << "Memory pool size exceeded (" << device << ") (" << res->mName << ": " << (memorypoolend ? (memorysize + ptrDiff(memorypool, memorypoolend)) : ptrDiff(memorypool, memorybase)) << " < " << memorysize << "\n";
585 throw std::bad_alloc();
586 }
587 if (mProcessingSettings.allocDebugLevel >= 2) {
588 std::cout << "Allocated (" << device << ") " << res->mName << ": " << retVal << " - available: " << (memorypoolend ? ptrDiff(memorypoolend, memorypool) : (memorysize - ptrDiff(memorypool, memorybase))) << "\n";
589 }
590 return retVal;
591}

593void GPUReconstruction::AllocateRegisteredMemoryInternal(GPUMemoryResource* res, GPUOutputControl* control, GPUReconstruction* recPool)
594{
595 if (mProcessingSettings.memoryAllocationStrategy == GPUMemoryResource::ALLOCATION_INDIVIDUAL && (control == nullptr || control->useInternal())) {
596 if (!(res->mType & GPUMemoryResource::MEMORY_EXTERNAL)) {
597 if (res->mPtrDevice && res->mReuse < 0) {
598 operator delete(res->mPtrDevice GPUCA_OPERATOR_NEW_ALIGNMENT);
599 }
600 res->mSize = std::max((size_t)res->SetPointers((void*)1) - 1, res->mOverrideSize);
601 if (res->mReuse >= 0) {
602 if (res->mSize > mMemoryResources[res->mReuse].mSize) {
603 GPUError("Invalid reuse, insufficient size: %ld < %ld", (int64_t)mMemoryResources[res->mReuse].mSize, (int64_t)res->mSize);
604 throw std::bad_alloc();
605 }
606 res->mPtrDevice = mMemoryResources[res->mReuse].mPtrDevice;
607 } else {
608 res->mPtrDevice = operator new(res->mSize + GPUCA_BUFFER_ALIGNMENT GPUCA_OPERATOR_NEW_ALIGNMENT);
609 }
610 res->mPtr = GPUProcessor::alignPointer<GPUCA_BUFFER_ALIGNMENT>(res->mPtrDevice);
611 res->SetPointers(res->mPtr);
612 if (mProcessingSettings.allocDebugLevel >= 2) {
613 std::cout << (res->mReuse >= 0 ? "Reused " : "Allocated ") << res->mName << ": " << res->mSize << "\n";
614 }
615 if (res->mType & GPUMemoryResource::MEMORY_STACK) {
616 mNonPersistentIndividualAllocations.emplace_back(res);
617 }
618 if ((size_t)res->mPtr % GPUCA_BUFFER_ALIGNMENT) {
619 GPUError("Got buffer with insufficient alignment");
620 throw std::bad_alloc();
621 }
622 }
623 } else {
624 if (res->mPtr != nullptr) {
625 GPUError("Double allocation! (%s)", res->mName);
626 throw std::bad_alloc();
627 }
628 if (IsGPU() && res->mOverrideSize < GPUCA_BUFFER_ALIGNMENT) {
629 res->mOverrideSize = GPUCA_BUFFER_ALIGNMENT;
630 }
631 if ((!IsGPU() || (res->mType & GPUMemoryResource::MEMORY_HOST) || mProcessingSettings.keepDisplayMemory) && !(res->mType & GPUMemoryResource::MEMORY_EXTERNAL)) { // keepAllMemory --> keepDisplayMemory
632 if (control && control->useExternal()) {
633 if (control->allocator) {
634 res->mSize = std::max((size_t)res->SetPointers((void*)1) - 1, res->mOverrideSize);
635 res->mPtr = control->allocator(CAMath::nextMultipleOf<GPUCA_BUFFER_ALIGNMENT>(res->mSize));
636 res->mSize = std::max<size_t>(ptrDiff(res->SetPointers(res->mPtr), res->mPtr), res->mOverrideSize);
637 if (mProcessingSettings.allocDebugLevel >= 2) {
638 std::cout << "Allocated (from callback) " << res->mName << ": " << res->mSize << "\n";
639 }
640 } else {
641 void* dummy = nullptr;
642 res->mSize = AllocateRegisteredMemoryHelper(res, res->mPtr, control->ptrCurrent, control->ptrBase, control->size, &GPUMemoryResource::SetPointers, dummy, "host");
643 }
644 } else {
645 res->mSize = AllocateRegisteredMemoryHelper(res, res->mPtr, recPool->mHostMemoryPool, recPool->mHostMemoryBase, recPool->mHostMemorySize, &GPUMemoryResource::SetPointers, recPool->mHostMemoryPoolEnd, "host");
646 }
647 if ((size_t)res->mPtr % GPUCA_BUFFER_ALIGNMENT) {
648 GPUError("Got buffer with insufficient alignment");
649 throw std::bad_alloc();
650 }
651 }
652 if (IsGPU() && (res->mType & GPUMemoryResource::MEMORY_GPU)) {
653 if (res->mProcessor->mLinkedProcessor == nullptr) {
654 GPUError("Device Processor not set (%s)", res->mName);
655 throw std::bad_alloc();
656 }
657 size_t size = AllocateRegisteredMemoryHelper(res, res->mPtrDevice, recPool->mDeviceMemoryPool, recPool->mDeviceMemoryBase, recPool->mDeviceMemorySize, &GPUMemoryResource::SetDevicePointers, recPool->mDeviceMemoryPoolEnd, " gpu");
658
659 if (!(res->mType & GPUMemoryResource::MEMORY_HOST) || (res->mType & GPUMemoryResource::MEMORY_EXTERNAL)) {
660 res->mSize = size;
661 } else if (size != res->mSize) {
662 GPUError("Inconsistent device memory allocation (%s: device %lu vs %lu)", res->mName, size, res->mSize);
663 throw std::bad_alloc();
664 }
665 if ((size_t)res->mPtrDevice % GPUCA_BUFFER_ALIGNMENT) {
666 GPUError("Got buffer with insufficient alignment");
667 throw std::bad_alloc();
668 }
669 }
670 UpdateMaxMemoryUsed();
671 }
672}

990void GPUReconstruction::RunPipelineWorker()
991{
992 if (!mInitialized || !mProcessingSettings.doublePipeline || mMaster != nullptr || !mSlaves.size()) {
993 throw std::invalid_argument("Cannot start double pipeline mode");
994 }
995 if (mProcessingSettings.debugLevel >= 3) {
996 GPUInfo("Pipeline worker started");
997 }
998 bool terminate = false;
999 while (!terminate) {
1000 {
1001 std::unique_lock<std::mutex> lk(mPipelineContext->mutex);
1002 mPipelineContext->cond.wait(lk, [this] { return this->mPipelineContext->queue.size() > 0; });
1003 }
1004 GPUReconstructionPipelineQueue* q;
1005 {
1006 std::lock_guard<std::mutex> lk(mPipelineContext->mutex);
1007 q = mPipelineContext->queue.front();
1008 mPipelineContext->queue.pop();
1009 }
1010 if (q->op == 1) {
1011 terminate = 1;
1012 } else {
1013 q->retVal = q->chain->RunChain();
1014 }
1015 {
1016 std::lock_guard<std::mutex> lk(q->m);
1017 q->done = true;
1018 }
1019 q->c.notify_one();
1020 }
1021 if (mProcessingSettings.debugLevel >= 3) {
1022 GPUInfo("Pipeline worker ended");
1023 }
1024}

1028 EnqueuePipeline(true);

1031int32_t GPUReconstruction::EnqueuePipeline(bool terminate)
1032{
1033 ClearAllocatedMemory(true);
1034 GPUReconstruction* rec = mMaster ? mMaster : this;
1035 std::unique_ptr<GPUReconstructionPipelineQueue> qu(new GPUReconstructionPipelineQueue);
1036 GPUReconstructionPipelineQueue* q = qu.get();
1037 q->chain = terminate ? nullptr : mChains[0].get();
1038 q->op = terminate ? 1 : 0;
1039 std::unique_lock<std::mutex> lkdone(q->m);
1040 {
1041 std::lock_guard<std::mutex> lkpipe(rec->mPipelineContext->mutex);
1042 if (rec->mPipelineContext->terminate) {
1043 throw std::runtime_error("Must not enqueue work after termination request");
1044 }
1045 rec->mPipelineContext->queue.push(q);
1046 rec->mPipelineContext->terminate = terminate;
1047 rec->mPipelineContext->cond.notify_one();
1048 }
1049 q->c.wait(lkdone, [&q]() { return q->done; });
1050 if (q->retVal) {
1051 return q->retVal;
1052 }
1053 if (terminate) {
1054 return 0;
1055 } else {
1056 return mChains[0]->FinalizePipelinedProcessing();
1057 }
1058}
