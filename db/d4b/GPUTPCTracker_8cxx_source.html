104void GPUTPCTracker::RegisterMemoryAllocation()
105{
106 AllocateAndInitializeLate();
107 bool reuseCondition = !mRec->GetProcessingSettings().keepDisplayMemory && mRec->GetProcessingSettings().trackletSelectorInPipeline && ((mRec->GetRecoStepsGPU() & GPUDataTypes::RecoStep::TPCSliceTracking) || mRec->GetProcessingSettings().ompKernels == 1 || mRec->GetProcessingSettings().ompThreads == 1);
108 GPUMemoryReuse reLinks{reuseCondition, GPUMemoryReuse::REUSE_1TO1, GPUMemoryReuse::TrackerDataLinks, (uint16_t)(mISlice % mRec->GetProcessingSettings().nStreams)};
109 mMemoryResLinks = mRec->RegisterMemoryAllocation(this, &GPUTPCTracker::SetPointersDataLinks, GPUMemoryResource::MEMORY_SCRATCH | GPUMemoryResource::MEMORY_STACK, "TPCSliceLinks", reLinks);
110 mMemoryResSliceScratch = mRec->RegisterMemoryAllocation(this, &GPUTPCTracker::SetPointersDataScratch, GPUMemoryResource::MEMORY_SCRATCH | GPUMemoryResource::MEMORY_STACK | GPUMemoryResource::MEMORY_CUSTOM, "TPCSliceScratch");
111 mMemoryResSliceInput = mRec->RegisterMemoryAllocation(this, &GPUTPCTracker::SetPointersDataInput, GPUMemoryResource::MEMORY_INPUT | GPUMemoryResource::MEMORY_STACK | GPUMemoryResource::MEMORY_CUSTOM, "TPCSliceInput");
112 GPUMemoryReuse reWeights{reuseCondition, GPUMemoryReuse::REUSE_1TO1, GPUMemoryReuse::TrackerDataWeights, (uint16_t)(mISlice % mRec->GetProcessingSettings().nStreams)};
113 mRec->RegisterMemoryAllocation(this, &GPUTPCTracker::SetPointersDataWeights, GPUMemoryResource::MEMORY_SCRATCH | GPUMemoryResource::MEMORY_STACK, "TPCSliceWeights", reWeights);
114 GPUMemoryReuse reScratch{reuseCondition, GPUMemoryReuse::REUSE_1TO1, GPUMemoryReuse::TrackerScratch, (uint16_t)(mISlice % mRec->GetProcessingSettings().nStreams)};
115 mRec->RegisterMemoryAllocation(this, &GPUTPCTracker::SetPointersScratch, GPUMemoryResource::MEMORY_SCRATCH | GPUMemoryResource::MEMORY_STACK, "TPCTrackerScratch", reScratch);
116 mRec->RegisterMemoryAllocation(this, &GPUTPCTracker::SetPointersScratchHost, GPUMemoryResource::MEMORY_SCRATCH_HOST, "TPCTrackerHost");
117 mMemoryResCommon = mRec->RegisterMemoryAllocation(this, &GPUTPCTracker::SetPointersCommon, GPUMemoryResource::MEMORY_PERMANENT, "TPCTrackerCommon");
118 mRec->RegisterMemoryAllocation(this, &GPUTPCTracker::SetPointersDataRows, GPUMemoryResource::MEMORY_PERMANENT, "TPCSliceRows");
119
120 uint32_t type = GPUMemoryResource::MEMORY_SCRATCH;
121 if (mRec->GetProcessingSettings().memoryAllocationStrategy == GPUMemoryResource::ALLOCATION_INDIVIDUAL) { // For individual scheme, we allocate tracklets separately, and change the type for the following allocations to custom
122 type |= GPUMemoryResource::MEMORY_CUSTOM;
123 mMemoryResTracklets = mRec->RegisterMemoryAllocation(this, &GPUTPCTracker::SetPointersTracklets, type, "TPCTrackerTracklets");
124 }
125 mMemoryResOutput = mRec->RegisterMemoryAllocation(this, &GPUTPCTracker::SetPointersOutput, type, "TPCTrackerTracks");

142void GPUTPCTracker::SetMaxData(const GPUTrackingInOutPointers& io)
143{
144 if (mRec->GetProcessingSettings().memoryAllocationStrategy == GPUMemoryResource::ALLOCATION_INDIVIDUAL) {
145 mNMaxStartHits = mData.NumberOfHits();
146 } else {
147 mNMaxStartHits = mRec->MemoryScalers()->NTPCStartHits(mData.NumberOfHits());
148 }
149 if (io.clustersNative) {
150 uint32_t maxRowHits = 0;
151 for (uint32_t i = 0; i < GPUCA_ROW_COUNT; i++) {
152 if (io.clustersNative->nClusters[mISlice][i] > maxRowHits) {
153 maxRowHits = io.clustersNative->nClusters[mISlice][i];
154 }
155 }
156 mNMaxRowStartHits = mRec->MemoryScalers()->NTPCRowStartHits(maxRowHits * GPUCA_ROW_COUNT);
157 } else {
158 mNMaxRowStartHits = mRec->MemoryScalers()->NTPCRowStartHits(mData.NumberOfHits());
159 }
160 mNMaxTracklets = mRec->MemoryScalers()->NTPCTracklets(mData.NumberOfHits());
161 mNMaxRowHits = mRec->MemoryScalers()->NTPCTrackletHits(mData.NumberOfHits());
162 mNMaxTracks = mRec->MemoryScalers()->NTPCSectorTracks(mData.NumberOfHits());
163 mNMaxTrackHits = mRec->MemoryScalers()->NTPCSectorTrackHits(mData.NumberOfHits(), mRec->GetProcessingSettings().tpcInputWithClusterRejection);
164#ifdef GPUCA_SORT_STARTHITS_GPU
165 if (mRec->GetRecoStepsGPU() & GPUDataTypes::RecoStep::TPCSliceTracking) {
166 if (mNMaxStartHits > mNMaxRowStartHits * GPUCA_ROW_COUNT) {
167 mNMaxStartHits = mNMaxRowStartHits * GPUCA_ROW_COUNT;
168 }
169 }
170#endif
171 mData.SetMaxData();
172}

206GPUh() void GPUTPCTracker::WriteOutput()
207{
208 mOutput->SetNTracks(0);
209 mOutput->SetNLocalTracks(0);
210 mOutput->SetNTrackClusters(0);
211
212 if (mCommonMem->nTracks == 0) {
213 return;
214 }
215 if (mCommonMem->nTracks > GPUCA_MAX_SLICE_NTRACK) {
216 GPUError("Maximum number of tracks exceeded, cannot store");
217 return;
218 }
219
220 int32_t nStoredHits = 0;
221 int32_t nStoredTracks = 0;
222 int32_t nStoredLocalTracks = 0;
223
224 GPUTPCTrack* out = mOutput->FirstTrack();
225
226 trackSortData* trackOrder = new trackSortData[mCommonMem->nTracks];
227 for (uint32_t i = 0; i < mCommonMem->nTracks; i++) {
228 trackOrder[i].fTtrack = i;
229 trackOrder[i].fSortVal = mTracks[trackOrder[i].fTtrack].NHits() / 1000.f + mTracks[trackOrder[i].fTtrack].Param().GetZ() * 100.f + mTracks[trackOrder[i].fTtrack].Param().GetY();
230 }
231 std::sort(trackOrder, trackOrder + mCommonMem->nLocalTracks, SortComparison<trackSortData>); // TODO: Check why this sorting affects the merging efficiency!
232 std::sort(trackOrder + mCommonMem->nLocalTracks, trackOrder + mCommonMem->nTracks, SortComparison<trackSortData>);
233
234 for (uint32_t iTrTmp = 0; iTrTmp < mCommonMem->nTracks; iTrTmp++) {
235 const int32_t iTr = trackOrder[iTrTmp].fTtrack;
236 GPUTPCTrack& iTrack = mTracks[iTr];
237
238 *out = iTrack;
239 int32_t nClu = 0;
240 int32_t iID = iTrack.FirstHitID();
241
242 for (int32_t ith = 0; ith < iTrack.NHits(); ith++) {
243 const GPUTPCHitId& ic = mTrackHits[iID + ith];
244 int32_t iRow = ic.RowIndex();
245 int32_t ih = ic.HitIndex();
246
247 const GPUTPCRow& row = mData.Row(iRow);
248 int32_t clusterIndex = mData.ClusterDataIndex(row, ih);
249#ifdef GPUCA_ARRAY_BOUNDS_CHECKS
250 if (ih >= row.NHits() || ih < 0) {
251 GPUError("Array out of bounds access (Sector Row) (Hit %d / %d - NumC %d): Sector %d Row %d Index %d", ith, iTrack.NHits(), NHitsTotal(), mISlice, iRow, ih);
252 fflush(stdout);
253 continue;
254 }
255 if (clusterIndex >= NHitsTotal() || clusterIndex < 0) {
256 GPUError("Array out of bounds access (Cluster Data) (Hit %d / %d - NumC %d): Sector %d Row %d Hit %d, Clusterdata Index %d", ith, iTrack.NHits(), NHitsTotal(), mISlice, iRow, ih, clusterIndex);
257 fflush(stdout);
258 continue;
259 }
260#endif
261
262 float origX, origY, origZ;
263 uint8_t flags;
264 uint16_t amp;
265 int32_t id;
266 if (Param().par.earlyTpcTransform) {
267 origX = mData.ClusterData()[clusterIndex].x;
268 origY = mData.ClusterData()[clusterIndex].y;
269 origZ = mData.ClusterData()[clusterIndex].z;
270 flags = mData.ClusterData()[clusterIndex].flags;
271 amp = mData.ClusterData()[clusterIndex].amp;
272 id = mData.ClusterData()[clusterIndex].id;
273 } else {
274 const ClusterNativeAccess& cls = *mConstantMem->ioPtrs.clustersNative;
275 id = clusterIndex + cls.clusterOffset[mISlice][0];
276 GPUTPCConvertImpl::convert(*mConstantMem, mISlice, iRow, cls.clustersLinear[id].getPad(), cls.clustersLinear[id].getTime(), origX, origY, origZ);
277 flags = cls.clustersLinear[id].getFlags();
278 amp = cls.clustersLinear[id].qTot;
279 }
280 GPUTPCSliceOutCluster c;
281 c.Set(id, iRow, flags, amp, origX, origY, origZ);
282#ifdef GPUCA_TPC_RAW_PROPAGATE_PAD_ROW_TIME
283 c.mPad = mData.ClusterData()[clusterIndex].pad;
284 c.mTime = mData.ClusterData()[clusterIndex].time;
285#endif
286 out->SetOutTrackCluster(nClu, c);
287 nClu++;
288 }
289
290 nStoredTracks++;
291 if (iTr < mCommonMem->nLocalTracks) {
292 nStoredLocalTracks++;
293 }
294 nStoredHits += nClu;
295 out->SetNHits(nClu);
296 out = out->NextTrack();
297 }
298 delete[] trackOrder;
299
300 mOutput->SetNTracks(nStoredTracks);
301 mOutput->SetNLocalTracks(nStoredLocalTracks);
302 mOutput->SetNTrackClusters(nStoredHits);
303 if (Param().par.debugLevel >= 3) {
304 GPUInfo("Slice %d, Output: Tracks %d, local tracks %d, hits %d", mISlice, nStoredTracks, nStoredLocalTracks, nStoredHits);
305 }
306}
