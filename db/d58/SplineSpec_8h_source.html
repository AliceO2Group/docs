197template <typename DataT>
198GPUdi() void SplineContainer<DataT>::getKnotU(int32_t iKnot, int32_t u[/* mXdim */]) const
199{
201 for (int32_t dim = 0; dim < mXdim; dim++) {
202 int32_t n = mGrid[dim].getNumberOfKnots();
203 u[dim] = mGrid[dim].getKnot(iKnot % n).getU();
204 iKnot /= n;
205 }
206}

208template <typename DataT>
209GPUdi() int32_t SplineContainer<DataT>::getKnotIndex(const int32_t iKnot[/* mXdim */]) const
210{
212 int32_t ind = iKnot[0];
213 int32_t n = 1;
214 for (int32_t dim = 1; dim < mXdim; dim++) {
215 n *= mGrid[dim - 1].getNumberOfKnots();
216 ind += n * iKnot[dim];
217 }
218 return ind;
219}

272 GPUd() void interpolate(const DataT x[/*mXdim*/], GPUgeneric() DataT S[/*mYdim*/]) const
273 {
274 const auto nXdimTmp = SplineUtil::getNdim<XdimT>(mXdim);
275 const auto nXdim = nXdimTmp.get();
276 const auto maxXdimTmp = SplineUtil::getMaxNdim<XdimT>(mXdim);
277 DataT u[maxXdimTmp.get()];
278 for (int32_t i = 0; i < nXdim; i++) {
279 u[i] = mGrid[i].convXtoU(x[i]);
280 }
281 interpolateU<SafetyLevel::kSafe>(mXdim, mYdim, mParameters, u, S);
282 }

285 template <SafetyLevel SafeT = SafetyLevel::kSafe>
286 GPUd() void interpolateU(int32_t inpXdim, int32_t inpYdim, GPUgeneric() const DataT Parameters[],
287 const DataT u[/*inpXdim*/], GPUgeneric() DataT S[/*inpYdim*/]) const
288 {
289 const auto nXdimTmp = SplineUtil::getNdim<XdimT>(mXdim);
290 const auto nXdim = nXdimTmp.get();
291 const auto maxXdimTmp = SplineUtil::getMaxNdim<XdimT>(mXdim);
292 const auto maxXdim = maxXdimTmp.get();
293 const auto nYdimTmp = SplineUtil::getNdim<YdimT>(mYdim);
294 const auto nYdim = nYdimTmp.get();
295 const auto maxYdimTmp = SplineUtil::getMaxNdim<XdimT>(mYdim);
296 const auto maxYdim = maxYdimTmp.get();
297
298 // const auto nParameters = 1 << (2 * nXdim); //total Nr of Parameters necessary for one interpolation
299 const auto nKnotParametersPerY = 1 << nXdim; // Nr of Parameters per Knot per Y dimension
300 const auto nKnotParameters = (1 << nXdim) * nYdim; // Nr of Parameters per Knot
301
302 DataT iParameters[(1 << (2 * maxXdim)) * maxYdim]; // Array for all parameters
303
304 //get the indices of the "most left" Knot:
305
306 int32_t indices[maxXdim]; // indices of the 'most left' knot
307 for (int32_t i = 0; i < nXdim; i++) {
308 indices[i] = mGrid[i].getLeftKnotIndexForU(u[i]);
309 }
310 // get all the needed parameters into one array iParameters[nParameters]:
311 int32_t indicestmp[maxXdim];
312 for (int32_t i = 0; i < nKnotParametersPerY; i++) { // for every necessary Knot
313 for (int32_t k = 0; k < nXdim; k++) {
314 indicestmp[k] = indices[k] + (i / (1 << k)) % 2; //get the knot-indices in every dimension (mirrored order binary counting)
315 }
316 int32_t index = TBase::getKnotIndex(indicestmp); // get index of the current Knot
317
318 for (int32_t j = 0; j < nKnotParameters; j++) { // and fill the iparameter array with according parameters
319 iParameters[i * nKnotParameters + j] = Parameters[index * nKnotParameters + j];
320 }
321 }
322 //now start with the interpolation loop:
323
324 constexpr auto maxInterpolations = (1 << (2 * maxXdim - 2)) * maxYdim;
325
326 DataT S0[maxInterpolations];
327 DataT D0[maxInterpolations];
328 DataT S1[maxInterpolations];
329 DataT D1[maxInterpolations];
330
331 int32_t nInterpolations = (1 << (2 * nXdim - 2)) * nYdim;
332 int32_t nKnots = 1 << (nXdim);
333
334 for (int32_t d = 0; d < nXdim; d++) { // for every dimension
335 DataT* pointer[4] = {S0, D0, S1, D1}; // pointers for interpolation arrays S0, D0, S1, D1 point to Arraystart
336 for (int32_t i = 0; i < nKnots; i++) { // for every knot
337 for (int32_t j = 0; j < nKnots; j++) { // for every parametertype
338 int32_t pointernr = 2 * (i % 2) + (j % 2); // to which array should it be delivered
339 for (int32_t k = 0; k < nYdim; k++) {
340 pointer[pointernr][0] = iParameters[(i * nKnots + j) * nYdim + k];
341 pointer[pointernr]++;
342 }
343 } // end for j (every parametertype)
344 } // end for i (every knot)
345
346 const typename Spline1D<DataT>::Knot& knotL = mGrid[d].getKnot(indices[d]);
347 DataT coordinate = u[d];
348 typedef Spline1DSpec<DataT, 0, 0> TGridX;
349 const TGridX& gridX = *((const TGridX*)&(mGrid[d]));
350 gridX.interpolateU(nInterpolations, knotL, S0, D0, S1, D1, coordinate, iParameters);
351 nInterpolations /= 4;
352 nKnots /= 2;
353 } // end d (every dimension)
354
355 for (int32_t i = 0; i < nYdim; i++) {
356 S[i] = iParameters[i]; // write into result-array
357 // LOG(info)<<iParameters[i] <<", ";
358 }
359 } // end interpolateU
360
361 protected:
362 using TBase::mGrid;
363 using TBase::mParameters;
364 using TBase::mXdim;
365 using TBase::mYdim;
366 using TBase::TBase; // inherit constructors and hide them
367};
