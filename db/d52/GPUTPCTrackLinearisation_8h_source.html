70GPUdi() GPUTPCTrackLinearisation::GPUTPCTrackLinearisation(const GPUTPCTrackParam& GPUrestrict() t) : mSinPhi(t.SinPhi()), mCosPhi(0), mDzDs(t.DzDs()), mQPt(t.QPt())
71{
72 if (mSinPhi > GPUCA_MAX_SIN_PHI) {
73 mSinPhi = GPUCA_MAX_SIN_PHI;
74 } else if (mSinPhi < -GPUCA_MAX_SIN_PHI) {
75 mSinPhi = -GPUCA_MAX_SIN_PHI;
76 }
77 mCosPhi = CAMath::Sqrt(1 - mSinPhi * mSinPhi);
78 if (t.SignCosPhi() < 0) {
79 mCosPhi = -mCosPhi;
80 }
81}
