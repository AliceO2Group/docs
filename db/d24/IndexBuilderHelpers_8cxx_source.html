45 IndexColumnBuilder::IndexColumnBuilder(std::shared_ptr<arrow::ChunkedArray> source, const char* name, int listSize, arrow::MemoryPool* pool)
46  : ChunkedArrayIterator{source},
47  SelfIndexColumnBuilder{name, pool},
48  mListSize{listSize},
49  mSourceSize{(size_t)source->length()}
50 {
51  switch (mListSize) {
52  case 1: {
53  mValueBuilder = mBuilder.get();
54  mArrowType = arrow::int32();
55  }; break;
56  case 2: {
57  if (preSlice().ok()) {
58  mListBuilder = std::make_unique<arrow::FixedSizeListBuilder>(pool, std::move(mBuilder), mListSize);
59  mValueBuilder = static_cast<arrow::FixedSizeListBuilder*>(mListBuilder.get())->value_builder();
60  mArrowType = arrow::fixed_size_list(arrow::int32(), 2);
61  } else {
62  throw runtime_error("Cannot pre-slice an array");
63  }
64  }; break;
65  case -1: {
66  if (preFind().ok()) {
67  mListBuilder = std::make_unique<arrow::ListBuilder>(pool, std::move(mBuilder));
68  mValueBuilder = static_cast<arrow::ListBuilder*>(mListBuilder.get())->value_builder();
69  mArrowType = arrow::list(arrow::int32());
70  } else {
71  throw runtime_error("Cannot pre-find array groups");
72  }
73  }; break;
74  default:
75  throw runtime_error_f("Invalid list size for index column: %d", mListSize);
76  }
77 }

79 arrow::Status IndexColumnBuilder::preSlice()
80 {
81  arrow::Datum value_counts;
82  auto options = arrow::compute::ScalarAggregateOptions::Defaults();
83  ARROW_ASSIGN_OR_RAISE(value_counts, arrow::compute::CallFunction("value_counts", {mSource}, &options));
84  auto pair = static_cast<arrow::StructArray>(value_counts.array());
85  mValuesArrow = std::make_shared<arrow::NumericArray<arrow::Int32Type>>(pair.field(0)->data());
86  mCounts = std::make_shared<arrow::NumericArray<arrow::Int64Type>>(pair.field(1)->data());
87  return arrow::Status::OK();
88 }

90 arrow::Status IndexColumnBuilder::preFind()
91 {
92  arrow::Datum max;
93  auto options = arrow::compute::ScalarAggregateOptions::Defaults();
94  ARROW_ASSIGN_OR_RAISE(max, arrow::compute::CallFunction("max", {mSource}, &options));
95  auto maxValue = std::dynamic_pointer_cast<arrow::Int32Scalar>(max.scalar())->value;
96  mIndices.resize(maxValue + 1);
97 
98  auto row = 0;
99  for (auto i = 0; i < mSource->length(); ++i) {
100  auto v = valueAt(i);
101  if (v >= 0) {
102  mValues.emplace_back(v);
103  mIndices[v].push_back(row);
104  }
105  ++row;
106  }
107  std::sort(mValues.begin(), mValues.end());
108 
109  return arrow::Status::OK();
110 }

142 bool IndexColumnBuilder::findSingle(int idx)
143 {
144  auto count = mSourceSize - mPosition;
145  while (count > 0) {
146  size_t step = count / 2;
147  mPosition += step;
148  if (valueAt(mPosition) <= idx) {
149  count -= step + 1;
150  } else {
151  mPosition -= step;
152  count = step;
153  }
154  }
155 
156  return (mPosition < mSourceSize && valueAt(mPosition) == idx);
157 }

191 void IndexColumnBuilder::fillSlice(int idx)
192 {
193  int data[2] = {-1, -1};
194  if (mValuePos < mValuesArrow->length() && mValuesArrow->Value(mValuePos) == idx) {
195  for (auto i = 0; i < mValuePos; ++i) {
196  data[0] += mCounts->Value(i);
197  }
198  data[0] += 1;
199  data[1] = data[0] + mCounts->Value(mValuePos) - 1;
200  }
201  (void)static_cast<arrow::FixedSizeListBuilder*>(mListBuilder.get())->AppendValues(1);
202  (void)static_cast<arrow::Int32Builder*>(mValueBuilder)->AppendValues(data, 2);
203 }

244 int ChunkedArrayIterator::valueAt(size_t pos)
245 {
246  while (O2_BUILTIN_UNLIKELY(mCurrent + pos >= mLast)) {
247  nextChunk();
248  }
249  while (O2_BUILTIN_UNLIKELY(pos < mFirstIndex)) {
250  prevChunk();
251  }
252  return *(mCurrent + pos);
253 }
