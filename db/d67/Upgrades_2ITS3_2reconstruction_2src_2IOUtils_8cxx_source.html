28 int loadROFrameDataITS3(its::TimeFrame* tf,
29  gsl::span<o2::itsmft::ROFRecord> rofs,
30  gsl::span<const itsmft::CompClusterExt> clusters,
31  gsl::span<const unsigned char>::iterator& pattIt,
32  const its3::TopologyDictionary* dict,
33  const dataformats::MCTruthContainer<MCCompLabel>* mcLabels)
34 {
35  its::GeometryTGeo* geom = its::GeometryTGeo::Instance();
36  geom->fillMatrixCache(o2::math_utils::bit2Mask(o2::math_utils::TransformType::T2L, o2::math_utils::TransformType::L2G));
37 
38  tf->mNrof = 0;
39  for (auto& rof : rofs) {
40  for (int clusterId{rof.getFirstEntry()}; clusterId < rof.getFirstEntry() + rof.getNEntries(); ++clusterId) {
41  auto& c = clusters[clusterId];
42  auto sensorID = c.getSensorID();
43  auto isITS3 = its3::constants::detID::isDetITS3(sensorID);
44  auto layer = geom->getLayer(sensorID);
45 
46  auto pattID = c.getPatternID();
47  o2::math_utils::Point3D<float> locXYZ;
48  float sigmaY2 = o2::its::ioutils::DefClusError2Row, sigmaZ2 = o2::its::ioutils::DefClusError2Col, sigmaYZ = 0; // Dummy COG errors (about half pixel size)
49  float pitchRow = isITS3 ? SSAlpide::mPitchRow : o2::itsmft::SegmentationAlpide::PitchRow;
50  float pitchCol = isITS3 ? SSAlpide::mPitchCol : o2::itsmft::SegmentationAlpide::PitchCol;
51  if (pattID != itsmft::CompCluster::InvalidPatternID) {
52  sigmaY2 = dict->getErr2X(pattID) * pitchRow * pitchRow;
53  sigmaZ2 = dict->getErr2Z(pattID) * pitchCol * pitchCol;
54  if (!dict->isGroup(pattID)) {
55  locXYZ = dict->getClusterCoordinates(c);
56  } else {
57  o2::itsmft::ClusterPattern patt(pattIt);
58  locXYZ = dict->getClusterCoordinates(c, patt);
59  sigmaY2 = patt.getRowSpan() * patt.getRowSpan() * pitchRow * pitchRow / 12.;
60  sigmaZ2 = patt.getColumnSpan() * patt.getColumnSpan() * pitchCol * pitchCol / 12.;
61  }
62  } else {
63  o2::itsmft::ClusterPattern patt(pattIt);
64  sigmaY2 = patt.getRowSpan() * patt.getRowSpan() * pitchRow * pitchRow / 12.;
65  sigmaZ2 = patt.getColumnSpan() * patt.getColumnSpan() * pitchCol * pitchCol / 12.;
66  locXYZ = dict->getClusterCoordinates(c, patt, false);
67  }
68 
69  // Transformation to the local --> global
70  auto gloXYZ = geom->getMatrixL2G(sensorID) * locXYZ;
71 
72  // for cylindrical layers we have a different alpha for each cluster, for regular silicon detectors instead a single alpha for the whole sensor
73  float alpha = 0.;
74  o2::math_utils::Point3D<float> trkXYZ;
75  if (isITS3) {
76  alpha = geom->getAlphaFromGlobalITS3(gloXYZ);
77  // Inverse transformation to the local --> tracking
78  trkXYZ = geom->getT2LMatrixITS3(sensorID, alpha) ^ locXYZ;
79  } else {
80  alpha = geom->getSensorRefAlpha(sensorID);
81  // Inverse transformation to the local --> tracking
82  trkXYZ = geom->getMatrixT2L(sensorID) ^ locXYZ;
83  }
84 
85  tf->addTrackingFrameInfoToLayer(layer, gloXYZ.x(), gloXYZ.y(), gloXYZ.z(), trkXYZ.x(), alpha,
86  std::array<float, 2>{trkXYZ.y(), trkXYZ.z()},
87  std::array<float, 3>{sigmaY2, sigmaYZ, sigmaZ2});
88 
90  tf->addClusterToLayer(layer, gloXYZ.x(), gloXYZ.y(), gloXYZ.z(), tf->getUnsortedClusters()[layer].size());
91  tf->addClusterExternalIndexToLayer(layer, clusterId);
92  }
93  for (unsigned int iL{0}; iL < tf->getUnsortedClusters().size(); ++iL) {
94  tf->mROframesClusters[iL].push_back(tf->getUnsortedClusters()[iL].size());
95  }
96  tf->mNrof++;
97  }
98 
99  for (auto& v : tf->mNTrackletsPerCluster) {
100  v.resize(tf->getUnsortedClusters()[1].size());
101  }
102 
103  if (mcLabels != nullptr) {
104  tf->mClusterLabels = mcLabels;
105  }
106  return tf->mNrof;
107 }
108 } // namespace o2::its3::ioutils
