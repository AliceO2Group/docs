200DataProcessingDevice::DataProcessingDevice(RunningDeviceRef running, ServiceRegistry& registry)
201 : mRunningDevice{running},
202 mConfigRegistry{nullptr},
203 mServiceRegistry{registry}
204{
205 GetConfig()->Subscribe<std::string>("dpl", [&registry = mServiceRegistry](const std::string& key, std::string value) {
206 if (key == "cleanup") {
207 auto ref = ServiceRegistryRef{registry, ServiceRegistry::globalDeviceSalt()};
208 auto& deviceState = ref.get<DeviceState>();
209 int64_t cleanupCount = deviceState.cleanupCount.load();
210 int64_t newCleanupCount = std::stoll(value);
211 if (newCleanupCount <= cleanupCount) {
212 return;
213 }
214 deviceState.cleanupCount.store(newCleanupCount);
215 for (auto& info : deviceState.inputChannelInfos) {
216 fair::mq::Parts parts;
217 while (info.channel->Receive(parts, 0)) {
218 LOGP(debug, "Dropping {} parts", parts.Size());
219 if (parts.Size() == 0) {
220 break;
221 }
222 }
223 }
224 }
225 });
226
227 std::function<void(const fair::mq::State)> stateWatcher = [this, &registry = mServiceRegistry](const fair::mq::State state) -> void {
228 auto ref = ServiceRegistryRef{registry, ServiceRegistry::globalDeviceSalt()};
229 auto& deviceState = ref.get<DeviceState>();
230 auto& control = ref.get<ControlService>();
231 auto& callbacks = ref.get<CallbackService>();
232 control.notifyDeviceState(fair::mq::GetStateName(state));
233 callbacks.call<CallbackService::Id::DeviceStateChanged>(ServiceRegistryRef{ref}, (int)state);
234
235 if (deviceState.nextFairMQState.empty() == false) {
236 auto state = deviceState.nextFairMQState.back();
237 (void)this->ChangeState(state);
238 deviceState.nextFairMQState.pop_back();
239 }
240 };
241
242 // 99 is to execute DPL callbacks last
243 this->SubscribeToStateChange("99-dpl", stateWatcher);
244
245 // One task for now.
246 mStreams.resize(1);
247 mHandles.resize(1);
248
249 ServiceRegistryRef ref{mServiceRegistry};
250
251 mAwakeHandle = (uv_async_t*)malloc(sizeof(uv_async_t));
252 auto& state = ref.get<DeviceState>();
253 assert(state.loop);
254 int res = uv_async_init(state.loop, mAwakeHandle, on_communication_requested);
255 mAwakeHandle->data = &state;
256 if (res < 0) {
257 LOG(error) << "Unable to initialise subscription";
258 }
259
261 SubscribeToNewTransition("dpl", [wakeHandle = mAwakeHandle](fair::mq::Transition t) {
262 int res = uv_async_send(wakeHandle);
263 if (res < 0) {
264 LOG(error) << "Unable to notify subscription";
265 }
266 LOG(debug) << "State transition requested";
267 });
268}

286// Once the processing in a thread is done, this is executed on the main thread.
287void run_completion(uv_work_t* handle, int status)
288{
289 auto* task = (TaskStreamInfo*)handle->data;
290 // Notice that the completion, while running on the main thread, still
291 // has a salt which is associated to the actual stream which was doing the computation
292 auto ref = ServiceRegistryRef{*task->registry, ServiceRegistry::globalStreamSalt(task->id.index + 1)};
293 auto& state = ref.get<DeviceState>();
294 auto& quotaEvaluator = ref.get<ComputingQuotaEvaluator>();
295
296 using o2::monitoring::Metric;
297 using o2::monitoring::Monitoring;
298 using o2::monitoring::tags::Key;
299 using o2::monitoring::tags::Value;

301 static std::function<void(ComputingQuotaOffer const&, ComputingQuotaStats&)> reportConsumedOffer = [ref](ComputingQuotaOffer const& accumulatedConsumed, ComputingQuotaStats& stats) {
302 auto& dpStats = ref.get<DataProcessingStats>();
303 stats.totalConsumedBytes += accumulatedConsumed.sharedMemory;
304
305 dpStats.updateStats({static_cast<short>(ProcessingStatsId::SHM_OFFER_BYTES_CONSUMED), DataProcessingStats::Op::Set, stats.totalConsumedBytes});
306 dpStats.processCommandQueue();
307 assert(stats.totalConsumedBytes == dpStats.metrics[(short)ProcessingStatsId::SHM_OFFER_BYTES_CONSUMED]);
308 };
309
310 static std::function<void(ComputingQuotaOffer const&, ComputingQuotaStats const&)> reportExpiredOffer = [ref](ComputingQuotaOffer const& offer, ComputingQuotaStats const& stats) {
311 auto& dpStats = ref.get<DataProcessingStats>();
312 dpStats.updateStats({static_cast<short>(ProcessingStatsId::RESOURCE_OFFER_EXPIRED), DataProcessingStats::Op::Set, stats.totalExpiredOffers});
313 dpStats.updateStats({static_cast<short>(ProcessingStatsId::ARROW_BYTES_EXPIRED), DataProcessingStats::Op::Set, stats.totalExpiredBytes});
314 dpStats.processCommandQueue();
315 };
316
317 for (auto& consumer : state.offerConsumers) {
318 quotaEvaluator.consume(task->id.index, consumer, reportConsumedOffer);
319 }
320 state.offerConsumers.clear();
321 quotaEvaluator.handleExpired(reportExpiredOffer);
322 quotaEvaluator.dispose(task->id.index);
323 task->running = false;
324}

343void on_socket_polled(uv_poll_t* poller, int status, int events)
344{
345 auto* context = (PollerContext*)poller->data;
346 assert(context);
347 O2_SIGNPOST_ID_FROM_POINTER(sid, sockets, poller);
348 context->state->loopReason |= DeviceState::DATA_SOCKET_POLLED;
349 switch (events) {
350 case UV_READABLE: {
351 O2_SIGNPOST_EVENT_EMIT(sockets, sid, "socket_state", "Data pending on socket for channel %{public}s", context->name);
352 context->state->loopReason |= DeviceState::DATA_INCOMING;
353 } break;
354 case UV_WRITABLE: {
355 O2_SIGNPOST_END(sockets, sid, "socket_state", "Socket connected for channel %{public}s", context->name);
356 if (context->read) {
357 O2_SIGNPOST_START(sockets, sid, "socket_state", "Socket connected for read in context %{public}s", context->name);
358 uv_poll_start(poller, UV_READABLE | UV_DISCONNECT | UV_PRIORITIZED, &on_socket_polled);
359 context->state->loopReason |= DeviceState::DATA_CONNECTED;
360 } else {
361 O2_SIGNPOST_START(sockets, sid, "socket_state", "Socket connected for write for channel %{public}s", context->name);
362 context->state->loopReason |= DeviceState::DATA_OUTGOING;
363 // If the socket is writable, fairmq will handle the rest, so we can stop polling and
364 // just wait for the disconnect.
365 uv_poll_start(poller, UV_DISCONNECT | UV_PRIORITIZED, &on_socket_polled);
366 }
367 context->pollerState = PollerContext::PollerState::Connected;
368 } break;
369 case UV_DISCONNECT: {
370 O2_SIGNPOST_END(sockets, sid, "socket_state", "Socket disconnected in context %{public}s", context->name);
371 } break;
372 case UV_PRIORITIZED: {
373 O2_SIGNPOST_EVENT_EMIT(sockets, sid, "socket_state", "Socket prioritized for context %{public}s", context->name);
374 } break;
375 }
376 // We do nothing, all the logic for now stays in DataProcessingDevice::doRun()
377}

379void on_out_of_band_polled(uv_poll_t* poller, int status, int events)
380{
381 O2_SIGNPOST_ID_FROM_POINTER(sid, sockets, poller);
382 auto* context = (PollerContext*)poller->data;
383 context->state->loopReason |= DeviceState::OOB_ACTIVITY;
384 if (status < 0) {
385 LOGP(fatal, "Error while polling {}: {}", context->name, status);
386 uv_poll_start(poller, UV_WRITABLE, &on_out_of_band_polled);
387 }
388 switch (events) {
389 case UV_READABLE: {
390 O2_SIGNPOST_EVENT_EMIT(sockets, sid, "socket_state", "Data pending on socket for channel %{public}s", context->name);
391 context->state->loopReason |= DeviceState::DATA_INCOMING;
392 assert(context->channelInfo);
393 context->channelInfo->readPolled = true;
394 } break;
395 case UV_WRITABLE: {
396 O2_SIGNPOST_END(sockets, sid, "socket_state", "OOB socket connected for channel %{public}s", context->name);
397 if (context->read) {
398 O2_SIGNPOST_START(sockets, sid, "socket_state", "OOB socket connected for read in context %{public}s", context->name);
399 uv_poll_start(poller, UV_READABLE | UV_DISCONNECT | UV_PRIORITIZED, &on_out_of_band_polled);
400 } else {
401 O2_SIGNPOST_START(sockets, sid, "socket_state", "OOB socket connected for write for channel %{public}s", context->name);
402 context->state->loopReason |= DeviceState::DATA_OUTGOING;
403 }
404 } break;
405 case UV_DISCONNECT: {
406 O2_SIGNPOST_END(sockets, sid, "socket_state", "OOB socket disconnected in context %{public}s", context->name);
407 uv_poll_start(poller, UV_WRITABLE, &on_out_of_band_polled);
408 } break;
409 case UV_PRIORITIZED: {
410 O2_SIGNPOST_EVENT_EMIT(sockets, sid, "socket_state", "OOB socket prioritized for context %{public}s", context->name);
411 } break;
412 }
413 // We do nothing, all the logic for now stays in DataProcessingDevice::doRun()

423void DataProcessingDevice::Init()
424{
425 auto ref = ServiceRegistryRef{mServiceRegistry};
426 auto& context = ref.get<DataProcessorContext>();
427 auto& spec = getRunningDevice(mRunningDevice, ref);
428
429 O2_SIGNPOST_ID_FROM_POINTER(cid, device, &context);
430 O2_SIGNPOST_START(device, cid, "Init", "Entering Init callback.");
431 context.statelessProcess = spec.algorithm.onProcess;
432 context.statefulProcess = nullptr;
433 context.error = spec.algorithm.onError;
434 context.initError = spec.algorithm.onInitError;
435
436 auto configStore = DeviceConfigurationHelpers::getConfiguration(mServiceRegistry, spec.name.c_str(), spec.options);
437 if (configStore == nullptr) {
438 std::vector<std::unique_ptr<ParamRetriever>> retrievers;
439 retrievers.emplace_back(std::make_unique<FairOptionsRetriever>(GetConfig()));
440 configStore = std::make_unique<ConfigParamStore>(spec.options, std::move(retrievers));
441 configStore->preload();
442 configStore->activate();
443 }
444
445 using boost::property_tree::ptree;
446
448 for (auto& entry : configStore->store()) {
449 std::stringstream ss;
450 std::string str;
451 if (entry.second.empty() == false) {
452 boost::property_tree::json_parser::write_json(ss, entry.second, false);
453 str = ss.str();
454 str.pop_back(); // remove EoL
455 } else {
456 str = entry.second.get_value<std::string>();
457 }
458 std::string configString = fmt::format("[CONFIG] {}={} 1 {}", entry.first, str, configStore->provenance(entry.first.c_str())).c_str();
459 mServiceRegistry.get<DriverClient>(ServiceRegistry::globalDeviceSalt()).tell(configString.c_str());
460 }
461
462 mConfigRegistry = std::make_unique<ConfigParamRegistry>(std::move(configStore));
463
464 // Setup the error handlers for init
465 if (context.initError) {
466 context.initErrorHandling = [&errorCallback = context.initError,
467 &serviceRegistry = mServiceRegistry](RuntimeErrorRef e) {
470 ServiceRegistryRef ref{serviceRegistry, ServiceRegistry::globalDeviceSalt()};
471 auto& context = ref.get<DataProcessorContext>();
472 auto& err = error_from_ref(e);
473 O2_SIGNPOST_ID_FROM_POINTER(cid, device, &context);
474 O2_SIGNPOST_EVENT_EMIT_ERROR(device, cid, "Init", "Exception caught while in Init: %{public}s. Invoking errorCallback.", err.what);
475 BacktraceHelpers::demangled_backtrace_symbols(err.backtrace, err.maxBacktrace, STDERR_FILENO);
476 auto& stats = ref.get<DataProcessingStats>();
477 stats.updateStats({(int)ProcessingStatsId::EXCEPTION_COUNT, DataProcessingStats::Op::Add, 1});
478 InitErrorContext errorContext{ref, e};
479 errorCallback(errorContext);
480 };
481 } else {
482 context.initErrorHandling = [&serviceRegistry = mServiceRegistry](RuntimeErrorRef e) {
483 auto& err = error_from_ref(e);
486 ServiceRegistryRef ref{serviceRegistry, ServiceRegistry::globalDeviceSalt()};
487 auto& context = ref.get<DataProcessorContext>();
488 O2_SIGNPOST_ID_FROM_POINTER(cid, device, &context);
489 O2_SIGNPOST_EVENT_EMIT_ERROR(device, cid, "Init", "Exception caught while in Init: %{public}s. Exiting with 1.", err.what);
490 BacktraceHelpers::demangled_backtrace_symbols(err.backtrace, err.maxBacktrace, STDERR_FILENO);
491 auto& stats = ref.get<DataProcessingStats>();
492 stats.updateStats({(int)ProcessingStatsId::EXCEPTION_COUNT, DataProcessingStats::Op::Add, 1});
493 exit(1);
494 };
495 }
496
497 context.expirationHandlers.clear();
498 context.init = spec.algorithm.onInit;
499 if (context.init) {
500 static bool noCatch = getenv("O2_NO_CATCHALL_EXCEPTIONS") && strcmp(getenv("O2_NO_CATCHALL_EXCEPTIONS"), "0");
501 InitContext initContext{*mConfigRegistry, mServiceRegistry};
502
503 if (noCatch) {
504 try {
505 context.statefulProcess = context.init(initContext);
506 } catch (o2::framework::RuntimeErrorRef e) {
507 if (context.initErrorHandling) {
508 (context.initErrorHandling)(e);
509 }
510 }
511 } else {
512 try {
513 context.statefulProcess = context.init(initContext);
514 } catch (std::exception& ex) {
518 auto e = runtime_error(ex.what());
519 (context.initErrorHandling)(e);
520 } catch (o2::framework::RuntimeErrorRef e) {
521 (context.initErrorHandling)(e);
522 }
523 }
524 }
525 auto& state = ref.get<DeviceState>();
526 state.inputChannelInfos.resize(spec.inputChannels.size());
530 int validChannelId = 0;
531 for (size_t ci = 0; ci < spec.inputChannels.size(); ++ci) {
532 auto& name = spec.inputChannels[ci].name;
533 if (name.find(spec.channelPrefix + "from_internal-dpl-clock") == 0) {
534 state.inputChannelInfos[ci].state = InputChannelState::Pull;
535 state.inputChannelInfos[ci].id = {ChannelIndex::INVALID};
536 validChannelId++;
537 } else {
538 state.inputChannelInfos[ci].id = {validChannelId++};
539 }
540 }
541
542 // Invoke the callback policy for this device.
543 if (spec.callbacksPolicy.policy != nullptr) {
544 InitContext initContext{*mConfigRegistry, mServiceRegistry};
545 spec.callbacksPolicy.policy(mServiceRegistry.get<CallbackService>(ServiceRegistry::globalDeviceSalt()), initContext);
546 }
547
548 // Services which are stream should be initialised now
549 auto* options = GetConfig();
550 for (size_t si = 0; si < mStreams.size(); ++si) {
551 ServiceRegistry::Salt streamSalt = ServiceRegistry::streamSalt(si + 1, ServiceRegistry::globalDeviceSalt().dataProcessorId);
552 mServiceRegistry.lateBindStreamServices(state, *options, streamSalt);
553 }
554 O2_SIGNPOST_END(device, cid, "Init", "Exiting Init callback.");
555}

557void on_signal_callback(uv_signal_t* handle, int signum)
558{
559 O2_SIGNPOST_ID_FROM_POINTER(sid, device, handle);
560 O2_SIGNPOST_START(device, sid, "signal_state", "Signal %d received.", signum);
561
562 auto* registry = (ServiceRegistry*)handle->data;
563 if (!registry) {
564 O2_SIGNPOST_END(device, sid, "signal_state", "No registry active. Ignoring signal.");
565 return;
566 }
567 ServiceRegistryRef ref{*registry};
568 auto& state = ref.get<DeviceState>();
569 auto& quotaEvaluator = ref.get<ComputingQuotaEvaluator>();
570 auto& stats = ref.get<DataProcessingStats>();
571 state.loopReason |= DeviceState::SIGNAL_ARRIVED;
572 size_t ri = 0;
573 while (ri != quotaEvaluator.mOffers.size()) {
574 auto& offer = quotaEvaluator.mOffers[ri];
575 // We were already offered some sharedMemory, so we
576 // do not consider the offer.
577 // FIXME: in principle this should account for memory
578 // available and being offered, however we
579 // want to get out of the woods for now.
580 if (offer.valid && offer.sharedMemory != 0) {
581 O2_SIGNPOST_END(device, sid, "signal_state", "Memory already offered.");
582 return;
583 }
584 ri++;
585 }

590 offer.memory = 0;

601static auto toBeForwardedHeader = [](void* header) -> bool {
602 // If is now possible that the record is not complete when
603 // we forward it, because of a custom completion policy.
604 // this means that we need to skip the empty entries in the
605 // record for being forwarded.
606 if (header == nullptr) {
607 return false;
608 }
609 auto sih = o2::header::get<SourceInfoHeader*>(header);
610 if (sih) {
611 return false;
612 }
613
614 auto dih = o2::header::get<DomainInfoHeader*>(header);
615 if (dih) {
616 return false;
617 }
618
619 auto dh = o2::header::get<DataHeader*>(header);
620 if (!dh) {
621 return false;
622 }
623 auto dph = o2::header::get<DataProcessingHeader*>(header);
624 if (!dph) {
625 return false;
626 }
627 return true;
628};

630static auto toBeforwardedMessageSet = [](std::vector<ChannelIndex>& cachedForwardingChoices,
631 FairMQDeviceProxy& proxy,
632 std::unique_ptr<fair::mq::Message>& header,
633 std::unique_ptr<fair::mq::Message>& payload,
634 size_t total,
635 bool consume) {
636 if (header.get() == nullptr) {
637 // Missing an header is not an error anymore.
638 // it simply means that we did not receive the
639 // given input, but we were asked to
640 // consume existing, so we skip it.
641 return false;
642 }
643 if (payload.get() == nullptr && consume == true) {
644 // If the payload is not there, it means we already
645 // processed it with ConsumeExisiting. Therefore we
646 // need to do something only if this is the last consume.
647 header.reset(nullptr);
648 return false;
649 }
650
651 auto fdph = o2::header::get<DataProcessingHeader*>(header->GetData());
652 if (fdph == nullptr) {
653 LOG(error) << "Data is missing DataProcessingHeader";
654 return false;
655 }
656 auto fdh = o2::header::get<DataHeader*>(header->GetData());
657 if (fdh == nullptr) {
658 LOG(error) << "Data is missing DataHeader";
659 return false;
660 }
661
662 // We need to find the forward route only for the first
663 // part of a split payload. All the others will use the same.
664 // but always check if we have a sequence of multiple payloads
665 if (fdh->splitPayloadIndex == 0 || fdh->splitPayloadParts <= 1 || total > 1) {
666 proxy.getMatchingForwardChannelIndexes(cachedForwardingChoices, *fdh, fdph->startTime);
667 }
668 return cachedForwardingChoices.empty() == false;
669};

686 for (int fi = 0; fi < proxy.getNumForwardChannels(); fi++) {
687 auto& info = proxy.getForwardChannelInfo(ChannelIndex{fi});
688 auto& state = proxy.getForwardChannelState(ChannelIndex{fi});

704// This is how we do the forwarding, i.e. we push
705// the inputs which are shared between this device and others
706// to the next one in the daisy chain.
707// FIXME: do it in a smarter way than O(N^2)
708static auto forwardInputs = [](ServiceRegistryRef registry, TimesliceSlot slot, std::vector<MessageSet>& currentSetOfInputs,
709 TimesliceIndex::OldestOutputInfo oldestTimeslice, bool copy, bool consume = true) {
710 auto& proxy = registry.get<FairMQDeviceProxy>();
711 // we collect all messages per forward in a map and send them together
712 std::vector<fair::mq::Parts> forwardedParts;
713 forwardedParts.resize(proxy.getNumForwards());
714 std::vector<ChannelIndex> cachedForwardingChoices{};
715 O2_SIGNPOST_ID_GENERATE(sid, forwarding);
716 O2_SIGNPOST_START(forwarding, sid, "forwardInputs", "Starting forwarding for slot %zu with oldestTimeslice %zu %{public}s%{public}s%{public}s",
717 slot.index, oldestTimeslice.timeslice.value, copy ? "with copy" : "", copy && consume ? " and " : "", consume ? "with consume" : "");
718
719 for (size_t ii = 0, ie = currentSetOfInputs.size(); ii < ie; ++ii) {
720 auto& messageSet = currentSetOfInputs[ii];
721 // In case the messageSet is empty, there is nothing to be done.
722 if (messageSet.size() == 0) {
723 continue;
724 }
725 if (!toBeForwardedHeader(messageSet.header(0)->GetData())) {
726 continue;
727 }
728 cachedForwardingChoices.clear();
729
730 for (size_t pi = 0; pi < currentSetOfInputs[ii].size(); ++pi) {
731 auto& messageSet = currentSetOfInputs[ii];
732 auto& header = messageSet.header(pi);
733 auto& payload = messageSet.payload(pi);
734 auto total = messageSet.getNumberOfPayloads(pi);
735
736 if (!toBeforwardedMessageSet(cachedForwardingChoices, proxy, header, payload, total, consume)) {
737 continue;
738 }
739
740 // In case of more than one forward route, we need to copy the message.
741 // This will eventually use the same mamory if running with the same backend.
742 if (cachedForwardingChoices.size() > 1) {
743 copy = true;
744 }
745 auto* dh = o2::header::get<DataHeader*>(header->GetData());
746 auto* dph = o2::header::get<DataProcessingHeader*>(header->GetData());
747
748 if (copy) {
749 for (auto& cachedForwardingChoice : cachedForwardingChoices) {
750 auto&& newHeader = header->GetTransport()->CreateMessage();
751 O2_SIGNPOST_EVENT_EMIT(forwarding, sid, "forwardInputs", "Forwarding a copy of %{public}s to route %d.",
752 fmt::format("{}/{}/{}@timeslice:{} tfCounter:{}", dh->dataOrigin, dh->dataDescription, dh->subSpecification, dph->startTime, dh->tfCounter).c_str(), cachedForwardingChoice.value);
753 newHeader->Copy(*header);
754 forwardedParts[cachedForwardingChoice.value].AddPart(std::move(newHeader));
755
756 for (size_t payloadIndex = 0; payloadIndex < messageSet.getNumberOfPayloads(pi); ++payloadIndex) {
757 auto&& newPayload = header->GetTransport()->CreateMessage();
758 newPayload->Copy(*messageSet.payload(pi, payloadIndex));
759 forwardedParts[cachedForwardingChoice.value].AddPart(std::move(newPayload));
760 }
761 }
762 } else {
763 O2_SIGNPOST_EVENT_EMIT(forwarding, sid, "forwardInputs", "Forwarding %{public}s to route %d.",
764 fmt::format("{}/{}/{}@timeslice:{} tfCounter:{}", dh->dataOrigin, dh->dataDescription, dh->subSpecification, dph->startTime, dh->tfCounter).c_str(), cachedForwardingChoices.back().value);
765 forwardedParts[cachedForwardingChoices.back().value].AddPart(std::move(messageSet.header(pi)));
766 for (size_t payloadIndex = 0; payloadIndex < messageSet.getNumberOfPayloads(pi); ++payloadIndex) {
767 forwardedParts[cachedForwardingChoices.back().value].AddPart(std::move(messageSet.payload(pi, payloadIndex)));
768 }
769 }
770 }
771 }
772 O2_SIGNPOST_EVENT_EMIT(forwarding, sid, "forwardInputs", "Forwarding %zu messages", forwardedParts.size());
773 for (int fi = 0; fi < proxy.getNumForwardChannels(); fi++) {
774 if (forwardedParts[fi].Size() == 0) {
775 continue;
776 }
777 ForwardChannelInfo info = proxy.getForwardChannelInfo(ChannelIndex{fi});
778 auto& parts = forwardedParts[fi];
779 if (info.policy == nullptr) {
780 O2_SIGNPOST_EVENT_EMIT_ERROR(forwarding, sid, "forwardInputs", "Forwarding to %{public}s %d has no policy.", info.name.c_str(), fi);
781 continue;
782 }
783 O2_SIGNPOST_EVENT_EMIT(forwarding, sid, "forwardInputs", "Forwarding to %{public}s %d", info.name.c_str(), fi);
784 info.policy->forward(parts, ChannelIndex{fi}, registry);
785 }
786
787 auto& asyncQueue = registry.get<AsyncQueue>();
788 auto& decongestion = registry.get<DecongestionService>();
789 O2_SIGNPOST_ID_GENERATE(aid, async_queue);
790 O2_SIGNPOST_EVENT_EMIT(async_queue, aid, "forwardInputs", "Queuing forwarding oldestPossible %zu", oldestTimeslice.timeslice.value);
791 AsyncQueueHelpers::post(asyncQueue, AsyncTask{.timeslice = oldestTimeslice.timeslice, .id = decongestion.oldestPossibleTimesliceTask, .debounce = -1, .callback = decongestionCallbackLate}
792 .user<DecongestionContext>({.ref = registry, .oldestTimeslice = oldestTimeslice}));
793 O2_SIGNPOST_END(forwarding, sid, "forwardInputs", "Forwarding done");
794};

828 auto ref = ServiceRegistryRef{mServiceRegistry};
829 auto& deviceContext = ref.get<DeviceContext>();
830 auto& context = ref.get<DataProcessorContext>();
831 auto& spec = ref.get<DeviceSpec const>();
832 auto& state = ref.get<DeviceState>();
833 // We add a timer only in case a channel poller is not there.
834 if ((context.statefulProcess != nullptr) || (context.statelessProcess != nullptr)) {
835 for (auto& [channelName, channel] : GetChannels()) {
836 InputChannelInfo* channelInfo;
837 for (size_t ci = 0; ci < spec.inputChannels.size(); ++ci) {
838 auto& channelSpec = spec.inputChannels[ci];
839 channelInfo = &state.inputChannelInfos[ci];
840 if (channelSpec.name != channelName) {
841 continue;
842 }
843 channelInfo->channel = &this->GetChannel(channelName, 0);
844 break;
845 }
846 if ((channelName.rfind("from_internal-dpl", 0) == 0) &&
847 (channelName.rfind("from_internal-dpl-aod", 0) != 0) &&
848 (channelName.rfind("from_internal-dpl-ccdb-backend", 0) != 0) &&
849 (channelName.rfind("from_internal-dpl-injected", 0)) != 0) {
850 LOGP(detail, "{} is an internal channel. Skipping as no input will come from there.", channelName);
851 continue;
852 }
853 // We only watch receiving sockets.
854 if (channelName.rfind("from_" + spec.name + "_", 0) == 0) {
855 LOGP(detail, "{} is to send data. Not polling.", channelName);
856 continue;
857 }
858
859 if (channelName.rfind("from_", 0) != 0) {
860 LOGP(detail, "{} is not a DPL socket. Not polling.", channelName);
861 continue;
862 }
863
864 // We assume there is always a ZeroMQ socket behind.
865 int zmq_fd = 0;
866 size_t zmq_fd_len = sizeof(zmq_fd);
867 // FIXME: I should probably save those somewhere... ;-)
868 auto* poller = (uv_poll_t*)malloc(sizeof(uv_poll_t));
869 channel[0].GetSocket().GetOption("fd", &zmq_fd, &zmq_fd_len);
870 if (zmq_fd == 0) {
871 LOG(error) << "Cannot get file descriptor for channel." << channelName;
872 continue;
873 }
874 LOGP(detail, "Polling socket for {}", channelName);
875 auto* pCtx = (PollerContext*)malloc(sizeof(PollerContext));
876 pCtx->name = strdup(channelName.c_str());
877 pCtx->loop = state.loop;
878 pCtx->device = this;
879 pCtx->state = &state;
880 pCtx->fd = zmq_fd;
881 assert(channelInfo != nullptr);
882 pCtx->channelInfo = channelInfo;
883 pCtx->socket = &channel[0].GetSocket();
884 pCtx->read = true;
885 poller->data = pCtx;
886 uv_poll_init(state.loop, poller, zmq_fd);
887 if (channelName.rfind("from_", 0) != 0) {
888 LOGP(detail, "{} is an out of band channel.", channelName);
889 state.activeOutOfBandPollers.push_back(poller);
890 } else {
891 channelInfo->pollerIndex = state.activeInputPollers.size();
892 state.activeInputPollers.push_back(poller);
893 }
894 }
895 // In case we do not have any input channel and we do not have
896 // any timers or signal watchers we still wake up whenever we can send data to downstream
897 // devices to allow for enumerations.
898 if (state.activeInputPollers.empty() &&
899 state.activeOutOfBandPollers.empty() &&
900 state.activeTimers.empty() &&
901 state.activeSignals.empty()) {
902 // FIXME: this is to make sure we do not reset the output timer
903 // for readout proxies or similar. In principle this should go once
904 // we move to OutOfBand InputSpec.
905 if (state.inputChannelInfos.empty()) {
906 LOGP(detail, "No input channels. Setting exit transition timeout to 0.");
907 deviceContext.exitTransitionTimeout = 0;
908 }
909 for (auto& [channelName, channel] : GetChannels()) {
910 if (channelName.rfind(spec.channelPrefix + "from_internal-dpl", 0) == 0) {
911 LOGP(detail, "{} is an internal channel. Not polling.", channelName);
912 continue;
913 }
914 if (channelName.rfind(spec.channelPrefix + "from_" + spec.name + "_", 0) == 0) {
915 LOGP(detail, "{} is an out of band channel. Not polling for output.", channelName);
916 continue;
917 }
918 // We assume there is always a ZeroMQ socket behind.
919 int zmq_fd = 0;
920 size_t zmq_fd_len = sizeof(zmq_fd);
921 // FIXME: I should probably save those somewhere... ;-)
922 auto* poller = (uv_poll_t*)malloc(sizeof(uv_poll_t));
923 channel[0].GetSocket().GetOption("fd", &zmq_fd, &zmq_fd_len);
924 if (zmq_fd == 0) {
925 LOGP(error, "Cannot get file descriptor for channel {}", channelName);
926 continue;
927 }
928 LOG(detail) << "Polling socket for " << channel[0].GetName();
929 // FIXME: leak
930 auto* pCtx = (PollerContext*)malloc(sizeof(PollerContext));
931 pCtx->name = strdup(channelName.c_str());
932 pCtx->loop = state.loop;
933 pCtx->device = this;
934 pCtx->state = &state;
935 pCtx->fd = zmq_fd;
936 pCtx->read = false;
937 poller->data = pCtx;
938 uv_poll_init(state.loop, poller, zmq_fd);
939 state.activeOutputPollers.push_back(poller);
940 }
941 }
942 } else {
943 LOGP(detail, "This is a fake device so we exit after the first iteration.");
944 deviceContext.exitTransitionTimeout = 0;
945 // This is a fake device, so we can request to exit immediately
946 ServiceRegistryRef ref{mServiceRegistry};
947 ref.get<ControlService>().readyToQuit(QuitRequest::Me);
948 // A two second timer to stop internal devices which do not want to
949 auto* timer = (uv_timer_t*)malloc(sizeof(uv_timer_t));
950 uv_timer_init(state.loop, timer);
951 timer->data = &state;
952 uv_update_time(state.loop);
953 uv_timer_start(timer, on_idle_timer, 2000, 2000);
954 state.activeTimers.push_back(timer);
955 }

1026void DataProcessingDevice::InitTask()
1027{
1028 auto ref = ServiceRegistryRef{mServiceRegistry};
1029 auto& deviceContext = ref.get<DeviceContext>();
1030 auto& context = ref.get<DataProcessorContext>();
1031
1032 O2_SIGNPOST_ID_FROM_POINTER(cid, device, &context);
1033 O2_SIGNPOST_START(device, cid, "InitTask", "Entering InitTask callback.");
1034 auto& spec = getRunningDevice(mRunningDevice, mServiceRegistry);
1035 auto distinct = DataRelayerHelpers::createDistinctRouteIndex(spec.inputs);
1036 auto& state = ref.get<DeviceState>();
1037 int i = 0;
1038 for (auto& di : distinct) {
1039 auto& route = spec.inputs[di];
1040 if (route.configurator.has_value() == false) {
1041 i++;
1042 continue;
1043 }
1044 ExpirationHandler handler{
1045 .name = route.configurator->name,
1046 .routeIndex = RouteIndex{i++},
1047 .lifetime = route.matcher.lifetime,
1048 .creator = route.configurator->creatorConfigurator(state, mServiceRegistry, *mConfigRegistry),
1049 .checker = route.configurator->danglingConfigurator(state, *mConfigRegistry),
1050 .handler = route.configurator->expirationConfigurator(state, *mConfigRegistry)};
1051 context.expirationHandlers.emplace_back(std::move(handler));
1052 }
1053
1054 if (state.awakeMainThread == nullptr) {
1055 state.awakeMainThread = (uv_async_t*)malloc(sizeof(uv_async_t));
1056 state.awakeMainThread->data = &state;
1057 uv_async_init(state.loop, state.awakeMainThread, on_awake_main_thread);
1058 }
1059
1060 deviceContext.expectedRegionCallbacks = std::stoi(fConfig->GetValue<std::string>("expected-region-callbacks"));
1061 deviceContext.exitTransitionTimeout = std::stoi(fConfig->GetValue<std::string>("exit-transition-timeout"));
1062 deviceContext.dataProcessingTimeout = std::stoi(fConfig->GetValue<std::string>("data-processing-timeout"));
1063
1064 for (auto& channel : GetChannels()) {
1065 channel.second.at(0).Transport()->SubscribeToRegionEvents([&context = deviceContext,
1066 &registry = mServiceRegistry,
1067 &pendingRegionInfos = mPendingRegionInfos,
1068 &regionInfoMutex = mRegionInfoMutex](fair::mq::RegionInfo info) {
1069 std::lock_guard<std::mutex> lock(regionInfoMutex);
1070 LOG(detail) << ">>> Region info event" << info.event;
1071 LOG(detail) << "id: " << info.id;
1072 LOG(detail) << "ptr: " << info.ptr;
1073 LOG(detail) << "size: " << info.size;
1074 LOG(detail) << "flags: " << info.flags;
1075 // Now we check for pending events with the mutex,
1076 // so the lines below are atomic.
1077 pendingRegionInfos.push_back(info);
1078 context.expectedRegionCallbacks -= 1;
1079 // We always want to handle these on the main loop,
1080 // so we awake it.
1081 ServiceRegistryRef ref{registry};
1082 uv_async_send(ref.get<DeviceState>().awakeMainThread);
1083 });
1084 }
1085
1086 // Add a signal manager for SIGUSR1 so that we can force
1087 // an event from the outside, making sure that the event loop can
1088 // be unblocked (e.g. by a quitting DPL driver) even when there
1089 // is no data pending to be processed.
1090 if (deviceContext.sigusr1Handle == nullptr) {
1091 deviceContext.sigusr1Handle = (uv_signal_t*)malloc(sizeof(uv_signal_t));
1092 deviceContext.sigusr1Handle->data = &mServiceRegistry;
1093 uv_signal_init(state.loop, deviceContext.sigusr1Handle);
1094 uv_signal_start(deviceContext.sigusr1Handle, on_signal_callback, SIGUSR1);
1095 }
1096 // If there is any signal, we want to make sure they are active
1097 for (auto& handle : state.activeSignals) {
1098 handle->data = &state;
1099 }
1100 // When we start, we must make sure that we do listen to the signal
1101 deviceContext.sigusr1Handle->data = &mServiceRegistry;
1102
1104 DataProcessingDevice::initPollers();
1105
1106 // Whenever we InitTask, we consider as if the previous iteration
1107 // was successful, so that even if there is no timer or receiving
1108 // channel, we can still start an enumeration.
1109 DataProcessorContext* initialContext = nullptr;
1110 bool idle = state.lastActiveDataProcessor.compare_exchange_strong(initialContext, (DataProcessorContext*)-1);
1111 if (!idle) {
1112 LOG(error) << "DataProcessor " << state.lastActiveDataProcessor.load()->spec->name << " was unexpectedly active";
1113 }
1114
1115 // We should be ready to run here. Therefore we copy all the
1116 // required parts in the DataProcessorContext. Eventually we should
1117 // do so on a per thread basis, with fine grained locks.
1118 // FIXME: this should not use ServiceRegistry::threadSalt, but
1119 // more a ServiceRegistry::globalDataProcessorSalt(N) where
1120 // N is the number of the multiplexed data processor.
1121 // We will get there.
1122 this->fillContext(mServiceRegistry.get<DataProcessorContext>(ServiceRegistry::globalDeviceSalt()), deviceContext);
1123
1124 O2_SIGNPOST_END(device, cid, "InitTask", "Exiting InitTask callback waiting for the remaining region callbacks.");
1125

1127 std::lock_guard<std::mutex> lock(mutex);
1128 return (pendingRegionInfos.empty() == false) || deviceContext.expectedRegionCallbacks > 0;
1129 };
1130 O2_SIGNPOST_START(device, cid, "InitTask", "Waiting for registation events.");
1135 while (hasPendingEvents(deviceContext)) {
1136 // Wait for the callback to signal its done, so that we do not busy wait.
1137 uv_run(state.loop, UV_RUN_ONCE);
1138 // Handle callbacks if any
1139 {
1140 O2_SIGNPOST_EVENT_EMIT(device, cid, "InitTask", "Memory registration event received.");
1141 std::lock_guard<std::mutex> lock(mRegionInfoMutex);
1142 handleRegionCallbacks(mServiceRegistry, mPendingRegionInfos);
1143 }
1144 }
1145 O2_SIGNPOST_END(device, cid, "InitTask", "Done waiting for registration events.");
1146}

1247 context.canForwardEarly = true;

1334TransitionHandlingState updateStateTransition(ServiceRegistryRef& ref, ProcessingPolicies const& policies)
1335{
1336 auto& state = ref.get<DeviceState>();
1337 auto& deviceProxy = ref.get<FairMQDeviceProxy>();
1338 if (state.transitionHandling != TransitionHandlingState::NoTransition || deviceProxy.newStateRequested() == false) {
1339 return state.transitionHandling;
1340 }
1341 O2_SIGNPOST_ID_FROM_POINTER(lid, device, state.loop);
1342 auto& deviceContext = ref.get<DeviceContext>();
1343 // Check if we only have timers
1344 auto& spec = ref.get<DeviceSpec const>();
1345 if (hasOnlyTimers(spec)) {
1346 switchState(ref, StreamingState::EndOfStreaming);
1347 }
1348
1349 // We do not do anything in particular if the data processing timeout would go past the exitTransitionTimeout
1350 if (deviceContext.dataProcessingTimeout > 0 && deviceContext.dataProcessingTimeout < deviceContext.exitTransitionTimeout) {
1351 uv_update_time(state.loop);
1352 O2_SIGNPOST_EVENT_EMIT(calibration, lid, "timer_setup", "Starting %d s timer for dataProcessingTimeout.", deviceContext.dataProcessingTimeout);
1353 uv_timer_start(deviceContext.dataProcessingGracePeriodTimer, on_data_processing_expired, deviceContext.dataProcessingTimeout * 1000, 0);
1354 }
1355 if (deviceContext.exitTransitionTimeout != 0 && state.streaming != StreamingState::Idle) {
1356 ref.get<CallbackService>().call<CallbackService::Id::ExitRequested>(ServiceRegistryRef{ref});
1357 uv_update_time(state.loop);
1358 O2_SIGNPOST_EVENT_EMIT(calibration, lid, "timer_setup", "Starting %d s timer for exitTransitionTimeout.",
1359 deviceContext.exitTransitionTimeout);
1360 uv_timer_start(deviceContext.gracePeriodTimer, on_transition_requested_expired, deviceContext.exitTransitionTimeout * 1000, 0);
1361 bool onlyGenerated = hasOnlyGenerated(spec);
1362 int timeout = onlyGenerated ? deviceContext.dataProcessingTimeout : deviceContext.exitTransitionTimeout;
1363 if (policies.termination == TerminationPolicy::QUIT && DefaultsHelpers::onlineDeploymentMode() == false) {
1364 O2_SIGNPOST_EVENT_EMIT_INFO(device, lid, "run_loop", "New state requested. Waiting for %d seconds before quitting.", timeout);
1365 } else {
1366 O2_SIGNPOST_EVENT_EMIT_INFO(device, lid, "run_loop",
1367 "New state requested. Waiting for %d seconds before %{public}s",
1368 timeout,
1369 onlyGenerated ? "dropping remaining input and switching to READY state." : "switching to READY state.");
1370 }
1371 return TransitionHandlingState::Requested;
1372 } else {
1373 if (deviceContext.exitTransitionTimeout == 0 && policies.termination == TerminationPolicy::QUIT) {
1374 O2_SIGNPOST_EVENT_EMIT_INFO(device, lid, "run_loop", "New state requested. No timeout set, quitting immediately as per --completion-policy");
1375 } else if (deviceContext.exitTransitionTimeout == 0 && policies.termination != TerminationPolicy::QUIT) {
1376 O2_SIGNPOST_EVENT_EMIT_INFO(device, lid, "run_loop", "New state requested. No timeout set, switching to READY state immediately");
1377 } else if (policies.termination == TerminationPolicy::QUIT) {
1378 O2_SIGNPOST_EVENT_EMIT_INFO(device, lid, "run_loop", "New state pending and we are already idle, quitting immediately as per --completion-policy");
1379 } else {
1380 O2_SIGNPOST_EVENT_EMIT_INFO(device, lid, "run_loop", "New state pending and we are already idle, switching to READY immediately.");
1381 }
1382 return TransitionHandlingState::Expired;
1383 }

1386void DataProcessingDevice::Run()
1387{
1388 ServiceRegistryRef ref{mServiceRegistry};
1389 auto& state = ref.get<DeviceState>();
1390 state.loopReason = DeviceState::LoopReason::FIRST_LOOP;
1391 bool firstLoop = true;
1392 O2_SIGNPOST_ID_FROM_POINTER(lid, device, state.loop);
1393 O2_SIGNPOST_START(device, lid, "device_state", "First iteration of the device loop");
1394
1395 bool dplEnableMultithreding = getenv("DPL_THREADPOOL_SIZE") != nullptr;
1396 if (dplEnableMultithreding) {
1397 setenv("UV_THREADPOOL_SIZE", "1", 1);
1398 }
1399
1400 while (state.transitionHandling != TransitionHandlingState::Expired) {
1401 if (state.nextFairMQState.empty() == false) {
1402 (void)this->ChangeState(state.nextFairMQState.back());
1403 state.nextFairMQState.pop_back();
1404 }
1405 // Notify on the main thread the new region callbacks, making sure
1406 // no callback is issued if there is something still processing.
1407 {
1408 std::lock_guard<std::mutex> lock(mRegionInfoMutex);
1409 handleRegionCallbacks(mServiceRegistry, mPendingRegionInfos);
1410 }
1411 // This will block for the correct delay (or until we get data
1412 // on a socket). We also do not block on the first iteration
1413 // so that devices which do not have a timer can still start an
1414 // enumeration.
1415 {
1416 ServiceRegistryRef ref{mServiceRegistry};
1417 ref.get<DriverClient>().flushPending(mServiceRegistry);
1418 DataProcessorContext* lastActive = state.lastActiveDataProcessor.load();
1419 // Reset to zero unless some other DataPorcessorContext completed in the meanwhile.
1420 // In such case we will take care of it at next iteration.
1421 state.lastActiveDataProcessor.compare_exchange_strong(lastActive, nullptr);
1422
1423 auto shouldNotWait = (lastActive != nullptr &&
1424 (state.streaming != StreamingState::Idle) && (state.activeSignals.empty())) ||
1425 (state.streaming == StreamingState::EndOfStreaming);
1426 if (firstLoop) {
1427 shouldNotWait = true;
1428 firstLoop = false;
1429 }
1430 if (lastActive != nullptr) {
1431 state.loopReason |= DeviceState::LoopReason::PREVIOUSLY_ACTIVE;
1432 }
1433 if (NewStatePending()) {
1434 O2_SIGNPOST_EVENT_EMIT(device, lid, "run_loop", "New state pending. Waiting for it to be handled.");
1435 shouldNotWait = true;
1436 state.loopReason |= DeviceState::LoopReason::NEW_STATE_PENDING;
1437 }
1438 state.transitionHandling = updateStateTransition(ref, ref.get<DeviceContext>().processingPolicies);
1439 // If we are Idle, we can then consider the transition to be expired.
1440 if (state.transitionHandling == TransitionHandlingState::Requested && state.streaming == StreamingState::Idle) {
1441 O2_SIGNPOST_EVENT_EMIT(device, lid, "run_loop", "State transition requested and we are now in Idle. We can consider it to be completed.");
1442 state.transitionHandling = TransitionHandlingState::Expired;
1443 }
1444 if (state.severityStack.empty() == false) {
1445 fair::Logger::SetConsoleSeverity((fair::Severity)state.severityStack.back());
1446 state.severityStack.pop_back();
1447 }
1448 // for (auto &info : mDeviceContext.state->inputChannelInfos) {
1449 // shouldNotWait |= info.readPolled;
1450 // }
1451 state.loopReason = DeviceState::NO_REASON;
1452 state.firedTimers.clear();
1453 if ((state.tracingFlags & DeviceState::LoopReason::TRACE_CALLBACKS) != 0) {
1454 state.severityStack.push_back((int)fair::Logger::GetConsoleSeverity());
1455 fair::Logger::SetConsoleSeverity(fair::Severity::trace);
1456 }
1457 // Run the asynchronous queue just before sleeping again, so that:
1458 // - we can trigger further events from the queue
1459 // - we can guarantee this is the last thing we do in the loop (
1460 // assuming no one else is adding to the queue before this point).
1461 auto onDrop = [&registry = mServiceRegistry, lid](TimesliceSlot slot, std::vector<MessageSet>& dropped, TimesliceIndex::OldestOutputInfo oldestOutputInfo) {
1462 O2_SIGNPOST_START(device, lid, "run_loop", "Dropping message from slot %" PRIu64 ". Forwarding as needed.", (uint64_t)slot.index);
1463 ServiceRegistryRef ref{registry};
1464 ref.get<AsyncQueue>();
1465 ref.get<DecongestionService>();
1466 ref.get<DataRelayer>();
1467 // Get the current timeslice for the slot.
1468 auto& variables = ref.get<TimesliceIndex>().getVariablesForSlot(slot);
1469 VariableContextHelpers::getTimeslice(variables);
1470 forwardInputs(registry, slot, dropped, oldestOutputInfo, false, true);
1471 };
1472 auto& relayer = ref.get<DataRelayer>();
1473 relayer.prunePending(onDrop);
1474 auto& queue = ref.get<AsyncQueue>();
1475 auto oldestPossibleTimeslice = relayer.getOldestPossibleOutput();
1476 AsyncQueueHelpers::run(queue, {oldestPossibleTimeslice.timeslice.value});
1477 if (shouldNotWait == false) {
1478 auto& dpContext = ref.get<DataProcessorContext>();
1479 dpContext.preLoopCallbacks(ref);
1480 }
1481 O2_SIGNPOST_END(device, lid, "run_loop", "Run loop completed. %{}s", shouldNotWait ? "Will immediately schedule a new one" : "Waiting for next event.");
1482 uv_run(state.loop, shouldNotWait ? UV_RUN_NOWAIT : UV_RUN_ONCE);
1483 O2_SIGNPOST_START(device, lid, "run_loop", "Run loop started. Loop reason %d.", state.loopReason);
1484 if ((state.loopReason & state.tracingFlags) != 0) {
1485 state.severityStack.push_back((int)fair::Logger::GetConsoleSeverity());
1486 fair::Logger::SetConsoleSeverity(fair::Severity::trace);
1487 } else if (state.severityStack.empty() == false) {
1488 fair::Logger::SetConsoleSeverity((fair::Severity)state.severityStack.back());
1489 state.severityStack.pop_back();
1490 }
1491 O2_SIGNPOST_EVENT_EMIT(device, lid, "run_loop", "Loop reason mask %x & %x = %x", state.loopReason, state.tracingFlags, state.loopReason & state.tracingFlags);
1492
1493 if ((state.loopReason & DeviceState::LoopReason::OOB_ACTIVITY) != 0) {
1494 O2_SIGNPOST_EVENT_EMIT(device, lid, "run_loop", "Out of band activity detected. Rescanning everything.");
1495 relayer.rescan();
1496 }
1497
1498 if (!state.pendingOffers.empty()) {
1499 O2_SIGNPOST_EVENT_EMIT(device, lid, "run_loop", "Pending %" PRIu64 " offers. updating the ComputingQuotaEvaluator.", (uint64_t)state.pendingOffers.size());
1500 ref.get<ComputingQuotaEvaluator>().updateOffers(state.pendingOffers, uv_now(state.loop));
1501 }
1502 }
1503
1504 // Notify on the main thread the new region callbacks, making sure
1505 // no callback is issued if there is something still processing.
1506 // Notice that we still need to perform callbacks also after
1507 // the socket epolled, because otherwise we would end up serving
1508 // the callback after the first data arrives is the system is too
1509 // fast to transition from Init to Run.
1510 {
1511 std::lock_guard<std::mutex> lock(mRegionInfoMutex);
1512 handleRegionCallbacks(mServiceRegistry, mPendingRegionInfos);
1513 }
1514
1515 assert(mStreams.size() == mHandles.size());
1517 TaskStreamRef streamRef{-1};
1518 for (size_t ti = 0; ti < mStreams.size(); ti++) {
1519 auto& taskInfo = mStreams[ti];
1520 if (taskInfo.running) {
1521 continue;
1522 }
1523 // Stream 0 is for when we run in
1524 streamRef.index = ti;
1525 }
1526 using o2::monitoring::Metric;
1527 using o2::monitoring::Monitoring;
1528 using o2::monitoring::tags::Key;
1529 using o2::monitoring::tags::Value;
1530 // We have an empty stream, let's check if we have enough
1531 // resources for it to run something
1532 if (streamRef.index != -1) {
1533 // Synchronous execution of the callbacks. This will be moved in the
1534 // moved in the on_socket_polled once we have threading in place.
1535 uv_work_t& handle = mHandles[streamRef.index];
1536 TaskStreamInfo& stream = mStreams[streamRef.index];
1537 handle.data = &mStreams[streamRef.index];
1538
1539 static std::function<void(ComputingQuotaOffer const&, ComputingQuotaStats const& stats)> reportExpiredOffer = [&registry = mServiceRegistry](ComputingQuotaOffer const& offer, ComputingQuotaStats const& stats) {
1540 ServiceRegistryRef ref{registry};
1541 auto& dpStats = ref.get<DataProcessingStats>();
1542 dpStats.updateStats({static_cast<short>(ProcessingStatsId::RESOURCE_OFFER_EXPIRED), DataProcessingStats::Op::Set, stats.totalExpiredOffers});
1543 dpStats.updateStats({static_cast<short>(ProcessingStatsId::ARROW_BYTES_EXPIRED), DataProcessingStats::Op::Set, stats.totalExpiredBytes});
1544 dpStats.processCommandQueue();
1545 };
1546 auto ref = ServiceRegistryRef{mServiceRegistry};
1547
1548 // Deciding wether to run or not can be done by passing a request to
1549 // the evaluator. In this case, the request is always satisfied and
1550 // we run on whatever resource is available.
1551 auto& spec = ref.get<DeviceSpec const>();
1552 bool enough = ref.get<ComputingQuotaEvaluator>().selectOffer(streamRef.index, spec.resourcePolicy.request, uv_now(state.loop));
1553
1554 if (enough) {
1555 stream.id = streamRef;
1556 stream.running = true;
1557 stream.registry = &mServiceRegistry;
1558 if (dplEnableMultithreding) [[unlikely]] {
1559 stream.task = &handle;
1560 uv_queue_work(state.loop, stream.task, run_callback, run_completion);
1561 } else {
1562 run_callback(&handle);
1563 run_completion(&handle, 0);
1564 }
1565 } else {
1566 auto ref = ServiceRegistryRef{mServiceRegistry};
1567 ref.get<ComputingQuotaEvaluator>().handleExpired(reportExpiredOffer);
1568 }
1569 }
1570 }
1571
1572 O2_SIGNPOST_END(device, lid, "run_loop", "Run loop completed. Transition handling state %d.", (int)state.transitionHandling);
1573 auto& spec = ref.get<DeviceSpec const>();
1575 for (size_t ci = 0; ci < spec.inputChannels.size(); ++ci) {
1576 auto& info = state.inputChannelInfos[ci];
1577 info.parts.fParts.clear();
1578 }
1579 state.transitionHandling = TransitionHandlingState::NoTransition;
1580}

1584void DataProcessingDevice::doPrepare(ServiceRegistryRef ref)
1585{
1586 auto& context = ref.get<DataProcessorContext>();
1587 O2_SIGNPOST_ID_FROM_POINTER(dpid, device, &context);
1588 O2_SIGNPOST_START(device, dpid, "do_prepare", "Starting DataProcessorContext::doPrepare.");
1589
1590 {
1591 ref.get<CallbackService>().call<CallbackService::Id::ClockTick>();
1592 }
1593 // Whether or not we had something to do.
1594
1595 // Initialise the value for context.allDone. It will possibly be updated
1596 // below if any of the channels is not done.
1597 //
1598 // Notice that fake input channels (InputChannelState::Pull) cannot possibly
1599 // expect to receive an EndOfStream signal. Thus we do not wait for these
1600 // to be completed. In the case of data source devices, as they do not have
1601 // real data input channels, they have to signal EndOfStream themselves.
1602 auto& state = ref.get<DeviceState>();
1603 auto& spec = ref.get<DeviceSpec const>();
1604 O2_SIGNPOST_ID_FROM_POINTER(cid, device, state.inputChannelInfos.data());
1605 O2_SIGNPOST_START(device, cid, "do_prepare", "Reported channel states.");
1606 context.allDone = std::any_of(state.inputChannelInfos.begin(), state.inputChannelInfos.end(), [cid](const auto& info) {
1607 if (info.channel) {
1608 O2_SIGNPOST_EVENT_EMIT(device, cid, "do_prepare", "Input channel %{public}s%{public}s has %zu parts left and is in state %d.",
1609 info.channel->GetName().c_str(), (info.id.value == ChannelIndex::INVALID ? " (non DPL)" : ""), info.parts.fParts.size(), (int)info.state);
1610 } else {
1611 O2_SIGNPOST_EVENT_EMIT(device, cid, "do_prepare", "External channel %d is in state %d.", info.id.value, (int)info.state);
1612 }
1613 return (info.parts.fParts.empty() == true && info.state != InputChannelState::Pull);
1614 });
1615 O2_SIGNPOST_END(device, cid, "do_prepare", "End report.");
1616 O2_SIGNPOST_EVENT_EMIT(device, dpid, "do_prepare", "Processing %zu input channels.", spec.inputChannels.size());
1619 static std::vector<int> pollOrder;
1620 pollOrder.resize(state.inputChannelInfos.size());
1621 std::iota(pollOrder.begin(), pollOrder.end(), 0);
1622 std::sort(pollOrder.begin(), pollOrder.end(), [&infos = state.inputChannelInfos](int a, int b) {
1623 return infos[a].oldestForChannel.value < infos[b].oldestForChannel.value;
1624 });
1625
1626 // Nothing to poll...
1627 if (pollOrder.empty()) {
1628 O2_SIGNPOST_END(device, dpid, "do_prepare", "Nothing to poll. Waiting for next iteration.");
1629 return;
1630 }
1631 auto currentOldest = state.inputChannelInfos[pollOrder.front()].oldestForChannel;
1632 auto currentNewest = state.inputChannelInfos[pollOrder.back()].oldestForChannel;
1633 auto delta = currentNewest.value - currentOldest.value;
1634 O2_SIGNPOST_EVENT_EMIT(device, dpid, "do_prepare", "Oldest possible timeframe range %" PRIu64 " => %" PRIu64 " delta %" PRIu64,
1635 (int64_t)currentOldest.value, (int64_t)currentNewest.value, (int64_t)delta);
1636 auto& infos = state.inputChannelInfos;
1637
1638 if (context.balancingInputs) {
1639 static int pipelineLength = DefaultsHelpers::pipelineLength();
1640 static uint64_t ahead = getenv("DPL_MAX_CHANNEL_AHEAD") ? std::atoll(getenv("DPL_MAX_CHANNEL_AHEAD")) : std::max(8, std::min(pipelineLength - 48, pipelineLength / 2));
1641 auto newEnd = std::remove_if(pollOrder.begin(), pollOrder.end(), [&infos, limitNew = currentOldest.value + ahead](int a) -> bool {
1642 return infos[a].oldestForChannel.value > limitNew;
1643 });
1644 for (auto it = pollOrder.begin(); it < pollOrder.end(); it++) {
1645 const auto& channelInfo = state.inputChannelInfos[*it];
1646 if (channelInfo.pollerIndex != -1) {
1647 auto& poller = state.activeInputPollers[channelInfo.pollerIndex];
1648 auto& pollerContext = *(PollerContext*)(poller->data);
1649 if (pollerContext.pollerState == PollerContext::PollerState::Connected || pollerContext.pollerState == PollerContext::PollerState::Suspended) {
1650 bool running = pollerContext.pollerState == PollerContext::PollerState::Connected;
1651 bool shouldBeRunning = it < newEnd;
1652 if (running != shouldBeRunning) {
1653 uv_poll_start(poller, shouldBeRunning ? UV_READABLE | UV_DISCONNECT | UV_PRIORITIZED : 0, &on_socket_polled);
1654 pollerContext.pollerState = shouldBeRunning ? PollerContext::PollerState::Connected : PollerContext::PollerState::Suspended;
1655 }
1656 }
1657 }
1658 }
1659 pollOrder.erase(newEnd, pollOrder.end());
1660 }
1661 O2_SIGNPOST_END(device, dpid, "do_prepare", "%zu channels pass the channel inbalance balance check.", pollOrder.size());
1662
1663 for (auto sci : pollOrder) {
1664 auto& info = state.inputChannelInfos[sci];
1665 auto& channelSpec = spec.inputChannels[sci];
1666 O2_SIGNPOST_ID_FROM_POINTER(cid, device, &info);
1667 O2_SIGNPOST_START(device, cid, "channels", "Processing channel %s", channelSpec.name.c_str());
1668
1669 if (info.state != InputChannelState::Completed && info.state != InputChannelState::Pull) {
1670 context.allDone = false;
1671 }
1672 if (info.state != InputChannelState::Running) {
1673 // Remember to flush data if we are not running
1674 // and there is some message pending.
1675 if (info.parts.Size()) {
1676 DataProcessingDevice::handleData(ref, info);
1677 }
1678 O2_SIGNPOST_END(device, cid, "channels", "Flushing channel %s which is in state %d and has %zu parts still pending.",
1679 channelSpec.name.c_str(), (int)info.state, info.parts.Size());
1680 continue;
1681 }
1682 if (info.channel == nullptr) {
1683 O2_SIGNPOST_END(device, cid, "channels", "Channel %s which is in state %d is nullptr and has %zu parts still pending.",
1684 channelSpec.name.c_str(), (int)info.state, info.parts.Size());
1685 continue;
1686 }
1687 // Only poll DPL channels for now.
1688 if (info.channelType != ChannelAccountingType::DPL) {
1689 O2_SIGNPOST_END(device, cid, "channels", "Channel %s which is in state %d is not a DPL channel and has %zu parts still pending.",
1690 channelSpec.name.c_str(), (int)info.state, info.parts.Size());
1691 continue;
1692 }
1693 auto& socket = info.channel->GetSocket();
1694 // If we have pending events from a previous iteration,
1695 // we do receive in any case.
1696 // Otherwise we check if there is any pending event and skip
1697 // this channel in case there is none.
1698 if (info.hasPendingEvents == 0) {
1699 socket.Events(&info.hasPendingEvents);
1700 // If we do not read, we can continue.
1701 if ((info.hasPendingEvents & 1) == 0 && (info.parts.Size() == 0)) {
1702 O2_SIGNPOST_END(device, cid, "channels", "No pending events and no remaining parts to process for channel %{public}s", channelSpec.name.c_str());
1703 continue;
1704 }
1705 }
1706 // We can reset this, because it means we have seen at least 1
1707 // message after the UV_READABLE was raised.
1708 info.readPolled = false;
1709 // Notice that there seems to be a difference between the documentation
1710 // of zeromq and the observed behavior. The fact that ZMQ_POLLIN
1711 // is raised does not mean that a message is immediately available to
1712 // read, just that it will be available soon, so the receive can
1713 // still return -2. To avoid this we keep receiving on the socket until
1714 // we get a message. In order not to overflow the DPL queue we process
1715 // one message at the time and we keep track of wether there were more
1716 // to process.
1717 bool newMessages = false;
1718 while (true) {
1719 O2_SIGNPOST_EVENT_EMIT(device, cid, "channels", "Receiving loop called for channel %{public}s (%d) with oldest possible timeslice %zu",
1720 channelSpec.name.c_str(), info.id.value, info.oldestForChannel.value);
1721 if (info.parts.Size() < 64) {
1722 fair::mq::Parts parts;
1723 info.channel->Receive(parts, 0);
1724 if (parts.Size()) {
1725 O2_SIGNPOST_EVENT_EMIT(device, cid, "channels", "Received %zu parts from channel %{public}s (%d).", parts.Size(), channelSpec.name.c_str(), info.id.value);
1726 }
1727 for (auto&& part : parts) {
1728 info.parts.fParts.emplace_back(std::move(part));
1729 }
1730 newMessages |= true;
1731 }
1732
1733 if (info.parts.Size() >= 0) {
1734 DataProcessingDevice::handleData(ref, info);
1735 // Receiving data counts as activity now, so that
1736 // We can make sure we process all the pending
1737 // messages without hanging on the uv_run.
1738 break;
1739 }
1740 }
1741 // We check once again for pending events, keeping track if this was the
1742 // case so that we can immediately repeat this loop and avoid remaining
1743 // stuck in uv_run. This is because we will not get notified on the socket
1744 // if more events are pending due to zeromq level triggered approach.
1745 socket.Events(&info.hasPendingEvents);
1746 if (info.hasPendingEvents) {
1747 info.readPolled = false;
1748 // In case there were messages, we consider it as activity
1749 if (newMessages) {
1750 state.lastActiveDataProcessor.store(&context);
1751 }
1752 }
1753 O2_SIGNPOST_END(device, cid, "channels", "Done processing channel %{public}s (%d).",
1754 channelSpec.name.c_str(), info.id.value);
1755 }
1756}

1758void DataProcessingDevice::doRun(ServiceRegistryRef ref)
1759{
1760 auto& context = ref.get<DataProcessorContext>();
1761 O2_SIGNPOST_ID_FROM_POINTER(dpid, device, &context);
1762 auto& state = ref.get<DeviceState>();
1763 auto& spec = ref.get<DeviceSpec const>();
1764
1765 if (state.streaming == StreamingState::Idle) {
1766 return;
1767 }
1768
1769 context.completed.clear();
1770 context.completed.reserve(16);
1771 if (DataProcessingDevice::tryDispatchComputation(ref, context.completed)) {
1772 state.lastActiveDataProcessor.store(&context);
1773 }
1774 DanglingContext danglingContext{*context.registry};
1775
1776 context.preDanglingCallbacks(danglingContext);
1777 if (state.lastActiveDataProcessor.load() == nullptr) {
1778 ref.get<CallbackService>().call<CallbackService::Id::Idle>();
1779 }
1780 auto activity = ref.get<DataRelayer>().processDanglingInputs(context.expirationHandlers, *context.registry, true);
1781 if (activity.expiredSlots > 0) {
1782 state.lastActiveDataProcessor = &context;
1783 }
1784
1785 context.completed.clear();
1786 if (DataProcessingDevice::tryDispatchComputation(ref, context.completed)) {
1787 state.lastActiveDataProcessor = &context;
1788 }
1789
1790 context.postDanglingCallbacks(danglingContext);
1791
1792 // If we got notified that all the sources are done, we call the EndOfStream
1793 // callback and return false. Notice that what happens next is actually
1794 // dependent on the callback, not something which is controlled by the
1795 // framework itself.
1796 if (context.allDone == true && state.streaming == StreamingState::Streaming) {
1797 switchState(ref, StreamingState::EndOfStreaming);
1798 state.lastActiveDataProcessor = &context;
1799 }
1800
1801 if (state.streaming == StreamingState::EndOfStreaming) {
1802 O2_SIGNPOST_EVENT_EMIT(device, dpid, "state", "We are in EndOfStreaming. Flushing queues.");
1803 // We keep processing data until we are Idle.
1804 // FIXME: not sure this is the correct way to drain the queues, but
1805 // I guess we will see.
1808 auto& relayer = ref.get<DataRelayer>();
1809
1810 bool shouldProcess = hasOnlyGenerated(spec) == false;
1811
1812 while (DataProcessingDevice::tryDispatchComputation(ref, context.completed) && shouldProcess) {
1813 relayer.processDanglingInputs(context.expirationHandlers, *context.registry, false);
1814 }
1815
1816 auto& timingInfo = ref.get<TimingInfo>();
1817 // We should keep the data generated at end of stream only for those
1818 // which are not sources.
1819 timingInfo.keepAtEndOfStream = shouldProcess;
1820 // Fill timinginfo with some reasonable values for data sent with endOfStream
1821 timingInfo.timeslice = relayer.getOldestPossibleOutput().timeslice.value;
1822 timingInfo.tfCounter = -1;
1823 timingInfo.firstTForbit = -1;
1824 // timingInfo.runNumber = ; // Not sure where to get this if not already set
1825 timingInfo.creation = std::chrono::time_point_cast<std::chrono::milliseconds>(std::chrono::system_clock::now()).time_since_epoch().count();
1826 O2_SIGNPOST_EVENT_EMIT(calibration, dpid, "calibration", "TimingInfo.keepAtEndOfStream %d", timingInfo.keepAtEndOfStream);
1827
1828 EndOfStreamContext eosContext{*context.registry, ref.get<DataAllocator>()};
1829
1830 context.preEOSCallbacks(eosContext);
1831 auto& streamContext = ref.get<StreamContext>();
1832 streamContext.preEOSCallbacks(eosContext);
1833 ref.get<CallbackService>().call<CallbackService::Id::EndOfStream>(eosContext);
1834 streamContext.postEOSCallbacks(eosContext);
1835 context.postEOSCallbacks(eosContext);
1836
1837 for (auto& channel : spec.outputChannels) {
1838 O2_SIGNPOST_EVENT_EMIT(device, dpid, "state", "Sending end of stream to %{public}s.", channel.name.c_str());
1839 DataProcessingHelpers::sendEndOfStream(ref, channel);
1840 }
1841 // This is needed because the transport is deleted before the device.
1842 relayer.clear();
1843 switchState(ref, StreamingState::Idle);
1844 // In case we should process, note the data processor responsible for it
1845 if (shouldProcess) {
1846 state.lastActiveDataProcessor = &context;
1847 }
1848 // On end of stream we shut down all output pollers.
1849 O2_SIGNPOST_EVENT_EMIT(device, dpid, "state", "Shutting down output pollers.");
1850 for (auto& poller : state.activeOutputPollers) {
1851 uv_poll_stop(poller);
1852 }
1853 return;
1854 }
1855
1856 if (state.streaming == StreamingState::Idle) {
1857 // On end of stream we shut down all output pollers.
1858 O2_SIGNPOST_EVENT_EMIT(device, dpid, "state", "Shutting down output pollers.");
1859 for (auto& poller : state.activeOutputPollers) {
1860 uv_poll_stop(poller);
1861 }
1862 }
1863
1864 return;
1865}

1896void DataProcessingDevice::handleData(ServiceRegistryRef ref, InputChannelInfo& info)
1897{
1898 using InputInfo = DataRelayer::InputInfo;
1899 using InputType = DataRelayer::InputType;
1900
1901 auto& context = ref.get<DataProcessorContext>();
1902 // This is the same id as the upper level function, so we get the events
1903 // associated with the same interval. We will simply use "handle_data" as
1904 // the category.
1905 O2_SIGNPOST_ID_FROM_POINTER(cid, device, &info);
1906
1907 // This is how we validate inputs. I.e. we try to enforce the O2 Data model
1908 // and we do a few stats. We bind parts as a lambda captured variable, rather
1909 // than an input, because we do not want the outer loop actually be exposed
1910 // to the implementation details of the messaging layer.
1911 auto getInputTypes = [&info, &context]() -> std::optional<std::vector<InputInfo>> {
1912 O2_SIGNPOST_ID_FROM_POINTER(cid, device, &info);
1913 auto ref = ServiceRegistryRef{*context.registry};
1914 auto& stats = ref.get<DataProcessingStats>();
1915 auto& state = ref.get<DeviceState>();
1916 auto& parts = info.parts;
1917 stats.updateStats({(int)ProcessingStatsId::TOTAL_INPUTS, DataProcessingStats::Op::Set, (int64_t)parts.Size()});
1918
1919 std::vector<InputInfo> results;
1920 // we can reserve the upper limit
1921 results.reserve(parts.Size() / 2);
1922 size_t nTotalPayloads = 0;
1923
1924 auto insertInputInfo = [&results, &nTotalPayloads](size_t position, size_t length, InputType type, ChannelIndex index) {
1925 results.emplace_back(position, length, type, index);
1926 if (type != InputType::Invalid && length > 1) {
1927 nTotalPayloads += length - 1;
1928 }
1929 };
1930
1931 for (size_t pi = 0; pi < parts.Size(); pi += 2) {
1932 auto* headerData = parts.At(pi)->GetData();
1933 auto sih = o2::header::get<SourceInfoHeader*>(headerData);
1934 if (sih) {
1935 O2_SIGNPOST_EVENT_EMIT(device, cid, "handle_data", "Got SourceInfoHeader with state %d", (int)sih->state);
1936 info.state = sih->state;
1937 insertInputInfo(pi, 2, InputType::SourceInfo, info.id);
1938 state.lastActiveDataProcessor = &context;
1939 continue;
1940 }
1941 auto dih = o2::header::get<DomainInfoHeader*>(headerData);
1942 if (dih) {
1943 O2_SIGNPOST_EVENT_EMIT(device, cid, "handle_data", "Got DomainInfoHeader with oldestPossibleTimeslice %d", (int)dih->oldestPossibleTimeslice);
1944 insertInputInfo(pi, 2, InputType::DomainInfo, info.id);
1945 state.lastActiveDataProcessor = &context;
1946 continue;
1947 }
1948 auto dh = o2::header::get<DataHeader*>(headerData);
1949 if (!dh) {
1950 insertInputInfo(pi, 0, InputType::Invalid, info.id);
1951 O2_SIGNPOST_EVENT_EMIT_ERROR(device, cid, "handle_data", "Header is not a DataHeader?");
1952 continue;
1953 }
1954 if (dh->payloadSize > parts.At(pi + 1)->GetSize()) {
1955 insertInputInfo(pi, 0, InputType::Invalid, info.id);
1956 O2_SIGNPOST_EVENT_EMIT_ERROR(device, cid, "handle_data", "DataHeader payloadSize mismatch");
1957 continue;
1958 }
1959 auto dph = o2::header::get<DataProcessingHeader*>(headerData);
1960 // We only deal with the tracking of parts if the log is enabled.
1961 // This is because in principle we should track the size of each of
1962 // the parts and sum it up. Not for now.
1963 O2_SIGNPOST_ID_FROM_POINTER(pid, parts, headerData);
1964 O2_SIGNPOST_START(parts, pid, "parts", "Processing DataHeader %{public}-4s/%{public}-16s/%d with splitPayloadParts %d and splitPayloadIndex %d",
1965 dh->dataOrigin.str, dh->dataDescription.str, dh->subSpecification, dh->splitPayloadParts, dh->splitPayloadIndex);
1966 if (!dph) {
1967 insertInputInfo(pi, 2, InputType::Invalid, info.id);
1968 O2_SIGNPOST_EVENT_EMIT_ERROR(device, cid, "handle_data", "Header stack does not contain DataProcessingHeader");
1969 continue;
1970 }
1971 if (dh->splitPayloadParts > 0 && dh->splitPayloadParts == dh->splitPayloadIndex) {
1972 // this is indicating a sequence of payloads following the header
1973 // FIXME: we will probably also set the DataHeader version
1974 insertInputInfo(pi, dh->splitPayloadParts + 1, InputType::Data, info.id);
1975 pi += dh->splitPayloadParts - 1;
1976 } else {
1977 // We can set the type for the next splitPayloadParts
1978 // because we are guaranteed they are all the same.
1979 // If splitPayloadParts = 0, we assume that means there is only one (header, payload)
1980 // pair.
1981 size_t finalSplitPayloadIndex = pi + (dh->splitPayloadParts > 0 ? dh->splitPayloadParts : 1) * 2;
1982 if (finalSplitPayloadIndex > parts.Size()) {
1983 O2_SIGNPOST_EVENT_EMIT_ERROR(device, cid, "handle_data", "DataHeader::splitPayloadParts invalid");
1984 insertInputInfo(pi, 0, InputType::Invalid, info.id);
1985 continue;
1986 }
1987 insertInputInfo(pi, 2, InputType::Data, info.id);
1988 for (; pi + 2 < finalSplitPayloadIndex; pi += 2) {
1989 insertInputInfo(pi + 2, 2, InputType::Data, info.id);
1990 }
1991 }
1992 }
1993 if (results.size() + nTotalPayloads != parts.Size()) {
1994 O2_SIGNPOST_EVENT_EMIT_ERROR(device, cid, "handle_data", "inconsistent number of inputs extracted. %zu vs parts (%zu)", results.size() + nTotalPayloads, parts.Size());
1995 return std::nullopt;
1996 }
1997 return results;
1998 };
1999
2000 auto reportError = [ref](const char* message) {
2001 auto& stats = ref.get<DataProcessingStats>();
2002 stats.updateStats({(int)ProcessingStatsId::ERROR_COUNT, DataProcessingStats::Op::Add, 1});
2003 };
2004
2005 auto handleValidMessages = [&info, ref, &reportError](std::vector<InputInfo> const& inputInfos) {
2006 auto& relayer = ref.get<DataRelayer>();
2007 auto& state = ref.get<DeviceState>();
2008 static WaitBackpressurePolicy policy;
2009 auto& parts = info.parts;
2010 // We relay execution to make sure we have a complete set of parts
2011 // available.
2012 bool hasBackpressure = false;
2013 size_t minBackpressureTimeslice = -1;
2014 bool hasData = false;
2015 size_t oldestPossibleTimeslice = -1;
2016 static std::vector<int> ordering;
2017 // Same as inputInfos but with iota.
2018 ordering.resize(inputInfos.size());
2019 std::iota(ordering.begin(), ordering.end(), 0);
2020 // stable sort orderings by type and position
2021 std::stable_sort(ordering.begin(), ordering.end(), [&inputInfos](int const& a, int const& b) {
2022 auto const& ai = inputInfos[a];
2023 auto const& bi = inputInfos[b];
2024 if (ai.type != bi.type) {
2025 return ai.type < bi.type;
2026 }
2027 return ai.position < bi.position;
2028 });
2029 for (size_t ii = 0; ii < inputInfos.size(); ++ii) {
2030 auto const& input = inputInfos[ordering[ii]];
2031 switch (input.type) {
2032 case InputType::Data: {
2033 hasData = true;
2034 auto headerIndex = input.position;
2035 auto nMessages = 0;
2036 auto nPayloadsPerHeader = 0;
2037 if (input.size > 2) {
2038 // header and multiple payload sequence
2039 nMessages = input.size;
2040 nPayloadsPerHeader = nMessages - 1;
2041 } else {
2042 // multiple header-payload pairs
2043 auto dh = o2::header::get<DataHeader*>(parts.At(headerIndex)->GetData());
2044 nMessages = dh->splitPayloadParts > 0 ? dh->splitPayloadParts * 2 : 2;
2045 nPayloadsPerHeader = 1;
2046 ii += (nMessages / 2) - 1;
2047 }
2048 auto onDrop = [ref](TimesliceSlot slot, std::vector<MessageSet>& dropped, TimesliceIndex::OldestOutputInfo oldestOutputInfo) {
2049 O2_SIGNPOST_ID_GENERATE(cid, async_queue);
2050 O2_SIGNPOST_EVENT_EMIT(async_queue, cid, "onDrop", "Dropping message from slot %zu. Forwarding as needed. Timeslice %zu",
2051 slot.index, oldestOutputInfo.timeslice.value);
2052 ref.get<AsyncQueue>();
2053 ref.get<DecongestionService>();
2054 ref.get<DataRelayer>();
2055 // Get the current timeslice for the slot.
2056 auto& variables = ref.get<TimesliceIndex>().getVariablesForSlot(slot);
2057 VariableContextHelpers::getTimeslice(variables);
2058 forwardInputs(ref, slot, dropped, oldestOutputInfo, false, true);
2059 };
2060 auto relayed = relayer.relay(parts.At(headerIndex)->GetData(),
2061 &parts.At(headerIndex),
2062 input,
2063 nMessages,
2064 nPayloadsPerHeader,
2065 onDrop);
2066 switch (relayed.type) {
2067 case DataRelayer::RelayChoice::Type::Backpressured:
2068 if (info.normalOpsNotified == true && info.backpressureNotified == false) {
2069 LOGP(alarm, "Backpressure on channel {}. Waiting.", info.channel->GetName());
2070 auto& monitoring = ref.get<o2::monitoring::Monitoring>();
2071 monitoring.send(o2::monitoring::Metric{1, fmt::format("backpressure_{}", info.channel->GetName())});
2072 info.backpressureNotified = true;
2073 info.normalOpsNotified = false;
2074 }
2075 policy.backpressure(info);
2076 hasBackpressure = true;
2077 minBackpressureTimeslice = std::min<size_t>(minBackpressureTimeslice, relayed.timeslice.value);
2078 break;
2079 case DataRelayer::RelayChoice::Type::Dropped:
2080 case DataRelayer::RelayChoice::Type::Invalid:
2081 case DataRelayer::RelayChoice::Type::WillRelay:
2082 if (info.normalOpsNotified == false && info.backpressureNotified == true) {
2083 LOGP(info, "Back to normal on channel {}.", info.channel->GetName());
2084 auto& monitoring = ref.get<o2::monitoring::Monitoring>();
2085 monitoring.send(o2::monitoring::Metric{0, fmt::format("backpressure_{}", info.channel->GetName())});
2086 info.normalOpsNotified = true;
2087 info.backpressureNotified = false;
2088 }
2089 break;
2090 }
2091 } break;
2092 case InputType::SourceInfo: {
2093 LOGP(detail, "Received SourceInfo");
2094 auto& context = ref.get<DataProcessorContext>();
2095 state.lastActiveDataProcessor = &context;
2096 auto headerIndex = input.position;
2097 auto payloadIndex = input.position + 1;
2098 assert(payloadIndex < parts.Size());
2099 // FIXME: the message with the end of stream cannot contain
2100 // split parts.
2101 parts.At(headerIndex).reset(nullptr);
2102 parts.At(payloadIndex).reset(nullptr);
2103 // for (size_t i = 0; i < dh->splitPayloadParts > 0 ? dh->splitPayloadParts * 2 - 1 : 1; ++i) {
2104 // parts.At(headerIndex + 1 + i).reset(nullptr);
2105 // }
2106 // pi += dh->splitPayloadParts > 0 ? dh->splitPayloadParts - 1 : 0;
2107
2108 } break;
2109 case InputType::DomainInfo: {
2112 auto& context = ref.get<DataProcessorContext>();
2113 state.lastActiveDataProcessor = &context;
2114 auto headerIndex = input.position;
2115 auto payloadIndex = input.position + 1;
2116 assert(payloadIndex < parts.Size());
2117 // FIXME: the message with the end of stream cannot contain
2118 // split parts.
2119
2120 auto dih = o2::header::get<DomainInfoHeader*>(parts.At(headerIndex)->GetData());
2121 if (hasBackpressure && dih->oldestPossibleTimeslice >= minBackpressureTimeslice) {
2122 break;
2123 }
2124 oldestPossibleTimeslice = std::min(oldestPossibleTimeslice, dih->oldestPossibleTimeslice);
2125 LOGP(debug, "Got DomainInfoHeader, new oldestPossibleTimeslice {} on channel {}", oldestPossibleTimeslice, info.id.value);
2126 parts.At(headerIndex).reset(nullptr);
2127 parts.At(payloadIndex).reset(nullptr);
2128 } break;
2129 case InputType::Invalid: {
2130 reportError("Invalid part found.");
2131 } break;
2132 }
2133 }
2136 if (oldestPossibleTimeslice != (size_t)-1) {
