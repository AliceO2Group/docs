323 void on_socket_polled(uv_poll_t* poller, int status, int events)
324 {
325  auto* context = (PollerContext*)poller->data;
326  assert(context);
327  O2_SIGNPOST_ID_FROM_POINTER(sid, device, poller);
328  context->state->loopReason |= DeviceState::DATA_SOCKET_POLLED;
329  switch (events) {
330  case UV_READABLE: {
331  O2_SIGNPOST_EVENT_EMIT(device, sid, "socket_state", "Data pending on socket for channel %{public}s", context->name);
332  context->state->loopReason |= DeviceState::DATA_INCOMING;
333  } break;
334  case UV_WRITABLE: {
335  O2_SIGNPOST_END(device, sid, "socket_state", "Socket connected for channel %{public}s", context->name);
336  if (context->read) {
337  O2_SIGNPOST_START(device, sid, "socket_state", "Socket connected for read in context %{public}s", context->name);
338  uv_poll_start(poller, UV_READABLE | UV_DISCONNECT | UV_PRIORITIZED, &on_socket_polled);
339  context->state->loopReason |= DeviceState::DATA_CONNECTED;
340  } else {
341  O2_SIGNPOST_START(device, sid, "socket_state", "Socket connected for write for channel %{public}s", context->name);
342  context->state->loopReason |= DeviceState::DATA_OUTGOING;
343  // If the socket is writable, fairmq will handle the rest, so we can stop polling and
344  // just wait for the disconnect.
345  uv_poll_start(poller, UV_DISCONNECT | UV_PRIORITIZED, &on_socket_polled);
346  }
347  context->pollerState = PollerContext::PollerState::Connected;
348  } break;
349  case UV_DISCONNECT: {
350  O2_SIGNPOST_END(device, sid, "socket_state", "Socket disconnected in context %{public}s", context->name);
351  } break;
352  case UV_PRIORITIZED: {
353  O2_SIGNPOST_EVENT_EMIT(device, sid, "socket_state", "Socket prioritized for context %{public}s", context->name);
354  } break;
355  }
356  // We do nothing, all the logic for now stays in DataProcessingDevice::doRun()
357 }

359 void on_out_of_band_polled(uv_poll_t* poller, int status, int events)
360 {
361  O2_SIGNPOST_ID_FROM_POINTER(sid, device, poller);
362  auto* context = (PollerContext*)poller->data;
363  context->state->loopReason |= DeviceState::OOB_ACTIVITY;
364  if (status < 0) {
365  LOGP(fatal, "Error while polling {}: {}", context->name, status);
366  uv_poll_start(poller, UV_WRITABLE, &on_out_of_band_polled);
367  }
368  switch (events) {
369  case UV_READABLE: {
370  O2_SIGNPOST_EVENT_EMIT(device, sid, "socket_state", "Data pending on socket for channel %{public}s", context->name);
371  context->state->loopReason |= DeviceState::DATA_INCOMING;
372  assert(context->channelInfo);
373  context->channelInfo->readPolled = true;
374  } break;
375  case UV_WRITABLE: {
376  O2_SIGNPOST_END(device, sid, "socket_state", "OOB socket connected for channel %{public}s", context->name);
377  if (context->read) {
378  O2_SIGNPOST_START(device, sid, "socket_state", "OOB socket connected for read in context %{public}s", context->name);
379  uv_poll_start(poller, UV_READABLE | UV_DISCONNECT | UV_PRIORITIZED, &on_out_of_band_polled);
380  } else {
381  O2_SIGNPOST_START(device, sid, "socket_state", "OOB socket connected for write for channel %{public}s", context->name);
382  context->state->loopReason |= DeviceState::DATA_OUTGOING;
383  }
384  } break;
385  case UV_DISCONNECT: {
386  O2_SIGNPOST_END(device, sid, "socket_state", "OOB socket disconnected in context %{public}s", context->name);
387  uv_poll_start(poller, UV_WRITABLE, &on_out_of_band_polled);
388  } break;
389  case UV_PRIORITIZED: {
390  O2_SIGNPOST_EVENT_EMIT(device, sid, "socket_state", "OOB socket prioritized for context %{public}s", context->name);
391  } break;
392  }
393  // We do nothing, all the logic for now stays in DataProcessingDevice::doRun()
394 }

403 void DataProcessingDevice::Init()
404 {
405  auto ref = ServiceRegistryRef{mServiceRegistry};
406  auto& context = ref.get<DataProcessorContext>();
407  auto& spec = getRunningDevice(mRunningDevice, ref);
408 
409  O2_SIGNPOST_ID_FROM_POINTER(cid, device, &context);
410  O2_SIGNPOST_START(device, cid, "Init", "Entering Init callback.");
411  context.statelessProcess = spec.algorithm.onProcess;
412  context.statefulProcess = nullptr;
413  context.error = spec.algorithm.onError;
414  context.initError = spec.algorithm.onInitError;
415 
416  auto configStore = DeviceConfigurationHelpers::getConfiguration(mServiceRegistry, spec.name.c_str(), spec.options);
417  if (configStore == nullptr) {
418  std::vector<std::unique_ptr<ParamRetriever>> retrievers;
419  retrievers.emplace_back(std::make_unique<FairOptionsRetriever>(GetConfig()));
420  configStore = std::make_unique<ConfigParamStore>(spec.options, std::move(retrievers));
421  configStore->preload();
422  configStore->activate();
423  }
424 
425  using boost::property_tree::ptree;
426 
428  for (auto& entry : configStore->store()) {
429  std::stringstream ss;
430  std::string str;
431  if (entry.second.empty() == false) {
432  boost::property_tree::json_parser::write_json(ss, entry.second, false);
433  str = ss.str();
434  str.pop_back(); // remove EoL
435  } else {
436  str = entry.second.get_value<std::string>();
437  }
438  std::string configString = fmt::format("[CONFIG] {}={} 1 {}", entry.first, str, configStore->provenance(entry.first.c_str())).c_str();
439  mServiceRegistry.get<DriverClient>(ServiceRegistry::globalDeviceSalt()).tell(configString.c_str());
440  }
441 
442  mConfigRegistry = std::make_unique<ConfigParamRegistry>(std::move(configStore));
443 
444  // Setup the error handlers for init
445  if (context.initError) {
446  context.initErrorHandling = [&errorCallback = context.initError,
447  &serviceRegistry = mServiceRegistry](RuntimeErrorRef e) {
450  ServiceRegistryRef ref{serviceRegistry, ServiceRegistry::globalDeviceSalt()};
451  auto& context = ref.get<DataProcessorContext>();
452  auto& err = error_from_ref(e);
453  O2_SIGNPOST_ID_FROM_POINTER(cid, device, &context);
454  O2_SIGNPOST_EVENT_EMIT_ERROR(device, cid, "Init", "Exception caught while in Init: %{public}s. Invoking errorCallback.", err.what);
455  BacktraceHelpers::demangled_backtrace_symbols(err.backtrace, err.maxBacktrace, STDERR_FILENO);
456  auto& stats = ref.get<DataProcessingStats>();
457  stats.updateStats({(int)ProcessingStatsId::EXCEPTION_COUNT, DataProcessingStats::Op::Add, 1});
458  InitErrorContext errorContext{ref, e};
459  errorCallback(errorContext);
460  };
461  } else {
462  context.initErrorHandling = [&serviceRegistry = mServiceRegistry](RuntimeErrorRef e) {
463  auto& err = error_from_ref(e);
466  ServiceRegistryRef ref{serviceRegistry, ServiceRegistry::globalDeviceSalt()};
467  auto& context = ref.get<DataProcessorContext>();
468  O2_SIGNPOST_ID_FROM_POINTER(cid, device, &context);
469  O2_SIGNPOST_EVENT_EMIT_ERROR(device, cid, "Init", "Exception caught while in Init: %{public}s. Exiting with 1.", err.what);
470  BacktraceHelpers::demangled_backtrace_symbols(err.backtrace, err.maxBacktrace, STDERR_FILENO);
471  auto& stats = ref.get<DataProcessingStats>();
472  stats.updateStats({(int)ProcessingStatsId::EXCEPTION_COUNT, DataProcessingStats::Op::Add, 1});
473  exit(1);
474  };
475  }
476 
477  context.expirationHandlers.clear();
478  context.init = spec.algorithm.onInit;
479  if (context.init) {
480  static bool noCatch = getenv("O2_NO_CATCHALL_EXCEPTIONS") && strcmp(getenv("O2_NO_CATCHALL_EXCEPTIONS"), "0");
481  InitContext initContext{*mConfigRegistry, mServiceRegistry};
482 
483  if (noCatch) {
484  try {
485  context.statefulProcess = context.init(initContext);
486  } catch (o2::framework::RuntimeErrorRef e) {
487  if (context.initErrorHandling) {
488  (context.initErrorHandling)(e);
489  }
490  }
491  } else {
492  try {
493  context.statefulProcess = context.init(initContext);
494  } catch (std::exception& ex) {
498  auto e = runtime_error(ex.what());
499  (context.initErrorHandling)(e);
500  } catch (o2::framework::RuntimeErrorRef e) {
501  (context.initErrorHandling)(e);
502  }
503  }
504  }
505  auto& state = ref.get<DeviceState>();
506  state.inputChannelInfos.resize(spec.inputChannels.size());
510  int validChannelId = 0;
511  for (size_t ci = 0; ci < spec.inputChannels.size(); ++ci) {
512  auto& name = spec.inputChannels[ci].name;
513  if (name.find(spec.channelPrefix + "from_internal-dpl-clock") == 0) {
514  state.inputChannelInfos[ci].state = InputChannelState::Pull;
515  state.inputChannelInfos[ci].id = {ChannelIndex::INVALID};
516  validChannelId++;
517  } else {
518  state.inputChannelInfos[ci].id = {validChannelId++};
519  }
520  }
521 
522  // Invoke the callback policy for this device.
523  if (spec.callbacksPolicy.policy != nullptr) {
524  InitContext initContext{*mConfigRegistry, mServiceRegistry};
525  spec.callbacksPolicy.policy(mServiceRegistry.get<CallbackService>(ServiceRegistry::globalDeviceSalt()), initContext);
526  }
527 
528  // Services which are stream should be initialised now
529  auto* options = GetConfig();
530  for (size_t si = 0; si < mStreams.size(); ++si) {
531  ServiceRegistry::Salt streamSalt = ServiceRegistry::streamSalt(si + 1, ServiceRegistry::globalDeviceSalt().dataProcessorId);
532  mServiceRegistry.lateBindStreamServices(state, *options, streamSalt);
533  }
534  O2_SIGNPOST_END(device, cid, "Init", "Exiting Init callback.");
535 }

537 void on_signal_callback(uv_signal_t* handle, int signum)
538 {
539  O2_SIGNPOST_ID_FROM_POINTER(sid, device, handle);
540  O2_SIGNPOST_START(device, sid, "signal_state", "Signal %d received.", signum);
541 
542  auto* registry = (ServiceRegistry*)handle->data;
543  if (!registry) {
544  O2_SIGNPOST_END(device, sid, "signal_state", "No registry active. Ignoring signal.");
545  return;
546  }
547  ServiceRegistryRef ref{*registry};
548  auto& state = ref.get<DeviceState>();
549  auto& quotaEvaluator = ref.get<ComputingQuotaEvaluator>();
550  auto& stats = ref.get<DataProcessingStats>();
551  state.loopReason |= DeviceState::SIGNAL_ARRIVED;
552  size_t ri = 0;
553  while (ri != quotaEvaluator.mOffers.size()) {
554  auto& offer = quotaEvaluator.mOffers[ri];
555  // We were already offered some sharedMemory, so we
556  // do not consider the offer.
557  // FIXME: in principle this should account for memory
558  // available and being offered, however we
559  // want to get out of the woods for now.
560  if (offer.valid && offer.sharedMemory != 0) {
561  O2_SIGNPOST_END(device, sid, "signal_state", "Memory already offered.");
562  return;
563  }
564  ri++;
565  }
566  // Find the first empty offer and have 1GB of shared memory there
567  for (auto& offer : quotaEvaluator.mOffers) {
568  if (offer.valid == false) {
569  offer.cpu = 0;
570  offer.memory = 0;
571  offer.sharedMemory = 1000000000;
572  offer.valid = true;
573  offer.user = -1;
574  break;
575  }
576  }
577  stats.updateStats({(int)ProcessingStatsId::TOTAL_SIGUSR1, DataProcessingStats::Op::Add, 1});
578  O2_SIGNPOST_END(device, sid, "signal_state", "Done processing signals.");
579 }

581 static auto toBeForwardedHeader = [](void* header) -> bool {
582  // If is now possible that the record is not complete when
583  // we forward it, because of a custom completion policy.
584  // this means that we need to skip the empty entries in the
585  // record for being forwarded.
586  if (header == nullptr) {
587  return false;
588  }
589  auto sih = o2::header::get<SourceInfoHeader*>(header);
590  if (sih) {
591  return false;
592  }
593 
594  auto dih = o2::header::get<DomainInfoHeader*>(header);
595  if (dih) {
596  return false;
597  }
598 
599  auto dh = o2::header::get<DataHeader*>(header);
600  if (!dh) {
601  return false;
602  }
603  auto dph = o2::header::get<DataProcessingHeader*>(header);
604  if (!dph) {
605  return false;
606  }
607  return true;

610 static auto toBeforwardedMessageSet = [](std::vector<ChannelIndex>& cachedForwardingChoices,
611  FairMQDeviceProxy& proxy,
612  std::unique_ptr<fair::mq::Message>& header,
613  std::unique_ptr<fair::mq::Message>& payload,
614  size_t total,
615  bool consume) {
616  if (header.get() == nullptr) {
617  // Missing an header is not an error anymore.
618  // it simply means that we did not receive the
619  // given input, but we were asked to
620  // consume existing, so we skip it.
621  return false;
622  }
623  if (payload.get() == nullptr && consume == true) {
624  // If the payload is not there, it means we already
625  // processed it with ConsumeExisiting. Therefore we
626  // need to do something only if this is the last consume.
627  header.reset(nullptr);
628  return false;
629  }
630 
631  auto fdph = o2::header::get<DataProcessingHeader*>(header->GetData());
632  if (fdph == nullptr) {
633  LOG(error) << "Data is missing DataProcessingHeader";
634  return false;
635  }
636  auto fdh = o2::header::get<DataHeader*>(header->GetData());
637  if (fdh == nullptr) {
638  LOG(error) << "Data is missing DataHeader";
639  return false;
640  }
641 
642  // We need to find the forward route only for the first
643  // part of a split payload. All the others will use the same.
644  // but always check if we have a sequence of multiple payloads
645  if (fdh->splitPayloadIndex == 0 || fdh->splitPayloadParts <= 1 || total > 1) {
646  proxy.getMatchingForwardChannelIndexes(cachedForwardingChoices, *fdh, fdph->startTime);
647  }
648  return cachedForwardingChoices.empty() == false;
649 };

656 auto decongestionCallbackLate = [](AsyncTask& task, size_t aid) -> void {

684 // This is how we do the forwarding, i.e. we push
685 // the inputs which are shared between this device and others
686 // to the next one in the daisy chain.
687 // FIXME: do it in a smarter way than O(N^2)
688 static auto forwardInputs = [](ServiceRegistryRef registry, TimesliceSlot slot, std::vector<MessageSet>& currentSetOfInputs,
689  TimesliceIndex::OldestOutputInfo oldestTimeslice, bool copy, bool consume = true) {
690  auto& proxy = registry.get<FairMQDeviceProxy>();
691  // we collect all messages per forward in a map and send them together
692  std::vector<fair::mq::Parts> forwardedParts;
693  forwardedParts.resize(proxy.getNumForwards());
694  std::vector<ChannelIndex> cachedForwardingChoices{};
695  O2_SIGNPOST_ID_GENERATE(sid, forwarding);
696  O2_SIGNPOST_START(forwarding, sid, "forwardInputs", "Starting forwarding for slot %zu with oldestTimeslice %zu %{public}s%{public}s%{public}s",
697  slot.index, oldestTimeslice.timeslice.value, copy ? "with copy" : "", copy && consume ? " and " : "", consume ? "with consume" : "");
698 
699  for (size_t ii = 0, ie = currentSetOfInputs.size(); ii < ie; ++ii) {
700  auto& messageSet = currentSetOfInputs[ii];
701  // In case the messageSet is empty, there is nothing to be done.
702  if (messageSet.size() == 0) {
703  continue;
704  }
705  if (!toBeForwardedHeader(messageSet.header(0)->GetData())) {
706  continue;
707  }
708  cachedForwardingChoices.clear();
709 
710  for (size_t pi = 0; pi < currentSetOfInputs[ii].size(); ++pi) {
711  auto& messageSet = currentSetOfInputs[ii];
712  auto& header = messageSet.header(pi);
713  auto& payload = messageSet.payload(pi);
714  auto total = messageSet.getNumberOfPayloads(pi);
715 
716  if (!toBeforwardedMessageSet(cachedForwardingChoices, proxy, header, payload, total, consume)) {
717  continue;
718  }
719 
720  // In case of more than one forward route, we need to copy the message.
721  // This will eventually use the same mamory if running with the same backend.
722  if (cachedForwardingChoices.size() > 1) {
723  copy = true;
724  }
725  auto* dh = o2::header::get<DataHeader*>(header->GetData());
726  auto* dph = o2::header::get<DataProcessingHeader*>(header->GetData());
727 
728  if (copy) {
729  for (auto& cachedForwardingChoice : cachedForwardingChoices) {
730  auto&& newHeader = header->GetTransport()->CreateMessage();
731  O2_SIGNPOST_EVENT_EMIT(forwarding, sid, "forwardInputs", "Forwarding a copy of %{public}s to route %d.",
732  fmt::format("{}/{}/{}@timeslice:{} tfCounter:{}", dh->dataOrigin, dh->dataDescription, dh->subSpecification, dph->startTime, dh->tfCounter).c_str(), cachedForwardingChoice.value);
733  newHeader->Copy(*header);
734  forwardedParts[cachedForwardingChoice.value].AddPart(std::move(newHeader));
735 
736  for (size_t payloadIndex = 0; payloadIndex < messageSet.getNumberOfPayloads(pi); ++payloadIndex) {
737  auto&& newPayload = header->GetTransport()->CreateMessage();
738  newPayload->Copy(*messageSet.payload(pi, payloadIndex));
739  forwardedParts[cachedForwardingChoice.value].AddPart(std::move(newPayload));
740  }
741  }
742  } else {
743  O2_SIGNPOST_EVENT_EMIT(forwarding, sid, "forwardInputs", "Forwarding %{public}s to route %d.",
744  fmt::format("{}/{}/{}@timeslice:{} tfCounter:{}", dh->dataOrigin, dh->dataDescription, dh->subSpecification, dph->startTime, dh->tfCounter).c_str(), cachedForwardingChoices.back().value);
745  forwardedParts[cachedForwardingChoices.back().value].AddPart(std::move(messageSet.header(pi)));
746  for (size_t payloadIndex = 0; payloadIndex < messageSet.getNumberOfPayloads(pi); ++payloadIndex) {
747  forwardedParts[cachedForwardingChoices.back().value].AddPart(std::move(messageSet.payload(pi, payloadIndex)));
748  }
749  }
750  }
751  }
752  O2_SIGNPOST_EVENT_EMIT(forwarding, sid, "forwardInputs", "Forwarding %zu messages", forwardedParts.size());
753  for (int fi = 0; fi < proxy.getNumForwardChannels(); fi++) {
754  if (forwardedParts[fi].Size() == 0) {
755  continue;
756  }
757  ForwardChannelInfo info = proxy.getForwardChannelInfo(ChannelIndex{fi});
758  auto& parts = forwardedParts[fi];
759  if (info.policy == nullptr) {
760  O2_SIGNPOST_EVENT_EMIT_ERROR(forwarding, sid, "forwardInputs", "Forwarding to %{public}s %d has no policy.", info.name.c_str(), fi);
761  continue;
762  }
763  O2_SIGNPOST_EVENT_EMIT(forwarding, sid, "forwardInputs", "Forwarding to %{public}s %d", info.name.c_str(), fi);
764  info.policy->forward(parts, ChannelIndex{fi}, registry);
765  }
766 
767  auto& asyncQueue = registry.get<AsyncQueue>();
768  auto& decongestion = registry.get<DecongestionService>();
769  O2_SIGNPOST_ID_GENERATE(aid, async_queue);
770  O2_SIGNPOST_EVENT_EMIT(async_queue, aid, "forwardInputs", "Queuing forwarding oldestPossible %zu", oldestTimeslice.timeslice.value);
771  AsyncQueueHelpers::post(asyncQueue, AsyncTask{.timeslice = oldestTimeslice.timeslice, .id = decongestion.oldestPossibleTimesliceTask, .debounce = -1, .callback = decongestionCallbackLate}
772  .user<DecongestionContext>({.ref = registry, .oldestTimeslice = oldestTimeslice}));
773  O2_SIGNPOST_END(forwarding, sid, "forwardInputs", "Forwarding done");
774 };

872  state.activeInputPollers.push_back(poller);

879  state.activeOutOfBandPollers.empty() &&

893  }

895  LOGP(detail, "{} is an out of band channel. Not polling for output.", channelName);
896  continue;
897  }
898  // We assume there is always a ZeroMQ socket behind.
899  int zmq_fd = 0;
900  size_t zmq_fd_len = sizeof(zmq_fd);
901  // FIXME: I should probably save those somewhere... ;-)
902  auto* poller = (uv_poll_t*)malloc(sizeof(uv_poll_t));

904  if (zmq_fd == 0) {
905  LOGP(error, "Cannot get file descriptor for channel {}", channelName);
906  continue;
907  }
908  LOG(detail) << "Polling socket for " << channel[0].GetName();
909  // FIXME: leak
910  auto* pCtx = (PollerContext*)malloc(sizeof(PollerContext));
911  pCtx->name = strdup(channelName.c_str());
912  pCtx->loop = state.loop;
913  pCtx->device = this;

920  }
921  }
922  } else {

1627  auto& info = state.inputChannelInfos[sci];

1644  }

1647  channelSpec.name.c_str(), (int)info.state, info.parts.Size());
1648  continue;
1649  }
1650  // Only poll DPL channels for now.
1651  if (info.channelType != ChannelAccountingType::DPL) {
1652  O2_SIGNPOST_END(device, cid, "channels", "Channel %s which is in state %d is not a DPL channel and has %zu parts still pending.",
1653  channelSpec.name.c_str(), (int)info.state, info.parts.Size());
1654  continue;
1655  }

1658  // we do receive in any case.
1659  // Otherwise we check if there is any pending event and skip

1673  // of zeromq and the observed behavior. The fact that ZMQ_POLLIN
1674  // is raised does not mean that a message is immediately available to
1675  // read, just that it will be available soon, so the receive can

1677  // we get a message. In order not to overflow the DPL queue we process

1699  // We can make sure we process all the pending

1701  break;

1703  }

1713  state.lastActiveDataProcessor.store(&context);
1714  }
1715  }

1717  channelSpec.name.c_str(), info.id.value);

1721 void DataProcessingDevice::doRun(ServiceRegistryRef ref)
1722 {
1723  auto& context = ref.get<DataProcessorContext>();
1724  O2_SIGNPOST_ID_FROM_POINTER(dpid, device, &context);
1725  auto switchState = [ref](StreamingState newState) {

1730  O2_SIGNPOST_START(device, dpid, "state", "Starting processing state %d", (int)newState);
1731  state.streaming = newState;
1732  ref.get<ControlService>().notifyStreamingState(state.streaming);
1733  };
1734  auto& state = ref.get<DeviceState>();
1735  auto& spec = ref.get<DeviceSpec const>();
1736 
1737  if (state.streaming == StreamingState::Idle) {
1738  return;
1739  }
1740 
1741  context.completed.clear();
1742  context.completed.reserve(16);
1743  if (DataProcessingDevice::tryDispatchComputation(ref, context.completed)) {
1744  state.lastActiveDataProcessor.store(&context);
1745  }
1746  DanglingContext danglingContext{*context.registry};
1747 
1748  context.preDanglingCallbacks(danglingContext);
1749  if (state.lastActiveDataProcessor.load() == nullptr) {
1750  ref.get<CallbackService>().call<CallbackService::Id::Idle>();
1751  }
1752  auto activity = ref.get<DataRelayer>().processDanglingInputs(context.expirationHandlers, *context.registry, true);
1753  if (activity.expiredSlots > 0) {
1754  state.lastActiveDataProcessor = &context;
1755  }
1756 
1757  context.completed.clear();
1758  if (DataProcessingDevice::tryDispatchComputation(ref, context.completed)) {
1759  state.lastActiveDataProcessor = &context;
1760  }
1761 
1762  context.postDanglingCallbacks(danglingContext);
1763 
1764  // If we got notified that all the sources are done, we call the EndOfStream
1765  // callback and return false. Notice that what happens next is actually
1766  // dependent on the callback, not something which is controlled by the
1767  // framework itself.
1768  if (context.allDone == true && state.streaming == StreamingState::Streaming) {
1769  switchState(StreamingState::EndOfStreaming);
1770  state.lastActiveDataProcessor = &context;
1771  }
1772

1774  O2_SIGNPOST_EVENT_EMIT(device, dpid, "state", "We are in EndOfStreaming. Flushing queues.");

1777  // I guess we will see.
1780  auto& relayer = ref.get<DataRelayer>();
1781 
1782  bool shouldProcess = hasOnlyGenerated(spec) == false;
1783 
1784  while (DataProcessingDevice::tryDispatchComputation(ref, context.completed) && shouldProcess) {
1785  relayer.processDanglingInputs(context.expirationHandlers, *context.registry, false);
1786  }
1787

1789  // We should keep the data generated at end of stream only for those
1790  // which are not sources.
1791  timingInfo.keepAtEndOfStream = shouldProcess;
1792  // Fill timinginfo with some reasonable values for data sent with endOfStream
1793  timingInfo.timeslice = relayer.getOldestPossibleOutput().timeslice.value;
1794  timingInfo.tfCounter = -1;
1795  timingInfo.firstTForbit = -1;
1796  // timingInfo.runNumber = ; // Not sure where to get this if not already set
1797  timingInfo.creation = std::chrono::time_point_cast<std::chrono::milliseconds>(std::chrono::system_clock::now()).time_since_epoch().count();
1798  O2_SIGNPOST_EVENT_EMIT(calibration, dpid, "calibration", "TimingInfo.keepAtEndOfStream %d", timingInfo.keepAtEndOfStream);
1799 
1800  EndOfStreamContext eosContext{*context.registry, ref.get<DataAllocator>()};
1801 
1802  context.preEOSCallbacks(eosContext);
1803  auto& streamContext = ref.get<StreamContext>();
1804  streamContext.preEOSCallbacks(eosContext);
1805  ref.get<CallbackService>().call<CallbackService::Id::EndOfStream>(eosContext);
1806  streamContext.postEOSCallbacks(eosContext);
1807  context.postEOSCallbacks(eosContext);
1808

1810  O2_SIGNPOST_EVENT_EMIT(device, dpid, "state", "Sending end of stream to %{public}s.", channel.name.c_str());
1811  DataProcessingHelpers::sendEndOfStream(ref, channel);
1812  }
1813  // This is needed because the transport is deleted before the device.
1814  relayer.clear();
1815  switchState(StreamingState::Idle);
1816  // In case we should process, note the data processor responsible for it
1817  if (shouldProcess) {
1818  state.lastActiveDataProcessor = &context;
1819  }
1820  // On end of stream we shut down all output pollers.
1821  O2_SIGNPOST_EVENT_EMIT(device, dpid, "state", "Shutting down output pollers.");
1822  for (auto& poller : state.activeOutputPollers) {
1823  uv_poll_stop(poller);
1824  }
1825  return;
1826  }
1827 
1828  if (state.streaming == StreamingState::Idle) {
1829  // On end of stream we shut down all output pollers.
1830  O2_SIGNPOST_EVENT_EMIT(device, dpid, "state", "Shutting down output pollers.");
1831  for (auto& poller : state.activeOutputPollers) {
1832  uv_poll_stop(poller);
1833  }
1834  }
1835 
1836  return;

1868 void DataProcessingDevice::handleData(ServiceRegistryRef ref, InputChannelInfo& info)
1869 {
1870  using InputInfo = DataRelayer::InputInfo;
1871  using InputType = DataRelayer::InputType;
1872 
1873  auto& context = ref.get<DataProcessorContext>();
1874  // This is the same id as the upper level function, so we get the events
1875  // associated with the same interval. We will simply use "handle_data" as
1876  // the category.
1877  O2_SIGNPOST_ID_FROM_POINTER(cid, device, &info);
1878 
1879  // This is how we validate inputs. I.e. we try to enforce the O2 Data model
1880  // and we do a few stats. We bind parts as a lambda captured variable, rather
1881  // than an input, because we do not want the outer loop actually be exposed
1882  // to the implementation details of the messaging layer.
1883  auto getInputTypes = [&info, &context]() -> std::optional<std::vector<InputInfo>> {
1884  O2_SIGNPOST_ID_FROM_POINTER(cid, device, &info);
1885  auto ref = ServiceRegistryRef{*context.registry};
1886  auto& stats = ref.get<DataProcessingStats>();
1887  auto& state = ref.get<DeviceState>();
1888  auto& parts = info.parts;
1889  stats.updateStats({(int)ProcessingStatsId::TOTAL_INPUTS, DataProcessingStats::Op::Set, (int64_t)parts.Size()});
1890 
1891  std::vector<InputInfo> results;
1892  // we can reserve the upper limit
1893  results.reserve(parts.Size() / 2);
1894  size_t nTotalPayloads = 0;
1895 
1896  auto insertInputInfo = [&results, &nTotalPayloads](size_t position, size_t length, InputType type, ChannelIndex index) {
1897  results.emplace_back(position, length, type, index);
1898  if (type != InputType::Invalid && length > 1) {
1899  nTotalPayloads += length - 1;
1900  }
1901  };
1902 
1903  for (size_t pi = 0; pi < parts.Size(); pi += 2) {
1904  auto* headerData = parts.At(pi)->GetData();
1905  auto sih = o2::header::get<SourceInfoHeader*>(headerData);
1906  if (sih) {
1907  O2_SIGNPOST_EVENT_EMIT(device, cid, "handle_data", "Got SourceInfoHeader with state %d", (int)sih->state);
1908  info.state = sih->state;
1909  insertInputInfo(pi, 2, InputType::SourceInfo, info.id);
1910  state.lastActiveDataProcessor = &context;
1911  continue;
1912  }
1913  auto dih = o2::header::get<DomainInfoHeader*>(headerData);
1914  if (dih) {
1915  O2_SIGNPOST_EVENT_EMIT(device, cid, "handle_data", "Got DomainInfoHeader with oldestPossibleTimeslice %d", (int)dih->oldestPossibleTimeslice);
1916  insertInputInfo(pi, 2, InputType::DomainInfo, info.id);
1917  state.lastActiveDataProcessor = &context;
1918  continue;
1919  }
1920  auto dh = o2::header::get<DataHeader*>(headerData);
1921  if (!dh) {
1922  insertInputInfo(pi, 0, InputType::Invalid, info.id);
1923  O2_SIGNPOST_EVENT_EMIT_ERROR(device, cid, "handle_data", "Header is not a DataHeader?");
1924  continue;
1925  }
1926  if (dh->payloadSize > parts.At(pi + 1)->GetSize()) {
1927  insertInputInfo(pi, 0, InputType::Invalid, info.id);
1928  O2_SIGNPOST_EVENT_EMIT_ERROR(device, cid, "handle_data", "DataHeader payloadSize mismatch");
1929  continue;
1930  }
1931  auto dph = o2::header::get<DataProcessingHeader*>(headerData);
1932  // We only deal with the tracking of parts if the log is enabled.
1933  // This is because in principle we should track the size of each of
1934  // the parts and sum it up. Not for now.
1935  O2_SIGNPOST_ID_FROM_POINTER(pid, parts, headerData);
1936  O2_SIGNPOST_START(parts, pid, "parts", "Processing DataHeader %{public}-4s/%{public}-16s/%d with splitPayloadParts %d and splitPayloadIndex %d",
1937  dh->dataOrigin.str, dh->dataDescription.str, dh->subSpecification, dh->splitPayloadParts, dh->splitPayloadIndex);
1938  if (!dph) {
1939  insertInputInfo(pi, 2, InputType::Invalid, info.id);
1940  O2_SIGNPOST_EVENT_EMIT_ERROR(device, cid, "handle_data", "Header stack does not contain DataProcessingHeader");
1941  continue;
1942  }
1943  if (dh->splitPayloadParts > 0 && dh->splitPayloadParts == dh->splitPayloadIndex) {
1944  // this is indicating a sequence of payloads following the header
1945  // FIXME: we will probably also set the DataHeader version
1946  insertInputInfo(pi, dh->splitPayloadParts + 1, InputType::Data, info.id);
1947  pi += dh->splitPayloadParts - 1;
1948  } else {
1949  // We can set the type for the next splitPayloadParts
1950  // because we are guaranteed they are all the same.
1951  // If splitPayloadParts = 0, we assume that means there is only one (header, payload)
1952  // pair.
1953  size_t finalSplitPayloadIndex = pi + (dh->splitPayloadParts > 0 ? dh->splitPayloadParts : 1) * 2;
1954  if (finalSplitPayloadIndex > parts.Size()) {
1955  O2_SIGNPOST_EVENT_EMIT_ERROR(device, cid, "handle_data", "DataHeader::splitPayloadParts invalid");
1956  insertInputInfo(pi, 0, InputType::Invalid, info.id);
1957  continue;
1958  }
1959  insertInputInfo(pi, 2, InputType::Data, info.id);
1960  for (; pi + 2 < finalSplitPayloadIndex; pi += 2) {
1961  insertInputInfo(pi + 2, 2, InputType::Data, info.id);
1962  }
1963  }
1964  }
1965  if (results.size() + nTotalPayloads != parts.Size()) {
1966  O2_SIGNPOST_EVENT_EMIT_ERROR(device, cid, "handle_data", "inconsistent number of inputs extracted. %zu vs parts (%zu)", results.size() + nTotalPayloads, parts.Size());
1967  return std::nullopt;
1968  }
1969  return results;
1970  };
