74 {
75  mTimer.Stop();

78  mVertexer.setValidateWithIR(mValidateWithIR);

87 void PrimaryVertexingSpec::run(ProcessingContext& pc)
88 {
89  double timeCPU0 = mTimer.CpuTime(), timeReal0 = mTimer.RealTime();
90  mTimer.Start(false);
91  std::vector<PVertex> vertices;
92  std::vector<GIndex> vertexTrackIDs;
93  std::vector<V2TRef> v2tRefs;
94  std::vector<o2::MCEventLabel> lblVtx;
95 
96  if (!mSkip) {
97  o2::globaltracking::RecoContainer recoData;
98  recoData.collectData(pc, *mDataRequest.get()); // select tracks of needed type, with minimal cuts, the real selected will be done in the vertexer
99  updateTimeDependentParams(pc); // Make sure this is called after recoData.collectData, which may load some conditions
100 
101  std::vector<TrackWithTimeStamp> tracks;
102  std::vector<o2::MCCompLabel> tracksMCInfo;
103  std::vector<o2d::GlobalTrackID> gids;
104  auto maxTrackTimeError = PVertexerParams::Instance().maxTimeErrorMUS;
105  auto trackMaxX = PVertexerParams::Instance().trackMaxX;
106  auto minIBHits = PVertexerParams::Instance().minIBHits;
107  auto halfROFITS = 0.5 * mITSROFrameLengthMUS + mITSROFBiasMUS;
108  auto hw2ErrITS = 2.f / std::sqrt(12.f) * mITSROFrameLengthMUS; // conversion from half-width to error for ITS
109 
110  auto creator = [maxTrackTimeError, hw2ErrITS, halfROFITS, trackMaxX, minIBHits, &tracks, &gids, &recoData](auto& _tr, GTrackID _origID, float t0, float terr) {
111  if constexpr (isBarrelTrack<decltype(_tr)>()) {
112  if (!_origID.includesDet(DetID::ITS) || _tr.getX() > trackMaxX) {
113  return true; // just in case this selection was not done on RecoContainer filling level
114  }
115  auto itsID = recoData.getITSContributorGID(_origID);
116  if (!itsID.isSourceSet() || o2::math_utils::numberOfBitsSet(recoData.getITSTrack(itsID).getPattern() & 7) < minIBHits) {
117  return true;
118  }
119  if constexpr (isITSTrack<decltype(_tr)>()) {
120  t0 += halfROFITS; // ITS time is supplied in \mus as beginning of ROF
121  terr *= hw2ErrITS; // error is supplied as a half-ROF duration, convert to \mus
122  }
123  // for all other barrel tracks the time is in \mus with gaussian error
124  if (terr < maxTrackTimeError) {
125  tracks.emplace_back(TrackWithTimeStamp{_tr, {t0, terr}});
126  gids.emplace_back(_origID);
127  }
128  }
129  return true;
130  };
131 
132  recoData.createTracksVariadic(creator, mTrackSrc); // create track sample considered for vertexing
133 
134  if (mUseMC) {
135  recoData.fillTrackMCLabels(gids, tracksMCInfo);
136  }
137  mVertexer.setStartIR(recoData.startIR);
138  static std::vector<InteractionCandidate> ft0Data;
139  if (mValidateWithIR) { // select BCs for validation
140  ft0Data.clear();
141  const o2::ft0::InteractionTag& ft0Params = o2::ft0::InteractionTag::Instance();
142  auto ft0all = recoData.getFT0RecPoints();
143  for (const auto& ftRP : ft0all) {
144  if (ft0Params.isSelected(ftRP)) {
145  ft0Data.emplace_back(InteractionCandidate{ftRP.getInteractionRecord(),
146  float(ftRP.getInteractionRecord().differenceInBC(recoData.startIR) * o2::constants::lhc::LHCBunchSpacingMUS),
147  float(ftRP.getTrigger().getAmplA() + ftRP.getTrigger().getAmplC()),
148  GTrackID::FT0});
149  }
150  }
151  }
152  mVertexer.process(tracks, gids, ft0Data, vertices, vertexTrackIDs, v2tRefs, tracksMCInfo, lblVtx);
153 
154  // flag vertices using UPC ITS mode
155  auto itsrofs = recoData.getITSTracksROFRecords();
156  std::vector<bool> itsTrUPC(recoData.getITSTracks().size());
157  for (auto& rof : itsrofs) {
158  if (rof.getFlag(o2::itsmft::ROFRecord::VtxUPCMode)) {
159  for (int i = rof.getFirstEntry(); i < rof.getFirstEntry() + rof.getNEntries(); i++) {
160  itsTrUPC[i] = true;
161  }
162  }
163  }
164  int nv = vertices.size();
165  for (int iv = 0; iv < nv; iv++) {
166  int idMin = v2tRefs[iv].getFirstEntry(), idMax = idMin + v2tRefs[iv].getEntries();
167  int nits = 0, nitsUPC = 0;
168  for (int id = idMin; id < idMax; id++) {
169  auto gid = recoData.getITSContributorGID(vertexTrackIDs[id]);
170  if (gid.getSource() == GIndex::ITS) {
171  nits++;
172  if (itsTrUPC[gid.getIndex()]) {
173  nitsUPC++;
174  }
175  }
176  }
177  if (nitsUPC > nits / 2) {
178  vertices[iv].setFlags(PVertex::UPCMode);
179  }
180  }
181  }
182 
183  pc.outputs().snapshot(Output{"GLO", "PVTX", 0}, vertices);
184  pc.outputs().snapshot(Output{"GLO", "PVTX_CONTIDREFS"}, v2tRefs);
185  pc.outputs().snapshot(Output{"GLO", "PVTX_CONTID", 0}, vertexTrackIDs);
186 
187  if (mUseMC) {
188  pc.outputs().snapshot(Output{"GLO", "PVTX_MCTR", 0}, lblVtx);
189  }
190 
191  mTimer.Stop();
192  LOGP(info, "Found {} PVs, Time CPU/Real:{:.3f}/{:.3f} (DBScan: {:.4f}, Finder:{:.4f}, MADSel:{:.4f}, Rej.Debris:{:.4f}, Reattach:{:.4f}) | {} trials for {} TZ-clusters, max.trials: {}, Slowest TZ-cluster: {} ms of mult {} | NInitial:{}, Rejections: NoFilledBC:{}, NoIntCand:{}, Debris:{}, Quality:{}, ITSOnly:{}",
193  vertices.size(), mTimer.CpuTime() - timeCPU0, mTimer.RealTime() - timeReal0,
194  mVertexer.getTimeDBScan().CpuTime(), mVertexer.getTimeVertexing().CpuTime(), mVertexer.getTimeMADSel().CpuTime(), mVertexer.getTimeDebris().CpuTime(),
195  mVertexer.getTimeReAttach().CpuTime(), mVertexer.getTotTrials(), mVertexer.getNTZClusters(), mVertexer.getMaxTrialsPerCluster(),
196  mVertexer.getLongestClusterTimeMS(), mVertexer.getLongestClusterMult(), mVertexer.getNIniFound(),
197  mVertexer.getNKilledBCValid(), mVertexer.getNKilledIntCand(), mVertexer.getNKilledDebris(), mVertexer.getNKilledQuality(), mVertexer.getNKilledITSOnly());
198 }

207 void PrimaryVertexingSpec::finaliseCCDB(ConcreteDataMatcher& matcher, void* obj)
208 {
209  if (o2::base::GRPGeomHelper::instance().finaliseCCDB(matcher, obj)) {
210  return;
211  }
212  // Note: strictly speaking, for Configurable params we don't need finaliseCCDB check, the singletons are updated at the CCDB fetcher level
213  if (matcher == ConcreteDataMatcher("ITS", "ALPIDEPARAM", 0)) {
214  LOG(info) << "ITS Alpide param updated";
215  const auto& par = o2::itsmft::DPLAlpideParam<o2::detectors::DetID::ITS>::Instance();
216  par.printKeyValues();
217  return;
218  }
219  if (matcher == ConcreteDataMatcher("GLO", "MEANVERTEX", 0)) {
220  LOG(info) << "Imposing new MeanVertex: " << ((const o2::dataformats::MeanVertexObject*)obj)->asString();
221  mVertexer.setMeanVertex((const o2::dataformats::MeanVertexObject*)obj);
222  return;
223  }

226 void PrimaryVertexingSpec::updateTimeDependentParams(ProcessingContext& pc)
227 {
228  o2::base::GRPGeomHelper::instance().checkUpdates(pc);
229  static bool initOnceDone = false;
230  if (!initOnceDone) { // this params need to be queried only once
231  initOnceDone = true;
232  // Note: reading of the ITS AlpideParam needed for ITS timing is done by the RecoContainer
233  auto grp = o2::base::GRPGeomHelper::instance().getGRPECS();
234  const auto& alpParams = o2::itsmft::DPLAlpideParam<o2::detectors::DetID::ITS>::Instance();
235  if (!grp->isDetContinuousReadOut(DetID::ITS)) {
236  mITSROFrameLengthMUS = alpParams.roFrameLengthTrig / 1.e3; // ITS ROFrame duration in \mus
237  } else {
238  mITSROFrameLengthMUS = alpParams.roFrameLengthInBC * o2::constants::lhc::LHCBunchSpacingNS * 1e-3; // ITS ROFrame duration in \mus
239  }
240  mITSROFBiasMUS = alpParams.roFrameBiasInBC * o2::constants::lhc::LHCBunchSpacingNS * 1e-3;
241  if (o2::base::GRPGeomHelper::instance().getGRPECS()->getRunType() != o2::parameters::GRPECSObject::RunType::COSMICS) {
242  mVertexer.setBunchFilling(o2::base::GRPGeomHelper::instance().getGRPLHCIF()->getBunchFilling());
243  }
244  mVertexer.setITSROFrameLength(mITSROFrameLengthMUS);
245  mVertexer.init();
246  if (pc.services().get<const o2::framework::DeviceSpec>().inputTimesliceId == 0) {
247  PVertexerParams::Instance().printKeyValues();
248  }
249  }
250  // we may have other params which need to be queried regularly
251  pc.inputs().get<o2::dataformats::MeanVertexObject*>("meanvtx");
252 }

254 DataProcessorSpec getPrimaryVertexingSpec(GTrackID::mask_t src, bool skip, bool validateWithFT0, bool useMC, bool useGeom)
255 {
256  std::vector<OutputSpec> outputs;
257  auto dataRequest = std::make_shared<DataRequest>();
258 
259  dataRequest->requestTracks(src, useMC);
260  if (validateWithFT0 && src[GTrackID::FT0]) {
261  dataRequest->requestFT0RecPoints(false);
262  }
263 
264  outputs.emplace_back("GLO", "PVTX", 0, Lifetime::Timeframe);
265  outputs.emplace_back("GLO", "PVTX_CONTID", 0, Lifetime::Timeframe);
266  outputs.emplace_back("GLO", "PVTX_CONTIDREFS", 0, Lifetime::Timeframe);
267 
268  if (useMC) {
269  outputs.emplace_back("GLO", "PVTX_MCTR", 0, Lifetime::Timeframe);
270  }
271 
272  auto ggRequest = std::make_shared<o2::base::GRPGeomRequest>(false, // orbitResetTime
273  true, // GRPECS=true
274  true, // GRPLHCIF
275  true, // GRPMagField
276  true, // askMatLUT
277  useGeom ? o2::base::GRPGeomRequest::Aligned : o2::base::GRPGeomRequest::None, // geometry
278  dataRequest->inputs,
279  true);
280  dataRequest->inputs.emplace_back("meanvtx", "GLO", "MEANVERTEX", 0, Lifetime::Condition, ccdbParamSpec("GLO/Calib/MeanVertex", {}, 1));
281 
282  return DataProcessorSpec{
283  "primary-vertexing",
284  dataRequest->inputs,
285  outputs,
286  AlgorithmSpec{adaptFromTask<PrimaryVertexingSpec>(dataRequest, ggRequest, src, skip, validateWithFT0, useMC)},
287  Options{{"pool-dumps-directory", VariantType::String, "", {"Destination directory for the tracks pool dumps"}}}};
288 }
