53// This template magic allow users to customize the behavior of the process
54// by (optionally) implementing a `configure` method which modifies one of the
55// objects in question.
56//
57// For example it can be optionally implemented by the user to specify the
58// channel policies for your setup. Use this if you want to customize the way
59// your devices communicate between themself, e.g. if you want to use REQ/REP
60// in place of PUB/SUB.
61//
62// The advantage of this approach is that we do not need to expose the
63// configurability / configuration object to the user, unless he really wants to
64// modify it. The drawback is that we need to declare the `customize` method
65// before include this file.
66
67// By default we leave the channel policies unchanged. Notice that the default still include
68// a "match all" policy which uses pub / sub
69// FIXME: add a debug statement saying that the default policy was used?
70

72void defaultConfiguration(std::vector<o2::framework::ConfigParamSpec>& globalWorkflowOptions)
73{
74 o2::framework::call_if_defined<struct WorkflowOptions>([&](auto* ptr) {
75 ptr = new std::decay_t<decltype(*ptr)>;
76 o2::framework::homogeneous_apply_refs([&globalWorkflowOptions](auto what) {
77 return o2::framework::ConfigurableHelpers::appendOption(globalWorkflowOptions, what);
78 },
79 *ptr);
80 });
81}

155int mainNoCatch(int argc, char** argv)
156{
157 using namespace o2::framework;
158 using namespace boost::program_options;
159
160 std::vector<o2::framework::ConfigParamSpec> workflowOptions;
161 UserCustomizationsHelper::userDefinedCustomization(workflowOptions, 0);
162 auto requiredWorkflowOptions = WorkflowCustomizationHelpers::requiredWorkflowOptions();
163 workflowOptions.insert(std::end(workflowOptions), std::begin(requiredWorkflowOptions), std::end(requiredWorkflowOptions));
164
165 std::vector<CompletionPolicy> completionPolicies;
166 UserCustomizationsHelper::userDefinedCustomization(completionPolicies, 0);
167 auto defaultCompletionPolicies = CompletionPolicy::createDefaultPolicies();
168 completionPolicies.insert(std::end(completionPolicies), std::begin(defaultCompletionPolicies), std::end(defaultCompletionPolicies));
169
170 std::vector<DispatchPolicy> dispatchPolicies;
171 UserCustomizationsHelper::userDefinedCustomization(dispatchPolicies, 0);
172 auto defaultDispatchPolicies = DispatchPolicy::createDefaultPolicies();
173 dispatchPolicies.insert(std::end(dispatchPolicies), std::begin(defaultDispatchPolicies), std::end(defaultDispatchPolicies));
174
175 std::vector<ResourcePolicy> resourcePolicies;
176 UserCustomizationsHelper::userDefinedCustomization(resourcePolicies, 0);
177 auto defaultResourcePolicies = ResourcePolicy::createDefaultPolicies();
178 resourcePolicies.insert(std::end(resourcePolicies), std::begin(defaultResourcePolicies), std::end(defaultResourcePolicies));
179
180 std::vector<CallbacksPolicy> callbacksPolicies;
181 UserCustomizationsHelper::userDefinedCustomization(callbacksPolicies, 0);
182 auto defaultCallbacksPolicies = CallbacksPolicy::createDefaultPolicies();
183 callbacksPolicies.insert(std::end(callbacksPolicies), std::begin(defaultCallbacksPolicies), std::end(defaultCallbacksPolicies));
184
185 std::vector<SendingPolicy> sendingPolicies;
186 UserCustomizationsHelper::userDefinedCustomization(sendingPolicies, 0);
187 auto defaultSendingPolicies = SendingPolicy::createDefaultPolicies();
188 sendingPolicies.insert(std::end(sendingPolicies), std::begin(defaultSendingPolicies), std::end(defaultSendingPolicies));
189
190 std::vector<std::unique_ptr<ParamRetriever>> retrievers;
191 std::unique_ptr<ParamRetriever> retriever{new BoostOptionsRetriever(true, argc, argv)};
192 retrievers.emplace_back(std::move(retriever));
193 auto workflowOptionsStore = std::make_unique<ConfigParamStore>(workflowOptions, std::move(retrievers));
194 workflowOptionsStore->preload();
195 workflowOptionsStore->activate();
196 ConfigParamRegistry workflowOptionsRegistry(std::move(workflowOptionsStore));
197 auto extraOptions = o2::framework::ConfigParamDiscovery::discover(workflowOptionsRegistry, argc, argv);
198 for (auto& extra : extraOptions) {
199 workflowOptions.push_back(extra);
200 }
201
202 ServiceRegistry configRegistry;
203 ConfigContext configContext(workflowOptionsRegistry, ServiceRegistryRef{configRegistry}, argc, argv);
204 o2::framework::WorkflowSpec specs = defineDataProcessing(configContext);
205 overrideCloning(configContext, specs);
206 overridePipeline(configContext, specs);
207 overrideLabels(configContext, specs);
208 for (auto& spec : specs) {
209 UserCustomizationsHelper::userDefinedCustomization(spec.requiredServices, 0);
210 }
211 std::vector<ChannelConfigurationPolicy> channelPolicies;
212 UserCustomizationsHelper::userDefinedCustomization(channelPolicies, 0);
213 auto defaultChannelPolicies = ChannelConfigurationPolicy::createDefaultPolicies(configContext);
214 channelPolicies.insert(std::end(channelPolicies), std::begin(defaultChannelPolicies), std::end(defaultChannelPolicies));
215 return doMain(argc, argv, specs,
216 channelPolicies, completionPolicies, dispatchPolicies,
217 resourcePolicies, callbacksPolicies, sendingPolicies, workflowOptions, extraOptions, configContext);
218}
