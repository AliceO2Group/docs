38 void DeadChannelMapCreator::loadFEEConfigViaRunInfoTS(long timeStamp)
39 {
40  // it seems the object validity for the TPC RunInfo is not correct. For safety add 1min
41  // since it has run validity this should be fine
42  timeStamp += 60000;
43  if (mObjectValidity[CDBType::ConfigRunInfo].isValid(timeStamp)) {
44  return;
45  }
46  const auto meta = mCCDBApi.retrieveHeaders(CDBTypeMap.at(CDBType::ConfigRunInfo), {}, timeStamp);
47  mObjectValidity[CDBType::ConfigRunInfo].startvalidity = std::stol(meta.at("Valid-From"));
48  mObjectValidity[CDBType::ConfigRunInfo].endvalidity = std::stol(meta.at("Valid-Until"));
49  const long tag = std::stol(meta.at("Tag"));
50  LOGP(info, "Loading FEE config for time stamp {}, via RunInfo with Tag {}, RunType {}, runNumber {}, validity: {} - {}",
51  timeStamp, tag, meta.at("RunType"), meta.at("runNumber"), meta.at("Valid-From"), meta.at("Valid-Until"));
52  loadFEEConfig(tag, timeStamp);
53 }

62 void DeadChannelMapCreator::loadFEEConfig(long tag, long createdNotAfter)
63 {
64  std::map<std::string, std::string> mm, meta;
65  const std::string createdNotAfterS = (createdNotAfter < 0) ? "" : std::to_string(createdNotAfter);
66  mFEEConfig.reset(mCCDBApi.retrieveFromTFileAny<o2::tpc::FEEConfig>(CDBTypeMap.at(CDBType::ConfigFEE), mm, tag, &meta, "", createdNotAfterS));
67  if (!mFEEConfig) {
68  LOGP(error, "Could not load {}/{}, createdNotAfter: {}", CDBTypeMap.at(CDBType::ConfigFEE), tag, createdNotAfterS);
69  return;
70  }
71  LOGP(info, "Using FEE config for Tag {}, ETag {}, Last-Modified {}", meta.at("Valid-From"), meta.at("ETag"), meta.at("Last-Modified"));
72  mDeadChannelMapFEE = mFEEConfig->getDeadChannelMap();
73 }

76 void DeadChannelMapCreator::loadIDCPadFlags(long timeStampOrRun)
77 {
78  // TODO: Implement validity interval handling
79  timeStampOrRun = getTimeStamp(timeStampOrRun);
80  if (mObjectValidity[CDBType::CalIDCPadStatusMapA].isValid(timeStampOrRun)) {
81  return;
82  }
83 
84  std::map<std::string, std::string> meta;
85  auto status = mCCDBApi.retrieveFromTFileAny<o2::tpc::CalDet<o2::tpc::PadFlags>>(CDBTypeMap.at(CDBType::CalIDCPadStatusMapA), {}, timeStampOrRun, &meta);
86  mObjectValidity[CDBType::CalIDCPadStatusMapA].startvalidity = std::stol(meta.at("Valid-From"));
87  mObjectValidity[CDBType::CalIDCPadStatusMapA].endvalidity = std::stol(meta.at("Valid-Until"));
88  if (!status) {
89  LOGP(error, "Could not load {}/{}", CDBTypeMap.at(CDBType::CalIDCPadStatusMapA), timeStampOrRun);
90  return;
91  }
92  setDeadChannelMapIDCPadStatus(*status);
93  mPadStatusMap.reset(status);
94 }

97 void DeadChannelMapCreator::setDeadChannelMapIDCPadStatus(const CalDetFlag_t& padStatusMap, PadFlags mask)
98 {
99  mDeadChannelMapIDC = false;
100  const auto& mapper = o2::tpc::Mapper::instance();
101  for (size_t iROC = 0; iROC < mDeadChannelMapIDC.getData().size(); ++iROC) {
102  auto& rocDeadMap = mDeadChannelMapIDC.getCalArray(iROC);
103  o2::tpc::ROC roc(iROC);
104  for (int iRow = 0; iRow < mapper.getNumberOfRowsROC(roc); ++iRow) {
105  for (int iPad = 0; iPad < mapper.getNumberOfPadsInRowROC(roc, iRow); ++iPad) {
106  if (std::underlying_type_t<PadFlags>(padStatusMap.getValue(roc, iRow, iPad)) & std::underlying_type_t<PadFlags>(mask)) {
107  mDeadChannelMapIDC.getCalArray(iROC).setValue(iRow, iPad, true);
108  }
109  }
110  }
111  }
112 }
