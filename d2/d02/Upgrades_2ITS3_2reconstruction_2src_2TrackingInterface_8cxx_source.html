38 void ITS3TrackingInterface::finaliseCCDB(framework::ConcreteDataMatcher& matcher, void* obj)
39 {
40  if (o2::base::GRPGeomHelper::instance().finaliseCCDB(matcher, obj)) {
41  return;
42  }
43  if (matcher == framework::ConcreteDataMatcher("IT3", "CLUSDICT", 0)) {
44  LOG(info) << "cluster dictionary updated";
45  setClusterDictionary((const o2::its3::TopologyDictionary*)obj);
46  return;
47  }
48  if (matcher == framework::ConcreteDataMatcher("ITS", "ALPIDEPARAM", 0)) {
49  LOG(info) << "Alpide param updated";
50  const auto& par = o2::itsmft::DPLAlpideParam<o2::detectors::DetID::ITS>::Instance();
51  par.printKeyValues();
52  return;
53  }
54  if (matcher == framework::ConcreteDataMatcher("GLO", "MEANVERTEX", 0)) {
55  LOGP(info, "Mean vertex acquired");
56  setMeanVertex((const o2::dataformats::MeanVertexObject*)obj);
57  return;
58  }
59  if (matcher == framework::ConcreteDataMatcher("ITS", "GEOMTGEO", 0)) {
60  LOG(info) << "ITS GeometryTGeo loaded from ccdb";
61  o2::its::GeometryTGeo::adopt((o2::its::GeometryTGeo*)obj);
62  return;
63  }
