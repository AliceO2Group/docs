58 void prepare(bool tpcZS, const CfFragment& fragmentMax)
59 {
60 abandonTimeframe = false;
61 nPagesTotal = nPagesFragmentMax = 0;
62 for (uint32_t i = 0; i < GPUCA_NSLICES; i++) {
63 nPagesSector[i] = 0;
64 nDigitsEndpointMax[i] = 0;
65 }
66
67 if (tpcZS) {
68 tpcMaxTimeBin = 0;
69 nFragments = fragmentMax.count();
70 if (fragmentData.size() < nFragments) {
71 fragmentData.resize(nFragments);
72 }
73
74 for (uint32_t i = 0; i < nFragments; i++) {
75 for (uint32_t j = 0; j < GPUCA_NSLICES; j++) {
76 for (uint32_t k = 0; k < GPUTrackingInOutZS::NENDPOINTS; k++) {
77 fragmentData[i].nDigits[j][k] = fragmentData[i].nPages[j][k] = 0;
78 fragmentData[i].pageDigits[j][k].clear();
79 }
80 }
81 }
82 }
83 }
