126 inline const int4 TrackerTraits::getBinsRect(const int layerIndex, float phi, float maxdeltaphi,
127  float z1, float z2, float maxdeltaz)
128 {
129  const float zRangeMin = o2::gpu::GPUCommonMath::Min(z1, z2) - maxdeltaz;
130  const float phiRangeMin = phi - maxdeltaphi;
131  const float zRangeMax = o2::gpu::GPUCommonMath::Max(z1, z2) + maxdeltaz;
132  const float phiRangeMax = phi + maxdeltaphi;
133 
134  if (zRangeMax < -mTrkParams[0].LayerZ[layerIndex + 1] ||
135  zRangeMin > mTrkParams[0].LayerZ[layerIndex + 1] || zRangeMin > zRangeMax) {
136 
137  return getEmptyBinsRect();
138  }
139 
140  const IndexTableUtils& utils{mTimeFrame->mIndexTableUtils};
141  return int4{o2::gpu::GPUCommonMath::Max(0, utils.getZBinIndex(layerIndex + 1, zRangeMin)),
142  utils.getPhiBinIndex(math_utils::getNormalizedPhi(phiRangeMin)),
143  o2::gpu::GPUCommonMath::Min(mTrkParams[0].ZBins - 1, utils.getZBinIndex(layerIndex + 1, zRangeMax)), // /!\ trkParams can potentially change across iterations
144  utils.getPhiBinIndex(math_utils::getNormalizedPhi(phiRangeMax))};
145 }
