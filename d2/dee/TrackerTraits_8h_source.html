103template <int nLayers>
104inline const int4 TrackerTraits<nLayers>::getBinsRect(const int layerIndex, float phi, float maxdeltaphi, float z1, float z2, float maxdeltaz) const noexcept
105{
106 const float zRangeMin = o2::gpu::GPUCommonMath::Min(z1, z2) - maxdeltaz;
107 const float phiRangeMin = (maxdeltaphi > constants::math::Pi) ? 0.f : phi - maxdeltaphi;
108 const float zRangeMax = o2::gpu::GPUCommonMath::Max(z1, z2) + maxdeltaz;
109 const float phiRangeMax = (maxdeltaphi > constants::math::Pi) ? constants::math::TwoPi : phi + maxdeltaphi;
110
111 if (zRangeMax < -mTrkParams[0].LayerZ[layerIndex] ||
112 zRangeMin > mTrkParams[0].LayerZ[layerIndex] || zRangeMin > zRangeMax) {
113 return getEmptyBinsRect();
114 }
115
116 const IndexTableUtils& utils{mTimeFrame->mIndexTableUtils};
117 return int4{o2::gpu::GPUCommonMath::Max(0, utils.getZBinIndex(layerIndex, zRangeMin)),
118 utils.getPhiBinIndex(math_utils::getNormalizedPhi(phiRangeMin)),
119 o2::gpu::GPUCommonMath::Min(mTrkParams[0].ZBins - 1, utils.getZBinIndex(layerIndex, zRangeMax)), // /!\ trkParams can potentially change across iterations
120 utils.getPhiBinIndex(math_utils::getNormalizedPhi(phiRangeMax))};
121}
