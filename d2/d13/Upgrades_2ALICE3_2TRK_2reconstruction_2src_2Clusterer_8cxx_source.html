25void Clusterer::process(gsl::span<const Digit> digits,
26 gsl::span<const DigROFRecord> digitROFs,
27 std::vector<o2::trk::Cluster>& clusters,
28 std::vector<unsigned char>& patterns,
29 std::vector<o2::trk::ROFRecord>& clusterROFs,
30 const ConstDigitTruth* digitLabels,
31 ClusterTruth* clusterLabels)
32{
33 if (!mThread) {
34 mThread = std::make_unique<ClustererThread>(this);
35 }
36
37 auto* geom = o2::trk::GeometryTGeo::Instance();
38
39 for (size_t iROF = 0; iROF < digitROFs.size(); ++iROF) {
40 const auto& inROF = digitROFs[iROF];
41 const auto outFirst = static_cast<int>(clusters.size());
42 const int first = inROF.getFirstEntry();
43 const int nEntries = inROF.getNEntries();
44
45 if (nEntries == 0) {
46 clusterROFs.emplace_back(inROF.getBCData(), inROF.getROFrame(), outFirst, 0);
47 continue;
48 }
49
50 // Sort digit indices within this ROF by (chipID, col, row) so we can process
51 // chip by chip, column by column -- the same ordering the ALPIDE scanner expects.
52 mSortIdx.resize(nEntries);
53 std::iota(mSortIdx.begin(), mSortIdx.end(), first);
54 std::sort(mSortIdx.begin(), mSortIdx.end(), [&digits](int a, int b) {
55 const auto& da = digits[a];
56 const auto& db = digits[b];
57 if (da.getChipIndex() != db.getChipIndex()) {
58 return da.getChipIndex() < db.getChipIndex();
59 }
60 if (da.getColumn() != db.getColumn()) {
61 return da.getColumn() < db.getColumn();
62 }
63 return da.getRow() < db.getRow();
64 });
65
66 // Process one chip at a time
67 int sliceStart = 0;
68 while (sliceStart < nEntries) {
69 const int chipFirst = sliceStart;
70 const uint16_t chipID = digits[mSortIdx[sliceStart]].getChipIndex();
71 while (sliceStart < nEntries && digits[mSortIdx[sliceStart]].getChipIndex() == chipID) {
72 ++sliceStart;
73 }
74 const int chipN = sliceStart - chipFirst;
75
76 mThread->processChip(digits, chipFirst, chipN, &clusters, &patterns, digitLabels, clusterLabels, geom);
77 }
78
79 clusterROFs.emplace_back(inROF.getBCData(), inROF.getROFrame(),
80 outFirst, static_cast<int>(clusters.size()) - outFirst);
81 }
82}

84//__________________________________________________
85void Clusterer::ClustererThread::processChip(gsl::span<const Digit> digits,
86 int chipFirst, int chipN,
87 std::vector<Cluster>* clustersOut,
88 std::vector<unsigned char>* patternsOut,
89 const ConstDigitTruth* labelsDigPtr,
90 ClusterTruth* labelsClusPtr,
91 GeometryTGeo* geom)
92{
93 // chipFirst and chipN are relative to mSortIdx (i.e. mSortIdx[chipFirst..chipFirst+chipN-1]
94 // are the global digit indices for this chip, already sorted by col then row).
95 // We use parent->mSortIdx to resolve the global index of each pixel.
96 const auto& sortIdx = parent->mSortIdx;
97
98 if (chipN == 1) {
99 finishChipSingleHitFast(digits, sortIdx[chipFirst], labelsDigPtr, labelsClusPtr, geom);
100 } else {
101 initChip(digits, sortIdx[chipFirst], geom);
102 for (int i = chipFirst + 1; i < chipFirst + chipN; ++i) {
103 updateChip(digits, sortIdx[i]);
104 }
105 finishChip(digits, labelsDigPtr, labelsClusPtr, geom);
106 }
107
108 // Flush per-thread output into the caller's containers
109 if (!clusters.empty()) {
110 clustersOut->insert(clustersOut->end(), clusters.begin(), clusters.end());
111 clusters.clear();
112 }
113 if (!patterns.empty()) {
114 patternsOut->insert(patternsOut->end(), patterns.begin(), patterns.end());
115 patterns.clear();
116 }
117 if (labelsClusPtr && labels.getNElements()) {
118 labelsClusPtr->mergeAtBack(labels);
119 labels.clear();
120 }
121}

124void Clusterer::ClustererThread::initChip(gsl::span<const Digit> digits, uint32_t first, GeometryTGeo* geom)
125{
126 const uint16_t chipID = digits[first].getChipIndex();
127
128 // Determine the number of rows for this chip's sensor type
129 size = constants::moduleMLOT::chip::nRows + 2; // default for ML/OT
130 if (geom) {
131 if (geom->getSubDetID(chipID) == 0) { // VD
132 const int layer = geom->getLayer(chipID);
133 size = constants::VD::petal::layer::nRows[layer] + 2;
134 }
135 }
136
137 delete[] column1;
138 delete[] column2;
139 column1 = new int[size];
140 column2 = new int[size];
141 column1[0] = column1[size - 1] = -1;
142 column2[0] = column2[size - 1] = -1;
143 prev = column1 + 1;
144 curr = column2 + 1;
145 resetColumn(curr);
146
147 pixels.clear();
148 preClusterHeads.clear();
149 preClusterIndices.clear();
150
151 const auto& pix = digits[first];
152 currCol = pix.getColumn();
153 curr[pix.getRow()] = 0;
154 preClusterHeads.push_back(0);
155 preClusterIndices.push_back(0);
156 pixels.emplace_back(-1, first);
157 noLeftCol = true;
158}

161void Clusterer::ClustererThread::updateChip(gsl::span<const Digit> digits, uint32_t ip)
162{
163 const auto& pix = digits[ip];
164 uint16_t row = pix.getRow();
165
166 if (currCol != pix.getColumn()) {
167 swapColumnBuffers();
168 resetColumn(curr);
169 noLeftCol = false;
170 if (pix.getColumn() > currCol + 1) {
171 // gap: no connection with previous column
172 currCol = pix.getColumn();
173 addNewPreCluster(ip, row);
174 noLeftCol = true;
175 return;
176 }
177 currCol = pix.getColumn();
178 }
179
180 bool orphan = true;
181
182 if (noLeftCol) {
183 if (curr[row - 1] >= 0) {
184 expandPreCluster(ip, row, curr[row - 1]);
185 return;
186 }
187 } else {
188#ifdef _ALLOW_DIAGONAL_TRK_CLUSTERS_
189 int neighbours[]{curr[row - 1], prev[row], prev[row + 1], prev[row - 1]};
190#else
191 int neighbours[]{curr[row - 1], prev[row]};
192#endif
193 for (auto pci : neighbours) {
194 if (pci < 0) {
195 continue;
196 }
197 if (orphan) {
198 expandPreCluster(ip, row, pci);
199 orphan = false;
200 continue;
201 }
202 // merge two pre-clusters: assign the smaller index to both
203 if (preClusterIndices[pci] < preClusterIndices[curr[row]]) {
204 preClusterIndices[curr[row]] = preClusterIndices[pci];
205 } else {
206 preClusterIndices[pci] = preClusterIndices[curr[row]];
207 }
208 }
209 }
210 if (orphan) {
211 addNewPreCluster(ip, row);
212 }
213}

216void Clusterer::ClustererThread::finishChip(gsl::span<const Digit> digits,
217 const ConstDigitTruth* labelsDigPtr,
218 ClusterTruth* labelsClusPtr,
219 GeometryTGeo* geom)
220{
221 const uint16_t chipID = digits[pixels[0].second].getChipIndex();
222
223 for (size_t i1 = 0; i1 < preClusterHeads.size(); ++i1) {
224 auto ci = preClusterIndices[i1];
225 if (ci < 0) {
226 continue;
227 }
228 BBox bbox(chipID);
229 int nlab = 0;
230 uint32_t totalCharge = 0;
231 pixArrBuff.clear();
232
233 // Walk the linked list for this pre-cluster head
234 auto collectPixels = [&](int head) {
235 int next = head;
236 while (next >= 0) {
237 const auto& pixEntry = pixels[next];
238 const auto& d = digits[pixEntry.second];
239 uint16_t r = d.getRow(), c = d.getColumn();
240 pixArrBuff.emplace_back(r, c);
241 bbox.adjust(r, c);
242 totalCharge += d.getCharge();
243 if (labelsClusPtr) {
244 fetchMCLabels(pixEntry.second, labelsDigPtr, nlab);
245 }
246 next = pixEntry.first;
247 }
248 };
249
250 collectPixels(preClusterHeads[i1]);
251 preClusterIndices[i1] = -1;
252
253 for (size_t i2 = i1 + 1; i2 < preClusterHeads.size(); ++i2) {
254 if (preClusterIndices[i2] != ci) {
255 continue;
256 }
257 collectPixels(preClusterHeads[i2]);
258 preClusterIndices[i2] = -1;
259 }
260
261 // Determine geometry info
262 int subDetID = -1, layer = -1, disk = -1;
263 if (geom) {
264 subDetID = geom->getSubDetID(chipID);
265 layer = geom->getLayer(chipID);
266 disk = geom->getDisk(chipID);
267 }
268
269 const bool doLabels = (labelsClusPtr != nullptr);
270 if (bbox.isAcceptableSize()) {
271 streamCluster(bbox, pixArrBuff, totalCharge, doLabels, nlab, chipID, subDetID, layer, disk);
272 } else {
273 // Huge cluster: split into MaxRowSpan x MaxColSpan tiles (same as ITS3)
274 auto warnLeft = MaxHugeClusWarn - parent->mNHugeClus;
275 if (warnLeft > 0) {
276 LOGP(warn, "Splitting huge TRK cluster: chipID {}, rows {}:{} cols {}:{}{}",
277 chipID, bbox.rowMin, bbox.rowMax, bbox.colMin, bbox.colMax,
278 warnLeft == 1 ? " (further warnings muted)" : "");
279 parent->mNHugeClus++;
280 }
281 BBox bboxT(chipID);
282 bboxT.colMin = bbox.colMin;
283 do {
284 bboxT.rowMin = bbox.rowMin;
285 bboxT.colMax = std::min(bbox.colMax, uint16_t(bboxT.colMin + o2::itsmft::ClusterPattern::MaxColSpan - 1));
286 do {
287 bboxT.rowMax = std::min(bbox.rowMax, uint16_t(bboxT.rowMin + o2::itsmft::ClusterPattern::MaxRowSpan - 1));
288 std::vector<std::pair<uint16_t, uint16_t>> subPix;
289 uint32_t subCharge = 0;
290 for (const auto& [r, c] : pixArrBuff) {
291 if (bboxT.isInside(r, c)) {
292 subPix.emplace_back(r, c);
293 subCharge += 1;
294 }
295 }
296 if (!subPix.empty()) {
297 streamCluster(bboxT, subPix, subCharge, doLabels, nlab, chipID, subDetID, layer, disk);
298 }
299 bboxT.rowMin = bboxT.rowMax + 1;
300 } while (bboxT.rowMin <= bbox.rowMax);
301 bboxT.colMin = bboxT.colMax + 1;
302 } while (bboxT.colMin <= bbox.colMax);
303 }
304 }
305 // flush per-thread output to the caller via processChip
306}

309void Clusterer::ClustererThread::finishChipSingleHitFast(gsl::span<const Digit> digits, uint32_t hit,
310 const ConstDigitTruth* labelsDigPtr,
311 ClusterTruth* labelsClusPtr,
312 GeometryTGeo* geom)
313{
314 const auto& d = digits[hit];
315 const uint16_t chipID = d.getChipIndex();
316 const uint16_t row = d.getRow();
317 const uint16_t col = d.getColumn();
318
319 if (labelsClusPtr) {
320 int nlab = 0;
321 fetchMCLabels(hit, labelsDigPtr, nlab);
322 const auto cnt = static_cast<uint32_t>(clusters.size());
323 for (int i = nlab; i--;) {
324 labels.addElement(cnt, labelsBuff[i]);
325 }
326 }
327
328 // 1×1 pattern: rowSpan=1, colSpan=1, one byte = 0x80
329 patterns.emplace_back(1);
330 patterns.emplace_back(1);
331 patterns.emplace_back(0x80);
332
333 Cluster cluster;
334 cluster.chipID = chipID;
335 cluster.row = row;
336 cluster.col = col;
337 cluster.size = 1;
338 if (geom) {
339 cluster.subDetID = geom->getSubDetID(chipID);
340 cluster.layer = geom->getLayer(chipID);
341 cluster.disk = geom->getDisk(chipID);
342 }
343 clusters.emplace_back(cluster);
344}

347void Clusterer::ClustererThread::streamCluster(const BBox& bbox,
348 const std::vector<std::pair<uint16_t, uint16_t>>& pixbuf,
349 uint32_t totalCharge,
350 bool doLabels, int nlab,
351 uint16_t chipID, int subDetID, int layer, int disk)
352{
353 if (doLabels) {
354 const auto cnt = static_cast<uint32_t>(clusters.size());
355 for (int i = nlab; i--;) {
356 labels.addElement(cnt, labelsBuff[i]); // accumulate in thread-local buffer
357 }
358 }
359
360 const uint16_t rowSpanW = bbox.rowSpan();
361 const uint16_t colSpanW = bbox.colSpan();
362
363 // Encode the pixel pattern bitmap (rowSpan, colSpan, bytes...)
364 std::array<unsigned char, o2::itsmft::ClusterPattern::MaxPatternBytes> patt{};
365 for (const auto& [r, c] : pixbuf) {
366 uint32_t ir = r - bbox.rowMin, ic = c - bbox.colMin;
367 int nbit = ir * colSpanW + ic;
368 patt[nbit >> 3] |= (0x1 << (7 - (nbit % 8)));
369 }
370 patterns.emplace_back(static_cast<unsigned char>(rowSpanW));
371 patterns.emplace_back(static_cast<unsigned char>(colSpanW));
372 int nBytes = (rowSpanW * colSpanW + 7) / 8;
373 patterns.insert(patterns.end(), patt.begin(), patt.begin() + nBytes);
374
375 Cluster cluster;
376 cluster.chipID = chipID;
377 cluster.row = bbox.rowMin;
378 cluster.col = bbox.colMin;
379 cluster.size = static_cast<uint16_t>(pixbuf.size());
380 cluster.subDetID = static_cast<int16_t>(subDetID);
381 cluster.layer = static_cast<int16_t>(layer);
382 cluster.disk = static_cast<int16_t>(disk);
383 clusters.emplace_back(cluster);
384}

387void Clusterer::ClustererThread::fetchMCLabels(uint32_t digID, const ConstDigitTruth* labelsDig, int& nfilled)
388{
389 if (nfilled >= MaxLabels) {
390 return;
391 }
392 if (!labelsDig || digID >= labelsDig->getIndexedSize()) {
393 return;
394 }
395 const auto& lbls = labelsDig->getLabels(digID);
396 for (int i = lbls.size(); i--;) {
397 int ic = nfilled;
398 for (; ic--;) {
399 if (labelsBuff[ic] == lbls[i]) {
400 return; // already present
401 }
402 }
403 labelsBuff[nfilled++] = lbls[i];
404 if (nfilled >= MaxLabels) {
405 break;
406 }
407 }
408}
