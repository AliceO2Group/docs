27std::pair<double, double> parsePoint(std::string ps)
28{
29 int ix = ps.find_first_of(' ');
30
31 auto first = ps.substr(0, ix);
32 auto second = ps.substr(ix + 1, ps.size() - ix - 1);
33 return std::make_pair(std::stod(first), std::stod(second));
34}
