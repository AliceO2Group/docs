40template <TrackletMode Mode, bool EvalRun, int NLayers>
41void trackleterKernelHost(
42 const gsl::span<const Cluster>& clustersNextLayer, // 0 2
43 const gsl::span<const Cluster>& clustersCurrentLayer, // 1 1
44 const gsl::span<uint8_t>& usedClustersNextLayer, // 0 2
45 const int* indexTableNext,
46 const float phiCut,
47 bounded_vector<Tracklet>& tracklets,
48 gsl::span<int> foundTracklets,
49 const IndexTableUtils<NLayers>& utils,
50 const TimeEstBC& timErr,
51 gsl::span<int> rofFoundTrackletsOffsets,
52 const int globalOffsetNextLayer,
53 const int globalOffsetCurrentLayer,
54 const int maxTrackletsPerCluster)
55{
56 const int PhiBins{utils.getNphiBins()};
57 const int ZBins{utils.getNzBins()};
58 // loop on layer1 clusters
59 for (int iCurrentLayerClusterIndex = 0; iCurrentLayerClusterIndex < clustersCurrentLayer.size(); ++iCurrentLayerClusterIndex) {
60 int storedTracklets{0};
61 const Cluster& currentCluster{clustersCurrentLayer[iCurrentLayerClusterIndex]};
62 const int4 selectedBinsRect{o2::its::getBinsRect(currentCluster, (int)Mode + 1, 0.f, 0.f, 100.f, phiCut / 2, utils)};
63 if (selectedBinsRect.x >= 0) {
64 int phiBinsNum{selectedBinsRect.w - selectedBinsRect.y + 1};
65 if (phiBinsNum < 0) {
66 phiBinsNum += PhiBins;
67 }
68 // loop on phi bins next layer
69 for (int iPhiBin{selectedBinsRect.y}, iPhiCount{0}; iPhiCount < phiBinsNum && storedTracklets < maxTrackletsPerCluster; iPhiBin = ++iPhiBin == PhiBins ? 0 : iPhiBin, iPhiCount++) {
70 const int firstBinIndex{utils.getBinIndex(selectedBinsRect.x, iPhiBin)};
71 const int firstRowClusterIndex{indexTableNext[firstBinIndex]};
72 const int maxRowClusterIndex{indexTableNext[firstBinIndex + ZBins]};
73 // loop on clusters next layer
74 for (int iNextLayerClusterIndex{firstRowClusterIndex}; iNextLayerClusterIndex < maxRowClusterIndex && iNextLayerClusterIndex < static_cast<int>(clustersNextLayer.size()) && storedTracklets < maxTrackletsPerCluster; ++iNextLayerClusterIndex) {
75 if (usedClustersNextLayer[iNextLayerClusterIndex]) {
76 continue;
77 }
78 const Cluster& nextCluster{clustersNextLayer[iNextLayerClusterIndex]};
79 if (math_utils::isPhiDifferenceBelow(currentCluster.phi, nextCluster.phi, phiCut)) {
80 if (storedTracklets < maxTrackletsPerCluster) {
81 if constexpr (!EvalRun) {
82 if constexpr (Mode == TrackletMode::Layer0Layer1) {
83 tracklets[rofFoundTrackletsOffsets[iCurrentLayerClusterIndex] + storedTracklets] = Tracklet{globalOffsetNextLayer + iNextLayerClusterIndex, globalOffsetCurrentLayer + iCurrentLayerClusterIndex, nextCluster, currentCluster, timErr};
84 } else {
85 tracklets[rofFoundTrackletsOffsets[iCurrentLayerClusterIndex] + storedTracklets] = Tracklet{globalOffsetCurrentLayer + iCurrentLayerClusterIndex, globalOffsetNextLayer + iNextLayerClusterIndex, currentCluster, nextCluster, timErr};
86 }
87 }
88 ++storedTracklets;
89 }
90 }
91 }
92 }
93 }
94 if constexpr (EvalRun) {
95 foundTracklets[iCurrentLayerClusterIndex] += storedTracklets;
96 } else {
97 rofFoundTrackletsOffsets[iCurrentLayerClusterIndex] += storedTracklets;
98 }
99 }
100}

102void trackletSelectionKernelHost(
103 const Cluster* clusters0, // global layer 0 clusters
104 const Cluster* clusters1, // global layer 1 clusters
105 gsl::span<unsigned char> usedClusters0, // global layer 0 used clusters
106 gsl::span<unsigned char> usedClusters2, // global layer 2 used clusters
107 const gsl::span<const Tracklet>& tracklets01,
108 const gsl::span<const Tracklet>& tracklets12,
109 bounded_vector<uint8_t>& usedTracklets,
110 const gsl::span<int> foundTracklets01,
111 const gsl::span<int> foundTracklets12,
112 bounded_vector<Line>& lines,
113 const gsl::span<const o2::MCCompLabel>& trackletLabels,
114 bounded_vector<o2::MCCompLabel>& linesLabels,
115 const int nLayer1Clusters,
116 const float tanLambdaCut,
117 const float phiCut,
118 const int maxTracklets)
119{
120 int offset01{0}, offset12{0};
121 for (int iCurrentLayerClusterIndex{0}; iCurrentLayerClusterIndex < nLayer1Clusters; ++iCurrentLayerClusterIndex) {
122 int validTracklets{0};
123 const int endTracklet01 = offset01 + foundTracklets01[iCurrentLayerClusterIndex];
124 const int endTracklet12 = offset12 + foundTracklets12[iCurrentLayerClusterIndex];
125 for (int iTracklet12{offset12}; iTracklet12 < endTracklet12 && validTracklets != maxTracklets; ++iTracklet12) {
126 const auto& tracklet12{tracklets12[iTracklet12]};
127 for (int iTracklet01{offset01}; iTracklet01 < endTracklet01 && validTracklets != maxTracklets; ++iTracklet01) {
128 if (usedTracklets[iTracklet01]) {
129 continue;
130 }
131
132 const auto& tracklet01{tracklets01[iTracklet01]};
133 if (!tracklet01.getTimeStamp().isCompatible(tracklet12.getTimeStamp())) {
134 continue;
135 }
136
137 const float deltaTanLambda{o2::gpu::GPUCommonMath::Abs(tracklet01.tanLambda - tracklet12.tanLambda)};
138 if (deltaTanLambda >= tanLambdaCut) {
139 continue;
140 }
141 if (math_utils::isPhiDifferenceBelow(tracklet01.phi, tracklet12.phi, phiCut) && validTracklets != maxTracklets) {
142 usedClusters0[tracklet01.firstClusterIndex] = 1;
143 usedClusters2[tracklet12.secondClusterIndex] = 1;
144 usedTracklets[iTracklet01] = true;
145 lines.emplace_back(tracklet01, clusters0, clusters1);
146 if (!trackletLabels.empty()) {
147 linesLabels.emplace_back(trackletLabels[iTracklet01]);
148 }
149 ++validTracklets;
150 }
151 }
152 }
153 offset01 += foundTracklets01[iCurrentLayerClusterIndex];
154 offset12 += foundTracklets12[iCurrentLayerClusterIndex];
155 }

160void VertexerTraits<NLayers>::updateVertexingParameters(const std::vector<VertexingParameters>& vrtPar)
161{
162 mVrtParams = vrtPar;
163 mIndexTableUtils.setTrackingParameters(vrtPar[0]);
164 for (auto& par : mVrtParams) {
165 par.phiSpan = static_cast<int>(std::ceil(mIndexTableUtils.getNphiBins() * par.phiCut / o2::constants::math::TwoPI));

290template <int NLayers>
291void VertexerTraits<NLayers>::computeTrackletMatching(const int iteration)
292{
293 mTaskArena->execute([&] {
294 tbb::combinable<int> totalLines{0};
295 tbb::parallel_for(
296 tbb::blocked_range<short>(0, (short)mTimeFrame->getNrof(1)),
297 [&](const tbb::blocked_range<short>& Rofs) {
298 for (short pivotRofId = Rofs.begin(); pivotRofId < Rofs.end(); ++pivotRofId) {
299 if (mTimeFrame->getFoundTracklets(pivotRofId, 0).empty() || skipROF(iteration, pivotRofId)) {
300 continue;
301 }
302 mTimeFrame->getLines(pivotRofId).reserve(std::min(mTimeFrame->getFoundTracklets(pivotRofId, 0).size(), mTimeFrame->getNTrackletsCluster(pivotRofId, 0).size() * constants::MaxSelectedTrackletsPerCluster));
303 bounded_vector<uint8_t> usedTracklets(mTimeFrame->getFoundTracklets(pivotRofId, 0).size(), 0, mMemoryPool.get());
304 trackletSelectionKernelHost(
305 mTimeFrame->getClusters()[0].data(),
306 mTimeFrame->getClusters()[1].data(),
307 mTimeFrame->getUsedClusters(0),
308 mTimeFrame->getUsedClusters(2),
309 mTimeFrame->getFoundTracklets(pivotRofId, 0),
310 mTimeFrame->getFoundTracklets(pivotRofId, 1),
311 usedTracklets,
312 mTimeFrame->getNTrackletsCluster(pivotRofId, 0),
313 mTimeFrame->getNTrackletsCluster(pivotRofId, 1),
314 mTimeFrame->getLines(pivotRofId),
315 mTimeFrame->getLabelsFoundTracklets(pivotRofId, 0),
316 mTimeFrame->getLinesLabel(pivotRofId),
317 static_cast<int>(mTimeFrame->getClustersOnLayer(pivotRofId, 1).size()),
318 mVrtParams[iteration].tanLambdaCut,
319 mVrtParams[iteration].phiCut,
320 constants::MaxSelectedTrackletsPerCluster);
321 totalLines.local() += mTimeFrame->getLines(pivotRofId).size();
322 }
323 });
324 mTimeFrame->setNLinesTotal(totalLines.combine(std::plus<int>()));
325 });
326
327 // from here on we do not use tracklets anymore, so let's free them
328 deepVectorClear(mTimeFrame->getTracklets());

331template <int NLayers>
332void VertexerTraits<NLayers>::computeVertices(const int iteration)
333{
334 const int nRofs = mTimeFrame->getNrof(1);
335 std::vector<std::vector<Vertex>> rofVertices(nRofs);
336 std::vector<std::vector<VertexLabel>> rofLabels(nRofs);
337 const float pairCut2 = mVrtParams[iteration].pairCut * mVrtParams[iteration].pairCut;
338 const float duplicateZCut = mVrtParams[iteration].duplicateZCut > 0.f ? mVrtParams[iteration].duplicateZCut : std::max(4.f * mVrtParams[iteration].pairCut, 0.5f * mVrtParams[iteration].clusterCut);
339 const float duplicateDistance2Cut = mVrtParams[iteration].duplicateDistance2Cut > 0.f ? mVrtParams[iteration].duplicateDistance2Cut : std::max(16.f * pairCut2, 0.0625f * mVrtParams[iteration].clusterCut * mVrtParams[iteration].clusterCut);
340 line_vertexer::Settings settings;
341 settings.beamX = mTimeFrame->getBeamX();
342 settings.beamY = mTimeFrame->getBeamY();
343 settings.pairCut = mVrtParams[iteration].pairCut;
344 settings.pairCut2 = pairCut2;
345 settings.clusterCut = mVrtParams[iteration].clusterCut;
346 settings.coarseZWindow = mVrtParams[iteration].coarseZWindow;
347 settings.seedDedupZCut = mVrtParams[iteration].seedDedupZCut;
348 settings.refitDedupZCut = mVrtParams[iteration].refitDedupZCut;
349 settings.duplicateZCut = duplicateZCut;
350 settings.duplicateDistance2Cut = duplicateDistance2Cut;
351 settings.finalSelectionZCut = mVrtParams[iteration].finalSelectionZCut;
352 settings.maxZ = mVrtParams[iteration].maxZPositionAllowed;
353 settings.seedMemberRadiusTime = mVrtParams[iteration].seedMemberRadiusTime;
354 settings.seedMemberRadiusZ = mVrtParams[iteration].seedMemberRadiusZ;
355 settings.memoryPool = mMemoryPool;
356
357 const auto processROF = [&](const int rofId) {
358 if (skipROF(iteration, rofId)) {
359 return;
360 }
361 auto& lines = mTimeFrame->getLines(rofId);
362 auto clusters = line_vertexer::buildClusters(std::span<const Line>{lines.data(), lines.size()}, settings);
363 deepVectorClear(lines); // not needed after
364 auto clusterBeamDistance2 = [&](const ClusterLines& cluster) {
365 return (mTimeFrame->getBeamX() - cluster.getVertex()[0]) * (mTimeFrame->getBeamX() - cluster.getVertex()[0]) +
366 (mTimeFrame->getBeamY() - cluster.getVertex()[1]) * (mTimeFrame->getBeamY() - cluster.getVertex()[1]);
367 };
368 auto clusterBetter = [&](const ClusterLines& lhs, const ClusterLines& rhs) {
369 if (lhs.getSize() != rhs.getSize()) {
370 return lhs.getSize() > rhs.getSize();
371 }
372 if (o2::gpu::GPUCommonMath::Abs(lhs.getAvgDistance2() - rhs.getAvgDistance2()) > constants::Tolerance) {
373 return lhs.getAvgDistance2() < rhs.getAvgDistance2();
374 }
375 const auto lhsBeam = clusterBeamDistance2(lhs);
376 const auto rhsBeam = clusterBeamDistance2(rhs);
377 if (o2::gpu::GPUCommonMath::Abs(lhsBeam - rhsBeam) > constants::Tolerance) {
378 return lhsBeam < rhsBeam;
379 }
380 return lhs.getVertex()[2] < rhs.getVertex()[2];
381 };
382
383 // Cluster deduplication by local non-maximum suppression in time/space
384 std::sort(clusters.begin(), clusters.end(), clusterBetter);
385 float minClusterZ = std::numeric_limits<float>::max();
386 for (const auto& cluster : clusters) {
387 minClusterZ = std::min(minClusterZ, cluster.getVertex()[2]);
388 }
389 bounded_vector<ClusterLines> deduplicated(mMemoryPool.get());
390 deduplicated.reserve(clusters.size());
391 std::unordered_map<int, std::vector<int>> keptByZBin;
392 for (auto& candidate : clusters) {
393 bool duplicate = false;
394 const auto candidateZ = candidate.getVertex()[2];
395 const auto zBin = static_cast<int>(std::floor((candidateZ - minClusterZ) / settings.duplicateZCut));
396 for (int neighborBin = zBin - 1; neighborBin <= zBin + 1 && !duplicate; ++neighborBin) {
397 const auto found = keptByZBin.find(neighborBin);
398 if (found == keptByZBin.end()) {
399 continue;
400 }
401 for (const auto ownerId : found->second) {
402 const auto& owner = deduplicated[ownerId];
403 if (!candidate.getTimeStamp().isCompatible(owner.getTimeStamp())) {
404 continue;
405 }
406 if (o2::gpu::GPUCommonMath::Abs(candidate.getVertex()[2] - owner.getVertex()[2]) >= settings.duplicateZCut) {
407 continue;
408 }
409 const auto dx = candidate.getVertex()[0] - owner.getVertex()[0];
410 const auto dy = candidate.getVertex()[1] - owner.getVertex()[1];
411 const auto dz = candidate.getVertex()[2] - owner.getVertex()[2];
412 const auto distance2 = math_utils::SqSum(dx, dy, dz);
413 if (distance2 < settings.duplicateDistance2Cut) {
414 duplicate = true;
415 break;
416 }
417 }
418 }
419 if (duplicate) {
420 continue;
421 }
422
423 const auto ownerId = static_cast<int>(deduplicated.size());
424 keptByZBin[zBin].push_back(ownerId);
425 deduplicated.push_back(std::move(candidate));
426 }
427 clusters = std::move(deduplicated);
428 int nClusters = static_cast<int>(clusters.size());
429
430 // Vertex filtering with score-based local NMS
431 std::sort(clusters.begin(), clusters.end(), clusterBetter);
432 std::vector<int> candidateIndices;
433 candidateIndices.reserve(nClusters);
434 for (int iCluster{0}; iCluster < nClusters; ++iCluster) {
435 const bool zCompatible = o2::gpu::GPUCommonMath::Abs(clusters[iCluster].getVertex()[2]) < mVrtParams[iteration].maxZPositionAllowed;
436
437 if (zCompatible) {
438 candidateIndices.push_back(iCluster);
439 }
440 }
441
442 if (candidateIndices.empty()) {
443 return;
444 }
445
446 auto countSharedLabels = [](const ClusterLines& lhs, const ClusterLines& rhs) {
447 size_t shared = 0;
448 auto lhsIt = lhs.getLabels().begin();
449 auto rhsIt = rhs.getLabels().begin();
450 while (lhsIt != lhs.getLabels().end() && rhsIt != rhs.getLabels().end()) {
451 if (*lhsIt == *rhsIt) {
452 ++shared;
453 ++lhsIt;
454 ++rhsIt;
455 } else if (*lhsIt < *rhsIt) {
456 ++lhsIt;
457 } else {
458 ++rhsIt;
459 }
460 }
461 return shared;
462 };
463
464 float minCandidateZ = std::numeric_limits<float>::max();
465 for (const auto clusterId : candidateIndices) {
466 minCandidateZ = std::min(minCandidateZ, clusters[clusterId].getVertex()[2]);
467 }
468 std::unordered_map<int, std::vector<int>> selectedByZBin;
469 std::vector<int> selectedIndices;
470 selectedIndices.reserve(candidateIndices.size());
471 for (const auto clusterId : candidateIndices) {
472 const auto& candidate = clusters[clusterId];
473 const auto candidateZ = candidate.getVertex()[2];
474 const auto zBin = static_cast<int>((candidateZ - minCandidateZ) / settings.finalSelectionZCut);
475 bool suppressed = false;
476 for (int neighborBin = zBin - 1; neighborBin <= zBin + 1 && !suppressed; ++neighborBin) {
477 const auto found = selectedByZBin.find(neighborBin);
478 if (found == selectedByZBin.end()) {
479 continue;
480 }
481 for (const auto selectedId : found->second) {
482 const auto& selected = clusters[selectedId];
483 if (!candidate.getTimeStamp().isCompatible(selected.getTimeStamp())) {
484 continue;
485 }
486 const auto zDelta = o2::gpu::GPUCommonMath::Abs(candidateZ - selected.getVertex()[2]);
487 const auto sharedLabels = countSharedLabels(candidate, selected);
488 const auto minSize = std::min(candidate.getSize(), selected.getSize());
489 const bool overlapDuplicate = sharedLabels > 0 && sharedLabels * 4 >= minSize;
490 const bool strongZDuplicate = zDelta < settings.finalSelectionZCut;
491 const bool clearlyBetterMultiplicity = selected.getSize() >= candidate.getSize() + 3;
492 const bool clearlyBetterQuality = selected.getSize() > candidate.getSize() &&
493 selected.getAvgDistance2() + constants::Tolerance < 0.8f * candidate.getAvgDistance2();
494 const bool weakCandidate = clearlyBetterMultiplicity || clearlyBetterQuality;
495 if (overlapDuplicate || (strongZDuplicate && weakCandidate)) {
496 suppressed = true;
497 break;
498 }
499 }
500 }
501 if (suppressed) {
502 continue;
503 }
504 selectedByZBin[zBin].push_back(clusterId);
505 selectedIndices.push_back(clusterId);
506 }
507
508 // sort vertices by their multiplicity to opt. suppress lower mult. debris
509 std::vector<int> sortedIndices(selectedIndices.size());
510 std::iota(sortedIndices.begin(), sortedIndices.end(), 0);
511 std::sort(sortedIndices.begin(), sortedIndices.end(), [&selectedIndices, &clusters](int i, int j) {
512 return clusters[selectedIndices[i]].getSize() > clusters[selectedIndices[j]].getSize();
513 });
514 for (const auto sortedId : sortedIndices) {
515 const auto& cluster = clusters[selectedIndices[sortedId]];
516 const auto beamDistance2 = clusterBeamDistance2(cluster);
517 if (!(beamDistance2 < mVrtParams[iteration].NSigmaCut)) {
518 continue;
519 }
520 if (cluster.getSize() < mVrtParams[iteration].clusterContributorsCut) {
521 continue;
522 }
523 if (!rofVertices[rofId].empty() && cluster.getSize() < mVrtParams[iteration].suppressLowMultDebris) {
524 continue;
525 }
526
527 Vertex vertex{cluster.getVertex().data(),
528 cluster.getRMS2(),
529 (ushort)cluster.getSize(),
530 cluster.getAvgDistance2()};
531 if (iteration) {
532 vertex.setFlags(Vertex::UPCMode);
533 }
534 vertex.setTimeStamp(cluster.getTimeStamp());
535 rofVertices[rofId].push_back(vertex);
536 if (mTimeFrame->hasMCinformation()) {
537 auto& lineLabels = mTimeFrame->getLinesLabel(rofId);
538 bounded_vector<o2::MCCompLabel> labels(mMemoryPool.get());
539 for (auto& index : cluster.getLabels()) {
540 labels.push_back(lineLabels[index]);
541 }
542 const auto mainLabel = computeMain(labels);
543 rofLabels[rofId].push_back(mainLabel);
544 }
545 }
546 };
547
548 if (mTaskArena->max_concurrency() <= 1) {
549 for (int rofId{0}; rofId < nRofs; ++rofId) {
550 processROF(rofId);
551 }
552 } else {
553 mTaskArena->execute([&] {
554 tbb::parallel_for(0, nRofs, [&](const int rofId) {
555 processROF(rofId);
556 });
557 });
558 }
559 // add vertices, these anyways get sorted afterward
560 for (int rofId{0}; rofId < nRofs; ++rofId) {
561 for (auto& vertex : rofVertices[rofId]) {
562 mTimeFrame->addPrimaryVertex(vertex);
563 }
564 if (mTimeFrame->hasMCinformation()) {
565 for (auto& label : rofLabels[rofId]) {
566 mTimeFrame->addPrimaryVertexLabel(label);
567 }
568 }
569 }
570}

572template <int NLayers>
573void VertexerTraits<NLayers>::addTruthSeedingVertices()
574{
575 LOGP(info, "Using truth seeds as vertices; will skip computations");
576 const auto dc = o2::steer::DigitizationContext::loadFromFile("collisioncontext.root");
577 const auto irs = dc->getEventRecords();
578 int64_t roFrameBiasInBC = o2::itsmft::DPLAlpideParam<o2::detectors::DetID::ITS>::Instance().getROFBiasInBC(1);
579 int64_t roFrameLengthInBC = o2::itsmft::DPLAlpideParam<o2::detectors::DetID::ITS>::Instance().getROFLengthInBC(1);
580 o2::steer::MCKinematicsReader mcReader(dc);
581 const int iSrc = 0; // take only events from collision generator
582 auto eveId2colId = dc->getCollisionIndicesForSource(iSrc);
583 for (int iEve{0}; iEve < mcReader.getNEvents(iSrc); ++iEve) {
584 const auto& ir = irs[eveId2colId[iEve]];
585 if (!ir.isDummy()) { // do we need this, is this for diffractive events?
586 const auto& eve = mcReader.getMCEventHeader(iSrc, iEve);
587 auto bc = (ir - raw::HBFUtils::Instance().getFirstSampledTFIR()).toLong() - roFrameBiasInBC;
588 if (bc < 0) { // event happened before TF
589 continue;
590 }
591 Vertex vert;
592 vert.getTimeStamp().setTimeStamp(bc);
593 vert.getTimeStamp().setTimeStampError(roFrameLengthInBC / 2);
594 // set minimum to 1 sometimes for diffractive events there is nothing acceptance
595 vert.setNContributors(std::max(1L, std::ranges::count_if(mcReader.getTracks(iSrc, iEve), [](const auto& trk) {
596 if (!trk.isPrimary() || trk.GetPt() < 0.05 || std::abs(trk.GetEta()) > 1.1) {
597 return false;
598 }
599 const auto* p = o2::O2DatabasePDG::Instance()->GetParticle(trk.GetPdgCode());
600 return (!p) ? false : p->Charge() != 0;
601 })));
602 vert.setXYZ((float)eve.GetX(), (float)eve.GetY(), (float)eve.GetZ());
603 vert.setChi2(1); // not used as constraint
604 constexpr float cov = 25e-4;
605 vert.setSigmaX(cov);
606 vert.setSigmaY(cov);
607 vert.setSigmaZ(cov);
608 mTimeFrame->addPrimaryVertex(vert);
609 o2::MCCompLabel mcLbl(o2::MCCompLabel::maxTrackID(), iEve, iSrc, false);
610 VertexLabel lbl(mcLbl, 1.0);
611 mTimeFrame->addPrimaryVertexLabel(lbl);
612 }
613 mcReader.releaseTracksForSourceAndEvent(iSrc, iEve);
614 }
615 LOGP(info, "Imposed {} pv collisions from mc-truth", mTimeFrame->getPrimaryVertices().size());
616}
