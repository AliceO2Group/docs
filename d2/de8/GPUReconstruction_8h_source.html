450template <class T>
451inline void GPUReconstruction::SetupGPUProcessor(T* proc, bool allocate)
452{
453 static_assert(sizeof(T) > sizeof(GPUProcessor), "Need to setup derived class");
454 if (allocate) {
455 proc->SetMaxData(GetIOPtrs());
456 }
457 if (proc->mGPUProcessorType != GPUProcessor::PROCESSOR_TYPE_DEVICE && proc->mLinkedProcessor) {
458 std::memcpy((void*)proc->mLinkedProcessor, (const void*)proc, sizeof(*proc));
459 proc->mLinkedProcessor->InitGPUProcessor((GPUReconstruction*)this, GPUProcessor::PROCESSOR_TYPE_DEVICE, proc);
460 }
461 if (allocate) {
462 AllocateRegisteredMemory(proc, true);
463 } else {
464 ResetRegisteredMemoryPointers(proc);
465 }
466}
