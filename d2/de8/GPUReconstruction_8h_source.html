398template <class T>
399inline T* GPUReconstruction::AllocateIOMemoryHelper(size_t n, const T*& ptr, std::unique_ptr<T[]>& u)
400{
401 if (n == 0) {
402 u.reset(nullptr);
403 return nullptr;
404 }
405 T* retVal;
406 if (mInputControl.useExternal()) {
407 u.reset(nullptr);
408 mInputControl.checkCurrent();
409 GPUProcessor::computePointerWithAlignment(mInputControl.ptrCurrent, retVal, n);
410 if ((size_t)((char*)mInputControl.ptrCurrent - (char*)mInputControl.ptrBase) > mInputControl.size) {
411 throw std::bad_alloc();
412 }
413 } else {
414 u.reset(new T[n]);
415 retVal = u.get();
416 if (mProcessingSettings.registerStandaloneInputMemory) {
417 if (registerMemoryForGPU(u.get(), n * sizeof(T))) {
418 GPUError("Error registering memory for GPU: %p - %ld bytes\n", (void*)u.get(), (int64_t)(n * sizeof(T)));
419 throw std::bad_alloc();
420 }
421 }
422 }
423 ptr = retVal;
424 return retVal;
425}

434template <class T>
435inline int16_t GPUReconstruction::RegisterMemoryAllocation(T* proc, void* (T::*setPtr)(void*), int32_t type, const char* name, const GPUMemoryReuse& re)
436{
437 if (!(type & (GPUMemoryResource::MEMORY_HOST | GPUMemoryResource::MEMORY_GPU))) {
438 if ((type & GPUMemoryResource::MEMORY_SCRATCH) && !mProcessingSettings.keepDisplayMemory) { // keepAllMemory --> keepDisplayMemory
439 type |= (proc->mGPUProcessorType == GPUProcessor::PROCESSOR_TYPE_CPU ? GPUMemoryResource::MEMORY_HOST : GPUMemoryResource::MEMORY_GPU);
440 } else {
441 type |= GPUMemoryResource::MEMORY_HOST | GPUMemoryResource::MEMORY_GPU;
442 }
443 }
444 if (proc->mGPUProcessorType == GPUProcessor::PROCESSOR_TYPE_CPU) {
445 type &= ~GPUMemoryResource::MEMORY_GPU;
446 }
447 mMemoryResources.emplace_back(proc, static_cast<void* (GPUProcessor::*)(void*)>(setPtr), (GPUMemoryResource::MemoryType)type, name);
448 if (mMemoryResources.size() >= 32768) {
449 throw std::bad_alloc();
450 }
451 uint16_t retVal = mMemoryResources.size() - 1;
452 if (re.type != GPUMemoryReuse::NONE && !mProcessingSettings.disableMemoryReuse) {
453 const auto& it = mMemoryReuse1to1.find(re.id);
454 if (it == mMemoryReuse1to1.end()) {
455 mMemoryReuse1to1[re.id] = {proc, retVal};
456 } else {
457 mMemoryResources[retVal].mReuse = it->second.res[0];
458 it->second.res.emplace_back(retVal);
459 }
460 }
461 return retVal;
462}

490template <class T, class S>
491inline uint32_t GPUReconstruction::DumpData(FILE* fp, const T* const* entries, const S* num, InOutPointerType type)
492{
493 int32_t count = getNIOTypeMultiplicity(type);
494 uint32_t numTotal = 0;
495 for (int32_t i = 0; i < count; i++) {
496 numTotal += num[i];
497 }
498 if (numTotal == 0) {
499 return 0;
500 }
501 fwrite(&type, sizeof(type), 1, fp);
502 for (int32_t i = 0; i < count; i++) {
503 fwrite(&num[i], sizeof(num[i]), 1, fp);
504 if (num[i]) {
505 fwrite(entries[i], sizeof(*entries[i]), num[i], fp);
506 }
507 }
508 if (mProcessingSettings.debugLevel >= 2) {
509 GPUInfo("Dumped %ld %s", (int64_t)numTotal, IOTYPENAMES[type]);
510 }
511 return numTotal;
512}

514template <class T, class S>
515inline size_t GPUReconstruction::ReadData(FILE* fp, const T** entries, S* num, std::unique_ptr<T[]>* mem, InOutPointerType type, T** nonConstPtrs)
516{
517 if (feof(fp)) {
518 return 0;
519 }
520 InOutPointerType inType;
521 size_t r, pos = ftell(fp);
522 r = fread(&inType, sizeof(inType), 1, fp);
523 if (r != 1 || inType != type) {
524 fseek(fp, pos, SEEK_SET);
525 return 0;
526 }
527
528 int32_t count = getNIOTypeMultiplicity(type);
529 size_t numTotal = 0;
530 for (int32_t i = 0; i < count; i++) {
531 r = fread(&num[i], sizeof(num[i]), 1, fp);
532 T* m = AllocateIOMemoryHelper(num[i], entries[i], mem[i]);
533 if (nonConstPtrs) {
534 nonConstPtrs[i] = m;
535 }
536 if (num[i]) {
537 r = fread(m, sizeof(*entries[i]), num[i], fp);
538 }
539 numTotal += num[i];
540 }
541 (void)r;
542 if (mProcessingSettings.debugLevel >= 2) {
543 GPUInfo("Read %ld %s", (int64_t)numTotal, IOTYPENAMES[type]);
544 }
545 return numTotal;
546}
