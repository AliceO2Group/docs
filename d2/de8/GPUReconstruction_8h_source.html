451template <class T>
452inline void GPUReconstruction::SetupGPUProcessor(T* proc, bool allocate)
453{
454 static_assert(sizeof(T) > sizeof(GPUProcessor), "Need to setup derived class");
455 if (allocate) {
456 proc->SetMaxData(GetIOPtrs());
457 }
458 if (proc->mGPUProcessorType != GPUProcessor::PROCESSOR_TYPE_DEVICE && proc->mLinkedProcessor) {
459 std::memcpy((void*)proc->mLinkedProcessor, (const void*)proc, sizeof(*proc));
460 proc->mLinkedProcessor->InitGPUProcessor((GPUReconstruction*)this, GPUProcessor::PROCESSOR_TYPE_DEVICE, proc);
461 }
462 if (allocate) {
463 AllocateRegisteredMemory(proc, true);
464 } else {
465 ResetRegisteredMemoryPointers(proc);
466 }
467}
