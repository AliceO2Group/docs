81 static VertexLabel computeMain(const bounded_vector<o2::MCCompLabel>& elements)
82 {
83 // we only care about the source&event of the tracks, not the trackId
84 auto composeVtxLabel = [](const o2::MCCompLabel& lbl) -> o2::MCCompLabel {
85 return {o2::MCCompLabel::maxTrackID(), lbl.getEventID(), lbl.getSourceID(), lbl.isFake()};
86 };
87 std::unordered_map<o2::MCCompLabel, size_t> frequency;
88 for (const auto& element : elements) {
89 ++frequency[composeVtxLabel(element)];
90 }
91 o2::MCCompLabel elem{};
92 size_t maxCount = 0;
93 for (const auto& [key, count] : frequency) {
94 if (count > maxCount) {
95 maxCount = count;
96 elem = key;
97 }
98 }
99 if (maxCount <= 1) { // need >50%
100 elem.setFakeFlag();
101 }
102 return std::make_pair(elem, static_cast<float>(maxCount) / static_cast<float>(elements.size()));
103 }
