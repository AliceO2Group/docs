60 // Creates the sink for FairLogger / InfoLogger integration
61 auto createInfoLoggerSinkHelper(InfoLogger* logger, InfoLoggerContext* ctx)
62 {
63  return [logger,
64  ctx](const std::string& content, const fair::LogMetaData& metadata) {
65  // translate FMQ metadata
66  InfoLogger::InfoLogger::Severity severity = InfoLogger::Severity::Undefined;
67  int level = InfoLogger::undefinedMessageOption.level;
68 
69  if (metadata.severity_name == fair::Logger::SeverityName(fair::Severity::nolog)) {
70  // discard
71  return;
72  } else if (metadata.severity_name == fair::Logger::SeverityName(fair::Severity::fatal)) {
73  severity = InfoLogger::Severity::Fatal;
74  level = 1;
75  } else if (metadata.severity_name == fair::Logger::SeverityName(fair::Severity::error)) {
76  severity = InfoLogger::Severity::Error;
77  level = 2;
78  } else if (metadata.severity_name == fair::Logger::SeverityName(fair::Severity::alarm)) {
79  severity = InfoLogger::Severity::Warning;
80  level = 6;
81  } else if (metadata.severity_name == fair::Logger::SeverityName(fair::Severity::important)) {
82  severity = InfoLogger::Severity::Info;
83  level = 7;
84  } else if (metadata.severity_name == fair::Logger::SeverityName(fair::Severity::warn)) {
85  severity = InfoLogger::Severity::Warning;
86  level = 11;
87  } else if (metadata.severity_name == fair::Logger::SeverityName(fair::Severity::state)) {
88  severity = InfoLogger::Severity::Info;
89  level = 12;
90  } else if (metadata.severity_name == fair::Logger::SeverityName(fair::Severity::info)) {
91  severity = InfoLogger::Severity::Info;
92  level = 13;
93  } else if (metadata.severity_name == fair::Logger::SeverityName(fair::Severity::debug)) {
94  severity = InfoLogger::Severity::Debug;
95  level = 14;
96  } else if (metadata.severity_name == fair::Logger::SeverityName(fair::Severity::detail)) {
97  severity = InfoLogger::Severity::Debug;
98  level = 15;
99  } else if (metadata.severity_name == fair::Logger::SeverityName(fair::Severity::debug1)) {
100  severity = InfoLogger::Severity::Debug;
101  level = 16;
102  } else if (metadata.severity_name == fair::Logger::SeverityName(fair::Severity::debug2)) {
103  severity = InfoLogger::Severity::Debug;
104  level = 17;
105  } else if (metadata.severity_name == fair::Logger::SeverityName(fair::Severity::debug3)) {
106  severity = InfoLogger::Severity::Debug;
107  level = 18;
108  } else if (metadata.severity_name == fair::Logger::SeverityName(fair::Severity::debug4)) {
109  severity = InfoLogger::Severity::Debug;
110  level = 19;
111  } else if (metadata.severity_name == fair::Logger::SeverityName(fair::Severity::trace)) {
112  severity = InfoLogger::Severity::Debug;
113  level = 21;
114  }
115 
116  InfoLogger::InfoLoggerMessageOption opt = {
117  severity,
118  level,
119  InfoLogger::undefinedMessageOption.errorCode,
120  metadata.file.data(),
121  atoi(metadata.line.data())};
122 
123  if (logger) {
124  logger->log(opt, *ctx, "%s", content.c_str());
125  }
126  };
127 };

129 struct InfoLoggerPlugin : o2::framework::ServicePlugin {
130  o2::framework::ServiceSpec* create() final
131  {
132  return new ServiceSpec{
133  .name = "infologger",
134  .init = [](ServiceRegistryRef services, DeviceState&, fair::mq::ProgOptions& options) -> ServiceHandle {
135  auto infoLoggerMode = options.GetPropertyAsString("infologger-mode");
136  auto infoLoggerSeverity = options.GetPropertyAsString("infologger-severity");
137  if (infoLoggerSeverity.empty() == false && options.GetPropertyAsString("infologger-mode") == "") {
138  LOGP(info, "Using O2_INFOLOGGER_MODE=infoLoggerD since infologger-severity is set");
139  infoLoggerMode = "infoLoggerD";
140  }
141  if (infoLoggerMode != "") {
142  setenv("O2_INFOLOGGER_MODE", infoLoggerMode.c_str(), 1);
143  }
144  char const* infoLoggerEnv = getenv("O2_INFOLOGGER_MODE");
145  if (infoLoggerEnv == nullptr || strcmp(infoLoggerEnv, "none") == 0) {
146  return ServiceHandle{.hash = TypeIdHelpers::uniqueId<MissingService>(),
147  .instance = nullptr,
148  .kind = ServiceKind::Serial,
149  .name = "infologger"};
150  }
151  InfoLogger* infoLoggerService = nullptr;
152  try {
153  infoLoggerService = new InfoLogger;
154  } catch (...) {
155  LOGP(error, "Unable to initialise InfoLogger with O2_INFOLOGGER_MODE={}.", infoLoggerMode);
156  return ServiceHandle{.hash = TypeIdHelpers::uniqueId<MissingService>(),
157  .instance = nullptr,
158  .kind = ServiceKind::Serial,
159  .name = "infologger"};
160  }
161  auto infoLoggerContext = &services.get<InfoLoggerContext>();
162  // Only print the first 10 characters and the last 18 if the
163  // string length is greater than 32 bytes.
164  auto truncate = [](std::string in) -> std::string {
165  if (in.size() < 32) {
166  return in;
167  }
168  char name[32];
169  memcpy(name, in.data(), 10);
170  name[10] = '.';
171  name[11] = '.';
172  name[12] = '.';
173  memcpy(name + 13, in.data() + in.size() - 18, 18);
174  name[31] = 0;
175  return name;
176  };
177  infoLoggerContext->setField(InfoLoggerContext::FieldName::Facility, truncate(services.get<DeviceSpec const>().name));
178  infoLoggerContext->setField(InfoLoggerContext::FieldName::System, std::string("DPL"));
179  infoLoggerService->setContext(*infoLoggerContext);
180 
181  if (infoLoggerSeverity != "") {
182  fair::Logger::AddCustomSink("infologger", infoLoggerSeverity, createInfoLoggerSinkHelper(infoLoggerService, infoLoggerContext));
183  }
184  return ServiceHandle{.hash = TypeIdHelpers::uniqueId<InfoLogger>(),
185  .instance = infoLoggerService,
186  .kind = ServiceKind::Serial,
187  .name = "infologger"};
188  },
189  .configure = CommonServices::noConfiguration(),
190  .kind = ServiceKind::Serial};
191  }
192 };
