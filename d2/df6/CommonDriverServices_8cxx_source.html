23std::vector<ServiceSpec> o2::framework::CommonDriverServices::defaultServices()
24{
25 std::vector<ServiceSpec> specs{

27 // Load plugins depending on the environment
28 std::vector<LoadablePlugin> loadableServices = {};

31 //
32 // library1:name1,library2:name2,...
33 if (loadableServicesEnv) {
34 loadableServices = PluginManager::parsePluginSpecString(loadableServicesEnv);
35 PluginManager::loadFromPlugin<ServiceSpec, ServicePlugin>(loadableServices, specs);
