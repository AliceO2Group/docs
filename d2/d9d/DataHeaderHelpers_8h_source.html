20 // Presentation format: 'f' - fixed, 'e' - exponential.
21 char presentation = 's';
22
23 // Parses format specifications of the form ['f' | 'e'].
24 constexpr auto parse(format_parse_context& ctx)
25 {
26 auto it = ctx.begin(), end = ctx.end();
27 if (it != end && (*it == 's')) {
28 presentation = *it++;
29 }

31 // Check if reached the end of the range:

33 throw format_error("invalid pick format");
34 }
35
36 // Return an iterator past the end of the parsed range:
37 return it;
38 }
