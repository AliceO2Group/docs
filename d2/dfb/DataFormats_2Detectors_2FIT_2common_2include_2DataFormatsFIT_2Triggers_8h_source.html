31 template <typename... TrgBits>
32 constexpr static uint64_t word(TrgBits&&... trgBits)
33 {
34 return ((1ull << std::forward<TrgBits>(trgBits)) | ...);
35 }
36
37 enum { bitA = 0,
38 bitC = 1, // alias of bitAOut (FT0/FDD)
39 bitAOut = 1, // alias of bitC (FV0)
40 bitSCen = 2, // alias of bitTrgNchan (FT0/FDD)
41 bitTrgNchan = 2, // alias of bitSCen (FV0)
42 bitCen = 3, // alias of bitTrgCharge (FT0/FDD)
43 bitTrgCharge = 3, // alias of bitCen (FV0)
44 bitVertex = 4, // alias of bitAIn (FT0/FDD)
45 bitAIn = 4, // alias of bitVertex (FV0)
46 bitLaser = 5, // indicates the laser was triggered in this BC
47 bitOutputsAreBlocked = 6, // indicates that laser-induced pulses should arrive from detector to FEE in this BC (and trigger outputs are blocked)
48 bitDataIsValid = 7, // data is valid for processing
49 bitMinBias = 8 }; // extra calculated bit, vrt & (cern || semicent)
50 static const int16_t DEFAULT_TIME = -5000; // for average of one side (A or C)
51 static const int16_t DEFAULT_AMP = 0;

65 inline static bool checkMinBiasFT0(uint64_t trgWord)
66 {
67 return static_cast<bool>(trgWord & word(bitVertex)) && static_cast<bool>(trgWord & word(bitSCen, bitCen));
68 }
69 static uint64_t makeExtendedTrgWord(uint64_t trgWord)
70 {
71 return trgWord | (static_cast<uint64_t>(checkMinBiasFT0(trgWord)) << bitMinBias);
72 }
73 static constexpr std::pair<uint8_t, uint8_t> parseDigitTriggerWord(uint8_t digitWord, bool shiftTechBitsToBegin = false)
74 {
75 const uint8_t techWordMask = word(bitLaser, bitOutputsAreBlocked, bitDataIsValid);
76 const uint8_t shiftTechWordPos = shiftTechBitsToBegin ? bitLaser : 0;
77 return {(digitWord & (~techWordMask)), (digitWord & techWordMask) >> shiftTechWordPos};
78 }

80 bool getOrA() const { return (triggersignals & (1 << bitA)) != 0; }

122 setTriggers(trgsig, chanA, chanC, aamplA, aamplC, atimeA, atimeC);
123 }
