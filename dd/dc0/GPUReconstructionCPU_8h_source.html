185template <class S, int32_t I, typename... Args>
186inline int32_t GPUReconstructionCPU::runKernel(krnlSetup&& setup, Args&&... args)
187{
188 HighResTimer* t = nullptr;
189 GPUCA_RECO_STEP myStep = S::GetRecoStep() == GPUCA_RECO_STEP::NoRecoStep ? setup.x.step : S::GetRecoStep();
190 if (myStep == GPUCA_RECO_STEP::NoRecoStep) {
191 throw std::runtime_error("Failure running general kernel without defining RecoStep");
192 }
193 int32_t cpuFallback = IsGPU() ? (setup.x.device == krnlDeviceType::CPU ? 2 : (mRecoSteps.stepsGPUMask & myStep) != myStep) : 0;
194 uint32_t& nThreads = setup.x.nThreads;
195 uint32_t& nBlocks = setup.x.nBlocks;
196 const uint32_t stream = setup.x.stream;
197 auto prop = getKernelProperties<S, I>();
198 const int32_t autoThreads = cpuFallback ? 1 : prop.nThreads;
199 const int32_t autoBlocks = cpuFallback ? 1 : (prop.forceBlocks ? prop.forceBlocks : (prop.minBlocks * mBlockCount));
200 if (nBlocks == (uint32_t)-1) {
201 nBlocks = (nThreads + autoThreads - 1) / autoThreads;
202 nThreads = autoThreads;
203 } else if (nBlocks == (uint32_t)-2) {
204 nBlocks = nThreads;
205 nThreads = autoThreads;
206 } else if (nBlocks == (uint32_t)-3) {
207 nBlocks = autoBlocks;
208 nThreads = autoThreads;
209 } else if ((int32_t)nThreads < 0) {
210 nThreads = cpuFallback ? 1 : -nThreads;
211 }
212 if (nThreads > GPUCA_MAX_THREADS) {
213 throw std::runtime_error("GPUCA_MAX_THREADS exceeded");
214 }
215 if (mProcessingSettings.debugLevel >= 3) {
216 GPUInfo("Running kernel %s (Stream %d, Range %d/%d, Grid %d/%d) on %s", GetKernelName<S, I>(), stream, setup.y.start, setup.y.num, nBlocks, nThreads, cpuFallback == 2 ? "CPU (forced)" : cpuFallback ? "CPU (fallback)" : mDeviceName.c_str());
217 }
218 if (nThreads == 0 || nBlocks == 0) {
219 return 0;
220 }
221 if (mProcessingSettings.debugLevel >= 1) {
222 t = &getKernelTimer<S, I>(myStep, !IsGPU() || cpuFallback ? getOMPThreadNum() : stream);
223 if ((!mProcessingSettings.deviceTimers || !IsGPU() || cpuFallback) && (mNestedLoopOmpFactor < 2 || getOMPThreadNum() == 0)) {
224 t->Start();
225 }
226 }
227 double deviceTimerTime = 0.;
228 int32_t retVal = runKernelImplWrapper(gpu_reconstruction_kernels::classArgument<S, I>(), cpuFallback, deviceTimerTime, std::forward<krnlSetup&&>(setup), std::forward<Args>(args)...);
229 if (GPUDebug(GetKernelName<S, I>(), stream, mProcessingSettings.serializeGPU & 1)) {
230 throw std::runtime_error("kernel failure");
231 }
232 if (mProcessingSettings.debugLevel >= 1) {
233 if (t) {
234 if (deviceTimerTime != 0.) {
235 t->AddTime(deviceTimerTime);
236 if (t->IsRunning()) {
237 t->Abort();
238 }
239 } else if (t->IsRunning()) {
240 t->Stop();
241 }
242 }
243 if (CheckErrorCodes(cpuFallback) && !mProcessingSettings.ignoreNonFatalGPUErrors) {
244 throw std::runtime_error("kernel error code");
245 }
246 }
247 return retVal;
248}

266template <class T, int32_t I>
267HighResTimer& GPUReconstructionCPU::getKernelTimer(RecoStep step, int32_t num, size_t addMemorySize, bool increment)
268{
269 static int32_t id = getNextTimerId();
270 timerMeta* timer = getTimerById(id, increment);
271 if (timer == nullptr) {
272 timer = insertTimer(id, GetKernelName<T, I>(), -1, NSLICES, 0, step);
273 }
274 if (addMemorySize) {
275 timer->memSize += addMemorySize;
276 }
277 if (num < 0 || num >= timer->num) {
278 throw std::runtime_error("Invalid timer requested");
279 }
280 return timer->timer[num];
281}

283template <class T, int32_t J>
284HighResTimer& GPUReconstructionCPU::getTimer(const char* name, int32_t num)
285{
286 static int32_t id = getNextTimerId();
287 timerMeta* timer = getTimerById(id);
288 if (timer == nullptr) {
289 int32_t max = std::max<int32_t>({getOMPMaxThreads(), mProcessingSettings.nStreams});
290 timer = insertTimer(id, name, J, max, 1, RecoStep::NoRecoStep);
291 }
292 if (num == -1) {
293 num = getOMPThreadNum();
294 }
295 if (num < 0 || num >= timer->num) {
296 throw std::runtime_error("Invalid timer requested");
297 }
298 return timer->timer[num];
299}
