193template <class S, int32_t I, typename... Args>
194inline int32_t GPUReconstructionCPU::runKernel(krnlSetup&& setup, Args&&... args)
195{
196 HighResTimer* t = nullptr;
197 GPUCA_RECO_STEP myStep = S::GetRecoStep() == GPUCA_RECO_STEP::NoRecoStep ? setup.x.step : S::GetRecoStep();
198 if (myStep == GPUCA_RECO_STEP::NoRecoStep) {
199 throw std::runtime_error("Failure running general kernel without defining RecoStep");
200 }
201 int32_t cpuFallback = IsGPU() ? (setup.x.device == krnlDeviceType::CPU ? 2 : (mRecoSteps.stepsGPUMask & myStep) != myStep) : 0;
202 uint32_t& nThreads = setup.x.nThreads;
203 uint32_t& nBlocks = setup.x.nBlocks;
204 const uint32_t stream = setup.x.stream;
205 auto prop = getKernelProperties<S, I>();
206 const int32_t autoThreads = cpuFallback ? 1 : prop.nThreads;
207 const int32_t autoBlocks = cpuFallback ? 1 : (prop.forceBlocks ? prop.forceBlocks : (prop.minBlocks * mBlockCount));
208 if (nBlocks == (uint32_t)-1) {
209 nBlocks = (nThreads + autoThreads - 1) / autoThreads;
210 nThreads = autoThreads;
211 } else if (nBlocks == (uint32_t)-2) {
212 nBlocks = nThreads;
213 nThreads = autoThreads;
214 } else if (nBlocks == (uint32_t)-3) {
215 nBlocks = autoBlocks;
216 nThreads = autoThreads;
217 } else if ((int32_t)nThreads < 0) {
218 nThreads = cpuFallback ? 1 : -nThreads;
219 }
220 if (nThreads > GPUCA_MAX_THREADS) {
221 throw std::runtime_error("GPUCA_MAX_THREADS exceeded");
222 }
223 if (mProcessingSettings.debugLevel >= 3) {
224 GPUInfo("Running kernel %s (Stream %d, Range %d/%d, Grid %d/%d) on %s", GetKernelName<S, I>(), stream, setup.y.start, setup.y.num, nBlocks, nThreads, cpuFallback == 2 ? "CPU (forced)" : cpuFallback ? "CPU (fallback)" : mDeviceName.c_str());
225 }
226 if (nThreads == 0 || nBlocks == 0) {
227 return 0;
228 }
229 if (mProcessingSettings.debugLevel >= 1) {
230 t = &getKernelTimer<S, I>(myStep, !IsGPU() || cpuFallback ? getOMPThreadNum() : stream);
231 if ((!mProcessingSettings.deviceTimers || !IsGPU() || cpuFallback) && (mNestedLoopOmpFactor < 2 || getOMPThreadNum() == 0)) {
232 t->Start();
233 }
234 }
235 double deviceTimerTime = 0.;
236 int32_t retVal = runKernelImplWrapper(gpu_reconstruction_kernels::classArgument<S, I>(), cpuFallback, deviceTimerTime, std::forward<krnlSetup&&>(setup), std::forward<Args>(args)...);
237 if (GPUDebug(GetKernelName<S, I>(), stream, mProcessingSettings.serializeGPU & 1)) {
238 throw std::runtime_error("kernel failure");
239 }
240 if (mProcessingSettings.debugLevel >= 1) {
241 if (t) {
242 if (deviceTimerTime != 0.) {
243 t->AddTime(deviceTimerTime);
244 if (t->IsRunning()) {
245 t->Abort();
246 }
247 } else if (t->IsRunning()) {
248 t->Stop();
249 }
250 }
251 if (CheckErrorCodes(cpuFallback) && !mProcessingSettings.ignoreNonFatalGPUErrors) {
252 throw std::runtime_error("kernel error code");
253 }
254 }
255 return retVal;
256}

274template <class T, int32_t I>
275HighResTimer& GPUReconstructionCPU::getKernelTimer(RecoStep step, int32_t num, size_t addMemorySize, bool increment)
276{
277 static int32_t id = getNextTimerId();
278 timerMeta* timer = getTimerById(id, increment);
279 if (timer == nullptr) {
280 timer = insertTimer(id, GetKernelName<T, I>(), -1, NSLICES, 0, step);
281 }
282 if (addMemorySize) {
283 timer->memSize += addMemorySize;
284 }
285 if (num < 0 || num >= timer->num) {
286 throw std::runtime_error("Invalid timer requested");
287 }
288 return timer->timer[num];
289}
290
291template <class T, int32_t J>
292HighResTimer& GPUReconstructionCPU::getTimer(const char* name, int32_t num)
293{
294 static int32_t id = getNextTimerId();
295 timerMeta* timer = getTimerById(id);
296 if (timer == nullptr) {
297 int32_t max = std::max<int32_t>({getOMPMaxThreads(), mProcessingSettings.nDeviceHelperThreads + 1, mProcessingSettings.nStreams});
298 timer = insertTimer(id, name, J, max, 1, RecoStep::NoRecoStep);
299 }
300 if (num == -1) {
301 num = getOMPThreadNum();
302 }
303 if (num < 0 || num >= timer->num) {
304 throw std::runtime_error("Invalid timer requested");
305 }
306 return timer->timer[num];
307}
