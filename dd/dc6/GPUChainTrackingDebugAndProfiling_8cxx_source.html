46int32_t GPUChainTracking::DoProfile()
47{
48#ifdef GPUCA_TRACKLET_CONSTRUCTOR_DO_PROFILE
49 std::unique_ptr<char[]> stageAtSync{new char[PROFILE_MAX_SIZE]};
50 mRec->GPUMemCpy(stageAtSync.get(), processorsShadow()->tpcTrackers[0].mStageAtSync, PROFILE_MAX_SIZE, -1, false);
51
52 FILE* fp = fopen("profile.txt", "w+");
53 FILE* fp2 = fopen("profile.bmp", "w+b");
54
55 const int32_t bmpheight = 8192;
56 BITMAPFILEHEADER bmpFH;
57 BITMAPINFOHEADER bmpIH;
58 memset(&bmpFH, 0, sizeof(bmpFH));
59 memset(&bmpIH, 0, sizeof(bmpIH));
60
61 bmpFH.bfType = 19778; //"BM"
62 bmpFH.bfSize = sizeof(bmpFH) + sizeof(bmpIH) + (ConstructorBlockCount() * ConstructorThreadCount() / 32 * 33 - 1) * bmpheight;
63 bmpFH.bfOffBits = sizeof(bmpFH) + sizeof(bmpIH);
64
65 bmpIH.biSize = sizeof(bmpIH);
66 bmpIH.biWidth = ConstructorBlockCount() * ConstructorThreadCount() / 32 * 33 - 1;
67 bmpIH.biHeight = bmpheight;
68 bmpIH.biPlanes = 1;
69 bmpIH.biBitCount = 32;
70
71 fwrite(&bmpFH, 1, sizeof(bmpFH), fp2);
72 fwrite(&bmpIH, 1, sizeof(bmpIH), fp2);
73
74 int32_t nEmptySync = 0;
75 for (uint32_t i = 0; i < bmpheight * ConstructorBlockCount() * ConstructorThreadCount(); i += ConstructorBlockCount() * ConstructorThreadCount()) {
76 int32_t fEmpty = 1;
77 for (uint32_t j = 0; j < ConstructorBlockCount() * ConstructorThreadCount(); j++) {
78 fprintf(fp, "%d\t", stageAtSync[i + j]);
79 int32_t color = 0;
80 if (stageAtSync[i + j] == 1) {
81 color = RGB(255, 0, 0);
82 }
83 if (stageAtSync[i + j] == 2) {
84 color = RGB(0, 255, 0);
85 }
86 if (stageAtSync[i + j] == 3) {
87 color = RGB(0, 0, 255);
88 }
89 if (stageAtSync[i + j] == 4) {
90 color = RGB(255, 255, 0);
91 }
92 fwrite(&color, 1, sizeof(int32_t), fp2);
93 if (j > 0 && j % 32 == 0) {
94 color = RGB(255, 255, 255);
95 fwrite(&color, 1, 4, fp2);
96 }
97 if (stageAtSync[i + j]) {
98 fEmpty = 0;
99 }
100 }
101 fprintf(fp, "\n");
102 if (fEmpty) {
103 nEmptySync++;
104 } else {
105 nEmptySync = 0;
106 }
107 (void)nEmptySync;
108 // if (nEmptySync == GPUCA_SCHED_ROW_STEP + 2) break;
109 }
110
111 fclose(fp);
112 fclose(fp2);
113#endif
114 return 0;
115}

248 nTRDTracks += trk.getNtracklets() != 0;

252 }

256void GPUChainTracking::SanityCheck()
257{
258#ifdef GPUCA_HAVE_O2HEADERS
259 size_t nErrors = 0;
260
261 for (uint32_t i = 0; i < mIOPtrs.nOutputTracksTPCO2; i++) {
262 const auto& trk = mIOPtrs.outputTracksTPCO2[i];
263 const auto& ref = trk.getClusterRef();
264 if (ref.getFirstEntry() > mIOPtrs.nOutputClusRefsTPCO2) {
265 if (nErrors++ < 1000) {
266 GPUError("Invalid getFirst() entry in cluster reference: %u > $u", ref.getFirstEntry(), mIOPtrs.nOutputClusRefsTPCO2);
267 continue;
268 }
269 }
270 if (ref.getFirstEntry() + (ref.getEntries() * 3 + 1) / 2 > mIOPtrs.nOutputClusRefsTPCO2) {
271 if (nErrors++ < 1000) {
272 GPUError("Invalid getEntries() entry in cluster reference: %u > $u", ref.getFirstEntry() + (ref.getEntries() * 3 + 1) / 2, mIOPtrs.nOutputClusRefsTPCO2);
273 continue;
274 }
275 }
276 for (int32_t j = 0; j < trk.getNClusters(); j++) {
277 uint8_t sector, row;
278 uint32_t cl;
279 trk.getClusterReference(mIOPtrs.outputClusRefsTPCO2, j, sector, row, cl);
280 if (sector >= GPUCA_NSLICES || row >= GPUCA_ROW_COUNT) {
281 if (nErrors++ < 1000) {
282 GPUError("Invalid sector / row %d / %d", (int32_t)sector, (int32_t)row);
283 continue;
284 }
285 }
286 if (cl >= mIOPtrs.clustersNative->nClusters[sector][row]) {
287 if (nErrors++ < 1000) {
288 GPUError("Invalid cluster index %d >= %d", cl, mIOPtrs.clustersNative->nClusters[sector][row]);
289 }
290 }
291 }
292 }
293
294 if (nErrors == 0) {
295 GPUInfo("Sanity check passed");
296 } else {
297 GPUError("Sanity check found %lu errors", nErrors);
298 }
299#endif
300}

302void GPUChainTracking::RunTPCClusterFilter(o2::tpc::ClusterNativeAccess* clusters, std::function<o2::tpc::ClusterNative*(size_t)> allocator, bool applyClusterCuts)
303{
304 GPUTPCClusterFilter clusterFilter(*clusters);
305 o2::tpc::ClusterNative* outputBuffer = nullptr;
306 for (int32_t iPhase = 0; iPhase < 2; iPhase++) {
307 uint32_t countTotal = 0;
308 for (uint32_t iSector = 0; iSector < GPUCA_NSLICES; iSector++) {
309 for (uint32_t iRow = 0; iRow < GPUCA_ROW_COUNT; iRow++) {
310 uint32_t count = 0;
311 for (uint32_t k = 0; k < clusters->nClusters[iSector][iRow]; k++) {
312 o2::tpc::ClusterNative cl = clusters->clusters[iSector][iRow][k];
313 bool keep = true;
314 if (applyClusterCuts) {
315 keep = keep && cl.qTot > param().rec.tpc.cfQTotCutoff && cl.qMax > param().rec.tpc.cfQMaxCutoff;
316 keep = keep && (!(cl.getFlags() & o2::tpc::ClusterNative::flagSingle) || ((cl.sigmaPadPacked || cl.qMax > param().rec.tpc.cfQMaxCutoffSinglePad) && (cl.sigmaTimePacked || cl.qMax > param().rec.tpc.cfQMaxCutoffSingleTime)));
317 }
318 if (param().tpcCutTimeBin > 0) {
319 keep = keep && cl.getTime() < param().tpcCutTimeBin;
320 }
321 keep = keep && (!GetProcessingSettings().tpcApplyDebugClusterFilter || clusterFilter.filter(iSector, iRow, cl));
322 if (iPhase && keep) {
323 outputBuffer[countTotal] = cl;
324 }
325 count += keep;
326 countTotal += keep;
327 }
328 if (iPhase) {
329 clusters->nClusters[iSector][iRow] = count;
330 }
331 }
332 }
333 if (iPhase) {
334 clusters->clustersLinear = outputBuffer;
335 clusters->setOffsetPtrs();
336 } else {
337 outputBuffer = allocator(countTotal);
338 }
339 }
