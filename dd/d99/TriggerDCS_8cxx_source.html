30bool TriggerDCS::isTRUEnabled(int itru) const
31{
32 if (itru < 32) {
33 return std::bitset<32>(mSTUEMCal.getRegion()).test(itru);
34 } else {
35 return std::bitset<32>(mSTUDCAL.getRegion()).test(itru - 32);
36 }
37}
