93 o2::framework::ServiceSpec CommonServices::monitoringSpec()
94 {
95  return ServiceSpec{
96  .name = "monitoring",
97  .init = [](ServiceRegistryRef registry, DeviceState&, fair::mq::ProgOptions& options) -> ServiceHandle {
98  void* service = nullptr;
99  bool isWebsocket = strncmp(options.GetPropertyAsString("driver-client-backend").c_str(), "ws://", 4) == 0;
100  bool isDefault = options.GetPropertyAsString("monitoring-backend") == "default";
101  bool useDPL = (isWebsocket && isDefault) || options.GetPropertyAsString("monitoring-backend") == "dpl://";
102  o2::monitoring::Monitoring* monitoring;
103  if (useDPL) {
104  monitoring = new Monitoring();
105  auto dplBackend = std::make_unique<DPLMonitoringBackend>(registry);
106  (dynamic_cast<o2::monitoring::Backend*>(dplBackend.get()))->setVerbosity(o2::monitoring::Verbosity::Debug);
107  monitoring->addBackend(std::move(dplBackend));
108  } else {
109  auto backend = isDefault ? "infologger://" : options.GetPropertyAsString("monitoring-backend");
110  monitoring = MonitoringFactory::Get(backend).release();
111  }
112  service = monitoring;
113  monitoring->enableBuffering(MONITORING_QUEUE_SIZE);
114  assert(registry.get<DeviceSpec const>().name.empty() == false);
115  monitoring->addGlobalTag("pipeline_id", std::to_string(registry.get<DeviceSpec const>().inputTimesliceId));
116  monitoring->addGlobalTag("dataprocessor_name", registry.get<DeviceSpec const>().name);
117  monitoring->addGlobalTag("dpl_instance", options.GetPropertyAsString("shm-segment-id"));
118  return ServiceHandle{TypeIdHelpers::uniqueId<Monitoring>(), service};
119  },
120  .configure = noConfiguration(),
121  .start = [](ServiceRegistryRef services, void* service) {
122  auto* monitoring = (o2::monitoring::Monitoring*)service;
123 
124  auto extRunNumber = services.get<RawDeviceService>().device()->fConfig->GetProperty<std::string>("runNumber", "unspecified");
125  if (extRunNumber == "unspecified") {
126  return;
127  }
128  try {
129  monitoring->setRunNumber(std::stoul(extRunNumber));
130  } catch (...) {
131  } },
132  .exit = [](ServiceRegistryRef registry, void* service) {

166 o2::framework::ServiceSpec CommonServices::streamContextSpec()
167 {
168  return ServiceSpec{
169  .name = "stream-context",
170  .uniqueId = simpleServiceId<StreamContext>(),
171  .init = simpleServiceInit<StreamContext, StreamContext, ServiceKind::Stream>(),
172  .configure = noConfiguration(),
173  .preProcessing = [](ProcessingContext& context, void* service) {
174  auto* stream = (StreamContext*)service;
175  auto& routes = context.services().get<DeviceSpec const>().outputs;
176  // Notice I need to do this here, because different invocation for
177  // the same stream might be referring to different data processors.
178  // We should probably have a context which is per stream of a specific
179  // data processor.
180  stream->routeDPLCreated.resize(routes.size());
181  stream->routeCreated.resize(routes.size());
182  // Reset the routeDPLCreated at every processing step
183  std::fill(stream->routeDPLCreated.begin(), stream->routeDPLCreated.end(), false);
184  std::fill(stream->routeCreated.begin(), stream->routeCreated.end(), false); },
185  .postProcessing = [](ProcessingContext& processingContext, void* service) {
186  auto* stream = (StreamContext*)service;
187  auto& routes = processingContext.services().get<DeviceSpec const>().outputs;
188  auto& timeslice = processingContext.services().get<TimingInfo>().timeslice;
189  auto& messageContext = processingContext.services().get<MessageContext>();
190  // Check if we never created any data for this timeslice
191  // if we did not, but we still have didDispatched set to true
192  // it means it was created out of band.
193  bool userDidCreate = false;
194  O2_SIGNPOST_ID_FROM_POINTER(cid, stream_context, service);
195  for (size_t ri = 0; ri < routes.size(); ++ri) {
196  if (stream->routeCreated[ri] == true && stream->routeDPLCreated[ri] == false) {
197  userDidCreate = true;
198  break;
199  }
200  }
201  O2_SIGNPOST_EVENT_EMIT(stream_context, cid, "postProcessingCallbacks", "userDidCreate == %d && didDispatch == %d",
202  userDidCreate,
203  messageContext.didDispatch());
204  if (userDidCreate == false && messageContext.didDispatch() == true) {
205  O2_SIGNPOST_EVENT_EMIT(stream_context, cid, "postProcessingCallbacks", "Data created out of band userDidCreate == %d && messageContext.didDispatch == %d",
206  userDidCreate,
207  messageContext.didDispatch());
208  return;
209  }
210  if (userDidCreate == false && messageContext.didDispatch() == false) {
211  O2_SIGNPOST_ID_FROM_POINTER(cid, stream_context, service);
212  O2_SIGNPOST_EVENT_EMIT(stream_context, cid, "postProcessingCallbacks", "No data created.");
213  return;
214  }
215  for (size_t ri = 0; ri < routes.size(); ++ri) {
216  auto &route = routes[ri];
217  auto &matcher = route.matcher;
218  if (stream->routeDPLCreated[ri] == true) {
219  O2_SIGNPOST_EVENT_EMIT(stream_context, cid, "postProcessingCallbacks", "Data created by DPL. ri = %" PRIu64 ", %{public}s",
220  (uint64_t)ri, DataSpecUtils::describe(matcher).c_str());
221  continue;
222  }
223  if (stream->routeCreated[ri] == true) {
224  continue;
225  } if ((timeslice % route.maxTimeslices) != route.timeslice) {
226  O2_SIGNPOST_EVENT_EMIT(stream_context, cid, "postProcessingCallbacks", "Route ri = %" PRIu64 ", skipped because of pipelining.",
227  (uint64_t)ri);
228  continue;
229  }
230  if (matcher.lifetime == Lifetime::Timeframe) {
231  O2_SIGNPOST_EVENT_EMIT(stream_context, cid, "postProcessingCallbacks",
232  "Expected Lifetime::Timeframe data %{public}s was not created for timeslice %" PRIu64 " and might result in dropped timeframes",
233  DataSpecUtils::describe(matcher).c_str(), (uint64_t)timeslice);
234  LOGP(error, "Expected Lifetime::Timeframe data {} was not created for timeslice {} and might result in dropped timeframes", DataSpecUtils::describe(matcher), timeslice);
235  }
236  } },
237  .preEOS = [](EndOfStreamContext& context, void* service) {
238  // We need to reset the routeDPLCreated / routeCreated because the end of stream
239  // uses a different context which does not know about the routes.
240  // FIXME: This should be fixed in a different way, but for now it will
241  // allow TPC IDC to work.
242  auto* stream = (StreamContext*)service;
243  auto& routes = context.services().get<DeviceSpec const>().outputs;
244  // Notice I need to do this here, because different invocation for
245  // the same stream might be referring to different data processors.
246  // We should probably have a context which is per stream of a specific
247  // data processor.
248  stream->routeDPLCreated.resize(routes.size());
249  stream->routeCreated.resize(routes.size());
250  // Reset the routeCreated / routeDPLCreated at every processing step
251  std::fill(stream->routeCreated.begin(), stream->routeCreated.end(), false);
252  std::fill(stream->routeDPLCreated.begin(), stream->routeDPLCreated.end(), false); },
253  .kind = ServiceKind::Stream};
254 }

477 // CCDB Support service
478 o2::framework::ServiceSpec CommonServices::ccdbSupportSpec()
479 {
480  return ServiceSpec{
481  .name = "ccdb-support",
482  .init = [](ServiceRegistryRef services, DeviceState&, fair::mq::ProgOptions&) -> ServiceHandle {
483  // iterate on all the outputs matchers
484  auto& spec = services.get<DeviceSpec const>();
485  for (auto& output : spec.outputs) {
486  if (DataSpecUtils::match(output.matcher, ConcreteDataTypeMatcher{"FLP", "DISTSUBTIMEFRAME"})) {
487  LOGP(debug, "Optional inputs support enabled");
488  return ServiceHandle{.hash = TypeIdHelpers::uniqueId<CCDBSupport>(), .instance = new CCDBSupport, .kind = ServiceKind::Serial};
489  }
490  }
491  return ServiceHandle{.hash = TypeIdHelpers::uniqueId<CCDBSupport>(), .instance = nullptr, .kind = ServiceKind::Serial};
492  },
493  .configure = noConfiguration(),
494  .finaliseOutputs = [](ProcessingContext& pc, void* service) {
495  if (!service) {
496  return;
497  }
498  if (pc.outputs().countDeviceOutputs(true) == 0) {
499  LOGP(debug, "We are w/o outputs, do not automatically add DISTSUBTIMEFRAME to outgoing messages");
500  return;
501  }
502  auto& timingInfo = pc.services().get<TimingInfo>();
503 
504  // For any output that is a FLP/DISTSUBTIMEFRAME with subspec != 0,
505  // we create a new message.
506  InputSpec matcher{"matcher", ConcreteDataTypeMatcher{"FLP", "DISTSUBTIMEFRAME"}};
507  auto& streamContext = pc.services().get<StreamContext>();
508  for (size_t oi = 0; oi < pc.services().get<DeviceSpec const>().outputs.size(); ++oi) {
509  OutputRoute const& output = pc.services().get<DeviceSpec const>().outputs[oi];
510  if ((output.timeslice % output.maxTimeslices) != 0) {
511  continue;
512  }
513  if (DataSpecUtils::match(output.matcher, ConcreteDataTypeMatcher{"FLP", "DISTSUBTIMEFRAME"})) {
514  auto concrete = DataSpecUtils::asConcreteDataMatcher(output.matcher);
515  if (concrete.subSpec == 0) {
516  continue;
517  }
518  auto& stfDist = pc.outputs().make<o2::header::STFHeader>(Output{concrete.origin, concrete.description, concrete.subSpec});
519  stfDist.id = timingInfo.timeslice;
520  stfDist.firstOrbit = timingInfo.firstTForbit;
521  stfDist.runNumber = timingInfo.runNumber;
522  // We mark it as not created, because we do should not account for it when
523  // checking if we created all the data for a timeslice.
524  O2_SIGNPOST_ID_FROM_POINTER(sid, stream_context, &streamContext);
525  O2_SIGNPOST_EVENT_EMIT(stream_context, sid, "finaliseOutputs", "Route %" PRIu64 " (%{public}s) was created by DPL.", (uint64_t)oi,
526  DataSpecUtils::describe(output.matcher).c_str());
527  streamContext.routeDPLCreated[oi] = true;
528  }
529  } },
530  .kind = ServiceKind::Global};
531 }

533 // Decongestion service
534 // If we do not have any Timeframe input, it means we must be creating timeslices
535 // in order and that we should propagate the oldest possible timeslice at the end
536 // of each processing step.
537 o2::framework::ServiceSpec CommonServices::decongestionSpec()
538 {
539  return ServiceSpec{
540  .name = "decongestion",
541  .init = [](ServiceRegistryRef services, DeviceState&, fair::mq::ProgOptions& options) -> ServiceHandle {
542  auto* decongestion = new DecongestionService();
543  for (auto& input : services.get<DeviceSpec const>().inputs) {
