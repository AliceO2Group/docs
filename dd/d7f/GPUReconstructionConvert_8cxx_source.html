239 inline void zsEncoder::ZSstreamOut(uint16_t* bufIn, uint32_t& lenIn, uint8_t* bufOut, uint32_t& lenOut, uint32_t nBits)
240 {
241  uint32_t byte = 0, bits = 0;
242  uint32_t mask = (1 << nBits) - 1;
243  for (uint32_t i = 0; i < lenIn; i++) {
244  byte |= (bufIn[i] & mask) << bits;
245  bits += nBits;
246  while (bits >= 8) {
247  bufOut[lenOut++] = (uint8_t)(byte & 0xFF);
248  byte = byte >> 8;
249  bits -= 8;
250  }
251  }
252  if (bits) {
253  bufOut[lenOut++] = byte;
254  }
255  lenIn = 0;
256 }

290 inline bool zsEncoderRow::sort(const o2::tpc::Digit a, const o2::tpc::Digit b)
291 {
292  int32_t endpointa = param->tpcGeometry.GetRegion(a.getRow());
293  int32_t endpointb = param->tpcGeometry.GetRegion(b.getRow());
294  endpointa = 2 * endpointa + (a.getRow() >= param->tpcGeometry.GetRegionStart(endpointa) + param->tpcGeometry.GetRegionRows(endpointa) / 2);
295  endpointb = 2 * endpointb + (b.getRow() >= param->tpcGeometry.GetRegionStart(endpointb) + param->tpcGeometry.GetRegionRows(endpointb) / 2);
296  if (endpointa != endpointb) {
297  return endpointa <= endpointb;
298  }
299  if (a.getTimeStamp() != b.getTimeStamp()) {
300  return a.getTimeStamp() < b.getTimeStamp();
301  }
302  if (a.getRow() != b.getRow()) {
303  return a.getRow() < b.getRow();
304  }
305  return a.getPad() < b.getPad();
306 }

308 bool zsEncoderRow::checkInput(std::vector<o2::tpc::Digit>& tmpBuffer, uint32_t k)
309 {
310  seqLen = 1;
311  if (lastRow != tmpBuffer[k].getRow()) {
312  endpointStart = param->tpcGeometry.GetRegionStart(curRegion);
313  endpoint = curRegion * 2;
314  if (tmpBuffer[k].getRow() >= endpointStart + param->tpcGeometry.GetRegionRows(curRegion) / 2) {
315  endpoint++;
316  endpointStart += param->tpcGeometry.GetRegionRows(curRegion) / 2;
317  }
318  }
319  for (uint32_t l = k + 1; l < tmpBuffer.size(); l++) {
320  if (tmpBuffer[l].getRow() == tmpBuffer[k].getRow() && tmpBuffer[l].getTimeStamp() == tmpBuffer[k].getTimeStamp() && tmpBuffer[l].getPad() == tmpBuffer[l - 1].getPad() + 1) {
321  seqLen++;
322  } else {
323  break;
324  }
325  }
326  if (lastEndpoint >= 0 && lastTime != -1 && (int32_t)hdr->nTimeBinSpan + tmpBuffer[k].getTimeStamp() - lastTime >= 256) {
327  lastEndpoint = -1;
328  }
329  if (endpoint == lastEndpoint) {
330  uint32_t sizeChk = (uint32_t)(pagePtr - reinterpret_cast<uint8_t*>(page)); // already written
331  sizeChk += 2 * (nRowsInTB + (tmpBuffer[k].getRow() != lastRow && tmpBuffer[k].getTimeStamp() == lastTime)); // TB HDR
332  sizeChk += streamSize8; // in stream buffer
333  sizeChk += (lastTime != tmpBuffer[k].getTimeStamp()) && ((sizeChk + (streamSize * encodeBits + 7) / 8) & 1); // time bin alignment
334  sizeChk += (tmpBuffer[k].getTimeStamp() != lastTime || tmpBuffer[k].getRow() != lastRow) ? 3 : 0; // new row overhead
335  sizeChk += (lastTime != -1 && tmpBuffer[k].getTimeStamp() > lastTime) ? ((tmpBuffer[k].getTimeStamp() - lastTime - 1) * 2) : 0; // empty time bins
336  sizeChk += 2; // sequence metadata
337  const uint32_t streamSizeChkBits = streamSize * encodeBits + ((lastTime != tmpBuffer[k].getTimeStamp() && (streamSize * encodeBits) % 8) ? (8 - (streamSize * encodeBits) % 8) : 0);
338  if (sizeChk + (encodeBits + streamSizeChkBits + 7) / 8 > TPCZSHDR::TPC_ZS_PAGE_SIZE) {
339  lastEndpoint = -1;
340  } else if (sizeChk + (seqLen * encodeBits + streamSizeChkBits + 7) / 8 > TPCZSHDR::TPC_ZS_PAGE_SIZE) {
341  seqLen = ((TPCZSHDR::TPC_ZS_PAGE_SIZE - sizeChk) * 8 - streamSizeChkBits) / encodeBits;
342  }
343  // sizeChk += (seqLen * encodeBits + streamSizeChkBits + 7) / 8;
344  // printf("Endpoint %d (%d), Pos %d, Chk %d, Len %d, rows %d, StreamSize %d %d, time %d (%d), row %d (%d), pad %d\n", endpoint, lastEndpoint, (int32_t) (pagePtr - reinterpret_cast<uint8_t*>(page)), sizeChk, seqLen, nRowsInTB, streamSize8, streamSize, (int32_t)tmpBuffer[k].getTimeStamp(), lastTime, (int32_t)tmpBuffer[k].getRow(), lastRow, tmpBuffer[k].getPad());
345  }
346  return endpoint != lastEndpoint || tmpBuffer[k].getTimeStamp() != lastTime;
347 }

366 uint32_t zsEncoderRow::encodeSequence(std::vector<o2::tpc::Digit>& tmpBuffer, uint32_t k)
367 {
368  if (tmpBuffer[k].getTimeStamp() != lastTime) {
369  if (lastTime != -1) {
370  hdr->nTimeBinSpan += tmpBuffer[k].getTimeStamp() - lastTime - 1;
371  pagePtr += (tmpBuffer[k].getTimeStamp() - lastTime - 1) * 2;
372  }
373  hdr->nTimeBinSpan++;
374  if ((pagePtr - reinterpret_cast<uint8_t*>(page)) & 1) {
375  pagePtr++;
376  }
377  curTBHdr = reinterpret_cast<TPCZSTBHDR*>(pagePtr);
378  curTBHdr->rowMask |= (endpoint & 1) << 15;
379  nRowsInTB = 0;
380  lastRow = GPUCA_ROW_COUNT;
381  }
382  if (tmpBuffer[k].getRow() != lastRow) {
383  curTBHdr->rowMask |= 1 << (tmpBuffer[k].getRow() - endpointStart);
384  ZSstreamOut(streamBuffer.data(), streamSize, streamBuffer8.data(), streamSize8, encodeBits);
385  if (nRowsInTB) {
386  curTBHdr->rowAddr1()[nRowsInTB - 1] = (pagePtr - reinterpret_cast<uint8_t*>(page)) + streamSize8;
387  }
388  nRowsInTB++;
389  nSeq = streamBuffer8.data() + streamSize8++;
390  *nSeq = 0;
391  }
392  (*nSeq)++;
393  streamBuffer8[streamSize8++] = tmpBuffer[k].getPad();
394  streamBuffer8[streamSize8++] = streamSize + seqLen;
395  for (int32_t l = 0; l < seqLen; l++) {
396  streamBuffer[streamSize++] = (uint16_t)(tmpBuffer[k + l].getChargeFloat() * encodeBitsFactor + 0.5f);
397  }
398  return seqLen;
399 }

401 void zsEncoderRow::decodePage(std::vector<o2::tpc::Digit>& outputBuffer, const zsPage* decPage, uint32_t decEndpoint, uint32_t firstOrbit, uint32_t triggerBC)
402 {
403  const uint8_t* decPagePtr = reinterpret_cast<const uint8_t*>(decPage);
404  const o2::header::RAWDataHeader* rdh = (const o2::header::RAWDataHeader*)decPagePtr;
405  if (o2::raw::RDHUtils::getMemorySize(*rdh) == sizeof(o2::header::RAWDataHeader)) {
406  return;
407  }
408  decPagePtr += sizeof(o2::header::RAWDataHeader);
409  const TPCZSHDR* decHDR = reinterpret_cast<const TPCZSHDR*>(decPagePtr);
410  decPagePtr += sizeof(*decHDR);
411  if (decHDR->version != 1 && decHDR->version != 2) {
412  throw std::runtime_error("invalid ZS version "s + std::to_string(decHDR->version) + " (1 or 2 expected)"s);
413  }
414  const float decodeBitsFactor = 1.f / (1 << (encodeBits - 10));
415  uint32_t mask = (1 << encodeBits) - 1;
416  int32_t cruid = decHDR->cruID;
417  uint32_t sector = cruid / 10;
418  if (sector != iSector) {
419  throw std::runtime_error("invalid TPC sector");
420  }
421  int32_t region = cruid % 10;
422  if ((uint32_t)region != decEndpoint / 2) {
423  throw std::runtime_error("CRU ID / endpoint mismatch");
424  }
425  int32_t nRowsRegion = param->tpcGeometry.GetRegionRows(region);
426 
427  int32_t timeBin = (decHDR->timeOffset + (uint64_t)(o2::raw::RDHUtils::getHeartBeatOrbit(*rdh) - firstOrbit) * o2::constants::lhc::LHCMaxBunches) / LHCBCPERTIMEBIN;
428  for (int32_t l = 0; l < decHDR->nTimeBinSpan; l++) {
429  if ((decPagePtr - reinterpret_cast<const uint8_t*>(decPage)) & 1) {
430  decPagePtr++;
431  }
432  const TPCZSTBHDR* tbHdr = reinterpret_cast<const TPCZSTBHDR*>(decPagePtr);
433  bool upperRows = tbHdr->rowMask & 0x8000;
434  if (tbHdr->rowMask != 0 && ((upperRows) ^ ((decEndpoint & 1) != 0))) {
435  throw std::runtime_error("invalid endpoint");
436  }
437  const int32_t rowOffset = param->tpcGeometry.GetRegionStart(region) + (upperRows ? (nRowsRegion / 2) : 0);
438  const int32_t nRows = upperRows ? (nRowsRegion - nRowsRegion / 2) : (nRowsRegion / 2);
439  const int32_t nRowsUsed = __builtin_popcount((uint32_t)(tbHdr->rowMask & 0x7FFF));
440  decPagePtr += nRowsUsed ? (2 * nRowsUsed) : 2;
441  int32_t rowPos = 0;
442  for (int32_t m = 0; m < nRows; m++) {
443  if ((tbHdr->rowMask & (1 << m)) == 0) {
444  continue;
445  }
446  const uint8_t* rowData = rowPos == 0 ? decPagePtr : (reinterpret_cast<const uint8_t*>(decPage) + tbHdr->rowAddr1()[rowPos - 1]);
447  const int32_t nSeqRead = *rowData;
448  const uint8_t* adcData = rowData + 2 * nSeqRead + 1;
449  int32_t nADC = (rowData[2 * nSeqRead] * encodeBits + 7) / 8;
450  decPagePtr += 1 + 2 * nSeqRead + nADC;
451  uint32_t byte = 0, bits = 0, posXbits = 0;
452  std::array<uint16_t, TPCZSHDR::TPC_ZS_PAGE_SIZE> decBuffer;
453  for (int32_t n = 0; n < nADC; n++) {
454  byte |= *(adcData++) << bits;
455  bits += 8;
456  while (bits >= encodeBits) {
457  decBuffer[posXbits++] = byte & mask;
458  byte = byte >> encodeBits;
459  bits -= encodeBits;

892  bool extendFailure = false;
893  uint32_t nOutput = 0;
894  uint32_t minTimeBin = -1, maxTimeBin = 0;
895  for (uint32_t i = 0; i < decHDR->nTimebinHeaders; i++) {
896  int32_t sizeLeftInPage = payloadEnd - decPagePtr;
897  if (sizeLeftInPage <= 0) {
898  throw std::runtime_error("Decoding ran beyond end of page before processing extended timebin");
899  }
900  if (i == decHDR->nTimebinHeaders - 1 && (decHDR->flags & o2::tpc::TPCZSHDRV2::ZSFlags::payloadExtendsToNextPage)) {
901  if (o2::raw::RDHUtils::getMemorySize(*rdh) != TPCZSHDR::TPC_ZS_PAGE_SIZE) {
902  throw std::runtime_error("pageExtends signaled, but current page is not full");
903  }
904 
905  const uint8_t* pageNext = ((const uint8_t*)decPage) + TPCZSHDR::TPC_ZS_PAGE_SIZE;
906  const o2::header::RAWDataHeader* rdhNext = (const o2::header::RAWDataHeader*)pageNext;
907 
908  if ((uint16_t)(o2::raw::RDHUtils::getPageCounter(*rdh) + 1) != o2::raw::RDHUtils::getPageCounter(*rdhNext)) {
909  GPUError("Incomplete HBF: Payload extended to next page, but next page missing in stream (packet counters %d %d)", (int32_t)o2::raw::RDHUtils::getPageCounter(*rdh), (int32_t)o2::raw::RDHUtils::getPageCounter(*rdhNext));
910  extendFailure = true;
911  decPagePtr = payloadEnd; // Next 8kb page is missing in stream, cannot decode remaining data, skip it
912  break;
913  }
914 
915  const TPCZSHDRV2* hdrNext = reinterpret_cast<const TPCZSHDRV2*>(pageNext + o2::raw::RDHUtils::getMemorySize(*rdhNext) - sizeof(TPCZSHDRV2));
916  tmpBuffer.resize(sizeLeftInPage + hdrNext->firstZSDataOffset - sizeof(o2::header::RAWDataHeader));
917  memcpy(tmpBuffer.data(), decPagePtr, sizeLeftInPage);
918  memcpy(tmpBuffer.data() + sizeLeftInPage, pageNext + sizeof(o2::header::RAWDataHeader), hdrNext->firstZSDataOffset - sizeof(o2::header::RAWDataHeader));
919  decPagePtr = tmpBuffer.data();
920  payloadEnd = decPagePtr + tmpBuffer.size();
921  }
922  uint8_t linkCount = *((const uint8_t*)decPagePtr) & 0x0F;
923  uint16_t linkBC = (*((const uint16_t*)decPagePtr) & 0xFFF0) >> 4;
924  bool v2Flag = decHDR->version == ZSVersion::ZSVersionDenseLinkBasedV2 && *((const uint8_t*)decPagePtr) & 0x10;
925  if (decHDR->version == ZSVersion::ZSVersionDenseLinkBasedV2) {
926  linkBC &= 0xFFC;
927  }
928  decPagePtr += sizeof(uint16_t);
929  std::vector<int32_t> links;
930  std::vector<std::bitset<80>> bitmasks;
931  uint32_t nTotalSamples = 0;
932  for (uint32_t l = 0; l < linkCount; l++) {
933  uint8_t decLinkX = *((const uint8_t*)decPagePtr);
934  decPagePtr += sizeof(uint8_t);
935  uint8_t decLink = decLinkX & 0b00011111;
936  std::bitset<10> bitmaskL2;
937  if (decLinkX & 0b00100000) {
938  bitmaskL2.set();
939  } else {
940  bitmaskL2 = std::bitset<10>(((((uint16_t)decLinkX) & 0b11000000) << 2) | (uint16_t) * ((const uint8_t*)decPagePtr));
941  decPagePtr += sizeof(uint8_t);
942  }
943 
944  std::bitset<80> bitmask(0);
945  for (int32_t i = 0; i < 10; i++) {
946  if (bitmaskL2.test(i)) {
947  bitmask |= std::bitset<80>(*((const uint8_t*)decPagePtr)) << i * 8;
948  decPagePtr += sizeof(uint8_t);
949  }
950  }
951  links.emplace_back(decLink);
952  bitmasks.emplace_back(bitmask);
953  nTotalSamples += bitmask.count();
954  }
955 
956  const uint8_t* adcData = (const uint8_t*)decPagePtr;
957  int32_t encodeBitsBlock = v2Flag ? v2nbits : encodeBits;
958  decPagePtr += (nTotalSamples * encodeBitsBlock + 7) / 8;
959 
960  // time bin might be smaller 0 due to triggerBC
961  int32_t timeBin = (int32_t(linkBC) + (int32_t)(o2::raw::RDHUtils::getHeartBeatOrbit(*rdh) - firstOrbit) * o2::constants::lhc::LHCMaxBunches - int32_t(triggerBC)) / LHCBCPERTIMEBIN;
962  if (timeBin < 0 || nTotalSamples == 0) {
963  if (timeBin < 0 && nTotalSamples > 0) {
964  LOGP(debug, "zsEncoderDenseLinkBased::decodePage skipping digits (linkBC {} + orbit {} - firstOrbit {}) * maxBunch {} - triggerBC {} = {} < 0, nTotalSamples {}", linkBC, o2::raw::RDHUtils::getHeartBeatOrbit(*rdh), firstOrbit, o2::constants::lhc::LHCMaxBunches, triggerBC, timeBin, nTotalSamples);
965  }
966  continue;
967  }
968  if (timeBin > maxTimeBin) {
969  maxTimeBin = timeBin;
970  }
971  if (timeBin < minTimeBin) {
972  minTimeBin = timeBin;
973  }
974 
975  std::vector<uint16_t> samples(nTotalSamples);
976  uint32_t mask = (1 << encodeBitsBlock) - 1;
977  uint32_t byte = 0, bits = 0, posXbits = 0;
978  while (posXbits < nTotalSamples) {
979  byte |= *(adcData++) << bits;
980  bits += 8;
981  while (bits >= encodeBitsBlock && posXbits < nTotalSamples) {
982  samples[posXbits++] = byte & mask;
983  byte = byte >> encodeBitsBlock;
984  bits -= encodeBitsBlock;
985  }
986  }
987  uint32_t samplePos = 0;
988 
989  for (uint32_t l = 0; l < linkCount; l++) {
990  uint8_t decLink = links[l];
991  const auto& bitmask = bitmasks[l];
992  int32_t nADC = bitmask.count();
993 
994  for (int32_t j = 0; j < bitmask.size(); j++) {
995  if (bitmask[j]) {
996  int32_t sampaOnFEC = 0, channelOnSAMPA = 0;
997  mapper.getSampaAndChannelOnFEC(cruid, j, sampaOnFEC, channelOnSAMPA);
998  const auto padSecPos = mapper.padSecPos(cruid, decLink, sampaOnFEC, channelOnSAMPA);
999  const auto& padPos = padSecPos.getPadPos();
1000  outputBuffer.emplace_back(o2::tpc::Digit{cruid, samples[samplePos++] * decodeBitsFactor, (tpccf::Row)padPos.getRow(), (tpccf::Pad)padPos.getPad(), timeBin});
1001  nOutput++;
1002  }
1003  }
1004  }
1005  }
1006 
1007  int32_t hdrMinTimeBin = (int32_t(decHDR->timeOffset) + int32_t(o2::raw::RDHUtils::getHeartBeatOrbit(*rdh) - firstOrbit) * o2::constants::lhc::LHCMaxBunches - triggerBC);
1008  if (triggerBC > 0 && hdrMinTimeBin < 0) {
1009  hdrMinTimeBin = 0;
1010  }
1011  hdrMinTimeBin /= LHCBCPERTIMEBIN;
1012  int32_t hdrMaxTimeBin = hdrMinTimeBin + decHDR->nTimeBinSpan + ((decHDR->flags & TPCZSHDRV2::ZSFlags::nTimeBinSpanBit8) ? 256 : 0);
1013 
1014  if (!extendFailure && nOutput != decHDR->nADCsamples) {
1015  std::ostringstream oss;
1016  oss << "Number of decoded digits " << nOutput << " does not match value from MetaInfo " << decHDR->nADCsamples;
1017  amendPageErrorMessage(oss, rdh, decHDR, nullptr, nullptr, nOutput);
1018  throw std::runtime_error(oss.str());
1019  }
1020 
1021  if (decHDR->nADCsamples && (minTimeBin < hdrMinTimeBin || maxTimeBin > hdrMaxTimeBin)) {
1022  std::ostringstream oss;
1023  oss << "Incorrect time bin range in MetaInfo, header reports " << hdrMinTimeBin << " - " << hdrMaxTimeBin << "(timeOffset: " << decHDR->timeOffset << " + (orbit: " << o2::raw::RDHUtils::getHeartBeatOrbit(*rdh) << " - firstOrbit " << firstOrbit << ") * LHCMaxBunches - triggerBC: " << triggerBC << ", decoded data is " << minTimeBin << " - " << maxTimeBin;
1024  amendPageErrorMessage(oss, rdh, decHDR, payloadEnd, decPagePtr, nOutput);
1025  throw std::runtime_error(oss.str());
1026  }
1027 
1028  if (decHDR->nTimebinHeaders && payloadEnd - decPagePtr < 0) {
1029  std::ostringstream oss;
1030  oss << "Decoding ran over end of page";
1031  amendPageErrorMessage(oss, rdh, decHDR, payloadEnd, decPagePtr, nOutput);
1032  throw std::runtime_error(oss.str());
1033  }
1034  if (decHDR->nTimebinHeaders && payloadEnd - decPagePtr >= 2 * o2::raw::RDHUtils::GBTWord128) {
1035  std::ostringstream oss;
1036  oss << "Decoding didn't reach end of page";
1037  amendPageErrorMessage(oss, rdh, decHDR, payloadEnd, decPagePtr, nOutput);
1038  throw std::runtime_error(oss.str());
1039  }
1040 }

1118  int32_t rawendpoint = 0;

1120  encodeBitsFactor = (1 << (encodeBits - 10));
1121

1125  if (needAnotherPage) {
1126  needAnotherPage = false;
1127  mustWritePage = true;
1128  } else {
1129  if (k < tmpBuffer.size()) {
1130  if (tmpBuffer[k].getTimeStamp() != lastTime) {

1199  } else
1200 #endif

1206  o2::raw::RDHUtils::setVersion(*rdh, o2::raw::RDHUtils::getVersion<o2::header::RAWDataHeader>());

1212  }
1213  }
1214  if (k >= tmpBuffer.size() && !needAnotherPage) {
1215  break;
1216  }
1217  }
1218  if (mustWritePage) {
1219  if (!needAnotherPage) {
1220  if (hbf != nexthbf) {
1221  pageCounter = 0;
1222  }
1223  outputRegion = curRegion;
1224  outputEndpoint = endpoint;
1225  hbf = nexthbf;

1228  }

1231  } else {

1237  }

1239  }

1242  }

1250  pagePtr += sizeof(*hdr);

1254  hdr->nTimeBinSpan = 0;
1255  hdr->nADCsamples = 0;

1257  rawendpoint = outputEndpoint & 1;

1261  totalPages++;
1262  }
1263  if (needAnotherPage) {
1264  continue;
1265  }

1267  lastTime = tmpBuffer[k].getTimeStamp();
1268  lastRow = tmpBuffer[k].getRow();
1269  hdr->nADCsamples += nEncoded;
1270  k += nEncoded;
1271  }
1272  if (raw) {
1273 #ifdef GPUCA_O2_LIB
1274  if (iSector == 0) {
1275  for (int32_t i = minhbf; i <= maxhbf; i++) {
1276  raw->addData(46208, 360, rdh_utils::SACLinkID, 0, *ir + i * o2::constants::lhc::LHCMaxBunches, gsl::span<char>((char*)&singleBuffer, (char*)&singleBuffer), true, 0, 4);
1277  }
1278  }
1279 #endif
1280  } else {
1281  for (uint32_t j = 0; j < GPUTrackingInOutZS::NENDPOINTS; j++) {
1282  if (buffer[j].size() == 0) {
1283  buffer[j].emplace_back();
1284  ZSfillEmpty(&buffer[j].back(), bcShiftInFirstHBF, rdh_utils::getFEEID(iSector * 10 + j / 2, j & 1, this->RAWLNK), orbitShift, this->RAWLNK);
1285  totalPages++;
1286  }
1287  }
1288  }
1289  return totalPages;
