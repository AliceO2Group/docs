54 void TrackerDPL::init(InitContext& /*ic*/)
55 {
56  mTimer.Stop();
57  mTimer.Reset();
58 
59  o2::base::GRPGeomHelper::instance().setRequest(mGGCCDBRequest);
60 
61  mChainITS.reset(mRecChain->AddChain<o2::gpu::GPUChainITS>());
62  mVertexer = std::make_unique<Vertexer>(mChainITS->GetITSVertexerTraits());
63  mTracker = std::make_unique<Tracker>(mChainITS->GetITSTrackerTraits());
64  mRunVertexer = true;
65  mCosmicsProcessing = false;
66  std::vector<TrackingParameters> trackParams;
67 
68  if (mMode == "async") {
69 
70  trackParams.resize(3);
71  trackParams[1].TrackletMinPt = 0.2f;
72  trackParams[1].CellDeltaTanLambdaSigma *= 2.;
73  trackParams[2].TrackletMinPt = 0.1f;
74  trackParams[2].CellDeltaTanLambdaSigma *= 4.;
75  trackParams[2].MinTrackLength = 4;
76  LOG(info) << "Initializing tracker in async. phase reconstruction with " << trackParams.size() << " passes";
77 
78  } else if (mMode == "sync_misaligned") {
79 
80  trackParams.resize(3);
81  trackParams[0].PhiBins = 32;
82  trackParams[0].ZBins = 64;
83  trackParams[0].CellDeltaTanLambdaSigma *= 3.;
84  trackParams[0].SystErrorZ2[0] = 1.e-4;
85  trackParams[0].SystErrorZ2[1] = 1.e-4;
86  trackParams[0].SystErrorZ2[2] = 1.e-4;
87  trackParams[0].SystErrorZ2[3] = 9.e-4;
88  trackParams[0].SystErrorZ2[4] = 9.e-4;
89  trackParams[0].SystErrorZ2[5] = 9.e-4;
90  trackParams[0].SystErrorZ2[6] = 9.e-4;
91  std::copy(trackParams[0].SystErrorZ2.begin(), trackParams[0].SystErrorZ2.end(), trackParams[0].SystErrorY2.begin());
92  trackParams[0].MaxChi2ClusterAttachment = 60.;
93  trackParams[0].MaxChi2NDF = 40.;
94  trackParams[1] = trackParams[0];
95  trackParams[2] = trackParams[0];
96  trackParams[1].MinTrackLength = 6;
97  trackParams[2].MinTrackLength = 4;
98  LOG(info) << "Initializing tracker in misaligned sync. phase reconstruction with " << trackParams.size() << " passes";
99 
100  } else if (mMode == "sync") {
101  trackParams.resize(1);
102  LOG(info) << "Initializing tracker in sync. phase reconstruction with " << trackParams.size() << " passes";
103  } else if (mMode == "cosmics") {
104  mCosmicsProcessing = true;
105  mRunVertexer = false;
106  trackParams.resize(1);
107  trackParams[0].MinTrackLength = 4;
108  trackParams[0].CellDeltaTanLambdaSigma *= 10;
109  trackParams[0].PhiBins = 4;
110  trackParams[0].ZBins = 16;
111  trackParams[0].PVres = 1.e5f;
112  trackParams[0].MaxChi2ClusterAttachment = 60.;
113  trackParams[0].MaxChi2NDF = 40.;
114  trackParams[0].TrackletsPerClusterLimit = 100.;
115  trackParams[0].CellsPerClusterLimit = 100.;
116  LOG(info) << "Initializing tracker in reconstruction for cosmics with " << trackParams.size() << " passes";
117 
118  } else {
119  throw std::runtime_error(fmt::format("Unsupported ITS tracking mode {:s} ", mMode));
120  }
121 
122  for (auto& params : trackParams) {
123  params.CorrType = o2::base::PropagatorImpl<float>::MatCorrType::USEMatCorrLUT;
124  for (int iLayer{0}; iLayer < params.NLayers - 4; ++iLayer) { // initialise ITS3 radii and lengths
125  params.LayerZ[iLayer] = constants::segment::lengthSensitive;
126  params.LayerRadii[iLayer] = constants::radii[iLayer];
127  }
128  }
129 
130  mTracker->setParameters(trackParams);
131 }

133 void TrackerDPL::run(ProcessingContext& pc)
134 {
135  mTimer.Start(false);
136  updateTimeDependentParams(pc);
137  auto compClusters = pc.inputs().get<gsl::span<o2::itsmft::CompClusterExt>>("compClusters");
138  gsl::span<const unsigned char> patterns = pc.inputs().get<gsl::span<unsigned char>>("patterns");
139  gsl::span<const o2::itsmft::PhysTrigger> physTriggers;
140  std::vector<o2::itsmft::PhysTrigger> fromTRD;
141  if (mUseTriggers == 2) { // use TRD triggers
142  o2::InteractionRecord ir{0, pc.services().get<o2::framework::TimingInfo>().firstTForbit};
143  auto trdTriggers = pc.inputs().get<gsl::span<o2::trd::TriggerRecord>>("phystrig");
144  for (const auto& trig : trdTriggers) {
145  if (trig.getBCData() >= ir && trig.getNumberOfTracklets()) {
146  ir = trig.getBCData();
147  fromTRD.emplace_back(o2::itsmft::PhysTrigger{ir, 0});
148  }
149  }
150  physTriggers = gsl::span<const o2::itsmft::PhysTrigger>(fromTRD.data(), fromTRD.size());
151  } else if (mUseTriggers == 1) { // use Phys triggers from ITS stream
152  physTriggers = pc.inputs().get<gsl::span<o2::itsmft::PhysTrigger>>("phystrig");
153  }
154 
155  // code further down does assignment to the rofs and the altered object is used for output
156  // we therefore need a copy of the vector rather than an object created directly on the input data,
157  // the output vector however is created directly inside the message memory thus avoiding copy by
158  // snapshot
159  auto orig = o2::header::gDataOriginITS;
160  auto rofsinput = pc.inputs().get<gsl::span<o2::itsmft::ROFRecord>>("ROframes");
161  auto& rofs = pc.outputs().make<std::vector<o2::itsmft::ROFRecord>>(Output{orig, "ITSTrackROF", 0}, rofsinput.begin(), rofsinput.end());
162 
163  auto& irFrames = pc.outputs().make<std::vector<o2::dataformats::IRFrame>>(Output{orig, "IRFRAMES", 0});
164 
165  const auto& alpParams = o2::itsmft::DPLAlpideParam<o2::detectors::DetID::ITS>::Instance(); // RS: this should come from CCDB
166  int nBCPerTF = alpParams.roFrameLengthInBC;
167 
168  LOG(info) << "ITS3Tracker pulled " << compClusters.size() << " clusters, " << rofs.size() << " RO frames";
169 
170  const dataformats::MCTruthContainer<MCCompLabel>* labels = nullptr;
171  gsl::span<itsmft::MC2ROFRecord const> mc2rofs;
172  if (mIsMC) {
173  labels = pc.inputs().get<const dataformats::MCTruthContainer<MCCompLabel>*>("labels").release();
174  // get the array as read-only span, a snapshot is send forward
175  mc2rofs = pc.inputs().get<gsl::span<itsmft::MC2ROFRecord>>("MC2ROframes");
176  LOG(info) << labels->getIndexedSize() << " MC label objects , in " << mc2rofs.size() << " MC events";
177  }
178 
179  auto& allClusIdx = pc.outputs().make<std::vector<int>>(Output{orig, "TRACKCLSID", 0});
180  std::vector<o2::MCCompLabel> trackLabels;
181  std::vector<MCCompLabel> verticesLabels;
182  auto& allTracks = pc.outputs().make<std::vector<o2::its::TrackITS>>(Output{orig, "TRACKS", 0});
183  std::vector<o2::MCCompLabel> allTrackLabels;
184  std::vector<o2::MCCompLabel> allVerticesLabels;
185  std::vector<float> allVerticesPurities;
186 
187  auto& vertROFvec = pc.outputs().make<std::vector<o2::itsmft::ROFRecord>>(Output{orig, "VERTICESROF", 0});
188  auto& vertices = pc.outputs().make<std::vector<Vertex>>(Output{orig, "VERTICES", 0});
189 
190  TimeFrame* timeFrame = mChainITS->GetITSTimeframe();
191  timeFrame->resizeVectors(7);
192  mTracker->adoptTimeFrame(*timeFrame);
193 
194  mTracker->setBz(o2::base::Propagator::Instance()->getNominalBz());
195  mVertexer->adoptTimeFrame(*timeFrame);
196  gsl::span<const unsigned char>::iterator pattIt = patterns.begin();
197 
198  gsl::span<itsmft::ROFRecord> rofspan(rofs);
199  ioutils::loadROFrameDataITS3(timeFrame, rofspan, compClusters, pattIt, mDict, labels);
200  pattIt = patterns.begin();
201  std::vector<int> savedROF;
202  auto logger = [&](const std::string& s) { LOG(info) << s; };
203  auto errorLogger = [&](const std::string& s) { LOG(error) << s; };
204 
205  // o2::its3::FastMultEst multEst; // mult estimator
206  std::vector<bool> processingMask(rofs.size(), true); // Override mult estimator
207  int cutVertexMult{0};
208  // int cutRandomMult = int(rofs.size()) - multEst.selectROFs(rofs, compClusters, physTriggers, processingMask);
209  timeFrame->setMultiplicityCutMask(processingMask);
210  float vertexerElapsedTime{0.f};
211  if (mRunVertexer) {
212  // Run seeding vertexer
213  vertexerElapsedTime = mVertexer->clustersToVertices(logger);
214  }
215  const auto& multEstConf = FastMultEstConfig::Instance(); // parameters for mult estimation and cuts
216  gsl::span<const std::pair<MCCompLabel, float>> vMCRecInfo;
217  for (size_t iRof{0}; iRof < rofspan.size(); ++iRof) {
218  std::vector<Vertex> vtxVecLoc;
219  auto& vtxROF = vertROFvec.emplace_back(rofspan[iRof]);
220  vtxROF.setFirstEntry(vertices.size());
221  if (mRunVertexer) {
222  auto vtxSpan = timeFrame->getPrimaryVertices(iRof);
223  if (mIsMC) {
224  vMCRecInfo = timeFrame->getPrimaryVerticesMCRecInfo(iRof);
225  }
226  vtxROF.setNEntries(vtxSpan.size());
227  bool selROF = vtxSpan.size() == 0;
228  for (size_t iV{0}; iV < vtxSpan.size(); ++iV) {
229  auto& v = vtxSpan[iV];
230  if (multEstConf.isVtxMultCutRequested() && !multEstConf.isPassingVtxMultCut(v.getNContributors())) {
231  continue; // skip vertex of unwanted multiplicity
232  }
233  selROF = true;
234  vertices.push_back(v);
235  if (mIsMC) {
236  allVerticesLabels.push_back(vMCRecInfo[iV].first);
237  allVerticesPurities.push_back(vMCRecInfo[iV].second);
238  }
239  }
240  if (processingMask[iRof] && !selROF) { // passed selection in clusters and not in vertex multiplicity
241  // LOG(debug) << fmt::format("ROF {} rejected by the vertex multiplicity selection [{},{}]",
242  // iRof,
243  // multEstConf.cutMultVtxLow,
244  // multEstConf.cutMultVtxHigh);
245  processingMask[iRof] = selROF;
246  cutVertexMult++;
247  }
248  } else { // cosmics
249  vtxVecLoc.emplace_back(Vertex());
250  vtxVecLoc.back().setNContributors(1);
251  vtxROF.setNEntries(vtxVecLoc.size());
252  for (auto& v : vtxVecLoc) {
253  vertices.push_back(v);
254  }
255  timeFrame->addPrimaryVertices(vtxVecLoc);
256  }
257  }
258  // LOG(info) << fmt::format(" - rejected {}/{} ROFs: random/mult.sel:{} (seed {}), vtx.sel:{}", cutRandomMult + cutVertexMult, rofspan.size(), cutRandomMult, multEst.lastRandomSeed, cutVertexMult);
259  LOG(info) << fmt::format(" - Vertex seeding total elapsed time: {} ms for {} vertices found in {} ROFs", vertexerElapsedTime, timeFrame->getPrimaryVerticesNum(), rofspan.size());
260  LOG(info) << fmt::format(" - Beam position computed for the TF: {}, {}", timeFrame->getBeamX(), timeFrame->getBeamY());
261 
262  if (mCosmicsProcessing && compClusters.size() > 1500 * rofspan.size()) {
263  LOG(error) << "Cosmics processing was requested with an average detector occupancy exceeding 1.e-7, skipping TF processing.";
264  } else {
265 
266  timeFrame->setMultiplicityCutMask(processingMask);
267  // Run CA tracker
268  mTracker->clustersToTracks(logger, errorLogger);
269  if (timeFrame->hasBogusClusters() != 0) {
270  LOG(warning) << fmt::format(" - The processed timeframe had {} clusters with wild z coordinates, check the dictionaries", timeFrame->hasBogusClusters());
271  }
272 
273  for (unsigned int iROF{0}; iROF < rofs.size(); ++iROF) {
274  auto& rof{rofs[iROF]};
275  auto& tracks = timeFrame->getTracks(iROF);
276  trackLabels = timeFrame->getTracksLabel(iROF);
277  auto number{tracks.size()};
278  auto first{allTracks.size()};
279  rof.setFirstEntry(first);
280  rof.setNEntries(number);
281  if (processingMask[iROF]) {
282  irFrames.emplace_back(rof.getBCData(), rof.getBCData() + nBCPerTF - 1).info = tracks.size();
283  }
284 
285  std::copy(trackLabels.begin(), trackLabels.end(), std::back_inserter(allTrackLabels));
286  // Some conversions that needs to be moved in the tracker internals
287  for (unsigned int iTrk{0}; iTrk < tracks.size(); ++iTrk) {
288  auto& trc{tracks[iTrk]};
289  trc.setFirstClusterEntry(allClusIdx.size()); // before adding tracks, create final cluster indices
290  int nclf = 0, ncl = (int)allClusIdx.size();
291  for (int ic = TrackITSExt::MaxClusters; (ic--) != 0;) { // track internally keeps in->out cluster indices, but we want to store the references as out->in!!!
292  auto clid = trc.getClusterIndex(ic);
293  if (clid >= 0) {
294  trc.setClusterSize(ic, timeFrame->getClusterSize(clid));
295  allClusIdx.push_back(clid);
296  nclf++;
297  }
298  }
299  assert(ncl == nclf);
300  allTracks.emplace_back(trc);
301  }
302  }
303 
304  LOGP(info, "ITS3Tracker pushed {} tracks and {} vertices", allTracks.size(), vertices.size());
305  if (mIsMC) {
306  LOGP(info, "ITS3Tracker pushed {} track labels", allTrackLabels.size());
307  LOGP(info, "ITS3Tracker pushed {} vertex labels", allVerticesLabels.size());
308 
309  pc.outputs().snapshot(Output{orig, "TRACKSMCTR", 0}, allTrackLabels);
310  pc.outputs().snapshot(Output{orig, "VERTICESMCTR", 0}, allVerticesLabels);
311  pc.outputs().snapshot(Output{orig, "VERTICESMCPUR", 0}, allVerticesPurities);
312  pc.outputs().snapshot(Output{orig, "ITSTrackMC2ROF", 0}, mc2rofs);
313  }
314  }
315  mTimer.Stop();
316 }

319 void TrackerDPL::updateTimeDependentParams(ProcessingContext& pc)
320 {
321  o2::base::GRPGeomHelper::instance().checkUpdates(pc);
322  static bool initOnceDone = false;
323  if (!initOnceDone) { // this params need to be queried only once
324  initOnceDone = true;
325  pc.inputs().get<o2::its3::TopologyDictionary*>("cldict"); // just to trigger the finaliseCCDB
326  pc.inputs().get<o2::itsmft::DPLAlpideParam<o2::detectors::DetID::ITS>*>("alppar");
327 
328  // Check if lightweight geometry was requested, otherwise full geometry is loaded
329  if (pc.inputs().getPos("itsTGeo") >= 0) {
330  pc.inputs().get<o2::its::GeometryTGeo*>("itsTGeo");
331  }
332  o2::its::GeometryTGeo* geom = o2::its::GeometryTGeo::Instance();
333  geom->fillMatrixCache(o2::math_utils::bit2Mask(o2::math_utils::TransformType::T2L, o2::math_utils::TransformType::T2GRot, o2::math_utils::TransformType::T2G));
334  mVertexer->getGlobalConfiguration();
335  mTracker->getGlobalConfiguration();
336  }

340 void TrackerDPL::finaliseCCDB(ConcreteDataMatcher& matcher, void* obj)
341 {
342  if (o2::base::GRPGeomHelper::instance().finaliseCCDB(matcher, obj)) {
343  return;
344  }
345  if (matcher == ConcreteDataMatcher("IT3", "CLUSDICT", 0)) {
346  LOG(info) << "cluster dictionary updated";
347  setClusterDictionary((o2::its3::TopologyDictionary*)obj);
348  return;
349  }
350  if (matcher == ConcreteDataMatcher("ITS", "GEOMTGEO", 0)) {
351  LOG(info) << "IT3 GeometryTGeo loaded from ccdb";
352  o2::its::GeometryTGeo::adopt((o2::its::GeometryTGeo*)obj);
353  return;
354  }
355  // Note: strictly speaking, for Configurable params we don't need finaliseCCDB check, the singletons are updated at the CCDB fetcher level
356  if (matcher == ConcreteDataMatcher("ITS", "ALPIDEPARAM", 0)) {
357  LOG(info) << "Alpide param updated";
358  const auto& par = o2::itsmft::DPLAlpideParam<o2::detectors::DetID::ITS>::Instance();
359  par.printKeyValues();
360  return;
361  }

370 DataProcessorSpec getTrackerSpec(bool useMC, bool useGeom, const int trgType, const std::string& trModeS, o2::gpu::GPUDataTypes::DeviceType dType)
371 {
372  std::vector<InputSpec> inputs;
373  inputs.emplace_back("compClusters", "ITS", "COMPCLUSTERS", 0, Lifetime::Timeframe);
374  inputs.emplace_back("patterns", "ITS", "PATTERNS", 0, Lifetime::Timeframe);
375  inputs.emplace_back("ROframes", "ITS", "CLUSTERSROF", 0, Lifetime::Timeframe);
376  if (trgType == 1) {
377  inputs.emplace_back("phystrig", "ITS", "PHYSTRIG", 0, Lifetime::Timeframe);
378  } else if (trgType == 2) {
379  inputs.emplace_back("phystrig", "TRD", "TRKTRGRD", 0, Lifetime::Timeframe);
380  }
381  inputs.emplace_back("cldict", "IT3", "CLUSDICT", 0, Lifetime::Condition, ccdbParamSpec("IT3/Calib/ClusterDictionary"));
382  inputs.emplace_back("alppar", "ITS", "ALPIDEPARAM", 0, Lifetime::Condition, ccdbParamSpec("ITS/Config/AlpideParam"));
383  auto ggRequest = std::make_shared<o2::base::GRPGeomRequest>(false, // orbitResetTime
384  false, // GRPECS
385  false, // GRPLHCIF
386  true, // GRPMagField
387  true, // askMatLUT
388  useGeom ? o2::base::GRPGeomRequest::Aligned : o2::base::GRPGeomRequest::None, // geometry
389  inputs,
390  true);
391 
392  if (!useGeom) { // load light-weight geometry
393  inputs.emplace_back("itsTGeo", "ITS", "GEOMTGEO", 0, Lifetime::Condition, ccdbParamSpec("ITS/Config/Geometry"));
394  }
395 
396  std::vector<OutputSpec> outputs;
397  outputs.emplace_back("ITS", "TRACKS", 0, Lifetime::Timeframe);
398  outputs.emplace_back("ITS", "TRACKCLSID", 0, Lifetime::Timeframe);
399  outputs.emplace_back("ITS", "ITSTrackROF", 0, Lifetime::Timeframe);
400  outputs.emplace_back("ITS", "VERTICES", 0, Lifetime::Timeframe);
401  outputs.emplace_back("ITS", "VERTICESROF", 0, Lifetime::Timeframe);
402  outputs.emplace_back("ITS", "IRFRAMES", 0, Lifetime::Timeframe);
403 
404  if (useMC) {
405  inputs.emplace_back("labels", "ITS", "CLUSTERSMCTR", 0, Lifetime::Timeframe);
406  inputs.emplace_back("MC2ROframes", "ITS", "CLUSTERSMC2ROF", 0, Lifetime::Timeframe);
407  outputs.emplace_back("ITS", "VERTICESMCTR", 0, Lifetime::Timeframe);
408  outputs.emplace_back("ITS", "TRACKSMCTR", 0, Lifetime::Timeframe);
409  outputs.emplace_back("ITS", "ITSTrackMC2ROF", 0, Lifetime::Timeframe);
410  }
411 
412  return DataProcessorSpec{
413  "its3-tracker",
414  inputs,
415  outputs,
416  AlgorithmSpec{adaptFromTask<TrackerDPL>(ggRequest, useMC, trgType, trModeS, dType)},
417  Options{}};
418 }
