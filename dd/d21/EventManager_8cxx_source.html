81 void EventManager::displayCurrentEvent()
82 {
83  auto start = std::chrono::high_resolution_clock::now();
84  const auto multiView = MultiView::getInstance();
85  const auto dataSource = getDataSource();
86  if (dataSource->getEventCount() > 0) {
87  if (!vizSettings.firstEvent) {
88  saveVisualisationSettings();
89  }
90 
91  multiView->destroyAllEvents();
92  int no = dataSource->getCurrentEvent();
93 
94  for (int i = 0; i < EVisualisationDataType::NdataTypes; ++i) {
95  dataTypeLists[i] = new TEveElementList(gDataTypeNames[i].c_str());
96  dataTypeListsPhi[i] = new TEveElementList(gDataTypeNames[i].c_str());
97  }
98 
99  VisualisationEvent event; // collect calorimeters in one drawing step
100  auto displayList = dataSource->getVisualisationList(no,
101  EventManagerFrame::getInstance().getMinTimeFrameSliderValue(),
102  EventManagerFrame::getInstance().getMaxTimeFrameSliderValue(),
103  EventManagerFrame::MaxRange);
104 
105  for (auto it = displayList.begin(); it != displayList.end(); ++it) {
106  if (it->second == EVisualisationGroup::EMC || it->second == EVisualisationGroup::PHS) {
107  displayCalorimeters(it->first, gVisualisationGroupName[it->second]);
108  } else {
109  displayVisualisationEvent(it->first, gVisualisationGroupName[it->second]);
110  }
111  }
112  multiView->registerElements(dataTypeLists, dataTypeListsPhi);
113 
114  if (vizSettings.firstEvent) {
115  ifstream s(TEMP_SETTINGS_PATH);
116  if (s.good()) {
117  restoreVisualisationSettings();
118  } else {
119  saveVisualisationSettings();
120  }
121  vizSettings.firstEvent = false;
122  } else {
123  restoreVisualisationSettings();
124  }
125 
126  if (dataSource->getRunNumber() != -1) {
127  if (this->mShowDate) {
128  multiView->getAnnotationTop()->SetText(
129  TString::Format("Run %d %s\n%s", dataSource->getRunNumber(),
130  std::string(parameters::GRPECS::RunTypeNames[dataSource->getRunType()]).c_str(),
131  dataSource->getFileTime().c_str()));
132  } else {
133  multiView->getAnnotationTop()->SetText(TString::Format("Run %d", dataSource->getRunNumber()));
134  }
135  auto detectors = detectors::DetID::getNames(dataSource->getDetectorsMask());
136  multiView->getAnnotationBottom()->SetText(
137  TString::Format("TFOrbit: %d\nDetectors: %s", dataSource->getFirstTForbit(), detectors.c_str()));
138  } else {
139  multiView->getAnnotationTop()->SetText("No Available Data to Display");
140  }
141  }
142  multiView->redraw3D();
143  auto stop = std::chrono::high_resolution_clock::now();
144  auto duration = std::chrono::duration_cast<std::chrono::microseconds>(stop - start);
145 }

220 void EventManager::displayVisualisationEvent(VisualisationEvent& event, const std::string& detectorName)
221 {
222  double eta = 0.1;
223  size_t trackCount = event.getTrackCount();
224 
225  // tracks
226  auto* list = new TEveTrackList(detectorName.c_str());
227  list->IncDenyDestroy();
228  // clusters
229  size_t clusterCount = 0;
230  auto* point_list = new TEvePointSet(detectorName.c_str());
231  point_list->IncDenyDestroy(); // don't delete if zero parent
232  point_list->SetMarkerColor(settings.GetValue("clusters.byType.unknown", kBlue));
233 
234  for (size_t i = 0; i < trackCount; ++i) {
235  VisualisationTrack track = event.getTrack(i);
236  TEveRecTrackD t;
237  t.fSign = track.getCharge() > 0 ? 1 : -1;
238  auto* vistrack = new TEveTrack(&t, &TEveTrackPropagator::fgDefault);
239  vistrack->SetLineColor(settings.GetValue("tracks.byType.unknown", kMagenta));
240  vistrack->SetName(track.getGIDAsString().c_str());
241  size_t pointCount = track.getPointCount();
242  vistrack->Reset(pointCount);
243 
244  int points = 0;
245  for (size_t j = 0; j < pointCount; ++j) {
246  auto point = track.getPoint(j);
247  if (point[2] > eta || point[2] < -1 * eta) {
248  vistrack->SetNextPoint(point[0], point[1], point[2]);
249  points++;
250  }
251  }
252  if (points > 0) {
253  list->AddElement(vistrack);
254  }
255 
256  // clusters connected with track
257  for (size_t i = 0; i < track.getClusterCount(); ++i) {
258  VisualisationCluster cluster = track.getCluster(i);
259  if (cluster.Z() > eta || cluster.Z() < -1 * eta) { // temporary remove eta=0 artefacts
260  point_list->SetNextPoint(cluster.X(), cluster.Y(), cluster.Z());
261  clusterCount++;
262  }
263  }
264  }
265 
266  if (trackCount != 0) {
267  dataTypeLists[EVisualisationDataType::Tracks]->AddElement(list);
268  if (detectorName != "MCH" && detectorName != "MFT" && detectorName != "MID") {
269  dataTypeListsPhi[EVisualisationDataType::Tracks]->AddElement(list);
270  }
271  }
272 
273  // global clusters (with no connection information)
274  for (size_t i = 0; i < event.getClusterCount(); ++i) {
275  VisualisationCluster cluster = event.getCluster(i);
276  if (cluster.Z() > eta || cluster.Z() < -1 * eta) { // temporary remove eta=0 artefacts
277  point_list->SetNextPoint(cluster.X(), cluster.Y(), cluster.Z());
278  clusterCount++;
279  }
280  }
281 
282  if (clusterCount != 0) {
283  dataTypeLists[EVisualisationDataType::Clusters]->AddElement(point_list);
284  if (detectorName != "MCH" && detectorName != "MFT" && detectorName != "MID") {
285  dataTypeListsPhi[EVisualisationDataType::Clusters]->AddElement(point_list);
286  }
287  }
288 
289  LOGF(info, "tracks: ", trackCount, " detector: ", detectorName, ":",
290  dataTypeLists[EVisualisationDataType::Tracks]->NumChildren());
291  LOGF(info, "clusters: ", clusterCount, " detector: ", detectorName, ":",
292  dataTypeLists[EVisualisationDataType::Clusters]->NumChildren());
293 }

295 void EventManager::displayCalorimeters(VisualisationEvent& event, const std::string& detectorName)
296 {
297  if (event.getCaloCount() > 0) {
298  struct CaloInfo {
299  std::string name;
300  std::string configColor;
301  int defaultColor;
302  std::string configSizeEta;
303  float sizeEta;
304  std::string configSizePhi;
305  float sizePhi;
306  std::string configNoise;
307  float defaultNoise;
308  std::string configTransparency;
309  int defaultTransparency;
310  std::string configBarrelRadius;
311  int defaultBarrelRadius;
312  std::string configTowerMaxHeight;
313  float defaultTowerMaxHeight;
314  std::string configMaxValAbs;
315  float defaultMaxValAbs;
316  };
317 
318  // TODO: calculate values based on info available in O2
319  static const std::unordered_map<o2::dataformats::GlobalTrackID::Source, CaloInfo> caloInfos =
320  {
321  {o2::dataformats::GlobalTrackID::EMC, {"emcal", "emcal.tower.color", kYellow, "emcal.tower.size.eta", 0.0143, "emcal.tower.size.phi", 0.0143, "emcal.tower.noise", 0, "emcal.tower.transparency", 101, "emcal.barrel.radius", 500, "emcal.tower.max.height", 80, "emcal.tower.max.val.abs", 100}},
322  {o2::dataformats::GlobalTrackID::PHS, {"phos", "phos.tower.color", kYellow, "phos.tower.size.eta", 0.0046, "phos.tower.size.phi", 0.00478, "phos.tower.noise", 200, "phos.tower.transparency", 101, "phos.barrel.radius", 550, "phos.tower.max.height", 80, "phos.tower.max.val.abs", 100}},
323  };
324 
325  auto data = new TEveCaloDataVec(1);
326  data->IncDenyDestroy();
327  auto key = detectorName == "EMC" ? o2::dataformats::GlobalTrackID::EMC : o2::dataformats::GlobalTrackID::PHS;
328 
329  const CaloInfo& info = caloInfos.at(key);
330  data->RefSliceInfo(0).Setup(info.name.c_str(),
331  settings.GetValue(info.configNoise.c_str(), info.defaultNoise),
332  settings.GetValue(info.configColor.c_str(), info.defaultColor),
333  settings.GetValue(info.configTransparency.c_str(), info.defaultTransparency));
334 
335  const auto dEta = settings.GetValue(info.configSizeEta.c_str(), info.sizeEta) / 2.0;
336  const auto dPhi = settings.GetValue(info.configSizePhi.c_str(), info.sizePhi) / 2.0;
337  const float barrelRadius = settings.GetValue(info.configBarrelRadius.c_str(), info.defaultBarrelRadius);
338 
339  struct pair_hash {
340  std::size_t operator()(const std::pair<float, float>& pair) const
341  {
342  return std::hash<float>()(pair.first + 1000.0 * pair.second);
343  }
344  };
345  std::unordered_map<std::pair<float, float>, float, pair_hash> map; // sum up entries for the same tower
346  for (const auto& calo : event.getCalorimetersSpan()) {
347  map[std::make_pair(calo.getEta(), calo.getPhi())] += calo.getEnergy();
348  }
349 
350  for (const auto& entry : map) {
351  auto [eta, phi] = entry.first;
352  data->AddTower(eta - dEta, eta + dEta, phi - dPhi, phi + dPhi);
353  data->FillSlice(0, entry.second);
354  }
355 
356  // remove artefacts
357  data->AddTower(-0.5, 0.5, -1.574 - 0.1, -1.574 + 0.1);
358  data->AddTower(-0.5, 0.5, 1.574 - 0.1, 1.574 + 0.1);
359  data->AddTower(-0.5, 0.5, -0.593 - 0.1, -0.593 + 0.1);
360  data->AddTower(-0.5, 0.5, -0.726 - 0.1, -0.726 + 0.1);
361  data->AddTower(-0.5, 0.5, -3.028 - 0.1, -3.028 + 0.1);
362  data->AddTower(-0.5, 0.5, -1.915 - 0.1, -1.915 + 0.1);
363 
364  data->DataChanged();
365  data->SetAxisFromBins();
366 
367  auto calo3d = new TEveCalo3D(data);
368  calo3d->SetName(detectorName.c_str());
369  calo3d->SetScaleAbs(kTRUE);
370  calo3d->SetMaxTowerH(settings.GetValue(info.configTowerMaxHeight.c_str(), info.defaultTowerMaxHeight));
371  calo3d->SetMaxValAbs(settings.GetValue(info.configMaxValAbs.c_str(), info.defaultMaxValAbs));
372  // calo3d->SetAutoRange(kTRUE);
373 
374  calo3d->SetBarrelRadius(barrelRadius);
375  calo3d->SetEndCapPos(barrelRadius);
376  calo3d->SetRnrFrame(false, false); // do not draw barrel grid
377 
378  dataTypeLists[EVisualisationDataType::Calorimeters]->AddElement(calo3d);
379  dataTypeListsPhi[EVisualisationDataType::Calorimeters]->AddElement(calo3d);
380  }

383 void EventManager::saveVisualisationSettings()
384 {
385  const auto& tracks = *dataTypeLists[EVisualisationDataType::Tracks];
386 
387  for (auto elm : tracks.RefChildren()) {
388  auto trackList = static_cast<TEveTrackList*>(elm);
389  int i = findGroupIndex(trackList->GetElementName());
390 
391  if (i != -1) {
392  vizSettings.trackVisibility[i] = trackList->GetRnrSelf();
393  vizSettings.trackColor[i] = trackList->GetLineColor();
394  vizSettings.trackStyle[i] = trackList->GetLineStyle();
395  vizSettings.trackWidth[i] = trackList->GetLineWidth();
396  }
397  }
398 
399  const auto& clusters = *dataTypeLists[EVisualisationDataType::Clusters];
400 
401  for (auto elm : clusters.RefChildren()) {
402  auto clusterSet = static_cast<TEvePointSet*>(elm);
403  int i = findGroupIndex(clusterSet->GetElementName());
404 
405  if (i != -1) {
406  vizSettings.clusterVisibility[i] = clusterSet->GetRnrSelf();
407  vizSettings.clusterColor[i] = clusterSet->GetMarkerColor();
408  vizSettings.clusterStyle[i] = clusterSet->GetMarkerStyle();
409  vizSettings.clusterSize[i] = clusterSet->GetMarkerSize();
410  }
411  }
412 
413  ofstream settings(TEMP_SETTINGS_PATH);
414 
415  if (settings.good()) {
416  Document d;
417  d.SetObject();
418  auto& allocator = d.GetAllocator();
419 
420  auto jsonArray = [](const auto& array, auto& allocator) {
421  Value arr(kArrayType);
422 
423  for (const auto& value : array) {
424  arr.PushBack(value, allocator);
425  }
426 
427  return arr;
428  };
429  d.AddMember("version", rapidjson::Value().SetString(o2_eve_version.c_str(), o2_eve_version.length()), allocator);
430  d.AddMember("trackVisibility", jsonArray(vizSettings.trackVisibility, allocator), allocator);
431  d.AddMember("trackColor", jsonArray(vizSettings.trackColor, allocator), allocator);
432  d.AddMember("trackStyle", jsonArray(vizSettings.trackStyle, allocator), allocator);
433  d.AddMember("trackWidth", jsonArray(vizSettings.trackWidth, allocator), allocator);
434  d.AddMember("clusterVisibility", jsonArray(vizSettings.clusterVisibility, allocator), allocator);
435  d.AddMember("clusterColor", jsonArray(vizSettings.clusterColor, allocator), allocator);
436  d.AddMember("clusterStyle", jsonArray(vizSettings.clusterStyle, allocator), allocator);
437  d.AddMember("clusterSize", jsonArray(vizSettings.clusterSize, allocator), allocator);
438 
439  auto jsonCamera = [&jsonArray](MultiView::EViews view, auto& allocator) {
440  Value obj(kObjectType);
441 
442  auto& camera = MultiView::getInstance()->getView(view)->GetGLViewer()->CurrentCamera();
443 
444  const gsl::span baseSpan(camera.RefCamBase().CArr(), 16);
445  obj.AddMember("base", jsonArray(baseSpan, allocator), allocator);
446 
447  const gsl::span transSpan(camera.GetCamTrans().CArr(), 16);
448  obj.AddMember("trans", jsonArray(transSpan, allocator), allocator);
449 
450  if (camera.IsOrthographic()) {
451  obj.AddMember("zoom", dynamic_cast<TGLOrthoCamera&>(camera).GetZoom(), allocator);
452  } else if (camera.IsPerspective()) {
453  obj.AddMember("fov", dynamic_cast<TGLPerspectiveCamera&>(camera).GetFOV(), allocator);
454  }
455 
456  return obj;
457  };
458 
459  d.AddMember("camera3d", jsonCamera(MultiView::View3d, allocator), allocator);
460  d.AddMember("cameraRphi", jsonCamera(MultiView::ViewRphi, allocator), allocator);
461  d.AddMember("cameraZY", jsonCamera(MultiView::ViewZY, allocator), allocator);
462 
463  StringBuffer strbuf;
464  Writer<StringBuffer> writer(strbuf);
465  d.Accept(writer);
466 
467  settings << strbuf.GetString();

471 void EventManager::restoreVisualisationSettings()
472 {
473  ifstream settings(TEMP_SETTINGS_PATH);
474 
475  if (settings.good()) {
476  string json((istreambuf_iterator<char>(settings)), istreambuf_iterator<char>());
477  Document d;
478  d.Parse(json.c_str());
479 
480  if (VisualisationEventJSONSerializer::getStringOrDefault(d, "version", "0.0") != o2_eve_version) {
481  LOGF(info, "visualisation settings has wrong version and was not restored");
482  return;
483  }

485  auto updateArray = [](auto& array, const auto& document, const char* name, const auto& accessor) {
486  for (size_t i = 0; i < elemof(array); ++i) {
487  array[i] = accessor(document[name][i]);
488  }
489  };
490 
491  auto getBool = [](const GenericValue<UTF8<char>>& v) { return v.GetBool(); };
492  auto getUint = [](const GenericValue<UTF8<char>>& v) { return v.GetUint(); };
493  auto getFloat = [](const GenericValue<UTF8<char>>& v) { return v.GetFloat(); };
494 
495  updateArray(vizSettings.trackVisibility, d, "trackVisibility", getBool);
496  updateArray(vizSettings.trackColor, d, "trackColor", getUint);
497  updateArray(vizSettings.trackStyle, d, "trackStyle", getUint);
498  updateArray(vizSettings.trackWidth, d, "trackWidth", getUint);
499  updateArray(vizSettings.clusterVisibility, d, "clusterVisibility", getBool);
500  updateArray(vizSettings.clusterColor, d, "clusterColor", getUint);
501  updateArray(vizSettings.clusterStyle, d, "clusterStyle", getUint);
502  updateArray(vizSettings.clusterSize, d, "clusterSize", getFloat);

504  auto updateCamera = [getFloat](MultiView::EViews view, const auto& document, const char* name) {
505  auto& camera = MultiView::getInstance()->getView(view)->GetGLViewer()->CurrentCamera();
506 
507  std::array<Double_t, 16> values;
508 
509  for (size_t i = 0; i < values.size(); ++i) {
510  values[i] = getFloat(document[name]["base"][i]);
511  }
512  camera.RefCamBase() = TGLMatrix(values.data());
513 
514  for (size_t i = 0; i < values.size(); ++i) {
515  values[i] = getFloat(document[name]["trans"][i]);
516  }
517  camera.RefCamTrans() = TGLMatrix(values.data());
518 
519  if (camera.IsOrthographic()) {
520  dynamic_cast<TGLOrthoCamera&>(camera).SetZoom(getFloat(document[name]["zoom"]));
521  } else if (camera.IsPerspective()) {
522  dynamic_cast<TGLPerspectiveCamera&>(camera).SetFOV(getFloat(document[name]["fov"]));
523  }
524 
525  camera.IncTimeStamp();
526  };
527 
528  updateCamera(MultiView::View3d, d, "camera3d");
529  updateCamera(MultiView::ViewRphi, d, "cameraRphi");
530  updateCamera(MultiView::ViewZY, d, "cameraZY");
531  }
532 
533  const auto& tracks = *dataTypeLists[EVisualisationDataType::Tracks];
534 
535  for (auto elm : tracks.RefChildren()) {
536  auto trackList = static_cast<TEveTrackList*>(elm);
537  int i = findGroupIndex(trackList->GetElementName());
538 
539  if (i != -1) {
540  const auto viz = vizSettings.trackVisibility[i];
541  trackList->SetRnrSelfChildren(viz, viz);
542  trackList->SetLineColor(vizSettings.trackColor[i]);
543  trackList->SetLineStyle(vizSettings.trackStyle[i]);
544  trackList->SetLineWidth(vizSettings.trackWidth[i]);
545  }
546  }
547 
548  const auto& clusters = *dataTypeLists[EVisualisationDataType::Clusters];
549 
550  for (auto elm : clusters.RefChildren()) {
551  auto clusterSet = static_cast<TEvePointSet*>(elm);
552  int i = findGroupIndex(clusterSet->GetElementName());
553 
554  if (i != -1) {
555  const auto viz = vizSettings.clusterVisibility[i];
556  clusterSet->SetRnrSelfChildren(viz, viz);
557  clusterSet->SetMarkerColor(vizSettings.clusterColor[i]);
558  clusterSet->SetMarkerStyle(vizSettings.clusterStyle[i]);
559  clusterSet->SetMarkerSize(vizSettings.clusterSize[i]);
560  }
561  }
562 
563  MultiView::getInstance()->redraw3D();
564 }
