76 static RunType string2RunType(const std::string& rts)
77 {
78  int rt = -1;
79  for (int i = 0; i < int(RunType::NRUNTYPES); i++) {
80  if (rts == RunTypeNames[i]) {
81  rt = i;
82  break;
83  }
84  }
85  return RunType(rt);
86 }
