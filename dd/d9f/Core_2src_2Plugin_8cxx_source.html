23 auto lookForAodFile = [](ConfigParamRegistry& registry, int argc, char** argv) -> bool {
24  O2_SIGNPOST_ID_GENERATE(sid, capabilities);
25  if (registry.hasOption("aod-file") && registry.isSet("aod-file")) {
26  for (size_t i = 0; i < argc; i++) {
27  std::string_view arg = argv[i];
28  if (arg.starts_with("--aod-metadata-")) {
29  return false;
30  }
31  }
32  O2_SIGNPOST_EVENT_EMIT(capabilities, sid, "DiscoverMetadataInAODCapability", "Metadata not found in arguments. Checking in AOD file.");
33  return true;
34  }
35  return false;
36 };

71 struct DiscoverMetadataInCommandLine : o2::framework::ConfigDiscoveryPlugin {
72  ConfigDiscovery* create() override
73  {
74  return new ConfigDiscovery{
75  .init = []() {},
76  .discover = [](ConfigParamRegistry& registry, int argc, char** argv) -> std::vector<ConfigParamSpec> {
77  O2_SIGNPOST_ID_GENERATE(sid, capabilities);
78  O2_SIGNPOST_EVENT_EMIT(capabilities, sid, "DiscoverMetadataInCommandLine",
79  "Discovering metadata for analysis from well known environment variables.");
80  std::vector<ConfigParamSpec> results;
81  for (size_t i = 0; i < argc; i++) {
82  std::string_view arg = argv[i];
83  if (!arg.starts_with("--aod-metadata")) {
84  continue;
85  }
86  std::string key = arg.data() + 2;
87  std::string value = argv[i + 1];
88  O2_SIGNPOST_EVENT_EMIT(capabilities, sid, "DiscoverMetadataInCommandLine",
89  "Found %{public}s with value %{public}s.", key.c_str(), value.c_str());
90  if (key == "aod-metadata-tables") {
91  std::stringstream is(value);
92  auto arrayValue = VariantJSONHelpers::read<VariantType::ArrayString>(is);
93  results.push_back(ConfigParamSpec{key, VariantType::ArrayString, arrayValue, {"Metadata in command line"}});
94  } else {
95  results.push_back(ConfigParamSpec{key, VariantType::String, value, {"Metadata in command line"}});
96  }
97  }
98  return results;
99  }};
100  }
101 };
