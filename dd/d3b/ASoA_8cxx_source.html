43 std::shared_ptr<arrow::Table> ArrowHelpers::joinTables(std::vector<std::shared_ptr<arrow::Table>>&& tables)
44 {
45  if (tables.size() == 1) {
46  return tables[0];
47  }
48  for (auto i = 0U; i < tables.size() - 1; ++i) {
49  if (tables[i]->num_rows() != tables[i + 1]->num_rows()) {
50  throw o2::framework::runtime_error_f("Tables %s and %s have different sizes (%d vs %d) and cannot be joined!",
51  tables[i]->schema()->metadata()->Get("label").ValueOrDie().c_str(),
52  tables[i + 1]->schema()->metadata()->Get("label").ValueOrDie().c_str(),
53  tables[i]->num_rows(),
54  tables[i + 1]->num_rows());
55  }
56  }
57  std::vector<std::shared_ptr<arrow::Field>> fields;
58  std::vector<std::shared_ptr<arrow::ChunkedArray>> columns;
59 
60  for (auto& t : tables) {
61  auto tf = t->fields();
62  std::copy(tf.begin(), tf.end(), std::back_inserter(fields));
63  }
64 
65  auto schema = std::make_shared<arrow::Schema>(fields);
66 
67  if (tables[0]->num_rows() != 0) {
68  for (auto& t : tables) {
69  auto tc = t->columns();
70  std::copy(tc.begin(), tc.end(), std::back_inserter(columns));
71  }
72  }
73  return arrow::Table::Make(schema, columns);
74 }

76 std::shared_ptr<arrow::Table> ArrowHelpers::concatTables(std::vector<std::shared_ptr<arrow::Table>>&& tables)
77 {
78  if (tables.size() == 1) {
79  return tables[0];
80  }
81  std::vector<std::shared_ptr<arrow::ChunkedArray>> columns;
82  assert(tables.size() > 1);
83  std::vector<std::shared_ptr<arrow::Field>> resultFields = tables[0]->schema()->fields();
84  auto compareFields = [](std::shared_ptr<arrow::Field> const& f1, std::shared_ptr<arrow::Field> const& f2) {
85  // Let's do this with stable sorting.
86  return (!f1->Equals(f2)) && (f1->name() < f2->name());
87  };
88  for (size_t i = 1; i < tables.size(); ++i) {
89  auto& fields = tables[i]->schema()->fields();
90  std::vector<std::shared_ptr<arrow::Field>> intersection;
91 
92  std::set_intersection(resultFields.begin(), resultFields.end(),
93  fields.begin(), fields.end(),
94  std::back_inserter(intersection), compareFields);
95  resultFields.swap(intersection);
96  }
97 
98  for (auto& field : resultFields) {
99  arrow::ArrayVector chunks;
100  for (auto& table : tables) {
101  auto ci = table->schema()->GetFieldIndex(field->name());
102  if (ci == -1) {
103  throw std::runtime_error("Unable to find field " + field->name());
104  }
105  auto column = table->column(ci);
106  auto otherChunks = column->chunks();
107  chunks.insert(chunks.end(), otherChunks.begin(), otherChunks.end());
108  }
109  columns.push_back(std::make_shared<arrow::ChunkedArray>(chunks));
110  }
111 
112  auto result = arrow::Table::Make(std::make_shared<arrow::Schema>(resultFields), columns);
113  return result;
114 }

153 void sliceByColumnGeneric(
154  char const* key,
155  char const* target,
156  std::shared_ptr<arrow::Table> const& input,
157  int32_t fullSize,
158  ListVector* groups,
159  ListVector* unassigned)
160 {
161  groups->resize(fullSize);
162  auto column = input->GetColumnByName(key);
163  int32_t row = 0;
164  for (auto iChunk = 0; iChunk < column->num_chunks(); ++iChunk) {
165  auto chunk = static_cast<arrow::NumericArray<arrow::Int32Type>>(column->chunk(iChunk)->data());
166  for (auto iElement = 0; iElement < chunk.length(); ++iElement) {
167  auto v = chunk.Value(iElement);
168  if (v >= 0) {
169  if (v >= groups->size()) {
170  throw runtime_error_f("Table %s has an entry with index (%d) that is larger than the grouping table size (%d)", target, v, fullSize);
171  }
172  (*groups)[v].push_back(row);
173  } else if (unassigned != nullptr) {
174  auto av = std::abs(v);
175  if (unassigned->size() < av + 1) {
176  unassigned->resize(av + 1);
177  }
178  (*unassigned)[av].push_back(row);
179  }
180  ++row;
181  }
182  }
183 }
