320 inline gsl::span<const Vertex> TimeFrame::getPrimaryVertices(int rof) const
321 {
322  const int start = mROframesPV[rof];
323  const int stop_idx = rof >= mNrof - 1 ? mNrof : rof + 1;
324  int delta = mMultiplicityCutMask[rof] ? mROframesPV[stop_idx] - start : 0; // return empty span if Rof is excluded
325  return {&mPrimaryVertices[start], static_cast<gsl::span<const Vertex>::size_type>(delta)};
326 }
