342 inline gsl::span<const Vertex> TimeFrame::getPrimaryVertices(int rofId) const
343 {
344  const int start = mROFramesPV[rofId];
345  const int stop_idx = rofId >= mNrof - 1 ? mNrof : rofId + 1;
346  int delta = mMultiplicityCutMask[rofId] ? mROFramesPV[stop_idx] - start : 0; // return empty span if Rof is excluded
347  return {&mPrimaryVertices[start], static_cast<gsl::span<const Vertex>::size_type>(delta)};
348 }
