228consteval size_t has_type_at_v(pack<Ts...>)
229{
230 constexpr size_t size = sizeof...(Ts);
231 constexpr bool found[size] = {std::same_as<T, Ts>...};
232 for (size_t i = 0; i < size; ++i) {
