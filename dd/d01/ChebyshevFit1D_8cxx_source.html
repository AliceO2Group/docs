67void ChebyshevFit1D::fit()
68{
69 for (int32_t i = 0; i < mN; i++) {
70 for (int32_t j = 0; j < i; j++) {
71 mA[i * mN + j] = mA[j * mN + i];
72 }
73 }
74 // print();
75 {
76 double* Ai = mA.data();
77 for (int32_t i = 0; i < mN; i++, Ai += mN) {
78 double a = Ai[i];
79 if (fabs(a) < 1.e-6) {
80 Ai[i] = 0;
81 continue;
82 }
83 double* Aj = Ai + mN;
84 for (int32_t j = i + 1; j < mN; j++, Aj += mN) {
85 double c = Aj[i] / a;
86 for (int32_t k = i + 1; k < mN; k++) {
87 Aj[k] -= c * Ai[k];
88 }
89 mB[j] -= c * mB[i];
90 }
91 // print();
92 }
93 }

99 s -= mC[k] * Ai[k];
