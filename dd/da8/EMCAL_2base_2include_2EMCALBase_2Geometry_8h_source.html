569 Float_t AngleFromEta(Float_t eta) const
570 { // returns theta in radians for a given pseudorapidity
571 return 2.0 * TMath::ATan(TMath::Exp(-eta));
572 }
573 Float_t ZFromEtaR(Float_t r, Float_t eta) const
574 { // returns z in for a given
575 // pseudorapidity and r=sqrt(x*x+y*y).
576 return r / TMath::Tan(AngleFromEta(eta));
577 }
