36WorkflowSpec defineDataProcessing(ConfigContext const& configcontext)
37{
38 WorkflowSpec specs;
39
40 bool newmctruth = configcontext.options().get<bool>("newmctruth");
41
42 // connect the source
43 specs.emplace_back(o2::getMCTruthSourceSpec(newmctruth));
44 // connect some consumers
45 for (int i = 0; i < configcontext.options().get<int>("consumers"); ++i) {
46 specs.emplace_back(o2::getMCTruthWriterSpec(i, i == 0, newmctruth));
47 }
48 // connect a device reading back the labels
49 specs.emplace_back(o2::getMCTruthReaderSpec(newmctruth));
50 // connect a device reading back the labels
51 specs.emplace_back(o2::getMCTruthWriterSpec(-1, false, newmctruth));
52
53 return specs;
54}
