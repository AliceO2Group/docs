24

46 int phiSpan = -1;

48 int ZBins = 1; // z-phi index table configutation: number of z bins

53
54 int nThreads = 1;
55 bool printMemory = false;
56 size_t maxMemory = std::numeric_limits<size_t>::max();
57 bool dropTFUponFailure = false;
58
59 O2ParamDef(VertexerParamConfig, "ITSVertexerParam");

62struct TrackerParamConfig : public o2::conf::ConfigurableParamHelper<TrackerParamConfig> {
63 // Use TGeo for mat. budget
64 static const int MaxIter = 4;
65 static const int MinTrackLength = 4;
66 static const int MaxTrackLength = 7;
67 bool useMatCorrTGeo = false; // use full geometry to corect for material budget accounting in the fits. Default is to use the material budget LUT.
68 bool useFastMaterial = false; // use faster material approximation for material budget accounting in the fits.
69 int deltaRof = 0; // configure the width of the window in ROFs to be considered for the tracking.
70 int minTrackLgtIter[MaxIter] = {}; // minimum track length at each iteration, used only if >0, otherwise use code defaults
71 uint8_t startLayerMask[MaxIter] = {}; // mask of start layer for this iteration (if >0)
72 float minPtIterLgt[MaxIter * (MaxTrackLength - MinTrackLength + 1)] = {}; // min.pT for given track length at this iteration, used only if >0, otherwise use code defaults
73 float sysErrY2[7] = {0}; // systematic error^2 in Y per layer
74 float sysErrZ2[7] = {0}; // systematic error^2 in Z per layer
75 float maxChi2ClusterAttachment = -1.f;
76 float maxChi2NDF = -1.f;
77 float nSigmaCut = -1.f;
78 float deltaTanLres = -1.f;
79 float minPt = -1.f;
80 float pvRes = -1.f;
81 int LUTbinsPhi = -1;
82 int LUTbinsZ = -1;
83 float diamondPos[3] = {0.f, 0.f, 0.f}; // override the position of the vertex
84 bool useDiamond = false; // enable overriding the vertex position
85 int useTrackFollower = -1; // bit 0: allow mixing implies bits 1&2; bit 1: topwards; bit2: downwards; => 0 off
86 float trackFollowerNSigmaZ = 1.f; // sigma in z-cut for track-following search rectangle
87 float trackFollowerNSigmaPhi = 1.f; // sigma in phi-cut for track-following search rectangle
88 float cellsPerClusterLimit = -1.f;
89 float trackletsPerClusterLimit = -1.f;
90 int findShortTracks = -1;
91 int nROFsPerIterations = 0; // size of the slice of ROFs to be processed at a time, preferably integer divisors of nROFs per TF, to balance the iterations.
92 int nOrbitsPerIterations = 0; // not implemented: size of the slice of ROFs to be processed at a time, computed using the number of ROFs per orbit.
93 bool perPrimaryVertexProcessing = false; // perform the full tracking considering the vertex hypotheses one at the time.
94 bool saveTimeBenchmarks = false; // dump metrics on file
95 bool overrideBeamEstimation = false; // use beam position from meanVertex CCDB object
96 int trackingMode = -1; // -1: unset, 0=sync, 1=async, 2=cosmics used by gpuwf only
97 bool doUPCIteration = false; // Perform an additional iteration for UPC events on tagged vertices. You want to combine this config with VertexerParamConfig.nIterations=2
98 int nIterations = MaxIter; // overwrite the number of iterations
99 int reseedIfShorter = 6; // for the final refit reseed the track with circle if they are shorter than this value
100 bool shiftRefToCluster{true}; // TrackFit: after update shift the linearization reference to cluster
101 bool repeatRefitOut{false}; // repeat outward refit using inward refit as a seed
102 bool createArtefactLabels{false}; // create on-the-fly labels for the artefacts
103
104 int nThreads = 1;
105 bool printMemory = false;
106 size_t maxMemory = std::numeric_limits<size_t>::max();
107 bool dropTFUponFailure = false;
108 bool fataliseUponFailure = true; // granular management of the fatalisation in async mode
109 bool allowSharingFirstCluster = false; // allow first cluster sharing among tracks
110
111 O2ParamDef(TrackerParamConfig, "ITSCATrackerParam");
112};
