42//_______________________________________________________________________
43void Digitizer::process(const std::vector<o2::itsmft::Hit>* hits, int evID, int srcID)
44{
45 // Digitize hits from a single event
46 LOG(debug) << "Digitizing IOTOF hits: " << hits->size() << " hits from event " << evID << " source " << srcID;
47
48 if (!hits || hits->empty()) {
49 return;
50 }
51
52 // Sort hits by detector ID for better cache locality
53 std::vector<int> hitIdx(hits->size());
54 std::iota(hitIdx.begin(), hitIdx.end(), 0);
55 std::sort(hitIdx.begin(), hitIdx.end(),
56 [hits](int lhs, int rhs) {
57 return (*hits)[lhs].GetDetectorID() < (*hits)[rhs].GetDetectorID();
58 });
59
60 // Process each hit
61 for (int i : hitIdx) {
62 processHit((*hits)[i], evID, srcID);
63 }
64
65 // In triggered mode, flush output after each event
66 if (!mContinuous) {
67 fillOutputContainer();
68 }
69}

71//_______________________________________________________________________
72void Digitizer::processHit(const o2::itsmft::Hit& hit, int evID, int srcID)
73{
74 // Process a single hit and create a digit if it passes all cuts
75
76 // Apply efficiency cut
77 if (!isEfficient()) {
78 LOG(debug) << "Hit rejected by efficiency cut";
79 return;
80 }
81
82 // Get detector element ID
83 int detID = hit.GetDetectorID();
84
85 // Convert energy loss to charge (number of electrons)
86 float energyLoss = hit.GetEnergyLoss(); // in GeV
87 int charge = energyToCharge(energyLoss);
88
89 // Apply charge threshold
90 if (charge < mChargeThreshold) {
91 LOG(debug) << "Hit rejected by charge threshold: " << charge << " < " << mChargeThreshold;
92 return;
93 }
94
95 // Get hit time and apply smearing
96 // Hit time is in seconds, convert to ns and add event time
97 double hitTime = hit.GetTime() * sec2ns; // convert to ns
98 double eventTimeNS = mEventTime.getTimeNS(); // event time since orbit 0
99 double absoluteTime = hitTime + eventTimeNS; // absolute time
100 double smearedTime = smearTime(absoluteTime); // apply detector resolution
101
102 // For now, use simple row/col mapping from detector ID
103 // TODO: Implement proper segmentation when geometry is finalized
104 uint16_t chipIndex = static_cast<uint16_t>(detID);
105 uint16_t row = 0; // Will be determined from hit position
106 uint16_t col = 0; // Will be determined from hit position
107
108 // Create the digit with time information
109 int digID = mDigits->size();
110 mDigits->emplace_back(chipIndex, row, col, charge, smearedTime);
111
112 LOG(debug) << "Created digit #" << digID << " chip=" << chipIndex
113 << " charge=" << charge << " time=" << smearedTime << " ns";
114
115 // Add MC truth label
116 if (mMCLabels) {
117 o2::MCCompLabel lbl(hit.GetTrackID(), evID, srcID, false);
118 mMCLabels->addElement(digID, lbl);
119 }
120}

122//_______________________________________________________________________
123double Digitizer::smearTime(double time) const
124{
125 // Apply Gaussian smearing to simulate detector time resolution
126 if (mTimeResolution > 0) {
127 return time + gRandom->Gaus(0, mTimeResolution);
128 }
129 return time;
130}

132//_______________________________________________________________________
133int Digitizer::energyToCharge(float energyLoss) const
134{
135 // Convert energy loss (GeV) to number of electrons
136 // Typical value: 3.6 eV per electron-hole pair in silicon
137 // energyLoss is in GeV, mEnergyToCharge is GeV per electron
138 return static_cast<int>(energyLoss / mEnergyToCharge);
139}

149void Digitizer::fillOutputContainer()
150{
151 // Create ROF record for the current event
152 if (mROFRecords && mDigits && !mDigits->empty()) {
153 o2::itsmft::ROFRecord rof;
154 rof.setFirstEntry(0);
155 rof.setNEntries(mDigits->size());
156 rof.setBCData(mEventTime);
157 mROFRecords->push_back(rof);
158 LOG(debug) << "Created ROF record with " << mDigits->size() << " digits";
159 }
160}
