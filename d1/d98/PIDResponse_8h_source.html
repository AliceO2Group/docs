80 GPUd() float PIDResponse::getExpectedSignal(const TrackTPC& track, const o2::track::PID::ID id) const
81 {
82  const float bg = static_cast<float>(track.getP() / o2::track::pid_constants::sMasses[id]);
83  if (bg < 0.05) {
84  return -999.;
85  }
86  const float bethe = mMIP * o2::tpc::BetheBlochAleph(bg, mBetheBlochParams[0], mBetheBlochParams[1], mBetheBlochParams[2], mBetheBlochParams[3], mBetheBlochParams[4]) * o2::gpu::GPUCommonMath::Pow(static_cast<float>(o2::track::pid_constants::sCharges[id]), mChargeFactor);
87  return bethe >= 0. ? bethe : -999.;
88 }

91 GPUd() o2::track::PID::ID PIDResponse::getMostProbablePID(const TrackTPC& track, float PID_EKrangeMin, float PID_EKrangeMax, float PID_EPrangeMin, float PID_EPrangeMax, float PID_EDrangeMin, float PID_EDrangeMax, float PID_ETrangeMin, float PID_ETrangeMax, char PID_useNsigma, float PID_sigma) const
92 {
93  const float dEdx = track.getdEdx().dEdxTotTPC;
94 
95  if (dEdx < 10.) {
96  return o2::track::PID::Pion;
97  }
98 
99  auto id = o2::track::PID::Electron;
100  float distanceMin = 0.;
101  float dEdxExpected = getExpectedSignal(track, id);
102  if (PID_useNsigma) {
103  // using nSigma
104  distanceMin = o2::gpu::GPUCommonMath::Abs((dEdx - dEdxExpected) / (PID_sigma * dEdxExpected));
105  } else {
106  // using absolute distance
107  distanceMin = o2::gpu::GPUCommonMath::Abs(dEdx - dEdxExpected);
108  }
109 
110  // calculate the distance to the expected dEdx signals
111  // start from Pion to exlude Muons
112  for (o2::track::PID::ID i = o2::track::PID::Pion; i < o2::track::PID::NIDs; i++) {
113  float distance = 0.;
114  dEdxExpected = getExpectedSignal(track, i);
115  if (PID_useNsigma) {
116  // using nSigma
117  distance = o2::gpu::GPUCommonMath::Abs((dEdx - dEdxExpected) / (PID_sigma * dEdxExpected));
118  } else {
119  // using absolute distance
120  distance = o2::gpu::GPUCommonMath::Abs(dEdx - dEdxExpected);
121  }
122  if (distance < distanceMin) {
123  id = i;
124  distanceMin = distance;
125  }
126  }
127 
128  // override the electrons to the closest alternative PID in the bands crossing regions
129  if (id == o2::track::PID::Electron) {
130  const float p = track.getP();
131  if ((p > PID_EKrangeMin) && (p < PID_EKrangeMax)) {
132  id = o2::track::PID::Kaon;
133  } else if ((p > PID_EPrangeMin) && (p < PID_EPrangeMax)) {
134  id = o2::track::PID::Proton;
135  } else if ((p > PID_EDrangeMin) && (p < PID_EDrangeMax)) {
136  id = o2::track::PID::Deuteron;
137  } else if ((p > PID_ETrangeMin) && (p < PID_ETrangeMax)) {
138  id = o2::track::PID::Triton;
139  }
140  }
141 
142  return id;
143 }
