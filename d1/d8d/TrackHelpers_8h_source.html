32GPUdi() int selectReseedMidLayer(int minLayer, int maxLayer, int nLayers, const float* layerRadii)
33{
34 if (maxLayer - minLayer == nLayers - 1) {
35 return (minLayer + maxLayer) / 2;
36 }
37 int midLayer = minLayer + 1;
38 const float midR = 0.5f * (layerRadii[maxLayer] + layerRadii[minLayer]);
39 float distanceToMidR = o2::gpu::CAMath::Abs(midR - layerRadii[midLayer]);
40 for (int iLayer = midLayer + 1; iLayer < maxLayer; ++iLayer) { // find the midpoint as closest to the midR
41 const float distance = o2::gpu::CAMath::Abs(midR - layerRadii[iLayer]);
42 if (distance < distanceToMidR) {
43 midLayer = iLayer;
44 distanceToMidR = distance;
45 }
46 }
47 return midLayer;
48}

56GPUdi() o2::track::TrackParCov buildTrackSeed(const Cluster& cluster1,
57 const Cluster& cluster2,
58 const TrackingFrameInfo& tf3,
59 const float bz,
60 const bool reverse = false)
61{
62 float ca = NAN, sa = NAN, snp = NAN, q2pt = NAN, q2pt2 = NAN;
63 o2::gpu::CAMath::SinCos(tf3.alphaTrackingFrame, sa, ca);
64 const float sign = reverse ? -1.f : 1.f;
65 const float x1 = (cluster1.xCoordinate * ca) + (cluster1.yCoordinate * sa);
66 const float y1 = (-cluster1.xCoordinate * sa) + (cluster1.yCoordinate * ca);
67 const float x2 = (cluster2.xCoordinate * ca) + (cluster2.yCoordinate * sa);
68 const float y2 = (-cluster2.xCoordinate * sa) + (cluster2.yCoordinate * ca);
69 const float x3 = tf3.xTrackingFrame;
70 const float y3 = tf3.positionTrackingFrame[0];
71 if (o2::gpu::CAMath::Abs(bz) < 0.01f) { // zero field
72 const float dx = x3 - x1;
73 const float dy = y3 - y1;
74 snp = sign * dy / o2::gpu::CAMath::Hypot(dx, dy);
75 q2pt = 1.f / o2::track::kMostProbablePt;
76 q2pt2 = 1.f;
77 } else {
78 const float crv = math_utils::computeCurvature(x3, y3, x2, y2, x1, y1);
79 snp = sign * crv * (x3 - math_utils::computeCurvatureCentreX(x3, y3, x2, y2, x1, y1));
80 q2pt = sign * crv / (bz * o2::constants::math::B2C);
81 q2pt2 = crv * crv;
82 }
83 const float tgl = -0.5f * sign * (math_utils::computeTanDipAngle(x1, y1, x2, y2, cluster1.zCoordinate, cluster2.zCoordinate) + math_utils::computeTanDipAngle(x2, y2, x3, y3, cluster2.zCoordinate, tf3.positionTrackingFrame[1]));
84 const float sg2q2pt = o2::track::kC1Pt2max * o2::gpu::CAMath::Clamp(q2pt2, 0.0005f, 1.0f);
85 return {x3, tf3.alphaTrackingFrame, {y3, tf3.positionTrackingFrame[1], snp, tgl, q2pt}, {tf3.covarianceTrackingFrame[0], tf3.covarianceTrackingFrame[1], tf3.covarianceTrackingFrame[2], 0.f, 0.f, o2::track::kCSnp2max, 0.f, 0.f, 0.f, o2::track::kCTgl2max, 0.f, 0.f, 0.f, 0.f, sg2q2pt}};
86}

88template <int NLayers>
89GPUdi() TrackITSExt seedTrackForRefit(const TrackSeed<NLayers>& seed,
90 const TrackingFrameInfo* const* foundTrackingFrameInfo,
91 const Cluster* const* unsortedClusters,
92 const float* layerRadii,
93 const float bz,
94 const int reseedIfShorter)
95{
96 TrackITSExt temporaryTrack(seed);
97 int lrMin = NLayers;
98 int lrMax = 0;
99 for (int iL{0}; iL < NLayers; ++iL) {
100 const int idx = seed.getCluster(iL);
101 temporaryTrack.setExternalClusterIndex(iL, idx, idx != constants::UnusedIndex);
102 if (idx != constants::UnusedIndex) {
103 lrMin = o2::gpu::CAMath::Min(lrMin, iL);
104 lrMax = o2::gpu::CAMath::Max(lrMax, iL);
105 }
106 }
107
108 const int ncl = temporaryTrack.getNClusters();
109 if (ncl < reseedIfShorter && ncl > 1) {
110 const int lrMid = selectReseedMidLayer(lrMin, lrMax, NLayers, layerRadii);
111 const auto& cluster0TF = foundTrackingFrameInfo[lrMin][seed.getCluster(lrMin)];
112 const auto& cluster1GL = unsortedClusters[lrMid][seed.getCluster(lrMid)];
113 const auto& cluster2GL = unsortedClusters[lrMax][seed.getCluster(lrMax)];
114 temporaryTrack.getParamIn() = buildTrackSeed(cluster2GL, cluster1GL, cluster0TF, bz, true);
115 }
116
117 resetTrackCovariance(temporaryTrack);
118 return temporaryTrack;
119}

121GPUdi() bool fitTrack(TrackITSExt& trk,
122 int start,
123 int end,
124 int step,
125 float chi2clcut,
126 float chi2ndfcut,
127 float maxQoverPt,
128 int nCl,
129 const float bz,
130 const TrackingFrameInfo* const* tfInfos,
131 const float* layerxX0,
132 const o2::base::Propagator* propagator,
133 const o2::base::PropagatorF::MatCorrType matCorrType,
134 o2::track::TrackPar* linRef = nullptr,
135 const bool shiftRefToCluster = false)
136{
137 for (int iLayer{start}; iLayer != end; iLayer += step) {
138 if (trk.getClusterIndex(iLayer) == constants::UnusedIndex) {
139 continue;
140 }
141
142 const TrackingFrameInfo& trackingHit = tfInfos[iLayer][trk.getClusterIndex(iLayer)];
143 if (linRef) {
144 if (!trk.o2::track::TrackParCovF::rotate(trackingHit.alphaTrackingFrame, *linRef, bz)) {
145 return false;
146 }
147 if (!propagator->propagateToX(trk, *linRef, trackingHit.xTrackingFrame, bz,
148 o2::base::PropagatorImpl<float>::MAX_SIN_PHI,
149 o2::base::PropagatorImpl<float>::MAX_STEP,
150 matCorrType)) {
151 return false;
152 }
153 if (matCorrType == o2::base::PropagatorF::MatCorrType::USEMatCorrNONE) {
154 if (!trk.correctForMaterial(*linRef, layerxX0[iLayer], layerxX0[iLayer] * constants::Radl * constants::Rho, true)) {
155 continue;
156 }
157 }
158 } else {
159 if (!trk.o2::track::TrackParCovF::rotate(trackingHit.alphaTrackingFrame)) {
160 return false;
161 }
162 if (!propagator->propagateToX(trk, trackingHit.xTrackingFrame, bz,
163 o2::base::PropagatorImpl<float>::MAX_SIN_PHI,
164 o2::base::PropagatorImpl<float>::MAX_STEP,
165 matCorrType)) {
166 return false;
167 }
168 if (matCorrType == o2::base::PropagatorF::MatCorrType::USEMatCorrNONE) {
169 if (!trk.correctForMaterial(layerxX0[iLayer], layerxX0[iLayer] * constants::Radl * constants::Rho, true)) {
170 continue;
171 }
172 }
173 }
174
175 const auto predChi2{trk.getPredictedChi2Quiet(trackingHit.positionTrackingFrame, trackingHit.covarianceTrackingFrame)};
176 if ((nCl >= 3 && predChi2 > chi2clcut) || predChi2 < 0.f) {
177 return false;
178 }
179 trk.setChi2(trk.getChi2() + predChi2);
180 if (!trk.o2::track::TrackParCov::update(trackingHit.positionTrackingFrame, trackingHit.covarianceTrackingFrame)) {
181 return false;
182 }
183 if (linRef && shiftRefToCluster) {
184 linRef->setY(trackingHit.positionTrackingFrame[0]);
185 linRef->setZ(trackingHit.positionTrackingFrame[1]);
186 }
187 nCl++;
188 }
189
190 return o2::gpu::CAMath::Abs(trk.getQ2Pt()) < maxQoverPt && trk.getChi2() < chi2ndfcut * (float)((nCl * 2) - 5);
191}

193template <int NLayers>
194GPUdi() bool refitTrack(const TrackSeed<NLayers>& trackSeed,
195 TrackITSExt& temporaryTrack,
196 float chi2clcut,
197 float chi2ndfcut,
198 const float bz,
199 const TrackingFrameInfo* const* tfInfos,
200 const Cluster* const* clusters,
201 const float* layerxX0,
202 const float* layerRadii,
203 const float* minPt,
204 const o2::base::Propagator* propagator,
205 const o2::base::PropagatorF::MatCorrType matCorrType,
206 const int reseedIfShorter,
207 const bool shiftRefToCluster,
208 const bool repeatRefitOut)
209{
210 temporaryTrack = seedTrackForRefit(trackSeed,
211 tfInfos,
212 clusters,
213 layerRadii,
214 bz,
215 reseedIfShorter);
216 o2::track::TrackPar linRef{temporaryTrack};
217 bool fitSuccess = fitTrack(temporaryTrack,
218 0,
219 NLayers,
220 1,
221 chi2clcut,
222 chi2ndfcut,
223 o2::constants::math::VeryBig,
224 0,
225 bz,
226 tfInfos,
227 layerxX0,
228 propagator,
229 matCorrType,
230 &linRef,
231 shiftRefToCluster);
232 if (!fitSuccess) {
233 return false;
234 }
235 temporaryTrack.getParamOut() = temporaryTrack.getParamIn();
236 linRef = temporaryTrack.getParamOut(); // use refitted track as lin.reference
237 resetTrackCovariance(temporaryTrack);
238 temporaryTrack.setChi2(0);
239 fitSuccess = fitTrack(temporaryTrack,
240 NLayers - 1,
241 -1,
242 -1,
243 chi2clcut,
244 chi2ndfcut,
245 50.f,
246 0,
247 bz,
248 tfInfos,
249 layerxX0,
250 propagator,
251 matCorrType,
252 &linRef,
253 shiftRefToCluster);
254 if (!fitSuccess || temporaryTrack.getPt() < minPt[NLayers - temporaryTrack.getNClusters()]) {
255 return false;
256 }
257 if (repeatRefitOut) { // repeat outward refit seeding and linearizing with the stable inward fit result
258 o2::track::TrackParCov saveInw{temporaryTrack};
259 linRef = saveInw; // use refitted track as lin.reference
260 float saveChi2 = temporaryTrack.getChi2();
261 track::resetTrackCovariance(temporaryTrack);
262 temporaryTrack.setChi2(0);
263 fitSuccess = o2::its::track::fitTrack(temporaryTrack,
264 0,
265 NLayers,
266 1,
267 chi2clcut,
268 chi2ndfcut,
269 o2::constants::math::VeryBig,
270 0,
271 bz,
272 tfInfos,
273 layerxX0,
274 propagator,
275 matCorrType,
276 &linRef,
277 shiftRefToCluster);
278 if (!fitSuccess) {
279 return false;
280 }
281 temporaryTrack.getParamOut() = temporaryTrack.getParamIn();
282 temporaryTrack.getParamIn() = saveInw;
283 temporaryTrack.setChi2(saveChi2);
284 }
285 return true;
286}
