66template <typename T>
67struct LifetimeHolder {
68 using type = T;
69 T* ptr = nullptr;
70 std::function<void(T&)> callback = nullptr;
71 LifetimeHolder(T* ptr_) : ptr(ptr_),
72 callback(nullptr)
73 {
74 }
75 LifetimeHolder() = delete;
76 // Never copy it, because there is only one LifetimeHolder pointer
77 // created object.
78 LifetimeHolder(const LifetimeHolder&) = delete;
79 LifetimeHolder& operator=(const LifetimeHolder&) = delete;
80 LifetimeHolder(LifetimeHolder&& other)
81 {
82 this->ptr = other.ptr;
83 other.ptr = nullptr;
84 if (other.callback) {
85 this->callback = std::move(other.callback);
86 } else {
87 this->callback = nullptr;
88 }
89 other.callback = nullptr;
90 }
91 LifetimeHolder& operator=(LifetimeHolder&& other)
92 {
93 this->ptr = other.ptr;
94 other.ptr = nullptr;
95 if (other.callback) {
96 this->callback = std::move(other.callback);
97 } else {
98 this->callback = nullptr;
99 }
100 other.callback = nullptr;
101 return *this;
102 }
103
104 // On deletion we invoke the callback and then delete the object,
105 // when prensent.
106 ~LifetimeHolder()
107 {
108 release();
109 }
110
111 T* operator->() { return ptr; }
112 T& operator*() { return *ptr; }
113
114 // release the owned object, if any. This allows to
115 // invoke the callback early (e.g. for the Product<> case)
116 void release()
117 {
118 if (ptr && callback) {
119 callback(*ptr);
120 delete ptr;
121 ptr = nullptr;
122 }
123 }
124};

310 template <typename T>
311 void snapshot(const Output& spec, T const& object)
312 {
313 auto& proxy = mRegistry.get<MessageContext>().proxy();
314 fair::mq::MessagePtr payloadMessage;
315 auto serializationType = o2::header::gSerializationMethodNone;
316 RouteIndex routeIndex = matchDataHeader(spec, mRegistry.get<TimingInfo>().timeslice);
317 if constexpr (is_messageable<T>::value == true) {
318 // Serialize a snapshot of a trivially copyable, non-polymorphic object,
319 payloadMessage = proxy.createOutputMessage(routeIndex, sizeof(T));
320 memcpy(payloadMessage->GetData(), &object, sizeof(T));
321
322 serializationType = o2::header::gSerializationMethodNone;
323 } else if constexpr (is_specialization_v<T, std::vector> == true ||
324 (gsl::details::is_span<T>::value && has_messageable_value_type<T>::value)) {
325 using ElementType = typename std::remove_pointer<typename T::value_type>::type;
326 if constexpr (is_messageable<ElementType>::value) {
327 // Serialize a snapshot of a std::vector of trivially copyable, non-polymorphic elements
328 // Note: in most cases it is better to use the `make` function und work with the provided
329 // reference object
330 constexpr auto elementSizeInBytes = sizeof(ElementType);
331 auto sizeInBytes = elementSizeInBytes * object.size();
332 payloadMessage = proxy.createOutputMessage(routeIndex, sizeInBytes);
333
334 if constexpr (std::is_pointer<typename T::value_type>::value == false) {
335 // vector of elements
336 if (object.data() && sizeInBytes) {
337 memcpy(payloadMessage->GetData(), object.data(), sizeInBytes);
338 }
339 } else {
340 // serialize vector of pointers to elements
341 auto target = reinterpret_cast<unsigned char*>(payloadMessage->GetData());
342 for (auto const& pointer : object) {
343 memcpy(target, pointer, elementSizeInBytes);
344 target += elementSizeInBytes;
345 }
346 }
347
348 serializationType = o2::header::gSerializationMethodNone;
349 } else if constexpr (has_root_dictionary<ElementType>::value) {
350 return snapshot(spec, ROOTSerialized<T const>(object));
351 } else {
352 static_assert(always_static_assert_v<T>,
353 "value type of std::vector not supported by API, supported types:"
354 "\n - messageable tyeps (trivially copyable, non-polymorphic structures)"
355 "\n - pointers to those"
356 "\n - types with ROOT dictionary and implementing ROOT ClassDef interface");
357 }
358 } else if constexpr (is_container<T>::value == true && has_messageable_value_type<T>::value == true) {
359 // Serialize a snapshot of a std::container of trivially copyable, non-polymorphic elements
360 // Note: in most cases it is better to use the `make` function und work with the provided
361 // reference object
362 constexpr auto elementSizeInBytes = sizeof(typename T::value_type);
363 auto sizeInBytes = elementSizeInBytes * object.size();
364 payloadMessage = proxy.createOutputMessage(routeIndex, sizeInBytes);
365
366 // serialize vector of pointers to elements
367 auto target = reinterpret_cast<unsigned char*>(payloadMessage->GetData());
368 for (auto const& entry : object) {
369 memcpy(target, (void*)&entry, elementSizeInBytes);
370 target += elementSizeInBytes;
371 }
372 serializationType = o2::header::gSerializationMethodNone;
373 } else if constexpr (has_root_dictionary<T>::value == true || is_specialization_v<T, ROOTSerialized> == true) {
374 // Serialize a snapshot of an object with root dictionary
375 payloadMessage = proxy.createOutputMessage(routeIndex);
376 payloadMessage->Rebuild(4096, {64});
377 if constexpr (is_specialization_v<T, ROOTSerialized> == true) {
378 // Explicitely ROOT serialize a snapshot of object.
379 // An object wrapped into type `ROOTSerialized` is explicitely marked to be ROOT serialized
380 // and is expected to have a ROOT dictionary. Availability can not be checked at compile time
381 // for all cases.
382 using WrappedType = typename T::wrapped_type;
383 static_assert(std::is_same<typename T::hint_type, const char>::value ||
384 std::is_same<typename T::hint_type, TClass>::value ||
385 std::is_void<typename T::hint_type>::value,
386 "class hint must be of type TClass or const char");
387
388 const TClass* cl = nullptr;
389 if (object.getHint() == nullptr) {
390 // get TClass info by wrapped type
391 cl = TClass::GetClass(typeid(WrappedType));
392 } else if (std::is_same<typename T::hint_type, TClass>::value) {
393 // the class info has been passed directly
394 cl = reinterpret_cast<const TClass*>(object.getHint());
395 } else if (std::is_same<typename T::hint_type, const char>::value) {
396 // get TClass info by optional name
397 cl = TClass::GetClass(reinterpret_cast<const char*>(object.getHint()));
398 }
399 if (has_root_dictionary<WrappedType>::value == false && cl == nullptr) {
400 if (std::is_same<typename T::hint_type, const char>::value) {
401 throw runtime_error_f("ROOT serialization not supported, dictionary not found for type %s",
402 reinterpret_cast<const char*>(object.getHint()));
403 } else {
404 throw runtime_error_f("ROOT serialization not supported, dictionary not found for type %s",
405 typeid(WrappedType).name());
406 }
407 }
408 typename root_serializer<T>::serializer().Serialize(*payloadMessage, &object(), cl);
409 } else {
410 typename root_serializer<T>::serializer().Serialize(*payloadMessage, &object, TClass::GetClass(typeid(T)));
411 }
412 serializationType = o2::header::gSerializationMethodROOT;
413 } else {
414 static_assert(always_static_assert_v<T>,
415 "data type T not supported by API, \n specializations available for"
416 "\n - trivially copyable, non-polymorphic structures"
417 "\n - std::vector of messageable structures or pointers to those"
418 "\n - types with ROOT dictionary and implementing ROOT ClassDef interface");
419 }
420 addPartToContext(routeIndex, std::move(payloadMessage), spec, serializationType);
421 }

553template <typename ContainerT>
554DataAllocator::CacheId DataAllocator::adoptContainer(const Output& spec, ContainerT&& container, CacheStrategy cache, header::SerializationMethod method)
555{
556 // Find a matching channel, extract the message for it form the container
557 // and put it in the queue to be sent at the end of the processing
558 auto& timingInfo = mRegistry.get<TimingInfo>();
559 auto routeIndex = matchDataHeader(spec, timingInfo.timeslice);
560
561 auto& context = mRegistry.get<MessageContext>();
562 auto* transport = mRegistry.get<FairMQDeviceProxy>().getOutputTransport(routeIndex);
563 fair::mq::MessagePtr payloadMessage = o2::pmr::getMessage(std::forward<ContainerT>(container), *transport);
564 fair::mq::MessagePtr headerMessage = headerMessageFromOutput(spec, routeIndex, //
565 method, //
566 payloadMessage->GetSize() //
567 );
568
569 CacheId cacheId{0}; //
570 if (cache == CacheStrategy::Always) {
571 // The message will be shallow cloned in the cache. Since the
572 // clone is indistinguishable from the original, we can keep sending
573 // the original.
574 cacheId.value = context.addToCache(payloadMessage);
575 }
576
577 context.add<MessageContext::TrivialObject>(std::move(headerMessage), std::move(payloadMessage), routeIndex);
578 return cacheId;
579}
