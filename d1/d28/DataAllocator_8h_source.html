73 template <typename T>
74 struct LifetimeHolder {
75  using type = T;
76  T* ptr = nullptr;
77  std::function<void(T&)> callback = nullptr;
78  LifetimeHolder(T* ptr_) : ptr(ptr_),
79  callback(nullptr)
80  {
81  }
82  LifetimeHolder() = delete;
83  // Never copy it, because there is only one LifetimeHolder pointer
84  // created object.
85  LifetimeHolder(const LifetimeHolder&) = delete;
86  LifetimeHolder& operator=(const LifetimeHolder&) = delete;
87  LifetimeHolder(LifetimeHolder&& other)
88  {
89  this->ptr = other.ptr;
90  other.ptr = nullptr;
91  if (other.callback) {
92  this->callback = std::move(other.callback);
93  } else {
94  this->callback = nullptr;
95  }
96  other.callback = nullptr;
97  }
98  LifetimeHolder& operator=(LifetimeHolder&& other)
99  {
100  this->ptr = other.ptr;
101  other.ptr = nullptr;
102  if (other.callback) {
103  this->callback = std::move(other.callback);
104  } else {
105  this->callback = nullptr;
106  }
107  other.callback = nullptr;
108  return *this;
109  }
110 
111  // On deletion we invoke the callback and then delete the object,
112  // when prensent.
113  ~LifetimeHolder()
114  {
115  release();
116  }
117 
118  T* operator->() { return ptr; }
119  T& operator*() { return *ptr; }
120 
121  // release the owned object, if any. This allows to
122  // invoke the callback early (e.g. for the Product<> case)
123  void release()
124  {
125  if (ptr && callback) {
126  callback(*ptr);
127  delete ptr;
128  ptr = nullptr;
129  }
130  }
131 };

174  {
175  auto& timingInfo = mRegistry.get<TimingInfo>();
176  auto& context = mRegistry.get<MessageContext>();

178  if constexpr (is_specialization_v<T, UninitializedVector>) {
179  auto routeIndex = matchDataHeader(spec, timingInfo.timeslice);
180  // plain buffer as polymorphic spectator std::vector, which does not run constructors / destructors
181  using ValueType = typename T::value_type;

184  fair::mq::MessagePtr headerMessage = headerMessageFromOutput(spec, routeIndex, o2::header::gSerializationMethodNone, 0);
185  return context.add<MessageContext::VectorObject<ValueType, MessageContext::ContainerRefObject<std::vector<ValueType, o2::pmr::NoConstructAllocator<ValueType>>>>>(
186  std::move(headerMessage), routeIndex, 0, std::forward<Args>(args)...)
187  .get();
188  } else if constexpr (is_specialization_v<T, std::vector> && has_messageable_value_type<T>::value) {
189  auto routeIndex = matchDataHeader(spec, timingInfo.timeslice);
190  // this catches all std::vector objects with messageable value type before checking if is also
191  // has a root dictionary, so non-serialized transmission is preferred
192  using ValueType = typename T::value_type;
193 
194  // Note: initial payload size is 0 and will be set by the context before sending
195  fair::mq::MessagePtr headerMessage = headerMessageFromOutput(spec, routeIndex, o2::header::gSerializationMethodNone, 0);
196  return context.add<MessageContext::VectorObject<ValueType>>(std::move(headerMessage), routeIndex, 0, std::forward<Args>(args)...).get();
197  } else if constexpr (has_root_dictionary<T>::value == true && is_messageable<T>::value == false) {
198  auto routeIndex = matchDataHeader(spec, timingInfo.timeslice);
199  // Extended support for types implementing the Root ClassDef interface, both TObject
200  // derived types and others
201  if constexpr (enable_root_serialization<T>::value) {
202  fair::mq::MessagePtr headerMessage = headerMessageFromOutput(spec, routeIndex, o2::header::gSerializationMethodROOT, 0);
203 
204  return context.add<typename enable_root_serialization<T>::object_type>(std::move(headerMessage), routeIndex, std::forward<Args>(args)...).get();
205  } else {
206  static_assert(enable_root_serialization<T>::value, "Please make sure you include RootMessageContext.h");
207  }
208  // Note: initial payload size is 0 and will be set by the context before sending
209  } else if constexpr (std::is_base_of_v<std::string, T>) {
210  auto* s = new std::string(args...);
211  adopt(spec, s);
212  return *s;
213  } else if constexpr (requires { static_cast<struct TableBuilder>(std::declval<std::decay_t<T>>()); }) {
214  auto tb = std::move(LifetimeHolder<TableBuilder>(new std::decay_t<T>(args...)));
215  adopt(spec, tb);
216  return tb;
217  } else if constexpr (requires { static_cast<struct TreeToTable>(std::declval<std::decay_t<T>>()); }) {
218  auto t2t = std::move(LifetimeHolder<TreeToTable>(new std::decay_t<T>(args...)));
219  adopt(spec, t2t);
220  return t2t;
221  } else if constexpr (sizeof...(Args) == 0) {
222  if constexpr (is_messageable<T>::value == true) {
223  return *reinterpret_cast<T*>(newChunk(spec, sizeof(T)).data());
224  } else {
225  static_assert(always_static_assert_v<T>, ERROR_STRING);
226  }
227  } else if constexpr (sizeof...(Args) == 1) {
228  using FirstArg = typename std::tuple_element<0, std::tuple<Args...>>::type;
229  if constexpr (std::is_integral_v<FirstArg>) {
230  if constexpr (is_messageable<T>::value == true) {
231  auto [nElements] = std::make_tuple(args...);
232  auto size = nElements * sizeof(T);
233  auto routeIndex = matchDataHeader(spec, timingInfo.timeslice);
234 
235  fair::mq::MessagePtr headerMessage = headerMessageFromOutput(spec, routeIndex, o2::header::gSerializationMethodNone, size);
236  return context.add<MessageContext::SpanObject<T>>(std::move(headerMessage), routeIndex, 0, nElements).get();

242  create(spec, &writer, schema);
243  return writer;

246  static_assert(always_static_assert_v<T>, ERROR_STRING);
247  }
248  } else {
249  static_assert(always_static_assert_v<T>, ERROR_STRING);

291  template <typename T>
292  void snapshot(const Output& spec, T const& object)
293  {
294  auto& proxy = mRegistry.get<MessageContext>().proxy();
295  fair::mq::MessagePtr payloadMessage;
296  auto serializationType = o2::header::gSerializationMethodNone;
297  RouteIndex routeIndex = matchDataHeader(spec, mRegistry.get<TimingInfo>().timeslice);
298  if constexpr (is_messageable<T>::value == true) {
299  // Serialize a snapshot of a trivially copyable, non-polymorphic object,
300  payloadMessage = proxy.createOutputMessage(routeIndex, sizeof(T));
301  memcpy(payloadMessage->GetData(), &object, sizeof(T));
302 
303  serializationType = o2::header::gSerializationMethodNone;
304  } else if constexpr (is_specialization_v<T, std::vector> == true ||
305  (gsl::details::is_span<T>::value && has_messageable_value_type<T>::value)) {
306  using ElementType = typename std::remove_pointer<typename T::value_type>::type;
307  if constexpr (is_messageable<ElementType>::value) {
308  // Serialize a snapshot of a std::vector of trivially copyable, non-polymorphic elements
309  // Note: in most cases it is better to use the `make` function und work with the provided
310  // reference object
311  constexpr auto elementSizeInBytes = sizeof(ElementType);
312  auto sizeInBytes = elementSizeInBytes * object.size();
313  payloadMessage = proxy.createOutputMessage(routeIndex, sizeInBytes);
314 
315  if constexpr (std::is_pointer<typename T::value_type>::value == false) {
316  // vector of elements
317  if (object.data() && sizeInBytes) {
318  memcpy(payloadMessage->GetData(), object.data(), sizeInBytes);
319  }
320  } else {
321  // serialize vector of pointers to elements
322  auto target = reinterpret_cast<unsigned char*>(payloadMessage->GetData());
323  for (auto const& pointer : object) {
324  memcpy(target, pointer, elementSizeInBytes);
325  target += elementSizeInBytes;
326  }
327  }
328 
329  serializationType = o2::header::gSerializationMethodNone;
330  } else if constexpr (has_root_dictionary<ElementType>::value) {
331  return snapshot(spec, ROOTSerialized<T const>(object));
332  } else {
333  static_assert(always_static_assert_v<T>,
334  "value type of std::vector not supported by API, supported types:"
335  "\n - messageable tyeps (trivially copyable, non-polymorphic structures)"
336  "\n - pointers to those"
337  "\n - types with ROOT dictionary and implementing ROOT ClassDef interface");
338  }
339  } else if constexpr (is_container<T>::value == true && has_messageable_value_type<T>::value == true) {
340  // Serialize a snapshot of a std::container of trivially copyable, non-polymorphic elements
341  // Note: in most cases it is better to use the `make` function und work with the provided
342  // reference object
343  constexpr auto elementSizeInBytes = sizeof(typename T::value_type);
344  auto sizeInBytes = elementSizeInBytes * object.size();
345  payloadMessage = proxy.createOutputMessage(routeIndex, sizeInBytes);
346 
347  // serialize vector of pointers to elements
348  auto target = reinterpret_cast<unsigned char*>(payloadMessage->GetData());
349  for (auto const& entry : object) {
350  memcpy(target, (void*)&entry, elementSizeInBytes);
351  target += elementSizeInBytes;
352  }
353  serializationType = o2::header::gSerializationMethodNone;
354  } else if constexpr (has_root_dictionary<T>::value == true || is_specialization_v<T, ROOTSerialized> == true) {
355  // Serialize a snapshot of an object with root dictionary
356  payloadMessage = proxy.createOutputMessage(routeIndex);
357  payloadMessage->Rebuild(4096, {64});
358  if constexpr (is_specialization_v<T, ROOTSerialized> == true) {
359  // Explicitely ROOT serialize a snapshot of object.
360  // An object wrapped into type `ROOTSerialized` is explicitely marked to be ROOT serialized
361  // and is expected to have a ROOT dictionary. Availability can not be checked at compile time
362  // for all cases.
363  using WrappedType = typename T::wrapped_type;
364  static_assert(std::is_same<typename T::hint_type, const char>::value ||
365  std::is_same<typename T::hint_type, TClass>::value ||
366  std::is_void<typename T::hint_type>::value,
367  "class hint must be of type TClass or const char");
368 
369  const TClass* cl = nullptr;
370  if (object.getHint() == nullptr) {
371  // get TClass info by wrapped type
372  cl = TClass::GetClass(typeid(WrappedType));
373  } else if (std::is_same<typename T::hint_type, TClass>::value) {
374  // the class info has been passed directly
375  cl = reinterpret_cast<const TClass*>(object.getHint());
376  } else if (std::is_same<typename T::hint_type, const char>::value) {
377  // get TClass info by optional name
378  cl = TClass::GetClass(reinterpret_cast<const char*>(object.getHint()));
379  }
380  if (has_root_dictionary<WrappedType>::value == false && cl == nullptr) {
381  if (std::is_same<typename T::hint_type, const char>::value) {
382  throw runtime_error_f("ROOT serialization not supported, dictionary not found for type %s",
383  reinterpret_cast<const char*>(object.getHint()));
384  } else {
385  throw runtime_error_f("ROOT serialization not supported, dictionary not found for type %s",
386  typeid(WrappedType).name());
387  }
388  }
389  typename root_serializer<T>::serializer().Serialize(*payloadMessage, &object(), cl);
390  } else {
391  typename root_serializer<T>::serializer().Serialize(*payloadMessage, &object, TClass::GetClass(typeid(T)));
392  }
393  serializationType = o2::header::gSerializationMethodROOT;
394  } else {
395  static_assert(always_static_assert_v<T>,
396  "data type T not supported by API, \n specializations available for"
397  "\n - trivially copyable, non-polymorphic structures"
398  "\n - std::vector of messageable structures or pointers to those"
399  "\n - types with ROOT dictionary and implementing ROOT ClassDef interface");
400  }
401  addPartToContext(routeIndex, std::move(payloadMessage), spec, serializationType);
402  }

534 template <typename ContainerT>
535 DataAllocator::CacheId DataAllocator::adoptContainer(const Output& spec, ContainerT&& container, CacheStrategy cache, header::SerializationMethod method)
536 {
537  // Find a matching channel, extract the message for it form the container
538  // and put it in the queue to be sent at the end of the processing
539  auto& timingInfo = mRegistry.get<TimingInfo>();
540  auto routeIndex = matchDataHeader(spec, timingInfo.timeslice);
541 
542  auto& context = mRegistry.get<MessageContext>();
543  auto* transport = mRegistry.get<FairMQDeviceProxy>().getOutputTransport(routeIndex);
544  fair::mq::MessagePtr payloadMessage = o2::pmr::getMessage(std::forward<ContainerT>(container), *transport);
545  fair::mq::MessagePtr headerMessage = headerMessageFromOutput(spec, routeIndex, //
546  method, //
547  payloadMessage->GetSize() //
548  );
549 
550  CacheId cacheId{0}; //
551  if (cache == CacheStrategy::Always) {
552  // The message will be shallow cloned in the cache. Since the
553  // clone is indistinguishable from the original, we can keep sending
554  // the original.
555  cacheId.value = context.addToCache(payloadMessage);
556  }
557 
558  context.add<MessageContext::TrivialObject>(std::move(headerMessage), std::move(payloadMessage), routeIndex);
559  return cacheId;
560 }
