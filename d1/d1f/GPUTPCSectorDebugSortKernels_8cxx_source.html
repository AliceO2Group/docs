35 const uint32_t n = jMax - jMin;

37 auto* GPUrestrict() tmp2 = tracker.Data().HitWeights() + row.HitNumberOffset() + jMin;
38 cahit2* GPUrestrict() hitData = tracker.Data().HitData(row) + jMin;
39 int32_t* GPUrestrict() clusterId = tracker.Data().ClusterDataIndex() + row.HitNumberOffset() + jMin;
40 for (uint32_t j = 0; j < n; j++) {
41 tmp1[j] = j;
42 }
43 GPUCommonAlgorithm::sort(tmp1, tmp1 + n, [&hitData, &clusterId](const calink& a, const calink& b) {
44 if (hitData[a].x != hitData[b].x) {
45 return hitData[a].x < hitData[b].x;
46 }
47 if (hitData[a].y != hitData[b].y) {
48 return hitData[a].y < hitData[b].y;
49 }
50 return clusterId[a] < clusterId[b];
51 });
