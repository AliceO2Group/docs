57void* GPUReconstructionDeviceBase::helperWrapper(GPUReconstructionHelpers::helperParam* par)
58{
59 if (mProcessingSettings.debugLevel >= 3) {
60 GPUInfo("\tHelper thread %d starting", par->num);
61 }
62
63 // cpu_set_t mask; //TODO add option
64 // CPU_ZERO(&mask);
65 // CPU_SET(par->num * 2 + 2, &mask);
66 // sched_setaffinity(0, sizeof(mask), &mask);
67
68 par->mutex[0].lock();
69 while (par->terminate == false) {
70 for (int32_t i = par->num + 1; i < par->count; i += mProcessingSettings.nDeviceHelperThreads + 1) {
71 // if (mProcessingSettings.debugLevel >= 3) GPUInfo("\tHelper Thread %d Running, Slice %d+%d, Phase %d", par->num, i, par->phase);
72 if ((par->functionCls->*par->function)(i, par->num + 1, par)) {
73 par->error = 1;
74 }
75 if (par->reset) {
76 break;
77 }
78 par->done = i + 1;
79 // if (mProcessingSettings.debugLevel >= 3) GPUInfo("\tHelper Thread %d Finished, Slice %d+%d, Phase %d", par->num, i, par->phase);
80 }
81 ResetThisHelperThread(par);
82 par->mutex[0].lock();
83 }
84 if (mProcessingSettings.debugLevel >= 3) {
85 GPUInfo("\tHelper thread %d terminating", par->num);
86 }
87 par->mutex[1].unlock();
88 pthread_exit(nullptr);
89 return (nullptr);
90}

101int32_t GPUReconstructionDeviceBase::GetGlobalLock(void*& pLock)
102{
103#ifdef _WIN32
104 HANDLE* semLock = new HANDLE;
105 *semLock = CreateSemaphore(nullptr, 1, 1, SemLockName);
106 if (*semLock == nullptr) {
107 GPUError("Error creating GPUInit Semaphore");
108 return (1);
109 }
110 WaitForSingleObject(*semLock, INFINITE);
111#elif !defined(__APPLE__) // GPU not supported on MacOS anyway
112 sem_t* semLock = sem_open(SemLockName, O_CREAT, 0x01B6, 1);
113 if (semLock == SEM_FAILED) {
114 GPUError("Error creating GPUInit Semaphore");
115 return (1);
116 }
117 timespec semtime;
118 clock_gettime(CLOCK_REALTIME, &semtime);
119 semtime.tv_sec += 10;
120 while (sem_timedwait(semLock, &semtime) != 0) {
121 GPUError("Global Lock for GPU initialisation was not released for 10 seconds, assuming another thread died");
122 GPUWarning("Resetting the global lock");
123 sem_post(semLock);
124 }
125#else
126 void* semLock = nullptr;
127#endif
128 pLock = semLock;
129 return 0;
130}

132void GPUReconstructionDeviceBase::ReleaseGlobalLock(void* sem)
133{
134// Release the global named semaphore that locks GPU Initialization
135#ifdef _WIN32
136 HANDLE* h = (HANDLE*)sem;
137 ReleaseSemaphore(*h, 1, nullptr);
138 CloseHandle(*h);
139 delete h;
140#elif !defined(__APPLE__) // GPU not supported on MacOS anyway
141 sem_t* pSem = (sem_t*)sem;
142 sem_post(pSem);
143 sem_unlink(SemLockName);
144#endif
145}

147void GPUReconstructionDeviceBase::ResetHelperThreads(int32_t helpers)
148{
149 GPUImportant("Error occurred, GPU tracker helper threads will be reset (Number of threads %d (%d))", mProcessingSettings.nDeviceHelperThreads, mNSlaveThreads);
150 SynchronizeGPU();
151 for (int32_t i = 0; i < mProcessingSettings.nDeviceHelperThreads; i++) {
152 mHelperParams[i].reset = true;
153 if (helpers || i >= mProcessingSettings.nDeviceHelperThreads) {
154 pthread_mutex_lock(&((pthread_mutex_t*)mHelperParams[i].mutex)[1]);
155 }
156 }
157 GPUImportant("GPU Tracker helper threads have ben reset");
158}

160int32_t GPUReconstructionDeviceBase::StartHelperThreads()
161{
162 int32_t nThreads = mProcessingSettings.nDeviceHelperThreads;
163 if (nThreads) {
164 mHelperParams = new GPUReconstructionHelpers::helperParam[nThreads];
165 if (mHelperParams == nullptr) {
166 GPUError("Memory allocation error");
167 ExitDevice();
168 return (1);
169 }
170 for (int32_t i = 0; i < nThreads; i++) {
171 mHelperParams[i].cls = this;
172 mHelperParams[i].terminate = false;
173 mHelperParams[i].reset = false;
174 mHelperParams[i].num = i;
175 for (int32_t j = 0; j < 2; j++) {
176 mHelperParams[i].mutex[j].lock();
177 }
178
179 if (pthread_create(&mHelperParams[i].threadId, nullptr, helperWrapper_static, &mHelperParams[i])) {
180 GPUError("Error starting slave thread");
181 ExitDevice();
182 return (1);
183 }
184 }
185 }
186 mNSlaveThreads = nThreads;
187 return (0);
188}

190int32_t GPUReconstructionDeviceBase::StopHelperThreads()
191{
192 if (mNSlaveThreads) {
193 for (int32_t i = 0; i < mNSlaveThreads; i++) {
194 mHelperParams[i].terminate = true;
195 mHelperParams[i].mutex[0].unlock();
196 mHelperParams[i].mutex[1].lock();
197 if (pthread_join(mHelperParams[i].threadId, nullptr)) {
198 GPUError("Error waiting for thread to terminate");
199 return (1);
200 }
201 }
202 delete[] mHelperParams;
203 }
204 mNSlaveThreads = 0;
205 return (0);
206}

227int32_t GPUReconstructionDeviceBase::InitDevice()
228{
229 // cpu_set_t mask;
230 // CPU_ZERO(&mask);
231 // CPU_SET(0, &mask);
232 // sched_setaffinity(0, sizeof(mask), &mask);
233
234 if (mProcessingSettings.memoryAllocationStrategy == GPUMemoryResource::ALLOCATION_INDIVIDUAL) {
235 GPUError("Individual memory allocation strategy unsupported for device\n");
236 return (1);
237 }
238 if (mProcessingSettings.nStreams > GPUCA_MAX_STREAMS) {
239 GPUError("Too many straems requested %d > %d\n", mProcessingSettings.nStreams, GPUCA_MAX_STREAMS);
240 return (1);
241 }
242 mThreadId = GetThread();
243
244 void* semLock = nullptr;
245 if (mProcessingSettings.globalInitMutex && GetGlobalLock(semLock)) {
246 return (1);
247 }
248
249 if (mProcessingSettings.deviceTimers) {
250 AddGPUEvents(mDebugEvents);
251 }
252
253 int32_t retVal = InitDevice_Runtime();
254 if (retVal) {
255 GPUImportant("GPU Tracker initialization failed");
256 return (1);
257 }
258
259 if (mProcessingSettings.globalInitMutex) {
260 ReleaseGlobalLock(semLock);
261 }
262
263 mDeviceMemoryPermanent = mDeviceMemoryBase;
264 mHostMemoryPermanent = mHostMemoryBase;
265 ClearAllocatedMemory();
266
267 mProcShadow.InitGPUProcessor(this, GPUProcessor::PROCESSOR_TYPE_SLAVE);
268 mProcShadow.mMemoryResProcessors = RegisterMemoryAllocation(&mProcShadow, &GPUProcessorProcessors::SetPointersDeviceProcessor, GPUMemoryResource::MEMORY_PERMANENT | GPUMemoryResource::MEMORY_HOST, "Processors");
269 AllocateRegisteredMemory(mProcShadow.mMemoryResProcessors);
270
271 if (StartHelperThreads()) {
272 return (1);
273 }
274
275 if (mMaster == nullptr || mProcessingSettings.debugLevel >= 2) {
276 GPUInfo("GPU Tracker initialization successfull"); // Verbosity reduced because GPU backend will print GPUImportant message!
277 }
278
279 return (retVal);
280}
