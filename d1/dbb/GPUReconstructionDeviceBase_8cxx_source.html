51void* GPUReconstructionDeviceBase::helperWrapper(GPUReconstructionHelpers::helperParam* par)
52{
53 if (mProcessingSettings.debugLevel >= 3) {
54 GPUInfo("\tHelper thread %d starting", par->num);
55 }
56
57 // cpu_set_t mask; //TODO add option
58 // CPU_ZERO(&mask);
59 // CPU_SET(par->num * 2 + 2, &mask);
60 // sched_setaffinity(0, sizeof(mask), &mask);
61
62 par->mutex[0].lock();
63 while (par->terminate == false) {
64 for (int32_t i = par->num + 1; i < par->count; i += mProcessingSettings.nDeviceHelperThreads + 1) {
65 // if (mProcessingSettings.debugLevel >= 3) GPUInfo("\tHelper Thread %d Running, Slice %d+%d, Phase %d", par->num, i, par->phase);
66 if ((par->functionCls->*par->function)(i, par->num + 1, par)) {
67 par->error = 1;
68 }
69 if (par->reset) {
70 break;
71 }
72 par->done = i + 1;
73 // if (mProcessingSettings.debugLevel >= 3) GPUInfo("\tHelper Thread %d Finished, Slice %d+%d, Phase %d", par->num, i, par->phase);
74 }
75 ResetThisHelperThread(par);
76 par->mutex[0].lock();
77 }
78 if (mProcessingSettings.debugLevel >= 3) {
79 GPUInfo("\tHelper thread %d terminating", par->num);
80 }
81 par->mutex[1].unlock();
82 pthread_exit(nullptr);
83 return (nullptr);
84}

95int32_t GPUReconstructionDeviceBase::GetGlobalLock(void*& pLock)
96{
97#ifdef _WIN32
98 HANDLE* semLock = new HANDLE;
99 *semLock = CreateSemaphore(nullptr, 1, 1, SemLockName);
100 if (*semLock == nullptr) {
101 GPUError("Error creating GPUInit Semaphore");
102 return (1);
103 }
104 WaitForSingleObject(*semLock, INFINITE);
105#elif !defined(__APPLE__) // GPU not supported on MacOS anyway
106 sem_t* semLock = sem_open(SemLockName, O_CREAT, 0x01B6, 1);
107 if (semLock == SEM_FAILED) {
108 GPUError("Error creating GPUInit Semaphore");
109 return (1);
110 }
111 timespec semtime;
112 clock_gettime(CLOCK_REALTIME, &semtime);
113 semtime.tv_sec += 10;
114 while (sem_timedwait(semLock, &semtime) != 0) {
115 GPUError("Global Lock for GPU initialisation was not released for 10 seconds, assuming another thread died");
116 GPUWarning("Resetting the global lock");
117 sem_post(semLock);
118 }
119#else
120 void* semLock = nullptr;
121#endif
122 pLock = semLock;
123 return 0;
124}

126void GPUReconstructionDeviceBase::ReleaseGlobalLock(void* sem)
127{
128// Release the global named semaphore that locks GPU Initialization
129#ifdef _WIN32
130 HANDLE* h = (HANDLE*)sem;
131 ReleaseSemaphore(*h, 1, nullptr);
132 CloseHandle(*h);
133 delete h;
134#elif !defined(__APPLE__) // GPU not supported on MacOS anyway
135 sem_t* pSem = (sem_t*)sem;
136 sem_post(pSem);
137 sem_unlink(SemLockName);
138#endif
139}

141void GPUReconstructionDeviceBase::ResetHelperThreads(int32_t helpers)
142{
143 GPUImportant("Error occurred, GPU tracker helper threads will be reset (Number of threads %d (%d))", mProcessingSettings.nDeviceHelperThreads, mNSlaveThreads);
144 SynchronizeGPU();
145 for (int32_t i = 0; i < mProcessingSettings.nDeviceHelperThreads; i++) {
146 mHelperParams[i].reset = true;
147 if (helpers || i >= mProcessingSettings.nDeviceHelperThreads) {
148 pthread_mutex_lock(&((pthread_mutex_t*)mHelperParams[i].mutex)[1]);
149 }
150 }
151 GPUImportant("GPU Tracker helper threads have ben reset");
152}

154int32_t GPUReconstructionDeviceBase::StartHelperThreads()
155{
156 int32_t nThreads = mProcessingSettings.nDeviceHelperThreads;
157 if (nThreads) {
158 mHelperParams = new GPUReconstructionHelpers::helperParam[nThreads];
159 if (mHelperParams == nullptr) {
160 GPUError("Memory allocation error");
161 ExitDevice();
162 return (1);
163 }
164 for (int32_t i = 0; i < nThreads; i++) {
165 mHelperParams[i].cls = this;
166 mHelperParams[i].terminate = false;
167 mHelperParams[i].reset = false;
168 mHelperParams[i].num = i;
169 for (int32_t j = 0; j < 2; j++) {
170 mHelperParams[i].mutex[j].lock();
171 }
172
173 if (pthread_create(&mHelperParams[i].threadId, nullptr, helperWrapper_static, &mHelperParams[i])) {
174 GPUError("Error starting slave thread");
175 ExitDevice();
176 return (1);
177 }
178 }
179 }
180 mNSlaveThreads = nThreads;
181 return (0);
182}

184int32_t GPUReconstructionDeviceBase::StopHelperThreads()
185{
186 if (mNSlaveThreads) {
187 for (int32_t i = 0; i < mNSlaveThreads; i++) {
188 mHelperParams[i].terminate = true;
189 mHelperParams[i].mutex[0].unlock();
190 mHelperParams[i].mutex[1].lock();
191 if (pthread_join(mHelperParams[i].threadId, nullptr)) {
192 GPUError("Error waiting for thread to terminate");
193 return (1);
194 }
195 }
196 delete[] mHelperParams;
197 }
198 mNSlaveThreads = 0;
199 return (0);
200}

221int32_t GPUReconstructionDeviceBase::InitDevice()
222{
223 // cpu_set_t mask;
224 // CPU_ZERO(&mask);
225 // CPU_SET(0, &mask);
226 // sched_setaffinity(0, sizeof(mask), &mask);
227
228 if (mProcessingSettings.memoryAllocationStrategy == GPUMemoryResource::ALLOCATION_INDIVIDUAL) {
229 GPUError("Individual memory allocation strategy unsupported for device\n");
230 return (1);
231 }
232 if (mProcessingSettings.nStreams > GPUCA_MAX_STREAMS) {
233 GPUError("Too many straems requested %d > %d\n", mProcessingSettings.nStreams, GPUCA_MAX_STREAMS);
234 return (1);
235 }
236 mThreadId = GetThread();
237
238 void* semLock = nullptr;
239 if (mProcessingSettings.globalInitMutex && GetGlobalLock(semLock)) {
240 return (1);
241 }
242
243 if (mProcessingSettings.deviceTimers) {
244 AddGPUEvents(mDebugEvents);
245 }
246
247 int32_t retVal = InitDevice_Runtime();
248 if (retVal) {
249 GPUImportant("GPU Tracker initialization failed");
250 return (1);
251 }
252
253 if (mProcessingSettings.globalInitMutex) {
254 ReleaseGlobalLock(semLock);
255 }
256
257 mDeviceMemoryPermanent = mDeviceMemoryBase;
258 mHostMemoryPermanent = mHostMemoryBase;
259 ClearAllocatedMemory();
260
261 mProcShadow.InitGPUProcessor(this, GPUProcessor::PROCESSOR_TYPE_SLAVE);
262 mProcShadow.mMemoryResProcessors = RegisterMemoryAllocation(&mProcShadow, &GPUProcessorProcessors::SetPointersDeviceProcessor, GPUMemoryResource::MEMORY_PERMANENT | GPUMemoryResource::MEMORY_HOST, "Processors");
263 AllocateRegisteredMemory(mProcShadow.mMemoryResProcessors);
264
265 if (StartHelperThreads()) {
266 return (1);
267 }
268
269 if (mMaster == nullptr || mProcessingSettings.debugLevel >= 2) {
270 GPUInfo("GPU Tracker initialization successfull"); // Verbosity reduced because GPU backend will print GPUImportant message!
271 }
272
273 return (retVal);
274}
