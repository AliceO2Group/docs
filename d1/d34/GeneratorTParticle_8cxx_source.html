93 Bool_t GeneratorTParticle::generateEvent()
94 {
95  // If this is the first entry, and we're executing a command, then
96  // wait until the input file exists and actually contain some data.
97  if (mEntry == 0 and not mCmd.empty()) {
98  waitForData(mTemporary);
99  }
100 
101  // Read in the next entry in the chain
102  int read = mChain->GetEntry(mEntry);
103  mEntry++;
104 
105  // If we got an error while reading, then give error message
106  if (read < 0) {
107  LOG(error) << "Failed to read entry " << mEntry << " of chain";
108  }
109 
110  // If we had an error or nothing was read back, then return false
111  if (read <= 0) {
112  return false;
113  }
114 
115  return true;
116 }

118 Bool_t GeneratorTParticle::importParticles()
119 {

122  auto statusCode = particle->GetStatusCode();
123  if (!mcgenstatus::isEncoded(statusCode)) {
124  statusCode = mcgenstatus::MCGenStatusEncoding(statusCode, 0)
125  .fullEncoding;
126  }
127 
128  mParticles.emplace_back(particle->GetPdgCode(),
129  statusCode,

131  particle->GetSecondMother(),
132  particle->GetFirstDaughter(),
133  particle->GetLastDaughter(),
134  particle->Px(),
135  particle->Py(),
136  particle->Pz(),
137  particle->Energy(),
138  particle->Vx(),
139  particle->Vy(),
140  particle->Vz(),
141  particle->T());
142  auto& tgt = mParticles[mParticles.size() - 1];
143  tgt.SetPolarTheta(particle->GetPolarTheta());
144  tgt.SetPolarPhi(particle->GetPolarPhi());
145  tgt.SetCalcMass(particle->GetCalcMass());
