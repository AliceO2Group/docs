27struct ChargePos {
28 tpccf::GlobalPad gpad;
29 tpccf::TPCFragmentTime timePadded;

31 GPUdDefault() ChargePos() CON_DEFAULT;
32
33 constexpr GPUhdi() ChargePos(tpccf::Row row, tpccf::Pad pad, tpccf::TPCFragmentTime t)
34 : gpad(tpcGlobalPadIdx(row, pad)), timePadded(t + GPUCF_PADDING_TIME)
35 {
36 }
37
38 GPUdi() ChargePos(const tpccf::GlobalPad& p, const tpccf::TPCFragmentTime& t) : gpad(p), timePadded(t) {}
39
40 GPUdi() ChargePos delta(const tpccf::Delta2& d) const
41 {
42 return {tpccf::GlobalPad(gpad + d.x), tpccf::TPCFragmentTime(timePadded + d.y)};
43 }
44
45 GPUdi() bool valid() const { return timePadded >= 0; }
46
47 GPUdi() tpccf::Row row() const { return gpad / TPC_PADS_PER_ROW_PADDED; }
48 GPUdi() tpccf::Pad pad() const { return gpad % TPC_PADS_PER_ROW_PADDED - GPUCF_PADDING_PAD; }
49 GPUdi() tpccf::TPCFragmentTime time() const { return timePadded - GPUCF_PADDING_TIME; }
50
51 private:
52 // Maps the position of a pad given as row and index in that row to a unique
53 // index between 0 and TPC_NUM_OF_PADS.
54 static constexpr GPUdi() tpccf::GlobalPad tpcGlobalPadIdx(tpccf::Row row, tpccf::Pad pad)
55 {
56 return TPC_PADS_PER_ROW_PADDED * row + pad + GPUCF_PADDING_PAD;
57 }
58};
