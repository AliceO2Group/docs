314 void TPCFastSpaceChargeCorrection::finishConstruction()
315 {
317 
318  assert(mConstructionMask & ConstructionState::InProgress);
319 
320  for (int i = 0; i < mGeo.getNumberOfRows(); i++) {
321  assert(mConstructionRowInfos[i].splineScenarioID >= 0);
322  }
323  for (int i = 0; i < mNumberOfScenarios; i++) {
324  assert(mConstructionScenarios[i].isConstructed());
325  }
326 
327  // organize memory for the flat buffer and caculate its size
328 
329  size_t rowsOffset = 0;
330  size_t rowsSize = sizeof(RowInfo) * mGeo.getNumberOfRows();
331 
332  size_t sliceRowsOffset = rowsSize;
333  size_t sliceRowsSize = sizeof(SliceRowInfo) * mGeo.getNumberOfRows() * mGeo.getNumberOfSlices();
334 
335  size_t scOffset = alignSize(sliceRowsOffset + sliceRowsSize, SplineType::getClassAlignmentBytes());
336  size_t scSize = sizeof(SplineType) * mNumberOfScenarios;
337 
338  size_t scBufferOffsets[mNumberOfScenarios];
339 
340  scBufferOffsets[0] = alignSize(scOffset + scSize, SplineType::getBufferAlignmentBytes());
341  size_t scBufferSize = 0;
342  for (int i = 0; i < mNumberOfScenarios; i++) {
343  SplineType& sp = mConstructionScenarios[i];
344  scBufferOffsets[i] = scBufferOffsets[0] + scBufferSize;
345  scBufferSize = alignSize(scBufferSize + sp.getFlatBufferSize(), sp.getBufferAlignmentBytes());
346  }
347  size_t bufferSize = scBufferOffsets[0] + scBufferSize;
348  size_t sliceDataOffset[3];
349  for (int is = 0; is < 3; is++) {
350  sliceDataOffset[is] = alignSize(bufferSize, SplineType::getParameterAlignmentBytes());
351  mSliceDataSizeBytes[is] = 0;
352  for (int i = 0; i < mGeo.getNumberOfRows(); i++) {
353  RowInfo& row = mConstructionRowInfos[i];
354  SplineType& spline = mConstructionScenarios[row.splineScenarioID];
355  row.dataOffsetBytes[is] = alignSize(mSliceDataSizeBytes[is], SplineType::getParameterAlignmentBytes());
356  mSliceDataSizeBytes[is] = row.dataOffsetBytes[is] + spline.getSizeOfParameters();
357  }
358  mSliceDataSizeBytes[is] = alignSize(mSliceDataSizeBytes[is], SplineType::getParameterAlignmentBytes());
359  bufferSize = sliceDataOffset[is] + mSliceDataSizeBytes[is] * mGeo.getNumberOfSlices();
360  }
361 
362  FlatObject::finishConstruction(bufferSize);
363 
364  mRowInfoPtr = reinterpret_cast<RowInfo*>(mFlatBufferPtr + rowsOffset);
365  for (int i = 0; i < mGeo.getNumberOfRows(); i++) {
366  mRowInfoPtr[i] = mConstructionRowInfos[i];
367  }
368 
369  mSliceRowInfoPtr = reinterpret_cast<SliceRowInfo*>(mFlatBufferPtr + sliceRowsOffset);
370  for (int s = 0; s < mGeo.getNumberOfSlices(); s++) {
371  for (int r = 0; r < mGeo.getNumberOfRows(); r++) {
372  mSliceRowInfoPtr[s * mGeo.getNumberOfRows() + r].gridCorrU0 = 0.;
373  mSliceRowInfoPtr[s * mGeo.getNumberOfRows() + r].scaleCorrUtoGrid = 0.;
374  mSliceRowInfoPtr[s * mGeo.getNumberOfRows() + r].scaleCorrVtoGrid = 0.;
375  }
376  }
377 
378  mScenarioPtr = reinterpret_cast<SplineType*>(mFlatBufferPtr + scOffset);
379 
380  for (int i = 0; i < mNumberOfScenarios; i++) {
381  SplineType& sp0 = mConstructionScenarios[i];
382  SplineType& sp1 = mScenarioPtr[i];
383  new (&sp1) SplineType(); // first, call a constructor
384  sp1.cloneFromObject(sp0, mFlatBufferPtr + scBufferOffsets[i]);
385  }
386 
387  for (int is = 0; is < 3; is++) {
388  mSplineData[is] = reinterpret_cast<char*>(mFlatBufferPtr + sliceDataOffset[is]);
389  }
390  releaseConstructionMemory();
391 
392  mTimeStamp = -1;
393 
394  setNoCorrection();
395 }

397 GPUd() void TPCFastSpaceChargeCorrection::setNoCorrection()
398 {
399  // initialise all corrections to 0.
400  for (int slice = 0; slice < mGeo.getNumberOfSlices(); slice++) {
401  double vLength = (slice < mGeo.getNumberOfSlicesA()) ? mGeo.getTPCzLengthA() : mGeo.getTPCzLengthC();
402  SliceInfo& sliceInfo = getSliceInfo(slice);
403  sliceInfo.vMax = vLength;
404  for (int row = 0; row < mGeo.getNumberOfRows(); row++) {
405  const SplineType& spline = getSpline(slice, row);
406 
407  for (int is = 0; is < 3; is++) {
408  float* data = getSplineData(slice, row, is);
409  int nPar = spline.getNumberOfParameters();
410  if (is == 1) {
411  nPar = nPar / 3;
412  }
413  if (is == 2) {
414  nPar = nPar * 2 / 3;
415  }
416  for (int i = 0; i < nPar; i++) {
417  data[i] = 0.f;
418  }
419  }
420 
421  SliceRowInfo& info = getSliceRowInfo(slice, row);
422  RowActiveArea& area = info.activeArea;
423  for (int i = 1; i < 5; i++) {
424  area.maxDriftLengthCheb[i] = 0;
425  }
426  area.maxDriftLengthCheb[0] = vLength;
427  area.cuMin = mGeo.convPadToU(row, 0.f);
428  area.cuMax = -area.cuMin;
429  area.vMax = vLength;
430  area.cvMax = vLength;
431  info.gridV0 = 0.f;
432  info.gridCorrU0 = area.cuMin;
433  info.gridCorrV0 = info.gridV0;
434  info.scaleCorrUtoGrid = spline.getGridX1().getUmax() / (area.cuMax - area.cuMin);
435  info.scaleCorrVtoGrid = spline.getGridX2().getUmax() / area.cvMax;
436  } // row
437  } // slice
438 }

456 double TPCFastSpaceChargeCorrection::testInverse(bool prn)
457 {
458  if (prn) {
459  LOG(info) << "Test inverse transform ";
460  }
461 
462  double tpcR2min = mGeo.getRowInfo(0).x - 1.;
463  tpcR2min = tpcR2min * tpcR2min;
464  double tpcR2max = mGeo.getRowInfo(mGeo.getNumberOfRows() - 1).x;
465  tpcR2max = tpcR2max / cos(2 * M_PI / mGeo.getNumberOfSlicesA() / 2) + 1.;
466  tpcR2max = tpcR2max * tpcR2max;
467 
468  double maxDtpc[3] = {0, 0, 0};
469  double maxD = 0;
470 
471  for (int slice = 0; slice < mGeo.getNumberOfSlices(); slice++) {
472  if (prn) {
473  LOG(info) << "check inverse transform for slice " << slice;
474  }
475  double vLength = (slice < mGeo.getNumberOfSlicesA()) ? mGeo.getTPCzLengthA() : mGeo.getTPCzLengthC();
476  double maxDslice[3] = {0, 0, 0};
477  for (int row = 0; row < mGeo.getNumberOfRows(); row++) {
478  float u0, u1, v0, v1;
479  mGeo.convScaledUVtoUV(slice, row, 0., 0., u0, v0);
480  mGeo.convScaledUVtoUV(slice, row, 1., 1., u1, v1);
481  double x = mGeo.getRowInfo(row).x;
482  double stepU = (u1 - u0) / 100.;
483  double stepV = (v1 - v0) / 100.;
484  double maxDrow[3] = {0, 0, 0};
485  for (double u = u0; u < u1; u += stepU) {
486  for (double v = v0; v < v1; v += stepV) {
487  float dx, du, dv;
488  getCorrection(slice, row, u, v, dx, du, dv);
489  double cx = x + dx;
490  double cu = u + du;
491  double cv = v + dv;
492  double r2 = cx * cx + cu * cu;
493  if (cv < 0 || cv > vLength || r2 < tpcR2min || r2 > tpcR2max) {
494  continue;
495  }
496  float nx, nu, nv;
497  getCorrectionInvCorrectedX(slice, row, cu, cv, nx);
498  getCorrectionInvUV(slice, row, cu, cv, nu, nv);
499  double d[3] = {nx - cx, nu - u, nv - v};
500  for (int i = 0; i < 3; i++) {
501  if (fabs(d[i]) > fabs(maxDrow[i])) {
502  maxDrow[i] = d[i];
503  }
504  }
505 
506  if (0 && prn && fabs(d[0]) + fabs(d[1]) + fabs(d[2]) > 0.1) {
507  LOG(info) << nx - cx << " " << nu - u << " " << nv - v
508  << " x,u,v " << x << ", " << u << ", " << v
509  << " dx,du,dv " << cx - x << ", " << cu - u << ", " << cv - v
510  << " nx,nu,nv " << nx - x << ", " << cu - nu << ", " << cv - nv;
511  }
512  }
513  }
514  if (0 && prn) {
515  LOG(info) << "slice " << slice << " row " << row
516  << " dx " << maxDrow[0] << " du " << maxDrow[1] << " dv " << maxDrow[2];
517  }
518  for (int i = 0; i < 3; i++) {
519  if (fabs(maxDslice[i]) < fabs(maxDrow[i])) {
520  maxDslice[i] = maxDrow[i];
521  }
522  if (fabs(maxDtpc[i]) < fabs(maxDrow[i])) {
523  maxDtpc[i] = maxDrow[i];
524  }
525  if (fabs(maxD) < fabs(maxDrow[i])) {
526  maxD = maxDrow[i];
527  }
528  }
529  }
530  if (prn) {
531  LOG(info) << "inverse correction: slice " << slice
532  << " dx " << maxDslice[0] << " du " << maxDslice[1] << " dv " << maxDslice[2];
533  }
534  } // slice
535 
536  LOG(info) << "Test inverse TPC correction. max deviations: "
537  << " dx " << maxDtpc[0] << " du " << maxDtpc[1] << " dv " << maxDtpc[2] << " cm";
538 
539  return maxD;
540 }
