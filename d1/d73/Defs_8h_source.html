152template <typename T>
153class Enum
154{
155 public:
156 class Iterator
157 {
158 public:
159 Iterator(int value) : m_value(value) {}
160
161 T operator*() const { return (T)m_value; }
162
163 void operator++() { ++m_value; }
164
165 bool operator!=(Iterator rhs) { return m_value != rhs.m_value; }
166
167 private:
168 int m_value;
169 };
170};
