31  ComputingQuotaConsumer dispose2MB = [bs = 2000000](int taskId,
32  std::array<ComputingQuotaOffer, 16>& offers,
33  ComputingQuotaStats& stats,
34  std::function<void(ComputingQuotaOffer const&, ComputingQuotaStats&)> accountDisposed) {
35  ComputingQuotaOffer disposed;
36  disposed.sharedMemory = 0;
37  int64_t bytesSent = bs;
38  for (size_t oi = 0; oi < offers.size(); oi++) {
39  auto& offer = offers[oi];
40  if (offer.user != taskId) {
41  continue;
42  }
43  int64_t toRemove = std::min((int64_t)bytesSent, offer.sharedMemory);
44  offer.sharedMemory -= toRemove;
45  bytesSent -= toRemove;
46  disposed.sharedMemory += toRemove;
47  if (bytesSent <= 0) {
48  break;
49  }
50  }
51  return accountDisposed(disposed, stats);
52  };
