47 void AnalysisSupportHelpers::addMissingOutputsToReader(std::vector<OutputSpec> const& providedOutputs,
48  std::vector<InputSpec> const& requestedInputs,
49  DataProcessorSpec& publisher)
50 {
51  auto matchingOutputFor = [](InputSpec const& requested) {
52  return [&requested](OutputSpec const& provided) {
53  return DataSpecUtils::match(requested, provided);
54  };
55  };
56  for (InputSpec const& requested : requestedInputs) {
57  auto provided = std::find_if(providedOutputs.begin(),
58  providedOutputs.end(),
59  matchingOutputFor(requested));
60 
61  if (provided != providedOutputs.end()) {
62  continue;
63  }
64 
65  auto inList = std::find_if(publisher.outputs.begin(),
66  publisher.outputs.end(),
67  matchingOutputFor(requested));
68  if (inList != publisher.outputs.end()) {
69  continue;
70  }
71 
72  auto concrete = DataSpecUtils::asConcreteDataMatcher(requested);
73  publisher.outputs.emplace_back(concrete.origin, concrete.description, concrete.subSpec, requested.lifetime, requested.metadata);
74  }
75 }

77 void AnalysisSupportHelpers::addMissingOutputsToSpawner(std::vector<OutputSpec> const& providedSpecials,
78  std::vector<InputSpec> const& requestedSpecials,
79  std::vector<InputSpec>& requestedAODs,
80  DataProcessorSpec& publisher)
81 {
82  for (auto& input : requestedSpecials) {
83  if (std::any_of(providedSpecials.begin(), providedSpecials.end(), [&input](auto const& x) {
84  return DataSpecUtils::match(input, x);
85  })) {
86  continue;
87  }
88  auto concrete = DataSpecUtils::asConcreteDataMatcher(input);
89  publisher.outputs.emplace_back(concrete.origin, concrete.description, concrete.subSpec);
90  for (auto& i : input.metadata) {
91  if ((i.type == VariantType::String) && (i.name.find("input:") != std::string::npos)) {
92  auto spec = DataSpecUtils::fromMetadataString(i.defaultValue.get<std::string>());
93  auto j = std::find(publisher.inputs.begin(), publisher.inputs.end(), spec);
94  if (j == publisher.inputs.end()) {
95  publisher.inputs.push_back(spec);
96  }
97  DataSpecUtils::updateInputList(requestedAODs, std::move(spec));
98  }
99  }
100  }

132 DataProcessorSpec AnalysisSupportHelpers::getOutputObjHistSink(std::vector<OutputObjectInfo> const& objmap, std::vector<OutputTaskInfo> const& tskmap)
133 {
134  auto writerFunction = [objmap, tskmap](InitContext& ic) -> std::function<void(ProcessingContext&)> {
135  auto& callbacks = ic.services().get<CallbackService>();
136  auto inputObjects = std::make_shared<std::vector<std::pair<InputObjectRoute, InputObject>>>();
137 
138  static TFile* f[OutputObjHandlingPolicy::numPolicies];
139  for (auto i = 0u; i < OutputObjHandlingPolicy::numPolicies; ++i) {
140  f[i] = nullptr;
141  }
142 
143  static std::string currentDirectory = "";
144  static std::string currentFile = "";
145 
146  auto endofdatacb = [inputObjects](EndOfStreamContext& context) {
147  LOG(debug) << "Writing merged objects and histograms to file";
148  if (inputObjects->empty()) {
149  LOG(error) << "Output object map is empty!";
150  context.services().get<ControlService>().readyToQuit(QuitRequest::Me);
151  return;
152  }
153  for (auto i = 0u; i < OutputObjHandlingPolicy::numPolicies; ++i) {
154  if (f[i] != nullptr) {
155  f[i]->Close();
156  }
157  }
158  LOG(debug) << "All outputs merged in their respective target files";
159  context.services().get<ControlService>().readyToQuit(QuitRequest::Me);
160  };
161 
162  callbacks.set<CallbackService::Id::EndOfStream>(endofdatacb);
163  return [inputObjects, objmap, tskmap](ProcessingContext& pc) mutable -> void {
164  auto const& ref = pc.inputs().get("x");
165  if (!ref.header) {
166  LOG(error) << "Header not found";
167  return;
168  }
169  if (!ref.payload) {
170  LOG(error) << "Payload not found";
171  return;
172  }
173  auto datah = o2::header::get<o2::header::DataHeader*>(ref.header);
174  if (!datah) {
175  LOG(error) << "No data header in stack";
176  return;
177  }
178 
179  auto objh = o2::header::get<o2::framework::OutputObjHeader*>(ref.header);
180  if (!objh) {
181  LOG(error) << "No output object header in stack";
182  return;
183  }
184 
185  InputObject obj;
186  FairInputTBuffer tm(const_cast<char*>(ref.payload), static_cast<int>(datah->payloadSize));
187  tm.InitMap();
188  obj.kind = tm.ReadClass();
189  tm.SetBufferOffset(0);
190  tm.ResetMap();
191  if (obj.kind == nullptr) {
192  LOG(error) << "Cannot read class info from buffer.";
193  return;
194  }
195 
196  auto policy = objh->mPolicy;
197  auto sourceType = objh->mSourceType;
198  auto hash = objh->mTaskHash;
199 
200  obj.obj = tm.ReadObjectAny(obj.kind);
201  auto* named = static_cast<TNamed*>(obj.obj);
202  obj.name = named->GetName();
203  auto hpos = std::find_if(tskmap.begin(), tskmap.end(), [&](auto&& x) { return x.id == hash; });
204  if (hpos == tskmap.end()) {
205  LOG(error) << "No task found for hash " << hash;
206  return;
207  }
208  auto taskname = hpos->name;
209  auto opos = std::find_if(objmap.begin(), objmap.end(), [&](auto&& x) { return x.id == hash; });
210  if (opos == objmap.end()) {
211  LOG(error) << "No object list found for task " << taskname << " (hash=" << hash << ")";
212  return;
213  }
214  auto objects = opos->bindings;
215  if (std::find(objects.begin(), objects.end(), obj.name) == objects.end()) {
216  LOG(error) << "No object " << obj.name << " in map for task " << taskname;
217  return;
218  }
219  auto nameHash = runtime_hash(obj.name.c_str());
220  InputObjectRoute key{obj.name, nameHash, taskname, hash, policy, sourceType};
221  auto existing = std::find_if(inputObjects->begin(), inputObjects->end(), [&](auto&& x) { return (x.first.uniqueId == nameHash) && (x.first.taskHash == hash); });
222  // If it's the first one, we just add it to the list.
223  if (existing == inputObjects->end()) {
224  obj.count = objh->mPipelineSize;
225  inputObjects->push_back(std::make_pair(key, obj));
226  existing = inputObjects->end() - 1;
227  } else {
228  obj.count = existing->second.count;
229  // Otherwise, we merge it with the existing one.
230  auto merger = existing->second.kind->GetMerge();
231  if (!merger) {
232  LOG(error) << "Already one unmergeable object found for " << obj.name;
233  return;
234  }
235  TList coll;
236  coll.Add(static_cast<TObject*>(obj.obj));
237  merger(existing->second.obj, &coll, nullptr);
238  }
239  // We expect as many objects as the pipeline size, for
240  // a given object name and task hash.
241  existing->second.count -= 1;
242 
243  if (existing->second.count != 0) {
244  return;
245  }
246  // Write the object here.
247  auto route = existing->first;
248  auto entry = existing->second;
249  auto file = ROOTfileNames.find(route.policy);
250  if (file == ROOTfileNames.end()) {
251  return;
252  }
253  auto filename = file->second;
254  if (f[route.policy] == nullptr) {
255  f[route.policy] = TFile::Open(filename.c_str(), "RECREATE");
256  }
257  auto nextDirectory = route.directory;
258  if ((nextDirectory != currentDirectory) || (filename != currentFile)) {
259  if (!f[route.policy]->FindKey(nextDirectory.c_str())) {
260  f[route.policy]->mkdir(nextDirectory.c_str());
261  }
262  currentDirectory = nextDirectory;
263  currentFile = filename;
264  }
265 
266  // translate the list-structure created by the registry into a directory structure within the file
267  std::function<void(TList*, TDirectory*)> writeListToFile;
268  writeListToFile = [&](TList* list, TDirectory* parentDir) {
269  TIter next(list);
270  TObject* object = nullptr;
271  while ((object = next())) {
272  if (object->InheritsFrom(TList::Class())) {
273  writeListToFile(static_cast<TList*>(object), parentDir->mkdir(object->GetName(), object->GetName(), true));
274  } else {
275  parentDir->WriteObjectAny(object, object->Class(), object->GetName());
276  auto* written = list->Remove(object);
277  delete written;
278  }
279  }
280  };
281 
282  TDirectory* currentDir = f[route.policy]->GetDirectory(currentDirectory.c_str());
283  if (route.sourceType == OutputObjSourceType::HistogramRegistrySource) {
284  auto* outputList = static_cast<TList*>(entry.obj);
285  outputList->SetOwner(false);
286 
287  // if registry should live in dedicated folder a TNamed object is appended to the list
288  if (outputList->Last() && outputList->Last()->IsA() == TNamed::Class()) {
289  delete outputList->Last();
290  outputList->RemoveLast();
291  currentDir = currentDir->mkdir(outputList->GetName(), outputList->GetName(), true);
292  }
293 
294  writeListToFile(outputList, currentDir);
295  outputList->SetOwner();
296  delete outputList;
297  entry.obj = nullptr;
298  } else {
299  currentDir->WriteObjectAny(entry.obj, entry.kind, entry.name.c_str());
300  delete (TObject*)entry.obj;
301  entry.obj = nullptr;
302  }
303  };
304  };
305 
306  char const* name = "internal-dpl-aod-global-analysis-file-sink";
307  // Lifetime is sporadic because we do not ask each analysis task to send its
308  // results every timeframe.
309  DataProcessorSpec spec{
310  .name = name,
311  .inputs = {InputSpec("x", DataSpecUtils::dataDescriptorMatcherFrom(header::DataOrigin{"ATSK"}), Lifetime::Sporadic)},
312  .algorithm = {writerFunction},
313  };
314 
315  return spec;
316 }

318 // add sink for the AODs
319 DataProcessorSpec
320  AnalysisSupportHelpers::getGlobalAODSink(std::shared_ptr<DataOutputDirector> dod,
321  std::vector<InputSpec> const& outputInputs)
322 {
323 
324  auto writerFunction = [dod, outputInputs](InitContext& ic) -> std::function<void(ProcessingContext&)> {
325  LOGP(debug, "======== getGlobalAODSink::Init ==========");
326 
327  // find out if any table needs to be saved
328  bool hasOutputsToWrite = false;
329  for (auto& outobj : outputInputs) {
330  auto ds = dod->getDataOutputDescriptors(outobj);
331  if (ds.size() > 0) {
332  hasOutputsToWrite = true;
333  break;
334  }
335  }
336 
337  // if nothing needs to be saved then return a trivial functor
338  // this happens when nothing needs to be saved but there are dangling outputs
339  if (!hasOutputsToWrite) {
340  return [](ProcessingContext&) mutable -> void {
341  static bool once = false;
342  if (!once) {
343  LOG(info) << "No AODs to be saved.";
344  once = true;
345  }
346  };
347  }
348 
349  // end of data functor is called at the end of the data stream
350  auto endofdatacb = [dod](EndOfStreamContext& context) {
351  dod->closeDataFiles();
352  context.services().get<ControlService>().readyToQuit(QuitRequest::Me);
353  };
354 
355  auto& callbacks = ic.services().get<CallbackService>();
356  callbacks.set<CallbackService::Id::EndOfStream>(endofdatacb);
357 
358  // prepare map<uint64_t, uint64_t>(startTime, tfNumber)
359  std::map<uint64_t, uint64_t> tfNumbers;
360  std::map<uint64_t, std::string> tfFilenames;
361 
362  std::vector<TString> aodMetaDataKeys;
363  std::vector<TString> aodMetaDataVals;
364 
365  // this functor is called once per time frame
366  return [dod, tfNumbers, tfFilenames, aodMetaDataKeys, aodMetaDataVals](ProcessingContext& pc) mutable -> void {
367  LOGP(debug, "======== getGlobalAODSink::processing ==========");
368  LOGP(debug, " processing data set with {} entries", pc.inputs().size());
369 
370  // return immediately if pc.inputs() is empty. This should never happen!
371  if (pc.inputs().size() == 0) {
372  LOGP(info, "No inputs available!");
373  return;
374  }
375 
376  // update tfNumbers
377  uint64_t startTime = 0;
378  uint64_t tfNumber = 0;
379  auto ref = pc.inputs().get("tfn");
380  if (ref.spec && ref.payload) {
381  startTime = DataRefUtils::getHeader<DataProcessingHeader*>(ref)->startTime;
382  tfNumber = pc.inputs().get<uint64_t>("tfn");
383  tfNumbers.insert(std::pair<uint64_t, uint64_t>(startTime, tfNumber));
384  }
385  // update tfFilenames
386  std::string aodInputFile;
387  auto ref2 = pc.inputs().get("tff");
388  if (ref2.spec && ref2.payload) {
389  startTime = DataRefUtils::getHeader<DataProcessingHeader*>(ref2)->startTime;
390  aodInputFile = pc.inputs().get<std::string>("tff");
391  tfFilenames.insert(std::pair<uint64_t, std::string>(startTime, aodInputFile));
392  }
393 
394  // close all output files if one has reached size limit
395  dod->checkFileSizes();
396 
397  // loop over the DataRefs which are contained in pc.inputs()
398  for (const auto& ref : pc.inputs()) {
399  if (!ref.spec) {
400  LOGP(debug, "Invalid input will be skipped!");
401  continue;
402  }
403 
404  // get metadata
405  if (DataSpecUtils::partialMatch(*ref.spec, header::DataDescription("AODMetadataKeys"))) {
406  aodMetaDataKeys = pc.inputs().get<std::vector<TString>>(ref.spec->binding);
407  }
408  if (DataSpecUtils::partialMatch(*ref.spec, header::DataDescription("AODMetadataVals"))) {
409  aodMetaDataVals = pc.inputs().get<std::vector<TString>>(ref.spec->binding);
410  }
411 
412  // skip non-AOD refs
413  if (!DataSpecUtils::partialMatch(*ref.spec, writableAODOrigins)) {
414  continue;
415  }
416  startTime = DataRefUtils::getHeader<DataProcessingHeader*>(ref)->startTime;
417 
418  // does this need to be saved?
419  auto dh = DataRefUtils::getHeader<header::DataHeader*>(ref);
420  auto tableName = dh->dataDescription.as<std::string>();
421  auto ds = dod->getDataOutputDescriptors(*dh);
422  if (ds.empty()) {
423  continue;
424  }
425 
426  // get TF number from startTime
427  auto it = tfNumbers.find(startTime);
428  if (it != tfNumbers.end()) {
429  tfNumber = (it->second / dod->getNumberTimeFramesToMerge()) * dod->getNumberTimeFramesToMerge();
430  } else {
431  LOGP(fatal, "No time frame number found for output with start time {}", startTime);
432  throw std::runtime_error("Processing is stopped!");
433  }
434  // get aod input file from startTime
435  auto it2 = tfFilenames.find(startTime);
436  if (it2 != tfFilenames.end()) {
437  aodInputFile = it2->second;
438  }
439 
440  // get the TableConsumer and corresponding arrow table
441  auto msg = pc.inputs().get(ref.spec->binding);
442  if (msg.header == nullptr) {
443  LOGP(error, "No header for message {}:{}", ref.spec->binding, DataSpecUtils::describe(*ref.spec));
444  continue;
445  }
446  auto s = pc.inputs().get<TableConsumer>(ref.spec->binding);
447  auto table = s->asArrowTable();
448  if (!table->Validate().ok()) {
449  LOGP(warning, "The table \"{}\" is not valid and will not be saved!", tableName);
450  continue;
451  }
452  if (table->schema()->fields().empty()) {
453  LOGP(debug, "The table \"{}\" is empty but will be saved anyway!", tableName);
454  }
455 
456  // loop over all DataOutputDescriptors
457  // a table can be saved in multiple ways
458  // e.g. different selections of columns to different files
459  for (auto d : ds) {
460  auto fileAndFolder = dod->getFileFolder(d, tfNumber, aodInputFile);
461  auto treename = fileAndFolder.folderName + "/" + d->treename;
462  TableToTree ta2tr(table,
463  fileAndFolder.file,
464  treename.c_str());
465 
466  // update metadata
467  if (fileAndFolder.file->FindObjectAny("metaData")) {
468  LOGF(debug, "Metadata: target file %s already has metadata, preserving it", fileAndFolder.file->GetName());
469  } else if (!aodMetaDataKeys.empty() && !aodMetaDataVals.empty()) {
470  TMap aodMetaDataMap;
471  for (uint32_t imd = 0; imd < aodMetaDataKeys.size(); imd++) {
472  aodMetaDataMap.Add(new TObjString(aodMetaDataKeys[imd]), new TObjString(aodMetaDataVals[imd]));
473  }
474  fileAndFolder.file->WriteObject(&aodMetaDataMap, "metaData", "Overwrite");
475  }
476 
477  if (!d->colnames.empty()) {
478  for (auto& cn : d->colnames) {
479  auto idx = table->schema()->GetFieldIndex(cn);
480  auto col = table->column(idx);
481  auto field = table->schema()->field(idx);
482  if (idx != -1) {
483  ta2tr.addBranch(col, field);
484  }
485  }
486  } else {
487  ta2tr.addAllBranches();
488  }
489  ta2tr.process();
490  }
491  }
492  };
493  }; // end of writerFunction
494 
495  // the command line options relevant for the writer are global
496  // see runDataProcessing.h
497  DataProcessorSpec spec{
498  "internal-dpl-aod-writer",
499  outputInputs,
500  Outputs{},
501  AlgorithmSpec(writerFunction),
502  {}};
503 
504  return spec;
505 }
