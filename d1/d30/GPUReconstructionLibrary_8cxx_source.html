48 GPUReconstruction* GPUReconstruction::CreateInstance(const GPUSettingsDeviceBackend& cfg)
49 {
50  GPUReconstruction* retVal = nullptr;
51  DeviceType type = (DeviceType)cfg.deviceType;
52 #ifdef DEBUG_STREAMER
53  if (type != DeviceType::CPU) {
54  GPUError("Cannot create GPUReconstruction for a non-CPU device if DEBUG_STREAMER are enabled");
55  return nullptr;
56  }
57 #endif
58  if (type == DeviceType::CPU) {
59  retVal = GPUReconstruction_Create_CPU(cfg);
60  } else {
61  auto* loader = GetLibraryInstance(type, true);
62  if (loader && (retVal = (*loader)->GetPtr(cfg))) {
63  retVal->mMyLib = *loader;
64  }
65  }
66 
67  if (retVal == nullptr) {
68  if (cfg.forceDeviceType) {
69  GPUError("Error: Could not load GPUReconstruction for specified device: %s (%u)", GPUDataTypes::DEVICE_TYPE_NAMES[type], cfg.deviceType);
70  } else if (type != DeviceType::CPU) {
71  GPUError("Could not load GPUReconstruction for device type %s (%u), falling back to CPU version", GPUDataTypes::DEVICE_TYPE_NAMES[type], cfg.deviceType);
72  GPUSettingsDeviceBackend cfg2 = cfg;
73  cfg2.deviceType = DeviceType::CPU;
74  retVal = CreateInstance(cfg2);
75  }
76  } else {
77  GPUInfo("Created GPUReconstruction instance for device type %s (%u)%s", GPUDataTypes::DEVICE_TYPE_NAMES[type], cfg.deviceType, cfg.master ? " (slave)" : "");
78  }
79 
80  return retVal;
81 }

102 #ifdef HIP_ENABLED

106 #ifdef OPENCL1_ENABLED

110 #ifdef OPENCL2_ENABLED

114  GPUError("Error: Invalid device type %u", (uint32_t)type);

118  GPUInfo("%s Support not compiled in for device type %u (%s)", GPUDataTypes::DEVICE_TYPE_NAMES[type], (uint32_t)type, GPUDataTypes::DEVICE_TYPE_NAMES[type]);
119  }
120  return nullptr;

142 int32_t GPUReconstruction::LibraryLoader::LoadLibrary()
143 {
144  static std::mutex mut;
145  std::lock_guard<std::mutex> lock(mut);
146 
147  if (mGPUEntry) {
148  return 0;
149  }
150 
151  LIBRARY_TYPE hGPULib;
152  hGPULib = LIBRARY_LOAD(mLibName);
153  if (hGPULib == nullptr) {
154 #ifndef _WIN32
155  GPUImportant("The following error occured during dlopen: %s", dlerror());
156 #endif
157  GPUError("Error Opening cagpu library for GPU Tracker (%s)", mLibName);
158  return 1;
159  } else {
160  void* createFunc = LIBRARY_FUNCTION(hGPULib, mFuncName);
161  if (createFunc == nullptr) {
162  GPUError("Error fetching entry function in GPU library\n");
163  LIBRARY_CLOSE(hGPULib);
164  return 1;
165  } else {
166  mGPULib = (void*)(size_t)hGPULib;
167  mGPUEntry = createFunc;
168  GPUInfo("GPU Tracker library loaded and GPU tracker object created sucessfully");
169  }
170  }
171  return 0;
172 }
