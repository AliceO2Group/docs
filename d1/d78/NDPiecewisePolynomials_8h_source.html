452 template <uint32_t Dim, uint32_t Degree, bool InteractionOnly>
453 template <uint32_t DimTmp>
454 GPUdi() void NDPiecewisePolynomials<Dim, Degree, InteractionOnly>::clamp(float x[/* Dim */], int32_t index[/* Dim */]) const
455 {
456  if (index[DimTmp] <= 0) {
457  index[DimTmp] = 0;
458 
459  if (x[DimTmp] < mMin[DimTmp]) {
460  x[DimTmp] = mMin[DimTmp];
461  }
462 
463  } else {
464  if (index[DimTmp] >= int32_t(mN[DimTmp] - 1)) {
465  index[DimTmp] = mN[DimTmp] - 2;
466  x[DimTmp] = mMax[DimTmp];
467  }
468  }
469 
470  if constexpr (DimTmp > 0) {
471  return clamp<DimTmp - 1>(x, index);
472  }
473 }

511 template <uint32_t Dim, uint32_t Degree, bool InteractionOnly>
512 void NDPiecewisePolynomials<Dim, Degree, InteractionOnly>::performFits(const std::function<double(const double x[/* Dim */])>& func, const uint32_t nAuxiliaryPoints[/* Dim */])
513 {
514  const int32_t nTotalFits = getNPolynomials();
515 #ifndef GPUCA_ALIROOT_LIB
516  LOGP(info, "Perform fitting of {}D-Polynomials of degree {} for a total of {} fits.", Dim, Degree, nTotalFits);
517 #endif
518 
519  MultivariatePolynomialHelper<0, 0, false> pol(Dim, Degree, InteractionOnly);
520  TLinearFitter fitter = pol.getTLinearFitter();
521 
522  uint32_t nPoints = 1;
523  for (uint32_t i = 0; i < Dim; ++i) {
524  nPoints *= nAuxiliaryPoints[i];
525  }
526 
527  std::vector<double> xCords;
528  std::vector<double> response;
529  xCords.reserve(Dim * nPoints);
530  response.reserve(nPoints);
531 
532  uint32_t nPolynomials[Dim]{0};
533  for (uint32_t i = 0; i < Dim; ++i) {
534  nPolynomials[i] = getNPolynomials(i);
535  }
536 
537  int32_t pos[Dim + 1]{0};
538  uint32_t counter = 0;
539  const int32_t printDebugForNFits = int32_t(nTotalFits / 20) + 1;
540 
541  for (;;) {
542  const bool debug = !(++counter % printDebugForNFits);
543  if (debug) {
544 #ifndef GPUCA_ALIROOT_LIB
545  LOGP(info, "Performing fit {} out of {}", counter, nTotalFits);
546 #endif
547  }
548 
549  checkPos(nPolynomials, pos);
550 
551  if (pos[Dim] == 1) {
552  break;
553  }
554 
555  xCords.clear();
556  response.clear();
557  fitInnerGrid(func, nAuxiliaryPoints, pos, fitter, xCords, response);
558  ++pos[0];
559  }
560 }

562 template <uint32_t Dim, uint32_t Degree, bool InteractionOnly>
563 void NDPiecewisePolynomials<Dim, Degree, InteractionOnly>::performFits(const std::vector<float>& x, const std::vector<float>& y)
564 {
565  const int32_t nTotalFits = getNPolynomials();
566 #ifndef GPUCA_ALIROOT_LIB
567  LOGP(info, "Perform fitting of {}D-Polynomials of degree {} for a total of {} fits.", Dim, Degree, nTotalFits);
568 #endif
569 
570  // approximate number of points
571  uint32_t nPoints = 2 * y.size() / nTotalFits;
572 
573  // polynomial index -> indices to datapoints
574  std::unordered_map<int32_t, std::vector<size_t>> dataPointsIndices;
575  for (int32_t i = 0; i < nTotalFits; ++i) {
576  dataPointsIndices[i].reserve(nPoints);
577  }
578 
579  // check for each data point which polynomial to use
580  for (size_t i = 0; i < y.size(); ++i) {
581  std::array<int32_t, Dim> index;
582  float xVal[Dim];
583  std::copy(x.begin() + i * Dim, x.begin() + i * Dim + Dim, xVal);
584  setIndex<Dim - 1>(xVal, index.data());
585 
586  std::array<int32_t, Dim> indexClamped{index};
587  clamp<Dim - 1>(xVal, indexClamped.data());
588 
589  // check if data points are in the grid
590  if (index == indexClamped) {
591  // index of the polyniomial
592  const uint32_t idx = getDataIndex(index.data()) / MultivariatePolynomialParametersHelper::getNParameters(Degree, Dim, InteractionOnly);
593 
594  // store index to data point
595  dataPointsIndices[idx].emplace_back(i);
596  }
597  }
598 
599  // for fitting
600  MultivariatePolynomialHelper<0, 0, false> pol(Dim, Degree, InteractionOnly);
601  TLinearFitter fitter = pol.getTLinearFitter();
602 
603  uint32_t counter = 0;
604  const int32_t printDebugForNFits = int32_t(nTotalFits / 20) + 1;
605 
606  // temp storage for x and y values for fitting
607  std::vector<double> xCords;
608  std::vector<double> response;
609 
610  for (int32_t i = 0; i < nTotalFits; ++i) {
611  const bool debug = !(++counter % printDebugForNFits);
612  if (debug) {
613 #ifndef GPUCA_ALIROOT_LIB
614  LOGP(info, "Performing fit {} out of {}", counter, nTotalFits);
615 #endif
616  }
617 
618  // store values for fitting
619  if (dataPointsIndices[i].empty()) {
620 #ifndef GPUCA_ALIROOT_LIB
621  LOGP(info, "No data points to fit");
622 #endif
623  continue;
624  }
625 
626  const auto nP = dataPointsIndices[i].size();
627  xCords.reserve(Dim * nP);
628  response.reserve(nP);
629  xCords.clear();
630  response.clear();
631 
632  // add datapoints to fit
633  for (size_t j = 0; j < nP; ++j) {
634  const size_t idxOrig = dataPointsIndices[i][j];
635 
636  // insert x values at the end of xCords
637  const int32_t idxXStart = idxOrig * Dim;
638  xCords.insert(xCords.end(), x.begin() + idxXStart, x.begin() + idxXStart + Dim);
639  response.emplace_back(y[idxOrig]);
640  }
641 
642  // perform the fit on the points TODO make errors configurable
643  std::vector<double> error;
644  const auto params = MultivariatePolynomialHelper<0, 0, false>::fit(fitter, xCords, response, error, true);
645 
646  // store parameters
647  std::copy(params.begin(), params.end(), &mParams[i * MultivariatePolynomialParametersHelper::getNParameters(Degree, Dim, InteractionOnly)]);
648  }
649 }

651 template <uint32_t Dim, uint32_t Degree, bool InteractionOnly>
652 void NDPiecewisePolynomials<Dim, Degree, InteractionOnly>::fitInnerGrid(const std::function<double(const double x[/* Dim */])>& func, const uint32_t nAuxiliaryPoints[/* Dim */], const int32_t currentIndex[/* Dim */], TLinearFitter& fitter, std::vector<double>& xCords, std::vector<double>& response)
653 {
654  int32_t pos[Dim + 1]{0};
655 
656  // add points which will be used for the fit
657  for (;;) {
658  checkPos(nAuxiliaryPoints, pos);
659 
660  if (pos[Dim] == 1) {
661  break;
662  }
663 
664  for (uint32_t iDim = 0; iDim < Dim; ++iDim) {
665  const double stepWidth = getStepWidth(iDim, nAuxiliaryPoints[iDim]);
666  const double vertexPos = getVertexPosition(currentIndex[iDim], iDim);
667  const double realPosTmp = vertexPos + pos[iDim] * stepWidth;
668  xCords.emplace_back(realPosTmp);
669  }
670 
671  // get response for last added points
672  const double responseTmp = func(&xCords[xCords.size() - Dim]);
673  response.emplace_back(responseTmp);
674  ++pos[0];
675  }
676 
677  // perform the fit on the points TODO make errors configurable
