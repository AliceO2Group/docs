329  setParams(container.mParams.data());

458 template <uint32_t Dim, uint32_t Degree, bool InteractionOnly>
459 template <uint32_t DimTmp>
460 GPUdi() void NDPiecewisePolynomials<Dim, Degree, InteractionOnly>::clamp(float x[/* Dim */], int32_t index[/* Dim */]) const
461 {
462  if (index[DimTmp] <= 0) {
463  index[DimTmp] = 0;
464 
465  if (x[DimTmp] < mMin[DimTmp]) {
466  x[DimTmp] = mMin[DimTmp];
467  }
468 
469  } else {
470  if (index[DimTmp] >= int32_t(mN[DimTmp] - 1)) {
471  index[DimTmp] = mN[DimTmp] - 2;
472  x[DimTmp] = mMax[DimTmp];
473  }
474  }
475 
476  if constexpr (DimTmp > 0) {
477  return clamp<DimTmp - 1>(x, index);
478  }
