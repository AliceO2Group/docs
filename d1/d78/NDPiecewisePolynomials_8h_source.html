307template <uint32_t Dim, uint32_t Degree, bool InteractionOnly>
308void NDPiecewisePolynomials<Dim, Degree, InteractionOnly>::setFromContainer(const NDPiecewisePolynomialContainer& container)
309{
310 if (Dim != container.mDim) {
311#ifndef GPUCA_ALIROOT_LIB
312 LOGP(info, "wrong number of dimensions! this {} container {}", Dim, container.mDim);
313#endif
314 return;
315 }
316 if (Degree != container.mDegree) {
317#ifndef GPUCA_ALIROOT_LIB
318 LOGP(info, "wrong number of degrees! this {} container {}", Degree, container.mDegree);
319#endif
320 return;
321 }
322 if (InteractionOnly != container.mInteractionOnly) {
323#ifndef GPUCA_ALIROOT_LIB
324 LOGP(info, "InteractionOnly is set for this object to {}, but stored as {} in the container", InteractionOnly, container.mInteractionOnly);
325#endif
326 return;
327 }
328 init(container.mMin.data(), container.mMax.data(), container.mN.data());
329 setParams(container.mParams.data());
330}

332template <uint32_t Dim, uint32_t Degree, bool InteractionOnly>
333void NDPiecewisePolynomials<Dim, Degree, InteractionOnly>::setDefault()
334{
335 const auto nParamsPerPol = MultivariatePolynomialParametersHelper::getNParameters(Degree, Dim, InteractionOnly);
336 const auto nPols = getNPolynomials();
337 std::vector<float> params(nParamsPerPol);
338 params.front() = 1;
339 for (auto i = 0; i < nPols; ++i) {
340 std::copy(params.begin(), params.end(), &mParams[i * nParamsPerPol]);
341 }
342}

458template <uint32_t Dim, uint32_t Degree, bool InteractionOnly>
459template <uint32_t DimTmp>
460GPUdi() void NDPiecewisePolynomials<Dim, Degree, InteractionOnly>::clamp(float x[/* Dim */], int32_t index[/* Dim */]) const
461{
462 if (index[DimTmp] <= 0) {
463 index[DimTmp] = 0;
464
465 if (x[DimTmp] < mMin[DimTmp]) {
466 x[DimTmp] = mMin[DimTmp];
467 }
468
469 } else {
470 if (index[DimTmp] >= int32_t(mN[DimTmp] - 1)) {
471 index[DimTmp] = mN[DimTmp] - 2;
472 x[DimTmp] = mMax[DimTmp];
473 }
474 }
475
476 if constexpr (DimTmp > 0) {
477 return clamp<DimTmp - 1>(x, index);
478 }
479}
