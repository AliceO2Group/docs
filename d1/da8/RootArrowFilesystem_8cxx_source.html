48 auto arrowTypeFromROOT(EDataType type, int size)
49 {
50  auto typeGenerator = [](std::shared_ptr<arrow::DataType> const& type, int size) -> std::shared_ptr<arrow::DataType> {
51  switch (size) {
52  case -1:
53  return arrow::list(type);
54  case 1:
55  return std::move(type);
56  default:
57  return arrow::fixed_size_list(type, size);
58  }
59  };
60 
61  switch (type) {
62  case EDataType::kBool_t:
63  return typeGenerator(arrow::boolean(), size);
64  case EDataType::kUChar_t:
65  return typeGenerator(arrow::uint8(), size);
66  case EDataType::kUShort_t:
67  return typeGenerator(arrow::uint16(), size);
68  case EDataType::kUInt_t:
69  return typeGenerator(arrow::uint32(), size);
70  case EDataType::kULong64_t:
71  return typeGenerator(arrow::uint64(), size);
72  case EDataType::kChar_t:
73  return typeGenerator(arrow::int8(), size);
74  case EDataType::kShort_t:
75  return typeGenerator(arrow::int16(), size);
76  case EDataType::kInt_t:
77  return typeGenerator(arrow::int32(), size);
78  case EDataType::kLong64_t:
79  return typeGenerator(arrow::int64(), size);
80  case EDataType::kFloat_t:
81  return typeGenerator(arrow::float32(), size);
82  case EDataType::kDouble_t:
83  return typeGenerator(arrow::float64(), size);
84  default:
85  throw o2::framework::runtime_error_f("Unsupported branch type: %d", static_cast<int>(type));
86  }
87 }

222 arrow::Result<std::shared_ptr<arrow::Schema>> TTreeFileFormat::Inspect(const arrow::dataset::FileSource& source) const
223 {
224  arrow::Schema schema{{}};
225  auto fs = std::dynamic_pointer_cast<VirtualRootFileSystemBase>(source.filesystem());
226  // Actually get the TTree from the ROOT file.
227  auto treeFs = std::dynamic_pointer_cast<TTreeFileSystem>(fs->GetSubFilesystem(source));
228  if (!treeFs.get()) {
229  throw runtime_error_f("Unknown filesystem %s\n", source.filesystem()->type_name().c_str());
230  }
231  TTree* tree = treeFs->GetTree(source);
232 
233  auto branches = tree->GetListOfBranches();
234  auto n = branches->GetEntries();
235 
236  std::vector<BranchInfo> branchInfos;
237  for (auto i = 0; i < n; ++i) {
238  auto branch = static_cast<TBranch*>(branches->At(i));
239  auto name = std::string{branch->GetName()};
240  auto pos = name.find("_size");
241  if (pos != std::string::npos) {
242  name.erase(pos);
243  branchInfos.emplace_back(BranchInfo{name, (TBranch*)nullptr, true});
244  } else {
245  auto lookup = std::find_if(branchInfos.begin(), branchInfos.end(), [&](BranchInfo const& bi) {
246  return bi.name == name;
247  });
248  if (lookup == branchInfos.end()) {
249  branchInfos.emplace_back(BranchInfo{name, branch, false});
250  } else {
251  lookup->ptr = branch;
252  }
253  }
254  }
255 
256  std::vector<std::shared_ptr<arrow::Field>> fields;
257  tree->SetCacheSize(25000000);
258  for (auto& bi : branchInfos) {
259  static TClass* cls;
260  EDataType type;
261  bi.ptr->GetExpectedType(cls, type);
262  auto listSize = -1;
263  if (!bi.mVLA) {
264  listSize = static_cast<TLeaf*>(bi.ptr->GetListOfLeaves()->At(0))->GetLenStatic();
265  }
266  auto field = std::make_shared<arrow::Field>(bi.ptr->GetName(), arrowTypeFromROOT(type, listSize));
267  fields.push_back(field);
268 
269  tree->AddBranchToCache(bi.ptr);
270  if (strncmp(bi.ptr->GetName(), "fIndexArray", strlen("fIndexArray")) == 0) {
271  std::string sizeBranchName = bi.ptr->GetName();
272  sizeBranchName += "_size";
273  auto* sizeBranch = (TBranch*)tree->GetBranch(sizeBranchName.c_str());
274  if (sizeBranch) {
275  tree->AddBranchToCache(sizeBranch);
276  }
277  }
278  }
279  tree->StopCacheLearningPhase();
280 
281  return std::make_shared<arrow::Schema>(fields);
282 }

366 char const* rootSuffixFromArrow(arrow::Type::type id)
367 {
368  switch (id) {
369  case arrow::Type::BOOL:
370  return "/O";
371  case arrow::Type::UINT8:
372  return "/b";
373  case arrow::Type::UINT16:
374  return "/s";
375  case arrow::Type::UINT32:
376  return "/i";
377  case arrow::Type::UINT64:
378  return "/l";
379  case arrow::Type::INT8:
380  return "/B";
381  case arrow::Type::INT16:
382  return "/S";
383  case arrow::Type::INT32:
384  return "/I";
385  case arrow::Type::INT64:
386  return "/L";
387  case arrow::Type::FLOAT:
388  return "/F";
389  case arrow::Type::DOUBLE:
390  return "/D";
391  default:
392  throw runtime_error("Unsupported arrow column type");
393  }

396 class TTreeFileWriter : public arrow::dataset::FileWriter
397 {
398  std::vector<TBranch*> branches;
399  std::vector<TBranch*> sizesBranches;
400  std::vector<std::shared_ptr<arrow::Array>> valueArrays;
401  std::vector<std::shared_ptr<arrow::Array>> sizeArrays;
402  std::vector<std::shared_ptr<arrow::DataType>> valueTypes;
403 
404  std::vector<int64_t> valuesIdealBasketSize;
405  std::vector<int64_t> sizeIdealBasketSize;
406 
407  std::vector<int64_t> typeSizes;
408  std::vector<int64_t> listSizes;
409  bool firstBasket = true;
410 
411  // This is to create a batsket size according to the first batch.
412  void finaliseBasketSize(std::shared_ptr<arrow::RecordBatch> firstBatch)
413  {
414  O2_SIGNPOST_ID_FROM_POINTER(sid, root_arrow_fs, this);
415  O2_SIGNPOST_START(root_arrow_fs, sid, "finaliseBasketSize", "First batch with %lli rows received and %zu columns",
416  firstBatch->num_rows(), firstBatch->columns().size());
417  for (size_t i = 0; i < branches.size(); i++) {
418  auto* branch = branches[i];
419  auto* sizeBranch = sizesBranches[i];
420 
421  int valueSize = valueTypes[i]->byte_width();
422  if (listSizes[i] == 1) {
423  O2_SIGNPOST_EVENT_EMIT(root_arrow_fs, sid, "finaliseBasketSize", "Branch %s exists and uses %d bytes per entry for %lli entries.",
424  branch->GetName(), valueSize, firstBatch->num_rows());
425  assert(sizeBranch == nullptr);
426  branch->SetBasketSize(1024 + firstBatch->num_rows() * valueSize);
427  } else if (listSizes[i] == -1) {
428  O2_SIGNPOST_EVENT_EMIT(root_arrow_fs, sid, "finaliseBasketSize", "Branch %s exists and uses %d bytes per entry.",
429  branch->GetName(), valueSize);
430  // This should probably lookup the
431  auto column = firstBatch->GetColumnByName(schema_->field(i)->name());
432  auto list = std::static_pointer_cast<arrow::ListArray>(column);
433  O2_SIGNPOST_EVENT_EMIT(root_arrow_fs, sid, "finaliseBasketSize", "Branch %s needed. Associated size branch %s and there are %lli entries of size %d in that list.",
434  branch->GetName(), sizeBranch->GetName(), list->length(), valueSize);
435  branch->SetBasketSize(1024 + firstBatch->num_rows() * valueSize * list->length());
436  sizeBranch->SetBasketSize(1024 + firstBatch->num_rows() * 4);
437  } else {
438  O2_SIGNPOST_EVENT_EMIT(root_arrow_fs, sid, "finaliseBasketSize", "Branch %s needed. There are %lli entries per array of size %d in that list.",
439  branch->GetName(), listSizes[i], valueSize);
440  assert(sizeBranch == nullptr);
441  branch->SetBasketSize(1024 + firstBatch->num_rows() * valueSize * listSizes[i]);
442  }
443 
444  auto field = firstBatch->schema()->field(i);
445  if (field->name().starts_with("fIndexArray")) {
446  // One int per array to keep track of the size
447  int idealBasketSize = 4 * firstBatch->num_rows() + 1024 + field->type()->byte_width() * firstBatch->num_rows(); // minimal additional size needed, otherwise we get 2 baskets
448  int basketSize = std::max(32000, idealBasketSize); // keep a minimum value
449  sizeBranch->SetBasketSize(basketSize);
450  branch->SetBasketSize(basketSize);
451  }
452  }
453  O2_SIGNPOST_END(root_arrow_fs, sid, "finaliseBasketSize", "Done");
454  }
455 
456  public:
457  // Create the TTree based on the physical_schema, not the one in the batch.
458  // The write method will have to reconcile the two schemas.
459  TTreeFileWriter(std::shared_ptr<arrow::Schema> schema, std::shared_ptr<arrow::dataset::FileWriteOptions> options,
460  std::shared_ptr<arrow::io::OutputStream> destination,
461  arrow::fs::FileLocator destination_locator)
462  : FileWriter(schema, options, destination, destination_locator)
463  {
464  // Batches have the same number of entries for each column.
465  auto directoryStream = std::dynamic_pointer_cast<TDirectoryFileOutputStream>(destination_);
466  auto treeStream = std::dynamic_pointer_cast<TTreeOutputStream>(destination_);
467 
468  if (directoryStream.get()) {
469  TDirectoryFile* dir = directoryStream->GetDirectory();
470  dir->cd();
471  auto* tree = new TTree(destination_locator_.path.c_str(), "");
472  treeStream = std::make_shared<TTreeOutputStream>(tree, "");
473  } else if (treeStream.get()) {
474  // We already have a tree stream, let's derive a new one
475  // with the destination_locator_.path as prefix for the branches
476  // This way we can multiplex multiple tables in the same tree.
477  auto tree = treeStream->GetTree();
478  treeStream = std::make_shared<TTreeOutputStream>(tree, destination_locator_.path);
479  } else {
480  // I could simply set a prefix here to merge to an already existing tree.
481  throw std::runtime_error("Unsupported backend.");
482  }
483 
484  for (auto i = 0u; i < schema->fields().size(); ++i) {
485  auto& field = schema->field(i);
486  listSizes.push_back(1);
487 
488  int valuesIdealBasketSize = 0;
489  // Construct all the needed branches.
490  switch (field->type()->id()) {
491  case arrow::Type::FIXED_SIZE_LIST: {
492  listSizes.back() = std::static_pointer_cast<arrow::FixedSizeListType>(field->type())->list_size();
493  valuesIdealBasketSize = 1024 + valueTypes.back()->byte_width() * listSizes.back();
494  valueTypes.push_back(field->type()->field(0)->type());
495  sizesBranches.push_back(nullptr);
496  std::string leafList = fmt::format("{}[{}]{}", field->name(), listSizes.back(), rootSuffixFromArrow(valueTypes.back()->id()));
497  branches.push_back(treeStream->CreateBranch(field->name().c_str(), leafList.c_str()));
498  } break;
499  case arrow::Type::LIST: {
500  valueTypes.push_back(field->type()->field(0)->type());
501  std::string leafList = fmt::format("{}[{}_size]{}", field->name(), field->name(), rootSuffixFromArrow(valueTypes.back()->id()));
502  listSizes.back() = -1; // VLA, we need to calculate it on the fly;
503  std::string sizeLeafList = field->name() + "_size/I";
504  sizesBranches.push_back(treeStream->CreateBranch((field->name() + "_size").c_str(), sizeLeafList.c_str()));
505  branches.push_back(treeStream->CreateBranch(field->name().c_str(), leafList.c_str()));
506  // Notice that this could be replaced by a better guess of the
507  // average size of the list elements, but this is not trivial.
508  } break;
509  default: {
510  valueTypes.push_back(field->type());
511  std::string leafList = field->name() + rootSuffixFromArrow(valueTypes.back()->id());
512  sizesBranches.push_back(nullptr);
513  branches.push_back(treeStream->CreateBranch(field->name().c_str(), leafList.c_str()));
514  } break;
515  }
516  }
517  // We create the branches from the schema
518  }

520  arrow::Status Write(const std::shared_ptr<arrow::RecordBatch>& batch) override
521  {
522  if (firstBasket) {
523  firstBasket = false;
524  finaliseBasketSize(batch);
525  }
526 
527  // Support writing empty tables
528  if (batch->columns().empty() || batch->num_rows() == 0) {
529  return arrow::Status::OK();
530  }
531 
532  // Batches have the same number of entries for each column.
533  auto directoryStream = std::dynamic_pointer_cast<TDirectoryFileOutputStream>(destination_);
534  TTree* tree = nullptr;
535  if (directoryStream.get()) {
536  TDirectoryFile* dir = directoryStream->GetDirectory();
537  tree = (TTree*)dir->Get(destination_locator_.path.c_str());
538  }
539  auto treeStream = std::dynamic_pointer_cast<TTreeOutputStream>(destination_);
540 
541  if (!tree) {
542  // I could simply set a prefix here to merge to an already existing tree.
543  throw std::runtime_error("Unsupported backend.");
544  }
545 
546  for (auto i = 0u; i < batch->columns().size(); ++i) {
547  auto column = batch->column(i);
548  auto& field = batch->schema()->field(i);
549 
550  valueArrays.push_back(nullptr);
551 
552  switch (field->type()->id()) {
553  case arrow::Type::FIXED_SIZE_LIST: {
554  auto list = std::static_pointer_cast<arrow::FixedSizeListArray>(column);
555  valueArrays.back() = list->values();
556  } break;
557  case arrow::Type::LIST: {
558  auto list = std::static_pointer_cast<arrow::ListArray>(column);
559  valueArrays.back() = list;
560  } break;
561  case arrow::Type::BOOL: {
562  // In case of arrays of booleans, we need to go back to their
563  // char based representation for ROOT to save them.
564  auto boolArray = std::static_pointer_cast<arrow::BooleanArray>(column);
565 
566  int64_t length = boolArray->length();
567  arrow::UInt8Builder builder;
568  auto ok = builder.Reserve(length);
569 
570  for (int64_t i = 0; i < length; ++i) {
571  if (boolArray->IsValid(i)) {
572  // Expand each boolean value (true/false) to uint8 (1/0)
573  uint8_t value = boolArray->Value(i) ? 1 : 0;
574  auto ok = builder.Append(value);
575  } else {
576  // Append null for invalid entries
577  auto ok = builder.AppendNull();
578  }
579  }
580  valueArrays.back() = *builder.Finish();
581  } break;
582  default:
583  valueArrays.back() = column;
584  }
585  }
586 
587  int64_t pos = 0;
588  while (pos < batch->num_rows()) {
589  for (size_t bi = 0; bi < branches.size(); ++bi) {
590  auto* branch = branches[bi];
591  auto* sizeBranch = sizesBranches[bi];
592  auto array = batch->column(bi);
593  auto& field = batch->schema()->field(bi);
594  auto& listSize = listSizes[bi];
595  auto valueType = valueTypes[bi];
596  auto valueArray = valueArrays[bi];
597 
598  switch (field->type()->id()) {
599  case arrow::Type::LIST: {

644 arrow::Result<arrow::RecordBatchGenerator> TTreeFileFormat::ScanBatchesAsync(
645  const std::shared_ptr<arrow::dataset::ScanOptions>& options,
646  const std::shared_ptr<arrow::dataset::FileFragment>& fragment) const
647 {
648  // Get the fragment as a TTreeFragment. This might be PART of a TTree.
649  auto treeFragment = std::dynamic_pointer_cast<TTreeFileFragment>(fragment);
650  // This is the schema we want to read
651  auto dataset_schema = options->dataset_schema;
652 
653  auto generator = [pool = options->pool, treeFragment, dataset_schema, &totalCompressedSize = mTotCompressedSize,
654  &totalUncompressedSize = mTotUncompressedSize]() -> arrow::Future<std::shared_ptr<arrow::RecordBatch>> {
655  auto schema = treeFragment->format()->Inspect(treeFragment->source());
656 
657  std::vector<std::shared_ptr<arrow::Array>> columns;
658  std::vector<std::shared_ptr<arrow::Field>> fields = dataset_schema->fields();
659  auto physical_schema = *treeFragment->ReadPhysicalSchema();
660 
661  static TBufferFile buffer{TBuffer::EMode::kWrite, 4 * 1024 * 1024};
662  auto containerFS = std::dynamic_pointer_cast<VirtualRootFileSystemBase>(treeFragment->source().filesystem());
663  auto fs = std::dynamic_pointer_cast<TTreeFileSystem>(containerFS->GetSubFilesystem(treeFragment->source()));
664 
665  int64_t rows = -1;
666  TTree* tree = fs->GetTree(treeFragment->source());
667  for (auto& field : fields) {
668  // The field actually on disk
669  auto physicalField = physical_schema->GetFieldByName(field->name());
670  TBranch* branch = tree->GetBranch(physicalField->name().c_str());
671  assert(branch);
672  buffer.Reset();
673  auto totalEntries = branch->GetEntries();
674  if (rows == -1) {
675  rows = totalEntries;
676  }
677  if (rows != totalEntries) {
678  throw runtime_error_f("Unmatching number of rows for branch %s", branch->GetName());
679  }
680  arrow::Status status;
681  int readEntries = 0;
682  std::shared_ptr<arrow::Array> array;
683  auto listType = std::dynamic_pointer_cast<arrow::FixedSizeListType>(physicalField->type());
684  if (physicalField->type() == arrow::boolean() ||
685  (listType && physicalField->type()->field(0)->type() == arrow::boolean())) {
686  if (listType) {
687  std::unique_ptr<arrow::ArrayBuilder> builder = nullptr;
688  auto status = arrow::MakeBuilder(pool, physicalField->type()->field(0)->type(), &builder);
689  if (!status.ok()) {
690  throw runtime_error("Cannot create value builder");
691  }
692  auto listBuilder = std::make_unique<arrow::FixedSizeListBuilder>(pool, std::move(builder), listType->list_size());
693  auto valueBuilder = listBuilder.get()->value_builder();
694  // boolean array special case: we need to use builder to create the bitmap
695  status = valueBuilder->Reserve(totalEntries * listType->list_size());
696  status &= listBuilder->Reserve(totalEntries);
697  if (!status.ok()) {
698  throw runtime_error("Failed to reserve memory for array builder");
699  }
700  while (readEntries < totalEntries) {
701  auto readLast = branch->GetBulkRead().GetBulkEntries(readEntries, buffer);
702  readEntries += readLast;
703  status &= static_cast<arrow::BooleanBuilder*>(valueBuilder)->AppendValues(reinterpret_cast<uint8_t const*>(buffer.GetCurrent()), readLast * listType->list_size());
704  }
705  status &= static_cast<arrow::FixedSizeListBuilder*>(listBuilder.get())->AppendValues(readEntries);
706  if (!status.ok()) {
707  throw runtime_error("Failed to append values to array");
708  }
709  status &= listBuilder->Finish(&array);
710  if (!status.ok()) {
711  throw runtime_error("Failed to create array");
712  }
713  } else if (listType == nullptr) {
714  std::unique_ptr<arrow::ArrayBuilder> builder = nullptr;
715  auto status = arrow::MakeBuilder(pool, physicalField->type(), &builder);
716  if (!status.ok()) {
717  throw runtime_error("Cannot create builder");
718  }
719  auto valueBuilder = static_cast<arrow::BooleanBuilder*>(builder.get());
720  // boolean array special case: we need to use builder to create the bitmap
721  status = valueBuilder->Reserve(totalEntries);
722  if (!status.ok()) {
723  throw runtime_error("Failed to reserve memory for array builder");
724  }
725  while (readEntries < totalEntries) {
726  auto readLast = branch->GetBulkRead().GetBulkEntries(readEntries, buffer);
727  readEntries += readLast;
728  status &= valueBuilder->AppendValues(reinterpret_cast<uint8_t const*>(buffer.GetCurrent()), readLast);
729  }
730  if (!status.ok()) {
731  throw runtime_error("Failed to append values to array");
732  }
733  status &= valueBuilder->Finish(&array);
734  if (!status.ok()) {
735  throw runtime_error("Failed to create array");
736  }
737  }
738  } else {
739  // other types: use serialized read to build arrays directly.
740  auto typeSize = physicalField->type()->byte_width();
741  // This is needed for branches which have not been persisted.
742  auto bytes = branch->GetTotBytes();
743  auto branchSize = bytes ? bytes : 1000000;
744  auto&& result = arrow::AllocateResizableBuffer(branchSize, pool);
745  if (!result.ok()) {
746  throw runtime_error("Cannot allocate values buffer");
747  }
748  std::shared_ptr<arrow::Buffer> arrowValuesBuffer = std::move(result).ValueUnsafe();
749  auto ptr = arrowValuesBuffer->mutable_data();
750  if (ptr == nullptr) {
751  throw runtime_error("Invalid buffer");
752  }
753 
754  std::unique_ptr<TBufferFile> offsetBuffer = nullptr;
755 
756  uint32_t offset = 0;
757  int count = 0;
758  std::shared_ptr<arrow::Buffer> arrowOffsetBuffer;
759  std::span<int> offsets;
760  int size = 0;
761  uint32_t totalSize = 0;
762  TBranch* mSizeBranch = nullptr;
763  int64_t listSize = 1;
764  if (auto fixedSizeList = std::dynamic_pointer_cast<arrow::FixedSizeListType>(physicalField->type())) {
765  listSize = fixedSizeList->list_size();
766  typeSize = fixedSizeList->field(0)->type()->byte_width();
767  } else if (auto vlaListType = std::dynamic_pointer_cast<arrow::ListType>(physicalField->type())) {
768  listSize = -1;
769  typeSize = vlaListType->field(0)->type()->byte_width();
770  }
771  if (listSize == -1) {
772  mSizeBranch = branch->GetTree()->GetBranch((std::string{branch->GetName()} + "_size").c_str());
773  offsetBuffer = std::make_unique<TBufferFile>(TBuffer::EMode::kWrite, 4 * 1024 * 1024);
774  result = arrow::AllocateResizableBuffer((totalEntries + 1) * (int64_t)sizeof(int), pool);
775  if (!result.ok()) {
776  throw runtime_error("Cannot allocate offset buffer");
777  }
778  arrowOffsetBuffer = std::move(result).ValueUnsafe();
779  unsigned char* ptrOffset = arrowOffsetBuffer->mutable_data();
780  auto* tPtrOffset = reinterpret_cast<int*>(ptrOffset);
781  offsets = std::span<int>{tPtrOffset, tPtrOffset + totalEntries + 1};
782 
783  // read sizes first
784  while (readEntries < totalEntries) {
785  auto readLast = mSizeBranch->GetBulkRead().GetEntriesSerialized(readEntries, *offsetBuffer);
786  readEntries += readLast;
787  for (auto i = 0; i < readLast; ++i) {
788  offsets[count++] = (int)offset;
789  offset += swap32_(reinterpret_cast<uint32_t*>(offsetBuffer->GetCurrent())[i]);
790  }
791  }
792  offsets[count] = (int)offset;
793  totalSize = offset;
794  readEntries = 0;
795  }
796 
797  while (readEntries < totalEntries) {
798  auto readLast = branch->GetBulkRead().GetEntriesSerialized(readEntries, buffer);
799  if (listSize == -1) {
800  size = offsets[readEntries + readLast] - offsets[readEntries];
801  } else {
802  size = readLast * listSize;
803  }
804  readEntries += readLast;
805  swapCopy(ptr, buffer.GetCurrent(), size, typeSize);
806  ptr += (ptrdiff_t)(size * typeSize);
807  }
808  if (listSize >= 1) {
809  totalSize = readEntries * listSize;
810  }
811  std::shared_ptr<arrow::PrimitiveArray> varray;
812  switch (listSize) {
813  case -1:
814  varray = std::make_shared<arrow::PrimitiveArray>(physicalField->type()->field(0)->type(), totalSize, arrowValuesBuffer);
815  array = std::make_shared<arrow::ListArray>(physicalField->type(), readEntries, arrowOffsetBuffer, varray);
816  break;
817  case 1:
818  array = std::make_shared<arrow::PrimitiveArray>(physicalField->type(), readEntries, arrowValuesBuffer);
819  break;
820  default:
821  varray = std::make_shared<arrow::PrimitiveArray>(physicalField->type()->field(0)->type(), totalSize, arrowValuesBuffer);
822  array = std::make_shared<arrow::FixedSizeListArray>(physicalField->type(), readEntries, varray);
823  }
824  }
825 
826  branch->SetStatus(false);
827  branch->DropBaskets("all");
828  branch->Reset();
829  branch->GetTransientBuffer(0)->Expand(0);
830 
831  columns.push_back(array);
832  }
833  auto batch = arrow::RecordBatch::Make(dataset_schema, rows, columns);
834  totalCompressedSize += tree->GetZipBytes();
835  totalUncompressedSize += tree->GetTotBytes();
836  return batch;
837  };
838  return generator;
839 }
