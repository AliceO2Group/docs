67 void TrackResiduals::setY2XBinning(const std::vector<float>& binning)
68 {
69  if (mIsInitialized) {
70  LOG(error) << "Binning already initialized, not changing y/x binning";
71  return;
72  }
73 
74  if (binning.size() == 0) {
75  LOGP(info, "Empty binning provided, will use default uniform y/x binning with {} bins", mNY2XBins);
76  return;
77  } else if (binning.size() == 1) {
78  const int bins = static_cast<int>(binning.at(0));
79  setNY2XBins(bins);
80  LOGP(info, "Setting uniform binning for y/x with {} bins", bins - 1);
81  return;
82  }
83 
84  const int nBins = binning.size() - 1;
85  if (fabsf(binning[0] + 1.f) > param::sEps || fabsf(binning[nBins] - 1.f) > param::sEps) {
86  LOG(error) << "Provided binning for y/x not in range -1 to 1: " << binning[0] << " - " << binning[nBins] << ". Not changing y/x binning";
87  return;
88  }
89 
90  LOGP(info, "Setting custom binning for y/x with {} bins", nBins);
91  setNY2XBins(nBins);
92  mUniformBins[VoxF] = false;
93  mY2XBinsDH.clear();
94  mY2XBinsDI.clear();
95  mY2XBinsCenter.clear();
96  for (int iBin = 0; iBin < nBins; ++iBin) {
97  mY2XBinsDH.push_back(.5f * (binning[iBin + 1] - binning[iBin]));
98  mY2XBinsDI.push_back(.5f / mY2XBinsDH[iBin]);
99  mY2XBinsCenter.push_back(binning[iBin] + mY2XBinsDH[iBin]);
100  LOGF(info, "Bin %i: center (%.3f), half bin width (%.3f)", iBin, mY2XBinsCenter.back(), mY2XBinsDH.back());
101  }
102 }

142 //______________________________________________________________________________
143 void TrackResiduals::initBinning()
144 {
145  // initialize binning structures
146  //
147  // X binning
148  if (mNXBins > 0 && mNXBins < param::NPadRows) {
149  // uniform binning in X
150  LOGF(info, "X-binning is uniform with %i bins from %.2f to %.2f", mNXBins, param::MinX, param::MaxX);
151  mDXI = mNXBins / (param::MaxX - param::MinX);
152  mDX = 1.0f / mDXI;
153  mUniformBins[VoxX] = true;
154  } else {
155  // binning per pad row
156  LOGF(info, "X-binning is per pad-row");
157  mNXBins = param::NPadRows;
158  mUniformBins[VoxX] = false;
159  mDX = param::RowDX[0];
160  mDXI = 1.f / mDX; // should not be used
161  }
162  //
163  // Y/X binning
164  mMaxY2X.resize(mNXBins);
165  mDY2XI.resize(mNXBins);
166  mDY2X.resize(mNXBins);
167  //
168  for (int ix = 0; ix < mNXBins; ++ix) {
169  float x = getX(ix);
170  mMaxY2X[ix] = tan(.5f * SECPHIWIDTH) - sDeadZone / x;
171  mDY2XI[ix] = mNY2XBins / (2.f * mMaxY2X[ix]);
172  mDY2X[ix] = 1.f / mDY2XI[ix];
173  }
174  if (mUniformBins[VoxF]) {
175  LOGF(info, "Y/X-binning is uniform with %i bins from -MaxY2X to +MaxY2X (values depend on X-bin)", mNY2XBins);
176  for (int ip = 0; ip < mNY2XBins; ++ip) {
177  mY2XBinsDH.push_back(1.f / mNY2XBins);
178  mY2XBinsDI.push_back(.5f / mY2XBinsDH[ip]);
179  mY2XBinsCenter.push_back(-1.f + (ip + 0.5f) * 2.f * mY2XBinsDH[ip]);
180  LOGF(info, "Bin %i: center (%.3f), half bin width (%.3f)", ip, mY2XBinsCenter.back(), mY2XBinsDH.back());
181  }
182  }
183  //
184  // Z/X binning
185  mDZ2XI = mNZ2XBins / sMaxZ2X;
186  mDZ2X = 1.0f / mDZ2XI; // for uniform case only
187  if (mUniformBins[VoxZ]) {
188  LOGF(info, "Z/X-binning is uniform with %i bins from 0 to %f", mNZ2XBins, sMaxZ2X);
189  for (int iz = 0; iz < mNZ2XBins; ++iz) {
190  mZ2XBinsDH.push_back(.5f * mDZ2X);
191  mZ2XBinsDI.push_back(mDZ2XI);
192  mZ2XBinsCenter.push_back((iz + 0.5f) * mDZ2X);
193  LOGF(info, "Bin %i: center (%.3f), half bin width (%.3f)", iz, mZ2XBinsCenter.back(), mZ2XBinsDH.back());
194  }
195  }
196  //
197  mNVoxPerSector = mNY2XBins * mNZ2XBins * mNXBins;
198  LOGF(info, "Each TPC sector is divided into %i voxels", mNVoxPerSector);
199 }

235 int TrackResiduals::getRowID(float x) const
236 {
237  int ix;
238  if (x < param::RowX[param::NRowsAccumulated[0] - 1] + param::RowDX[0]) {
239  // we are in the IROC
240  ix = (x - (param::RowX[0] - .5f * param::RowDX[0])) / param::RowDX[0];
241  if (ix < 0) {
242  // x is smaller than the inner radius of the first pad row
243  ix = -1;
244  }
245  } else if (x >= param::RowX[param::NRowsAccumulated[param::NROCTypes - 2]] - .5f * param::RowDX[param::NROCTypes - 1]) {
246  // we are in the OROC3
247  ix = (x - (param::RowX[param::NRowsAccumulated[param::NROCTypes - 2]] - .5f * param::RowDX[param::NROCTypes - 1])) / param::RowDX[param::NROCTypes - 1] + param::NRowsAccumulated[param::NROCTypes - 2];
248  if (ix >= param::NPadRows) {
249  // x is larger than the outer radius of the last OROC pad row
250  ix = -1;
251  }
252  } else if (x > param::RowX[param::NRowsAccumulated[0]] - .5f * param::RowDX[1] && x < param::RowX[param::NRowsAccumulated[1] - 1] + .5f * param::RowDX[1]) {
253  // we are in the OROC1
254  ix = (x - (param::RowX[param::NRowsAccumulated[0]] - .5f * param::RowDX[1])) / param::RowDX[1] + param::NRowsAccumulated[0];
255  } else if (x > param::RowX[param::NRowsAccumulated[1]] - .5f * param::RowDX[2] && x < param::RowX[param::NRowsAccumulated[2] - 1] + .5f * param::RowDX[2]) {
256  // we are in the OROC2
257  ix = (x - (param::RowX[param::NRowsAccumulated[1]] - .5f * param::RowDX[2])) / param::RowDX[2] + param::NRowsAccumulated[1];
258  } else {
259  // x is in one of the gaps between the ROCs
260  ix = -1;
261  }
262  return ix;
263 }

265 bool TrackResiduals::findVoxelBin(int secID, float x, float y, float z, std::array<unsigned char, VoxDim>& bvox) const
266 {
267  // Z/X bin
268  if (fabs(z / x) > sMaxZ2X) {
269  return false;
270  }
271  int bz = getZ2XBinExact(secID < SECTORSPERSIDE ? z / x : -z / x);
272  if (bz < 0) {
273  return false;
274  }
275  // X bin
276  int bx = getXBinExact(x);
277  if (bx < 0 || bx >= mNXBins) {
278  return false;
279  }
280  // Y/X bin
281  int bp = getY2XBinExact(y / x, bx);
282  if (bp < 0 || bp >= mNY2XBins) {
283  return false;
284  }
285 
286  bvox[VoxZ] = bz;
287  bvox[VoxX] = bx;
288  bvox[VoxF] = bp;
289  return true;
290 }

292 void TrackResiduals::setKernelType(KernelType kernel, float bwX, float bwP, float bwZ, float scX, float scP, float scZ)
293 {
294  // set kernel type and widths in terms of binning in x, y/x, z/x and define aux variables
295  mKernelType = kernel;
296 
297  mKernelScaleEdge[VoxX] = scX;
298  mKernelScaleEdge[VoxF] = scP;
299  mKernelScaleEdge[VoxZ] = scZ;
300 
301  mKernelWInv[VoxX] = (bwX > 0) ? 1. / bwX : 1.;
302  mKernelWInv[VoxF] = (bwP > 0) ? 1. / bwP : 1.;
303  mKernelWInv[VoxZ] = (bwZ > 0) ? 1. / bwZ : 1.;
304 
305  if (mKernelType == KernelType::Epanechnikov) {
306  // bandwidth 1
307  mStepKern[VoxX] = static_cast<int>(nearbyint(bwX + 0.5));
308  mStepKern[VoxF] = static_cast<int>(nearbyint(bwP + 0.5));
309  mStepKern[VoxZ] = static_cast<int>(nearbyint(bwZ + 0.5));
310  } else if (mKernelType == KernelType::Gaussian) {
311  // look in ~5 sigma
312  mStepKern[VoxX] = static_cast<int>(nearbyint(bwX * 5. + 0.5));
313  mStepKern[VoxF] = static_cast<int>(nearbyint(bwP * 5. + 0.5));
314  mStepKern[VoxZ] = static_cast<int>(nearbyint(bwZ * 5. + 0.5));
315  } else {
316  LOG(error) << "given kernel type is not defined";
317  }
318  for (int i = VoxDim; i--;) {
319  if (mStepKern[i] < 1) {
320  mStepKern[i] = 1;
321  }
322  }
323 }

343 void TrackResiduals::fillStats(int iSec)
344 {
345  initResultsContainer(iSec);
346  std::vector<VoxRes>& secDataTmp = mVoxelResults[iSec];
347  for (int iVox = 0; iVox < mNVoxPerSector; ++iVox) {
348  const auto& voxStat = mVoxStatsIn[iVox];
349  VoxRes& resVox = secDataTmp[iVox];
350  for (int iDim = VoxDim; iDim--;) {
351  const auto sumStat = (resVox.stat[VoxDim] + voxStat.nEntries);
352  if (sumStat == 0) {
353  continue;
354  }
355  double norm = 1. / sumStat;
356  resVox.stat[iDim] = (resVox.stat[iDim] * resVox.stat[VoxDim] + voxStat.meanPos[iDim] * voxStat.nEntries) * norm;
357  }
358  resVox.stat[VoxDim] += voxStat.nEntries;
359  }

362 //______________________________________________________________________________
363 void TrackResiduals::processSectorResiduals(int iSec)
364 {
365  LOGP(info, "Processing {} voxel residuals for sector {}", mLocalResidualsIn.size(), iSec);
366  initResultsContainer(iSec);
367  // effective t0 correction changes sign between A-/C-side
368  float effT0corr = (iSec < SECTORSPERSIDE) ? mEffT0Corr : -1. * mEffT0Corr;
369  std::vector<size_t> binData;
370  for (const auto& res : mLocalResidualsIn) {
371  binData.push_back(getGlbVoxBin(res.bvox));
372  }
373  // sort in voxel increasing order
374  std::vector<size_t> binIndices(binData.size());
375  o2::math_utils::SortData(binData, binIndices);
376  // fill the voxel statistics into the results container
377  std::vector<VoxRes>& secData = mVoxelResults[iSec];
378 
379  // vectors holding the data for one voxel at a time
380  std::vector<float> dyVec;
381  std::vector<float> dzVec;
382  std::vector<float> tgVec;
383  // assuming we will always have around 1000 entries per voxel
384  dyVec.reserve(1e3);
385  dzVec.reserve(1e3);
386  tgVec.reserve(1e3);
387  size_t currVoxBin = -1;
388  unsigned int nPointsInVox = 0;
389  unsigned int nProcessed = 0;
390  while (nProcessed < binData.size()) {
391  // read all points, voxel by voxel
392  int idx = binIndices[nProcessed];
393  if (currVoxBin != binData[idx]) {
394  if (nPointsInVox) {
395  VoxRes& resVox = secData[currVoxBin];
396  processVoxelResiduals(dyVec, dzVec, tgVec, resVox);
397  }
398  currVoxBin = binData[idx];
399  nPointsInVox = 0;
400  dyVec.clear();
401  dzVec.clear();
402  tgVec.clear();
403  }
404  dyVec.push_back(mLocalResidualsIn[idx].dy * param::MaxResid / 0x7fff);
405  dzVec.push_back(mLocalResidualsIn[idx].dz * param::MaxResid / 0x7fff -
406  mEffVdriftCorr * secData[currVoxBin].stat[VoxZ] * secData[currVoxBin].stat[VoxX] -
407  effT0corr);
408  tgVec.push_back(mLocalResidualsIn[idx].tgSlp * param::MaxTgSlp / 0x7fff);
409 
410  ++nPointsInVox;
411  ++nProcessed;
412  }
413  if (nPointsInVox) {
414  // process last voxel
415  VoxRes& resVox = secData[currVoxBin];
416  processVoxelResiduals(dyVec, dzVec, tgVec, resVox);
417  }
418  LOG(info) << "extracted residuals for sector " << iSec;
419 
420  int nRowsOK = validateVoxels(iSec);
421  LOG(info) << "number of validated X rows: " << nRowsOK;
422  if (!nRowsOK) {
423  LOG(warning) << "sector " << iSec << ": all X-bins disabled, abandon smoothing";
424  return;
425  } else {
426  smooth(iSec);
427  }
428 
429  // process dispersions
430  dyVec.clear();
431  tgVec.clear();
432  currVoxBin = -1;
433  nProcessed = 0;
434  nPointsInVox = 0;
435  while (nProcessed < binData.size()) {
436  int idx = binIndices[nProcessed];
437  if (currVoxBin != binData[idx]) {
438  if (nPointsInVox) {
439  VoxRes& resVox = secData[currVoxBin];
440  if (!getXBinIgnored(iSec, resVox.bvox[VoxX])) {
441  processVoxelDispersions(tgVec, dyVec, resVox);
442  }
443  }
444  currVoxBin = binData[idx];
445  nPointsInVox = 0;
446  dyVec.clear();
447  tgVec.clear();
448  }
449  dyVec.push_back(mLocalResidualsIn[idx].dy * param::MaxResid / 0x7fff);
450  tgVec.push_back(mLocalResidualsIn[idx].tgSlp * param::MaxTgSlp / 0x7fff);
451  ++nPointsInVox;
452  ++nProcessed;
453  }
454  if (nPointsInVox) {
455  // process last voxel
456  VoxRes& resVox = secData[currVoxBin];
457  if (!getXBinIgnored(iSec, resVox.bvox[VoxX])) {
458  processVoxelDispersions(tgVec, dyVec, resVox);
459  }
460  }
461  // smooth dispersions
462  for (int ix = 0; ix < mNXBins; ++ix) {
463  if (getXBinIgnored(iSec, ix)) {
464  continue;
465  }
466  for (int iz = 0; iz < mNZ2XBins; ++iz) {
467  for (int ip = 0; ip < mNY2XBins; ++ip) {
468  int voxBin = getGlbVoxBin(ix, ip, iz);
469  VoxRes& resVox = secData[voxBin];
470  getSmoothEstimate(iSec, resVox.stat[VoxX], resVox.stat[VoxF], resVox.stat[VoxZ], resVox.DS, 0x1 << VoxV);
471  }
472  }
473  }
474  LOG(info) << "Done processing residuals for sector " << iSec;
475  dumpResults(iSec);
476 }

478 //______________________________________________________________________________
479 void TrackResiduals::processVoxelResiduals(std::vector<float>& dy, std::vector<float>& dz, std::vector<float>& tg, VoxRes& resVox)
480 {
481  int nPoints = dy.size();
482  if (nPoints < mParams->minEntriesPerVoxel) {
483  LOG(debug) << "voxel " << getGlbVoxBin(resVox.bvox) << " is skipped due to too few entries (" << nPoints << " < " << mParams->minEntriesPerVoxel << ")";
484  return;
485  } else {
486  LOGF(debug, "Processing voxel %i with %i entries", getGlbVoxBin(resVox.bvox), nPoints);
487  }
488  std::array<float, 7> zResults;
489  resVox.flags = 0;
490  std::vector<size_t> indices(dz.size());
491  if (!o2::math_utils::LTMUnbinned(dz, indices, zResults, mParams->LTMCut)) {
492  LOG(debug) << "failed trimming input array for voxel " << getGlbVoxBin(resVox.bvox);
493  return;
494  }
495  if (!mParams->isBfieldZero) {
496  std::array<float, 2> res{0.f};
497  std::array<float, 3> err{0.f};
498  float sigMAD = fitPoly1Robust(tg, dy, res, err, mParams->LTMCut);
499  if (sigMAD < 0) {
500  LOG(debug) << "failed robust linear fit, sigMAD = " << sigMAD;
501  return;
502  }
503  float corrErr = err[0] * err[2];
504  corrErr = corrErr > 0 ? err[1] / std::sqrt(corrErr) : -999;
505  //
506  resVox.D[ResX] = -res[1];
507  resVox.D[ResY] = res[0];
508  resVox.D[ResZ] = zResults[1];
509  resVox.E[ResX] = std::sqrt(err[2]);
510  resVox.E[ResY] = std::sqrt(err[0]);
511  resVox.E[ResZ] = zResults[4];
512  resVox.EXYCorr = corrErr;
513  resVox.D[ResD] = resVox.dYSigMAD = sigMAD; // later will be overwritten by real dispersion
514  resVox.dZSigLTM = zResults[2];
515  } else {
516  // for B=0 we cannot disentangle radial distortions from distortions in y,
517  // so simply use average for dy as well and set distortion in X to zero
518  std::array<float, 7> yResults;
519  std::vector<size_t> indicesY(dy.size());
520  if (!o2::math_utils::LTMUnbinned(dy, indicesY, yResults, mParams->LTMCut)) {
521  LOG(debug) << "failed trimming input array for voxel " << getGlbVoxBin(resVox.bvox);
522  return;
523  }
524  resVox.D[ResX] = 0; // force to zero
525  resVox.D[ResY] = yResults[1];
526  resVox.D[ResZ] = zResults[1];
527  resVox.E[ResX] = 0;
528  resVox.E[ResY] = yResults[4];
529  resVox.E[ResZ] = zResults[4];
530  resVox.EXYCorr = 0;
531  resVox.D[ResD] = resVox.dYSigMAD = yResults[2];
532  resVox.dZSigLTM = zResults[2];
533  }
534 
535  LOGF(debug, "D[0]=%.2f, D[1]=%.2f, D[2]=%.2f, E[0]=%.2f, E[1]=%.2f, E[2]=%.2f, EXYCorr=%.4f, dYSigMAD=%.3f, dZSigLTM=%.3f",
536  resVox.D[0], resVox.D[1], resVox.D[2], resVox.E[0], resVox.E[1], resVox.E[2], resVox.EXYCorr, resVox.dYSigMAD, resVox.dZSigLTM);
537 
538  resVox.flags |= DistDone;
539 
540  return;

543 void TrackResiduals::processVoxelDispersions(std::vector<float>& tg, std::vector<float>& dy, VoxRes& resVox)

562  // return number of good voxels for given sector
563  int cntMasked = 0; // number of voxels masked for any reason (either low statistics or bad fit)

567 
568  int cntMaskedFit = 0;
569  int cntMaskedSigma = 0;

572  for (int ix = 0; ix < mNXBins; ++ix) {
573  int cntValid = 0;
574  for (int ip = 0; ip < mNY2XBins; ++ip) {
575  for (int iz = 0; iz < mNZ2XBins; ++iz) {
576  int binGlb = getGlbVoxBin(ix, ip, iz);
577  VoxRes& resVox = secData[binGlb];

579  ++cntLowStat;
580  }

587  voxelOK = false;

589  }
590  // check raw distribution sigmas
591  if (resVox.dYSigMAD > mParams->maxSigY ||
592  resVox.dZSigLTM > mParams->maxSigZ) {
593  voxelOK = false;
594  ++cntMaskedSigma;
595  }
596  }
597  if (voxelOK) {
598  ++cntValid;
599  } else {
600  ++cntMasked;
601  resVox.flags |= Masked;
602  }
603  } // loop over Z
604  } // loop over Y/X
605  mValidFracXBins[iSec][ix] = static_cast<float>(cntValid) / (mNY2XBins * mNZ2XBins);
606  LOGP(debug, "Sector {}: xBin {} has {} % of voxels valid. Total masked due to fit: {} ,and sigma: {}",
607  iSec, ix, mValidFracXBins[iSec][ix] * 100., cntMaskedFit, cntMaskedSigma);
608  } // loop over X
609 
610  // mask X-bins which cannot be smoothed
611 
612  short nBadReg = 0; // count bad regions (one or more consecutive bad X-bins)
613  std::array<short, param::NPadRows> badStart; // to store indices to the beginnings of the bad regions
614  std::array<short, param::NPadRows> badEnd; // to store indices to the end of the bad regions
615  bool prevBad = false;
616  float fracBadRows = 0.f;
617  for (int ix = 0; ix < mNXBins; ++ix) {
618  if (mValidFracXBins[iSec][ix] < mParams->minValidVoxFracDrift) {
619  LOG(debug) << "row " << ix << " is bad";
620  ++fracBadRows;
621  if (prevBad) {
622  badEnd[nBadReg] = ix;
623  } else {
624  badStart[nBadReg] = ix;
625  badEnd[nBadReg] = ix;
626  prevBad = true;
627  }
628  } else {
629  if (prevBad) {
630  ++nBadReg;
631  prevBad = false;
632  }
633  }
634  }
635  if (prevBad) {
636  ++nBadReg;
637  }

640  LOG(warning) << "sector " << iSec << ": Fraction of bad X-bins: " << fracBadRows << " -> masking whole sector";
641  mXBinsIgnore[iSec].set();
642  } else {
643  for (int iBad = 0; iBad < nBadReg; ++iBad) {
644  LOG(debug) << "masking bad region " << iBad;
645  short badInReg = badEnd[iBad] - badStart[iBad] + 1;

648  // disable too large bad patches
649  for (int i = 0; i < badInReg; ++i) {

651  mXBinsIgnore[iSec].set(badStart[iBad] + i);
652  }
653  }
654  if (badInNextReg > mParams->maxBadXBinsToCover && (badStart[iBad + 1] - badEnd[iBad] - 1) < mParams->minGoodXBinsToCover) {
655  // disable too small isolated good patches`
656  for (int i = badEnd[iBad] + 1; i < badStart[iBad + 1]; ++i) {
657  LOG(debug) << "disabling too small good patch before bad region " << iBad + 1 << ", badStart(" << badEnd[iBad] << "), badEnd(" << badStart[iBad + 1] << ")";
658  mXBinsIgnore[iSec].set(i);
659  }
660  }
661  }
662  if (nBadReg) {
663  if (mXBinsIgnore[iSec].test(badStart[0]) && badStart[0] < mParams->minGoodXBinsToCover) {
664  // 1st good patch is too small
665  for (int i = 0; i < badStart[0]; ++i) {
666  LOG(debug) << "disabling too small first good patch badStart(0), badEnd(" << badStart[0] << ")";
667  mXBinsIgnore[iSec].set(i);
668  }
669  }
670  if (mXBinsIgnore[iSec].test(badStart[nBadReg - 1]) && (mNXBins - badEnd[nBadReg - 1] - 1) < mParams->minGoodXBinsToCover) {
671  // last good patch is too small
672  for (int i = badEnd[nBadReg - 1] + 1; i < mNXBins; ++i) {
673  LOG(debug) << "disabling too small last good patch badStart(" << badEnd[nBadReg - 1] << "), badEnd(" << mNXBins << ")";
674  mXBinsIgnore[iSec].set(i);
675  }
676  }
677  }
678  }
679  //
680  int nMaskedRows = mXBinsIgnore[iSec].count();
681  LOGP(info, "Sector {}: out of {} voxels {} are masked. {} (low stat), {} (invalid fit) and {} (raw distrib sigma)",
682  iSec, mNVoxPerSector, cntMasked, cntLowStat, cntMaskedFit, cntMaskedSigma);
683  //
684  return mNXBins - nMaskedRows;

687 void TrackResiduals::smooth(int iSec)
688 {
689  std::vector<VoxRes>& secData = mVoxelResults[iSec];
690  for (int ix = 0; ix < mNXBins; ++ix) {
691  if (getXBinIgnored(iSec, ix)) {
692  continue;
693  }
694  for (int ip = 0; ip < mNY2XBins; ++ip) {
695  for (int iz = 0; iz < mNZ2XBins; ++iz) {
696  int voxBin = getGlbVoxBin(ix, ip, iz);
697  VoxRes& resVox = secData[voxBin];
698  resVox.flags &= ~SmoothDone;
699  bool res = getSmoothEstimate(resVox.bsec, resVox.stat[VoxX], resVox.stat[VoxF], resVox.stat[VoxZ], resVox.DS, (0x1 << VoxX | 0x1 << VoxF | 0x1 << VoxZ));
700  if (!res) {
701  mNSmoothingFailedBins[iSec]++;
702  } else {
703  resVox.flags |= SmoothDone;
704  }
705  }
706  }
707  }
708  // substract dX contribution to dZ
709  for (int ix = 0; ix < mNXBins; ++ix) {
710  if (getXBinIgnored(iSec, ix)) {
711  continue;
712  }
713  for (int ip = 0; ip < mNY2XBins; ++ip) {
714  for (int iz = 0; iz < mNZ2XBins; ++iz) {
715  int voxBin = getGlbVoxBin(ix, ip, iz);
716  VoxRes& resVox = secData[voxBin];
717  if (!(resVox.flags & SmoothDone)) {
718  continue;
719  }
720  resVox.DS[ResZ] += resVox.stat[VoxZ] * resVox.DS[ResX]; // remove slope*dX contribution from dZ
721  resVox.D[ResZ] += resVox.stat[VoxZ] * resVox.DS[ResX]; // remove slope*dX contribution from dZ
722  }
723  }
724  }

727 bool TrackResiduals::getSmoothEstimate(int iSec, float x, float p, float z, std::array<float, ResDim>& res, int whichDim)
728 {
729  // get smooth estimate for distortions for point in sector coordinates
731 
732  std::array<int, VoxDim> minPointsDir{0}; // min number of points per direction
733  const float kTrialStep = 0.5;
734  std::array<bool, ResDim> doDim{false};
735  for (int i = 0; i < ResDim; ++i) {
736  doDim[i] = (whichDim & (0x1 << i)) > 0;
737  if (doDim[i]) {
738  res[i] = 0.f;
739  }
740  }
741 
742  int matSize = sSmtLinDim;
743  for (int i = 0; i < VoxDim; ++i) {
744  minPointsDir[i] = 3; // for pol1 smoothing require at least 3 points
745  if (mSmoothPol2[i]) {
746  ++minPointsDir[i];
747  ++matSize;
748  }
749  }
750 
751  int ix0, ip0, iz0;
752  findVoxel(x, p, iSec < SECTORSPERSIDE ? z : -z, ix0, ip0, iz0); // find nearest voxel
753  std::vector<VoxRes>& secData = mVoxelResults[iSec];
754  int binCenter = getGlbVoxBin(ix0, ip0, iz0); // global bin of nearest voxel
755  VoxRes& voxCenter = secData[binCenter]; // nearest voxel
756  LOG(debug) << "getting smooth estimate around voxel " << binCenter;
757 
758  // cache
759  // \todo maybe a 1-D cache would be more efficient?
760  std::array<std::array<double, sMaxSmtDim*(sMaxSmtDim + 1) / 2>, ResDim> cmat;
761  int maxNeighb = 10 * 10 * 10;
762  std::vector<VoxRes*> currVox;
763  currVox.reserve(maxNeighb);
764  std::vector<float> currCache;
765  currCache.reserve(maxNeighb * VoxHDim);
766 
767  std::array<int, VoxDim> maxTrials;
768  maxTrials[VoxZ] = mNZ2XBins / 2;
769  maxTrials[VoxF] = mNY2XBins / 2;
770  maxTrials[VoxX] = mParams->maxBadXBinsToCover * 2;
771 
772  std::array<int, VoxDim> trial{0};
773 
774  while (true) {
775  std::fill(mLastSmoothingRes.begin(), mLastSmoothingRes.end(), 0);
776  memset(&cmat[0][0], 0, sizeof(cmat));
777 
778  int nbOK = 0; // accounted neighbours
779 
780  float stepX = mStepKern[VoxX] * (1. + kTrialStep * trial[VoxX]);
781  float stepF = mStepKern[VoxF] * (1. + kTrialStep * trial[VoxF]);
782  float stepZ = mStepKern[VoxZ] * (1. + kTrialStep * trial[VoxZ]);
783 
784  if (!(voxCenter.flags & DistDone) || (voxCenter.flags & Masked) || getXBinIgnored(iSec, ix0)) {
785  // closest voxel has no data -> increase smoothing step
786  stepX += kTrialStep * mStepKern[VoxX];
787  stepF += kTrialStep * mStepKern[VoxF];
788  stepZ += kTrialStep * mStepKern[VoxZ];
789  }
790 
791  // effective kernel widths accounting for the increased bandwidth at the edges and missing data
792  float kWXI = getDXI(ix0) * mKernelWInv[VoxX] * mStepKern[VoxX] / stepX;
793  float kWFI = getDY2XI(ix0, ip0) * mKernelWInv[VoxF] * mStepKern[VoxF] / stepF;
794  float kWZI = getDZ2XI(iz0) * mKernelWInv[VoxZ] * mStepKern[VoxZ] / stepZ;
795  int iStepX = static_cast<int>(nearbyint(stepX + 0.5));
796  int iStepF = static_cast<int>(nearbyint(stepF + 0.5));
797  int iStepZ = static_cast<int>(nearbyint(stepZ + 0.5));
798  int ixMin = ix0 - iStepX;
799  int ixMax = ix0 + iStepX;
800  if (ixMin < 0) {
801  ixMin = 0;
802  ixMax = std::min(static_cast<int>(nearbyint(ix0 + stepX * mKernelScaleEdge[VoxX])), mNXBins - 1);
803  kWXI /= mKernelScaleEdge[VoxX];
804  }
805  if (ixMax >= mNXBins) {
806  ixMax = mNXBins - 1;
807  ixMin = std::max(static_cast<int>(nearbyint(ix0 - stepX * mKernelScaleEdge[VoxX])), 0);
808  kWXI /= mKernelScaleEdge[VoxX];
809  }
810 
811  int ipMin = ip0 - iStepF;
812  int ipMax = ip0 + iStepF;
813  if (ipMin < 0) {
814  ipMin = 0;
815  ipMax = std::min(static_cast<int>(nearbyint(ip0 + stepF * mKernelScaleEdge[VoxF])), mNY2XBins - 1);
816  kWFI /= mKernelScaleEdge[VoxF];
817  }
818  if (ipMax >= mNY2XBins) {
819  ipMax = mNY2XBins - 1;
820  ipMin = std::max(static_cast<int>(nearbyint(ip0 - stepF * mKernelScaleEdge[VoxF])), 0);
821  kWFI /= mKernelScaleEdge[VoxF];
822  }
823 
824  int izMin = iz0 - iStepZ;
825  int izMax = iz0 + iStepZ;
826  if (izMin < 0) {
827  izMin = 0;
828  izMax = std::min(static_cast<int>(nearbyint(iz0 + stepZ * mKernelScaleEdge[VoxZ])), mNZ2XBins - 1);
829  kWZI /= mKernelScaleEdge[VoxZ];
830  }
831  if (izMax >= mNZ2XBins) {
832  izMax = mNZ2XBins - 1;
833  izMin = std::max(static_cast<int>(nearbyint(iz0 - stepZ * mKernelScaleEdge[VoxZ])), 0);
834  kWZI /= mKernelScaleEdge[VoxZ];
835  }
836 
837  std::vector<unsigned short> nOccX(ixMax - ixMin + 1, 0);
838  std::vector<unsigned short> nOccF(ipMax - ipMin + 1, 0);
839  std::vector<unsigned short> nOccZ(izMax - izMin + 1, 0);
840 
841  int nbCheck = (ixMax - ixMin + 1) * (ipMax - ipMin + 1) * (izMax - izMin + 1);
842  if (nbCheck >= maxNeighb) {
843  maxNeighb = nbCheck + 100;
844  currCache.reserve(maxNeighb * VoxHDim);
845  currVox.reserve(maxNeighb);
846  }
847  std::array<double, 3> u2Vec;
848 
849  // first loop, check presence of enough points
850  for (int ix = ixMin; ix <= ixMax; ++ix) {
851  for (int ip = ipMin; ip <= ipMax; ++ip) {
852  for (int iz = izMin; iz <= izMax; ++iz) {
853  int binNb = getGlbVoxBin(ix, ip, iz);
854  VoxRes& voxNb = secData[binNb];
855  if (!(voxNb.flags & DistDone) ||
856  (voxNb.flags & Masked) ||
857  getXBinIgnored(iSec, ix)) {
858  // skip voxels w/o data
859  continue;
860  }
861  // estimate weighted distance
862  float dx = voxNb.stat[VoxX] - x;
863  float df = voxNb.stat[VoxF] - p;
864  float dz = voxNb.stat[VoxZ] - z;
865  float dxw = dx * kWXI;
866  float dfw = df * kWFI;
867  float dzw = dz * kWZI;
868  u2Vec[0] = dxw * dxw;
869  u2Vec[1] = dfw * dfw;
870  u2Vec[2] = dzw * dzw;
871  double kernelWeight = getKernelWeight(u2Vec);
872  if (kernelWeight < 1e-6) {
873  continue;
874  }
875  // new point is validated
876  ++nOccX[ix - ixMin];
877  ++nOccF[ip - ipMin];
878  ++nOccZ[iz - izMin];
879  currVox[nbOK] = &voxNb;
880  currCache[nbOK * VoxHDim + VoxX] = dx;
881  currCache[nbOK * VoxHDim + VoxF] = df;
882  currCache[nbOK * VoxHDim + VoxZ] = dz;
883  currCache[nbOK * VoxHDim + VoxV] = kernelWeight;
884  ++nbOK;
885  }
886  }
887  }
888 
889  // check if we have enough points in every dimension
890  std::array<int, VoxDim> nPoints{0};
891  for (int i = ixMax - ixMin + 1; i--;) {
892  if (nOccX[i]) {
893  ++nPoints[VoxX];
894  }
895  }
896  for (int i = ipMax - ipMin + 1; i--;) {
897  if (nOccF[i]) {
898  ++nPoints[VoxF];
899  }
900  }
901  for (int i = izMax - izMin + 1; i--;) {
902  if (nOccZ[i]) {
903  ++nPoints[VoxZ];
904  }
905  }
906  bool enoughPoints = true;
907  std::array<bool, VoxDim> incrDone{false};
908  for (int i = 0; i < VoxDim; ++i) {
909  if (nPoints[i] < minPointsDir[i]) {
910  // need to extend smoothing neighbourhood
911  enoughPoints = false;
912  if (trial[i] < maxTrials[i] && !incrDone[i]) {
913  // try to increment only missing direction
914  ++trial[i];
915  incrDone[i] = true;
916  } else if (trial[i] == maxTrials[i]) {
917  // cannot increment missing direction, try others
918  for (int j = VoxDim; j--;) {
919  if (i != j && trial[j] < maxTrials[j] && !incrDone[j]) {
920  ++trial[j];
921  incrDone[j] = true;
922  }
923  }
924  }
925  }
926  }
927 
928  if (!enoughPoints) {
929  if (!(incrDone[VoxX] || incrDone[VoxF] || incrDone[VoxZ])) {
930  LOG(error) << fmt::format("trial limit reached, skipping this voxel: incrDone[VoxX] {}, incrDone[VoxF] {}, incrDone[VoxZ] {}", incrDone[VoxX], incrDone[VoxF], incrDone[VoxZ]);
931  return false;
932  }
933  LOG(debug) << "sector " << iSec << ": increasing filter bandwidth around voxel " << binCenter;
934  // printf("Sector:%2d x=%.2f y/x=%.2f z/x=%.2f (iX: %d iY2X:%d iZ2X:%d)\n", iSec, x, p, z, ix0, ip0, iz0);
935  // printf("not enough neighbours (need min %d) %d %d %d (tot: %d) | Steps: %.1f %.1f %.1f\n", 2, nPoints[VoxX], nPoints[VoxF], nPoints[VoxZ], nbOK, stepX, stepF, stepZ);
936  // printf("trying to increase filter bandwidth (trialXFZ: %d %d %d)\n", trial[VoxX], trial[VoxF], trial[VoxZ]);
937  continue;
938  }
939 
940  // now fill matrices and solve
941  for (int iNb = 0; iNb < nbOK; ++iNb) {
942  double wiCache = currCache[iNb * VoxHDim + VoxV];
943  double dxi = currCache[iNb * VoxHDim + VoxX];
944  double dfi = currCache[iNb * VoxHDim + VoxF];
945  double dzi = currCache[iNb * VoxHDim + VoxZ];
946  double dxi2 = dxi * dxi;
947  double dfi2 = dfi * dfi;
948  double dzi2 = dzi * dzi;
949  const VoxRes* voxNb = currVox[iNb];
950  for (int iDim = 0; iDim < ResDim; ++iDim) {
951  if (!doDim[iDim]) {
952  continue;
953  }
954  double vi = voxNb->D[iDim];
955  double wi = wiCache;
956  if (mUseErrInSmoothing && fabs(voxNb->E[iDim]) > 1e-6) {
957  // account for point error apart from kernel value
958  wi /= (voxNb->E[iDim] * voxNb->E[iDim]);
959  }
960  std::array<double, sMaxSmtDim*(sMaxSmtDim + 1) / 2>& cmatD = cmat[iDim];
961  double* rhsD = &mLastSmoothingRes[iDim * sMaxSmtDim];
962  unsigned short iMat = 0;
963  unsigned short iRhs = 0;
964  // linear part
965  cmatD[iMat++] += wi;
966  rhsD[iRhs++] += wi * vi;
967  //
968  cmatD[iMat++] += wi * dxi;
969  cmatD[iMat++] += wi * dxi2;
970  rhsD[iRhs++] += wi * dxi * vi;
971  //
972  cmatD[iMat++] += wi * dfi;
973  cmatD[iMat++] += wi * dxi * dfi;
974  cmatD[iMat++] += wi * dfi2;
975  rhsD[iRhs++] += wi * dfi * vi;
976  //
977  cmatD[iMat++] += wi * dzi;
978  cmatD[iMat++] += wi * dxi * dzi;
979  cmatD[iMat++] += wi * dfi * dzi;
980  cmatD[iMat++] += wi * dzi2;
981  rhsD[iRhs++] += wi * dzi * vi;
982  //
983  // check if quadratic part is needed
984  if (mSmoothPol2[VoxX]) {
985  cmatD[iMat++] += wi * dxi2;
986  cmatD[iMat++] += wi * dxi * dxi2;
987  cmatD[iMat++] += wi * dfi * dxi2;
988  cmatD[iMat++] += wi * dzi * dxi2;
989  cmatD[iMat++] += wi * dxi2 * dxi2;
990  rhsD[iRhs++] += wi * dxi2 * vi;
991  }
992  if (mSmoothPol2[VoxF]) {
993  cmatD[iMat++] += wi * dfi2;
994  cmatD[iMat++] += wi * dxi * dfi2;

1169 void TrackResiduals::medFit(int nPoints, int offset, const std::vector<float>& x, const std::vector<float>& y, float& a, float& b, std::array<float, 3>& err) const
1170 {
1171  // fitting a straight line y(x|a, b) = a + b * x
1172  // to given x and y data minimizing the absolute deviation
1173  float aa, bb, chi2 = 0.f;
1174  if (nPoints < 2) {
1175  a = b = 0.f;
1176  err[0] = err[1] = err[2] = 999.f;
1177  return;
1178  }
1179  // do least squares minimization as first guess
1180  float sx = 0.f, sxx = 0.f, sy = 0.f, sxy = 0.f;
1181  for (int j = nPoints + offset; j-- > offset;) { // same order as in AliRoot version such that resulting sums are identical
1182  sx += x[j];
1183  sxx += x[j] * x[j];
1184  sy += y[j];
1185  sxy += x[j] * y[j];
1186  }
1187  float del = nPoints * sxx - sx * sx;
1188  float delI = 1. / del;
1189  aa = (sxx * sy - sx * sxy) * delI;
1190  bb = (nPoints * sxy - sx * sy) * delI;
1191  err[0] = sxx * delI;
1192  err[1] = sx * delI;
1193  err[2] = nPoints * delI;
1194 
1195  for (int j = nPoints + offset; j-- > offset;) {
1196  float tmp = y[j] - (aa + bb * x[j]);
1197  chi2 += tmp * tmp;
1198  }
1199  float sigb = std::sqrt(chi2 * delI); // expected sigma for b
1200  float b1 = bb;
1201  float f1 = roFunc(nPoints, offset, x, y, b1, aa);
1202  if (sigb > 0) {
1203  float b2 = bb + std::copysign(3.f * sigb, f1);
1204  float f2 = roFunc(nPoints, offset, x, y, b2, aa);
1205  if (fabs(f1 - f2) < sFloatEps) {
1206  a = aa;
1207  b = bb;
1208  return;
1209  }
1210  while (f1 * f2 > 0.f) {
1211  bb = b2 + 1.6f * (b2 - b1);
1212  b1 = b2;
1213  f1 = f2;
1214  b2 = bb;
1215  f2 = roFunc(nPoints, offset, x, y, b2, aa);
1216  }
1217  sigb = .01f * sigb;
1218  while (fabs(b2 - b1) > sigb) {
1219  bb = b1 + .5f * (b2 - b1);
1220  if (bb == b1 || bb == b2) {
1221  break;
1222  }
1223  float f = roFunc(nPoints, offset, x, y, bb, aa);
1224  if (f * f1 >= .0f) {
1225  f1 = f;
1226  b1 = bb;
1227  } else {
1228  f2 = f;
1229  b2 = bb;
1230  }
1231  }
1232  }
1233  a = aa;
1234  b = bb;
1235 }

1237 float TrackResiduals::roFunc(int nPoints, int offset, const std::vector<float>& x, const std::vector<float>& y, float b, float& aa) const
1238 {
1239  // calculate sum(x_i * sgn(y_i - a - b * x_i)) for given b
1240  // see numberical recipies paragraph 15.7.3
1241  std::vector<float> vecTmp(nPoints);
1242  float sum = 0.f;
1243  for (int j = nPoints; j-- > 0;) {
1244  vecTmp[j] = y[j + offset] - b * x[j + offset];
1245  }
1246  int nPointsHalf = nPoints / 2;
1247  if (nPoints < 20) { // it is faster to do insertion sort
1248  for (int i = 1; i < nPoints; i++) {
1249  float v = vecTmp[i];
1250  int j;
1251  for (j = i; j--;) {
1252  if (vecTmp[j] > v) {
1253  vecTmp[j + 1] = vecTmp[j];
1254  } else {
1255  break;
1256  }
1257  }
1258  vecTmp[j + 1] = v;
1259  }
1260  aa = (nPoints & 0x1) ? vecTmp[nPointsHalf] : .5f * (vecTmp[nPointsHalf - 1] + vecTmp[nPointsHalf]);
1261  } else {
1262  std::vector<float>::iterator nth = vecTmp.begin() + vecTmp.size() / 2;
1263  if (nPoints & 0x1) {
1264  std::nth_element(vecTmp.begin(), nth, vecTmp.end());
1265  aa = *nth;
1266  } else {
1267  std::nth_element(vecTmp.begin(), nth - 1, vecTmp.end());
1268  std::nth_element(nth, nth, vecTmp.end());
1269  aa = 0.5 * (*(nth - 1) + *(nth));
1270  }
1271  // aa = (nPoints & 0x1) ? selectKthMin(nPointsHalf, vecTmp) : .5f * (selectKthMin(nPointsHalf - 1, vecTmp) + selectKthMin(nPointsHalf, vecTmp));
1272  }
1273  for (int j = nPoints; j-- > 0;) {
1274  float d = y[j + offset] - (b * x[j + offset] + aa);
1275  if (y[j + offset] != 0.f) {
1276  d /= fabs(y[j + offset]);
1277  }
1278  if (fabs(d) > sFloatEps) {
1279  sum += (d >= 0.f ? x[j + offset] : -x[j + offset]);
1280  }
1281  }
1282  return sum;
1283 }

1286 float TrackResiduals::selectKthMin(const int k, std::vector<float>& data) const
1287 {
1288  // Returns the k th smallest value in the vector. The input vector will be rearranged
1289  // to have this value in location data[k] , with all smaller elements moved before it
1290  // (in arbitrary order) and all larger elements after (also in arbitrary order).
1291  // From Numerical Recipes in C++ (paragraph 8.5)
1292  // Probably it is not needed anymore, since std::nth_element() can also be used
1293 
1294  int i, ir, j, l, mid, n = data.size();
1295  float a; // partitioning element
1296  l = 0; // left hand side of active partition
1297  ir = n - 1; // right hand side of active partition
1298 
1299  while (true) {
1300  if (ir <= l + 1) { // active partition with 1 or 2 elements
1301  if (ir == l + 1 && data[ir] < data[l]) { // case of 2 elements
1302  std::swap(data[l], data[ir]);
1303  }
1304  return data[k];
1305  } else {
1306  mid = (l + ir) >> 1;
1307  std::swap(data[mid], data[l + 1]);
1308  if (data[l] > data[ir]) {
1309  std::swap(data[l], data[ir]);
1310  }
1311  if (data[l + 1] > data[ir]) {
1312  std::swap(data[l + 1], data[ir]);
1313  }
1314  if (data[l] > data[l + 1]) {
1315  std::swap(data[l], data[l + 1]);
1316  }
1317  i = l + 1; // initialize pointers for partitioning
1318  j = ir;
1319  a = data[l + 1];
1320  while (true) {
1321  // innermost loop used for partitioning
1322  do {
1323  i++;
1324  } while (data[i] < a);
1325  do {
1326  j--;
1327  } while (data[j] > a);
1328  if (j < i) {
1329  // pointers crossed -> partitioning complete
1330  break;
1331  }
1332  std::swap(data[i], data[j]);
1333  }
1334  data[l + 1] = data[j];
1335  data[j] = a;
1336  // keep the partition which contains kth element active
1337  if (j >= k) {
1338  ir = j - 1;
1339  }
1340  if (j <= k) {
1341  l = i;
1342  }
1343  }
1344  }

1349 {
1350  // Sigma calculated from median absolute deviations
1351  // see: https://en.wikipedia.org/wiki/Median_absolute_deviation
1352  // the data is passed by value (copied!), such that the original vector
1353  // is not rearranged
1354

1357  return 0;
1358  }
1359

1368  std::nth_element(nth, nth, data.end());

1392 void TrackResiduals::fitCircle(int nCl, std::array<float, param::NPadRows>& x, std::array<float, param::NPadRows>& y, float& xc, float& yc, float& r, std::array<float, param::NPadRows>& residHelixY)
1393 {
1394  // this fast algebraic circle fit is described here:
1395  // https://dtcenter.org/met/users/docs/write_ups/circle_fit.pdf
1396  float xMean = 0.f;
1397  float yMean = 0.f;
1398 
1399  for (int i = nCl; i--;) {
1400  xMean += x[i];
1401  yMean += y[i];
1402  }
1403  xMean /= nCl;
1404  yMean /= nCl;
1405  // define sums needed for circular fit
1406  float su2 = 0.f, sv2 = 0.f, suv = 0.f, su3 = 0.f, sv3 = 0.f, su2v = 0.f, suv2 = 0.f;
1407  for (int i = nCl; i--;) {
1408  float ui = x[i] - xMean;
1409  float vi = y[i] - yMean;
1410  float ui2 = ui * ui;
1411  float vi2 = vi * vi;
1412  suv += ui * vi;
1413  su2 += ui2;
1414  sv2 += vi2;
1415  su3 += ui2 * ui;
1416  sv3 += vi2 * vi;
1417  su2v += ui2 * vi;
1418  suv2 += ui * vi2;
1419  }
1420  float rhsU = .5f * (su3 + suv2);
1421  float rhsV = .5f * (sv3 + su2v);
1422  float det = su2 * sv2 - suv * suv;
1423  float uc = (rhsU * sv2 - rhsV * suv) / det;
1424  float vc = (su2 * rhsV - suv * rhsU) / det;
1425  float r2 = uc * uc + vc * vc + (su2 + sv2) / nCl;
1426  xc = uc + xMean;
1427  yc = vc + yMean;
1428  r = sqrt(r2);
1429  // write residuals to residHelixY
1430  for (int i = nCl; i--;) {
1431  float dx = x[i] - xc;
1432  float dxr = r2 - dx * dx;
1433  float ys = dxr > 0 ? sqrt(dxr) : 0.f; // distance of point in y from the circle center (using fit results for r and xc)
1434  float dy = y[i] - yc; // distance of point in y from the circle center (using fit result for yc)
1435  float dysp = dy - ys;
1436  float dysm = dy + ys;
1437  residHelixY[i] = fabsf(dysp) < fabsf(dysm) ? dysp : dysm;
1438  }
1439  // printf("r = %.4f m, xc = %.4f, yc = %.4f\n", r/100.f, xc, yc);
1440 }
