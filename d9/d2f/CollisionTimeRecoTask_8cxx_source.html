34void CollisionTimeRecoTask::processTF(const gsl::span<const o2::ft0::Digit>& digits,
35 const gsl::span<const o2::ft0::ChannelData>& channels,
36 std::vector<o2::ft0::RecPoints>& vecRecPoints,
37 std::vector<o2::ft0::ChannelDataFloat>& vecChData)
38{
39 for (const auto& digit : digits) {
40 if (!ChannelFilterParam::Instance().checkTCMbits(digit.getTriggers().getTriggersignals())) {
41 continue;
42 }
43 const auto channelsPerDigit = digit.getBunchChannelData(channels);
44 vecRecPoints.emplace_back(processDigit(digit, channelsPerDigit, vecChData));
45 }
46}
47RP CollisionTimeRecoTask::processDigit(const o2::ft0::Digit& digit,
48 const gsl::span<const o2::ft0::ChannelData> inChData,
49 std::vector<o2::ft0::ChannelDataFloat>& outChData)
50{
51 LOG(debug) << "Running reconstruction on new event";
52 const int firstEntry = outChData.size();
53 unsigned int ndigitsA = 0;
54 unsigned int ndigitsC = 0;
55 float sideAtime = 0;
56 float sideCtime = 0;
57
58 constexpr int nMCPsA = 4 * Geometry::NCellsA;
59
60 int nch{0};
61 bool isActiveA = false;
62 bool isActiveC = false;
63 bool isFlangeEvent = false;
64
65 for (const auto& channelData : inChData) {
66 if (channelData.ChId >= NCHANNELS) {
67 // Reference channels shouldn't participate in reco at all!
68 continue;
69 }
70 if (mDeadChannelMap && !mDeadChannelMap->isChannelAlive(channelData.ChId)) {
71 LOG(debug) << "Channel " << channelData.ChId << " is dead - discarding data";
72 continue;
73 }
74 const float timeInPS = getTimeInPS(channelData);
75 if (ChannelFilterParam::Instance().checkAll(channelData)) {
76 outChData.emplace_back(channelData.ChId, timeInPS, (float)channelData.QTCAmpl, channelData.ChainQTC);
77 nch++;
78 }
79 const bool isOkForTimeCalc = TimeFilterParam::Instance().checkAll(channelData);
80 // only signals which satisfy conditions may participate in time calculation
81 if (channelData.ChId < nMCPsA) {
82 // A-side
83 if (isOkForTimeCalc) {
84 sideAtime += timeInPS;
85 ndigitsA++;
86 }
87 isActiveA = true;
88 } else {
89 // C-side
90 if (isOkForTimeCalc) {
91 sideCtime += timeInPS;
92 ndigitsC++;
93 }
94 isActiveC = true;
95 isFlangeEvent |= channelData.CFDTime < -350 && channelData.CFDTime > -450;
96 }
97 }
98 std::array<short, 4> mCollisionTime = {RP::sDummyCollissionTime, RP::sDummyCollissionTime, RP::sDummyCollissionTime, RP::sDummyCollissionTime};
99
100 mCollisionTime[TimeA] = (ndigitsA > 0) ? std::round(sideAtime / ndigitsA) : RP::sDummyCollissionTime; // 2 * o2::InteractionRecord::DummyTime;
101 mCollisionTime[TimeC] = (ndigitsC > 0) ? std::round(sideCtime / ndigitsC) : RP::sDummyCollissionTime; // 2 * o2::InteractionRecord::DummyTime;
102
103 if (ndigitsA > 0 && ndigitsC > 0) {
104 mCollisionTime[Vertex] = (mCollisionTime[TimeA] - mCollisionTime[TimeC]) / 2.;
105 mCollisionTime[TimeMean] = (mCollisionTime[TimeA] + mCollisionTime[TimeC]) / 2.;
106 } else {
107 mCollisionTime[TimeMean] = std::min(mCollisionTime[TimeA], mCollisionTime[TimeC]);
108 }
109 const uint8_t extraTrgWord = RecPoints::makeExtraTrgWord(isActiveA, isActiveC, isFlangeEvent);
110 return RecPoints(firstEntry, nch, digit.mIntRecord, mCollisionTime, digit.mTriggers, extraTrgWord);
111}

119float CollisionTimeRecoTask::getTimeInPS(const o2::ft0::ChannelData& channelData)
120{
121 // Getting time offset
122 float offsetChannel{0};
123 if (mTimeCalibObject) {
124 // Temporary, will be changed to status bit checking
125 // Check statistics
126 const auto& stat = mTimeCalibObject->mTime[channelData.ChId].mStat;
127 const bool isEnoughStat = stat > CalibParam::Instance().mMaxEntriesThreshold;
128 const bool isNotGoogStat = stat > CalibParam::Instance().mMinEntriesThreshold && !isEnoughStat;
129 // Check fit quality
130 const auto& meanGaus = mTimeCalibObject->mTime[channelData.ChId].mGausMean;
131 const auto& meanHist = mTimeCalibObject->mTime[channelData.ChId].mStatMean;
132 const auto& sigmaGaus = mTimeCalibObject->mTime[channelData.ChId].mGausRMS;
133 const auto& rmsHist = mTimeCalibObject->mTime[channelData.ChId].mStatRMS;
134 const bool isGoodFitResult = (mTimeCalibObject->mTime[channelData.ChId].mStatusBits & 1) > 0;
135 const bool isBadFit = std::abs(meanGaus - meanHist) > CalibParam::Instance().mMaxDiffMean || rmsHist < CalibParam::Instance().mMinRMS || sigmaGaus > CalibParam::Instance().mMaxSigma;
136
137 if (isEnoughStat && isGoodFitResult && !isBadFit) {
138 offsetChannel = meanGaus;
139 } else if ((isNotGoogStat || isEnoughStat) && isBadFit) {
140 offsetChannel = meanHist;
141 }
142 }
143 // Getting slewing offset
144 float slewoffset{0};
145 const auto& gr = mCalibSlew[static_cast<int>(channelData.getFlag(o2::ft0::ChannelData::EEventDataBit::kNumberADC))][channelData.ChId];
146 slewoffset = gr.Eval(channelData.QTCAmpl);
147
148 // Final calculation
149 const float globalOffset = offsetChannel + slewoffset;
150 return (static_cast<float>(channelData.CFDTime) - globalOffset) * Geometry::ChannelWidth;
151}
