46class IOTOFDPLDigitizerTask : o2::base::BaseDPLDigitizer
47{
48 public:
49 using BaseDPLDigitizer::init;
50
51 IOTOFDPLDigitizerTask(bool mctruth = true) : BaseDPLDigitizer(o2::base::InitServices::FIELD | o2::base::InitServices::GEOM),
52 mWithMCTruth(mctruth) {}
53
54 void initDigitizerTask(framework::InitContext& ic) override
55 {
56 mDisableQED = ic.options().get<bool>("disable-qed");
57
58 auto geom = GeometryTGeo::Instance();
59 geom->fillMatrixCache(o2::math_utils::bit2Mask(o2::math_utils::TransformType::L2G)); // make sure L2G matrices are loaded
60
61 mDigitizer.setGeometry(geom);
62 mDigitizer.setChargeThreshold(-1000.f);
63 mDigitizer.init();
64 }
65
66 void run(framework::ProcessingContext& pc)
67 {
68 if (mFinished) {
69 return;
70 }
71 mFirstOrbitTF = pc.services().get<o2::framework::TimingInfo>().firstTForbit;
72 const o2::InteractionRecord firstIR(0, mFirstOrbitTF);
73
74 // read collision context from input
75 auto context = pc.inputs().get<o2::steer::DigitizationContext*>("collisioncontext");
76 context->initSimChains(mID, mSimChains);
77 const bool withQED = context->isQEDProvided() && !mDisableQED;
78 auto& timesview = context->getEventRecords();
79 LOG(info) << "GOT " << timesview.size() << " COLLISION TIMES";
80 LOG(info) << "SIMCHAINS " << mSimChains.size();
81
82 // if there is nothing to do ... return
83 if (timesview.empty()) {
84 return;
85 }
86
87 TStopwatch timer;
88 timer.Start();
89 LOG(info) << " CALLING TF3 DIGITIZATION ";
90
91 mDigitizer.setDigits(&mDigits);
92 mDigitizer.setROFRecords(&mROFRecords);
93 mDigitizer.setROFRecordIR(firstIR);
94 if (mWithMCTruth) {
95 mDigitizer.setMCLabels(&mLabels);
96 }
97
98 auto& eventParts = context->getEventParts(withQED);
99 // loop over all composite collisions given from context
100 // (aka loop over all the interaction records)
101 // o2::InteractionTimeRecord firstorbit(o2::InteractionRecord(0, o2::raw::HBFUtils::Instance().orbitFirstSampled), 0.0);
102 for (int collID = 0; collID < timesview.size(); ++collID) {
103 o2::InteractionTimeRecord orbit(timesview[collID]);
104 // orbit += firstorbit
105 mDigitizer.setEventTime(orbit);
106
107 // for each collision, loop over the constituents event and source IDs
108 // (background signal merging is basically taking place here)
109 for (const auto& part : eventParts[collID]) {
110
111 // get the hits for this event and this source
112 mHits.clear();
113 context->retrieveHits(mSimChains, o2::detectors::SimTraits::DETECTORBRANCHNAMES[mID][0].c_str(), part.sourceID, part.entryID, &mHits);
114
115 if (mHits.size() > 0) {
116 mDigits.clear();
117 if (mWithMCTruth) {
118 mLabels.clear();
119 }
120
121 LOG(debug) << "For collision " << collID << " eventID " << part.entryID << " found " << mHits.size() << " hits ";
122 mDigitizer.process(&mHits, part.entryID, part.sourceID); // call actual digitization procedure
123 }
124 }
125 }
126 if (mDigitizer.isContinuous()) {
127 LOG(debug) << "Number of digits before final flush: " << mDigits.size();
128 mDigits.clear();
129 if (mWithMCTruth) {
130 mLabels.clear();
131 }
132 LOG(debug) << "Final flushing for continuous mode";
133 mDigitizer.fillOutputContainer();
134 LOG(debug) << "Number of digits after final flush: " << mDigits.size();
135 }
136
137 // here we have all digits and we can send them to consumer (aka snapshot it onto output)
138 LOG(debug) << "Digitization finished with " << mDigits.size() << " digits and " << mROFRecords.size() << " ROF records";
139 pc.outputs().snapshot(Output{mOrigin, "DIGITS", 0}, mDigits);
140 pc.outputs().snapshot(Output{mOrigin, "DIGITSROF", 0}, mROFRecords);
141 if (mWithMCTruth) {
142 auto& sharedlabels = pc.outputs().make<o2::dataformats::ConstMCTruthContainer<o2::MCCompLabel>>(Output{mOrigin, "DIGITSMCTR", 0});
143 mLabels.flatten_to(sharedlabels);
144 // free space of existing label containers
145 mLabels.clear_andfreememory();
146
147 // write dummy MC2ROF vector to keep writer/readers backward compatible
148 // NOTE: Steer/DigitizerWorkflow/src/ITSMFTDigitizerSpec.cxx also uses dummy MC2ROF
149 static std::vector<o2::itsmft::MC2ROFRecord> dummyMC2ROF;
150 pc.outputs().snapshot(Output{mOrigin, "DIGITSMC2ROF", 0}, dummyMC2ROF);
151 }
152
153 timer.Stop();
154 LOG(info) << "Digitization took " << timer.CpuTime() << "s";
155
156 // we should be only called once; tell DPL that this process is ready to exit
157 pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
158
159 mFinished = true;
160 }
161
162 private:
163 bool mDisableQED = false;
164 bool mWithMCTruth{true};
165 bool mFinished{false};
166 unsigned long mFirstOrbitTF = 0x0;
167 const o2::detectors::DetID mID{o2::detectors::DetID::TF3};
168 const o2::header::DataOrigin mOrigin{o2::header::gDataOriginTF3};
169 o2::iotof::Digitizer mDigitizer{};
170 std::vector<o2::iotof::Digit> mDigits{};
171 std::vector<o2::itsmft::ROFRecord> mROFRecords{};
172 std::vector<o2::itsmft::Hit> mHits{};
173 std::vector<o2::itsmft::Hit>* mHitsP{&mHits};
174 o2::dataformats::MCTruthContainer<o2::MCCompLabel> mLabels{};
175 std::vector<TChain*> mSimChains{};
176 o2::parameters::GRPObject::ROMode mROMode = o2::parameters::GRPObject::PRESENT; // readout mode
177};

179std::vector<o2::framework::OutputSpec> makeOutChannels(o2::header::DataOrigin detOrig, bool mctruth)
180{
181 std::vector<o2::framework::OutputSpec> outputs;
182 outputs.emplace_back(detOrig, "DIGITS", o2::framework::Lifetime::Timeframe);
183 outputs.emplace_back(detOrig, "DIGITSROF", o2::framework::Lifetime::Timeframe);
184 if (mctruth) {
185 outputs.emplace_back(detOrig, "DIGITSMC2ROF", o2::framework::Lifetime::Timeframe);
186 outputs.emplace_back(detOrig, "DIGITSMCTR", o2::framework::Lifetime::Timeframe);
187 }
188 outputs.emplace_back(detOrig, "ROMode", 0, o2::framework::Lifetime::Timeframe);
189 return outputs;
190}

192o2::framework::DataProcessorSpec getIOTOFDigitizerSpec(int channel, bool mctruth)
193{
194 std::vector<o2::framework::InputSpec> inputs;
195 inputs.emplace_back("collisioncontext", "SIM", "COLLISIONCONTEXT", static_cast<o2::header::DataHeader::SubSpecificationType>(channel), o2::framework::Lifetime::Timeframe);
196 inputs.emplace_back("IOTOF_aptsresp", "TF3", "APTSRESP", 0, o2::framework::Lifetime::Condition, o2::framework::ccdbParamSpec("IT3/Calib/APTSResponse"));
197
198 const std::string detStr = o2::detectors::DetID::getName(o2::detectors::DetID::TF3);
199 return o2::framework::DataProcessorSpec{detStr + "Digitizer",
200 inputs,
201 makeOutChannels(o2::header::gDataOriginTF3, mctruth),
202 o2::framework::AlgorithmSpec{o2::framework::adaptFromTask<IOTOFDPLDigitizerTask>(mctruth)},
203 o2::framework::Options{
204 {"disable-qed", o2::framework::VariantType::Bool, false, {"disable QED handling"}},
205 {"local-response-file", o2::framework::VariantType::String, "", {"use response file saved locally at this path/filename"}}}};
206}
