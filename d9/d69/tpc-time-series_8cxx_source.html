38WorkflowSpec defineDataProcessing(ConfigContext const& config)
39{
40 WorkflowSpec workflow;
41 o2::conf::ConfigurableParam::updateFromString(config.options().get<std::string>("configKeyValues"));
42 const bool disableWriter = config.options().get<bool>("disable-root-output");
43 const bool enableUnbinnedWriter = config.options().get<bool>("enable-unbinned-root-output");
44 auto src = o2::dataformats::GlobalTrackID::getSourcesMask(config.options().get<std::string>("track-sources"));
45 auto materialType = static_cast<o2::base::Propagator::MatCorrType>(config.options().get<int>("material-type"));
46 workflow.emplace_back(o2::tpc::getTPCTimeSeriesSpec(disableWriter, materialType, enableUnbinnedWriter, src));
47 if (!disableWriter) {
48 workflow.emplace_back(o2::tpc::getTPCTimeSeriesWriterSpec());
49 }
50 return workflow;
51}
