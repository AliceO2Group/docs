168GPUdi() void GPUTPCGMPolynomialField::GetField(float x, float y, float z, float* B) const
169{
170 const float* fBxS = &mTpcBx[1];
171 const float* fByS = &mTpcBy[1];
172 const float* fBzS = &mTpcBz[1];
173
174 const float f[NTPCM - 1] = {x, y, z, x * x, x * y, x * z, y * y, y * z, z * z};
175 float bx = mTpcBx[0], by = mTpcBy[0], bz = mTpcBz[0];
176 for (int32_t i = NTPCM - 1; i--;) {
177 // for (int32_t i=0;i<NTPCM-1; i++){
178 bx += fBxS[i] * f[i];
179 by += fByS[i] * f[i];
180 bz += fBzS[i] * f[i];
181 }
182 B[0] = bx;
183 B[1] = by;
184 B[2] = bz;
185}

199GPUdi() void GPUTPCGMPolynomialField::GetPolynomsTrd(float x, float y, float z, float f[NTRDM])
200{
201 float xx = x * x, xy = x * y, xz = x * z, yy = y * y, yz = y * z, zz = z * z;
202 f[0] = 1.f;
203 f[1] = x;
204 f[2] = y;
205 f[3] = z;
206 f[4] = xx;
207 f[5] = xy;
208 f[6] = xz;
209 f[7] = yy;
210 f[8] = yz;
211 f[9] = zz;
212 f[10] = x * xx;
213 f[11] = x * xy;
214 f[12] = x * xz;
215 f[13] = x * yy;
216 f[14] = x * yz;
217 f[15] = x * zz;
218 f[16] = y * yy;
219 f[17] = y * yz;
220 f[18] = y * zz;
221 f[19] = z * zz;
222}

224GPUdi() void GPUTPCGMPolynomialField::GetFieldTrd(float x, float y, float z, float* B) const
225{
226 float f[NTRDM];
227 GetPolynomsTrd(x, y, z, f);
228 float bx = 0.f, by = 0.f, bz = 0.f;
229 for (int32_t i = 0; i < NTRDM; i++) {
230 bx += mTrdBx[i] * f[i];
231 by += mTrdBy[i] * f[i];
232 bz += mTrdBz[i] * f[i];
233 }
234 B[0] = bx;
235 B[1] = by;
236 B[2] = bz;
237}
