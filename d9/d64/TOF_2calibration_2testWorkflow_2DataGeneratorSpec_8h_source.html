37class TFDispatcher : public o2::framework::Task
38{
39 public:
40 TFDispatcher(int slot, int ngen, int nlanes, int latency) : mSlot(slot), mNGen(ngen), mNLanes(nlanes), mLatency(latency) {}
41
42 void init(o2::framework::InitContext& ic) final
43 {
44 mMaxTF = ic.options().get<int64_t>("max-timeframes");
45 mMinSize = ic.options().get<int>("min-number-of-info");
46 }
47
48 void run(o2::framework::ProcessingContext& pc) final
49 {
50 for (auto& input : pc.inputs()) {
51 auto tfid = header::get<o2::framework::DataProcessingHeader*>(input.header)->startTime;
52 if (tfid >= mMaxTF - 1) {
53 LOG(info) << "Data generator reached TF " << tfid << ", stopping";
54 pc.services().get<o2::framework::ControlService>().endOfStream();
55 pc.services().get<o2::framework::ControlService>().readyToQuit(o2::framework::QuitRequest::Me);
56 if (!acceptTF(tfid)) {
57 return;
58 }
59 break;
60 }
61 if (!acceptTF(tfid)) {
62 return;
63 }
64 }
65 int size = mMinSize + gRandom->Integer(100); // push dummy output
66 usleep(mLatency);
67 pc.outputs().snapshot(o2::framework::OutputRef{"output", 0}, size);
68 }
69
70 bool acceptTF(int tfid)
71 {
72
73 // check if the current TF should be processed by this instance of the generator
74
75 int targetSlot = (tfid / mNLanes) % mNGen;
76 if (targetSlot != mSlot) {
77 LOG(info) << "tfid = " << tfid << ", mNLanes = " << mNLanes << ", mNGen = " << mNGen << ", mSlot = " << mSlot << " target slot = " << targetSlot << ": discarded";
78 return false;
79 }
80 LOG(info) << "tfid = " << tfid << ", mNLanes = " << mNLanes << ", mNGen = " << mNGen << ", mSlot = " << mSlot << " target slot = " << targetSlot << ": accepted";
81 return true;
82 }
83
84 private:
85 uint64_t mMaxTF = 1;
86 int mSlot = 1;
87 int mNGen = 1;
88 int mNLanes = 1;
89 int mLatency = 0;
90 int mMinSize = 100;
91};

93class TFProcessorCalibInfoTOF : public o2::framework::Task
94{
95 public:
96 TFProcessorCalibInfoTOF(int latency, int latencyRMS) : mMeanLatency(latency), mLatencyRMS(latencyRMS){};
97
98 void init(o2::framework::InitContext& ic) final
99 {
100 mDevCopy = ic.services().get<const o2::framework::DeviceSpec>().inputTimesliceId;
101 gRandom->SetSeed(mDevCopy);
102 mTOFChannelCalib = ic.options().get<bool>("do-TOF-channel-calib");
103 mTOFChannelCalibInTestMode = ic.options().get<bool>("do-TOF-channel-calib-in-test-mode");
104 LOG(info) << "TFProcessorCalibInfoTOFCopy: " << mDevCopy << " MeanLatency: " << mMeanLatency << " LatencyRMS: " << mLatencyRMS << " DoTOFChannelCalib: " << mTOFChannelCalib
105 << " DoTOFChannelCalibInTestMode: " << mTOFChannelCalibInTestMode;
106
107 for (int i = 0; i < o2::tof::Geo::NCHANNELS; i++) {
108 mChannelShifts[i] = (2000. / o2::tof::Geo::NCHANNELS) * i + (-1000.); // shift needs to be always the same for a channel; in this way we make them all in [-1000, 1000], and shifting with the channel index
109 }
110 }
111
112 void run(o2::framework::ProcessingContext& pc) final
113 {
114 auto tfcounter = o2::header::get<o2::framework::DataProcessingHeader*>(pc.inputs().get("input").header)->startTime;
115 // introduceDelay
116 uint32_t delay = std::abs(gRandom->Gaus(mMeanLatency, mLatencyRMS));
117 LOG(info) << "TFProcessorCalibInfoTOFCopy: " << mDevCopy << " Simulate latency of " << delay << " mcs for TF " << tfcounter;
118 usleep(delay);
119
120 // push dummy output
121 auto size = pc.inputs().get<int>("input");
122 auto& output = pc.outputs().make<std::vector<o2::dataformats::CalibInfoTOF>>(o2::framework::OutputRef{"output", 0});
123 output.reserve(size);
124
125 double clockShift = 1e3 * std::sin(tfcounter / 100. * o2::constants::math::PI); // in ps
126
127 for (int i = size; i--;) {
128 if (!mTOFChannelCalib) {
129 output.emplace_back(gRandom->Integer(o2::tof::Geo::NCHANNELS), 0, gRandom->Gaus(clockShift, 100.), 0, 0);
130 } else {
131 int channel = mTOFChannelCalibInTestMode ? gRandom->Integer(100) : gRandom->Integer(o2::tof::Geo::NCHANNELS);
132 double value = gRandom->Gaus(mChannelShifts[channel], 100.); // in ps
133 double tot = gRandom->Gaus(12, 2); // in ns
134 output.emplace_back(channel, 0, value, tot, 0);
135 }
136 }
137 }
138
139 private:
140 int mDevCopy = 0;
141 uint32_t mMeanLatency = 0;
142 uint32_t mLatencyRMS = 1;
143 bool mTOFChannelCalib = false;
144 bool mTOFChannelCalibInTestMode = false;
145 double mChannelShifts[o2::tof::Geo::NCHANNELS];
146};

148class TFProcessorDiagnostic : public o2::framework::Task
149{
150 public:
151 TFProcessorDiagnostic(int latency, int latencyRMS) : mMeanLatency(latency), mLatencyRMS(latencyRMS){};
152
153 void init(o2::framework::InitContext& ic) final
154 {
155
156 mDevCopy = ic.services().get<const o2::framework::DeviceSpec>().inputTimesliceId;
157 gRandom->SetSeed(mDevCopy);
158 LOG(info) << "TFProcessorDiagnosticCopy: " << mDevCopy;
159
160 auto size = ic.options().get<int>("n-diag-words"); // number of diagnostic words that we want to simulate; only these will then be present
161 mProb.resize(size);
162 mDiagnosticPattern.resize(size);
163
164 for (int i = 0; i < size; ++i) {
165 int crateId = gRandom->Integer(72);
166 int trmId = gRandom->Integer(10);
167 mProb[i] = 0;
168 mDiagnosticPattern[i] = (ULong64_t(trmId) << 32) + (ULong64_t(crateId) << 36);
169 for (int j = 0; j < 28; ++j) { // adding diagnostic patterns for each word; we can have 28 at max, but we don't want all of them (we use a cut on a probability of 30%)
170 if (gRandom->Rndm() < 0.3) {
171 mDiagnosticPattern[i] += 1 << j;
172 mProb[i] = gRandom->Rndm(); // probability for each diagnostic word
173 }
174 }
175 }
176 }
177
178 void run(o2::framework::ProcessingContext& pc) final
179 {
180 auto tfcounter = o2::header::get<o2::framework::DataProcessingHeader*>(pc.inputs().get("input").header)->startTime;
181 // introduceDelay
182 uint32_t delay = std::abs(gRandom->Gaus(mMeanLatency, mLatencyRMS));
183 LOG(info) << "TFProcessorDiagnosticCopy: " << mDevCopy << " Simulate latency of " << delay << " mcs for TF " << tfcounter;
184 usleep(delay);
185
186 // push dummy output
187 auto& outputDiagnostic = pc.outputs().make<o2::tof::Diagnostic>(o2::framework::OutputRef{"output", 0});
188 for (int iOrbit = 0; iOrbit < o2::tof::Utils::getNOrbitInTF(); ++iOrbit) {
189 for (int iROwindow = 0; iROwindow < o2::tof::Geo::NWINDOW_IN_ORBIT; ++iROwindow) {
190 outputDiagnostic.fillROW();
191 for (int i = 0; i < mDiagnosticPattern.size(); ++i) {
192 if (gRandom->Rndm() < mProb[i]) {
193 outputDiagnostic.fill(mDiagnosticPattern[i]);
194 }
195 }
196 }
197 }
198 LOG(debug) << "diagnostic for TF " << tfcounter << " --> ";
199 outputDiagnostic.print();
200 }
201
202 private:
203 int mDevCopy = 0;
204 std::vector<uint64_t> mDiagnosticPattern;
205 std::vector<float> mProb;
206 o2::tof::Diagnostic mDiagnosticFrequency;
207 uint32_t mMeanLatency = 0;
208 uint32_t mLatencyRMS = 1;
209};
