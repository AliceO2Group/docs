119 DataProcessorSpec getITSTPCMatchingQCDevice(bool useMC, bool doK0QC, std::string trkSources)
120 {
121  std::vector<OutputSpec> outputs;
122  outputs.emplace_back("GLO", "ITSTPCMATCHQC", 0, Lifetime::Sporadic);
123 
124  auto dataRequest = std::make_shared<o2::globaltracking::DataRequest>();
125  GID::mask_t srcMask = GID::getSourcesMask(trkSources);
126  dataRequest->requestTracks(srcMask, useMC);
127  if (doK0QC) {
128  dataRequest->requestPrimaryVertices(useMC);
129  dataRequest->requestSecondaryVertices(useMC);
130  }
131  auto ccdbRequest = std::make_shared<o2::base::GRPGeomRequest>(false, // orbitResetTime
132  false, // GRPECS=true
133  false, // GRPLHCIF
134  true, // GRPMagField
135  false, // askMatLUT
136  o2::base::GRPGeomRequest::None, // geometry
137  dataRequest->inputs);
138 
139  dataRequest->inputs.emplace_back("SVParam", "GLO", "SVPARAM", 0, Lifetime::Condition, ccdbParamSpec("GLO/Config/SVertexerParam"));
140  return DataProcessorSpec{
141  .name = "itstpc-matching-qc",
142  .inputs = dataRequest->inputs,
143  .outputs = outputs,
144  .algorithm = AlgorithmSpec{adaptFromTask<o2::globaltracking::ITSTPCMatchingQCDevice>(dataRequest, ccdbRequest, useMC, doK0QC, trkSources)},
145  };
146 }
