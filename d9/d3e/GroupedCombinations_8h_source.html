24 // Create an instance of a tuple interleaved from given tuples
25 template <typename... T1s, typename... T2s, std::size_t... Is>
26 auto interleaveTuplesImpl(std::tuple<T1s...>& t1, std::tuple<T2s...>& t2, std::index_sequence<Is...>)
27 {
28  return std::tuple_cat(std::make_tuple(std::get<Is>(t1), std::get<Is>(t2))...);
29 }

31 template <typename... T1s, typename... T2s>
32 auto interleaveTuples(std::tuple<T1s...>& t1, std::tuple<T2s...>& t2)
33 {
34  return interleaveTuplesImpl(t1, t2, std::index_sequence_for<T1s...>());

40 template <typename G, typename A>
41 expressions::BindingNode getMatchingIndexNode()
42 {
43  using external_index_columns_pack = typename A::external_index_columns_t;
44  using selected_indices_t = selected_pack_multicondition<is_index_to_g_t, pack<G>, external_index_columns_pack>;
45  static_assert(pack_size(selected_indices_t{}) == 1, "No matching index column from associated to grouping");
46  using index_column_t = pack_head_t<selected_indices_t>;
47  return expressions::BindingNode{index_column_t::mLabel, o2::framework::TypeIdHelpers::uniqueId<typename index_column_t::column_t>(), expressions::selectArrowType<typename index_column_t::type>()};
48 }

50 template <typename T1, typename GroupingPolicy, typename BP, typename G, typename... As>
51 struct GroupedCombinationsGenerator {
52  using GroupedIteratorType = pack_to_tuple_t<interleaved_pack_t<repeated_type_pack_t<typename G::iterator, sizeof...(As)>, pack<As...>>>;
53 
54  struct GroupedIterator : public GroupingPolicy {
55  public:
56  using reference = GroupedIteratorType&;
57  using value_type = GroupedIteratorType;
58  using pointer = GroupedIteratorType*;
59  using iterator_category = std::forward_iterator_tag;
60 
61  GroupedIterator(const GroupingPolicy& groupingPolicy, SliceCache* cache_)
62  : GroupingPolicy(groupingPolicy), mIndexColumns{getMatchingIndexNode<G, As>()...}, cache{cache_}
63  {
64  }
65  template <typename... T2s>
66  GroupedIterator(const GroupingPolicy& groupingPolicy, const G& grouping, const std::tuple<T2s...>& associated, SliceCache* cache_)
67  : GroupingPolicy(groupingPolicy),
68  mGrouping{std::make_shared<G>(std::vector{grouping.asArrowTable()})},
69  mAssociated{std::make_shared<std::tuple<As...>>(std::make_tuple(std::get<has_type_at<As>(pack<T2s...>{})>(associated)...))},
70  mIndexColumns{getMatchingIndexNode<G, As>()...},
71  cache{cache_}
72  {
73  if constexpr (soa::is_soa_filtered_v<std::decay_t<G>>) {
74  mGrouping = std::make_shared<G>(std::vector{grouping.asArrowTable()}, grouping.getSelectedRows());
75  } else {
76  mGrouping = std::make_shared<G>(std::vector{grouping.asArrowTable()});
77  }
78  setMultipleGroupingTables<sizeof...(As)>(grouping);
79  if (!this->mIsEnd) {
80  setCurrentGroupedCombination();
81  }
82  }
83 
84  GroupedIterator(GroupedIterator const&) = default;
85  GroupedIterator& operator=(GroupedIterator const&) = default;
86  ~GroupedIterator() = default;

88  template <typename... T2s>
89  void setTables(const G& grouping, const std::tuple<T2s...>& associated)
90  {
91  if constexpr (soa::is_soa_filtered_v<std::decay_t<G>>) {
92  mGrouping = std::make_shared<G>(std::vector{grouping.asArrowTable()}, grouping.getSelectedRows());
93  } else {
94  mGrouping = std::make_shared<G>(std::vector{grouping.asArrowTable()});
95  }
96  mAssociated = std::make_shared<std::tuple<As...>>(std::make_tuple(std::get<has_type_at<As>(pack<T2s...>{})>(associated)...));
97  setMultipleGroupingTables<sizeof...(As)>(grouping);
98  if (!this->mIsEnd) {
99  setCurrentGroupedCombination();
100  }
101  }
102 
103  template <std::size_t N, typename T, typename... Args>
104  void setMultipleGroupingTables(const T& param, const Args&... args)
105  {
106  if constexpr (N == 1) {
107  GroupingPolicy::setTables(param, args...);
108  } else {
109  setMultipleGroupingTables<N - 1>(param, param, args...);
110  }
111  }
112 
113  void moveToEnd()
114  {
115  GroupingPolicy::moveToEnd();
116  }
117 
118  // prefix increment
119  GroupedIterator& operator++()
120  {
121  if (!this->mIsEnd) {
122  this->addOne();
123  setCurrentGroupedCombination();
124  }
125  return *this;
126  }
127  // postfix increment
128  GroupedIterator operator++(int /*unused*/)
129  {
130  GroupedIterator copy(*this);
131  operator++();
132  return copy;
133  }
134  // return reference
135  reference operator*()
136  {
137  return *mCurrentGrouped;
138  }
139  bool operator==(const GroupedIterator& rh) const
140  {
141  return (this->mIsEnd && rh.mIsEnd) || (this->mCurrent == rh.mCurrent);
142  }
143  bool operator!=(const GroupedIterator& rh) const
144  {
145  return !(*this == rh);
146  }
147 
148  private:
149  std::tuple<As...> getAssociatedTables()
150  {
151  return doGetAssociatedTables(std::make_index_sequence<sizeof...(As)>());
152  }
153 
154  template <std::size_t... Is>
155  std::tuple<As...> doGetAssociatedTables(std::index_sequence<Is...>)
156  {
157  return std::make_tuple(getAssociatedTable<Is>()...);
158  }
159 
160  template <std::size_t I>
161  auto getAssociatedTable()
162  {
163  uint64_t ind = *std::get<0>(std::get<I>(this->mCurrent).getIndices());
164  if (std::get<I>(*mAssociated).size() == 0) {
165  return std::get<I>(*mAssociated);
166  }
167  return std::get<I>(*mAssociated).sliceByCached(mIndexColumns[I], ind, *cache);
168  }
169 
170  void setCurrentGroupedCombination()
171  {
172  if (!this->mIsEnd) {
173  auto& currentGrouping = GroupingPolicy::mCurrent;
174  mSlices.emplace(getAssociatedTables());
175  mCurrentGrouped.emplace(interleaveTuples(currentGrouping, *mSlices));
176  }
177  }
178 
179  std::array<expressions::BindingNode, sizeof...(As)> mIndexColumns;
180  std::shared_ptr<G> mGrouping;
181  std::shared_ptr<std::tuple<As...>> mAssociated;
182  std::optional<std::tuple<As...>> mSlices;
183  std::optional<GroupedIteratorType> mCurrentGrouped;
184  SliceCache* cache = nullptr;
185  };
