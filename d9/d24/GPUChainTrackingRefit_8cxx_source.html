25int32_t GPUChainTracking::RunRefit()
26{
27 bool doGPU = GetRecoStepsGPU() & RecoStep::Refit;
28 GPUTrackingRefitProcessor& Refit = processors()->trackingRefit;
29 GPUTrackingRefitProcessor& RefitShadow = doGPU ? processorsShadow()->trackingRefit : Refit;
30
31 [[maybe_unused]] const auto& threadContext = GetThreadContext();
32 SetupGPUProcessor(&Refit, false);
33 RefitShadow.SetPtrsFromGPUConstantMem(processorsShadow(), doGPU ? &processorsDevice()->param : nullptr);
34 RefitShadow.SetPropagator(doGPU ? processorsShadow()->calibObjects.o2Propagator : GetO2Propagator());
35 RefitShadow.mPTracks = (doGPU ? processorsShadow() : processors())->tpcMerger.MergedTracks();
36 WriteToConstantMemory(RecoStep::Refit, (char*)&processors()->trackingRefit - (char*)processors(), &RefitShadow, sizeof(RefitShadow), 0);
37 // TransferMemoryResourcesToGPU(RecoStep::Refit, &Refit, 0);
38 if (param().rec.trackingRefitGPUModel) {
39 runKernel<GPUTrackingRefitKernel, GPUTrackingRefitKernel::mode0asGPU>(GetGrid(mIOPtrs.nMergedTracks, 0));
40 } else {
41 runKernel<GPUTrackingRefitKernel, GPUTrackingRefitKernel::mode1asTrackParCov>(GetGrid(mIOPtrs.nMergedTracks, 0));
42 }
43 // TransferMemoryResourcesToHost(RecoStep::Refit, &Refit, 0);
44 SynchronizeStream(0);
45 return 0;
46}
