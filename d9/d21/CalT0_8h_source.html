41  float calcT0av() const
42  {
43  if (mT0.size() == 0) {
44  return -1;
45  }
46  float sum = 0;
47  int counts = 0;
48  for (int iDet = 0; iDet < constants::MAXCHAMBER; ++iDet) {
49  if (mT0[iDet] > -5) {
50  sum += mT0[iDet];
51  ++counts;
52  }
53  }
54 
55  if (counts > 0) {
56  return sum / counts;
57  } else {
58  return -2;
59  }
60  }
