139template <typename T, typename... Args>
140static inline int64_t OCLsetKernelParameters_helper(cl_kernel& kernel, int32_t i, const T& firstParameter, const Args&... restOfParameters)
141{
142 int64_t retVal = clSetKernelArg(kernel, i, sizeof(T), &firstParameter);
143 if (retVal) {
144 return retVal;
145 }
146 return OCLsetKernelParameters_helper(kernel, i + 1, restOfParameters...);
147}

149template <typename... Args>
150static inline int64_t OCLsetKernelParameters(cl_kernel& kernel, const Args&... args)
151{
152 return OCLsetKernelParameters_helper(kernel, 0, args...);
153}

175template <typename K, typename... Args>
176inline int32_t GPUReconstructionOCLBackend::runKernelBackendInternal(const krnlSetupTime& _xyz, K& k, const Args&... args)
177{
178 auto& x = _xyz.x;
179 auto& y = _xyz.y;
180 auto& z = _xyz.z;
181 if (y.num <= 1) {
182 GPUFailedMsg(OCLsetKernelParameters(k, mInternals->mem_gpu, mInternals->mem_constant, y.start, args...));
183 } else {
184 GPUFailedMsg(OCLsetKernelParameters(k, mInternals->mem_gpu, mInternals->mem_constant, y.start, y.num, args...));
185 }
186
187 cl_event ev;
188 cl_event* evr;
189 bool tmpEvent = false;
190 if (z.ev == nullptr && mProcessingSettings.deviceTimers && mProcessingSettings.debugLevel > 0) {
191 evr = &ev;
192 tmpEvent = true;
193 } else {
194 evr = (cl_event*)z.ev;
195 }
196 GPUFailedMsg(clExecuteKernelA(mInternals->command_queue[x.stream], k, x.nThreads, x.nThreads * x.nBlocks, evr, (cl_event*)z.evList, z.nEvents));
197 if (mProcessingSettings.deviceTimers && mProcessingSettings.debugLevel > 0) {
198 cl_ulong time_start, time_end;
199 GPUFailedMsg(clWaitForEvents(1, evr));
200 GPUFailedMsg(clGetEventProfilingInfo(*evr, CL_PROFILING_COMMAND_START, sizeof(time_start), &time_start, nullptr));
201 GPUFailedMsg(clGetEventProfilingInfo(*evr, CL_PROFILING_COMMAND_END, sizeof(time_end), &time_end, nullptr));
202 _xyz.t = (time_end - time_start) * 1.e-9f;
203 if (tmpEvent) {
204 GPUFailedMsg(clReleaseEvent(ev));
205 }
206 }
207 return 0;
208}

210template <class T, int32_t I>
211int32_t GPUReconstructionOCLBackend::AddKernel(bool multi)
212{
213 std::string name(GetKernelName<T, I>());
214 if (multi) {
215 name += "_multi";
216 }
217 std::string kname("krnl_" + name);
218
219 cl_int ocl_error;
220 cl_kernel krnl = clCreateKernel(mInternals->program, kname.c_str(), &ocl_error);
221 if (GPUFailedMsgI(ocl_error)) {
222 GPUError("Error creating OPENCL Kernel: %s", name.c_str());
223 return 1;
224 }
225 mInternals->kernels.emplace_back(krnl, name);
226 return 0;
227}

229template <class T, int32_t I>
230inline uint32_t GPUReconstructionOCLBackend::FindKernel(int32_t num)
231{
232 std::string name(GetKernelName<T, I>());
233 if (num > 1) {
234 name += "_multi";
235 }
236
237 for (uint32_t k = 0; k < mInternals->kernels.size(); k++) {
238 if (mInternals->kernels[k].second == name) {
239 return (k);
240 }
241 }
242 GPUError("Could not find OpenCL kernel %s", name.c_str());
243 throw ::std::runtime_error("Requested unsupported OpenCL kernel");
244}
