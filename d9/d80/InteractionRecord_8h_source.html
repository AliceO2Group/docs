27struct InteractionRecord {
28 // information about bunch crossing and orbit
29 static constexpr uint16_t DummyBC = 0xffff;
30 static constexpr uint32_t DummyOrbit = 0xffffffff;
31 static constexpr double DummyTime = DummyBC * o2::constants::lhc::LHCBunchSpacingNS + DummyOrbit * o2::constants::lhc::LHCOrbitNS;
32 static constexpr int64_t MaxGlobalBCs = (int64_t(DummyOrbit) * o2::constants::lhc::LHCMaxBunches) + (o2::constants::lhc::LHCMaxBunches - 1);
33 static constexpr InteractionRecord getIRMaxBC() { return {o2::constants::lhc::LHCMaxBunches - 1, DummyOrbit}; }
34 uint16_t bc = DummyBC;
35 uint32_t orbit = DummyOrbit;
36
37 constexpr InteractionRecord() = default;
38
39 InteractionRecord(double tNS)
40 {
41 setFromNS(tNS);
42 }

44 constexpr InteractionRecord(uint16_t b, uint32_t orb) : bc(b), orbit(orb)
45 {
46 }

48 InteractionRecord(const InteractionRecord& src) = default;
49 InteractionRecord& operator=(const InteractionRecord& src) = default;

51 void clear()
52 {
53 bc = 0xffff;
54 orbit = 0xffffffff;
55 }
56
57 bool isDummy() const
58 {
59 return bc > o2::constants::lhc::LHCMaxBunches;
60 }

62 void setFromNS(double ns)
63 {
64 bc = ns2bc(ns, orbit);

67 static double bc2ns(int bc, unsigned int orbit)
68 {
69 return bc * o2::constants::lhc::LHCBunchSpacingNS + orbit * o2::constants::lhc::LHCOrbitNS;
70 }
71
72 static int ns2bc(double ns, unsigned int& orb)
73 {
74 long nb = std::round(ns / o2::constants::lhc::LHCBunchSpacingNS);
75 orb = nb / o2::constants::lhc::LHCMaxBunches;
76 return nb % o2::constants::lhc::LHCMaxBunches;
77 }
78
79 double bc2ns() const
80 {
81 return bc2ns(bc, orbit);
82 }
83
84 bool operator==(const InteractionRecord& other) const
85 {
86 return (bc == other.bc) && (orbit == other.orbit);
87 }
88
89 bool operator!=(const InteractionRecord& other) const
90 {
91 return (bc != other.bc) || (orbit != other.orbit);
92 }
93
94 int64_t differenceInBC(const InteractionRecord& other) const
95 {
96 // return difference in bunch-crossings
97 int64_t diffBC = int(bc) - other.bc;
98 if (orbit != other.orbit) {
99 diffBC += (int64_t(orbit) - other.orbit) * o2::constants::lhc::LHCMaxBunches;
100 }
101 return diffBC;
102 }
103
104 float differenceInBCNS(const InteractionRecord& other) const
105 {
106 // return difference in bunch-crossings in ns
107 return differenceInBC(other) * o2::constants::lhc::LHCBunchSpacingNS;
108 }
109
110 float differenceInBCMUS(const InteractionRecord& other) const
111 {
112 // return difference in bunch-crossings in us
113 return differenceInBC(other) * o2::constants::lhc::LHCBunchSpacingMUS;
114 }
115
116 int64_t toLong() const
117 {
118 // return as single long number
119 return (int64_t(orbit) * o2::constants::lhc::LHCMaxBunches) + bc;
120 }
121
122 void setFromLong(int64_t l)
123 {
124 // set from long BC counter
125 bc = l % o2::constants::lhc::LHCMaxBunches;
126 orbit = l / o2::constants::lhc::LHCMaxBunches;
127 }
128
129 static InteractionRecord long2IR(int64_t l)
130 {
131 // set from long BC counter
132 return {uint16_t(l % o2::constants::lhc::LHCMaxBunches), uint32_t(l / o2::constants::lhc::LHCMaxBunches)};
133 }
134
135 bool operator>(const InteractionRecord& other) const
136 {
137 return (orbit == other.orbit) ? (bc > other.bc) : (orbit > other.orbit);
138 }
139
140 bool operator>=(const InteractionRecord& other) const
141 {
142 return !((*this) < other);
143 }

145 bool operator<(const InteractionRecord& other) const
146 {
147 return (orbit == other.orbit) ? (bc < other.bc) : (orbit < other.orbit);
148 }
149
150 bool operator<=(const InteractionRecord& other) const
151 {
152 return !((*this) > other);
153 }
154
155 InteractionRecord operator--()
156 {
157 // prefix decrement operator
158 if (!bc--) {
159 orbit--;
160 bc = o2::constants::lhc::LHCMaxBunches - 1;
161 if (orbit == DummyOrbit) { // wrapped?
162 orbit = 0;
163 bc = 0;
164 }
165 }
166 return InteractionRecord(*this);
167 }
168
169 InteractionRecord operator--(int)
170 {
171 // postfix decrement operator, no check for orbit wrap
172 InteractionRecord tmp(*this);
173 if (!bc--) {
174 orbit--;
175 bc = o2::constants::lhc::LHCMaxBunches - 1;
176 if (orbit == DummyOrbit) { // wrapped?
177 orbit = 0;
178 bc = 0;
179 }
180 }
181 return tmp;
182 }
183
184 InteractionRecord operator++()
185 {
186 // prefix increment operator
187 if ((++bc) == o2::constants::lhc::LHCMaxBunches) {
188 orbit++;
189 bc = 0;
190 if (orbit == 0) { // wrapped?
191 orbit = DummyOrbit;
192 bc = o2::constants::lhc::LHCMaxBunches - 1;
193 }
194 }
195 return InteractionRecord(*this);
196 }
197
198 InteractionRecord operator++(int)
199 {
200 // postfix increment operator
201 InteractionRecord tmp(*this);
202 if ((++bc) == o2::constants::lhc::LHCMaxBunches) {
203 orbit++;
204 bc = 0;
205 if (orbit == 0) { // wrapped?
206 orbit = DummyOrbit;
207 bc = o2::constants::lhc::LHCMaxBunches - 1;
208 }
209 }
210 return tmp;
211 }
212
213 InteractionRecord& operator+=(int64_t dbc)
214 {
215 // bc self-addition operator, avoid wrapping
216 auto l = toLong();
217 if (dbc >= 0) {
218 if (MaxGlobalBCs - dbc < l) {
219 l = MaxGlobalBCs;
220 dbc = 0;
221 }
222 } else {
223 if (l < -dbc) {
224 l = 0;
225 dbc = 0;
226 }
227 }
228 l += dbc;
229 bc = l % o2::constants::lhc::LHCMaxBunches;
230 orbit = l / o2::constants::lhc::LHCMaxBunches;
231 return *this;
232 }
233
234 InteractionRecord& operator-=(int64_t dbc)
235 {
236 // bc self-subtraction operator
237 return operator+=(-dbc);
238 }
239
240 InteractionRecord& operator+=(const InteractionRecord& add)
241 {
242 // InteractionRecord self-addition operator
243 return operator+=(add.toLong());
244 }
245
246 InteractionRecord& operator-=(const InteractionRecord& add)
247 {
248 // InteractionRecord self-subtraction operator
249 return operator-=(add.toLong());
250 }
251

258 }

298 InteractionTimeRecord(const InteractionRecord& ir, double t_in_bc) : InteractionRecord(ir), timeInBCNS(t_in_bc)

303 InteractionTimeRecord(double tNS) : InteractionRecord(tNS)
304 {
305 timeInBCNS = tNS - bc2ns();

307
309 void setFromNS(double tNS)
310 {
311 InteractionRecord::setFromNS(tNS);

314
315 void clear()
316 {
317 InteractionRecord::clear();
318 timeInBCNS = 0.;
319 }
320
321 double getTimeOffsetWrtBC() const

327 double getTimeNS() const
328 {
329 return timeInBCNS + bc2ns();

331

333 {
334 return this->InteractionRecord::operator==(other) && (timeInBCNS == other.timeInBCNS);

338 {

340 }

349 return !((*this) < other);
