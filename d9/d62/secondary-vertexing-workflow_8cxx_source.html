74WorkflowSpec defineDataProcessing(ConfigContext const& configcontext)
75{
76 GID::mask_t allowedSources = GID::getSourcesMask("ITS,TPC,ITS-TPC,TPC-TOF,TPC-TRD,ITS-TPC-TRD,ITS-TPC-TOF,TPC-TRD-TOF,ITS-TPC-TRD-TOF");
77
78 // Update the (declared) parameters if changed from the command line
79 o2::conf::ConfigurableParam::updateFromString(configcontext.options().get<std::string>("configKeyValues"));
80 // write the configuration used for the workflow
81 o2::conf::ConfigurableParam::writeINI("o2secondary-vertexing-workflow_configuration.ini");
82 bool useMC = !configcontext.options().get<bool>("disable-mc");
83 auto disableRootOut = configcontext.options().get<bool>("disable-root-output");
84 auto enableCCDBParams = !configcontext.options().get<bool>("disable-ccdb-params");
85 auto enableCasc = !configcontext.options().get<bool>("disable-cascade-finder");
86 auto enable3body = !configcontext.options().get<bool>("disable-3body-finder");
87 auto enableStrTr = !configcontext.options().get<bool>("disable-strangeness-tracker");
88 auto useGeom = configcontext.options().get<bool>("use-full-geometry");
89 auto sclOpt = o2::tpc::CorrectionMapsLoader::parseGlobalOptions(configcontext.options());
90 GID::mask_t src = allowedSources & GID::getSourcesMask(configcontext.options().get<std::string>("vertexing-sources"));
91 GID::mask_t dummy, srcClus = GID::includesDet(DetID::TOF, src) ? GID::getSourceMask(GID::TOF) : dummy; // eventually, TPC clusters will be needed for refit
92 if (enableStrTr) {
93 srcClus |= GID::getSourceMask(GID::ITS);
94 }
95 if (src[GID::TPC]) {
96 srcClus |= GID::getSourceMask(GID::TPC);
97 }
98 if (sclOpt.requestCTPLumi) {
99 src = src | GID::getSourcesMask("CTP");
100 }
101 WorkflowSpec specs;
102 if (sclOpt.needTPCScalersWorkflow() && !configcontext.options().get<bool>("disable-root-input")) {
103 specs.emplace_back(o2::tpc::getTPCScalerSpec(sclOpt.lumiType == 2, sclOpt.enableMShapeCorrection));
104 }
105 specs.emplace_back(o2::vertexing::getSecondaryVertexingSpec(src, enableCasc, enable3body, enableStrTr, enableCCDBParams, useMC, useGeom, sclOpt));
106
107 // only TOF clusters are needed if TOF is involved, no clusters MC needed
108 WorkflowSpec inputspecs;
109 o2::globaltracking::InputHelper::addInputSpecs(configcontext, inputspecs, srcClus, src, src, useMC, srcClus);
110 o2::globaltracking::InputHelper::addInputSpecsPVertex(configcontext, inputspecs, useMC); // P-vertex is always needed
111
112 if (configcontext.options().get<bool>("combine-source-devices")) {
113 std::vector<DataProcessorSpec> unmerged;
114 specs.push_back(specCombiner("SV-Input-Reader", inputspecs, unmerged));
115 if (unmerged.size() > 0) {
116 // LOG(fatal) << "unexpected DPL merge error";
117 for (auto& s : unmerged) {
118 specs.push_back(s);
119 LOG(info) << " adding unmerged spec " << s.name;
120 }
121 }
122 } else {
123 for (auto& s : inputspecs) {
124 specs.push_back(s);
125 }
126 }
127
128 if (!disableRootOut) {
129 specs.emplace_back(o2::vertexing::getSecondaryVertexWriterSpec());
130 if (enableStrTr) {
131 specs.emplace_back(o2::strangeness_tracking::getStrangenessTrackingWriterSpec(useMC));
132 }
133 }
134
135 // configure dpl timer to inject correct firstTForbit: start from the 1st orbit of TF containing 1st sampled orbit
136 o2::raw::HBFUtilsInitializer hbfIni(configcontext, specs);
137
138 return std::move(specs);
139}
