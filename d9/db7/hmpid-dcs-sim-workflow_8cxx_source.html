37void createHints(const char* type, int i, double mean, double sd, std::vector<o2::dcs::test::HintType>& dphints)
38{
39 double lLimit = mean - 2 * sd;
40 double uLimit = mean + 2 * sd;
41 if (i > 9) {
42 const char* specifier = Form("HMP_TRANPLANT_MEASURE_%i_%s", i, type);
43 dphints.emplace_back(o2::dcs::test::DataPointHint<double>{specifier, lLimit, uLimit});
44 } else {
45 const char* specifier = Form("HMP_TRANPLANT_MEASURE_0%i_%s", i, type);
46 dphints.emplace_back(o2::dcs::test::DataPointHint<double>{specifier, lLimit, uLimit});
47 }
48}
