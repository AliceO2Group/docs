99 static auto updateNumericInfo(DeviceMetricsInfo& metrics, size_t metricIndex, float value, size_t timestamp) {
100 metrics.minDomain[metricIndex] = std::min(metrics.minDomain[metricIndex], timestamp);
101 metrics.maxDomain[metricIndex] = std::max(metrics.maxDomain[metricIndex], timestamp);
102 metrics.max[metricIndex] = std::max(metrics.max[metricIndex], (float)value);
103 metrics.min[metricIndex] = std::min(metrics.min[metricIndex], (float)value);
104 metrics.changed.at(metricIndex) = true;
105 }

107 template <typename T>
108 static auto getNumericMetricCursor(size_t metricIndex)
109 {
110 return [metricIndex](DeviceMetricsInfo& metrics, T value, size_t timestamp) {
111 MetricInfo& metric = metrics.metrics[metricIndex];
112 updateNumericInfo(metrics, metricIndex, (float)value, timestamp);
113
114 auto& store = getMetricsStore<T>(metrics);
115 auto& timestamps = getTimestampsStore<T>(metrics);
116 size_t pos = metric.pos++ % store[metric.storeIdx].size();
117 timestamps[metric.storeIdx][pos] = timestamp;
118 store[metric.storeIdx][pos] = value;
119 metric.filledMetrics++;
120 };
121 }

126 template <typename T>
127 static size_t
128 bookNumericMetric(DeviceMetricsInfo& metrics,
129 char const* name,
130 NewMetricCallback newMetricsCallback = nullptr)
131 {
132 static_assert(std::is_same_v<T, int> || std::is_same_v<T, uint64_t> || std::is_same_v<T, float>, "Unsupported metric type");
133 size_t metricIndex = bookMetricInfo(metrics, name, getMetricType<T>());
134 auto& metricInfo = metrics.metrics[metricIndex];
135 if (newMetricsCallback != nullptr) {
136 newMetricsCallback(name, metricInfo, 0, metricIndex);
137 }
138 return metricIndex;
139 }
