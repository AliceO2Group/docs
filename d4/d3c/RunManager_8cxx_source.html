38 int CTPRunManager::loadRun(const std::string& cfg)
39 {
40  LOG(info) << "Loading run: " << cfg;
41  auto now = std::chrono::system_clock::now();
42  long timeStamp = std::chrono::duration_cast<std::chrono::milliseconds>(now.time_since_epoch()).count();
43  LOG(info) << "Timestamp real time:" << timeStamp;
44  size_t pos = cfg.find(" ");
45  std::string cfgmod = cfg;
46  if (pos == std::string::npos) {
47  LOG(warning) << "Space not found in CTP config";
48  } else {
49  std::string f = cfg.substr(0, pos);
50  if (f.find("run") == std::string::npos) {
51  double_t tt = std::stold(f);
52  timeStamp = (tt * 1000.);
53  LOG(info) << "Timestamp file:" << timeStamp;
54  cfgmod = cfg.substr(pos, cfg.size());
55  LOG(info) << "ctpcfg: using ctp time";
56  }
57  }
58  CTPActiveRun* activerun = new CTPActiveRun;
59  activerun->timeStart = timeStamp;
60  activerun->cfg.loadConfigurationRun3(cfgmod);
61  activerun->cfg.printStream(std::cout);
62  //
63  uint32_t runnumber = activerun->cfg.getRunNumber();
64  activerun->scalers.setRunNumber(runnumber);
65  activerun->scalers.setClassMask(activerun->cfg.getTriggerClassMask());
66  o2::detectors::DetID::mask_t detmask = activerun->cfg.getDetectorMask();
67  activerun->scalers.setDetectorMask(detmask);
68  //
69  mRunsLoaded[runnumber] = activerun;
70  saveRunConfigToCCDB(&activerun->cfg, timeStamp);
71  return 0;

159 int CTPRunManager::processMessage(std::string& topic, const std::string& message)
160 {
161  LOG(info) << "Processing message with topic:" << topic;
162  std::string firstcounters;
163  if (topic.find("clear") != std::string::npos) {
164  mRunsLoaded.clear();
165  LOG(info) << "Loaded runs cleared.";
166  return 0;
167  }
168  if (topic.find("ctpconfig") != std::string::npos) {
169  LOG(info) << "ctpconfig received";
170  loadRun(message);
171  return 0;
172  }
173  if (topic.find("sox") != std::string::npos) {
174  // get config
175  size_t irun = message.find("run");
176  if (irun == std::string::npos) {
177  LOG(warning) << "run keyword not found in SOX";
178  irun = message.size();
179  }
180  LOG(info) << "SOX received, Run keyword position:" << irun;
181  std::string cfg = message.substr(irun, message.size() - irun);
182  startRun(cfg);
183  firstcounters = message.substr(0, irun);
184  }
185  if (topic.find("eox") != std::string::npos) {
186  LOG(info) << "EOX received";
187  mEOX = 1;
188  }
189  //
190  std::vector<std::string> tokens;
191  if (firstcounters.size() > 0) {
192  tokens = o2::utils::Str::tokenize(firstcounters, ' ');
193  } else {
194  tokens = o2::utils::Str::tokenize(message, ' ');
195  }
196  if (tokens.size() != (CTPRunScalers::NCOUNTERS + 1)) {
197  if (tokens.size() == (CTPRunScalers::NCOUNTERSv2 + 1)) {
198  mNew = 0;
199  LOG(warning) << "v2 scaler size";
200  } else {
201  LOG(error) << "Scalers size wrong:" << tokens.size() << " expected:" << CTPRunScalers::NCOUNTERS + 1 << " or " << CTPRunScalers::NCOUNTERSv2 + 1;
202  return 1;
203  }
204  }
205  double timeStamp = std::stold(tokens.at(0));
206  std::time_t tt = timeStamp;
207  LOG(info) << "Processing scalers, all good, time:" << tokens.at(0) << " " << std::asctime(std::localtime(&tt));
208  for (uint32_t i = 1; i < tokens.size(); i++) {
209  mCounters[i - 1] = std::stoull(tokens.at(i));
210  if (i < (NRUNS + 1)) {
211  std::cout << mCounters[i - 1] << " ";
212  }
213  }
214  std::cout << std::endl;
215  LOG(info) << "Counter size:" << tokens.size();
216  //
217  for (uint32_t i = 0; i < NRUNS; i++) {
218  if ((mCounters[i] == 0) && (mActiveRunNumbers[i] == 0)) {
219  // not active
220  } else if ((mCounters[i] != 0) && (mActiveRunNumbers[i] == mCounters[i])) {
221  // active , do scalers
222  LOG(info) << "Run continue:" << mCounters[i];
223  addScalers(i, tt);
224  } else if ((mCounters[i] != 0) && (mActiveRunNumbers[i] == 0)) {
225  LOG(info) << "Run started:" << mCounters[i];
226  auto run = mRunsLoaded.find(mCounters[i]);
227  if (run != mRunsLoaded.end()) {
228  mActiveRunNumbers[i] = mCounters[i];
229  mActiveRuns[i] = run->second;
230  mRunsLoaded.erase(run);
231  addScalers(i, tt);
232  } else {
233  LOG(error) << "Trying to start run which is not loaded:" << mCounters[i];
234  }
235  } else if ((mCounters[i] == 0) && (mActiveRunNumbers[i] != 0)) {
236  if (mEOX != 1) {
237  LOG(error) << "Internal error in processMessage: mEOX != 1 expected 0: mEOX:" << mEOX;
238  }
239  LOG(info) << "Run stopped:" << mActiveRunNumbers[i];
240  addScalers(i, tt);
241  mActiveRunNumbers[i] = 0;
242  mEOX = 0;
243  stopRun(i, tt);
244  }
245  }
246  mEOX = 0;
247  printActiveRuns();
248  return 0;
249 }

262 int CTPRunManager::saveRunScalersToCCDB(int i)
263 {
264  // data base
265  CTPActiveRun* run = mActiveRuns[i];
266  using namespace std::chrono_literals;
267  std::chrono::seconds days3 = 259200s;
268  std::chrono::seconds min10 = 600s;
269  long time3days = std::chrono::duration_cast<std::chrono::milliseconds>(days3).count();
270  long time10min = std::chrono::duration_cast<std::chrono::milliseconds>(min10).count();
271  long tmin = run->timeStart - time10min;
272  long tmax = run->timeStop + time3days;
273  o2::ccdb::CcdbApi api;
274  map<string, string> metadata; // can be empty
275  metadata["runNumber"] = std::to_string(run->cfg.getRunNumber());
276  api.init(mCCDBHost.c_str()); // or http://localhost:8080 for a local installation
277  // store abitrary user object in strongly typed manner
278  api.storeAsTFileAny(&(run->scalers), mCCDBPathCTPScalers, metadata, tmin, tmax);
279  LOG(info) << "CTP scalers saved in ccdb:" << mCCDBHost << " run:" << run->cfg.getRunNumber() << " tmin:" << tmin << " tmax:" << tmax;
280  return 0;
281 }
