73 bool TrackInterpolation::isInputTrackAccepted(const GTrackID& gid, const o2::globaltracking::RecoContainer::GlobalIDSet& gidTable, const o2::dataformats::PrimaryVertex& pv) const
74 {
75  LOGP(debug, "Check if input track {} is accepted", gid.asString());
76  bool hasOuterPoint = gidTable[GTrackID::TRD].isIndexSet() || gidTable[GTrackID::TOF].isIndexSet();
77  if (!hasOuterPoint && !mProcessITSTPConly) {
78  return false; // don't do ITS-only extrapolation through TPC
79  }
80  const auto itsTrk = mRecoCont->getITSTrack(gidTable[GTrackID::ITS]);
81  const auto tpcTrk = mRecoCont->getTPCTrack(gidTable[GTrackID::TPC]);
82 
83  if (gidTable[GTrackID::TRD].isIndexSet()) {
84  // TRD specific cuts
85  const auto& trdTrk = mRecoCont->getITSTPCTRDTrack<o2::trd::TrackTRD>(gidTable[GTrackID::ITSTPCTRD]);
86  if (trdTrk.getNtracklets() < mParams->minTRDNTrklts) {
87  return false;
88  }
89  }
90  // reduced chi2 cut is the same for all track types
91  if (itsTrk.getChi2() / itsTrk.getNumberOfClusters() > mParams->maxITSChi2 || tpcTrk.getChi2() / tpcTrk.getNClusterReferences() > mParams->maxTPCChi2) {
92  return false;
93  }
94  if (!hasOuterPoint) {
95  // ITS-TPC track (does not have outer points in TRD or TOF)
96  if (itsTrk.getNumberOfClusters() < mParams->minITSNClsNoOuterPoint || tpcTrk.getNClusterReferences() < mParams->minTPCNClsNoOuterPoint) {
97  return false;
98  }
99  if (itsTrk.getPt() < mParams->minPtNoOuterPoint) {
100  return false;
101  }
102  } else {
103  if (itsTrk.getNumberOfClusters() < mParams->minITSNCls || tpcTrk.getNClusterReferences() < mParams->minTPCNCls) {
104  return false;
105  }
106  }
107 
108  auto trc = mRecoCont->getTrackParam(gid);
109  o2::dataformats::DCA dca;
110  auto prop = o2::base::Propagator::Instance();
111  if (!prop->propagateToDCA(pv, trc, prop->getNominalBz(), 2., o2::base::PropagatorF::MatCorrType::USEMatCorrLUT, &dca)) {
112  return false;
113  }
114  if (dca.getR2() > mParams->maxDCA * mParams->maxDCA) {
115  return false;
116  }
117  return true;
118 }

142 void TrackInterpolation::prepareInputTrackSample(const o2::globaltracking::RecoContainer& inp)
143 {
144  mRecoCont = &inp;
145  uint32_t nTrackSeeds = 0;
146  uint32_t countSeedCandidates[4] = {0};
147  auto pvvec = mRecoCont->getPrimaryVertices();
148  auto trackIndex = mRecoCont->getPrimaryVertexMatchedTracks(); // Global ID's for associated tracks
149  auto vtxRefs = mRecoCont->getPrimaryVertexMatchedTrackRefs(); // references from vertex to these track IDs
150  int nv = vtxRefs.size() - 1;
151  GTrackID::mask_t allowedSources = GTrackID::getSourcesMask("ITS-TPC,ITS-TPC-TRD,ITS-TPC-TOF,ITS-TPC-TRD-TOF");
152  constexpr std::array<int, 3> SrcFast = {int(GTrackID::ITSTPCTRD), int(GTrackID::ITSTPCTOF), int(GTrackID::ITSTPCTRDTOF)};
153 
154  for (int iv = 0; iv < nv; iv++) {
155  LOGP(debug, "processing PV {} of {}", iv, nv);
156 
157  const auto& vtref = vtxRefs[iv];
158  auto pv = pvvec[iv];
159  if (mParams->minTOFTRDPVContributors > 0) { // we want only PVs constrained by fast detectors
160  int nfound = 0;
161  bool usePV = false;
162  for (uint32_t is = 0; is < SrcFast.size() && !usePV; is++) {
163  int src = SrcFast[is], idMin = vtref.getFirstEntryOfSource(src), idMax = idMin + vtref.getEntriesOfSource(src);
164  for (int i = idMin; i < idMax; i++) {
165  if (trackIndex[i].isPVContributor() && (++nfound == mParams->minTOFTRDPVContributors)) {
166  usePV = true;
167  break;
168  }
169  }
170  }
171  if (!usePV) {
172  continue;
173  }
174  }
175 
176  for (int is = GTrackID::NSources; is >= 0; is--) {
177  if (!allowedSources[is]) {
178  continue;
179  }
180  LOGP(debug, "Checking source {}", is);
181  int idMin = vtref.getFirstEntryOfSource(is), idMax = idMin + vtref.getEntriesOfSource(is);
182  for (int i = idMin; i < idMax; i++) {
183  auto vid = trackIndex[i];
184  auto vidOrig = vid; // in case only ITS-TPC tracks are configured vid might be overwritten. We need to remember it for the PID
185  if (mParams->ignoreNonPVContrib && !vid.isPVContributor()) {
186  continue;
187  }
188  if (vid.isAmbiguous()) {
189  continue;
190  }
191  auto gidTable = mRecoCont->getSingleDetectorRefs(vid);
192  if (!mSourcesConfigured[is]) {
193  auto src = findValidSource(mSourcesConfigured, static_cast<GTrackID::Source>(vid.getSource()));
194  if (src == GTrackID::ITSTPCTRD || src == GTrackID::ITSTPC) {
195  LOGP(debug, "prepareInputTrackSample: Found valid source {}", GTrackID::getSourceName(src));
196  vid = gidTable[src];
197  gidTable = mRecoCont->getSingleDetectorRefs(vid);
198  } else {
199  break; // no valid source for this vertex track source exists
200  }
201  }
202  ++countSeedCandidates[mTrackTypes[vid.getSource()]];
203  LOGP(debug, "Checking vid {}", vid.asString());
204  if (!isInputTrackAccepted(vid, gidTable, pv)) {
205  continue;
206  }
207  mSeeds.push_back(mRecoCont->getITSTrack(gidTable[GTrackID::ITS]).getParamOut());
208  mSeeds.back().setPID(mRecoCont->getTrackParam(vidOrig).getPID(), true);
209  mGIDs.push_back(vid);
210  mGIDtables.push_back(gidTable);
211  mTrackTimes.push_back(pv.getTimeStamp().getTimeStamp());
212  mTrackIndices[mTrackTypes[vid.getSource()]].push_back(nTrackSeeds++);
213  }
214  }
215  }
216 
217  LOGP(info, "Created {} seeds. {} out of {} ITS-TPC-TRD-TOF, {} out of {} ITS-TPC-TRD, {} out of {} ITS-TPC-TOF, {} out of {} ITS-TPC",
218  nTrackSeeds,
219  mTrackIndices[mTrackTypes[GTrackID::ITSTPCTRDTOF]].size(), countSeedCandidates[mTrackTypes[GTrackID::ITSTPCTRDTOF]],
220  mTrackIndices[mTrackTypes[GTrackID::ITSTPCTRD]].size(), countSeedCandidates[mTrackTypes[GTrackID::ITSTPCTRD]],
221  mTrackIndices[mTrackTypes[GTrackID::ITSTPCTOF]].size(), countSeedCandidates[mTrackTypes[GTrackID::ITSTPCTOF]],
222  mTrackIndices[mTrackTypes[GTrackID::ITSTPC]].size(), countSeedCandidates[mTrackTypes[GTrackID::ITSTPC]]);
223 }

234 void TrackInterpolation::process()
235 {
236  // main processing function
237  if (!mInitDone) {
238  LOG(error) << "Initialization not yet done. Aborting...";
239  return;
240  }
241  // set the input containers
242  mTPCTracksClusIdx = mRecoCont->getTPCTracksClusterRefs();
243  mTPCClusterIdxStruct = &mRecoCont->getTPCClusters();
244  if (mDumpTrackPoints) {
245  if (!mITSDict) {
246  LOG(error) << "No ITS dictionary available";
247  return;
248  }
249  mITSTrackClusIdx = mRecoCont->getITSTracksClusterRefs();
250  const auto clusITS = mRecoCont->getITSClusters();
251  const auto patterns = mRecoCont->getITSClustersPatterns();
252  auto pattIt = patterns.begin();
253  mITSClustersArray.clear();
254  mITSClustersArray.reserve(clusITS.size());
255  LOGP(info, "We have {} ITS clusters and the number of patterns is {}", clusITS.size(), patterns.size());
256  o2::its::ioutils::convertCompactClusters(clusITS, pattIt, mITSClustersArray, mITSDict);
257  }
258 
259  // In case we have more input tracks available than are required per TF
260  // we want to sample them. But we still prefer global ITS-TPC-TRD-TOF tracks
261  // over ITS-TPC-TRD tracks and so on. So we have to shuffle the indices
262  // in blocks.
263  std::random_device rd;
264  std::mt19937 g(rd());
265  std::vector<uint32_t> trackIndices; // here we keep the GIDs for all track types in a single vector to use in loop
266  std::shuffle(mTrackIndices[mTrackTypes[GTrackID::ITSTPCTRDTOF]].begin(), mTrackIndices[mTrackTypes[GTrackID::ITSTPCTRDTOF]].end(), g);
267  std::shuffle(mTrackIndices[mTrackTypes[GTrackID::ITSTPCTRD]].begin(), mTrackIndices[mTrackTypes[GTrackID::ITSTPCTRD]].end(), g);
268  std::shuffle(mTrackIndices[mTrackTypes[GTrackID::ITSTPCTOF]].begin(), mTrackIndices[mTrackTypes[GTrackID::ITSTPCTOF]].end(), g);
269  std::shuffle(mTrackIndices[mTrackTypes[GTrackID::ITSTPC]].begin(), mTrackIndices[mTrackTypes[GTrackID::ITSTPC]].end(), g);
270  trackIndices.insert(trackIndices.end(), mTrackIndices[mTrackTypes[GTrackID::ITSTPCTRDTOF]].begin(), mTrackIndices[mTrackTypes[GTrackID::ITSTPCTRDTOF]].end());
271  trackIndices.insert(trackIndices.end(), mTrackIndices[mTrackTypes[GTrackID::ITSTPCTRD]].begin(), mTrackIndices[mTrackTypes[GTrackID::ITSTPCTRD]].end());
272  trackIndices.insert(trackIndices.end(), mTrackIndices[mTrackTypes[GTrackID::ITSTPCTOF]].begin(), mTrackIndices[mTrackTypes[GTrackID::ITSTPCTOF]].end());
273  trackIndices.insert(trackIndices.end(), mTrackIndices[mTrackTypes[GTrackID::ITSTPC]].begin(), mTrackIndices[mTrackTypes[GTrackID::ITSTPC]].end());
274 
275  int nSeeds = mSeeds.size();
276  int maxOutputTracks = (mMaxTracksPerTF >= 0) ? mMaxTracksPerTF + mAddTracksForMapPerTF : nSeeds;
277  mTrackData.reserve(maxOutputTracks);
278  mClRes.reserve(maxOutputTracks * param::NPadRows);
279  bool maxTracksReached = false;
280  for (int iSeed = 0; iSeed < nSeeds; ++iSeed) {
281  if (mMaxTracksPerTF >= 0 && mTrackDataCompact.size() >= mMaxTracksPerTF + mAddTracksForMapPerTF) {
282  LOG(info) << "Maximum number of tracks per TF reached. Skipping the remaining " << nSeeds - iSeed << " tracks.";
283  break;
284  }
285  int seedIndex = trackIndices[iSeed];
286  if (mParams->enableTrackDownsampling && !isTrackSelected(mSeeds[seedIndex])) {
287  continue;
288  }
289  if (!mSingleSourcesConfigured && !mSourcesConfiguredMap[mGIDs[seedIndex].getSource()]) {
290  auto src = findValidSource(mSourcesConfiguredMap, static_cast<GTrackID::Source>(mGIDs[seedIndex].getSource()));
291  if (src == GTrackID::ITSTPCTRD || src == GTrackID::ITSTPC) {
292  LOGP(debug, "process: Found valid source {}", GTrackID::getSourceName(src));
293  mGIDs.push_back(mGIDtables[seedIndex][src]);
294  mGIDtables.push_back(mRecoCont->getSingleDetectorRefs(mGIDs.back()));
295  mTrackTimes.push_back(mTrackTimes[seedIndex]);
296  mSeeds.push_back(mSeeds[seedIndex]);
297  }
298  }
299  if (mMaxTracksPerTF >= 0 && mTrackDataCompact.size() >= mMaxTracksPerTF) {
300  LOG(debug) << "We already have reached mMaxTracksPerTF, but we continue to create seeds until mAddTracksForMapPerTF is also reached";
301  continue;
302  }
303  if (mGIDs[seedIndex].includesDet(DetID::TRD) || mGIDs[seedIndex].includesDet(DetID::TOF)) {
304  interpolateTrack(seedIndex);
305  if (mProcessSeeds) {
306  if (mGIDs[seedIndex].includesDet(DetID::TRD) && mGIDs[seedIndex].includesDet(DetID::TOF)) {
307  mGIDs.push_back(mGIDtables[seedIndex][GTrackID::ITSTPCTRD]);
308  mGIDtables.push_back(mRecoCont->getSingleDetectorRefs(mGIDs.back()));
309  mTrackTimes.push_back(mTrackTimes[seedIndex]);
310  mSeeds.push_back(mSeeds[seedIndex]);
311  }
312  mGIDs.push_back(mGIDtables[seedIndex][GTrackID::ITSTPC]);
313  mGIDtables.push_back(mRecoCont->getSingleDetectorRefs(mGIDs.back()));
314  mTrackTimes.push_back(mTrackTimes[seedIndex]);
315  mSeeds.push_back(mSeeds[seedIndex]);
316  }
317  } else {
318  extrapolateTrack(seedIndex);
319  }
320  }
321  if (mSeeds.size() > nSeeds) {
322  LOGP(info, "Up to {} tracks out of {} additional seeds will be processed", mAddTracksForMapPerTF, mSeeds.size() - nSeeds);
323  }
324  for (int iSeed = nSeeds; iSeed < (int)mSeeds.size(); ++iSeed) {
325  if (!mProcessSeeds && mAddTracksForMapPerTF > 0 && mTrackDataCompact.size() >= mMaxTracksPerTF + mAddTracksForMapPerTF) {
326  LOG(info) << "Maximum number of additional tracks per TF reached. Skipping the remaining " << mSeeds.size() - iSeed << " tracks.";
327  break;
328  }
329  // this loop will only be entered in case mProcessSeeds is set
330  LOGP(debug, "Processing additional track {}", mGIDs[iSeed].asString());
331  if (mGIDs[iSeed].includesDet(DetID::TRD) || mGIDs[iSeed].includesDet(DetID::TOF)) {
332  interpolateTrack(iSeed);
333  } else {
334  extrapolateTrack(iSeed);
335  }
336  }
337  LOG(info) << "Could process " << mTrackData.size() << " tracks successfully";
338 }

340 void TrackInterpolation::interpolateTrack(int iSeed)
341 {
342  LOGP(debug, "Starting track interpolation for GID {}", mGIDs[iSeed].asString());
343  TrackData trackData;
344  std::unique_ptr<TrackDataExtended> trackDataExtended;
345  std::vector<TPCClusterResiduals> clusterResiduals;
346  auto propagator = o2::base::Propagator::Instance();
347  const auto& gidTable = mGIDtables[iSeed];
348  const auto& trkTPC = mRecoCont->getTPCTrack(gidTable[GTrackID::TPC]);
349  const auto& trkITS = mRecoCont->getITSTrack(gidTable[GTrackID::ITS]);
350  if (mDumpTrackPoints) {
351  trackDataExtended = std::make_unique<TrackDataExtended>();
352  (*trackDataExtended).gid = mGIDs[iSeed];
353  (*trackDataExtended).clIdx.setFirstEntry(mClRes.size());
354  (*trackDataExtended).trkITS = trkITS;
355  (*trackDataExtended).trkTPC = trkTPC;
356  auto nCl = trkITS.getNumberOfClusters();
357  auto clEntry = trkITS.getFirstClusterEntry();
358  for (int iCl = nCl - 1; iCl >= 0; iCl--) { // clusters are stored from outer to inner layers
359  const auto& clsITS = mITSClustersArray[mITSTrackClusIdx[clEntry + iCl]];
360  (*trackDataExtended).clsITS.push_back(clsITS);
361  }
362  }
363  trackData.gid = mGIDs[iSeed];
364  trackData.par = mSeeds[iSeed];
365  auto& trkWork = mSeeds[iSeed];
366  // reset the cache array (sufficient to set cluster available to zero)
367  for (auto& elem : mCache) {
368  elem.clAvailable = 0;
369  }
370  trackData.clIdx.setFirstEntry(mClRes.size()); // reference the first cluster residual belonging to this track
371  float clusterTimeBinOffset = mTrackTimes[iSeed] / mTPCTimeBinMUS;
372 
373  // store the TPC cluster positions in the cache
374  for (int iCl = trkTPC.getNClusterReferences(); iCl--;) {
375  uint8_t sector, row;
376  uint32_t clusterIndexInRow;
377  const auto& clTPC = trkTPC.getCluster(mTPCTracksClusIdx, iCl, *mTPCClusterIdxStruct, sector, row);
378  float clTPCX;
379  std::array<float, 2> clTPCYZ;
380  mFastTransform->TransformIdeal(sector, row, clTPC.getPad(), clTPC.getTime(), clTPCX, clTPCYZ[0], clTPCYZ[1], clusterTimeBinOffset);
381  mCache[row].clSec = sector;
382  mCache[row].clAvailable = 1;
383  mCache[row].clY = clTPCYZ[0];
384  mCache[row].clZ = clTPCYZ[1];
385  mCache[row].clAngle = o2::math_utils::sector2Angle(sector);
386  }
387 
388  // extrapolate seed through TPC and store track position at each pad row
389  for (int iRow = 0; iRow < param::NPadRows; ++iRow) {
390  if (!mCache[iRow].clAvailable) {
391  continue;
392  }
393  if (!trkWork.rotate(mCache[iRow].clAngle)) {
394  LOG(debug) << "Failed to rotate track during first extrapolation";
395  return;
396  }
397  if (!propagator->PropagateToXBxByBz(trkWork, param::RowX[iRow], mParams->maxSnp, mParams->maxStep, mMatCorr)) {
398  LOG(debug) << "Failed on first extrapolation";
399  return;
400  }
401  mCache[iRow].y[ExtOut] = trkWork.getY();
402  mCache[iRow].z[ExtOut] = trkWork.getZ();
403  mCache[iRow].sy2[ExtOut] = trkWork.getSigmaY2();
404  mCache[iRow].szy[ExtOut] = trkWork.getSigmaZY();
405  mCache[iRow].sz2[ExtOut] = trkWork.getSigmaZ2();
406  mCache[iRow].snp[ExtOut] = trkWork.getSnp();
407  //printf("Track alpha at row %i: %.2f, Y(%.2f), Z(%.2f)\n", iRow, trkWork.getAlpha(), trkWork.getY(), trkWork.getZ());
408  }
409 
410  // start from outermost cluster with outer refit and back propagation
411  if (gidTable[GTrackID::TOF].isIndexSet()) {
412  LOG(debug) << "TOF point available";
413  const auto& clTOF = mRecoCont->getTOFClusters()[gidTable[GTrackID::TOF]];
414  if (mDumpTrackPoints) {
415  (*trackDataExtended).clsTOF = clTOF;
416  (*trackDataExtended).matchTOF = mRecoCont->getTOFMatch(mGIDs[iSeed]);
417  }
418  const int clTOFSec = clTOF.getCount();
419  const float clTOFAlpha = o2::math_utils::sector2Angle(clTOFSec);
420  if (!trkWork.rotate(clTOFAlpha)) {
421  LOG(debug) << "Failed to rotate into TOF cluster sector frame";
422  return;
423  }
424  float clTOFX = clTOF.getX();
425  std::array<float, 2> clTOFYZ{clTOF.getY(), clTOF.getZ()};
426  std::array<float, 3> clTOFCov{mParams->sigYZ2TOF, 0.f, mParams->sigYZ2TOF}; // assume no correlation between y and z and equal cluster error sigma^2 = (3cm)^2 / 12
427  if (!propagator->PropagateToXBxByBz(trkWork, clTOFX, mParams->maxSnp, mParams->maxStep, mMatCorr)) {
428  LOG(debug) << "Failed final propagation to TOF radius";
429  return;
430  }
431  // TODO: check if reset of covariance matrix is needed here (or, in case TOF point is not available at outermost TRD layer)
432  if (!trkWork.update(clTOFYZ, clTOFCov)) {
433  LOG(debug) << "Failed to update extrapolated ITS track with TOF cluster";
434  //LOGF(info, "trkWork.y=%f, cl.y=%f, trkWork.z=%f, cl.z=%f", trkWork.getY(), clTOFYZ[0], trkWork.getZ(), clTOFYZ[1]);
435  return;
436  }
437  }
438  if (gidTable[GTrackID::TRD].isIndexSet()) {
439  LOG(debug) << "TRD available";
440  const auto& trkTRD = mRecoCont->getITSTPCTRDTrack<o2::trd::TrackTRD>(gidTable[GTrackID::ITSTPCTRD]);
441  if (mDumpTrackPoints) {
442  (*trackDataExtended).trkTRD = trkTRD;
443  }
444  for (int iLayer = o2::trd::constants::NLAYER - 1; iLayer >= 0; --iLayer) {
445  int trkltIdx = trkTRD.getTrackletIndex(iLayer);
446  if (trkltIdx < 0) {
447  // no TRD tracklet in this layer
448  continue;
449  }
450  const auto& trdSP = mRecoCont->getTRDCalibratedTracklets()[trkltIdx];
451  const auto& trdTrklt = mRecoCont->getTRDTracklets()[trkltIdx];
452  if (mDumpTrackPoints) {
453  (*trackDataExtended).trkltTRD.push_back(trdTrklt);
454  (*trackDataExtended).clsTRD.push_back(trdSP);
455  }
456  auto trkltDet = trdTrklt.getDetector();
457  auto trkltSec = trkltDet / (o2::trd::constants::NLAYER * o2::trd::constants::NSTACK);
458  if (trkltSec != o2::math_utils::angle2Sector(trkWork.getAlpha())) {
459  if (!trkWork.rotate(o2::math_utils::sector2Angle(trkltSec))) {
460  LOG(debug) << "Track could not be rotated in TRD tracklet coordinate system in layer " << iLayer;
461  return;
462  }
463  }
464  if (!propagator->PropagateToXBxByBz(trkWork, trdSP.getX(), mParams->maxSnp, mParams->maxStep, mMatCorr)) {
465  LOG(debug) << "Failed propagation to TRD layer " << iLayer;
466  return;
467  }
468 
469  const auto* pad = mGeoTRD->getPadPlane(trkltDet);
470  float tilt = tan(TMath::DegToRad() * pad->getTiltingAngle()); // tilt is signed! and returned in degrees
471  float tiltCorrUp = tilt * (trdSP.getZ() - trkWork.getZ());
472  float zPosCorrUp = trdSP.getZ() + mRecoParam.getZCorrCoeffNRC() * trkWork.getTgl(); // maybe Z can be corrected on avarage already by the tracklet transformer?
473  float padLength = pad->getRowSize(trdTrklt.getPadRow());
474  if (!((trkWork.getSigmaZ2() < (padLength * padLength / 12.f)) && (std::fabs(trdSP.getZ() - trkWork.getZ()) < padLength))) {
475  tiltCorrUp = 0.f;
476  }
477  std::array<float, 2> trkltTRDYZ{trdSP.getY() - tiltCorrUp, zPosCorrUp};
478  std::array<float, 3> trkltTRDCov;
479  mRecoParam.recalcTrkltCov(tilt, trkWork.getSnp(), pad->getRowSize(trdTrklt.getPadRow()), trkltTRDCov);
480  if (!trkWork.update(trkltTRDYZ, trkltTRDCov)) {
481  LOG(debug) << "Failed to update track at TRD layer " << iLayer;
482  return;
483  }
484  }
485  }
486 
487  if (mDumpTrackPoints) {
488  (*trackDataExtended).trkOuter = trkWork;
489  }
490 
491  // go back through the TPC and store updated track positions
492  bool outerParamStored = false;
493  for (int iRow = param::NPadRows; iRow--;) {
494  if (!mCache[iRow].clAvailable) {
495  continue;
496  }
497  if (mProcessSeeds && !outerParamStored) {
498  // for debug purposes we store the track parameters
499  // of the refitted ITS-(TRD)-(TOF) track at the
500  // outermose TPC cluster if we are processing all seeds
501  // i.e. if we in any case also process the ITS-TPC only
502  // part of the same track
503  trackData.par = trkWork;
504  outerParamStored = true;
505  }
506  if (!trkWork.rotate(mCache[iRow].clAngle)) {
507  LOG(debug) << "Failed to rotate track during back propagation";
508  return;
509  }
510  if (!propagator->PropagateToXBxByBz(trkWork, param::RowX[iRow], mParams->maxSnp, mParams->maxStep, mMatCorr)) {
511  LOG(debug) << "Failed on back propagation";
512  //printf("trkX(%.2f), clX(%.2f), clY(%.2f), clZ(%.2f), alphaTOF(%.2f)\n", trkWork.getX(), param::RowX[iRow], clTOFYZ[0], clTOFYZ[1], clTOFAlpha);
513  return;
514  }
515  mCache[iRow].y[ExtIn] = trkWork.getY();
516  mCache[iRow].z[ExtIn] = trkWork.getZ();
517  mCache[iRow].sy2[ExtIn] = trkWork.getSigmaY2();
518  mCache[iRow].szy[ExtIn] = trkWork.getSigmaZY();
519  mCache[iRow].sz2[ExtIn] = trkWork.getSigmaZ2();
520  mCache[iRow].snp[ExtIn] = trkWork.getSnp();
521  }
522 
523  // calculate weighted mean at each pad row (assume for now y and z are uncorrelated) and store residuals to TPC clusters
524  unsigned short deltaRow = 0;
525  for (int iRow = 0; iRow < param::NPadRows; ++iRow) {
526  if (!mCache[iRow].clAvailable) {
527  ++deltaRow;
528  continue;
529  }
530  float wTotY = 1.f / mCache[iRow].sy2[ExtOut] + 1.f / mCache[iRow].sy2[ExtIn];
531  float wTotZ = 1.f / mCache[iRow].sz2[ExtOut] + 1.f / mCache[iRow].sz2[ExtIn];
532  mCache[iRow].y[Int] = (mCache[iRow].y[ExtOut] / mCache[iRow].sy2[ExtOut] + mCache[iRow].y[ExtIn] / mCache[iRow].sy2[ExtIn]) / wTotY;
533  mCache[iRow].z[Int] = (mCache[iRow].z[ExtOut] / mCache[iRow].sz2[ExtOut] + mCache[iRow].z[ExtIn] / mCache[iRow].sz2[ExtIn]) / wTotZ;
534 
535  // simple average w/o weighting for angle
536  mCache[iRow].snp[Int] = (mCache[iRow].snp[ExtOut] + mCache[iRow].snp[ExtIn]) / 2.f;
537 
538  TPCClusterResiduals res;
539  res.setDY(mCache[iRow].clY - mCache[iRow].y[Int]);
540  res.setDZ(mCache[iRow].clZ - mCache[iRow].z[Int]);
541  res.setY(mCache[iRow].y[Int]);
542  res.setZ(mCache[iRow].z[Int]);
543  res.setSnp(mCache[iRow].snp[Int]);
544  res.sec = mCache[iRow].clSec;
545  res.dRow = deltaRow;
546  clusterResiduals.push_back(std::move(res));
547  deltaRow = 1;
548  }
549  trackData.chi2TRD = gidTable[GTrackID::TRD].isIndexSet() ? mRecoCont->getITSTPCTRDTrack<o2::trd::TrackTRD>(gidTable[GTrackID::ITSTPCTRD]).getChi2() : 0;
550  trackData.chi2TPC = trkTPC.getChi2();
551  trackData.chi2ITS = trkITS.getChi2();
552  trackData.nClsTPC = trkTPC.getNClusterReferences();
553  trackData.nClsITS = trkITS.getNumberOfClusters();
554  trackData.nTrkltsTRD = gidTable[GTrackID::TRD].isIndexSet() ? mRecoCont->getITSTPCTRDTrack<o2::trd::TrackTRD>(gidTable[GTrackID::ITSTPCTRD]).getNtracklets() : 0;
555  trackData.clAvailTOF = gidTable[GTrackID::TOF].isIndexSet() ? 1 : 0;
556  trackData.dEdxTPC = trkTPC.getdEdx().dEdxTotTPC;
557 
558  TrackParams params; // for refitted track parameters and flagging rejected clusters
559  if (mParams->skipOutlierFiltering || validateTrack(trackData, params, clusterResiduals)) {
560  // track is good
561  int nClValidated = 0;
562  int iRow = 0;
563  for (unsigned int iCl = 0; iCl < clusterResiduals.size(); ++iCl) {
564  iRow += clusterResiduals[iCl].dRow;
565  if (params.flagRej[iCl]) {
566  // skip masked cluster residual
567  continue;
568  }
569  ++nClValidated;
570  float tgPhi = clusterResiduals[iCl].snp / std::sqrt((1.f - clusterResiduals[iCl].snp) * (1.f + clusterResiduals[iCl].snp));
571  mClRes.emplace_back(clusterResiduals[iCl].dy, clusterResiduals[iCl].dz, tgPhi, clusterResiduals[iCl].y, clusterResiduals[iCl].z, iRow, clusterResiduals[iCl].sec);
572  }
573  trackData.clIdx.setEntries(nClValidated);
574  mTrackData.push_back(std::move(trackData));
575  if (mDumpTrackPoints) {
576  (*trackDataExtended).clIdx.setEntries(nClValidated);
577  mTrackDataExtended.push_back(std::move(*trackDataExtended));
578  }
579  mGIDsSuccess.push_back(mGIDs[iSeed]);
580  mTrackDataCompact.emplace_back(mClRes.size() - nClValidated, nClValidated, mGIDs[iSeed].getSource());
581  }
582  if (mParams->writeUnfiltered) {
583  TrackData trkDataTmp = trackData;
584  trkDataTmp.clIdx.setFirstEntry(mClResUnfiltered.size());
585  trkDataTmp.clIdx.setEntries(clusterResiduals.size());
586  mTrackDataUnfiltered.push_back(std::move(trkDataTmp));
587  mClResUnfiltered.insert(mClResUnfiltered.end(), clusterResiduals.begin(), clusterResiduals.end());
588  }

594  LOGP(debug, "Starting track extrapolation for GID {}", mGIDs[iSeed].asString());
595  const auto& gidTable = mGIDtables[iSeed];

610  for (int iCl = nCl - 1; iCl >= 0; iCl--) { // clusters are stored from outer to inner layers
611  const auto& clsITS = mITSClustersArray[mITSTrackClusIdx[clEntry + iCl]];
612  (*trackDataExtended).clsITS.push_back(clsITS);
613  }
614  }
615  trackData.gid = mGIDs[iSeed];
616  trackData.par = mSeeds[iSeed];
617 
618  auto& trkWork = mSeeds[iSeed];
619  float clusterTimeBinOffset = mTrackTimes[iSeed] / mTPCTimeBinMUS;

622  unsigned short nMeasurements = 0;

634  mSeeds[iSeed].getPt(), trkTPC.getNClusterReferences(), iCl, row, clRowPrev);

662  trackData.nClsTPC = trkTPC.getNClusterReferences();
663  trackData.nClsITS = trkITS.getNumberOfClusters();
664  trackData.clIdx.setEntries(nMeasurements);
665  trackData.dEdxTPC = trkTPC.getdEdx().dEdxTotTPC;
666  if (mDumpTrackPoints) {
667  (*trackDataExtended).trkOuter = trkWork;
668  }
669 
670  TrackParams params; // for refitted track parameters and flagging rejected clusters
671  if (clusterResiduals.size() > constants::MAXGLOBALPADROW) {
672  LOGP(warn, "Extrapolated ITS-TPC track and found more reesiduals than possible ({})", clusterResiduals.size());
673  return;
674  }
675  if (mParams->skipOutlierFiltering || validateTrack(trackData, params, clusterResiduals)) {
676  // track is good
677  int nClValidated = 0;
678  int iRow = 0;
679  for (unsigned int iCl = 0; iCl < clusterResiduals.size(); ++iCl) {
680  iRow += clusterResiduals[iCl].dRow;
681  if (params.flagRej[iCl]) {
682  // skip masked cluster residual
683  continue;

688  }
689  trackData.clIdx.setEntries(nClValidated);
690  mTrackData.push_back(std::move(trackData));
691  mGIDsSuccess.push_back(mGIDs[iSeed]);

707 bool TrackInterpolation::validateTrack(const TrackData& trk, TrackParams& params, const std::vector<TPCClusterResiduals>& clsRes) const
708 {
709  if (clsRes.size() < mParams->minNCl) {
710  // no enough clusters for this track to be considered
711  LOG(debug) << "Skipping track with too few clusters: " << clsRes.size();
712  return false;
713  }
714 
715  bool resHelix = compareToHelix(trk, params, clsRes);
716  if (!resHelix) {
717  LOG(debug) << "Skipping track too far from helix approximation";
718  return false;
719  }
720  if (fabsf(mBz) > 0.01 && fabsf(params.qpt) > mParams->maxQ2Pt) {
721  LOG(debug) << "Skipping track with too high q/pT: " << params.qpt;

723  }
724  if (!outlierFiltering(trk, params, clsRes)) {
725  return false;
726  }
727  return true;

730 bool TrackInterpolation::compareToHelix(const TrackData& trk, TrackParams& params, const std::vector<TPCClusterResiduals>& clsRes) const
731 {
732  std::array<float, param::NPadRows> residHelixY;
733  std::array<float, param::NPadRows> residHelixZ;
734 
735  std::array<float, param::NPadRows> xLab;
736  std::array<float, param::NPadRows> yLab;
737  std::array<float, param::NPadRows> sPath;
738 
739  float curvature = fabsf(trk.par.getQ2Pt() * mBz * o2::constants::physics::LightSpeedCm2S * 1e-14f);
740  int secFirst = clsRes[0].sec;
741  float phiSect = (secFirst + .5f) * o2::constants::math::SectorSpanRad;
742  float snPhi = sin(phiSect);
743  float csPhi = cos(phiSect);
744  sPath[0] = 0.f;
745 
746  int iRow = 0;
747  int nCl = clsRes.size();
748  for (unsigned int iP = 0; iP < nCl; ++iP) {
749  iRow += clsRes[iP].dRow;
750  float yTrk = clsRes[iP].y;
751  // LOGF(info, "iRow(%i), yTrk(%f)", iRow, yTrk);
752  xLab[iP] = param::RowX[iRow];
753  if (clsRes[iP].sec != secFirst) {
754  float phiSectCurrent = (clsRes[iP].sec + .5f) * o2::constants::math::SectorSpanRad;
755  float cs = cos(phiSectCurrent - phiSect);
756  float sn = sin(phiSectCurrent - phiSect);
757  xLab[iP] = param::RowX[iRow] * cs - yTrk * sn;
758  yLab[iP] = yTrk * cs + param::RowX[iRow] * sn;
759  } else {
760  xLab[iP] = param::RowX[iRow];
761  yLab[iP] = yTrk;
762  }
763  // this is needed only later, but we retrieve it already now to save another loop
764  params.zTrk[iP] = clsRes[iP].z;
765  params.xTrk[iP] = param::RowX[iRow];
766  params.dy[iP] = clsRes[iP].dy;
767  params.dz[iP] = clsRes[iP].dz;
768  // done retrieving values for later
769  if (iP > 0) {
770  float dx = xLab[iP] - xLab[iP - 1];
771  float dy = yLab[iP] - yLab[iP - 1];
772  float ds2 = dx * dx + dy * dy;
773  float ds = sqrt(ds2); // circular path (linear approximation)
774  // if the curvature of the track or the (approximated) chord length is too large the more exact formula is used:
775  // chord length = 2r * asin(ds/(2r))
776  // using the first two terms of the tailer expansion for asin(x) ~ x + x^3 / 6
777  if (ds * curvature > 0.05) {
778  ds *= (1.f + ds2 * curvature * curvature / 24.f);
779  }
780  sPath[iP] = sPath[iP - 1] + ds;
781  }
782  }
783  if (fabsf(mBz) < 0.01) {
784  // for B=0 we don't need to try a circular fit...
785  return true;
786  }
787  float xcSec = 0.f;
788  float ycSec = 0.f;
789  float r = 0.f;
790  TrackResiduals::fitCircle(nCl, xLab, yLab, xcSec, ycSec, r, residHelixY);
791  // LOGF(info, "Done with circle fit. nCl(%i), xcSec(%f), ycSec(%f), r(%f).", nCl, xcSec, ycSec, r);
792  /*
793  for (int i=0; i<nCl; ++i) {
794  LOGF(info, "i(%i), xLab(%f), yLab(%f).", i, xLab[i], yLab[i]);
795  }
796  */
797  // determine curvature
798  float phiI = TMath::ATan2(yLab[0], xLab[0]);
799  float phiF = TMath::ATan2(yLab[nCl - 1], xLab[nCl - 1]);
800  if (phiI < 0) {
801  phiI += o2::constants::math::TwoPI;
802  }
803  if (phiF < 0) {
804  phiF += o2::constants::math::TwoPI;
805  }
806  float dPhi = phiF - phiI;
807  float curvSign = -1.f;
808  if (dPhi > 0) {
809  if (dPhi < o2::constants::math::PI) {
810  curvSign = 1.f;
811  }
812  } else if (dPhi < -o2::constants::math::PI) {
813  curvSign = 1.f;
814  }
815  params.qpt = std::copysign(1.f / (r * mBz * o2::constants::physics::LightSpeedCm2S * 1e-14f), curvSign);
816 
817  // calculate circle coordinates in the lab frame
818  float xc = xcSec * csPhi - ycSec * snPhi;
819  float yc = xcSec * snPhi + ycSec * csPhi;
820 
821  std::array<float, 2> pol1Z;
822  TrackResiduals::fitPoly1(nCl, sPath, params.zTrk, pol1Z);
823 
824  params.tgl = pol1Z[0];
825 
826  // max deviations in both directions from helix fit in y and z
827  float hMinY = 1e9f;
828  float hMaxY = -1e9f;
829  float hMinZ = 1e9f;
830  float hMaxZ = -1e9f;
831  // extract residuals in Z and fill track slopes in sector frame
832  int secCurr = secFirst;
833  iRow = 0;
834  for (unsigned int iCl = 0; iCl < nCl; ++iCl) {
835  iRow += clsRes[iCl].dRow;
836  float resZ = params.zTrk[iCl] - (pol1Z[1] + sPath[iCl] * pol1Z[0]);
837  residHelixZ[iCl] = resZ;
838  if (resZ < hMinZ) {
839  hMinZ = resZ;
840  }
841  if (resZ > hMaxZ) {
842  hMaxZ = resZ;
843  }
844  if (residHelixY[iCl] < hMinY) {
845  hMinY = residHelixY[iCl];
846  }
847  if (residHelixY[iCl] > hMaxY) {
848  hMaxY = residHelixY[iCl];
849  }
850  int sec = clsRes[iCl].sec;
851  if (sec != secCurr) {
852  secCurr = sec;
853  phiSect = (.5f + sec) * o2::constants::math::SectorSpanRad;
854  snPhi = sin(phiSect);
855  csPhi = cos(phiSect);
856  xcSec = xc * csPhi + yc * snPhi; // recalculate circle center in the sector frame
857  }
858  float cstalp = (param::RowX[iRow] - xcSec) / r;
859  if (fabsf(cstalp) > 1.f - sFloatEps) {
860  // track cannot reach this pad row
861  cstalp = std::copysign(1.f - sFloatEps, cstalp);
862  }
863  params.tglArr[iCl] = cstalp / sqrt((1 - cstalp) * (1 + cstalp)); // 1 / tan(acos(cstalp)) = cstalp / sqrt(1 - cstalp^2)
864 
865  // In B+ the slope of q- should increase with x. Just look on q * B
866  if (params.qpt * mBz > 0) {
867  params.tglArr[iCl] *= -1.f;
868  }
869  }
870  // LOGF(info, "CompareToHelix: hMaxY(%f), hMinY(%f), hMaxZ(%f), hMinZ(%f). Max deviation allowed: y(%.2f), z(%.2f)", hMaxY, hMinY, hMaxZ, hMinZ, mParams->maxDevHelixY, mParams->maxDevHelixZ);
871  // LOGF(info, "New pt/Q (%f), old pt/Q (%f)", 1./params.qpt, 1./trk.qPt);
872  return fabsf(hMaxY - hMinY) < mParams->maxDevHelixY && fabsf(hMaxZ - hMinZ) < mParams->maxDevHelixZ;
873 }

875 bool TrackInterpolation::outlierFiltering(const TrackData& trk, TrackParams& params, const std::vector<TPCClusterResiduals>& clsRes) const
876 {
877  if (clsRes.size() < mParams->nMALong) {
878  LOG(debug) << "Skipping track with too few clusters for long moving average: " << clsRes.size();
879  return false;
880  }
881  float rmsLong = checkResiduals(trk, params, clsRes);

893 float TrackInterpolation::checkResiduals(const TrackData& trk, TrackParams& params, const std::vector<TPCClusterResiduals>& clsRes) const
894 {
895  float rmsLong = 0.f;
896 
897  int nCl = clsRes.size();
898  int iClFirst = 0;
899  int iClLast = nCl - 1;
900  int secStart = clsRes[0].sec;
901 
902  // arrays with differences / abs(differences) of points to their neighbourhood, initialized to zero
903  std::array<float, param::NPadRows> yDiffLL{};
904  std::array<float, param::NPadRows> zDiffLL{};
905  std::array<float, param::NPadRows> absDevY{};
906  std::array<float, param::NPadRows> absDevZ{};
907 
908  for (unsigned int iCl = 0; iCl < nCl; ++iCl) {
909  if (iCl < iClLast && clsRes[iCl].sec == secStart) {
910  continue;
911  }
912  // sector changed or last cluster reached
913  // now run estimators for all points in the same sector
914  int nClSec = iCl - iClFirst;
915  if (iCl == iClLast) {
916  ++nClSec;
917  }
918  diffToLocLine(nClSec, iClFirst, params.xTrk, params.dy, yDiffLL);
919  diffToLocLine(nClSec, iClFirst, params.xTrk, params.dz, zDiffLL);
920  iClFirst = iCl;
921  secStart = clsRes[iCl].sec;
922  }
923  // store abs deviations
924  int nAccY = 0;
925  int nAccZ = 0;
926  for (int iCl = nCl; iCl--;) {
927  if (fabsf(yDiffLL[iCl]) > param::sEps) {
928  absDevY[nAccY++] = fabsf(yDiffLL[iCl]);
929  }
930  if (fabsf(zDiffLL[iCl]) > param::sEps) {
931  absDevZ[nAccZ++] = fabsf(zDiffLL[iCl]);
932  }
933  }
934  if (nAccY < mParams->minNumberOfAcceptedResiduals || nAccZ < mParams->minNumberOfAcceptedResiduals) {
935  // mask all clusters
936  LOGP(debug, "Accepted {} clusters for dY {} clusters for dZ, but required at least {} for both", nAccY, nAccZ, mParams->minNumberOfAcceptedResiduals);
937  params.flagRej.set();
938  return 0.f;
939  }
940  // estimate rms on 90% of the smallest deviations
941  int nKeepY = static_cast<int>(.9 * nAccY);
942  int nKeepZ = static_cast<int>(.9 * nAccZ);
943  std::nth_element(absDevY.begin(), absDevY.begin() + nKeepY, absDevY.begin() + nAccY);
944  std::nth_element(absDevZ.begin(), absDevZ.begin() + nKeepZ, absDevZ.begin() + nAccZ);
945  float rmsYkeep = 0.f;
946  float rmsZkeep = 0.f;
947  for (int i = nKeepY; i--;) {
948  rmsYkeep += absDevY[i] * absDevY[i];
949  }
950  for (int i = nKeepZ; i--;) {
951  rmsZkeep += absDevZ[i] * absDevZ[i];
952  }
953  rmsYkeep = std::sqrt(rmsYkeep / nKeepY);
954  rmsZkeep = std::sqrt(rmsZkeep / nKeepZ);
955  if (rmsYkeep < param::sEps || rmsZkeep < param::sEps) {
956  LOG(warning) << "Too small RMS: " << rmsYkeep << "(y), " << rmsZkeep << "(z).";
957  params.flagRej.set();
958  return 0.f;
959  }
960  float rmsYkeepI = 1.f / rmsYkeep;
961  float rmsZkeepI = 1.f / rmsZkeep;
962  int nAcc = 0;
963  std::array<float, param::NPadRows> yAcc;
964  std::array<float, param::NPadRows> yDiffLong;
965  for (int iCl = 0; iCl < nCl; ++iCl) {
966  yDiffLL[iCl] *= rmsYkeepI;
967  zDiffLL[iCl] *= rmsZkeepI;
968  if (yDiffLL[iCl] * yDiffLL[iCl] + zDiffLL[iCl] * zDiffLL[iCl] > mParams->maxStdDevMA) {
969  params.flagRej.set(iCl);
970  } else {
971  yAcc[nAcc++] = params.dy[iCl];
972  }
973  }
974  if (nAcc > mParams->nMALong) {
975  diffToMA(nAcc, yAcc, yDiffLong);
976  float average = 0.f;
977  float rms = 0.f;
978  for (int i = 0; i < nAcc; ++i) {
979  average += yDiffLong[i];
980  rms += yDiffLong[i] * yDiffLong[i];
981  }
982  average /= nAcc;
983  rmsLong = rms / nAcc - average * average;
984  rmsLong = (rmsLong > 0) ? std::sqrt(rmsLong) : 0.f;
985  }
986  return rmsLong;
987 }

989 void TrackInterpolation::diffToLocLine(const int np, int idxOffset, const std::array<float, param::NPadRows>& x, const std::array<float, param::NPadRows>& y, std::array<float, param::NPadRows>& diffY) const
990 {
991  // Calculate the difference between the points and the linear extrapolations from the neighbourhood.
992  // Nothing more than multiple 1-d fits at once. Instead of building 4 sums (x, x^2, y, xy), 4 * nPoints sums are calculated at once
993  // compare to TrackResiduals::fitPoly1() method
994 
995  // adding one entry to the vectors saves an additional if statement when calculating the cumulants
996  std::vector<float> sumX1vec(np + 1);
997  std::vector<float> sumX2vec(np + 1);
998  std::vector<float> sumY1vec(np + 1);
999  std::vector<float> sumXYvec(np + 1);
1000  auto sumX1 = &(sumX1vec[1]);
1001  auto sumX2 = &(sumX2vec[1]);
1002  auto sumY1 = &(sumY1vec[1]);
1003  auto sumXY = &(sumXYvec[1]);
1004 
1005  // accumulate sums for all points
1006  for (int iCl = 0; iCl < np; ++iCl) {
1007  int idx = iCl + idxOffset;
1008  sumX1[iCl] = sumX1[iCl - 1] + x[idx];
1009  sumX2[iCl] = sumX2[iCl - 1] + x[idx] * x[idx];
1010  sumY1[iCl] = sumY1[iCl - 1] + y[idx];
1011  sumXY[iCl] = sumXY[iCl - 1] + x[idx] * y[idx];
1012  }
1013 
1014  for (int iCl = 0; iCl < np; ++iCl) {
1015  int iClLeft = iCl - mParams->nMAShort;
1016  int iClRight = iCl + mParams->nMAShort;
1017  if (iClLeft < 0) {
1018  iClLeft = 0;
1019  }
1020  if (iClRight >= np) {
1021  iClRight = np - 1;
1022  }
1023  int nPoints = iClRight - iClLeft;
1024  if (nPoints < mParams->nMAShort) {
1025  continue;
1026  }
1027  float nPointsInv = 1.f / nPoints;
1028  int iClLeftP = iClLeft - 1;
1029  int iClCurrP = iCl - 1;
1030  // extract sum from iClLeft to iClRight from cumulants, excluding iCl from the fit
1031  float sX1 = sumX1[iClRight] - sumX1[iClLeftP] - (sumX1[iCl] - sumX1[iClCurrP]);
1032  float sX2 = sumX2[iClRight] - sumX2[iClLeftP] - (sumX2[iCl] - sumX2[iClCurrP]);
1033  float sY1 = sumY1[iClRight] - sumY1[iClLeftP] - (sumY1[iCl] - sumY1[iClCurrP]);
1034  float sXY = sumXY[iClRight] - sumXY[iClLeftP] - (sumXY[iCl] - sumXY[iClCurrP]);
1035  float det = sX2 - nPointsInv * sX1 * sX1;
1036  if (fabsf(det) < 1e-12f) {
1037  continue;
1038  }
1039  float slope = (sXY - nPointsInv * sX1 * sY1) / det;
1040  float offset = nPointsInv * sY1 - nPointsInv * slope * sX1;
1041  diffY[iCl + idxOffset] = y[iCl + idxOffset] - slope * x[iCl + idxOffset] - offset;
1042  }
1043 }

1045 void TrackInterpolation::diffToMA(const int np, const std::array<float, param::NPadRows>& y, std::array<float, param::NPadRows>& diffMA) const
1046 {

1048  std::vector<float> sumVec(np + 1);
1049  auto sum = &(sumVec[1]);
1050  for (int i = 0; i < np; ++i) {
1051  sum[i] = sum[i - 1] + y[i];
1052  }
1053  for (int i = 0; i < np; ++i) {
1054  diffMA[i] = 0;
1055  int iLeft = i - mParams->nMALong;
1056  int iRight = i + mParams->nMALong;
1057  if (iLeft < 0) {
1058  iLeft = 0;
1059  }
1060  if (iRight >= np) {
1061  iRight = np - 1;

1066  continue;
1067  }

1069  diffMA[i] = y[i] - movingAverage;
1070  }
