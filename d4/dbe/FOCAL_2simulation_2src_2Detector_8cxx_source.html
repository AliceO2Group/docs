87 Bool_t Detector::ProcessHits(FairVolume* v)
88 {
89  int track = fMC->GetStack()->GetCurrentTrackNumber(),
90  directparent = fMC->GetStack()->GetCurrentParentTrackNumber();
91  // Like other calorimeters FOCAL will create a huge amount of shower particles during tracking
92  // Instead, the hits should be assigned to the incoming particle in FOCAL.
93  // Implementation of the incoming particle search taken from implementation in EMCAL.
94  if (track != mCurrentTrack) {
95  LOG(debug4) << "Doing new track " << track << " current (" << mCurrentTrack << "), direct parent (" << directparent << ")";
96  // new current track - check parentage
97  auto hasSuperParent = mSuperParentsIndices.find(directparent);
98  if (hasSuperParent != mSuperParentsIndices.end()) {
99  // same superparent as direct parent
100  mCurrentParentID = hasSuperParent->second;
101  mSuperParentsIndices[track] = hasSuperParent->second;
102  auto superparent = mSuperParents.find(mCurrentParentID);
103  if (superparent != mSuperParents.end()) {
104  mCurrentSuperparent = &(superparent->second);
105  } else {
106  LOG(error) << "Attention: No superparent object found (parent " << mCurrentParentID << ")";
107  mCurrentSuperparent = nullptr;
108  }
109  LOG(debug4) << "Found superparent " << mCurrentParentID;
110  } else {
111  // start of new chain
112  // for new incoming tracks the super parent index is equal to the track ID (for recursion)
113  mSuperParentsIndices[track] = track;
114  mCurrentSuperparent = AddSuperparent(track, fMC->TrackPid(), fMC->Etot());
115  mCurrentParentID = track;
116  }
117  mCurrentTrack = track;
118  }
119 
120  // Processing HCAL hits
121  if (fMC->CurrentMedium() == mMedSensHCal) {
122  LOG(debug) << "We are in sensitive volume " << v->GetName() << ": " << fMC->CurrentVolPath();
123 
124  double eloss = fMC->Edep() * 1e9; // energy in eV (GeV->eV)
125  if (eloss < DBL_EPSILON) {
126  return false; // only process hits which actually deposit some energy in the FOCAL
127  }
128 
129  // In case of new parent track create new track reference
130  auto o2stack = static_cast<o2::data::Stack*>(fMC->GetStack());
131  if (!mCurrentSuperparent->mHasTrackReference) {
132  float x, y, z, px, py, pz, e;
133  fMC->TrackPosition(x, y, z);
134  fMC->TrackMomentum(px, py, pz, e);
135  o2::TrackReference trackref(x, y, z, px, py, pz, fMC->TrackLength(), fMC->TrackTime(), mCurrentParentID, GetDetId());
136  o2stack->addTrackReference(trackref);
137  mCurrentSuperparent->mHasTrackReference = true;
138  }
139 
140  float posX, posY, posZ;
141  fMC->TrackPosition(posX, posY, posZ);
142 
143  auto [indetector, col, row, layer, segment] = mGeometry->getVirtualInfo(posX, posY, posZ);
144 
145  if (!indetector) {
146  // particle outside the detector
147  return true;
148  }
149 
150  auto currenthit = FindHit(mCurrentParentID, col, row, layer);
151  if (!currenthit) {
152  // Condition for new hit:
153  // - Processing different partent track (parent track must be produced outside FOCAL)
154  // - Inside different cell
155  // - First track of the event
156  Double_t time = fMC->TrackTime() * 1e9; // time in ns
157  LOG(debug3) << "Adding new hit for parent " << mCurrentParentID << " and cell Col: " << col << " Row: " << row << " segment: " << segment;
158 
160  AddHit(mCurrentParentID, mCurrentPrimaryID, mCurrentSuperparent->mEnergy, row * col + col, o2::focal::Hit::Subsystem_t::HCAL, math_utils::Point3D<float>(posX, posY, posZ), time, eloss);
161  o2stack->addHit(GetDetId());
162  } else {
163  LOG(debug3) << "Adding energy to the current hit";
164  currenthit->SetEnergyLoss(currenthit->GetEnergyLoss() + eloss);
165  }
166  }
167 
168  return true;
169 }

171 Hit* Detector::AddHit(int trackID, int primary, double initialEnergy, int detID, o2::focal::Hit::Subsystem_t subsystem,
172  const math_utils::Point3D<float>& pos, double time, double eLoss)
173 {
174  LOG(debug3) << "Adding hit for track " << trackID << " with position (" << pos.X() << ", "

176  mHits->emplace_back(primary, trackID, detID, subsystem, initialEnergy, pos, time, eLoss);
177  return &(mHits->back());

180 Hit* Detector::FindHit(int parentID, int col, int row, int layer)
181 {
182 
183  auto HitComparison = [&](const Hit& hit) {
184  auto information = mGeometry->getVirtualInfo(hit.GetX(), hit.GetY(), hit.GetZ());
185  // FIXME Should we compare segments instead of layers ???
186  return hit.GetTrackID() == parentID && col == std::get<1>(information) && row == std::get<2>(information) && layer == std::get<3>(information);
187  };
188 
189  auto result = std::find_if(mHits->begin(), mHits->end(), HitComparison);
190  if (result == mHits->end()) {
191  return nullptr;
192  }
193  return &(*result);
194 }

421 void Detector::ConstructGeometry()
422 {
423 
424  LOG(debug) << "Creating FOCAL geometry\n";
425 
426  CreateMaterials();
427 
429  mGeoCompositions = mGeometry->getFOCALMicroModule(-1);
430  if (!mGeoCompositions.size()) {
431  LOG(error) << "FOCAL compositions not found!!";

433  }
434 
435  float pars[4];
436  pars[0] = (mGeometry->getFOCALSizeX() + 2 * mGeometry->getMiddleTowerOffset()) / 2;

444  }

446  pars[2] += (1.0 + 0.5); // place Aluminium 1cm thick box (0.5 means half) at 2cm behind HCal to simulate SiPM readout material
447  pars[1] += (10.0 + 1.0); // place Aluminium 1cm thick box at 10 cm below FOCAL
448  }
449  pars[3] = 0;
450 
451  LOG(info) << "Creating FOCAL with dimensions X: " << (mGeometry->getFOCALSizeX() + 2 * mGeometry->getMiddleTowerOffset()) << ", Y: "
452  << mGeometry->getFOCALSizeY() << ", Z: " << mGeometry->getFOCALSizeZ() + (mGeometry->getInsertFrontPadLayers() ? 2.0 : 0.0) + (mGeometry->getInsertHCalReadoutMaterial() ? 3.0 : 0.0);
453 
454  TVirtualMC::GetMC()->Gsvolu("FOCAL", "BOX", getMediumID(ID_AIR), pars, 4);
455  mSensitiveHCAL.push_back("FOCAL");
456 
457  if (mGeometry->getUseHCALSandwich()) {
458  CreateHCALSandwich();
459  } else {
460  CreateHCALSpaghetti();
461  }
462  const float z0 = 1312.5; // center of caveRB24 mother volume
463  TVirtualMC::GetMC()->Gspos("FOCAL", 1, "caveRB24", 0, 0, mGeometry->getFOCALZ0() - (mGeometry->getInsertFrontPadLayers() ? 2.0 : 0.0) + (mGeometry->getInsertHCalReadoutMaterial() ? 1.5 : 0.0) - z0, 0, "ONLY");
464 }

466 void Detector::CreateHCALSpaghetti()
467 {
468  TGeoVolumeAssembly* volHCAL = new TGeoVolumeAssembly("HCAL");
469 
470  TGeoVolumeAssembly* HcalTube = gGeoManager->MakeVolumeAssembly("ScintCuTubes");
471 
472  TGeoVolume* volCuTube;
473  TGeoVolume* volSciFi;
474 
475  float RScint = 0.;
476  float Rin = 0.;
477  float Rout = 0.;
478  float Length = 0.;
479 
480  for (auto& icomp : mGeoCompositions) {
481  Length = icomp->sizeZ() / 2;
482 
483  if (icomp->material() == "Pb") {
484  Rout = icomp->sizeX() / 2;
485  TGeoMedium* medium = gGeoManager->GetMedium(getMediumID(ID_PB));
486  volCuTube = gGeoManager->MakeTube("Tube", medium, Rin, Rout, Length); // The Scintillator must be first in order in vector for Rin to be set
487  volCuTube->SetLineWidth(2);
488  volCuTube->SetLineColor(kRed);
489  mSensitiveHCAL.push_back(volCuTube->GetName());
490  HcalTube->AddNode(volCuTube, 1, nullptr);
491  }
492  if (icomp->material() == "Scint") {
493  RScint = icomp->sizeX() / 2;
494  Rin = RScint + 0.005;
495  TGeoMedium* medium = gGeoManager->GetMedium(getMediumID(ID_SC));
496  volSciFi = gGeoManager->MakeTube("ScintFiber", medium, 0., RScint, Length);
497  volSciFi->SetLineWidth(2);
498  volSciFi->SetLineColor(kBlue);
499  mSensitiveHCAL.push_back(volSciFi->GetName());
500  HcalTube->AddNode(volSciFi, 1, nullptr);
501  }
502  if (icomp->material() == "CuHCAL") {
503  Rout = icomp->sizeX() / 2;
504  TGeoMedium* medium = gGeoManager->GetMedium(getMediumID(ID_COPPER));
505  volCuTube = gGeoManager->MakeTube("Tube", medium, Rin, Rout, Length); // The Scintillator must be first in order in vector for Rin to be set
506  volCuTube->SetLineWidth(2);
507  volCuTube->SetLineColor(kRed);
508  mSensitiveHCAL.push_back(volCuTube->GetName());
509  HcalTube->AddNode(volCuTube, 1, nullptr);
510  }
511  }
512 
513  double TowerSize = mGeometry->getHCALTowerSize();
514  double CuBoxThickness = 0.3; // Thickness of the Cu box carrying capillary tubes
515 
516  TGeoBBox* ODBox = new TGeoBBox("TowerOD", TowerSize / 2, TowerSize / 2, Length);
517  TGeoBBox* IDBox = new TGeoBBox("TowerID", (TowerSize - CuBoxThickness) / 2, (TowerSize - CuBoxThickness) / 2, Length + 0.01);
518  TGeoCompositeShape* TowerHCAL = new TGeoCompositeShape("TowerHCAL", "TowerOD - TowerID");
519  TGeoVolume* volTower = new TGeoVolume("volTower", TowerHCAL, gGeoManager->GetMedium(getMediumID(ID_COPPER)));
520  volTower->SetLineWidth(2);
521  volTower->SetLineColor(42);
522  mSensitiveHCAL.push_back(volTower->GetName());
523 
524  TGeoVolumeAssembly* volTowerHCAL = new TGeoVolumeAssembly("volTowerHCAL");
525  volTowerHCAL->AddNode(volTower, 1, nullptr);
526 
527  int Rows = 0;
528  float RowPos = 0.;
529  int Columns = 0;
530  int NumTubes = 1;
531 
532  // Packing circles in Hexagonal shape
533  while (RowPos + CuBoxThickness / 2 + Rout + 2 * Rout < TowerSize) {
534  Columns = 0;
535  float ColumnPos = (Rows % 2 == 0) ? 0. : Rout;
536  while (ColumnPos + CuBoxThickness / 2 + Rout + 2 * Rout < TowerSize) {
537
