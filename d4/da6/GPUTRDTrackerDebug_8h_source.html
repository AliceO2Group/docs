45 int32_t GetSector(float alpha)
46 {
47 if (alpha < 0) {
48 alpha += 2.f * M_PI;
49 }
50 return (int32_t)(alpha * 18 / (2.f * M_PI));
51 }

136 // general information
137 void SetGeneralInfo(int32_t iEv, int32_t nTPCtracks, int32_t iTrk, float pt)
138 {

145 // track parameters
146 void SetTrackParameter(const T& trk, int32_t ly)
147 {
148 fTrackX(ly) = trk.getX();
149 fTrackY(ly) = trk.getY();
150 fTrackZ(ly) = trk.getZ();
151 fTrackPhi(ly) = trk.getSnp();
152 fTrackLambda(ly) = trk.getTgl();
153 fTrackPt(ly) = trk.getPt();
154 fTrackQPt(ly) = trk.getQ2Pt();
155 fTrackSector(ly) = GetSector(trk.getAlpha());
156 fTrackYerr(ly) = trk.getSigmaY2();
157 fTrackZerr(ly) = trk.getSigmaZ2();
158 }
159 void SetTrackParameterNoUp(const T& trk, int32_t ly)
160 {
161 fTrackNoUpX(ly) = trk.getX();
162 fTrackNoUpY(ly) = trk.getY();
163 fTrackNoUpZ(ly) = trk.getZ();
164 fTrackNoUpPhi(ly) = trk.getSnp();
165 fTrackNoUpLambda(ly) = trk.getTgl();
166 fTrackNoUpPt(ly) = trk.getPt();
167 fTrackNoUpSector(ly) = GetSector(trk.getAlpha());
168 fTrackNoUpYerr(ly) = trk.getSigmaY2();
169 fTrackNoUpZerr(ly) = trk.getSigmaZ2();
170 }
171 void SetTrack(const T& trk)
172 {
173 fChi2 = trk.getChi2();
174 fNlayers = trk.getNlayers();
175 fNtrklts = trk.getNtracklets();
176 for (int32_t iLy = 0; iLy < 6; iLy++) {
177 if (trk.getIsFindable(iLy)) {
178 fFindable(iLy) = 1;
179 }
180 }
181 }

187 fTrackletY(ly) = fY;
188 fTrackletZ(ly) = fZ;
189 }
190 void SetCorrectedTrackletPosition(const My_Float* fYZ, int32_t ly)
191 {
192 fTrackletYcorr(ly) = fYZ[0];
193 fTrackletZcorr(ly) = fYZ[1];
194 }
195 void SetTrackletCovariance(const My_Float* fCov, int32_t ly)
196 {
197 fTrackletY2err(ly) = fCov[0];
198 fTrackletYZerr(ly) = fCov[1];
199 fTrackletZ2err(ly) = fCov[2];
200 }
201 void SetTrackletProperties(const float dy, const int32_t det, int32_t ly)
202 {
203 fTrackletDy(ly) = dy;
204 fTrackletDet(ly) = det;
205 }
206
207 // update information
208 void SetChi2Update(float chi2, int32_t ly) { fChi2Update(ly) = chi2; }
209
210 // other infos
211 void SetRoad(float roadY, float roadZ, int32_t ly)
212 {
213 fRoadY(ly) = roadY;
214 fRoadZ(ly) = roadZ;
215 }

217 void Output()
218 {
219 (*fStreamer) << "tracksFinal"
220 << "event=" << fEv << // event number
221 "nTPCtracks=" << fNTPCtracks << // total number of TPC tracks for this event
222 "iTrack=" << fTrk << // track index in event
223 "trackPtTPC=" << fPtTPC << // track pT before any propagation
224 "trackX.=" << &fTrackX << // x-pos of track (layerwise)
225 "trackY.=" << &fTrackY << // y-pos of track (layerwise)
226 "trackZ.=" << &fTrackZ << // z-pos of track (layerwise)
227 "trackPhi.=" << &fTrackPhi << // phi angle of track (track.fP[2])
228 "trackLambda.=" << &fTrackLambda << // lambda angle of track (track.fP[3])
229 "trackQPt.=" << &fTrackQPt << // track q/pT (track.fP[4])
230 "trackPt.=" << &fTrackPt << // track pT (layerwise)
231 "trackYerr.=" << &fTrackYerr << // sigma_y^2 for track
232 "trackZerr.=" << &fTrackZerr << // sigma_z^2 for track
233 "trackSec.=" << &fTrackSector << // TRD sector of track
234 "trackNoUpX.=" << &fTrackNoUpX << // x-pos of track w/o updates (layerwise)
235 "trackNoUpY.=" << &fTrackNoUpY << // y-pos of track w/o updates (layerwise)
236 "trackNoUpZ.=" << &fTrackNoUpZ << // z-pos of track w/o updates (layerwise)
237 "trackNoUpPhi.=" << &fTrackNoUpPhi << // phi angle of track w/o updates (track.fP[2])
238 "trackNoUpLambda.=" << &fTrackNoUpLambda << // lambda angle of track w/o updates (track.fP[3])
239 "trackNoUpPt.=" << &fTrackNoUpPt << // track pT w/o updates (layerwise)
240 "trackNoUpYerr.=" << &fTrackNoUpYerr << // sigma_y^2 for track w/o updates
241 "trackNoUpZerr.=" << &fTrackNoUpZerr << // sigma_z^2 for track w/o updates
242 "trackNoUpSec.=" << &fTrackNoUpSector << // TRD sector of track w/o updates
243 "trackletX.=" << &fTrackletX << // x position of tracklet used for update (sector coords)
244 "trackletY.=" << &fTrackletYcorr << // y position of tracklet used for update (sector coords, tilt corrected position)
245 "trackletZ.=" << &fTrackletZcorr << // z position of tracklet used for update (sector coords, tilt corrected position)
246 "trackletYRaw.=" << &fTrackletY << // y position of tracklet used for update (sector coords)
247 "trackletZRaw.=" << &fTrackletZ << // z position of tracklet used for update (sector coords)
248 "trackletYerr.=" << &fTrackletY2err << // sigma_y^2 for tracklet
249 "trackletYZerr.=" << &fTrackletYZerr << // sigma_yz for tracklet
250 "trackletZerr.=" << &fTrackletZ2err << // sigma_z^2 for tracklet
251 "trackletDy.=" << &fTrackletDy << // deflection for tracklet
252 "trackletDet.=" << &fTrackletDet << // TRD chamber of tracklet
253 "chi2Update.=" << &fChi2Update << // chi2 for update
254 "chi2Total=" << fChi2 << // total chi2 for track
255 "nLayers=" << fNlayers << // number of layers in which track was findable
256 "nTracklets=" << fNtrklts << // number of attached tracklets
257 "roadY.=" << &fRoadY << // search road width in Y
258 "roadZ.=" << &fRoadZ << // search road width in Z
259 "findable.=" << &fFindable << // whether or not track was in active TRD volume (layerwise)
260 "\n";
261 }
