278template <typename T>
279GPUdi() T IrregularSpline1D::getSpline(const IrregularSpline1D::Knot& knot1, T f0, T f1, T f2, T f3, float u)
280{
283
284 f0 -= f1;
285 f2 -= f1;
286 f3 -= f1;
287
288 T x = T((u - knot1.u) * knot1.scale); // scaled u
289 T z1 = T(f0 * knot1.scaleL0 + f2 * knot1.scaleL2); // scaled u derivative at the knot 1
290 T z2 = T(f2 * knot1.scaleR2 + f3 * knot1.scaleR3); // scaled u derivative at the knot 2
291
292 T x2 = x * x;
293
294 // f(x) = ax^3 + bx^2 + cx + d
295 //
296 // f(0) = f1
297 // f(1) = f2
298 // f'(0) = z1
299 // f'(1) = z2
300 //
301
302 // T d = f1;
303 // T c = z1;
304 T a = -f2 - f2 + z1 + z2;
305 T b = f2 - z1 - a;
306 return a * x * x2 + b * x2 + z1 * x + f1;
307}

309template <typename T>
310GPUdi() T IrregularSpline1D::getSpline(const T correctedData[], float u) const
311{
313 int32_t iknot = getKnotIndex(u);
314 const IrregularSpline1D::Knot& knot = getKnot(iknot);
315 const T* f = correctedData + iknot - 1;
316 return getSpline(knot, f[0], f[1], f[2], f[3], u);
317}

332GPUdi() void IrregularSpline1D::getEdgeCorrectionCoefficients(double u0, double u1, double u2, double u3, double& c0, double& c1, double& c2, double& c3)
333{
337
338 double du = u2 - u1;
339 double x0 = (u0 - u1) / du;
340 // double x1 = ( u1 - u1 )/du = 0
341 // double x2 = ( u2 - u1 )/du = 1
342 double x3 = (u3 - u1) / du;
343
344 double cL0 = -1. / (x0 * (x0 - 1.));
345 double cL2 = x0 / (x0 - 1.);
346 double cR2 = (x3 - 2.) / (x3 - 1.);
347 double cR3 = 1. / (x3 * (x3 - 1.));
348
349 // Fi = fi - f1
350
351 // z1 = F0corr*cL0 + F2*cL2; // scaled u derivative at the knot 1
352 // z2 = F2*cR2 + F3*cR3; // scaled u derivative at the knot 2
353
354 // a = -2F2 + z1 + z2
355 // b = 3F2 - 2*z1 - z2 ;
356 // c = z1
357 // d = 0
358
359 // F(x) = ax^3 + bx^2 + cx + d
360 // F(x) = (-2*x^3+3*x^2)*F2 + (x^3-2*x^2+x)*z1 + (x^3-x^2)*z2
361 // F(x) = (-2*x^3+3*x^2)*F2 + (x^3-2*x^2+x)*z1 + (x^3-x^2)*(cR2*F2 + cR3*F3)
362 // F(x) = (-2*x^3+3*x^2+(x^3-x^2)*cR2)*F2 + (x^3-2*x^2+x)*z1 + (x^3-x^2)*cR3*F3
363 // F(x) = x^2(-2*x+3+(x-1)*cR2)*F2 + x*(x-1)^2*z1 + x^2(x-1)*cR3*F3
364
365 // z1*x*(x-1)^2 = F(x) - x^2(-2*x+3+(x-1)*cR2)*F2 - x^2(x-1)*cR3*F3
366 // z1*x0*(x0-1)^2 = F0 - x0^2(-2*x0+3+(x0-1)*cR2)*F2 - x0^2(x0-1)*cR3*F3
367
368 double x01 = x0 - 1.;
369
370 // coeff. for z1 at F0, F1, F2, F3:
371
372 c0 = 1. / (x0 * x01 * x01);
373 c1 = 0;
374 c2 = -(-2 * x0 + 3. + x01 * cR2) * x0 / (x01 * x01);
375 c3 = -x0 * cR3 / x01;
376
377 // F0corr = (z1 - F2*cL2)/cL0;
378
379 c0 = c0 / cL0;
380 c1 = c1 / cL0;
381 c2 = (c2 - cL2) / cL0;
382 c3 = c3 / cL0;
383
384 // F0corr = c0*F0 + c1*F1 + C2*F2 + c3*F3;
385 // f0corr-f1 = c0*(f0-f1) + c1*(f1-f1) + c2*(f2-f1) + c3*(f3-f1);
386 // f0corr-f1 = c0*(f0-f1) + c2*(f2-f1) + c3*(f3-f1);
387 // f0corr = c0*f0 + c2*f2 + c3*f3 + (1-c0-c2-c3)*f1;
388
389 c1 = 1. - c0 - c2 - c3;
390}

392template <typename T>
393GPUdi() void IrregularSpline1D::correctEdges(T* data) const
394{
395 const IrregularSpline1D::Knot* s = getKnots();
396 double c0, c1, c2, c3;
397 getEdgeCorrectionCoefficients(s[0].u, s[1].u, s[2].u, s[3].u, c0, c1, c2, c3);
398 data[0] = c0 * data[0] + c1 * data[1] + c2 * data[2] + c3 * data[3];
399 int32_t i = mNumberOfKnots - 1;
400 getEdgeCorrectionCoefficients(s[i - 0].u, s[i - 1].u, s[i - 2].u, s[i - 3].u, c0, c1, c2, c3);
401 data[i] = c0 * data[i - 0] + c1 * data[i - 1] + c2 * data[i - 2] + c3 * data[i - 3];
402}
