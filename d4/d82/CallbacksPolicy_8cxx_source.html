30 static bool checkPrescale(const TimingInfo& info, int prescale, bool startProcessing)
31 {
32  if (prescale <= 1) {
33  static size_t counter = 0;
34  static size_t downscaleFactor = 1;
35  if (startProcessing) {
36  counter++;
37  }
38  if (counter <= 100000) {
39  return true;
40  }
41  if (counter > 100000 * downscaleFactor) {
42  downscaleFactor *= 10;
43  LOG(info) << "Processed " << counter << " timeslices / timers, increasing reporting downscale factor to " << downscaleFactor;
44  }
45  return counter % downscaleFactor == 0;
46  }
47  return info.isTimer() || !(info.timeslice % prescale);
48 }

50 CallbacksPolicy epnProcessReporting()
51 {
52  int prescale = 1;
53  bool forceReport = false;
54  if (getenv("DPL_REPORT_PROCESSING") != nullptr && (prescale = std::abs(atoi(getenv("DPL_REPORT_PROCESSING"))))) {
55  forceReport = true;
56  }
57  if (!prescale) {
58  prescale = 1;
59  }
60  return {
61  .matcher = [forceReport](DeviceSpec const&, ConfigContext const& context) -> bool {
62  static bool report = DefaultsHelpers::deploymentMode() == DeploymentMode::OnlineDDS || forceReport;
63  return report;
64  },
65  .policy = [prescale](CallbackService& callbacks, InitContext& context) -> void {
66  callbacks.set<CallbackService::Id::PreProcessing>([prescale](ServiceRegistryRef registry, int op) {
67  auto& info = registry.get<TimingInfo>();
68  if ((int)info.firstTForbit != -1 && checkPrescale(info, prescale, true)) {
69  char const* what = info.isTimer() ? "timer" : "timeslice";
70  LOGP(info, "Processing {}:{}, tfCounter:{}, firstTForbit:{}, runNumber:{}, creation:{}, action:{}",
71  what, info.timeslice, info.tfCounter, info.firstTForbit, info.runNumber, info.creation, op);
72  }
73  info.lapse = uv_hrtime();
74  });
75  callbacks.set<CallbackService::Id::PostProcessing>([prescale](ServiceRegistryRef registry, int op) {
76  auto& info = registry.get<TimingInfo>();
77  if ((int)info.firstTForbit != -1 && checkPrescale(info, prescale, false)) {
78  char const* what = info.isTimer() ? "timer" : "timeslice";
79  LOGP(info, "Done processing {}:{}, tfCounter:{}, firstTForbit:{}, runNumber:{}, creation:{}, action:{}, wall:{}",
80  what, info.timeslice, info.tfCounter, info.firstTForbit, info.runNumber, info.creation, op, uv_hrtime() - info.lapse);
81  }
82  });
83  }};
84 }
