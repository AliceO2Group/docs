353 GPUdi() void TPCFastTransform::convPadTimeToUV(int32_t slice, int32_t row, float pad, float time, float& u, float& v, float vertexTime) const
354 {
355  bool sideC = (slice >= getGeometry().getNumberOfSlicesA());
356 
357  const TPCFastTransformGeo::RowInfo& rowInfo = getGeometry().getRowInfo(row);
358  const TPCFastTransformGeo::SliceInfo& sliceInfo = getGeometry().getSliceInfo(slice);
359 
360  float x = rowInfo.x;
361  u = (pad - 0.5f * rowInfo.maxPad) * rowInfo.padWidth;
362 
363  float y = sideC ? -u : u; // pads are mirrorred on C-side
364  float yLab = y * sliceInfo.cosAlpha + x * sliceInfo.sinAlpha;
365 
366  v = (time - mT0 - vertexTime) * (mVdrift + mVdriftCorrY * yLab) + mLdriftCorr; // drift length cm
367 }

370 {
371  v = (time - mT0 - maxTimeBin) * mVdrift + mLdriftCorr; // drift length cm
372  if (slice < getGeometry().getNumberOfSlicesA()) {
373  v += getGeometry().getTPCzLengthA();
374  } else {
375  v += getGeometry().getTPCzLengthC();
376  }

404 GPUdi() void TPCFastTransform::convUVtoPadTime(int32_t slice, int32_t row, float u, float v, float& pad, float& time, float vertexTime) const
405 {
406  bool sideC = (slice >= getGeometry().getNumberOfSlicesA());
407 
408  const TPCFastTransformGeo::RowInfo& rowInfo = getGeometry().getRowInfo(row);
409  const TPCFastTransformGeo::SliceInfo& sliceInfo = getGeometry().getSliceInfo(slice);
410 
411  pad = u / rowInfo.padWidth + 0.5f * rowInfo.maxPad;
412 
413  float x = rowInfo.x;
414  float y = sideC ? -u : u; // pads are mirrorred on C-side
415  float yLab = y * sliceInfo.cosAlpha + x * sliceInfo.sinAlpha;
416  time = mT0 + vertexTime + (v - mLdriftCorr) / (mVdrift + mVdriftCorrY * yLab);
417 }

420 {

422  time = mT0 + vertexTime + (v - mLdriftCorr) / (mVdrift + mVdriftCorrY * yLab);
423 }

425 GPUdi() void TPCFastTransform::convUVtoPadTimeInTimeFrame(int32_t slice, int32_t row, float u, float v, float& pad, float& time, float maxTimeBin) const
426 {
427  if (slice < getGeometry().getNumberOfSlicesA()) {
428  v -= getGeometry().getTPCzLengthA();
429  } else {
430  v -= getGeometry().getTPCzLengthC();
431  }
432  const TPCFastTransformGeo::RowInfo& rowInfo = getGeometry().getRowInfo(row);
433  pad = u / rowInfo.padWidth + 0.5f * rowInfo.maxPad;
434  time = mT0 + maxTimeBin + (v - mLdriftCorr) / mVdrift;
435 }

437 GPUdi() void TPCFastTransform::getTOFcorrection(int32_t slice, int32_t /*row*/, float x, float y, float z, float& dz) const
438 {
439  // calculate time of flight correction for z coordinate
440 
441  bool sideC = (slice >= getGeometry().getNumberOfSlicesA());
442  float distZ = z - mPrimVtxZ;
443  float dv = -GPUCommonMath::Sqrt(x * x + y * y + distZ * distZ) * mTOFcorr;
444  dz = sideC ? dv : -dv;
445 }

447 GPUdi() void TPCFastTransform::TransformInternal(int32_t slice, int32_t row, float& u, float& v, float& x, const TPCFastTransform* ref, const TPCFastTransform* ref2, float scale, float scale2, int32_t scaleMode) const
448 {
449  GPUCA_RTC_SPECIAL_CODE(ref2 = nullptr; scale2 = 0.f;);
450  if (mApplyCorrection) {
451  float dx = 0.f, du = 0.f, dv = 0.f;
452  if ((scale >= 0.f) || (scaleMode == 1) || (scaleMode == 2)) {
453 #ifndef GPUCA_GPUCODE
454  if (mCorrectionSlow) {
455  float ly, lz;
456  getGeometry().convUVtoLocal(slice, u, v, ly, lz);
457  float gx, gy, gz;
458  getGeometry().convLocalToGlobal(slice, x, ly, lz, gx, gy, gz);
459 
460  float gdxC, gdyC, gdzC;
461  mCorrectionSlow->getCorrections(gx, gy, gz, slice, gdxC, gdyC, gdzC);
462  getGeometry().convGlobalToLocal(slice, gdxC, gdyC, gdzC, dx, du, dv);
463 
464  if (slice >= 18) {
465  du = -du; // mirror for c-Side
466  } else {
467  dv = -dv; // mirror z for A-Side
468  }
469  } else
470 #endif // GPUCA_GPUCODE
471  {
472  mCorrection.getCorrection(slice, row, u, v, dx, du, dv);
473  if (ref) {
474  if ((scale > 0.f) && (scaleMode == 0)) { // scaling was requested
475  float dxRef, duRef, dvRef;
476  ref->mCorrection.getCorrection(slice, row, u, v, dxRef, duRef, dvRef);
477  dx = (dx - dxRef) * scale + dxRef;
478  du = (du - duRef) * scale + duRef;
479  dv = (dv - dvRef) * scale + dvRef;
480  } else if ((scale != 0.f) && ((scaleMode == 1) || (scaleMode == 2))) {
481  float dxRef, duRef, dvRef;
482  ref->mCorrection.getCorrection(slice, row, u, v, dxRef, duRef, dvRef);
483  dx = dxRef * scale + dx;
484  du = duRef * scale + du;
485  dv = dvRef * scale + dv;
486  }
487  }
488  if (ref2 && (scale2 != 0)) {
489  float dxRef, duRef, dvRef;
490  ref2->mCorrection.getCorrection(slice, row, u, v, dxRef, duRef, dvRef);
491  dx = dxRef * scale2 + dx;
492  du = duRef * scale2 + du;
493  dv = dvRef * scale2 + dv;
494  }
495  }
496  }
497  GPUCA_DEBUG_STREAMER_CHECK(if (o2::utils::DebugStreamer::checkStream(o2::utils::StreamFlags::streamFastTransform)) {
498  float ly, lz;
499  getGeometry().convUVtoLocal(slice, u, v, ly, lz);
500

503 
504  float lyT, lzT;
505  float uCorr = u + du;
506  float vCorr = v + dv;
507  float lxT = x + dx;
508  getGeometry().convUVtoLocal(slice, uCorr, vCorr, lyT, lzT);
509 
510  float invYZtoXScaled;
511  InverseTransformYZtoX(slice, row, lyT, lzT, invYZtoXScaled, ref, ref2, scale, scale2, scaleMode);
512 
513  float invYZtoX;
514  InverseTransformYZtoX(slice, row, lyT, lzT, invYZtoX);
515 
516  float YZtoNominalY;

518  InverseTransformYZtoNominalYZ(slice, row, lyT, lzT, YZtoNominalY, YZtoNominalZ);

523 
524  float dxRef, duRef, dvRef;
525  if (ref) {
526  ref->mCorrection.getCorrection(slice, row, u, v, dxRef, duRef, dvRef);
527  }
528
