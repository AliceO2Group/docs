72 void DigitReader::run(ProcessingContext& pc)
73 {
74  auto ent = mTree->GetReadEntry() + 1;
75  assert(ent < mTree->GetEntries()); // this should not happen
76 
77  mTree->GetEntry(ent);
78  LOG(info) << "DigitReader pushes " << mDigits.size() << " digits at entry " << ent;
79  pc.outputs().snapshot(Output{"CTP", "DIGITS", 0}, mDigits);
80  pc.outputs().snapshot(Output{"CTP", "LUMI", 0}, mLumi);
81  if (mTree->GetReadEntry() + 1 >= mTree->GetEntries()) {
82  pc.services().get<ControlService>().endOfStream();
83  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
84  }
85 }

87 void DigitReader::connectTree(const std::string& filename)
88 {
89  mTree.reset(nullptr); // in case it was already loaded
90  mFile.reset(TFile::Open(filename.c_str()));
91  assert(mFile && !mFile->IsZombie());
92  mTree.reset((TTree*)mFile->Get(mDigTreeName.c_str()));
93  assert(mTree);
94  if (mTree->GetBranch(mDigitBranchName.c_str())) {
95  mTree->SetBranchAddress(mDigitBranchName.c_str(), &mDigitsPtr);
96  } else {
97  LOGP(warn, "Digits branch {} is absent", mDigitBranchName);
98  }
99  if (mTree->GetBranch(mLumiBranchName.c_str())) {
100  mTree->SetBranchAddress(mLumiBranchName.c_str(), &mLumiPtr);
101  } else {
102  LOGP(warn, "Lumi branch {} is absent", mLumiBranchName);
103  }
104  mTree->SetBranchAddress(mDigitBranchName.c_str(), &mDigitsPtr);
105  LOG(info) << "Loaded tree from " << filename << " with " << mTree->GetEntries() << " entries";
106 }
