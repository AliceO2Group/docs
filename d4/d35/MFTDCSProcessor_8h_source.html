47  std::pair<uint64_t, double> midValue; // mid value seen by the MFT DCS processor
48  std::pair<uint64_t, double> maxChange; // maximum variation seen by the MFT DCS processor
49  std::pair<uint64_t, double> summary; // number of entries by the MFT DCS processor
50 
51  MFTDCSinfo()

53  firstValue = std::make_pair(0, -999999999);
54  lastValue = std::make_pair(0, -999999999);
55  meanValue = std::make_pair(0, -999999999);
56  stddevValue = std::make_pair(0, -999999999);
57  midValue = std::make_pair(0, -999999999);
58  maxChange = std::make_pair(0, -999999999);
59  summary = std::make_pair(0, -999999999);
60  }
61  void makeEmpty()
62  {
63  firstValue.first = lastValue.first = midValue.first = meanValue.first = stddevValue.first = maxChange.first = summary.first = 0.;
64  firstValue.second = lastValue.second = midValue.second = meanValue.second = stddevValue.second = maxChange.second = summary.second = -999999999;
65  }
66  void print() const;
67 
68  ClassDefNV(MFTDCSinfo, 2);
69 };
