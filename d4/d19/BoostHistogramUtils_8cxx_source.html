22std::string createErrorMessageFitGaus(o2::utils::FitGausError_t errorcode)
23{
24 switch (errorcode) {
25 case FitGausError_t::FIT_ERROR_MIN:
26 return "Gaus fit failed! xMax < 4";
27 case FitGausError_t::FIT_ERROR_MAX:
28 return "Gaus fit failed! xMax too large";
29 case FitGausError_t::FIT_ERROR_ENTRIES:
30 return "Gaus fit failed! entries < 12";
31 case FitGausError_t::FIT_ERROR_KTOL_MEAN:
32 return "Gaus fit failed! std::abs(par[1]) < kTol";
33 case FitGausError_t::FIT_ERROR_KTOL_SIGMA:
34 return "Gaus fit failed! std::abs(par[2]) < kTol";
35 case FitGausError_t::FIT_ERROR_KTOL_RMS:
36 return "Gaus fit failed! RMS < kTol";
37 default:
38 return "Gaus fit failed! Unknown error code";
39 }
40 return "Gaus fit failed! Unknown error code";
41}
