204 void TrackExtensionStudy::process()
205 {
206  LOGP(info, "** Filling particle table ... ");
207  mParticleInfo.resize(mKineReader->getNSources()); // sources
208  for (int iSource{0}; iSource < mKineReader->getNSources(); ++iSource) {
209  mParticleInfo[iSource].resize(mKineReader->getNEvents(iSource)); // events
210  for (int iEvent{0}; iEvent < mKineReader->getNEvents(iSource); ++iEvent) {
211  mParticleInfo[iSource][iEvent].resize(mKineReader->getTracks(iSource, iEvent).size()); // tracks
212  for (auto iPart{0}; iPart < mKineReader->getTracks(iEvent).size(); ++iPart) {
213  auto& part = mKineReader->getTracks(iSource, iEvent)[iPart];
214  mParticleInfo[iSource][iEvent][iPart].event = iEvent;
215  mParticleInfo[iSource][iEvent][iPart].pdg = part.GetPdgCode();
216  mParticleInfo[iSource][iEvent][iPart].pt = part.GetPt();
217  mParticleInfo[iSource][iEvent][iPart].phi = part.GetPhi();
218  mParticleInfo[iSource][iEvent][iPart].eta = part.GetEta();
219  mParticleInfo[iSource][iEvent][iPart].vx = part.Vx();
220  mParticleInfo[iSource][iEvent][iPart].vy = part.Vy();
221  mParticleInfo[iSource][iEvent][iPart].vz = part.Vz();
222  mParticleInfo[iSource][iEvent][iPart].isPrimary = part.isPrimary();
223  mParticleInfo[iSource][iEvent][iPart].mother = part.getMotherTrackId();
224  mParticleInfo[iSource][iEvent][iPart].prodProcess = part.getProcess();
225  }
226  }
227  }
228  LOGP(info, "** Creating particle/clusters correspondance ... ");
229  for (auto iSource{0}; iSource < mParticleInfo.size(); ++iSource) {
230  for (auto iCluster{0}; iCluster < mClusters.size(); ++iCluster) {
231  auto labs = mClustersMCLCont->getLabels(iCluster); // ideally I can have more than one label per cluster
232  for (auto& lab : labs) {
233  if (!lab.isValid()) {
234  continue; // We want to skip channels related to noise, e.g. sID = 99: QED
235  }
236  int trackID, evID, srcID;
237  bool fake;
238  lab.get(trackID, evID, srcID, fake);
239  auto& cluster = mClusters[iCluster];
240  auto layer = mGeometry->getLayer(cluster.getSensorID());
241  mParticleInfo[srcID][evID][trackID].clusters |= (1 << layer);
242  if (fake) {
243  mParticleInfo[srcID][evID][trackID].fakeClusters |= (1 << layer);
244  }
245  }
246  }
247  }
248 
249  LOGP(info, "** Analysing tracks ... ");
250  int unaccounted{0}, good{0}, fakes{0}, extended{0};
251  for (auto iTrack{0}; iTrack < mTracks.size(); ++iTrack) {
252  const auto& lab = mTracksMCLabels[iTrack];
253  if (!lab.isValid()) {
254  unaccounted++;
255  continue;
256  }
257  int trackID, evID, srcID;
258  bool fake;
259  lab.get(trackID, evID, srcID, fake);
260 
261  if (srcID == 99) { // skip QED
262  unaccounted++;
263  continue;
264  }
265 
266  for (int iLayer{0}; iLayer < 7; ++iLayer) {
267  if (mTracks[iTrack].isExtendedOnLayer(iLayer)) {
268  ++extended;
269  break;
270  }
271  }
272 
273  mParticleInfo[srcID][evID][trackID].isReco += !fake;
274  mParticleInfo[srcID][evID][trackID].isFake += fake;
275  if (mTracks[iTrack].isBetter(mParticleInfo[srcID][evID][trackID].track, 1.e9)) {
276  mParticleInfo[srcID][evID][trackID].storedStatus = fake;
277  mParticleInfo[srcID][evID][trackID].track = mTracks[iTrack];
278  mParticleInfo[srcID][evID][trackID].mcTrack = *mKineReader->getTrack(lab);
279  }
280  fakes += fake;
281  good += !fake;
282  }
283  LOGP(info, "** Some statistics:");
284  LOGP(info, "\t- Total number of tracks: {}", mTracks.size());
285  LOGP(info, "\t- Total number of tracks not corresponding to particles: {} ({:.2f} %)", unaccounted, unaccounted * 100. / mTracks.size());
286  LOGP(info, "\t- Total number of fakes: {} ({:.2f} %)", fakes, fakes * 100. / mTracks.size());
287  LOGP(info, "\t- Total number of good: {} ({:.2f} %)", good, good * 100. / mTracks.size());
288  LOGP(info, "\t- Total number of extensions: {} ({:.2f} %)", extended, extended * 100. / mTracks.size());
289 
290  LOGP(info, "** Filling histograms ... ");
291  for (auto iTrack{0}; iTrack < mTracks.size(); ++iTrack) {
292  auto& lab = mTracksMCLabels[iTrack];
293  if (!lab.isValid()) {
294  unaccounted++;
295  continue;
296  }
297  int trackID, evID, srcID;
298  bool fake;
299  lab.get(trackID, evID, srcID, fake);
300  const auto& part = mParticleInfo[srcID][evID][trackID];
301  if (!part.isPrimary) {
302  continue;
303  }
304  const auto& trk = part.track;
305  mHTrackCounts->Fill(0);
306 
307  std::bitset<7> extPattern{0};
308  for (int iLayer{0}; iLayer < 7; ++iLayer) {
309  if (trk.isExtendedOnLayer(iLayer)) {
310  extPattern.set(iLayer);
311  }
312  }
313 
314  if (!extPattern.any()) {
315  mHTrackCounts->Fill(1);
316  if (part.isReco || !part.isFake) {
317  mHTrackCounts->Fill(2);
318  } else {
319  mHTrackCounts->Fill(3);
320  }
321  continue;
322  }
323 
324  if (mWithTree) {
325  std::array<float, 3> xyz{(float)part.mcTrack.GetStartVertexCoordinatesX(), (float)part.mcTrack.GetStartVertexCoordinatesY(), (float)part.mcTrack.GetStartVertexCoordinatesZ()};
326  std::array<float, 3> pxyz{(float)part.mcTrack.GetStartVertexMomentumX(), (float)part.mcTrack.GetStartVertexMomentumY(), (float)part.mcTrack.GetStartVertexMomentumZ()};
327  auto pdg = O2DatabasePDG::Instance()->GetParticle(part.pdg);
328  if (pdg == nullptr) {
329  LOGP(fatal, "MC info not available");
330  }
331  auto mcTrk = o2::track::TrackPar(xyz, pxyz, TMath::Nint(pdg->Charge() / 3.), true);
332  (*mStream) << "tree"
333  << "trk=" << trk
334  << "mcTrk=" << mcTrk
335  << "\n";
336  }
337 
338  mHTrackCounts->Fill(4);
339  mHLengthAny->Fill(trk.getNClusters());
340  mHChi2Any->Fill(trk.getChi2());
341  mHPtAny->Fill(trk.getPt());
342  if (part.isReco || !part.isFake) {
343  mHTrackCounts->Fill(5);
344  mHLengthGood->Fill(trk.getNClusters());
345  mHChi2Good->Fill(trk.getChi2());
346  mHPtGood->Fill(trk.getPt());
347  } else {
348  mHTrackCounts->Fill(6);
349  mHLengthFake->Fill(trk.getNClusters());
350  mHChi2Fake->Fill(trk.getChi2());
351  mHPtFake->Fill(trk.getPt());
352  }
353 
354  std::bitset<7> clusPattern{static_cast<uint8_t>(trk.getPattern())};
355  for (int iLayer{0}; iLayer < 7; ++iLayer) {
356  if (extPattern.test(iLayer)) {
357  extPattern.set(iLayer);
358  mHExtensionAny->Fill(iLayer);
359  if (part.isReco || !part.isFake) {
360  mHExtensionGood->Fill(iLayer);
361  } else {
362  mHExtensionFake->Fill(iLayer);
363  }
364 
365  if ((part.fakeClusters & (0x1 << iLayer)) == 0) {
366  mHClustersCounts->Fill(4);
367  } else {
368  mHClustersCounts->Fill(5);
369  }
370  }
371  }
372  std::bitset<7> oldPattern{clusPattern & ~extPattern};
373  auto clusN = clusPattern.to_ulong();
374  auto clusIdx = std::distance(std::begin(mBitPatternsAfter), std::find(std::begin(mBitPatternsAfter), std::end(mBitPatternsAfter), clusN));
375  auto oldN = oldPattern.to_ulong();
376  auto oldIdx = std::distance(std::begin(mBitPatternsBefore), std::find(std::begin(mBitPatternsBefore), std::end(mBitPatternsBefore), oldN));
377  mHExtensionPatternsAny->Fill(oldIdx, clusIdx);
378  if (part.isReco || !part.isFake) {
379  mHExtensionPatternsGood->Fill(oldIdx, clusIdx);
380  } else {
381  mHExtensionPatternsFake->Fill(oldIdx, clusIdx);
382  }
383 
384  // old pattern
385  bool oldFake{false}, newFake{false};
386  for (int iLayer{0}; iLayer < 7; ++iLayer) {
387  if (trk.isFakeOnLayer(iLayer)) {
388  if (oldPattern.test(iLayer)) {
389  oldFake = true;
390  } else if (extPattern.test(iLayer)) {
391  newFake = true;
392  }
393  }
394  }
395 
396  if (oldFake && newFake) {
397  mHTrackCounts->Fill(9);
398  } else if (oldFake) {
399  mHTrackCounts->Fill(7);
400  } else if (newFake) {
401  mHTrackCounts->Fill(8);
402  }
403 
404  mHClustersCounts->SetBinContent(2, mHClustersCounts->GetBinContent(2) + oldPattern.count());
405  mHClustersCounts->SetBinContent(3, mHClustersCounts->GetBinContent(3) + extPattern.count());
406  }

452 {

456  dataRequest->requestClusters(srcClustersMask, useMC);
457 
458  auto ggRequest = std::make_shared<o2::base::GRPGeomRequest>(false, // orbitResetTime
459  true, // GRPECS=true
460  false, // GRPLHCIF
461  true, // GRPMagField
462  true, // askMatLUT
463  o2::base::GRPGeomRequest::Aligned, // geometry
464  dataRequest->inputs,
465  true);
466 
467  return DataProcessorSpec{
468  "its-study-track-extension",
469  dataRequest->inputs,
470  outputs,
471  AlgorithmSpec{adaptFromTask<TrackExtensionStudy>(dataRequest, srcTracksMask, useMC, kineReader, ggRequest)},
