51 std::pair<uint32_t, uint32_t> GPUChainTracking::TPCClusterizerDecodeZSCountUpdate(uint32_t iSlice, const CfFragment& fragment)
52 {
53  bool doGPU = mRec->GetRecoStepsGPU() & GPUDataTypes::RecoStep::TPCClusterFinding;
54  GPUTPCClusterFinder& clusterer = processors()->tpcClusterer[iSlice];
55  GPUTPCClusterFinder::ZSOffset* o = processors()->tpcClusterer[iSlice].mPzsOffsets;
56  uint32_t digits = 0;
57  uint32_t pages = 0;
58  for (uint16_t j = 0; j < GPUTrackingInOutZS::NENDPOINTS; j++) {
59  clusterer.mMinMaxCN[j] = mCFContext->fragmentData[fragment.index].minMaxCN[iSlice][j];
60  if (doGPU) {
61  uint16_t posInEndpoint = 0;
62  uint16_t pagesEndpoint = 0;
63  for (uint32_t k = clusterer.mMinMaxCN[j].zsPtrFirst; k < clusterer.mMinMaxCN[j].zsPtrLast; k++) {
64  const uint32_t pageFirst = (k == clusterer.mMinMaxCN[j].zsPtrFirst) ? clusterer.mMinMaxCN[j].zsPageFirst : 0;
65  const uint32_t pageLast = (k + 1 == clusterer.mMinMaxCN[j].zsPtrLast) ? clusterer.mMinMaxCN[j].zsPageLast : mIOPtrs.tpcZS->slice[iSlice].nZSPtr[j][k];
66  for (uint32_t l = pageFirst; l < pageLast; l++) {
67  uint16_t pageDigits = mCFContext->fragmentData[fragment.index].pageDigits[iSlice][j][posInEndpoint++];
68  if (pageDigits) {
69  *(o++) = GPUTPCClusterFinder::ZSOffset{digits, j, pagesEndpoint};
70  digits += pageDigits;
71  }
72  pagesEndpoint++;
73  }
74  }
75  if (pagesEndpoint != mCFContext->fragmentData[fragment.index].pageDigits[iSlice][j].size()) {
76  if (GetProcessingSettings().ignoreNonFatalGPUErrors) {
77  GPUError("TPC raw page count mismatch in TPCClusterizerDecodeZSCountUpdate: expected %d / buffered %lu", pagesEndpoint, mCFContext->fragmentData[fragment.index].pageDigits[iSlice][j].size());
78  return {0, 0};
79  } else {
80  GPUFatal("TPC raw page count mismatch in TPCClusterizerDecodeZSCountUpdate: expected %d / buffered %lu", pagesEndpoint, mCFContext->fragmentData[fragment.index].pageDigits[iSlice][j].size());
81  }
82  }
83  } else {
84  clusterer.mPzsOffsets[j] = GPUTPCClusterFinder::ZSOffset{digits, j, 0};
85  digits += mCFContext->fragmentData[fragment.index].nDigits[iSlice][j];
86  pages += mCFContext->fragmentData[fragment.index].nPages[iSlice][j];
87  }
88  }
89  if (doGPU) {
90  pages = o - processors()->tpcClusterer[iSlice].mPzsOffsets;
91  }
92  if (!doGPU && GetProcessingSettings().debugLevel >= 4 && mCFContext->zsVersion >= ZSVersion::ZSVersionDenseLinkBased) {
93  TPCClusterizerEnsureZSOffsets(iSlice, fragment);
94  }
95  return {digits, pages};
96 }

98 void GPUChainTracking::TPCClusterizerEnsureZSOffsets(uint32_t iSlice, const CfFragment& fragment)
99 {
100  GPUTPCClusterFinder& clusterer = processors()->tpcClusterer[iSlice];
101  uint32_t nAdcs = 0;
102  for (uint16_t endpoint = 0; endpoint < GPUTrackingInOutZS::NENDPOINTS; endpoint++) {
103  const auto& data = mCFContext->fragmentData[fragment.index];
104  uint32_t pagesEndpoint = 0;
105  const uint32_t nAdcsExpected = data.nDigits[iSlice][endpoint];
106  const uint32_t nPagesExpected = data.nPages[iSlice][endpoint];
107 
108  uint32_t nAdcDecoded = 0;
109  const auto& zs = mIOPtrs.tpcZS->slice[iSlice];
110  for (uint32_t i = data.minMaxCN[iSlice][endpoint].zsPtrFirst; i < data.minMaxCN[iSlice][endpoint].zsPtrLast; i++) {
111  const uint32_t pageFirst = (i == data.minMaxCN[iSlice][endpoint].zsPtrFirst) ? data.minMaxCN[iSlice][endpoint].zsPageFirst : 0;
112  const uint32_t pageLast = (i + 1 == data.minMaxCN[iSlice][endpoint].zsPtrLast) ? data.minMaxCN[iSlice][endpoint].zsPageLast : zs.nZSPtr[endpoint][i];
113  for (uint32_t j = pageFirst; j < pageLast; j++) {
114  const uint8_t* page = static_cast<const uint8_t*>(zs.zsPtr[endpoint][i]) + j * TPCZSHDR::TPC_ZS_PAGE_SIZE;
115  const header::RAWDataHeader* rawDataHeader = reinterpret_cast<const header::RAWDataHeader*>(page);
116  const TPCZSHDRV2* decHdr = reinterpret_cast<const TPCZSHDRV2*>(page + raw::RDHUtils::getMemorySize(*rawDataHeader) - sizeof(TPCZSHDRV2));
117  const uint16_t nSamplesInPage = decHdr->nADCsamples;
118 
119  nAdcDecoded += nSamplesInPage;
120  pagesEndpoint++;
121  }
122  }
123 
124  if (pagesEndpoint != nPagesExpected) {
125  GPUFatal("Sector %d, Endpoint %d, Fragment %d: TPC raw page count mismatch: expected %d / buffered %lu", iSlice, endpoint, fragment.index, pagesEndpoint, nPagesExpected);
126  }
127 
128  if (nAdcDecoded != nAdcsExpected) {
129  GPUFatal("Sector %d, Endpoint %d, Fragment %d: TPC ADC count mismatch: expected %u, buffered %u", iSlice, endpoint, fragment.index, nAdcsExpected, nAdcDecoded);
130  }
131 
132  if (nAdcs != clusterer.mPzsOffsets[endpoint].offset) {
133  GPUFatal("Sector %d, Endpoint %d, Fragment %d: TPC ADC offset mismatch: expected %u, buffered %u", iSlice, endpoint, fragment.index, nAdcs, clusterer.mPzsOffsets[endpoint].offset);
134  }
135 
136  nAdcs += nAdcsExpected;
137  }

147 std::pair<uint32_t, uint32_t> GPUChainTracking::TPCClusterizerDecodeZSCount(uint32_t iSlice, const CfFragment& fragment)
148 {
149  mRec->getGeneralStepTimer(GeneralStep::Prepare).Start();
150  uint32_t nDigits = 0;
151  uint32_t nPages = 0;
152  uint32_t endpointAdcSamples[GPUTrackingInOutZS::NENDPOINTS];
153  memset(endpointAdcSamples, 0, sizeof(endpointAdcSamples));
154  bool doGPU = mRec->GetRecoStepsGPU() & GPUDataTypes::RecoStep::TPCClusterFinding;
155  int32_t firstHBF = (mIOPtrs.settingsTF && mIOPtrs.settingsTF->hasTfStartOrbit) ? mIOPtrs.settingsTF->tfStartOrbit : (mIOPtrs.tpcZS->slice[iSlice].count[0] && mIOPtrs.tpcZS->slice[iSlice].nZSPtr[0][0]) ? o2::raw::RDHUtils::getHeartBeatOrbit(*(const o2::header::RAWDataHeader*)mIOPtrs.tpcZS->slice[iSlice].zsPtr[0][0]) : 0;
156 
157  for (uint16_t j = 0; j < GPUTrackingInOutZS::NENDPOINTS; j++) {
158 #ifndef GPUCA_NO_VC
159  if (GetProcessingSettings().prefetchTPCpageScan >= 3 && j < GPUTrackingInOutZS::NENDPOINTS - 1) {
160  for (uint32_t k = 0; k < mIOPtrs.tpcZS->slice[iSlice].count[j + 1]; k++) {
161  for (uint32_t l = 0; l < mIOPtrs.tpcZS->slice[iSlice].nZSPtr[j + 1][k]; l++) {
162  Vc::Common::prefetchMid(((const uint8_t*)mIOPtrs.tpcZS->slice[iSlice].zsPtr[j + 1][k]) + l * TPCZSHDR::TPC_ZS_PAGE_SIZE);
163  Vc::Common::prefetchMid(((const uint8_t*)mIOPtrs.tpcZS->slice[iSlice].zsPtr[j + 1][k]) + l * TPCZSHDR::TPC_ZS_PAGE_SIZE + sizeof(o2::header::RAWDataHeader));
164  }
165  }
166  }
167 #endif
168 
169  std::vector<std::pair<CfFragment, TPCCFDecodeScanTmp>> fragments;
170  fragments.reserve(mCFContext->nFragments);
171  fragments.emplace_back(std::pair<CfFragment, TPCCFDecodeScanTmp>{fragment, {0, 0, 0, 0, 0, -1}});
172  for (uint32_t i = 1; i < mCFContext->nFragments; i++) {
173  fragments.emplace_back(std::pair<CfFragment, TPCCFDecodeScanTmp>{fragments.back().first.next(), {0, 0, 0, 0, 0, -1}});
174  }
175  std::vector<bool> fragmentExtends(mCFContext->nFragments, false);
176 
177  uint32_t firstPossibleFragment = 0;
178  uint32_t pageCounter = 0;
179  uint32_t emptyPages = 0;
180  for (uint32_t k = 0; k < mIOPtrs.tpcZS->slice[iSlice].count[j]; k++) {
181  if (GetProcessingSettings().tpcSingleSector != -1 && GetProcessingSettings().tpcSingleSector != (int32_t)iSlice) {
182  break;
183  }
184  nPages += mIOPtrs.tpcZS->slice[iSlice].nZSPtr[j][k];
185  for (uint32_t l = 0; l < mIOPtrs.tpcZS->slice[iSlice].nZSPtr[j][k]; l++) {
186 #ifndef GPUCA_NO_VC
187  if (GetProcessingSettings().prefetchTPCpageScan >= 2 && l + 1 < mIOPtrs.tpcZS->slice[iSlice].nZSPtr[j][k]) {
188  Vc::Common::prefetchForOneRead(((const uint8_t*)mIOPtrs.tpcZS->slice[iSlice].zsPtr[j][k]) + (l + 1) * TPCZSHDR::TPC_ZS_PAGE_SIZE);
189  Vc::Common::prefetchForOneRead(((const uint8_t*)mIOPtrs.tpcZS->slice[iSlice].zsPtr[j][k]) + (l + 1) * TPCZSHDR::TPC_ZS_PAGE_SIZE + sizeof(o2::header::RAWDataHeader));
190  }
191 #endif
192  const uint8_t* const page = ((const uint8_t*)mIOPtrs.tpcZS->slice[iSlice].zsPtr[j][k]) + l * TPCZSHDR::TPC_ZS_PAGE_SIZE;
193  const o2::header::RAWDataHeader* rdh = (const o2::header::RAWDataHeader*)page;
194  if (o2::raw::RDHUtils::getMemorySize(*rdh) == sizeof(o2::header::RAWDataHeader)) {
195  emptyPages++;
196  continue;
197  }
198  pageCounter++;
199  const TPCZSHDR* const hdr = (const TPCZSHDR*)(rdh_utils::getLink(o2::raw::RDHUtils::getFEEID(*rdh)) == rdh_utils::DLBZSLinkID ? (page + o2::raw::RDHUtils::getMemorySize(*rdh) - sizeof(TPCZSHDRV2)) : (page + sizeof(o2::header::RAWDataHeader)));
200  if (mCFContext->zsVersion == -1) {
201  mCFContext->zsVersion = hdr->version;
202  if (GetProcessingSettings().param.tpcTriggerHandling && mCFContext->zsVersion < ZSVersion::ZSVersionDenseLinkBased) { // TODO: Move tpcTriggerHandling to recoSteps bitmask
203  static bool errorShown = false;
204  if (errorShown == false) {
205  GPUAlarm("Trigger handling only possible with TPC Dense Link Based data, received version %d, disabling", mCFContext->zsVersion);
206  }
207  errorShown = true;
208  }
209  } else if (mCFContext->zsVersion != (int32_t)hdr->version) {
210  GPUError("Received TPC ZS 8kb page of mixed versions, expected %d, received %d (linkid %d, feeCRU %d, feeEndpoint %d, feelinkid %d)", mCFContext->zsVersion, (int32_t)hdr->version, (int32_t)o2::raw::RDHUtils::getLinkID(*rdh), (int32_t)rdh_utils::getCRU(*rdh), (int32_t)rdh_utils::getEndPoint(*rdh), (int32_t)rdh_utils::getLink(*rdh));
211  constexpr size_t bufferSize = 3 * std::max(sizeof(*rdh), sizeof(*hdr)) + 1;
212  char dumpBuffer[bufferSize];
213  for (size_t i = 0; i < sizeof(*rdh); i++) {
214  // "%02X " guaranteed to be 3 chars + ending 0.
215  snprintf(dumpBuffer + 3 * i, 4, "%02X ", (int32_t)((uint8_t*)rdh)[i]);
216  }
217  GPUAlarm("RDH of page: %s", dumpBuffer);
218  for (size_t i = 0; i < sizeof(*hdr); i++) {
219  // "%02X " guaranteed to be 3 chars + ending 0.
220  snprintf(dumpBuffer + 3 * i, 4, "%02X ", (int32_t)((uint8_t*)hdr)[i]);
221  }
222  GPUAlarm("Metainfo of page: %s", dumpBuffer);
223  if (GetProcessingSettings().ignoreNonFatalGPUErrors) {
224  mCFContext->abandonTimeframe = true;
225  return {0, 0};
226  } else {
227  GPUFatal("Cannot process with invalid TPC ZS data, exiting");
228  }
229  }
230  if (GetProcessingSettings().param.tpcTriggerHandling) {
231  const TPCZSHDRV2* const hdr2 = (const TPCZSHDRV2*)hdr;
232  if (hdr2->flags & TPCZSHDRV2::ZSFlags::TriggerWordPresent) {
233  const char* triggerWord = (const char*)hdr - TPCZSHDRV2::TRIGGER_WORD_SIZE;
234  o2::tpc::TriggerInfoDLBZS tmp;
235  memcpy((void*)&tmp.triggerWord, triggerWord, TPCZSHDRV2::TRIGGER_WORD_SIZE);
236  tmp.orbit = o2::raw::RDHUtils::getHeartBeatOrbit(*rdh);
237  if (tmp.triggerWord.isValid(0)) {
238  mTriggerBuffer->triggers.emplace(tmp);
239  }
240  }
241  }
242  nDigits += hdr->nADCsamples;
243  endpointAdcSamples[j] += hdr->nADCsamples;
244  uint32_t timeBin = (hdr->timeOffset + (o2::raw::RDHUtils::getHeartBeatOrbit(*rdh) - firstHBF) * o2::constants::lhc::LHCMaxBunches) / LHCBCPERTIMEBIN;
245  uint32_t maxTimeBin = timeBin + hdr->nTimeBinSpan;
246  if (mCFContext->zsVersion >= ZSVersion::ZSVersionDenseLinkBased) {
247  const TPCZSHDRV2* const hdr2 = (const TPCZSHDRV2*)hdr;
248  if (hdr2->flags & TPCZSHDRV2::ZSFlags::nTimeBinSpanBit8) {
249  maxTimeBin += 256;
250  }
251  }
252  if (maxTimeBin > mCFContext->tpcMaxTimeBin) {
253  mCFContext->tpcMaxTimeBin = maxTimeBin;
254  }
255  bool extendsInNextPage = false;
256  if (mCFContext->zsVersion >= ZSVersion::ZSVersionDenseLinkBased) {
257  if (l + 1 < mIOPtrs.tpcZS->slice[iSlice].nZSPtr[j][k] && o2::raw::RDHUtils::getMemorySize(*rdh) == TPCZSHDR::TPC_ZS_PAGE_SIZE) {
258  const o2::header::RAWDataHeader* nextrdh = (const o2::header::RAWDataHeader*)(page + TPCZSHDR::TPC_ZS_PAGE_SIZE);
259  extendsInNextPage = o2::raw::RDHUtils::getHeartBeatOrbit(*nextrdh) == o2::raw::RDHUtils::getHeartBeatOrbit(*rdh) && o2::raw::RDHUtils::getMemorySize(*nextrdh) > sizeof(o2::header::RAWDataHeader);
260  }
261  }
262  while (firstPossibleFragment && (uint32_t)fragments[firstPossibleFragment - 1].first.last() > timeBin) {
263  firstPossibleFragment--;
264  }
265  auto handleExtends = [&](uint32_t ff) {
266  if (fragmentExtends[ff]) {
267  if (doGPU) {
268  // Only add extended page on GPU. On CPU the pages are in consecutive memory anyway.
269  // Not adding the page prevents an issue where a page is decoded twice on CPU, when only the extend should be decoded.
270  fragments[ff].second.zsPageLast++;
271  mCFContext->fragmentData[ff].nPages[iSlice][j]++;
272  mCFContext->fragmentData[ff].pageDigits[iSlice][j].emplace_back(0);
273  }
274  fragmentExtends[ff] = false;
275  }
276  };
277  if (mCFContext->zsVersion >= ZSVersion::ZSVersionDenseLinkBased) {
278  for (uint32_t ff = 0; ff < firstPossibleFragment; ff++) {
279  handleExtends(ff);
280  }
281  }
282  for (uint32_t f = firstPossibleFragment; f < mCFContext->nFragments; f++) {
283  if (timeBin < (uint32_t)fragments[f].first.last() && (uint32_t)fragments[f].first.first() <= maxTimeBin) {
284  if (!fragments[f].second.hasData) {
285  fragments[f].second.hasData = 1;
286  fragments[f].second.zsPtrFirst = k;
287  fragments[f].second.zsPageFirst = l;
288  } else {
289  if (pageCounter > (uint32_t)fragments[f].second.pageCounter + 1) {
290  mCFContext->fragmentData[f].nPages[iSlice][j] += emptyPages + pageCounter - fragments[f].second.pageCounter - 1;
291  for (uint32_t k2 = fragments[f].second.zsPtrLast - 1; k2 <= k; k2++) {
292  for (uint32_t l2 = ((int32_t)k2 == fragments[f].second.zsPtrLast - 1) ? fragments[f].second.zsPageLast : 0; l2 < (k2 < k ? mIOPtrs.tpcZS->slice[iSlice].nZSPtr[j][k2] : l); l2++) {
293  if (doGPU) {
294  mCFContext->fragmentData[f].pageDigits[iSlice][j].emplace_back(0);
295  } else {
296  // CPU cannot skip unneeded pages, so we must keep space to store the invalid dummy clusters
297  const uint8_t* const pageTmp = ((const uint8_t*)mIOPtrs.tpcZS->slice[iSlice].zsPtr[j][k2]) + l2 * TPCZSHDR::TPC_ZS_PAGE_SIZE;
298  const o2::header::RAWDataHeader* rdhTmp = (const o2::header::RAWDataHeader*)pageTmp;
299  if (o2::raw::RDHUtils::getMemorySize(*rdhTmp) != sizeof(o2::header::RAWDataHeader)) {
300  const TPCZSHDR* const hdrTmp = (const TPCZSHDR*)(rdh_utils::getLink(o2::raw::RDHUtils::getFEEID(*rdhTmp)) == rdh_utils::DLBZSLinkID ? (pageTmp + o2::raw::RDHUtils::getMemorySize(*rdhTmp) - sizeof(TPCZSHDRV2)) : (pageTmp + sizeof(o2::header::RAWDataHeader)));
301  mCFContext->fragmentData[f].nDigits[iSlice][j] += hdrTmp->nADCsamples;
302  }
303  }
304  }
305  }
306  } else if (emptyPages) {
307  mCFContext->fragmentData[f].nPages[iSlice][j] += emptyPages;
308  if (doGPU) {
309  for (uint32_t m = 0; m < emptyPages; m++) {
310  mCFContext->fragmentData[f].pageDigits[iSlice][j].emplace_back(0);
311  }
312  }
313  }
314  }
315  fragments[f].second.zsPtrLast = k + 1;
316  fragments[f].second.zsPageLast = l + 1;
317  fragments[f].second.pageCounter = pageCounter;
318  mCFContext->fragmentData[f].nPages[iSlice][j]++;
319  mCFContext->fragmentData[f].nDigits[iSlice][j] += hdr->nADCsamples;
320  if (doGPU) {
321  mCFContext->fragmentData[f].pageDigits[iSlice][j].emplace_back(hdr->nADCsamples);
322  }
323  fragmentExtends[f] = extendsInNextPage;
324  } else {
325  handleExtends(f);
326  if (timeBin < (uint32_t)fragments[f].first.last()) {
327  if (mCFContext->zsVersion >= ZSVersion::ZSVersionDenseLinkBased) {
328  for (uint32_t ff = f + 1; ff < mCFContext->nFragments; ff++) {
329  handleExtends(ff);
330  }
331  }
332  break;
333  } else {
334  firstPossibleFragment = f + 1;
335  }
336  }
337  }
338  emptyPages = 0;
339  }
340  }
341  for (uint32_t f = 0; f < mCFContext->nFragments; f++) {
342  mCFContext->fragmentData[f].minMaxCN[iSlice][j].zsPtrLast = fragments[f].second.zsPtrLast;
343  mCFContext->fragmentData[f].minMaxCN[iSlice][j].zsPtrFirst = fragments[f].second.zsPtrFirst;
344  mCFContext->fragmentData[f].minMaxCN[iSlice][j].zsPageLast = fragments[f].second.zsPageLast;
345  mCFContext->fragmentData[f].minMaxCN[iSlice][j].zsPageFirst = fragments[f].second.zsPageFirst;
346  }
347  }
348  mCFContext->nPagesTotal += nPages;
349  mCFContext->nPagesSector[iSlice] = nPages;
350 
351  mCFContext->nDigitsEndpointMax[iSlice] = 0;
352  for (uint32_t i = 0; i < GPUTrackingInOutZS::NENDPOINTS; i++) {
353  if (endpointAdcSamples[i] > mCFContext->nDigitsEndpointMax[iSlice]) {
354  mCFContext->nDigitsEndpointMax[iSlice] = endpointAdcSamples[i];
355  }
356  }
357  uint32_t nDigitsFragmentMax = 0;
358  for (uint32_t i = 0; i < mCFContext->nFragments; i++) {
359  uint32_t pagesInFragment = 0;
360  uint32_t digitsInFragment = 0;
361  for (uint16_t j = 0; j < GPUTrackingInOutZS::NENDPOINTS; j++) {
362  pagesInFragment += mCFContext->fragmentData[i].nPages[iSlice][j];
363  digitsInFragment += mCFContext->fragmentData[i].nDigits[iSlice][j];
364  }
365  mCFContext->nPagesFragmentMax = std::max(mCFContext->nPagesFragmentMax, pagesInFragment);
366  nDigitsFragmentMax = std::max(nDigitsFragmentMax, digitsInFragment);
367  }
368  mRec->getGeneralStepTimer(GeneralStep::Prepare).Stop();
369  return {nDigits, nDigitsFragmentMax};
370 }

372 void GPUChainTracking::RunTPCClusterizer_compactPeaks(GPUTPCClusterFinder& clusterer, GPUTPCClusterFinder& clustererShadow, int32_t stage, bool doGPU, int32_t lane)
373 {
374  auto& in = stage ? clustererShadow.mPpeakPositions : clustererShadow.mPpositions;
375  auto& out = stage ? clustererShadow.mPfilteredPeakPositions : clustererShadow.mPpeakPositions;
376  if (doGPU) {
377  const uint32_t iSlice = clusterer.mISlice;
378  auto& count = stage ? clusterer.mPmemory->counters.nPeaks : clusterer.mPmemory->counters.nPositions;
379 
380  std::vector<size_t> counts;
381 
382  uint32_t nSteps = clusterer.getNSteps(count);
383  if (nSteps > clusterer.mNBufs) {
384  GPUError("Clusterer buffers exceeded (%u > %u)", nSteps, (int32_t)clusterer.mNBufs);
385  exit(1);
386  }
387 
388  size_t tmpCount = count;
389  if (nSteps > 1) {
390  for (uint32_t i = 1; i < nSteps; i++) {
391  counts.push_back(tmpCount);
392  if (i == 1) {
393  runKernel<GPUTPCCFStreamCompaction, GPUTPCCFStreamCompaction::scanStart>({GetGrid(tmpCount, clusterer.mScanWorkGroupSize, lane), {iSlice}}, i, stage);
394  } else {
