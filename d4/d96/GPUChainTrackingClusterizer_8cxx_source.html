52std::pair<uint32_t, uint32_t> GPUChainTracking::TPCClusterizerDecodeZSCountUpdate(uint32_t iSlice, const CfFragment& fragment)
53{
54 bool doGPU = mRec->GetRecoStepsGPU() & GPUDataTypes::RecoStep::TPCClusterFinding;
55 GPUTPCClusterFinder& clusterer = processors()->tpcClusterer[iSlice];
56 GPUTPCClusterFinder::ZSOffset* o = processors()->tpcClusterer[iSlice].mPzsOffsets;
57 uint32_t digits = 0;
58 uint32_t pages = 0;
59 for (uint16_t j = 0; j < GPUTrackingInOutZS::NENDPOINTS; j++) {
60 clusterer.mMinMaxCN[j] = mCFContext->fragmentData[fragment.index].minMaxCN[iSlice][j];
61 if (doGPU) {
62 uint16_t posInEndpoint = 0;
63 uint16_t pagesEndpoint = 0;
64 for (uint32_t k = clusterer.mMinMaxCN[j].zsPtrFirst; k < clusterer.mMinMaxCN[j].zsPtrLast; k++) {
65 const uint32_t pageFirst = (k == clusterer.mMinMaxCN[j].zsPtrFirst) ? clusterer.mMinMaxCN[j].zsPageFirst : 0;
66 const uint32_t pageLast = (k + 1 == clusterer.mMinMaxCN[j].zsPtrLast) ? clusterer.mMinMaxCN[j].zsPageLast : mIOPtrs.tpcZS->slice[iSlice].nZSPtr[j][k];
67 for (uint32_t l = pageFirst; l < pageLast; l++) {
68 uint16_t pageDigits = mCFContext->fragmentData[fragment.index].pageDigits[iSlice][j][posInEndpoint++];
69 if (pageDigits) {
70 *(o++) = GPUTPCClusterFinder::ZSOffset{digits, j, pagesEndpoint};
71 digits += pageDigits;
72 }
73 pagesEndpoint++;
74 }
75 }
76 if (pagesEndpoint != mCFContext->fragmentData[fragment.index].pageDigits[iSlice][j].size()) {
77 if (GetProcessingSettings().ignoreNonFatalGPUErrors) {
78 GPUError("TPC raw page count mismatch in TPCClusterizerDecodeZSCountUpdate: expected %d / buffered %lu", pagesEndpoint, mCFContext->fragmentData[fragment.index].pageDigits[iSlice][j].size());
79 return {0, 0};
80 } else {
81 GPUFatal("TPC raw page count mismatch in TPCClusterizerDecodeZSCountUpdate: expected %d / buffered %lu", pagesEndpoint, mCFContext->fragmentData[fragment.index].pageDigits[iSlice][j].size());
82 }
83 }
84 } else {
85 clusterer.mPzsOffsets[j] = GPUTPCClusterFinder::ZSOffset{digits, j, 0};
86 digits += mCFContext->fragmentData[fragment.index].nDigits[iSlice][j];
87 pages += mCFContext->fragmentData[fragment.index].nPages[iSlice][j];
88 }
89 }
90 if (doGPU) {
91 pages = o - processors()->tpcClusterer[iSlice].mPzsOffsets;
92 }
93 if (!doGPU && GetProcessingSettings().debugLevel >= 4 && mCFContext->zsVersion >= ZSVersion::ZSVersionDenseLinkBased) {
94 TPCClusterizerEnsureZSOffsets(iSlice, fragment);
95 }
96 return {digits, pages};
97}

99void GPUChainTracking::TPCClusterizerEnsureZSOffsets(uint32_t iSlice, const CfFragment& fragment)
100{
101 GPUTPCClusterFinder& clusterer = processors()->tpcClusterer[iSlice];
102 uint32_t nAdcs = 0;
103 for (uint16_t endpoint = 0; endpoint < GPUTrackingInOutZS::NENDPOINTS; endpoint++) {
104 const auto& data = mCFContext->fragmentData[fragment.index];
105 uint32_t pagesEndpoint = 0;
106 const uint32_t nAdcsExpected = data.nDigits[iSlice][endpoint];
107 const uint32_t nPagesExpected = data.nPages[iSlice][endpoint];
108
109 uint32_t nAdcDecoded = 0;
110 const auto& zs = mIOPtrs.tpcZS->slice[iSlice];
111 for (uint32_t i = data.minMaxCN[iSlice][endpoint].zsPtrFirst; i < data.minMaxCN[iSlice][endpoint].zsPtrLast; i++) {
112 const uint32_t pageFirst = (i == data.minMaxCN[iSlice][endpoint].zsPtrFirst) ? data.minMaxCN[iSlice][endpoint].zsPageFirst : 0;
113 const uint32_t pageLast = (i + 1 == data.minMaxCN[iSlice][endpoint].zsPtrLast) ? data.minMaxCN[iSlice][endpoint].zsPageLast : zs.nZSPtr[endpoint][i];
114 for (uint32_t j = pageFirst; j < pageLast; j++) {
115 const uint8_t* page = static_cast<const uint8_t*>(zs.zsPtr[endpoint][i]) + j * TPCZSHDR::TPC_ZS_PAGE_SIZE;
116 const header::RAWDataHeader* rawDataHeader = reinterpret_cast<const header::RAWDataHeader*>(page);
117 const TPCZSHDRV2* decHdr = reinterpret_cast<const TPCZSHDRV2*>(page + raw::RDHUtils::getMemorySize(*rawDataHeader) - sizeof(TPCZSHDRV2));
118 const uint16_t nSamplesInPage = decHdr->nADCsamples;
119
120 nAdcDecoded += nSamplesInPage;
121 pagesEndpoint++;
122 }
123 }
124
125 if (pagesEndpoint != nPagesExpected) {
126 GPUFatal("Sector %d, Endpoint %d, Fragment %d: TPC raw page count mismatch: expected %d / buffered %lu", iSlice, endpoint, fragment.index, pagesEndpoint, nPagesExpected);
127 }
128
129 if (nAdcDecoded != nAdcsExpected) {
130 GPUFatal("Sector %d, Endpoint %d, Fragment %d: TPC ADC count mismatch: expected %u, buffered %u", iSlice, endpoint, fragment.index, nAdcsExpected, nAdcDecoded);
131 }
132
133 if (nAdcs != clusterer.mPzsOffsets[endpoint].offset) {
134 GPUFatal("Sector %d, Endpoint %d, Fragment %d: TPC ADC offset mismatch: expected %u, buffered %u", iSlice, endpoint, fragment.index, nAdcs, clusterer.mPzsOffsets[endpoint].offset);
135 }
136
137 nAdcs += nAdcsExpected;
138 }

148std::pair<uint32_t, uint32_t> GPUChainTracking::TPCClusterizerDecodeZSCount(uint32_t iSlice, const CfFragment& fragment)
149{
150 mRec->getGeneralStepTimer(GeneralStep::Prepare).Start();
151 uint32_t nDigits = 0;
152 uint32_t nPages = 0;
153 uint32_t endpointAdcSamples[GPUTrackingInOutZS::NENDPOINTS];
154 memset(endpointAdcSamples, 0, sizeof(endpointAdcSamples));
155 bool doGPU = mRec->GetRecoStepsGPU() & GPUDataTypes::RecoStep::TPCClusterFinding;
156 int32_t firstHBF = (mIOPtrs.settingsTF && mIOPtrs.settingsTF->hasTfStartOrbit) ? mIOPtrs.settingsTF->tfStartOrbit : (mIOPtrs.tpcZS->slice[iSlice].count[0] && mIOPtrs.tpcZS->slice[iSlice].nZSPtr[0][0]) ? o2::raw::RDHUtils::getHeartBeatOrbit(*(const o2::header::RAWDataHeader*)mIOPtrs.tpcZS->slice[iSlice].zsPtr[0][0]) : 0;
157
158 for (uint16_t j = 0; j < GPUTrackingInOutZS::NENDPOINTS; j++) {
159#ifndef GPUCA_NO_VC
160 if (GetProcessingSettings().prefetchTPCpageScan >= 3 && j < GPUTrackingInOutZS::NENDPOINTS - 1) {
161 for (uint32_t k = 0; k < mIOPtrs.tpcZS->slice[iSlice].count[j + 1]; k++) {
162 for (uint32_t l = 0; l < mIOPtrs.tpcZS->slice[iSlice].nZSPtr[j + 1][k]; l++) {
163 Vc::Common::prefetchMid(((const uint8_t*)mIOPtrs.tpcZS->slice[iSlice].zsPtr[j + 1][k]) + l * TPCZSHDR::TPC_ZS_PAGE_SIZE);
164 Vc::Common::prefetchMid(((const uint8_t*)mIOPtrs.tpcZS->slice[iSlice].zsPtr[j + 1][k]) + l * TPCZSHDR::TPC_ZS_PAGE_SIZE + sizeof(o2::header::RAWDataHeader));
165 }
166 }
167 }
168#endif
169
170 std::vector<std::pair<CfFragment, TPCCFDecodeScanTmp>> fragments;
171 fragments.reserve(mCFContext->nFragments);
172 fragments.emplace_back(std::pair<CfFragment, TPCCFDecodeScanTmp>{fragment, {0, 0, 0, 0, 0, -1}});
173 for (uint32_t i = 1; i < mCFContext->nFragments; i++) {
174 fragments.emplace_back(std::pair<CfFragment, TPCCFDecodeScanTmp>{fragments.back().first.next(), {0, 0, 0, 0, 0, -1}});
175 }
176 std::vector<bool> fragmentExtends(mCFContext->nFragments, false);
177
178 uint32_t firstPossibleFragment = 0;
179 uint32_t pageCounter = 0;
180 uint32_t emptyPages = 0;
181 for (uint32_t k = 0; k < mIOPtrs.tpcZS->slice[iSlice].count[j]; k++) {
182 if (GetProcessingSettings().tpcSingleSector != -1 && GetProcessingSettings().tpcSingleSector != (int32_t)iSlice) {
183 break;
184 }
185 nPages += mIOPtrs.tpcZS->slice[iSlice].nZSPtr[j][k];
186 for (uint32_t l = 0; l < mIOPtrs.tpcZS->slice[iSlice].nZSPtr[j][k]; l++) {
187#ifndef GPUCA_NO_VC
188 if (GetProcessingSettings().prefetchTPCpageScan >= 2 && l + 1 < mIOPtrs.tpcZS->slice[iSlice].nZSPtr[j][k]) {
189 Vc::Common::prefetchForOneRead(((const uint8_t*)mIOPtrs.tpcZS->slice[iSlice].zsPtr[j][k]) + (l + 1) * TPCZSHDR::TPC_ZS_PAGE_SIZE);
190 Vc::Common::prefetchForOneRead(((const uint8_t*)mIOPtrs.tpcZS->slice[iSlice].zsPtr[j][k]) + (l + 1) * TPCZSHDR::TPC_ZS_PAGE_SIZE + sizeof(o2::header::RAWDataHeader));
191 }
192#endif
193 const uint8_t* const page = ((const uint8_t*)mIOPtrs.tpcZS->slice[iSlice].zsPtr[j][k]) + l * TPCZSHDR::TPC_ZS_PAGE_SIZE;
194 const o2::header::RAWDataHeader* rdh = (const o2::header::RAWDataHeader*)page;
195 if (o2::raw::RDHUtils::getMemorySize(*rdh) == sizeof(o2::header::RAWDataHeader)) {
196 emptyPages++;
197 continue;
198 }
199 pageCounter++;
200 const TPCZSHDR* const hdr = (const TPCZSHDR*)(rdh_utils::getLink(o2::raw::RDHUtils::getFEEID(*rdh)) == rdh_utils::DLBZSLinkID ? (page + o2::raw::RDHUtils::getMemorySize(*rdh) - sizeof(TPCZSHDRV2)) : (page + sizeof(o2::header::RAWDataHeader)));
201 if (mCFContext->zsVersion == -1) {
202 mCFContext->zsVersion = hdr->version;
203 if (GetProcessingSettings().param.tpcTriggerHandling && mCFContext->zsVersion < ZSVersion::ZSVersionDenseLinkBased) { // TODO: Move tpcTriggerHandling to recoSteps bitmask
204 static bool errorShown = false;
205 if (errorShown == false) {
206 GPUAlarm("Trigger handling only possible with TPC Dense Link Based data, received version %d, disabling", mCFContext->zsVersion);
207 }
208 errorShown = true;
209 }
210 } else if (mCFContext->zsVersion != (int32_t)hdr->version) {
211 GPUError("Received TPC ZS 8kb page of mixed versions, expected %d, received %d (linkid %d, feeCRU %d, feeEndpoint %d, feelinkid %d)", mCFContext->zsVersion, (int32_t)hdr->version, (int32_t)o2::raw::RDHUtils::getLinkID(*rdh), (int32_t)rdh_utils::getCRU(*rdh), (int32_t)rdh_utils::getEndPoint(*rdh), (int32_t)rdh_utils::getLink(*rdh));
212 constexpr size_t bufferSize = 3 * std::max(sizeof(*rdh), sizeof(*hdr)) + 1;
213 char dumpBuffer[bufferSize];
214 for (size_t i = 0; i < sizeof(*rdh); i++) {
215 // "%02X " guaranteed to be 3 chars + ending 0.
216 snprintf(dumpBuffer + 3 * i, 4, "%02X ", (int32_t)((uint8_t*)rdh)[i]);
217 }
218 GPUAlarm("RDH of page: %s", dumpBuffer);
219 for (size_t i = 0; i < sizeof(*hdr); i++) {
220 // "%02X " guaranteed to be 3 chars + ending 0.
221 snprintf(dumpBuffer + 3 * i, 4, "%02X ", (int32_t)((uint8_t*)hdr)[i]);
222 }
223 GPUAlarm("Metainfo of page: %s", dumpBuffer);
224 if (GetProcessingSettings().ignoreNonFatalGPUErrors) {
225 mCFContext->abandonTimeframe = true;
226 return {0, 0};
227 } else {
228 GPUFatal("Cannot process with invalid TPC ZS data, exiting");
229 }
230 }
231 if (GetProcessingSettings().param.tpcTriggerHandling) {
232 const TPCZSHDRV2* const hdr2 = (const TPCZSHDRV2*)hdr;
233 if (hdr2->flags & TPCZSHDRV2::ZSFlags::TriggerWordPresent) {
234 const char* triggerWord = (const char*)hdr - TPCZSHDRV2::TRIGGER_WORD_SIZE;
235 o2::tpc::TriggerInfoDLBZS tmp;
236 memcpy((void*)&tmp.triggerWord, triggerWord, TPCZSHDRV2::TRIGGER_WORD_SIZE);
237 tmp.orbit = o2::raw::RDHUtils::getHeartBeatOrbit(*rdh);
238 if (tmp.triggerWord.isValid(0)) {
239 mTriggerBuffer->triggers.emplace(tmp);
240 }
241 }
242 }
243 nDigits += hdr->nADCsamples;
244 endpointAdcSamples[j] += hdr->nADCsamples;
245 uint32_t timeBin = (hdr->timeOffset + (o2::raw::RDHUtils::getHeartBeatOrbit(*rdh) - firstHBF) * o2::constants::lhc::LHCMaxBunches) / LHCBCPERTIMEBIN;
246 uint32_t maxTimeBin = timeBin + hdr->nTimeBinSpan;
247 if (mCFContext->zsVersion >= ZSVersion::ZSVersionDenseLinkBased) {
248 const TPCZSHDRV2* const hdr2 = (const TPCZSHDRV2*)hdr;
249 if (hdr2->flags & TPCZSHDRV2::ZSFlags::nTimeBinSpanBit8) {
250 maxTimeBin += 256;
251 }
252 }
253 if (maxTimeBin > mCFContext->tpcMaxTimeBin) {
254 mCFContext->tpcMaxTimeBin = maxTimeBin;
255 }
256 bool extendsInNextPage = false;
257 if (mCFContext->zsVersion >= ZSVersion::ZSVersionDenseLinkBased) {
258 if (l + 1 < mIOPtrs.tpcZS->slice[iSlice].nZSPtr[j][k] && o2::raw::RDHUtils::getMemorySize(*rdh) == TPCZSHDR::TPC_ZS_PAGE_SIZE) {
259 const o2::header::RAWDataHeader* nextrdh = (const o2::header::RAWDataHeader*)(page + TPCZSHDR::TPC_ZS_PAGE_SIZE);
260 extendsInNextPage = o2::raw::RDHUtils::getHeartBeatOrbit(*nextrdh) == o2::raw::RDHUtils::getHeartBeatOrbit(*rdh) && o2::raw::RDHUtils::getMemorySize(*nextrdh) > sizeof(o2::header::RAWDataHeader);
261 }
262 }
263 while (firstPossibleFragment && (uint32_t)fragments[firstPossibleFragment - 1].first.last() > timeBin) {
264 firstPossibleFragment--;
265 }
266 auto handleExtends = [&](uint32_t ff) {
267 if (fragmentExtends[ff]) {
268 if (doGPU) {
269 // Only add extended page on GPU. On CPU the pages are in consecutive memory anyway.
270 // Not adding the page prevents an issue where a page is decoded twice on CPU, when only the extend should be decoded.
271 fragments[ff].second.zsPageLast++;
272 mCFContext->fragmentData[ff].nPages[iSlice][j]++;
273 mCFContext->fragmentData[ff].pageDigits[iSlice][j].emplace_back(0);
274 }
275 fragmentExtends[ff] = false;
276 }
277 };
278 if (mCFContext->zsVersion >= ZSVersion::ZSVersionDenseLinkBased) {
279 for (uint32_t ff = 0; ff < firstPossibleFragment; ff++) {
280 handleExtends(ff);
281 }
282 }
283 for (uint32_t f = firstPossibleFragment; f < mCFContext->nFragments; f++) {
284 if (timeBin < (uint32_t)fragments[f].first.last() && (uint32_t)fragments[f].first.first() <= maxTimeBin) {
285 if (!fragments[f].second.hasData) {
286 fragments[f].second.hasData = 1;
287 fragments[f].second.zsPtrFirst = k;
288 fragments[f].second.zsPageFirst = l;
289 } else {
290 if (pageCounter > (uint32_t)fragments[f].second.pageCounter + 1) {
291 mCFContext->fragmentData[f].nPages[iSlice][j] += emptyPages + pageCounter - fragments[f].second.pageCounter - 1;
292 for (uint32_t k2 = fragments[f].second.zsPtrLast - 1; k2 <= k; k2++) {
293 for (uint32_t l2 = ((int32_t)k2 == fragments[f].second.zsPtrLast - 1) ? fragments[f].second.zsPageLast : 0; l2 < (k2 < k ? mIOPtrs.tpcZS->slice[iSlice].nZSPtr[j][k2] : l); l2++) {
294 if (doGPU) {
295 mCFContext->fragmentData[f].pageDigits[iSlice][j].emplace_back(0);
296 } else {
297 // CPU cannot skip unneeded pages, so we must keep space to store the invalid dummy clusters
298 const uint8_t* const pageTmp = ((const uint8_t*)mIOPtrs.tpcZS->slice[iSlice].zsPtr[j][k2]) + l2 * TPCZSHDR::TPC_ZS_PAGE_SIZE;
299 const o2::header::RAWDataHeader* rdhTmp = (const o2::header::RAWDataHeader*)pageTmp;
300 if (o2::raw::RDHUtils::getMemorySize(*rdhTmp) != sizeof(o2::header::RAWDataHeader)) {
301 const TPCZSHDR* const hdrTmp = (const TPCZSHDR*)(rdh_utils::getLink(o2::raw::RDHUtils::getFEEID(*rdhTmp)) == rdh_utils::DLBZSLinkID ? (pageTmp + o2::raw::RDHUtils::getMemorySize(*rdhTmp) - sizeof(TPCZSHDRV2)) : (pageTmp + sizeof(o2::header::RAWDataHeader)));
302 mCFContext->fragmentData[f].nDigits[iSlice][j] += hdrTmp->nADCsamples;
303 }
304 }
305 }
306 }
307 } else if (emptyPages) {
308 mCFContext->fragmentData[f].nPages[iSlice][j] += emptyPages;
309 if (doGPU) {
310 for (uint32_t m = 0; m < emptyPages; m++) {
311 mCFContext->fragmentData[f].pageDigits[iSlice][j].emplace_back(0);
312 }
313 }
314 }
315 }
316 fragments[f].second.zsPtrLast = k + 1;
317 fragments[f].second.zsPageLast = l + 1;
318 fragments[f].second.pageCounter = pageCounter;
319 mCFContext->fragmentData[f].nPages[iSlice][j]++;
320 mCFContext->fragmentData[f].nDigits[iSlice][j] += hdr->nADCsamples;
321 if (doGPU) {
322 mCFContext->fragmentData[f].pageDigits[iSlice][j].emplace_back(hdr->nADCsamples);
323 }
324 fragmentExtends[f] = extendsInNextPage;
325 } else {
326 handleExtends(f);
327 if (timeBin < (uint32_t)fragments[f].first.last()) {
328 if (mCFContext->zsVersion >= ZSVersion::ZSVersionDenseLinkBased) {
329 for (uint32_t ff = f + 1; ff < mCFContext->nFragments; ff++) {
330 handleExtends(ff);
331 }
332 }
333 break;
334 } else {
335 firstPossibleFragment = f + 1;
336 }
337 }
338 }
339 emptyPages = 0;
340 }
341 }
342 for (uint32_t f = 0; f < mCFContext->nFragments; f++) {
343 mCFContext->fragmentData[f].minMaxCN[iSlice][j].zsPtrLast = fragments[f].second.zsPtrLast;
344 mCFContext->fragmentData[f].minMaxCN[iSlice][j].zsPtrFirst = fragments[f].second.zsPtrFirst;
345 mCFContext->fragmentData[f].minMaxCN[iSlice][j].zsPageLast = fragments[f].second.zsPageLast;
346 mCFContext->fragmentData[f].minMaxCN[iSlice][j].zsPageFirst = fragments[f].second.zsPageFirst;
347 }
348 }
349 mCFContext->nPagesTotal += nPages;
350 mCFContext->nPagesSector[iSlice] = nPages;
351
352 mCFContext->nDigitsEndpointMax[iSlice] = 0;
353 for (uint32_t i = 0; i < GPUTrackingInOutZS::NENDPOINTS; i++) {
354 if (endpointAdcSamples[i] > mCFContext->nDigitsEndpointMax[iSlice]) {
355 mCFContext->nDigitsEndpointMax[iSlice] = endpointAdcSamples[i];
356 }
357 }
358 uint32_t nDigitsFragmentMax = 0;
359 for (uint32_t i = 0; i < mCFContext->nFragments; i++) {
360 uint32_t pagesInFragment = 0;
361 uint32_t digitsInFragment = 0;
362 for (uint16_t j = 0; j < GPUTrackingInOutZS::NENDPOINTS; j++) {
363 pagesInFragment += mCFContext->fragmentData[i].nPages[iSlice][j];
364 digitsInFragment += mCFContext->fragmentData[i].nDigits[iSlice][j];
365 }
366 mCFContext->nPagesFragmentMax = std::max(mCFContext->nPagesFragmentMax, pagesInFragment);
367 nDigitsFragmentMax = std::max(nDigitsFragmentMax, digitsInFragment);
368 }
369 mRec->getGeneralStepTimer(GeneralStep::Prepare).Stop();
370 return {nDigits, nDigitsFragmentMax};
371}

373void GPUChainTracking::RunTPCClusterizer_compactPeaks(GPUTPCClusterFinder& clusterer, GPUTPCClusterFinder& clustererShadow, int32_t stage, bool doGPU, int32_t lane)
374{
375 auto& in = stage ? clustererShadow.mPpeakPositions : clustererShadow.mPpositions;
376 auto& out = stage ? clustererShadow.mPfilteredPeakPositions : clustererShadow.mPpeakPositions;
377 if (doGPU) {
378 const uint32_t iSlice = clusterer.mISlice;
379 auto& count = stage ? clusterer.mPmemory->counters.nPeaks : clusterer.mPmemory->counters.nPositions;
380
381 std::vector<size_t> counts;
382
383 uint32_t nSteps = clusterer.getNSteps(count);
384 if (nSteps > clusterer.mNBufs) {
385 GPUError("Clusterer buffers exceeded (%u > %u)", nSteps, (int32_t)clusterer.mNBufs);
386 exit(1);
387 }
388
389 size_t tmpCount = count;
390 if (nSteps > 1) {
391 for (uint32_t i = 1; i < nSteps; i++) {
392 counts.push_back(tmpCount);
393 if (i == 1) {
394 runKernel<GPUTPCCFStreamCompaction, GPUTPCCFStreamCompaction::scanStart>({GetGrid(tmpCount, clusterer.mScanWorkGroupSize, lane), {iSlice}}, i, stage);
395 } else {
396 runKernel<GPUTPCCFStreamCompaction, GPUTPCCFStreamCompaction::scanUp>({GetGrid(tmpCount, clusterer.mScanWorkGroupSize, lane), {iSlice}}, i, tmpCount);
397 }
398 tmpCount = (tmpCount + clusterer.mScanWorkGroupSize - 1) / clusterer.mScanWorkGroupSize;
399 }
400
401 runKernel<GPUTPCCFStreamCompaction, GPUTPCCFStreamCompaction::scanTop>({GetGrid(tmpCount, clusterer.mScanWorkGroupSize, lane), {iSlice}}, nSteps, tmpCount);
402
403 for (uint32_t i = nSteps - 1; i > 1; i--) {
404 tmpCount = counts[i - 1];
405 runKernel<GPUTPCCFStreamCompaction, GPUTPCCFStreamCompaction::scanDown>({GetGrid(tmpCount - clusterer.mScanWorkGroupSize, clusterer.mScanWorkGroupSize, lane), {iSlice}}, i, clusterer.mScanWorkGroupSize, tmpCount);
406 }
407 }
408
409 runKernel<GPUTPCCFStreamCompaction, GPUTPCCFStreamCompaction::compactDigits>({GetGrid(count, clusterer.mScanWorkGroupSize, lane), {iSlice}}, 1, stage, in, out);
410 } else {
411 auto& nOut = stage ? clusterer.mPmemory->counters.nClusters : clusterer.mPmemory->counters.nPeaks;
