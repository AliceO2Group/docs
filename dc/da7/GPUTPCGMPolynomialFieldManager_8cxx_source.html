186int32_t GPUTPCGMPolynomialFieldManager::GetPolynomialField(float nominalFieldkG, GPUTPCGMPolynomialField& field)
187{
188 //
189 // get closest pre-calculated polynomial field approximation of the TPC region for the given field value nominalFieldkG
190 // returns !=0 in case of error
191 //
192 // check which field map is in use
193
194 field.Reset();
195
196 StoredField_t type = kUnknown;
197
198 if (fabsf(nominalFieldkG) < gpu_common_constants::kZeroFieldCut) {
199 type = kUniform;
200 nominalFieldkG = 0;
201 } else if (fabsf(fabsf(nominalFieldkG) - 5.00668f) <= fabsf(fabsf(nominalFieldkG) - 2.f)) {
202 type = k5kG;
203 } else {
204 type = k2kG;
205 }
206
207 return GetPolynomialField(type, nominalFieldkG, field);
