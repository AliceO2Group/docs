24template <class T, int32_t I, typename... Args>
25inline void GPUReconstructionOCLBackend::runKernelBackendInternal(const krnlSetupTime& _xyz, const Args&... args)
26{
27 cl_kernel k = _xyz.y.num > 1 ? getKernelObject<cl_kernel, T, I, true>() : getKernelObject<cl_kernel, T, I, false>();
28 auto& x = _xyz.x;
29 auto& y = _xyz.y;
30 auto& z = _xyz.z;
31 if (y.num <= 1) {
32 GPUFailedMsg(OCLsetKernelParameters(k, mInternals->mem_gpu, mInternals->mem_constant, y.start, args...));
33 } else {
34 GPUFailedMsg(OCLsetKernelParameters(k, mInternals->mem_gpu, mInternals->mem_constant, y.start, y.num, args...));
35 }
36
37 cl_event ev;
38 cl_event* evr;
39 bool tmpEvent = false;
40 if (z.ev == nullptr && mProcessingSettings.deviceTimers && mProcessingSettings.debugLevel > 0) {
41 evr = &ev;
42 tmpEvent = true;
43 } else {
44 evr = (cl_event*)z.ev;
45 }
46 GPUFailedMsg(clExecuteKernelA(mInternals->command_queue[x.stream], k, x.nThreads, x.nThreads * x.nBlocks, evr, (cl_event*)z.evList, z.nEvents));
47 if (mProcessingSettings.deviceTimers && mProcessingSettings.debugLevel > 0) {
48 cl_ulong time_start, time_end;
49 GPUFailedMsg(clWaitForEvents(1, evr));
50 GPUFailedMsg(clGetEventProfilingInfo(*evr, CL_PROFILING_COMMAND_START, sizeof(time_start), &time_start, nullptr));
51 GPUFailedMsg(clGetEventProfilingInfo(*evr, CL_PROFILING_COMMAND_END, sizeof(time_end), &time_end, nullptr));
52 _xyz.t = (time_end - time_start) * 1.e-9f;
53 if (tmpEvent) {
54 GPUFailedMsg(clReleaseEvent(ev));
55 }
56 }
57}

65template <class T, int32_t I>
66inline uint32_t GPUReconstructionOCLBackend::FindKernel(int32_t num)
67{
68 std::string name(GetKernelName<T, I>());
69 if (num > 1) {
70 name += "_multi";
71 }
72
73 for (uint32_t k = 0; k < mInternals->kernels.size(); k++) {
74 if (mInternals->kernels[k].second == name) {
75 return (k);
76 }
77 }
78 GPUError("Could not find OpenCL kernel %s", name.c_str());
79 throw ::std::runtime_error("Requested unsupported OpenCL kernel");
80}

82template <class T, int32_t I>
83int32_t GPUReconstructionOCLBackend::AddKernel(bool multi)
84{
85 std::string name(GetKernelName<T, I>());
86 if (multi) {
87 name += "_multi";
88 }
89 std::string kname("krnl_" + name);
90
91 cl_int ocl_error;
92 cl_kernel krnl = clCreateKernel(mInternals->program, kname.c_str(), &ocl_error);
93 if (GPUFailedMsgI(ocl_error)) {
94 GPUError("Error creating OPENCL Kernel: %s", name.c_str());
95 return 1;
96 }
97 mInternals->kernels.emplace_back(krnl, name);
98 return 0;
99}

101template <class S, class T, int32_t I, bool MULTI>
102S& GPUReconstructionOCLBackend::getKernelObject()
