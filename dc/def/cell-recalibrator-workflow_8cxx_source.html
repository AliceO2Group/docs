47WorkflowSpec defineDataProcessing(ConfigContext const& cfgc)
48{
49 // Calibration types
50 auto disableBadchannels = cfgc.options().get<bool>("no-badchannelcalib"),
51 disableTime = cfgc.options().get<bool>("no-timecalib"),
52 disableEnergy = cfgc.options().get<bool>("no-gaincalib");
53
54 // subpsecs for input and output
55 auto inputsubspec = cfgc.options().get<uint32_t>("input-subspec"),
56 outputsubspec = cfgc.options().get<uint32_t>("output-subspec");
57
58 // LED event handling
59 auto dropled = cfgc.options().get<bool>("drop-led"),
60 redirectled = cfgc.options().get<bool>("redirect-led");
61
62 if (dropled && redirectled) {
63 LOG(fatal) << "Ambiguous handling of LED events";
64 }
65 uint32_t ledsetting = 0;
66 if (dropled) {
67 ledsetting = 1;
68 }
69 if (redirectled) {
70 ledsetting = 2;
71 }
72
73 o2::conf::ConfigurableParam::updateFromString(cfgc.options().get<std::string>("configKeyValues"));
74
75 WorkflowSpec specs;
76 specs.emplace_back(o2::emcal::getCellRecalibratorSpec(inputsubspec, outputsubspec, ledsetting, !disableBadchannels, !disableTime, !disableEnergy, cfgc.options().get<bool>("isMC")));
77
78 // configure dpl timer to inject correct firstTForbit: start from the 1st orbit of TF containing 1st sampled orbit
79 o2::raw::HBFUtilsInitializer hbfIni(cfgc, specs);
80 return specs;
81}
