51template <typename T, size_t Size>
52inline std::string to_string(gsl::span<T, Size> span)
53{
54 std::stringstream ss{};
55 for (size_t i = 0; auto v : span) {
56 ss << v;
57 if (++i != span.size()) {
58 ss << " ";
59 }
60 }
61 return std::move(ss).str();
62}
