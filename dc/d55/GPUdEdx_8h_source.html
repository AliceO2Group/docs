76GPUdi() void GPUdEdx::checkSubThresh(int32_t roc)
77{
78 if (roc != mLastROC) {
79 if (mNSubThresh && mCount + mNSubThresh <= MAX_NCL) {
80 for (int32_t i = 0; i < mNSubThresh; i++) {
81 mChargeTot[mCount] = (GPUCA_PAR_DEDX_STORAGE_TYPE_A)(mSubThreshMinTot * scalingFactor<GPUCA_PAR_DEDX_STORAGE_TYPE_A>::factor + scalingFactor<GPUCA_PAR_DEDX_STORAGE_TYPE_A>::round);
82 mChargeMax[mCount++] = (GPUCA_PAR_DEDX_STORAGE_TYPE_A)(mSubThreshMinMax * scalingFactor<GPUCA_PAR_DEDX_STORAGE_TYPE_A>::factor + scalingFactor<GPUCA_PAR_DEDX_STORAGE_TYPE_A>::round);
83 }
84 mNClsROC[mLastROC] += mNSubThresh;
85 mNClsROCSubThresh[mLastROC] += mNSubThresh;
86 }
87 mNSubThresh = 0;
88 mSubThreshMinTot = 1e10f;
89 mSubThreshMinMax = 1e10f;
90 }
91
92 mLastROC = roc;
93}

95GPUdnii() void GPUdEdx::fillCluster(float qtot, float qmax, int32_t padRow, uint8_t sector, float trackSnp, float trackTgl, const GPUCalibObjectsConst& calib, float z, float pad, float relTime)
96{
97 if (mCount >= MAX_NCL) {
98 return;
99 }
100
101 // container containing all the dE/dx corrections
102 auto calibContainer = calib.dEdxCalibContainer;
103 constexpr GPUTPCGeometry geo;
104
105 const int32_t roc = geo.GetROC(padRow);
106 checkSubThresh(roc);
107 float snp2 = trackSnp * trackSnp;
108 if (snp2 > GPUCA_MAX_SIN_PHI_LOW) {
109 snp2 = GPUCA_MAX_SIN_PHI_LOW;
110 }
111
112 // setting maximum for snp for which the calibration object was created
113 const float snp = CAMath::Abs(trackSnp);
114
115 // tanTheta local dip angle: z angle - dz/dx (cm/cm)
116 const float sec2 = 1.f / (1.f - snp2);
117 const float tgl2 = trackTgl * trackTgl;
118 const float tanTheta = CAMath::Sqrt(tgl2 * sec2);
119
120 // getting the topology correction
121 const uint32_t padPos = CAMath::Min<uint32_t>(GPUTPCGeometry::NPads(padRow) - 1, CAMath::Float2UIntRn(pad)); // position of the pad is shifted half a pad ( pad=3 -> centre position of third pad)
122 const float absRelPad = CAMath::Abs(pad - padPos);
123 const int32_t region = geo.GetRegion(padRow);
124 z = CAMath::Abs(z);
125 const float threshold = calibContainer->getZeroSupressionThreshold(sector, padRow, padPos); // TODO: Use the mean zero supresion threshold of all pads in the cluster?
126 const bool useFullGainMap = calibContainer->isUsageOfFullGainMap();
127 float qTotIn = qtot;
128 const float fullGainMapGain = calibContainer->getGain(sector, padRow, padPos);
129 if (useFullGainMap) {
130 qmax /= fullGainMapGain;
131 qtot /= fullGainMapGain;
132 } else {
133 qTotIn *= fullGainMapGain;
134 }
135
136 const float qMaxTopologyCorr = calibContainer->getTopologyCorrection(region, o2::tpc::ChargeType::Max, tanTheta, snp, z, absRelPad, relTime, threshold, qTotIn);
137 const float qTotTopologyCorr = calibContainer->getTopologyCorrection(region, o2::tpc::ChargeType::Tot, tanTheta, snp, z, absRelPad, relTime, threshold, qTotIn);
138 qmax /= qMaxTopologyCorr;
139 qtot /= qTotTopologyCorr;
140
141 tpc::StackID stack{
142 sector,
143 static_cast<tpc::GEMstack>(roc)};
144
145 const float qMaxResidualCorr = calibContainer->getResidualCorrection(stack, tpc::ChargeType::Max, trackTgl, trackSnp);
146 const float qTotResidualCorr = calibContainer->getResidualCorrection(stack, tpc::ChargeType::Tot, trackTgl, trackSnp);
147 qmax /= qMaxResidualCorr;
148 qtot /= qTotResidualCorr;
149
150 const float residualGainMapGain = calibContainer->getResidualGain(sector, padRow, padPos);
151 qmax /= residualGainMapGain;
152 qtot /= residualGainMapGain;
153
154 mChargeTot[mCount] = (GPUCA_PAR_DEDX_STORAGE_TYPE_A)(qtot * scalingFactor<GPUCA_PAR_DEDX_STORAGE_TYPE_A>::factor + scalingFactor<GPUCA_PAR_DEDX_STORAGE_TYPE_A>::round);
155 mChargeMax[mCount++] = (GPUCA_PAR_DEDX_STORAGE_TYPE_A)(qmax * scalingFactor<GPUCA_PAR_DEDX_STORAGE_TYPE_A>::factor + scalingFactor<GPUCA_PAR_DEDX_STORAGE_TYPE_A>::round);
156 mNClsROC[roc]++;
157 if (qtot < mSubThreshMinTot) {
158 mSubThreshMinTot = qtot;
159 }
160 if (qmax < mSubThreshMinMax) {
161 mSubThreshMinMax = qmax;
162 }
163
164 GPUCA_DEBUG_STREAMER_CHECK(if (o2::utils::DebugStreamer::checkStream(o2::utils::StreamFlags::streamdEdx)) {
165 float padlx = geo.Row2X(padRow);
166 float padly = geo.LinearPad2Y(sector, padRow, padPos);
167 o2::utils::DebugStreamer::instance()->getStreamer("debug_dedx", "UPDATE") << o2::utils::DebugStreamer::instance()->getUniqueTreeName("tree_dedx").data()
168 << "qTot=" << mChargeTot[mCount - 1]
169 << "qMax=" << mChargeMax[mCount - 1]
170 << "region=" << region
171 << "padRow=" << padRow
172 << "sector=" << sector
173 << "lx=" << padlx
174 << "ly=" << padly
175 << "tanTheta=" << tanTheta
176 << "trackTgl=" << trackTgl
177 << "sinPhi=" << trackSnp
178 << "z=" << z
179 << "absRelPad=" << absRelPad
180 << "relTime=" << relTime
181 << "threshold=" << threshold
182 << "qTotIn=" << qTotIn
183 << "qMaxTopologyCorr=" << qMaxTopologyCorr
184 << "qTotTopologyCorr=" << qTotTopologyCorr
185 << "qMaxResidualCorr=" << qMaxResidualCorr
186 << "qTotResidualCorr=" << qTotResidualCorr
187 << "residualGainMapGain=" << residualGainMapGain
188 << "fullGainMapGain=" << fullGainMapGain
189 << "\n";
190 })
191}
