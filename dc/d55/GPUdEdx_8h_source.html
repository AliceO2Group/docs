92GPUdi() void GPUdEdx::checkSubThresh(int32_t roc)
93{
94 if (roc != mLastROC) {
95 if (mNSubThresh && mCount + mNSubThresh <= MAX_NCL) {
96 for (int32_t i = 0; i < mNSubThresh; i++) {
97 mChargeTot[mCount] = (GPUCA_DEDX_STORAGE_TYPE)(mSubThreshMinTot * scalingFactor<GPUCA_DEDX_STORAGE_TYPE>::factor + scalingFactor<GPUCA_DEDX_STORAGE_TYPE>::round);
98 mChargeMax[mCount++] = (GPUCA_DEDX_STORAGE_TYPE)(mSubThreshMinMax * scalingFactor<GPUCA_DEDX_STORAGE_TYPE>::factor + scalingFactor<GPUCA_DEDX_STORAGE_TYPE>::round);
99 }
100 mNClsROC[mLastROC] += mNSubThresh;
101 mNClsROCSubThresh[mLastROC] += mNSubThresh;
102 }
103 mNSubThresh = 0;
104 mSubThreshMinTot = 1e10f;
105 mSubThreshMinMax = 1e10f;
106 }
107
108 mLastROC = roc;
109}

111GPUdnii() void GPUdEdx::fillCluster(float qtot, float qmax, int32_t padRow, uint8_t slice, float trackSnp, float trackTgl, const GPUParam& GPUrestrict() param, const GPUCalibObjectsConst& calib, float z, float pad, float relTime)
112{
113 if (mCount >= MAX_NCL) {
114 return;
115 }
116
117 // container containing all the dE/dx corrections
118 auto calibContainer = calib.dEdxCalibContainer;
119
120 const int32_t roc = param.tpcGeometry.GetROC(padRow);
121 checkSubThresh(roc);
122 float snp2 = trackSnp * trackSnp;
123 if (snp2 > GPUCA_MAX_SIN_PHI_LOW) {
124 snp2 = GPUCA_MAX_SIN_PHI_LOW;
125 }
126
127 // setting maximum for snp for which the calibration object was created
128 const float snp = CAMath::Abs(trackSnp);
129
130 // tanTheta local dip angle: z angle - dz/dx (cm/cm)
131 const float sec2 = 1.f / (1.f - snp2);
132 const float tgl2 = trackTgl * trackTgl;
133 const float tanTheta = CAMath::Sqrt(tgl2 * sec2);
134
135 // getting the topology correction
136 const uint32_t padPos = CAMath::Float2UIntRn(pad); // position of the pad is shifted half a pad ( pad=3 -> centre position of third pad)
137 const float absRelPad = CAMath::Abs(pad - padPos);
138 const int32_t region = param.tpcGeometry.GetRegion(padRow);
139 z = CAMath::Abs(z);
140 const float threshold = calibContainer->getZeroSupressionThreshold(slice, padRow, padPos); // TODO: Use the mean zero supresion threshold of all pads in the cluster?
141 const bool useFullGainMap = calibContainer->isUsageOfFullGainMap();
142 float qTotIn = qtot;
143 const float fullGainMapGain = calibContainer->getGain(slice, padRow, padPos);
144 if (useFullGainMap) {
145 qmax /= fullGainMapGain;
146 qtot /= fullGainMapGain;
147 } else {
148 qTotIn *= fullGainMapGain;
149 }
150
151 const float qMaxTopologyCorr = calibContainer->getTopologyCorrection(region, o2::tpc::ChargeType::Max, tanTheta, snp, z, absRelPad, relTime, threshold, qTotIn);
152 const float qTotTopologyCorr = calibContainer->getTopologyCorrection(region, o2::tpc::ChargeType::Tot, tanTheta, snp, z, absRelPad, relTime, threshold, qTotIn);
153 qmax /= qMaxTopologyCorr;
154 qtot /= qTotTopologyCorr;
155
156 tpc::StackID stack{
157 slice,
158 static_cast<tpc::GEMstack>(roc)};
159
160 const float qMaxResidualCorr = calibContainer->getResidualCorrection(stack, tpc::ChargeType::Max, trackTgl, trackSnp);
161 const float qTotResidualCorr = calibContainer->getResidualCorrection(stack, tpc::ChargeType::Tot, trackTgl, trackSnp);
162 qmax /= qMaxResidualCorr;
163 qtot /= qTotResidualCorr;
164
165 const float residualGainMapGain = calibContainer->getResidualGain(slice, padRow, padPos);
166 qmax /= residualGainMapGain;
167 qtot /= residualGainMapGain;
168
169 mChargeTot[mCount] = (GPUCA_DEDX_STORAGE_TYPE)(qtot * scalingFactor<GPUCA_DEDX_STORAGE_TYPE>::factor + scalingFactor<GPUCA_DEDX_STORAGE_TYPE>::round);
170 mChargeMax[mCount++] = (GPUCA_DEDX_STORAGE_TYPE)(qmax * scalingFactor<GPUCA_DEDX_STORAGE_TYPE>::factor + scalingFactor<GPUCA_DEDX_STORAGE_TYPE>::round);
171 mNClsROC[roc]++;
172 if (qtot < mSubThreshMinTot) {
173 mSubThreshMinTot = qtot;
174 }
175 if (qmax < mSubThreshMinMax) {
176 mSubThreshMinMax = qmax;
177 }
178
179 GPUCA_DEBUG_STREAMER_CHECK(if (o2::utils::DebugStreamer::checkStream(o2::utils::StreamFlags::streamdEdx)) {
180 float padlx = param.tpcGeometry.Row2X(padRow);
181 float padly = param.tpcGeometry.LinearPad2Y(slice, padRow, padPos);
182 o2::utils::DebugStreamer::instance()->getStreamer("debug_dedx", "UPDATE") << o2::utils::DebugStreamer::instance()->getUniqueTreeName("tree_dedx").data()
183 << "qTot=" << mChargeTot[mCount - 1]
184 << "qMax=" << mChargeMax[mCount - 1]
185 << "region=" << region
186 << "padRow=" << padRow
187 << "sector=" << slice
188 << "lx=" << padlx
189 << "ly=" << padly
190 << "tanTheta=" << tanTheta
191 << "trackTgl=" << trackTgl
192 << "sinPhi=" << trackSnp
193 << "z=" << z
194 << "absRelPad=" << absRelPad
195 << "relTime=" << relTime
196 << "threshold=" << threshold
197 << "qTotIn=" << qTotIn
198 << "qMaxTopologyCorr=" << qMaxTopologyCorr
199 << "qTotTopologyCorr=" << qTotTopologyCorr
200 << "qMaxResidualCorr=" << qMaxResidualCorr
201 << "qTotResidualCorr=" << qTotResidualCorr
202 << "residualGainMapGain=" << residualGainMapGain
203 << "fullGainMapGain=" << fullGainMapGain
204 << "\n";
205 })
206}
