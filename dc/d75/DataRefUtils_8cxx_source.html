27 void* extractFromTFile(TFile& file, TClass const* cl, const char* what)
28 {
29  if (!cl) {
30  return nullptr;
31  }
32  auto object = file.GetObjectChecked(what, cl);
33  if (!object) {
34  // it could be that object was stored with previous convention
35  // where the classname was taken as key
36  std::string objectName(cl->GetName());
37  objectName.erase(std::find_if(objectName.rbegin(), objectName.rend(), [](unsigned char ch) {
38  return !std::isspace(ch);
39  }).base(),
40  objectName.end());
41  objectName.erase(objectName.begin(), std::find_if(objectName.begin(), objectName.end(), [](unsigned char ch) {
42  return !std::isspace(ch);
43  }));
44 
45  object = file.GetObjectChecked(objectName.c_str(), cl);
46  LOG(warn) << "Did not find object under expected name " << what;
47  if (!object) {
48  return nullptr;
49  }
50  LOG(warn) << "Found object under deprecated name " << cl->GetName();
51  }
52  auto result = object;
53  // We need to handle some specific cases as ROOT ties them deeply
54  // to the file they are contained in
55  if (cl->InheritsFrom("TObject")) {
56  // make a clone
57  // detach from the file
58  auto tree = dynamic_cast<TTree*>((TObject*)object);
59  if (tree) {
60  tree->LoadBaskets(0x1L << 32); // make tree memory based
61  tree->SetDirectory(nullptr);
62  result = tree;
63  } else {
64  auto h = dynamic_cast<TH1*>((TObject*)object);
65  if (h) {
66  h->SetDirectory(nullptr);
67  result = h;
68  }
69  }
70  }
71  return result;
72 }

77 void* DataRefUtils::decodeCCDB(DataRef const& ref, std::type_info const& tinfo)
78 {
79  void* result = nullptr;
80  Int_t previousErrorLevel = gErrorIgnoreLevel;
81  gErrorIgnoreLevel = kFatal;
82  auto* dh = o2::header::get<o2::header::DataHeader*>(ref.header);
83  TMemFile memFile("name", const_cast<char*>(ref.payload), dh->payloadSize, "READ");
84  gErrorIgnoreLevel = previousErrorLevel;
85  if (memFile.IsZombie()) {
86  return nullptr;
87  }
88  TClass* tcl = TClass::GetClass(tinfo);
89  result = extractFromTFile(memFile, tcl, "ccdb_object");
90  if (!result) {
91  throw runtime_error_f("Couldn't retrieve object corresponding to %s from TFile", tcl->GetName());
92  }
93  memFile.Close();
94  return result;
95 }
