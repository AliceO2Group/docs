20ROFFilter createIRFrameFilter(gsl::span<const o2::dataformats::IRFrame> irframes)
21{
22 return [irframes](const ROFRecord& rof) {
23 InteractionRecord rofStart{rof.getBCData()};
24 InteractionRecord rofEnd = rofStart + rof.getBCWidth() - 1;
25 IRFrame ref(rofStart, rofEnd);
26 for (const auto& ir : irframes) {
27 if (ir.info > 0) {
28 auto overlap = ref.getOverlap(ir);
29 if (overlap.isValid()) {
30 return true;
31 }
32 }
33 }
34 return false;
35 };
36}
