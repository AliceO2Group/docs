22GeometryError_t getGeometryErrorFromErrorCode(unsigned int errorcode)
23{
24 assert(errorcode < getNumberOfGeometryErrorCodes());
25 switch (errorcode) {
26 case 0:
27 return GeometryError_t::CELL_RANGE_EXCEED;
28 case 1:
29 return GeometryError_t::CELL_INDEX_NEGATIVE;
30 default:
31 return GeometryError_t::UNKNOWN_ERROR;
32 }
33}
