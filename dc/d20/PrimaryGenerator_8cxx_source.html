79 Bool_t PrimaryGenerator::GenerateEvent(FairGenericStack* pStack)
80 {
84  if (!mEmbedTree) {
85  fixInteractionVertex(); // <-- always fixes vertex outside of FairROOT
86  auto ret = FairPrimaryGenerator::GenerateEvent(pStack);
87  if (ret) {
88  setGeneratorInformation();
89  }
90  return ret;
91  }
92 
96  mEmbedTree->GetEntry(mEmbedIndex);
97  setInteractionVertex(mEmbedEvent);
98 
100  auto genList = GetListOfGenerators();
101  for (int igen = 0; igen < genList->GetEntries(); ++igen) {
102  auto o2gen = dynamic_cast<Generator*>(genList->At(igen));
103  if (o2gen) {
104  o2gen->notifyEmbedding(mEmbedEvent);
105  }
106  }
107 
109  if (!FairPrimaryGenerator::GenerateEvent(pStack)) {
110  return kFALSE;
111  }
112 
114  auto o2event = dynamic_cast<MCEventHeader*>(fEvent);
115  if (o2event) {
116  o2event->setEmbeddingFileName(mEmbedFile->GetName());
117  o2event->setEmbeddingEventIndex(mEmbedIndex);
118  }
119  setGeneratorInformation();
120 
122  mEmbedIndex++;
123  mEmbedIndex %= mEmbedEntries;
124 
126  return kTRUE;
127 }

131 void PrimaryGenerator::AddTrack(Int_t pdgid, Double_t px, Double_t py, Double_t pz,
132  Double_t vx, Double_t vy, Double_t vz,
133  Int_t mother1, Int_t mother2,
134  Int_t daughter1, Int_t daughter2,
135  Bool_t wanttracking,
136  Double_t e, Double_t tof,
137  Double_t weight, TMCProcess proc, Int_t generatorStatus)
138 {
141  // check the status encoding
142  if (!mcgenstatus::isEncoded(generatorStatus) && proc == TMCProcess::kPPrimary) {
143  LOG(fatal) << "Generator status " << generatorStatus << " of particle is not encoded properly.";
144  }
145 
147  vx += fVertex.X();
148  vy += fVertex.Y();
149  vz += fVertex.Z();
150 
152  auto particlePDG = TDatabasePDG::Instance()->GetParticle(pdgid);
153  if (wanttracking && !particlePDG) {
154  LOG(warn) << "Particle to be tracked is not defined in PDG: pdg = " << pdgid << " (disabling tracking)";
155  wanttracking = false;
156  }
157 
159  Int_t doTracking = 0; // Go to tracking
160  if (fdoTracking && wanttracking) {
161  doTracking = 1;
162  }
163  Int_t dummyparent = -1; // Primary particle (now the value is -1 by default)
164  Double_t polx = 0.; // Polarisation
165  Double_t poly = 0.;
166  Double_t polz = 0.;
167  Int_t ntr = 0; // Track number; to be filled by the stack
168  Int_t status = generatorStatus; // Generation status
169 
170  // correct for tracks which are in list before generator is called
171  if (mother1 != -1) {
172  mother1 += fMCIndexOffset;
173  }
174  if (mother2 != -1) {
175  mother2 += fMCIndexOffset;
176  }
177  if (daughter1 != -1) {
178  daughter1 += fMCIndexOffset;
179  }
180  if (daughter2 != -1) {
181  daughter2 += fMCIndexOffset;
182  }
183 
192  if (abs(pdgid) == 311 && doTracking) {
193  LOG(warn) << "K0/antiK0 requested for tracking: converting into K0s/K0L";
194  pdgid = gRandom->Uniform() < 0.5 ? 310 : 130;
195  }
196 
198  if (e < 0) {
199  double mass = particlePDG ? particlePDG->Mass() : 0.;
200  e = std::sqrt(mass * mass + px * px + py * py + pz * pz);
201  }
202 
204  auto stack = dynamic_cast<o2::data::Stack*>(fStack);
205  if (!stack) {
206  LOG(fatal) << "Stack must be an o2::data:Stack";
207  return; // must be the o2 stack
208  }
209  stack->PushTrack(doTracking, mother1, pdgid, px, py, pz,
210  e, vx, vy, vz, tof, polx, poly, polz, TMCProcess::kPPrimary, ntr,
211  weight, status, mother2, daughter1, daughter2, proc);
212 
213  fNTracks++;
214 }

277 void PrimaryGenerator::fixInteractionVertex()
278 {
279  if (!mApplyVertex) {
280  SetBeam(0., 0., 0., 0.);
281  SetTarget(0., 0.);
282  return;
283  }
284 
285  // if someone gave vertex from outside; we will take it
286  if (mHaveExternalVertex) {
287  SetBeam(mExternalVertexX, mExternalVertexY, 0., 0.);
288  SetTarget(mExternalVertexZ, 0.);
289  mHaveExternalVertex = false; // the vertex is now consumed
290  return;
291  }
292 
293  // sampling a vertex and fixing for next event; no smearing will be done
294  // inside FairPrimaryGenerator;
295  SmearVertexXY(false);
296  SmearVertexZ(false);
297  SmearGausVertexXY(false);
298  SmearGausVertexZ(false);
299 
300  // we use the mMeanVertexObject if initialized (initialize first)
301  if (!mMeanVertex) {
302  if (mVertexMode == o2::conf::VertexMode::kDiamondParam) {
303  auto const& param = InteractionDiamondParam::Instance();
304  const auto& xyz = param.position;
305  const auto& sigma = param.width;
306  mMeanVertex = std::move(std::unique_ptr<o2::dataformats::MeanVertexObject>(new o2::dataformats::MeanVertexObject(xyz[0], xyz[1], xyz[2], sigma[0], sigma[1], sigma[2], param.slopeX, param.slopeY)));
307  }
308  if (mVertexMode == o2::conf::VertexMode::kNoVertex) {
309  mMeanVertex = std::move(std::unique_ptr<o2::dataformats::MeanVertexObject>(new o2::dataformats::MeanVertexObject(0, 0, 0, 0, 0, 0, 0, 0)));
310  }
311  if (mVertexMode == o2::conf::VertexMode::kCCDB) {
312  // fatal.. then the object should have been passed with setting
313  LOG(fatal) << "MeanVertexObject is null ... but mode is kCCDB. Please inject the valid CCDB object via setVertexMode";
314  }
315  }
316  auto sampledvertex = mMeanVertex->sample();
317 
318  if (PrimaryGeneratorParam::Instance().verbose) {
319  LOG(info) << "Sampled interacting vertex " << sampledvertex;
320  }
321  SetBeam(sampledvertex.X(), sampledvertex.Y(), 0., 0.);
322  SetTarget(sampledvertex.Z(), 0.);
323 }

327 Bool_t PrimaryGenerator::embedInto(TString fname)
328 {
332  if (mEmbedFile && mEmbedFile->IsOpen()) {
333  LOG(error) << "Another embedding file is currently open";
334  return kFALSE;
335  }
336 
338  mEmbedFile = TFile::Open(fname);
339  if (!mEmbedFile || !mEmbedFile->IsOpen()) {
340  LOG(error) << "Cannot open file for embedding: " << fname;
341  return kFALSE;
342  }
343 
345  mEmbedTree = (TTree*)mEmbedFile->Get("o2sim");
346  if (!mEmbedTree) {
347  LOG(error) << R"(Cannot find "o2sim" tree for embedding in )" << fname;
348  return kFALSE;
349  }
350 
352  mEmbedEntries = mEmbedTree->GetEntries();
353  if (mEmbedEntries <= 0) {
354  LOG(error) << "Invalid number of entries found in tree for embedding: " << mEmbedEntries;
355  return kFALSE;
356  }
357 
359  mEmbedEvent = new MCEventHeader;
360  mEmbedTree->SetBranchAddress("MCEventHeader.", &mEmbedEvent);
361 
363  return kTRUE;
364 }
