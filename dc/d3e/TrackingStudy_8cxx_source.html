141 void TrackingStudySpec::run(ProcessingContext& pc)
142 {
143  o2::globaltracking::RecoContainer recoData;
144  recoData.collectData(pc, *mDataRequest.get()); // select tracks of needed type, with minimal cuts, the real selected will be done in the vertexer
145  updateTimeDependentParams(pc); // Make sure this is called after recoData.collectData, which may load some conditions
146  if (recoData.inputsTPCclusters) {
147  mTPCRefitter = std::make_unique<o2::gpu::GPUO2InterfaceRefit>(&recoData.inputsTPCclusters->clusterIndex, &mTPCCorrMapsLoader, o2::base::Propagator::Instance()->getNominalBz(),
148  recoData.getTPCTracksClusterRefs().data(), 0, recoData.clusterShMapTPC.data(), recoData.occupancyMapTPC.data(),
149  recoData.occupancyMapTPC.size(), nullptr, o2::base::Propagator::Instance());
150  mTPCRefitter->setTrackReferenceX(900); // disable propagation after refit by setting reference to value > 500
151  mNTPCOccBinLength = mTPCRefitter->getParam()->rec.tpc.occupancyMapTimeBins;
152  mTBinClOccBef.clear();
153  mTBinClOccAft.clear();
154  }
155  // prepare TPC occupancy data
156  if (mNTPCOccBinLength > 1 && recoData.occupancyMapTPC.size()) {
157  mNTPCOccBinLengthInv = 1. / mNTPCOccBinLength;
158  int nTPCBins = mNHBPerTF * o2::constants::lhc::LHCMaxBunches / 8, ninteg = 0;
159  int nTPCOccBins = nTPCBins * mNTPCOccBinLengthInv, sumBins = std::max(1, int(o2::constants::lhc::LHCMaxBunches / 8 * mNTPCOccBinLengthInv));
160  mTBinClOccAft.resize(nTPCOccBins);
161  mTBinClOccBef.resize(nTPCOccBins);
162  std::vector<float> mltHistTB(nTPCOccBins);
163  float sm = 0., tb = 0.5 * mNTPCOccBinLength;
164  for (int i = 0; i < nTPCOccBins; i++) {
165  mltHistTB[i] = mTPCRefitter->getParam()->GetUnscaledMult(tb);
166  tb += mNTPCOccBinLength;
167  }
168  for (int i = nTPCOccBins; i--;) {
169  sm += mltHistTB[i];
170  if (i + sumBins < nTPCOccBins) {
171  sm -= mltHistTB[i + sumBins];
172  }
173  mTBinClOccAft[i] = sm;
174  }
175  sm = 0;
176  for (int i = 0; i < nTPCOccBins; i++) {
177  sm += mltHistTB[i];
178  if (i - sumBins > 0) {
179  sm -= mltHistTB[i - sumBins];
180  }
181  mTBinClOccBef[i] = sm;
182  }
183  } else {
184  mTBinClOccBef.resize(1);
185  mTBinClOccAft.resize(1);
186  }
187 
188  process(recoData);
189 }

191 void TrackingStudySpec::updateTimeDependentParams(ProcessingContext& pc)
192 {
193  o2::base::GRPGeomHelper::instance().checkUpdates(pc);
194  mTPCVDriftHelper.extractCCDBInputs(pc);
195  mTPCCorrMapsLoader.extractCCDBInputs(pc);
196  static bool initOnceDone = false;
197  if (!initOnceDone) { // this params need to be queried only once
198  initOnceDone = true;
199  // Note: reading of the ITS AlpideParam needed for ITS timing is done by the RecoContainer
200  auto grp = o2::base::GRPGeomHelper::instance().getGRPECS();
201  const auto& alpParams = o2::itsmft::DPLAlpideParam<o2::detectors::DetID::ITS>::Instance();
202  if (!grp->isDetContinuousReadOut(DetID::ITS)) {
203  mITSROFrameLengthMUS = alpParams.roFrameLengthTrig / 1.e3; // ITS ROFrame duration in \mus
204  } else {
205  mITSROFrameLengthMUS = alpParams.roFrameLengthInBC * o2::constants::lhc::LHCBunchSpacingNS * 1e-3; // ITS ROFrame duration in \mus
206  }
207  pc.inputs().get<o2::dataformats::MeanVertexObject*>("meanvtx");
208  mNHBPerTF = o2::base::GRPGeomHelper::instance().getGRPECS()->getNHBFPerTF();
209  auto& elParam = o2::tpc::ParameterElectronics::Instance();
210  mTPCTBinMUS = elParam.ZbinWidth; // TPC bin in microseconds
211  mTPCTBinMUSInv = 1. / mTPCTBinMUS;
212  }
213  bool updateMaps = false;
214  if (mTPCCorrMapsLoader.isUpdated()) {
215  mTPCCorrMapsLoader.acknowledgeUpdate();
216  updateMaps = true;
217  }
218  if (mTPCVDriftHelper.isUpdated()) {
219  LOGP(info, "Updating TPC fast transform map with new VDrift factor of {} wrt reference {} and DriftTimeOffset correction {} wrt {} from source {}",
220  mTPCVDriftHelper.getVDriftObject().corrFact, mTPCVDriftHelper.getVDriftObject().refVDrift,
221  mTPCVDriftHelper.getVDriftObject().timeOffsetCorr, mTPCVDriftHelper.getVDriftObject().refTimeOffset,
222  mTPCVDriftHelper.getSourceName());
223  mTPCVDriftHelper.acknowledgeUpdate();
224  updateMaps = true;
225  }
226  if (updateMaps) {
227  mTPCCorrMapsLoader.updateVDrift(mTPCVDriftHelper.getVDriftObject().corrFact, mTPCVDriftHelper.getVDriftObject().refVDrift, mTPCVDriftHelper.getVDriftObject().getTimeOffset());
228  }

231 void TrackingStudySpec::process(o2::globaltracking::RecoContainer& recoData)
232 {
233  auto pvvec = recoData.getPrimaryVertices();
234  auto trackIndex = recoData.getPrimaryVertexMatchedTracks(); // Global ID's for associated tracks
235  auto vtxRefs = recoData.getPrimaryVertexMatchedTrackRefs(); // references from vertex to these track IDs
236  auto prop = o2::base::Propagator::Instance();
237  auto FITInfo = recoData.getFT0RecPoints();
238  static int TFCount = 0;
239  int nv = vtxRefs.size();
240  o2::dataformats::PrimaryVertexExt pveDummy;
241  o2::dataformats::PrimaryVertexExt vtxDummy(mMeanVtx.getPos(), {}, {}, 0);
242  std::vector<o2::dataformats::PrimaryVertexExt> pveVec(nv);
243  pveVec.back() = vtxDummy;
244  const auto& alpParams = o2::itsmft::DPLAlpideParam<o2::detectors::DetID::ITS>::Instance();
245  float tBiasITS = alpParams.roFrameBiasInBC * o2::constants::lhc::LHCBunchSpacingMUS;
246  const o2::ft0::InteractionTag& ft0Params = o2::ft0::InteractionTag::Instance();
247  std::vector<o2::dataformats::TrackInfoExt> trcExtVec;
248  auto vdrit = mTPCVDriftHelper.getVDriftObject().getVDrift();
249  bool tpcTrackOK = recoData.isTrackSourceLoaded(GTrackID::TPC);
250 
251  auto fillTPCClInfo = [&recoData, this](const o2::tpc::TrackTPC& trc, o2::dataformats::TrackInfoExt& trExt, float timestampTB = -1e9) {
252  const auto clRefs = recoData.getTPCTracksClusterRefs();
253  const auto shMap = recoData.clusterShMapTPC.data();
254  if (recoData.inputsTPCclusters) {
255  uint8_t clSect = 0, clRow = 0, clRowP = -1;
256  uint32_t clIdx = 0;
257  for (int ic = 0; ic < trc.getNClusterReferences(); ic++) {
258  trc.getClusterReference(clRefs, ic, clSect, clRow, clIdx);
259  if (clRow != clRowP) {
260  trExt.rowCountTPC++;
261  clRowP = clRow;
262  }
263  if (shMap[clRefs[ic + trc.getClusterRef().getFirstEntry()]]) {
264  trExt.nClTPCShared++;
265  }
266  }
267  trc.getClusterReference(clRefs, trc.getNClusterReferences() - 1, clSect, clRow, clIdx);
268  trExt.rowMinTPC = clRow;
269  const auto& clus = recoData.inputsTPCclusters->clusterIndex.clusters[clSect][clRow][clIdx];
270  this->mTPCCorrMapsLoader.Transform(clSect, clRow, clus.getPad(), clus.getTime(), trExt.innerTPCPos0[0], trExt.innerTPCPos0[1], trExt.innerTPCPos0[2], trc.getTime0()); // nominal time of the track
271  if (timestampTB > -1e8) {
272  this->mTPCCorrMapsLoader.Transform(clSect, clRow, clus.getPad(), clus.getTime(), trExt.innerTPCPos[0], trExt.innerTPCPos[1], trExt.innerTPCPos[2], timestampTB); // time assigned from the global track track
273  } else {
274  trExt.innerTPCPos = trExt.innerTPCPos0;
275  }
276  trc.getClusterReference(clRefs, 0, clSect, clRow, clIdx);
277  trExt.rowMaxTPC = clRow;
278  }
279  };
280 
281  for (int iv = 0; iv < nv; iv++) {
282  LOGP(debug, "processing PV {} of {}", iv, nv);
283  const auto& vtref = vtxRefs[iv];
284  if (iv != nv - 1) {
285  auto& pve = pveVec[iv];
286  static_cast<o2::dataformats::PrimaryVertex&>(pve) = pvvec[iv];
287  // find best matching FT0 signal
288  float bestTimeDiff = 1000, bestTime = -999;
289  int bestFTID = -1;
290  if (mTracksSrc[GTrackID::FT0]) {
291  for (int ift0 = vtref.getFirstEntryOfSource(GTrackID::FT0); ift0 < vtref.getFirstEntryOfSource(GTrackID::FT0) + vtref.getEntriesOfSource(GTrackID::FT0); ift0++) {
292  const auto& ft0 = FITInfo[trackIndex[ift0]];
293  if (ft0Params.isSelected(ft0)) {
294  auto fitTime = ft0.getInteractionRecord().differenceInBCMUS(recoData.startIR);
295  if (std::abs(fitTime - pve.getTimeStamp().getTimeStamp()) < bestTimeDiff) {
296  bestTimeDiff = fitTime - pve.getTimeStamp().getTimeStamp();
297  bestFTID = trackIndex[ift0];
298  }
299  }
300  }
301  } else {
302  LOGP(warn, "FT0 is not requested, cannot set complete vertex info");
303  }
304  if (bestFTID >= 0) {
305  pve.FT0A = FITInfo[bestFTID].getTrigger().getAmplA();
306  pve.FT0C = FITInfo[bestFTID].getTrigger().getAmplC();
307  pve.FT0Time = double(FITInfo[bestFTID].getInteractionRecord().differenceInBCMUS(recoData.startIR)) + FITInfo[bestFTID].getCollisionTimeMean() * 1e-6; // time in \mus
308  }
309  pve.VtxID = iv;
310  }
311  trcExtVec.clear();
312  float q2ptITS, q2ptTPC, q2ptITSTPC, q2ptITSTPCTRD;
313  for (int is = 0; is < GTrackID::NSources; is++) {
314  DetID::mask_t dm = GTrackID::getSourceDetectorsMask(is);
315  bool skipTracks = !mTracksSrc[is] || !recoData.isTrackSourceLoaded(is) || !(dm[DetID::ITS] || dm[DetID::TPC]);
316  int idMin = vtref.getFirstEntryOfSource(is), idMax = idMin + vtref.getEntriesOfSource(is);
317  for (int i = idMin; i < idMax; i++) {
318  auto vid = trackIndex[i];
319  bool pvCont = vid.isPVContributor();
320  if (pvCont) {
321  pveVec[iv].nSrc[is]++;
322  }
323  if (skipTracks) {
324  continue;
325  }
326  GTrackID tpcTrID;
327  const o2::tpc::TrackTPC* tpcTr = nullptr;
328  int nclTPC = 0;
329  if (dm[DetID::TPC] && tpcTrackOK) {
330  tpcTrID = recoData.getTPCContributorGID(vid);
331  tpcTr = &recoData.getTPCTrack(tpcTrID);
332  nclTPC = tpcTr->getNClusters();
333  if (nclTPC < mMinTPCClusters) {
334  continue;
335  }
336  }
337  bool ambig = vid.isAmbiguous();
338  auto trc = recoData.getTrackParam(vid);
339  if (abs(trc.getEta()) > mMaxEta) {
340  continue;
341  }
342  if (iv < nv - 1 && is == GTrackID::TPC && tpcTr && !tpcTr->hasBothSidesClusters()) { // for unconstrained TPC tracks correct track Z
343  float corz = vdrit * (tpcTr->getTime0() * mTPCTBinMUS - pvvec[iv].getTimeStamp().getTimeStamp());
344  if (tpcTr->hasASideClustersOnly()) {
345  corz = -corz; // A-side
346  }
347  trc.setZ(trc.getZ() + corz);
348  }
349  float xmin = trc.getX();
350  o2::dataformats::DCA dca;
351  if (!prop->propagateToDCA(iv == nv - 1 ? vtxDummy : pvvec[iv], trc, prop->getNominalBz(), 2., o2::base::PropagatorF::MatCorrType::USEMatCorrLUT, &dca)) {
352  continue;
353  }
354  bool hasITS = GTrackID::getSourceDetectorsMask(is)[GTrackID::ITS];
355  if (std::abs(dca.getY()) > (hasITS ? getDCAYCut(trc.getPt()) : mTPCDCAYCut) ||
356  std::abs(dca.getZ()) > (hasITS ? getDCAZCut(trc.getPt()) : mTPCDCAZCut)) {
357  continue;
358  }
359  if (trc.getPt() < mMinPt) {
360  continue;
361  }
362  if (iv != nv - 1) {
363  pveVec[iv].nSrcA[is]++;
364  if (ambig) {
365  pveVec[iv].nSrcAU[is]++;
366  }
367  }
368  if (!hasITS && mStoreWithITSOnly) {
369  continue;
370  }
371  {
372  auto& trcExt = trcExtVec.emplace_back();
373  recoData.getTrackTime(vid, trcExt.ttime, trcExt.ttimeE);
374  trcExt.track = trc;
375  trcExt.dca = dca;
376  trcExt.gid = vid;
377  trcExt.xmin = xmin;
378  trcExt.dcaTPC.set(-999.f, -999.f);
379 
380  if (tpcTr) {
381  float tsuse = trcExt.ttime / (8 * o2::constants::lhc::LHCBunchSpacingMUS);
382  if (is == GTrackID::TPC) {
383  trcExt.dcaTPC = dca;
384  tsuse = -1e9;
385  } else {
386  o2::track::TrackParCov tmpTPC(*tpcTr);
387  if (iv < nv - 1 && is == GTrackID::TPC && tpcTr && !tpcTr->hasBothSidesClusters()) { // for unconstrained TPC tracks correct track Z
388  float corz = vdrit * (tpcTr->getTime0() * mTPCTBinMUS - pvvec[iv].getTimeStamp().getTimeStamp());
389  if (tpcTr->hasASideClustersOnly()) {
390  corz = -corz; // A-side
391  }
392  tmpTPC.setZ(tmpTPC.getZ() + corz);
393  }
394  if (!prop->propagateToDCA(iv == nv - 1 ? vtxDummy : pvvec[iv], tmpTPC, prop->getNominalBz(), 2., o2::base::PropagatorF::MatCorrType::USEMatCorrLUT, &trcExt.dcaTPC)) {
395  trcExt.dcaTPC.set(-999.f, -999.f);
396  }
397  }
398  fillTPCClInfo(*tpcTr, trcExt, tsuse);
399  }
400  auto gidRefs = recoData.getSingleDetectorRefs(vid);
401  if (gidRefs[GTrackID::ITS].isIndexSet()) {
402  const auto& itsTr = recoData.getITSTrack(gidRefs[GTrackID::ITS]);
403  trcExt.q2ptITS = itsTr.getQ2Pt();
404  trcExt.nClITS = itsTr.getNClusters();
405  for (int il = 0; il < 7; il++) {
406  if (itsTr.hasHitOnLayer(il)) {
407  trcExt.pattITS |= 0x1 << il;
408  }
409  }
410  } else if (gidRefs[GTrackID::ITSAB].isIndexSet()) {
411  const auto& itsTrf = recoData.getITSABRefs()[gidRefs[GTrackID::ITSAB]];
412  trcExt.nClITS = itsTrf.getNClusters();
413  for (int il = 0; il < 7; il++) {
414  if (itsTrf.hasHitOnLayer(il)) {
415  trcExt.pattITS |= 0x1 << il;
416  }
417  }
418  }
419  if (gidRefs[GTrackID::TPC].isIndexSet()) {
420  trcExt.q2ptTPC = recoData.getTrackParam(gidRefs[GTrackID::TPC]).getQ2Pt();
421  trcExt.nClTPC = nclTPC;
422  }
423  if (gidRefs[GTrackID::ITSTPC].isIndexSet()) {
424  const auto& trTPCITS = recoData.getTPCITSTrack(gidRefs[GTrackID::ITSTPC]);
425  trcExt.q2ptITSTPC = trTPCITS.getQ2Pt();
426  trcExt.chi2ITSTPC = trTPCITS.getChi2Match();
427  }
428  if (gidRefs[GTrackID::TRD].isIndexSet()) {
429  trcExt.q2ptITSTPCTRD = recoData.getTrackParam(gidRefs[GTrackID::TRD]).getQ2Pt();
430  }
431  if (gidRefs[GTrackID::TOF].isIndexSet()) {
432  trcExt.infoTOF = recoData.getTOFMatch(vid);
433  }
434  }
435  }
436  }
437  float tpcOccBef = 0., tpcOccAft = 0.;
438  if (iv != nv - 1) {
439  int tb = pveVec[iv].getTimeStamp().getTimeStamp() * mTPCTBinMUSInv * mNTPCOccBinLengthInv;
440  tpcOccBef = tb < 0 ? mTBinClOccBef[0] : (tb >= mTBinClOccBef.size() ? mTBinClOccBef.back() : mTBinClOccBef[tb]);
441  tpcOccAft = tb < 0 ? mTBinClOccAft[0] : (tb >= mTBinClOccAft.size() ? mTBinClOccAft.back() : mTBinClOccAft[tb]);
442  }
443  (*mDBGOut) << "trpv"
444  << "orbit=" << recoData.startIR.orbit << "tfID=" << TFCount
445  << "tpcOccBef=" << tpcOccBef << "tpcOccAft=" << tpcOccAft
446  << "pve=" << pveVec[iv] << "trc=" << trcExtVec << "\n";
447  }
448 
449  int nvtot = mMaxNeighbours < 0 ? -1 : (int)pveVec.size();
450 
451  auto insSlot = [maxSlots = mMaxNeighbours](std::vector<float>& vc, float v, int slot, std::vector<int>& vid, int id) {
452  for (int i = maxSlots - 1; i > slot; i--) {
453  std::swap(vc[i], vc[i - 1]);
454  std::swap(vid[i], vid[i - 1]);
455  }
456  vc[slot] = v;
457  vid[slot] = id;
458  };
459 
460  for (int cnt = 0; cnt < nvtot; cnt++) {
461  const auto& pve = pveVec[cnt];
462  float tv = pve.getTimeStamp().getTimeStamp();
463  std::vector<o2::dataformats::PrimaryVertexExt> pveT(mMaxNeighbours); // neighbours in time
464  std::vector<o2::dataformats::PrimaryVertexExt> pveZ(mMaxNeighbours); // neighbours in Z
465  std::vector<int> idT(mMaxNeighbours), idZ(mMaxNeighbours);
466  std::vector<float> dT(mMaxNeighbours), dZ(mMaxNeighbours);
467  for (int i = 0; i < mMaxNeighbours; i++) {
468  idT[i] = idZ[i] = -1;
469  dT[i] = mMaxVTTimeDiff;
470  dZ[i] = 1e9;
471  }
472  int cntM = cnt - 1, cntP = cnt + 1;
473  for (; cntM >= 0; cntM--) { // backward
474  const auto& vt = pveVec[cntM];
475  auto dtime = std::abs(tv - vt.getTimeStamp().getTimeStamp());
476  if (dtime > mMaxVTTimeDiff) {
477  continue;
478  }
479  for (int i = 0; i < mMaxNeighbours; i++) {
480  if (dT[i] > dtime) {
481  insSlot(dT, dtime, i, idT, cntM);
482  break;
483  }
484  }
485  auto dz = std::abs(pve.getZ() - vt.getZ());
486  for (int i = 0; i < mMaxNeighbours; i++) {
487  if (dZ[i] > dz) {
488  insSlot(dZ, dz, i, idZ, cntM);
489  break;
490  }
491  }
492  }
493  for (; cntP < nvtot; cntP++) { // forward
494  const auto& vt = pveVec[cntP];
495  auto dtime = std::abs(tv - vt.getTimeStamp().getTimeStamp());
496  if (dtime > mMaxVTTimeDiff) {
497  continue;
498  }
499  for (int i = 0; i < mMaxNeighbours; i++) {
500  if (dT[i] > dtime) {
501  insSlot(dT, dtime, i, idT, cntP);
502  break;
503  }
504  }
505  auto dz = std::abs(pve.getZ() - vt.getZ());
506  for (int i = 0; i < mMaxNeighbours; i++) {
507  if (dZ[i] > dz) {
508  insSlot(dZ, dz, i, idZ, cntP);
509  break;
510  }
511  }
512  }
513  for (int i = 0; i < mMaxNeighbours; i++) {
514  if (idT[i] != -1) {
515  pveT[i] = pveVec[idT[i]];
516  } else {
517  break;
518  }
519  }
520  for (int i = 0; i < mMaxNeighbours; i++) {
521  if (idZ[i] != -1) {
522  pveZ[i] = pveVec[idZ[i]];
523  } else {
524  break;
525  }
526  }
527  (*mDBGOutVtx) << "pvExt"
528  << "pve=" << pve
529  << "pveT=" << pveT
530  << "pveZ=" << pveZ
531  << "tfID=" << TFCount
532  << "\n";
533  }
534 
535  TFCount++;
536 }
