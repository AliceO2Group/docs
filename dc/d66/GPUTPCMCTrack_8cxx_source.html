29GPUTPCMCTrack::GPUTPCMCTrack(const TParticle* part) : fPDG(0), fP(0), fPt(0), mNHits(0), fNMCPoints(0), fFirstMCPointID(0), fNReconstructed(0), fSet(0), fNTurns(0)
30{
31 //* Constructor from TParticle
32
33 for (int32_t i = 0; i < 7; i++) {
34 fPar[i] = 0;
35 }
36 for (int32_t i = 0; i < 7; i++) {
37 fTPCPar[i] = 0;
38 }
39 fP = 0;
40 fPt = 0;
41
42 if (!part) {
43 return;
44 }
45 TLorentzVector mom, vtx;
46 part->ProductionVertex(vtx);
47 part->Momentum(mom);
48 fPar[0] = part->Vx();
49 fPar[1] = part->Vy();
50 fPar[2] = part->Vz();
51 fP = part->P();
52 fPt = part->Pt();
53 double pi = (fP > 1.e-4) ? 1. / fP : 0;
54 fPar[3] = part->Px() * pi;
55 fPar[4] = part->Py() * pi;
56 fPar[5] = part->Pz() * pi;
57 fPar[6] = 0;
58 fPDG = part->GetPdgCode();
59 if (CAMath::Abs(fPDG) < 100000) {
60 TParticlePDG* pPDG = TDatabasePDG::Instance()->GetParticle(fPDG);
61 if (pPDG) {
62 fPar[6] = pPDG->Charge() / 3.0 * pi;
63 }
64 }
65}

68{
69 //* Set parameters at TPC entrance
70
71 for (int32_t i = 0; i < 7; i++) {

76 fTPCPar[1] = Y;
77 fTPCPar[2] = Z;
78 double p = CAMath::Sqrt(Px * Px + Py * Py + Pz * Pz);
79 double pi = (p > 1.e-4) ? 1. / p : 0;
80 fTPCPar[3] = Px * pi;
81 fTPCPar[4] = Py * pi;
