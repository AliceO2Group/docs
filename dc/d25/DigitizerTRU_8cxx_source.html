43 void DigitizerTRU::init()
44 {
45  setPatches();
46  auto mSimParam = &(o2::emcal::SimParam::Instance());
47  // mRandomGenerator = new TRandom3(std::chrono::high_resolution_clock::now().time_since_epoch().count());
48 
49  float tau = mSimParam->getTimeResponseTauTRU();
50  float N = mSimParam->getTimeResponsePowerTRU();
51 
52  mSmearEnergy = mSimParam->doSmearEnergy();
53  mSimulateTimeResponse = mSimParam->doSimulateTimeResponse();
54 
55  mDigits.init();
56  mDigits.reserve(15);
57 
58  LZERO.setGeometry(mGeometry);
59  LZERO.init();
60 
61  // Parameters from data (@Martin Poghosyan)
62  // tau = 61.45 / 25.; // 61.45 ns, according to the fact that the
63  // N = 2.;
64 
65  if (mSimulateTimeResponse) {
66  // for each phase create a template distribution
67  TF1 RawResponse("RawResponse", rawResponseFunction, 0, 256, 5);
68  RawResponse.SetParameters(1., 0., tau, N, 0.);
69  RawResponse.SetParameter(1, 425. / o2::emcal::constants::EMCAL_TIMESAMPLE);
70 
71  // parameter 1: Handling phase + delay
72  // phase: 25 ns * phase index (-4)
73  // delay: Average signal delay
74  RawResponse.SetParameter(1, mSimParam->getSignalDelay() / constants::EMCAL_TIMESAMPLE);
75  for (int sample = 0; sample < constants::EMCAL_MAXTIMEBINS; sample++) {
76  mAmplitudeInTimeBins[sample] = RawResponse.Eval(sample);
77  LOG(info) << "DIG TRU init in DigitizerTRU: amplitudes[" << sample << "] = " << mAmplitudeInTimeBins[sample];
78  }
79 
80  // Rescale the TRU time response function
81  // so that the sum of the samples at [5], [6], [7] and [8]
82  // is equal to 1
83  auto maxElement = std::max_element(mAmplitudeInTimeBins.begin(), mAmplitudeInTimeBins.end());
84  double rescalingFactor = *(maxElement - 1) + *maxElement + *(maxElement + 1) + *(maxElement + 2);
85  double normalisationTRU = mSimParam->getTimeResponseNormalisationTRU();
86  rescalingFactor /= normalisationTRU;
87  // rescalingFactor /= (5. / 3.93); // the slope seen in the single fastOr correlation
88  // rescalingFactor /= (45. / 39.25); // the slope seen in the single fastOr correlation
89  for (int sample = 0; sample < constants::EMCAL_MAXTIMEBINS; sample++) {
90  mAmplitudeInTimeBins[sample] /= rescalingFactor;
91  LOG(info) << "DIG TRU init in DigitizerTRU after RESCALING: amplitudes[" << sample << "] = " << mAmplitudeInTimeBins[sample];
92  }
93  } else {
94  }
95 
96  if (mEnableDebugStreaming) {
97  mDebugStream = std::make_unique<o2::utils::TreeStreamRedirector>("emcaldigitsDebugTRU.root", "RECREATE");
98  // mDebugStreamPatch = std::make_unique<o2::utils::TreeStreamRedirector>("emcaldigitsDebugPatchTRU.root", "RECREATE");
99  }
100 }

102 double DigitizerTRU::rawResponseFunction(double* x, double* par)
103 {
104  double signal = 0.;
105  double tau = par[2];
106  double n = par[3];
107  double ped = par[4];
108  double xx = (x[0] - par[1] + tau) / tau;
109 
110  // par[0] amp, par[1] peak time
111 
112  if (xx <= 0) {
113  signal = ped;
114  } else {
115  signal = ped + par[0] * std::pow(xx, n) * std::exp(n * (1 - xx));
116  }
117 
118  return signal;
119 }

148 std::vector<std::tuple<int, Digit>> DigitizerTRU::makeAnaloguesFastorSums(const gsl::span<const Digit> sdigits)
149 {
150  TriggerMappingV2 mTriggerMap(mGeometry);
151  std::unordered_map<int, Digit> sdigitsFastOR;
152  std::vector<int> fastorIndicesFound;
153  for (const auto& dig : sdigits) {
154  o2::emcal::TriggerMappingV2::IndexCell towerid = dig.getTower();
155  int fastorIndex = mTriggerMap.getAbsFastORIndexFromCellIndex(towerid);
156  auto whichTRU = std::get<0>(mTriggerMap.getTRUFromAbsFastORIndex(fastorIndex));
157  auto whichFastOrTRU = std::get<1>(mTriggerMap.getTRUFromAbsFastORIndex(fastorIndex));
158  auto found = sdigitsFastOR.find(fastorIndex);
159  if (found != sdigitsFastOR.end()) {
160  // sum energy
161  Digit digitToSum((found->second).getTower(), dig.getAmplitude(), (found->second).getTimeStamp());
162  (found->second) += digitToSum;
163  } else {
164  // create new digit
165  fastorIndicesFound.emplace_back(fastorIndex);
166  sdigitsFastOR.emplace(fastorIndex, dig);
167  }
168  }
169  // sort digits for output
170  std::sort(fastorIndicesFound.begin(), fastorIndicesFound.end(), std::less<>());
171 
172  for (auto& elem : sdigitsFastOR) {
173  auto dig = elem.second;
174  int fastorIndex = elem.first;
175  auto whichTRU = std::get<0>(mTriggerMap.getTRUFromAbsFastORIndex(fastorIndex));
176  auto whichFastOrTRU = std::get<1>(mTriggerMap.getTRUFromAbsFastORIndex(fastorIndex));
177  }
178 
179  // Setting them to be TRU digits
180  for (auto& elem : sdigitsFastOR) {
181  (elem.second).setTRU();
182  }
183 
184  for (auto& elem : sdigitsFastOR) {
185  auto dig = elem.second;
186  int fastorIndex = elem.first;
187  auto whichTRU = std::get<0>(mTriggerMap.getTRUFromAbsFastORIndex(fastorIndex));
188  auto whichFastOrTRU = std::get<1>(mTriggerMap.getTRUFromAbsFastORIndex(fastorIndex));
189  }
190 
191  std::vector<std::tuple<int, Digit>> outputFastorSDigits;
192  std::for_each(fastorIndicesFound.begin(), fastorIndicesFound.end(), [&outputFastorSDigits, &sdigitsFastOR](int fastorIndex) { outputFastorSDigits.emplace_back(fastorIndex, sdigitsFastOR[fastorIndex]); });
193  return outputFastorSDigits;
194 }

197 void DigitizerTRU::sampleSDigit(const Digit& sDigit)
198 {
199  mTempDigitVector.clear();
200  Int_t tower = sDigit.getTower();
201  Double_t energy = sDigit.getAmplitude();
202 
203  if (mSmearEnergy) {
204  energy = smearEnergy(energy);
205  }
206 
207  if (energy < __DBL_EPSILON__) {
208  return;
209  }
210 
211  Double_t energies[15];
212  if (mSimulateTimeResponse) {
213  for (int sample = 0; sample < mAmplitudeInTimeBins.size(); sample++) {
214 
215  double val = energy * (mAmplitudeInTimeBins[sample]);
216  energies[sample] = val;
217  double digitTime = mEventTimeOffset * constants::EMCAL_TIMESAMPLE;
218  Digit digit(tower, val, digitTime);
219  digit.setTRU();
220  mTempDigitVector.push_back(digit);
221  }
222 
223  } else {
224  Digit digit(tower, energy, smearTime(sDigit.getTimeStamp(), energy));
225  digit.setTRU();
226  mTempDigitVector.push_back(digit);
227  }
228 
229  if (mEnableDebugStreaming) {
230  double timeStamp = sDigit.getTimeStamp();
231  (*mDebugStream).GetFile()->cd();
232  (*mDebugStream) << "DigitsTimeSamples"
233  << "Tower=" << tower
234  << "Time=" << timeStamp
235  << "DigitEnergy=" << energy
236  << "Sample0=" << energies[0]
237  << "Sample1=" << energies[1]
238  << "Sample2=" << energies[2]
239  << "Sample3=" << energies[3]
240  << "Sample4=" << energies[4]
241  << "Sample5=" << energies[5]
242  << "Sample6=" << energies[6]
243  << "Sample7=" << energies[7]
244  << "Sample8=" << energies[8]
245  << "Sample9=" << energies[9]
246  << "Sample10=" << energies[10]
247  << "Sample11=" << energies[11]
248  << "Sample12=" << energies[12]
249  << "Sample13=" << energies[13]
250  << "Sample14=" << energies[14]
251  << "\n";
252  }
253 }

256 double DigitizerTRU::smearEnergy(double energy)
257 {
258  auto mSimParam = &(o2::emcal::SimParam::Instance());
259  Double_t fluct = (energy * mSimParam->getMeanPhotonElectron()) / mSimParam->getGainFluctuations();
260  TRandom3 mRandomGenerator(std::chrono::high_resolution_clock::now().time_since_epoch().count());
261  energy *= mRandomGenerator.Poisson(fluct) / fluct;
262  return energy;
263 }

265 double DigitizerTRU::smearTime(double time, double energy)
266 {
267  auto mSimParam = &(o2::emcal::SimParam::Instance());
268  TRandom3 mRandomGenerator(std::chrono::high_resolution_clock::now().time_since_epoch().count());
269  return mRandomGenerator.Gaus(time + mSimParam->getSignalDelay(), mSimParam->getTimeResolution(energy));
270 }
