296inline T& CDBInterface::getObjectFromCDB(std::string_view path)
297{
298 static auto& cdb = o2::ccdb::BasicCCDBManager::instance();
299 auto* object = cdb.get<T>(path.data());
300 if (!object) {
301 LOGP(fatal, "Could not get {} from cdb", path);
302 }
303 return *object;
304}
