154 void DCSConfigDevice::run(o2::framework::ProcessingContext& pc)
155 {
156  auto configBuff = pc.inputs().get<gsl::span<char>>("inputConfig");
157  auto configFileName = pc.inputs().get<std::string>("inputConfigFileName");
158  const auto creation = pc.services().get<o2::framework::TimingInfo>().creation;
159  LOG(info) << "received input file " << configFileName << " of size " << configBuff.size();
160 
161  // either we receive a run info file or an update of a FEE config tag
162  if (configFileName == RunInfoFileName) {
163  updateRunInfo(configBuff);
164  } else { // update tag
165  std::string objName = fs::path(configFileName).stem().c_str();
166  objName = objName.substr(3, objName.size());
167  // first buffer all the datao
168  auto& dataVec = mDataBuffer[objName];
169  auto& objTime = mDataBufferTimes[objName];
170  if (dataVec.size()) {
171  LOGP(warning, "Another object with name {} was already received before: {}, now: {}, old object will be overwritten", objName, objTime, creation);
172  dataVec.clear();
173  }
174  dataVec.insert(dataVec.begin(), configBuff.begin(), configBuff.end());
175  objTime = creation;
176 
177  if (objName == TagInfoName) {
178  updateTags(pc.outputs());
179  }
180  }

183 void DCSConfigDevice::updateRunInfo(gsl::span<const char> configBuff)
184 {
185  const std::string line(configBuff.data(), configBuff.size());
186  const auto data = Str::tokenize(line, ',');
187  const std::string_view runInfoConf("Run number;SOX;Tag number;Run type");
188  if (data.size() != 4) {
189  LOGP(error, "{} has wrong format: {}, expected: {}, not writing RunInformation to CCDB", RunInfoFileName, line, runInfoConf);
190  return;
191  }
192  std::map<std::string, std::string> md;
193  md[o2::base::NameConf::CCDBRunTag.data()] = data[0];
194  md["Tag"] = data[2];
195  md["RunType"] = data[3];
196  md[o2::ccdb::CcdbObjectInfo::AdjustableEOV] = "true";
197  char tempChar{static_cast<char>(std::stoi(md["Tag"]))};
198 
199  const long startValRCT = std::stol(data[1]);
200  const long endValRCT = startValRCT + 48l * 60l * 60l * 1000l;
201  if (!mDontWriteRunInfo) {
202  o2::ccdb::CcdbObjectInfo w(CDBTypeMap.at(CDBType::ConfigRunInfo), "", "", md, startValRCT, endValRCT);
203  mCCDBApi.storeAsBinaryFile(&tempChar, sizeof(tempChar), "tmp.dat", "char", CDBTypeMap.at(CDBType::ConfigRunInfo), md, startValRCT, endValRCT);
204  if (!mCCDBApi.isSnapshotMode()) {
205  o2::ccdb::adjustOverriddenEOV(mCCDBApi, w);
206  }
207  }
208 
209  std::string mdInfo = "[";
210  for (const auto& [key, val] : md) {
211  mdInfo += fmt::format("{} = {}, ", key, val);
212  }
213  mdInfo += "]";
214  LOGP(info, "Updated {} with {} for validity range {}, {}", CDBTypeMap.at(CDBType::ConfigRunInfo), mdInfo, startValRCT, endValRCT);
215 }

218 {
219  mCDBStorage.setReason(tagInfo.comment);
220  std::map<std::string, std::string> md = mCDBStorage.getMetaData();
221  o2::ccdb::CcdbObjectInfo w(false);
222  o2::calibration::Utils::prepareCCDBobjectInfo(mFEEConfig, w, CDBTypeMap.at(CDBType::ConfigFEE), md, tagInfo.tag, tagInfo.tag + 1);
223 
224  mFEEConfig.tag = FEEConfig::Tags(tagInfo.tag);
225  auto image = o2::ccdb::CcdbApi::createObjectImage(&mFEEConfig, &w);
226 
227  LOGP(info, "Sending object {} / {} of size {} bytes, valid for {} : {} ", w.getPath(), w.getFileName(), image->size(), w.getStartValidityTimestamp(), w.getEndValidityTimestamp());
228  output.snapshot(Output{CDBPayload, CDBDescMap.at(CDBType::ConfigFEE), 0}, *image.get());
229  output.snapshot(Output{CDBWrapper, CDBDescMap.at(CDBType::ConfigFEE), 0}, w);
230 
231  mFEEPadDataReceived.reset();
232  mCRUConfigReceived = false;
233 
234  if (mDumpToRootFile) {
235  writeRootFile(tagInfo.tag);
236  }
237 
238  mFEEConfig.clear();

241 void DCSConfigDevice::fillFEEPad(std::string_view configFileName, gsl::span<const char> configBuff, bool update)
242 {
243  auto& calPad = mFEEConfig.padMaps[configFileName.data()];
244  int nLines = 0;
245  if (configFileName == "ITfraction") {
246  nLines = cru_calib_helpers::fillCalPad<0>(calPad, configBuff);
247  mFEEPadDataReceived.set(0);
248  } else if (configFileName == "ITexpLambda") {
249  nLines = cru_calib_helpers::fillCalPad<0>(calPad, configBuff);
250  mFEEPadDataReceived.set(1);
251  } else if (configFileName == "ThresholdMap") {
252  nLines = cru_calib_helpers::fillCalPad<2>(calPad, configBuff);
253  mFEEPadDataReceived.set(2);
254  } else if (configFileName == "Pedestals") {
255  nLines = cru_calib_helpers::fillCalPad<2>(calPad, configBuff);
256  mFEEPadDataReceived.set(3);
257  } else if (configFileName == "CMkValues") {
258  nLines = cru_calib_helpers::fillCalPad<6>(calPad, configBuff);
259  mFEEPadDataReceived.set(4);
260  }
261 
262  if (!update && (nLines != NLinksTotal)) {
263  LOGP(error, "Full FEEConfig expected, but only {} / {} lines read for object {}", nLines, NLinksTotal, configFileName);
264  } else {
265  LOGP(info, "updating CalDet object {} for {} links", configFileName, nLines);
266  }
267 }

269 void DCSConfigDevice::fillCRUConfig(gsl::span<const char> configBuff, bool update)
270 {
271  membuf sbuf((char*)configBuff.data(), configBuff.size());
272  std::istream in(&sbuf);
273  std::string line;
274 
275  int nLines = 0;
276  while (std::getline(in, line)) {
277  const auto cru = std::stoi(line.substr(0, line.find_first_of(' ')));
278  if ((cru < 0) || (cru >= CRU::MaxCRU)) {
279  LOGP(error, "unexpected CRU number {} in line {}", cru, line);
280  continue;
281  }
282 
283  const std::string cruData(line.substr(line.find_first_of(' ') + 1, line.size()));
284 
285  auto& cruConfig = mFEEConfig.cruConfig[cru];
286  if (cruConfig.setValues(cruData)) {
287  ++nLines;
288  }
289  }
290 
291  if (!update && (nLines != CRU::MaxCRU)) {
292  LOGP(error, "Full FEEConfig expected, but only {} / {} lines read for CRUConfig", nLines, (int)CRU::MaxCRU);
293  } else {
294  LOGP(info, "updating CRUConfig for {} crus", nLines);
295  }
296  mCRUConfigReceived = true;
297 }

311 void DCSConfigDevice::updateTags(DataAllocator& outputs)
312 {
313  const auto tagInfos = readTagInfos(mDataBuffer.at(TagInfoName.data()));
314  std::string tagNames;
315 
316  for (const auto& tagInfo : tagInfos) {
317  if (tagNames.size()) {
318  tagNames += "_";
319  }
320  tagNames += std::to_string(tagInfo.tag);
321  LOGP(info, "Updating TPC FEE config for tag {}", tagInfo.tag);
322  const auto update = loadFEEConfig(tagInfo.tag);
323 
324  for (const auto& [objName, configBuff] : mDataBuffer) {
325  if (CalDetNames.find(objName) != std::string_view::npos) {
326  fillFEEPad(objName, configBuff, update);
327  } else if (objName == CRUConfigName) {
328  fillCRUConfig(configBuff, update);
329  }
330  }
331  if (!update && ((mFEEPadDataReceived.count() != NCalDets) || !mCRUConfigReceived)) {
332  std::string errorMessage;
333  if (mFEEPadDataReceived.count() != NCalDets) {
334  errorMessage = fmt::format("not all CalDet objects received: {} ({})", mFEEPadDataReceived.to_string(), CalDetNames);
335  }
336  if (!mCRUConfigReceived) {
337  if (errorMessage.size()) {
338  errorMessage += " and ";
339  }
340  errorMessage += "CRUConfig not reveived";
341  }
342 
343  LOGP(error, "Full FEEConfig expected, but {}", errorMessage);
344  }
345  updateCCDB(outputs, tagInfo);
346  }
347 
348  if (mDumpToTextFiles) {
349  for (const auto& [objName, configBuff] : mDataBuffer) {
350  dumpToTextFile(objName, configBuff, tagNames);
351  }
352  }
353 
354  mDataBuffer.clear();
355 }

366 DCSConfigDevice::TagInfos DCSConfigDevice::readTagInfos(gsl::span<const char> configBuff)
367 {
368 
369  membuf sbuf((char*)configBuff.data(), configBuff.size());
370  std::istream in(&sbuf);
371  std::string line;
372 
373  TagInfos infos;
374  while (std::getline(in, line)) {
375  const auto firstComma = line.find_first_of(',');
376  int tag = -1;
377  try {
378  tag = std::stoi(std::string(line.substr(0, firstComma)));
379  } catch (...) {
380  }
381  const std::string comment(line.substr(firstComma + 1, line.size()));
382  if ((tag < 0) || comment.empty()) {
383  LOGP(warning, "Ill formatted line in 'TPC{}.txt': {}, expecting <tag,comment>, skipping", TagInfoName, line);
384  continue;
385  }
386  if (tag == 0) {
387  LOGP(warning, "Tag '{}' not accepted, must be > 0, skipping", tag);
388  continue;
389  }
390  infos.emplace_back(tag, comment);
391  }
392 
393  return infos;
394 }

396 bool DCSConfigDevice::loadFEEConfig(int tag)
397 {
398  FEEConfig* feeConfig = nullptr;
399  if (mReadFromRootFile) {
400  // find last file of a tag in the local directory, assuming
401  std::vector<std::string> filenames;
402  for (const auto& entry : fs::directory_iterator("./")) {
403  const auto fileName = entry.path().filename().string();
404  // chek if root file and contains tag number
405  if (entry.is_regular_file() &&
406  fileName.find(".root") == (fileName.size() - 5) &&
407  fileName.find(fmt::format(".{}.", tag)) != std::string::npos) {
408  filenames.emplace_back(fileName);
409  }
410  }
411  if (filenames.size()) {
412  std::sort(filenames.begin(), filenames.end());
413  const auto& lastFile = filenames.back();
414  std::unique_ptr<TFile> inFile(TFile::Open(lastFile.data()));
415  if (inFile->IsOpen() && !inFile->IsZombie()) {
416  inFile->GetObject("FEEConfig", feeConfig);
417  }
418  if (feeConfig) {
419  LOGP(info, "Read FEEConfig from file {} for updating", lastFile);
420  }
421  }
422  } else {
423  std::map<std::string, std::string> headers;
424  feeConfig = mCCDBApi.retrieveFromTFileAny<FEEConfig>(CDBTypeMap.at(CDBType::ConfigFEE), std::map<std::string, std::string>(), tag, &headers);
425  if (feeConfig) {
426  LOGP(info, "Read FEEConfig from ccdb for updating: URL: {}, Validity: {} - {}, Last modfied: {}, ETag: {}", mCCDBApi.getURL(), headers["Valid-From"], headers["Valid-Until"], headers["Last-Modified"], headers["ETag"]);
427  }
428  }
429 
430  if (!feeConfig) {
431  LOGP(warning, "Could not retrieve FEE config for tag {}, a new entry will be created, full config should be sent", tag);
432  mFEEConfig.clear();
433  return false;
434  }
435  mFEEConfig = *feeConfig;
436  if (mReadFromRootFile) {
437  delete feeConfig;
438  }
439  return true;
440 }
