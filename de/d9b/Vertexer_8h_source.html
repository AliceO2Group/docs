123template <int NLayers>
124template <typename... T>
125float Vertexer<NLayers>::evaluateTask(void (Vertexer<NLayers>::*task)(T...), std::string_view taskName, int iteration, LogFunc& logger, T&&... args)
126{
127 float diff{0.f};
128
129 if constexpr (constants::DoTimeBenchmarks) {
130 auto start = std::chrono::high_resolution_clock::now();
131 (this->*task)(std::forward<T>(args)...);
132 auto end = std::chrono::high_resolution_clock::now();
133
134 std::chrono::duration<double, std::milli> diff_t{end - start};
135 diff = diff_t.count();
136
137 std::stringstream sstream;
138 if (taskName.empty()) {
139 sstream << diff << "\t";
140 } else {
141 sstream << std::setw(2) << " - " << taskName << " completed in: " << diff << " ms";
142 }
143 logger(sstream.str());
144
145 if (mVertParams[iteration].SaveTimeBenchmarks) {
146 std::string taskNameStr(taskName);
147 std::transform(taskNameStr.begin(), taskNameStr.end(), taskNameStr.begin(),
148 [](unsigned char c) { return std::tolower(c); });
149 std::replace(taskNameStr.begin(), taskNameStr.end(), ' ', '_');
150 if (std::ofstream file{"its_time_benchmarks.txt", std::ios::app}) {
151 file << "vtx:" << iteration << '\t' << taskNameStr << '\t' << diff << '\n';
152 }
153 addTimingStatCurStep(iteration, diff);
154 }
155 } else {
156 (this->*task)(std::forward<T>(args)...);
157 }
158
159 if (mVertParams[iteration].PrintMemory) {
160 LOGP(info, "iter:{}:{}: {}", iteration, StateNames[mCurStep], mMemoryPool->asString());
161 }
162
163 return diff;
164}
