54 void CalibdEdxCorrection::dumpToTree(const char* outFileName) const
55 {
56  o2::utils::TreeStreamRedirector pcstream(outFileName, "RECREATE");
57  pcstream.GetFile()->cd();
58 
59  for (int sector = 0; sector < 2 * SECTORSPERSIDE; ++sector) {
60  for (int roc = 0; roc < GEMSTACKSPERSECTOR; ++roc) {
61  tpc::StackID stack{
62  sector,
63  static_cast<tpc::GEMstack>(roc)};
64 
65  std::vector<float> qMaxCorrOut;
66  std::vector<float> qTotCorrOut;
67  std::vector<float> tglOut;
68  std::vector<float> snpOut;
69 
70  for (float tgl = 0; tgl < 2; tgl += 0.01) {
71  for (float snp = 0; snp < 1; snp += 0.1) {
72  qMaxCorrOut.emplace_back(getCorrection(stack, ChargeType::Max, tgl, snp));
73  qTotCorrOut.emplace_back(getCorrection(stack, ChargeType::Tot, tgl, snp));
74  tglOut.emplace_back(tgl);
75  snpOut.emplace_back(snp);
76  }
77  }
78 
79  pcstream << "tree"
80  << "qMaxCorr=" << qMaxCorrOut
81  << "qTotCorr=" << qTotCorrOut
82  << "tgl=" << tglOut
83  << "snp=" << snpOut
84  << "roc=" << roc
85  << "sector=" << sector
86  << "\n";
87  }
88  }
89 }
