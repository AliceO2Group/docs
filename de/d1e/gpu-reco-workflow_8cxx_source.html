140WorkflowSpec defineDataProcessing(ConfigContext const& cfgc)
141{
142 WorkflowSpec specs;
143 std::vector<int32_t> tpcSectors(o2::tpc::Sector::MAXSECTOR);
144 std::iota(tpcSectors.begin(), tpcSectors.end(), 0);
145
146 auto inputType = cfgc.options().get<std::string>("input-type");
147 bool doMC = !cfgc.options().get<bool>("disable-mc");
148 auto sclOpt = o2::tpc::CorrectionMapsLoader::parseGlobalOptions(cfgc.options());
149 o2::conf::ConfigurableParam::updateFromFile(cfgc.options().get<std::string>("configFile"));
150 o2::conf::ConfigurableParam::updateFromString(cfgc.options().get<std::string>("configKeyValues"));
151 o2::conf::ConfigurableParam::writeINI("o2gpurecoworkflow_configuration.ini");
152
153 std::vector<ioType> outputTypes, inputTypes;
154 try {
155 outputTypes = o2::RangeTokenizer::tokenize<ioType>(cfgc.options().get<std::string>("output-type"), [](std::string const& token) { return OutputMap.at(token); });
156 inputTypes = o2::RangeTokenizer::tokenize<ioType>(cfgc.options().get<std::string>("input-type"), [](std::string const& token) { return InputMap.at(token); });
157 } catch (std::out_of_range&) {
158 throw std::invalid_argument("invalid input / output type");
159 }
160
161 auto isEnabled = [](auto& list, ioType type) {
162 return std::find(list.begin(), list.end(), type) != list.end();
163 };
164
165 GPURecoWorkflowSpec::Config cfg;
166 cfg.runTPCTracking = true;
167 cfg.lumiScaleType = sclOpt.lumiType;
168 cfg.lumiScaleMode = sclOpt.lumiMode;
169 cfg.enableMShape = sclOpt.enableMShapeCorrection;
170 cfg.enableCTPLumi = sclOpt.requestCTPLumi;
171 cfg.decompressTPCFromROOT = isEnabled(inputTypes, ioType::CompClustROOT);
172 cfg.decompressTPC = isEnabled(inputTypes, ioType::CompClustFlat) || cfg.decompressTPCFromROOT;
173 cfg.zsDecoder = isEnabled(inputTypes, ioType::ZSRaw);
174 cfg.zsOnTheFly = isEnabled(inputTypes, ioType::ZSRawOTF);
175 cfg.caClusterer = cfg.zsDecoder || cfg.zsOnTheFly || isEnabled(inputTypes, ioType::Digits);
176 cfg.outputTracks = isEnabled(outputTypes, ioType::Tracks);
177 cfg.outputCompClustersRoot = isEnabled(outputTypes, ioType::CompClustROOT);
178 cfg.outputCompClustersFlat = isEnabled(outputTypes, ioType::CompClustFlat);
179 cfg.outputCAClusters = isEnabled(outputTypes, ioType::Clusters);
180 cfg.outputQA = isEnabled(outputTypes, ioType::QA);
181 cfg.outputErrorQA = isEnabled(outputTypes, ioType::ErrorQA);
182 cfg.outputSharedClusterMap = (cfg.outputCAClusters || cfg.caClusterer || isEnabled(inputTypes, ioType::Clusters)) && cfg.outputTracks && !isEnabled(outputTypes, ioType::NoSharedMap);
183 cfg.processMC = doMC;
184 cfg.sendClustersPerSector = isEnabled(outputTypes, ioType::SendClustersPerSector);
185 cfg.askDISTSTF = !cfgc.options().get<bool>("ignore-dist-stf");
186 cfg.readTRDtracklets = isEnabled(inputTypes, ioType::TRDTracklets);
187 cfg.runTRDTracking = isEnabled(outputTypes, ioType::TRDTracks);
188 cfg.tpcTriggerHandling = isEnabled(outputTypes, ioType::TPCTriggers) || cfg.caClusterer;
189 cfg.enableDoublePipeline = cfgc.options().get<bool>("enableDoublePipeline");
190 cfg.tpcDeadMapSources = cfgc.options().get<int32_t>("tpc-deadMap-sources");
191 cfg.tpcUseMCTimeGain = cfgc.options().get<bool>("tpc-mc-time-gain");
192 cfg.runITSTracking = isEnabled(outputTypes, ioType::ITSTracks);
193 cfg.itsOverrBeamEst = isEnabled(inputTypes, ioType::MeanVertex);
194 cfg.useFilteredOutputSpecs = cfgc.options().get<bool>("filtered-output-specs");
195
196 Inputs ggInputs;
197 auto ggRequest = std::make_shared<o2::base::GRPGeomRequest>(false, true, false, true, true, o2::base::GRPGeomRequest::Aligned, ggInputs, true);
198
199 auto task = std::make_shared<GPURecoWorkflowSpec>(&gPolicyData, cfg, tpcSectors, gTpcSectorMask, ggRequest, &gPolicyOrderCheck);
200 Inputs taskInputs = task->inputs();
201 std::move(ggInputs.begin(), ggInputs.end(), std::back_inserter(taskInputs));
202 gTask = task;
203
204 specs.emplace_back(DataProcessorSpec{
205 "gpu-reconstruction",
206 taskInputs,
207 task->outputs(),
208 AlgorithmSpec{adoptTask<GPURecoWorkflowSpec>(task)},
209 task->options()});
210
211 if (cfg.enableDoublePipeline) {
212 cfg.enableDoublePipeline = 2;
213 Inputs ggInputsPrepare;
214 auto ggRequestPrepare = std::make_shared<o2::base::GRPGeomRequest>(false, true, false, false, false, o2::base::GRPGeomRequest::None, ggInputsPrepare, true);
215 auto taskPrepare = std::make_shared<GPURecoWorkflowSpec>(&gPolicyData, cfg, tpcSectors, gTpcSectorMask, ggRequestPrepare);
216 Inputs taskInputsPrepare = taskPrepare->inputs();
217 std::move(ggInputsPrepare.begin(), ggInputsPrepare.end(), std::back_inserter(taskInputsPrepare));
218 specs.emplace_back(DataProcessorSpec{
219 "gpu-reconstruction-prepare",
220 taskInputsPrepare,
221 taskPrepare->outputs(),
222 AlgorithmSpec{adoptTask<GPURecoWorkflowSpec>(taskPrepare)},
223 taskPrepare->options()});
224 }
225
226 // configure dpl timer to inject correct firstTForbit: start from the 1st orbit of TF containing 1st sampled orbit
227 o2::raw::HBFUtilsInitializer hbfIni(cfgc, specs);
228
229 return specs;
230}
