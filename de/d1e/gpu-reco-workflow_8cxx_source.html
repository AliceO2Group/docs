59  {"ignore-dist-stf", VariantType::Bool, false, {"do not subscribe to FLP/DISTSUBTIMEFRAME/0 message (no lost TF recovery)"}},
60  {"configKeyValues", VariantType::String, "", {"Semicolon separated key=value strings (e.g.: 'TPCHwClusterer.peakChargeThreshold=4;...')"}},
61  {"configFile", VariantType::String, "", {"configuration file for configurable parameters"}},
62  {"enableDoublePipeline", VariantType::Bool, false, {"enable GPU double pipeline mode"}},

134 WorkflowSpec defineDataProcessing(ConfigContext const& cfgc)
135 {
136  WorkflowSpec specs;
137  std::vector<int> tpcSectors(o2::tpc::Sector::MAXSECTOR);
138  std::iota(tpcSectors.begin(), tpcSectors.end(), 0);
139 
140  auto inputType = cfgc.options().get<std::string>("input-type");
141  bool doMC = !cfgc.options().get<bool>("disable-mc");
142  auto sclOpt = o2::tpc::CorrectionMapsLoader::parseGlobalOptions(cfgc.options());
143  o2::conf::ConfigurableParam::updateFromFile(cfgc.options().get<std::string>("configFile"));
144  o2::conf::ConfigurableParam::updateFromString(cfgc.options().get<std::string>("configKeyValues"));
145  o2::conf::ConfigurableParam::writeINI("o2gpurecoworkflow_configuration.ini");
146 
147  std::vector<ioType> outputTypes, inputTypes;
148  try {
149  outputTypes = o2::RangeTokenizer::tokenize<ioType>(cfgc.options().get<std::string>("output-type"), [](std::string const& token) { return OutputMap.at(token); });
150  inputTypes = o2::RangeTokenizer::tokenize<ioType>(cfgc.options().get<std::string>("input-type"), [](std::string const& token) { return InputMap.at(token); });
151  } catch (std::out_of_range&) {
152  throw std::invalid_argument("invalid input / output type");
153  }
154 
155  auto isEnabled = [](auto& list, ioType type) {
156  return std::find(list.begin(), list.end(), type) != list.end();
157  };
158 
159  GPURecoWorkflowSpec::Config cfg;
160  cfg.runTPCTracking = true;
161  cfg.lumiScaleType = sclOpt.lumiType;
162  cfg.lumiScaleMode = sclOpt.lumiMode;
163  cfg.enableMShape = sclOpt.enableMShapeCorrection;
164  cfg.enableCTPLumi = sclOpt.requestCTPLumi;
165  cfg.decompressTPC = isEnabled(inputTypes, ioType::CompClustCTF);
166  cfg.decompressTPCFromROOT = isEnabled(inputTypes, ioType::CompClustROOT);
167  cfg.zsDecoder = isEnabled(inputTypes, ioType::ZSRaw);
168  cfg.zsOnTheFly = isEnabled(inputTypes, ioType::ZSRawOTF);
169  cfg.caClusterer = cfg.zsDecoder || cfg.zsOnTheFly || isEnabled(inputTypes, ioType::Digits);
170  cfg.outputTracks = isEnabled(outputTypes, ioType::Tracks);
171  cfg.outputCompClusters = isEnabled(outputTypes, ioType::CompClustROOT);
172  cfg.outputCompClustersFlat = isEnabled(outputTypes, ioType::CompClustCTF);
173  cfg.outputCAClusters = isEnabled(outputTypes, ioType::Clusters);
174  cfg.outputQA = isEnabled(outputTypes, ioType::QA);
175  cfg.outputErrorQA = isEnabled(outputTypes, ioType::ErrorQA);
176  cfg.outputSharedClusterMap = (cfg.outputCAClusters || cfg.caClusterer || isEnabled(inputTypes, ioType::Clusters)) && cfg.outputTracks && !isEnabled(outputTypes, ioType::NoSharedMap);
177  cfg.processMC = doMC;
178  cfg.sendClustersPerSector = isEnabled(outputTypes, ioType::SendClustersPerSector);
179  cfg.askDISTSTF = !cfgc.options().get<bool>("ignore-dist-stf");
180  cfg.readTRDtracklets = isEnabled(inputTypes, ioType::TRDTracklets);
181  cfg.runTRDTracking = isEnabled(outputTypes, ioType::TRDTracks);
182  cfg.tpcTriggerHandling = isEnabled(outputTypes, ioType::TPCTriggers) || cfg.caClusterer;
183  cfg.enableDoublePipeline = cfgc.options().get<bool>("enableDoublePipeline");
184  cfg.tpcDeadMapSources = cfgc.options().get<int>("tpc-deadMap-sources");
185  cfg.runITSTracking = isEnabled(outputTypes, ioType::ITSTracks);
186 
187  Inputs ggInputs;
188  auto ggRequest = std::make_shared<o2::base::GRPGeomRequest>(false, true, false, true, true, o2::base::GRPGeomRequest::Aligned, ggInputs, true);
189 
190  auto task = std::make_shared<GPURecoWorkflowSpec>(&gPolicyData, cfg, tpcSectors, gTpcSectorMask, ggRequest, &gPolicyOrderCheck);
191  Inputs taskInputs = task->inputs();
192  std::move(ggInputs.begin(), ggInputs.end(), std::back_inserter(taskInputs));
193  gTask = task;
194 
195  specs.emplace_back(DataProcessorSpec{
196  "gpu-reconstruction",
197  taskInputs,
198  task->outputs(),
199  AlgorithmSpec{adoptTask<GPURecoWorkflowSpec>(task)},
200  task->options()});
201 
202  if (cfg.enableDoublePipeline) {
203  cfg.enableDoublePipeline = 2;
204  Inputs ggInputsPrepare;
205  auto ggRequestPrepare = std::make_shared<o2::base::GRPGeomRequest>(false, true, false, false, false, o2::base::GRPGeomRequest::None, ggInputsPrepare, true);
206  auto taskPrepare = std::make_shared<GPURecoWorkflowSpec>(&gPolicyData, cfg, tpcSectors, gTpcSectorMask, ggRequestPrepare);
207  Inputs taskInputsPrepare = taskPrepare->inputs();
208  std::move(ggInputsPrepare.begin(), ggInputsPrepare.end(), std::back_inserter(taskInputsPrepare));
209  specs.emplace_back(DataProcessorSpec{
210  "gpu-reconstruction-prepare",
211  taskInputsPrepare,
212  taskPrepare->outputs(),
213  AlgorithmSpec{adoptTask<GPURecoWorkflowSpec>(taskPrepare)},
214  taskPrepare->options()});
215  }
216 
217  if (!cfgc.options().get<bool>("ignore-dist-stf")) {
218  GlobalTrackID::mask_t srcTrk = GlobalTrackID::getSourcesMask("none");
219  GlobalTrackID::mask_t srcCl = GlobalTrackID::getSourcesMask("TPC");
220  o2::globaltracking::InputHelper::addInputSpecs(cfgc, specs, srcCl, srcTrk, srcTrk, doMC);
221  }
222 
223  // configure dpl timer to inject correct firstTForbit: start from the 1st orbit of TF containing 1st sampled orbit
224  o2::raw::HBFUtilsInitializer hbfIni(cfgc, specs);
225 
226  return specs;
227 }
