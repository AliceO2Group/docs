56 void CalculatedEdx::fillMissingClusters(int missingClusters[4], float minChargeTot, float minChargeMax, int method)
57 {
58  if (method != 0 && method != 1) {
59  LOGP(info, "Unrecognized subthreshold cluster treatment. Not adding virtual charges to the track!");
60  return;
61  }
62 
63  for (int roc = 0; roc < 4; roc++) {
64  for (int i = 0; i < missingClusters[roc]; i++) {
65  float chargeTot = (method == 1) ? minChargeTot / 2.f : minChargeTot;
66  float chargeMax = (method == 1) ? minChargeMax / 2.f : minChargeMax;
67 
68  mChargeTotROC[roc].emplace_back(chargeTot);
69  mChargeTotROC[4].emplace_back(chargeTot);
70 
71  mChargeMaxROC[roc].emplace_back(chargeMax);
72  mChargeMaxROC[4].emplace_back(chargeMax);
73  }
74  }

146  // for missing clusters
147  unsigned char rowIndexOld = 0;
148  unsigned char sectorIndexOld = 0;
149  float minChargeTot = 100000.f;
150  float minChargeMax = 100000.f;
151 
152  // loop over the clusters
153  for (int iCl = 0; iCl < nClusters; iCl++) {
154 
155  const o2::tpc::ClusterNative& cl = track.getCluster(*mTPCTrackClIdxVecInput, iCl, *mClusterIndex);
156 
157  unsigned char sectorIndex = 0;
158  unsigned char rowIndex = 0;
159  unsigned int clusterIndexNumb = 0;
160 
161  // set sectorIndex, rowIndex, clusterIndexNumb
162  track.getClusterReference(*mTPCTrackClIdxVecInput, iCl, sectorIndex, rowIndex, clusterIndexNumb);
163 
164  // check if the cluster is shared
165  const unsigned int absoluteIndex = mClusterIndex->clusterOffset[sectorIndex][rowIndex] + clusterIndexNumb;
166  const bool isShared = mRefit ? (mTPCRefitterShMap[absoluteIndex] & GPUCA_NAMESPACE::gpu::GPUTPCGMMergedTrackHit::flagShared) : 0;
167 
168  // get region, pad, stack and stack ID
169  const int region = Mapper::REGION[rowIndex];
170  const unsigned char pad = std::clamp(static_cast<unsigned int>(cl.getPad() + 0.5f), static_cast<unsigned int>(0), Mapper::PADSPERROW[region][Mapper::getLocalRowFromGlobalRow(rowIndex)] - 1); // the left side of the pad is defined at e.g. 3.5 and the right side at 4.5
171  const CRU cru(Sector(sectorIndex), region);
172  const auto stack = cru.gemStack();
173  StackID stackID{sectorIndex, stack};
174  // the stack number for debugging
175  const int stackNumber = static_cast<int>(stack);
176 
177  // get local coordinates, offset and flags
178  const float localX = o2::tpc::Mapper::instance().getPadCentre(PadPos(rowIndex, pad)).X();
179  const float localY = Mapper::instance().getPadCentre(PadPos(rowIndex, pad)).Y();
180  const float offsPad = (cl.getPad() - pad) * o2::tpc::Mapper::instance().getPadRegionInfo(Mapper::REGION[rowIndex]).getPadWidth();
181  const auto flagsCl = cl.getFlags();
182 
183  int excludeCl = 0; // works as a bit mask
184  if (((clusterMask & ClusterFlags::ExcludeSingleCl) == ClusterFlags::ExcludeSingleCl) && ((flagsCl & ClusterNative::flagSingle) == ClusterNative::flagSingle)) {
185  excludeCl += 0b001; // 1 for single cluster
186  }
187  if (((clusterMask & ClusterFlags::ExcludeSplitCl) == ClusterFlags::ExcludeSplitCl) && (((flagsCl & ClusterNative::flagSplitPad) == ClusterNative::flagSplitPad) || ((flagsCl & ClusterNative::flagSplitTime) == ClusterNative::flagSplitTime))) {
188  excludeCl += 0b010; // 2 for split cluster
189  }
190  if (((clusterMask & ClusterFlags::ExcludeEdgeCl) == ClusterFlags::ExcludeEdgeCl) && ((flagsCl & ClusterNative::flagEdge) == ClusterNative::flagEdge)) {
191  excludeCl += 0b100; // 4 for edge cluster
192  }
193  if (((clusterMask & ClusterFlags::ExcludeSharedCl) == ClusterFlags::ExcludeSharedCl) && isShared) {
194  excludeCl += 0b10000; // for shared cluster
195  }
196 
197  // get the x position of the track
198  const float xPosition = Mapper::instance().getPadCentre(PadPos(rowIndex, 0)).X();
199 
200  bool check = true;
201  if (!mPropagateTrack) {
202  if (mRefit == nullptr) {
203  LOGP(error, "mRefit is a nullptr, call the function setRefit() before looping over the tracks.");
204  }
205  mRefit->setTrackReferenceX(xPosition);
206  check = (mRefit->RefitTrackAsGPU(track, false, true) < 0) ? false : true;
207  } else {
208  // propagate this track to the plane X=xk (cm) in the field "b" (kG)
209  track.rotate(o2::math_utils::detail::sector2Angle<float>(sectorIndex));
210  check = o2::base::Propagator::Instance()->PropagateToXBxByBz(track, xPosition, 0.999f, 2., o2::base::Propagator::MatCorrType::USEMatCorrLUT);
211  }
212 
213  if (!check || std::isnan(track.getParam(1))) {
214  excludeCl += 0b1000; // 8 for failure of track propagation or refit
215  }
216 
217  if (excludeCl != 0) {
218  // for debugging
219  if (mDebug) {
220  excludeClVector.emplace_back(excludeCl);
221  regionVector.emplace_back(region);
222  rowIndexVector.emplace_back(rowIndex);
223  padVector.emplace_back(pad);
224  sectorVector.emplace_back(sectorIndex);
225  stackVector.emplace_back(stackNumber);
226  localXVector.emplace_back(localX);
227  localYVector.emplace_back(localY);
228  offsPadVector.emplace_back(offsPad);
229  trackVector.emplace_back(track);
230  clVector.emplace_back(cl);
231  occupancyVector.emplace_back(getOccupancy(cl));
232  isClusterShared.emplace_back(isShared);
233 
234  topologyCorrVector.emplace_back(-999.f);
235  topologyCorrTotVector.emplace_back(-999.f);
236  topologyCorrMaxVector.emplace_back(-999.f);
237  gainVector.emplace_back(-999.f);
238  gainResidualVector.emplace_back(-999.f);
239  residualCorrTotVector.emplace_back(-999.f);
240  residualCorrMaxVector.emplace_back(-999.f);
241  scCorrVector.emplace_back(-999.f);
242  }
243  // to avoid counting the skipped cluster as a subthreshold cluster
244  rowIndexOld = rowIndex;
245  sectorIndexOld = sectorIndex;
246  continue;
247  }
248 
249  // get charge values
250  float chargeTot = cl.qTot;
251  float chargeMax = cl.qMax;
252 
253  // get threshold
254  const float threshold = mCalibCont.getZeroSupressionThreshold(sectorIndex, rowIndex, pad);
255 
256  // find missing clusters
257  int missingClusters = rowIndexOld - rowIndex - 1;
258  if ((missingClusters > 0) && (missingClusters <= mMaxMissingCl)) {
259  if ((clusterMask & ClusterFlags::ExcludeSectorBoundaries) == ClusterFlags::ExcludeSectorBoundaries) {
260  if (sectorIndexOld == sectorIndex) {
261  if (stack == GEMstack::IROCgem) {
262  nClsSubThreshROC[0] += missingClusters;
263  nClsROC[0] += missingClusters;
264  } else if (stack == GEMstack::OROC1gem) {
265  nClsSubThreshROC[1] += missingClusters;
266  nClsROC[1] += missingClusters;
267  } else if (stack == GEMstack::OROC2gem) {
268  nClsSubThreshROC[2] += missingClusters;
269  nClsROC[2] += missingClusters;
270  } else if (stack == GEMstack::OROC3gem) {
271  nClsSubThreshROC[3] += missingClusters;
272  nClsROC[3] += missingClusters;
273  }
274  }
275  } else {
276  if (stack == GEMstack::IROCgem) {
277  nClsSubThreshROC[0] += missingClusters;
278  nClsROC[0] += missingClusters;
279  } else if (stack == GEMstack::OROC1gem) {
280  nClsSubThreshROC[1] += missingClusters;
281  nClsROC[1] += missingClusters;
282  } else if (stack == GEMstack::OROC2gem) {
283  nClsSubThreshROC[2] += missingClusters;
284  nClsROC[2] += missingClusters;
285  } else if (stack == GEMstack::OROC3gem) {
286  nClsSubThreshROC[3] += missingClusters;
287  nClsROC[3] += missingClusters;
288  }
289  }
290  };
291  rowIndexOld = rowIndex;
292  sectorIndexOld = sectorIndex;
293 
294  // get effective length
295  float effectiveLength = 1.0f;

297  float effectiveLengthMax = 1.0f;
298  if ((correctionMask & CorrectionFlags::TopologySimple) == CorrectionFlags::TopologySimple) {

301  chargeMax /= effectiveLength;
302  };
303  if ((correctionMask & CorrectionFlags::TopologyPol) == CorrectionFlags::TopologyPol) {

305  effectiveLengthMax = getTrackTopologyCorrectionPol(track, cl, region, chargeMax, ChargeType::Max, threshold);
306  chargeTot /= effectiveLengthTot;
307  chargeMax /= effectiveLengthMax;
308  };
309

312  float gainResidual = 1.0f;
313  if ((correctionMask & CorrectionFlags::GainFull) == CorrectionFlags::GainFull) {

315  };
316  if ((correctionMask & CorrectionFlags::GainResidual) == CorrectionFlags::GainResidual) {
317  gainResidual = mCalibCont.getResidualGain(sectorIndex, rowIndex, pad);
318  };
319  chargeTot /= gain * gainResidual;
320  chargeMax /= gain * gainResidual;
321 
322  // get dEdx correction on tgl and sector plane
323  float corrTot = 1.0f;
324  float corrMax = 1.0f;
325  if ((correctionMask & CorrectionFlags::dEdxResidual) == CorrectionFlags::dEdxResidual) {
326  corrTot = mCalibCont.getResidualCorrection(stackID, ChargeType::Tot, track.getTgl(), track.getSnp());
327  corrMax = mCalibCont.getResidualCorrection(stackID, ChargeType::Max, track.getTgl(), track.getSnp());
328  if (corrTot > 0) {
329  chargeTot /= corrTot;
330  };
331  if (corrMax > 0) {
332  chargeMax /= corrMax;
333  };
334  };
335 
336  // set the min charge
337  if (chargeTot < minChargeTot) {
338  minChargeTot = chargeTot;
339  };
340 
341  if (chargeMax < minChargeMax) {
342  minChargeMax = chargeMax;
343  };
344 
345  // space-charge dEdx corrections
346  const float time = cl.getTime() - track.getTime0(); // ToDo: get correct time from ITS-TPC track if possible
347  float scCorr = 1.0f;
348  if ((correctionMask & CorrectionFlags::dEdxSC) == CorrectionFlags::dEdxSC) {
349  scCorr = mSCdEdxCorrection.getCorrection(time, sectorIndex, rowIndex, pad);
350  if (scCorr > 0) {
351  chargeTot /= scCorr;
352  };
353  if (corrMax > 0) {
354  chargeMax /= scCorr;
355  };
356  }
357 
358  if (stack == GEMstack::IROCgem) {
359  mChargeTotROC[0].emplace_back(chargeTot);
360  mChargeMaxROC[0].emplace_back(chargeMax);
361  nClsROC[0]++;
362  } else if (stack == GEMstack::OROC1gem) {
363  mChargeTotROC[1].emplace_back(chargeTot);
364  mChargeMaxROC[1].emplace_back(chargeMax);
365  nClsROC[1]++;
366  } else if (stack == GEMstack::OROC2gem) {
367  mChargeTotROC[2].emplace_back(chargeTot);
368  mChargeMaxROC[2].emplace_back(chargeMax);
369  nClsROC[2]++;
370  } else if (stack == GEMstack::OROC3gem) {
371  mChargeTotROC[3].emplace_back(chargeTot);
372  mChargeMaxROC[3].emplace_back(chargeMax);
373  nClsROC[3]++;
374  };
375 
376  mChargeTotROC[4].emplace_back(chargeTot);
377  mChargeMaxROC[4].emplace_back(chargeMax);
378 
379  // for debugging
380  if (mDebug) {
381  excludeClVector.emplace_back(0); // cl is successfully processed
382  regionVector.emplace_back(region);
383  rowIndexVector.emplace_back(rowIndex);
384  padVector.emplace_back(pad);
385  sectorVector.emplace_back(sectorIndex);
386  stackVector.emplace_back(stackNumber);
387  localXVector.emplace_back(localX);
388  localYVector.emplace_back(localY);
389  offsPadVector.emplace_back(offsPad);

391  clVector.emplace_back(cl);
392  occupancyVector.emplace_back(getOccupancy(cl));
393  isClusterShared.emplace_back(isShared);

395  topologyCorrVector.emplace_back(effectiveLength);
396  topologyCorrTotVector.emplace_back(effectiveLengthTot);
397  topologyCorrMaxVector.emplace_back(effectiveLengthMax);
398  gainVector.emplace_back(gain);
399  gainResidualVector.emplace_back(gainResidual);
400  residualCorrTotVector.emplace_back(corrTot);
401  residualCorrMaxVector.emplace_back(corrMax);
402  scCorrVector.emplace_back(scCorr);
403  };
404  }
405 
406  // number of clusters
407  output.NHitsSubThresholdIROC = nClsROC[0];
408  output.NHitsSubThresholdOROC1 = nClsROC[1];
409  output.NHitsSubThresholdOROC2 = nClsROC[2];

411 
412  // check if the lost clusters are subthreshold clusters based on the charge thresholds
413  if (minChargeTot <= mMinChargeTotThreshold && minChargeMax <= mMinChargeMaxThreshold) {
414  output.NHitsIROC = nClsROC[0] - nClsSubThreshROC[0];

423  } else {

437  output.dEdxTotOROC2 = getTruncMean(mChargeTotROC[2], low, high);

443  output.dEdxMaxOROC2 = getTruncMean(mChargeMaxROC[2], low, high);
444  output.dEdxMaxOROC3 = getTruncMean(mChargeMaxROC[3], low, high);
445  output.dEdxMaxTPC = getTruncMean(mChargeMaxROC[4], low, high);
446 
447  // for debugging
448  if (mDebug) {
449  if (mStreamer == nullptr) {
450  setStreamer(debugRootFile);
451  }
452

454  << "Ncl=" << nClusters

485 float CalculatedEdx::getTruncMean(std::vector<float>& charge, float low, float high) const
486 {
487  // sort the charge vector
488  std::sort(charge.begin(), charge.end());
489 
490  // calculate truncated mean
491  int nCl = 0;
492  float sum = 0;
493  size_t firstCl = charge.size() * low;
494  size_t lastCl = charge.size() * high;
495 
496  for (size_t iCl = firstCl; iCl < lastCl; ++iCl) {
497  sum += charge[iCl];
498  ++nCl;
499  }
500 
501  if (nCl > 0) {
502  sum /= nCl;
503  }
504  return sum;
505 }

507 float CalculatedEdx::getTrackTopologyCorrection(const o2::tpc::TrackTPC& track, const unsigned int region) const
508 {
509  const float padLength = Mapper::instance().getPadRegionInfo(region).getPadHeight();
510  const float snp = track.getSnp();
511  const float tgl = track.getTgl();
512  const float snp2 = snp * snp;
513  const float tgl2 = tgl * tgl;
514  // calculate the trace length of the track over the pad
515  const float effectiveLength = padLength * std::sqrt((1 + tgl2) / (1 - snp2));
516  return effectiveLength;
517 }

519 float CalculatedEdx::getTrackTopologyCorrectionPol(const o2::tpc::TrackTPC& track, const o2::tpc::ClusterNative& cl, const unsigned int region, const float charge, ChargeType chargeType, const float threshold) const
520 {
521  const float snp = std::abs(track.getSnp());
522  const float tgl = track.getTgl();
523  const float snp2 = snp * snp;
524  const float tgl2 = tgl * tgl;
525  const float sec2 = 1.f / (1.f - snp2);
526  const float tanTheta = std::sqrt(tgl2 * sec2);
527
