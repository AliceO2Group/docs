47 void EntropyEncoderSpec::run(ProcessingContext& pc)
48 {
49  auto cput = mTimer.CpuTime();
50  mTimer.Start(false);
51  mCTFCoder.updateTimeDependentParams(pc, true);
52  auto digits = pc.inputs().get<gsl::span<o2::ft0::Digit>>("digits");
53  auto channels = pc.inputs().get<gsl::span<o2::ft0::ChannelData>>("channels");
54  if (mSelIR) {
55  mCTFCoder.setSelectedIRFrames(pc.inputs().get<gsl::span<o2::dataformats::IRFrame>>("selIRFrames"));
56  }
57 
58  auto& buffer = pc.outputs().make<std::vector<o2::ctf::BufferType>>(Output{"FT0", "CTFDATA", 0});
59  auto iosize = mCTFCoder.encode(buffer, digits, channels);
60  if (mSelIR) {
61  mCTFCoder.getIRFramesSelector().clear();
62  }
63  mTimer.Stop();
64  LOG(info) << iosize.asString() << " in " << mTimer.CpuTime() - cput << " s";
65 }

73 DataProcessorSpec getEntropyEncoderSpec(bool selIR)
74 {
75  std::vector<InputSpec> inputs;
76  inputs.emplace_back("digits", "FT0", "DIGITSBC", 0, Lifetime::Timeframe);
77  inputs.emplace_back("channels", "FT0", "DIGITSCH", 0, Lifetime::Timeframe);
78  inputs.emplace_back("ctfdict", "FT0", "CTFDICT", 0, Lifetime::Condition, ccdbParamSpec("FT0/Calib/CTFDictionaryTree"));
79  if (selIR) {
80  inputs.emplace_back("selIRFrames", "CTF", "SELIRFRAMES", 0, Lifetime::Timeframe);
81  }
82  return DataProcessorSpec{
83  "ft0-entropy-encoder",
84  inputs,
85  Outputs{{"FT0", "CTFDATA", 0, Lifetime::Timeframe}},
86  AlgorithmSpec{adaptFromTask<EntropyEncoderSpec>(selIR)},
87  Options{{"ctf-dict", VariantType::String, "ccdb", {"CTF dictionary: empty or ccdb=CCDB, none=no external dictionary otherwise: local filename"}},
88  {"irframe-margin-bwd", VariantType::UInt32, 0u, {"margin in BC to add to the IRFrame lower boundary when selection is requested"}},
89  {"irframe-margin-fwd", VariantType::UInt32, 0u, {"margin in BC to add to the IRFrame upper boundary when selection is requested"}},
90  {"mem-factor", VariantType::Float, 1.f, {"Memory allocation margin factor"}},
91  {"ans-version", VariantType::String, {"version of ans entropy coder implementation to use"}}}};
92 }
