52  {

54  if (mDeviceTimers) {
55  mRec->GPUFailedMsg(cudaEventRecord(mDeviceTimers[0].get<cudaEvent_t>(), mStreams[mXYZ.x.stream]));
56  } else {
57  mTimer.ResetStart();
58  }
59  }
60  }
61  ~GPUDebugTiming()
62  {
63  if (mDo && mXYZ.t == 0.) {
64  if (mDeviceTimers) {
65  mRec->GPUFailedMsg(cudaEventRecord(mDeviceTimers[1].get<cudaEvent_t>(), mStreams[mXYZ.x.stream]));
66  mRec->GPUFailedMsg(cudaEventSynchronize(mDeviceTimers[1].get<cudaEvent_t>()));
67  float v;
68  mRec->GPUFailedMsg(cudaEventElapsedTime(&v, mDeviceTimers[0].get<cudaEvent_t>(), mDeviceTimers[1].get<cudaEvent_t>()));
69  mXYZ.t = v * 1.e-3f;
70  } else {
71  mRec->GPUFailedMsg(cudaStreamSynchronize(mStreams[mXYZ.x.stream]));
72  mXYZ.t = mTimer.GetCurrentElapsedTime();
73  }
74  }
75  }

79  cudaStream_t* mStreams;
