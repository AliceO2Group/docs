38 float x[HALFSECTORS];
39 float y[HALFSECTORS];

41 const auto startphi = 10. * M_PI / 180.;

44 x[s] = std::cos(s * deltaphi + startphi);

47
48 for (int s = 0; s < HALFSECTORS; ++s) {
