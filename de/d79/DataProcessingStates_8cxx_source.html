34 void DataProcessingStates::processCommandQueue()
35 {
36  int position = lastInsertedState.load(std::memory_order_relaxed);
37  // Process the commands in the order we have just computed.
38  while (position < DataProcessingStates::STATES_BUFFER_SIZE) {
39  DataProcessingStates::CommandHeader header;
40  // Avoid alignment issues.
41  memcpy(&header, &store[position], sizeof(DataProcessingStates::CommandHeader));
42  int id = header.id;
43  int64_t timestamp = header.timestamp;
44  int size = header.size;
45  assert(id < updateInfos.size());
46  auto& update = updateInfos[id];
47  // We need to update only once per invoked callback,
48  // because the metrics are stored in reverse insertion order.
49  if (generation > update.generation) {
50  // If we have enough capacity, we reuse the buffer.
51  // If the size is the same, we mark it as updated only if different.
52  if (statesViews[id].size == size) {
53  int diff = memcmp(statesBuffer.data() + statesViews[id].first, &store[position + sizeof(DataProcessingStates::CommandHeader)], size);
54  if (diff) {
55  memcpy(statesBuffer.data() + statesViews[id].first, &store[position + sizeof(DataProcessingStates::CommandHeader)], size);
56  updated[id] = true;
57  update.timestamp = timestamp;
58  update.generation = generation;
59  pushedMetricsLapse++;
60  }
61  } else if (statesViews[id].capacity >= size) {
62  memcpy(statesBuffer.data() + statesViews[id].first, &store[position + sizeof(DataProcessingStates::CommandHeader)], size);
63  statesViews[id].size = size;
64  updated[id] = true;
65  update.timestamp = timestamp;
66  update.generation = generation;
67  pushedMetricsLapse++;
68  } else if (statesViews[id].capacity < size) {
69  // Otherwise we need to reallocate.
70  int newCapacity = std::max(size, 64);
71  int first = statesBuffer.size();
72  statesBuffer.resize(statesBuffer.size() + newCapacity);
73  assert(first < statesBuffer.size());
74  memcpy(statesBuffer.data() + first, &store[position + sizeof(DataProcessingStates::CommandHeader)], size);
75  statesViews[id].first = first;
76  statesViews[id].size = size;
77  statesViews[id].capacity = newCapacity;
78  updated[id] = true;
79  update.timestamp = timestamp;
80  update.generation = generation;
81  pushedMetricsLapse++;
82  }
83  }
84  position += sizeof(DataProcessingStates::CommandHeader) + header.size;
85  }
86  assert(position == DataProcessingStates::STATES_BUFFER_SIZE);
87  // We reset the queue. Once again, the queue is filled in reverse order.
88  nextState.store(STATES_BUFFER_SIZE, std::memory_order_relaxed);
89  lastInsertedState.store(STATES_BUFFER_SIZE, std::memory_order_relaxed);
90  generation++;
91 }

93 void DataProcessingStates::updateState(CommandSpec cmd)
94 {
95  LOGP(debug, "Updating state {} with {}", cmd.id, std::string_view(cmd.data, cmd.size));
96  if (stateSpecs[cmd.id].name.empty()) {
97  throw runtime_error_f("StateID %d was not registered", (int)cmd.id);
98  }
99  if (cmd.id >= stateNames.size()) {
100  throw runtime_error_f("StateID %d is out of range", (int)cmd.id);
101  }
102  if (cmd.id >= updateInfos.size()) {
103  throw runtime_error_f("MetricID %d is out of range", (int)cmd.id);
104  }
105  // Do not update currently disabled states
106  if (enabled[cmd.id] == false) {
107  return;
108  }
109  pendingStates++;
110  // Add a static mutex to protect the queue
111  // Get the next available operation in an atomic way.
112  int size = sizeof(CommandHeader) + cmd.size;
113  if (size > 16384) {
114  throw runtime_error_f("State size is %d for state %s. States larger than 16384 bytes not supported for now.",
115  size, stateSpecs[cmd.id].name.c_str());
116  }
117  int idx = nextState.fetch_sub(size, std::memory_order_relaxed);
118  if (idx - size < 0) {
119  // We abort this command
120  pendingStates--;
121  while (pendingStates.load(std::memory_order_relaxed) > 0) {
122  // We need to wait for all the pending commands to be processed.
123  // This is needed because we are going to flush the queue.
124  // We cannot flush the queue while there are pending commands
125  // as we might end up with a command being processed twice.
126  std::this_thread::sleep_for(std::chrono::milliseconds(1));
127  }
128  processCommandQueue();
129  lastInsertedState.store(STATES_BUFFER_SIZE, std::memory_order_relaxed);
130  nextState.store(STATES_BUFFER_SIZE, std::memory_order_relaxed);
131  pendingStates++;
132  idx = nextState.fetch_sub(size, std::memory_order_relaxed);
133  } else if (idx < 0) {
134  while (idx < 0) {
135  // We need to wait for the flushing of the queue
136  std::this_thread::sleep_for(std::chrono::milliseconds(1));
137  idx = nextState.load(std::memory_order_relaxed);
138  }
139  return updateState(cmd);
140  }
141  // Save the state in the queue
142  assert(idx >= 0);
143  assert(cmd.id < statesViews.size());
144  int64_t timestamp = getTimestamp(realTimeBase, initialTimeOffset);
145  // We also write starting from idx - size, because we know this is
146  // reserved for us.
147  idx -= size;
148  CommandHeader header{(short)cmd.id, cmd.size, timestamp};
149  assert(idx >= 0);
150  assert(idx + sizeof(CommandHeader) + cmd.size <= store.size());
151  memcpy(&store.data()[idx], &header, sizeof(CommandHeader));
152  memcpy(&store.data()[idx + sizeof(CommandHeader)], cmd.data, cmd.size);
153 
154  lastInsertedState = idx;
155  pendingStates--;
156  // Keep track of the number of commands we have received.
157  updatedMetricsLapse++;
158 }

160 void DataProcessingStates::flushChangedStates(std::function<void(std::string const&, int64_t, std::string_view)> const& callback)
161 {
162  publishingInvokedTotal++;
163  bool publish = false;
164  auto currentTimestamp = getTimestamp(realTimeBase, initialTimeOffset);
165  for (auto mi : registeredStates) {
166  auto& update = updateInfos[mi];
167  auto& spec = stateSpecs[mi];
168  auto& view = statesViews[mi];
169  if (updated[mi] == false && (currentTimestamp - update.timestamp) > spec.maxRefreshLatency) {
170  updated[mi] = true;
171  update.timestamp = currentTimestamp;
172  }
173  if (updated[mi] == false) {
174  continue;
175  }
176  int64_t delayPublished = (currentTimestamp - update.lastPublished);
177  assert(delayPublished >= 0);
178  if (delayPublished < spec.minPublishInterval) {
179  continue;
180  }
181  publish = true;
182  assert(view.first + view.size <= statesBuffer.size());
183  assert(view.first <= statesBuffer.size());
184  auto msg = std::string_view(statesBuffer.data() + view.first, view.size);
185 
186  callback(spec.name.data(), update.timestamp, msg);
187  publishedMetricsLapse++;
188  update.lastPublished = currentTimestamp;
189  updated[mi] = false;
190  }
191  if (publish) {
192  publishingDoneTotal++;
193  }
194  static int64_t startTime = uv_hrtime();
195  int64_t now = uv_hrtime();
196  double averageInvocations = (publishingInvokedTotal * 1000000000) / (now - startTime);
197  double averagePublishing = (publishedMetricsLapse * 1000000000) / (now - startTime);
198 
199  LOGP(debug, "Publishing invoked {} times / s, {} metrics published / s", (int)averageInvocations, (int)averagePublishing);
200 }

202 void DataProcessingStates::repack()
203 {
204  // Everytime we publish, we repack the states buffer so that we can minimize the
205  // amount of memory me use.
206  std::array<int, MAX_STATES> order;
207  std::iota(order.begin(), order.end(), 0);
208  std::stable_sort(order.begin(), order.begin() + statesViews.size(), [&](int a, int b) {
209  return statesViews[a].first < statesViews[b].first;
210  });
211  int position = 0;
212  for (size_t i = 0; i < order.size(); ++i) {
213  auto& view = statesViews[order[i]];
214  // If we have no size, we do not need to move anything.
215  if (view.size == 0) {
216  continue;
217  }
218  // If we are already in the correct place, do nothing.
219  if (view.first == position) {
220  continue;
221  }
222  memcpy(statesBuffer.data() + position, statesBuffer.data() + view.first, view.size);
223  view.first = position;
224  view.capacity = view.size;
225  position += view.size;
226  }
227 }

229 void DataProcessingStates::registerState(StateSpec const& spec)
230 {
231  if (stateSpecs[spec.stateId].name.size() != 0 && spec.name != stateSpecs[spec.stateId].name) {
232  auto currentName = stateSpecs[spec.stateId].name;
233  throw runtime_error_f("Metric %d already registered with name %s", spec.stateId, currentName.data(), spec.name.data());
234  }
235  auto currentMetric = std::find_if(stateSpecs.begin(), stateSpecs.end(), [&spec](StateSpec const& s) { return s.name == spec.name && s.stateId != spec.stateId; });
236  if (currentMetric != stateSpecs.end()) {
237  throw runtime_error_f("Metric %s already registered with id %d. Cannot reregister with %d.", spec.name.data(), currentMetric->stateId, spec.stateId);
238  }
239  stateSpecs[spec.stateId] = spec;
240  stateNames[spec.stateId] = spec.name;
241  int64_t currentTime = getTimestamp(realTimeBase, initialTimeOffset);
242  updateInfos[spec.stateId] = UpdateInfo{currentTime, currentTime};
243  updated[spec.stateId] = spec.sendInitialValue;
244  enabled[spec.stateId] = spec.defaultEnabled;
245  registeredStates.push_back(spec.stateId);
246 }
