62 void GPUDisplay::DoScreenshot(const char* filename, std::vector<char>& pixels, float animateTime)
63 {
64  size_t screenshot_x = mBackend->mScreenWidth * mCfgR.screenshotScaleFactor;
65  size_t screenshot_y = mBackend->mScreenHeight * mCfgR.screenshotScaleFactor;
66  size_t size = 4 * screenshot_x * screenshot_y;
67  if (size != pixels.size()) {
68  GPUError("Pixel array of incorrect size obtained");
69  filename = nullptr;
70  }
71 
72  if (filename) {
73  FILE* fp = fopen(filename, "w+b");
74  if (fp == nullptr) {
75  GPUError("Error opening screenshot file %s", filename);
76  return;
77  }
78 
79  BITMAPFILEHEADER bmpFH;
80  BITMAPINFOHEADER bmpIH;
81  memset(&bmpFH, 0, sizeof(bmpFH));
82  memset(&bmpIH, 0, sizeof(bmpIH));
83 
84  bmpFH.bfType = 19778; //"BM"
85  bmpFH.bfSize = sizeof(bmpFH) + sizeof(bmpIH) + size;
86  bmpFH.bfOffBits = sizeof(bmpFH) + sizeof(bmpIH);
87 
88  bmpIH.biSize = sizeof(bmpIH);
89  bmpIH.biWidth = screenshot_x;
90  bmpIH.biHeight = screenshot_y;
91  bmpIH.biPlanes = 1;
92  bmpIH.biBitCount = 32;
93  bmpIH.biCompression = BI_RGB;
94  bmpIH.biSizeImage = size;
95  bmpIH.biXPelsPerMeter = 5670;
96  bmpIH.biYPelsPerMeter = 5670;
97 
98  fwrite(&bmpFH, 1, sizeof(bmpFH), fp);
99  fwrite(&bmpIH, 1, sizeof(bmpIH), fp);
100  fwrite(pixels.data(), 1, size, fp);
101  fclose(fp);
102  }
103 }
