29// fast bit count
30inline int numberOfBitsSet(uint32_t x)
31{
32 // count number of non-0 bits in 32bit word
33 x = x - ((x >> 1) & 0x55555555);
34 x = (x & 0x33333333) + ((x >> 2) & 0x33333333);
35 return (((x + (x >> 4)) & 0x0F0F0F0F) * 0x01010101) >> 24;
36}

38// recursive creation of bitmask
39template <typename T>
40constexpr uint32_t bit2Mask(T v)
41{
42 return 0x1 << v;

45template <typename T, typename... Args>
46constexpr uint32_t bit2Mask(T first, Args... args)
47{
