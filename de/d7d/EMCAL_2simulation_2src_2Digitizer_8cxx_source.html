38 void Digitizer::init()
39 {
40  mSimParam = &(o2::emcal::SimParam::Instance());
41  auto randomSeed = std::chrono::high_resolution_clock::now().time_since_epoch().count();
42  if (o2::conf::DigiParams::Instance().seed != 0) {
43  randomSeed = o2::conf::DigiParams::Instance().seed;
44  }
45  mRandomGenerator = new TRandom3(randomSeed);
46 
47  float tau = mSimParam->getTimeResponseTau();
48  float N = mSimParam->getTimeResponsePower();
49 
50  mSmearEnergy = mSimParam->doSmearEnergy();
51  mSimulateTimeResponse = mSimParam->doSimulateTimeResponse();
52 
53  mDigits.init();
54  mDigits.reserve();
55  /*
56  if ((mDelay - mTimeWindowStart) != 0)
57  {
58  mDigits.setBufferSize(mDigits.getBufferSize() + (mDelay - mTimeWindowStart));
59  mDigits.reserve();
60  }
61  */
62 
63  if (mSimulateTimeResponse) {
64  // for each phase create a template distribution
65  TF1 RawResponse("RawResponse", rawResponseFunction, 0, 256, 5);
66  RawResponse.SetParameters(1., 0., tau, N, 0.);
67 
68  for (int phase = 0; phase < 4; phase++) {
69  // parameter 1: Handling phase + delay
70  // phase: 25 ns * phase index (-4)
71  // delay: Average signal delay
72  for (int itofbin = 0; itofbin < EMC_TOF_BINS; itofbin++) {
73  double tofbincenter = itofbin * EMC_TOF_BINWITH + 0.5 * EMC_TOF_BINWITH;
74  RawResponse.SetParameter(1, 0.25 * phase + (tofbincenter + mSimParam->getSignalDelay()) / constants::EMCAL_TIMESAMPLE);
75  for (int sample = 0; sample < constants::EMCAL_MAXTIMEBINS; sample++) {
76  mAmplitudeInTimeBins[phase][itofbin][sample] = RawResponse.Eval(sample);
77  }
78  }
79  }
80  } else {
81  }
82 
83  if (mEnableDebugStreaming) {
84  mDebugStream = std::make_unique<o2::utils::TreeStreamRedirector>("emcaldigitsDebug.root", "RECREATE");
85  }
86 }

89 double Digitizer::rawResponseFunction(double* x, double* par)
90 {
91  double signal = 0.;
92  double tau = par[2];
93  double n = par[3];
94  double ped = par[4];
95  double xx = (x[0] - par[1] + tau) / tau;
96 
97  // par[0] amp, par[1] peak time
98 
99  if (xx <= 0) {
100  signal = ped;
101  } else {
102  signal = ped + par[0] * std::pow(xx, n) * std::exp(n * (1 - xx));
103  }
104 
105  return signal;
106 }

115 void Digitizer::process(const std::vector<LabeledDigit>& labeledSDigits)
116 {
117 
118  for (auto labeleddigit : labeledSDigits) {
119 
120  int tower = labeleddigit.getTower();
121 
122  sampleSDigit(labeleddigit.getDigit());
123 
124  if (mTempDigitVector.size() == 0) {
125  continue;
126  }
127 
128  std::vector<LabeledDigit> listofLabeledDigit;
129 
130  for (auto& digit : mTempDigitVector) {
131  Int_t id = digit.getTower();
132 
133  auto labels = labeleddigit.getLabels();
134  LabeledDigit d(digit, labels[0]);
135  int iLabel(0);
136  for (auto& label : labels) {
137  if (digit.getAmplitude() < __DBL_EPSILON__) {
138  label.setAmplitudeFraction(0);
139  }
140  if (iLabel == 0) {
141  continue;
142  }
143  d.addLabel(label);
144  iLabel++;
145  }
146  listofLabeledDigit.push_back(d);
147  }
148  mDigits.addDigits(tower, listofLabeledDigit);
149  }
150 }

153 void Digitizer::sampleSDigit(const Digit& sDigit)
154 {
155  mTempDigitVector.clear();
156  Int_t tower = sDigit.getTower();
157  Double_t energy = sDigit.getAmplitude();
158 
159  if (mSmearEnergy) {
160  energy = smearEnergy(energy);
161  }
162 
163  if (energy < __DBL_EPSILON__) {
164  return;
165  }
166 
167  Double_t energies[15];
168  if (mSimulateTimeResponse) {
169  if (sDigit.getTimeStamp() + mSimParam->getSignalDelay() + mPhase * 25 > EMC_TOF_MAX) {
170  // Digit time larger than sampling window, will not be sampled
171  // For time response simulation take also signal delay and phase into account
172  return;
173  }
174  int tofbin = static_cast<int>(sDigit.getTimeStamp() / EMC_TOF_BINWITH);
175  if (tofbin >= EMC_TOF_BINS) {
176  tofbin = EMC_TOF_BINS - 1;
177  }
178  for (int sample = 0; sample < mAmplitudeInTimeBins[mPhase][tofbin].size(); sample++) {
179 
180  double val = energy * (mAmplitudeInTimeBins[mPhase][tofbin][sample]);
181  energies[sample] = val;
182  double digitTime = mEventTimeOffset * constants::EMCAL_TIMESAMPLE;
183  Digit digit(tower, val, digitTime);
184  mTempDigitVector.push_back(digit);
185  }
186  } else {
187  if (sDigit.getTimeStamp() > EMC_TOF_MAX) {
188  // Digit time larger than sampling window, will not be sampled
189  // In non-sampled mode only apply the max. time window
190  return;
191  }
192  Digit digit(tower, energy, smearTime(sDigit.getTimeStamp(), energy));
193  mTempDigitVector.push_back(digit);
194  }
195 
196  if (mEnableDebugStreaming) {
197  double timeStamp = sDigit.getTimeStamp();
198  (*mDebugStream).GetFile()->cd();
199  (*mDebugStream) << "DigitsTimeSamples"
200  << "Tower=" << tower
201  << "Time=" << timeStamp
202  << "DigitEnergy=" << energy
203  << "Sample0=" << energies[0]
204  << "Sample1=" << energies[1]
205  << "Sample2=" << energies[2]
206  << "Sample3=" << energies[3]
207  << "Sample4=" << energies[4]
208  << "Sample5=" << energies[5]
209  << "Sample6=" << energies[6]
210  << "Sample7=" << energies[7]
211  << "Sample8=" << energies[8]
212  << "Sample9=" << energies[9]
213  << "Sample10=" << energies[10]
214  << "Sample11=" << energies[11]
215  << "Sample12=" << energies[12]
216  << "Sample13=" << energies[13]
217  << "Sample14=" << energies[14]
218  << "\n";
219  }
220 }

223 double Digitizer::smearEnergy(double energy)
224 {
225  Double_t fluct = (energy * mSimParam->getMeanPhotonElectron()) / mSimParam->getGainFluctuations();
226  energy *= mRandomGenerator->Poisson(fluct) / fluct;
227  return energy;
228 }
