44WorkflowSpec defineDataProcessing(ConfigContext const& configcontext)
45{
46 // Update the (declared) parameters if changed from the command line
47 o2::conf::ConfigurableParam::updateFromString(configcontext.options().get<std::string>("configKeyValues"));
48 // write the configuration used for the workflow
49 o2::conf::ConfigurableParam::writeINI("o2trdkrcluster_workflow_configuration.ini");
50
51 o2::framework::WorkflowSpec specs;
52
53 // input
54 if (!configcontext.options().get<bool>("disable-root-input")) {
55 specs.emplace_back(o2::trd::getTRDDigitReaderSpec(false));
56 }
57
58 // processing devices
59 specs.emplace_back(o2::trd::getKrClustererSpec());
60
61 // output devices
62 if (!configcontext.options().get<bool>("disable-root-output")) {
63 specs.emplace_back(o2::trd::getKrClusterWriterSpec());
64 }
65
66 return specs;
67}
