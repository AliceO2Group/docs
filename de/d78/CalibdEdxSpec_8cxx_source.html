45  void init(framework::InitContext& ic) final
46  {
47  o2::base::GRPGeomHelper::instance().setRequest(mCCDBRequest);
48  const auto minEntriesSector = ic.options().get<int>("min-entries-sector");
49  const auto minEntries1D = ic.options().get<int>("min-entries-1d");
50  const auto minEntries2D = ic.options().get<int>("min-entries-2d");
51  const auto fitPasses = ic.options().get<int>("fit-passes");
52  const auto fitThreshold = ic.options().get<float>("fit-threshold");
53  const auto fitThresholdLowFactor = ic.options().get<float>("fit-threshold-low-factor");
54 
55  const auto dEdxBins = ic.options().get<int>("dedxbins");
56  const auto mindEdx = ic.options().get<float>("min-dedx");
57  const auto maxdEdx = ic.options().get<float>("max-dedx");
58  const auto angularBins = ic.options().get<int>("angularbins");
59  const auto fitSnp = ic.options().get<bool>("fit-snp");
60  mMakeGaussianFits = !ic.options().get<bool>("disable-gaussian-fits");
61 
62  mDumpToFile = ic.options().get<int>("file-dump");
63 
64  mCalib = std::make_unique<CalibdEdx>(dEdxBins, mindEdx, maxdEdx, angularBins, fitSnp);
65  mCalib->setApplyCuts(false);
66  mCalib->setSectorFitThreshold(minEntriesSector);
67  mCalib->set1DFitThreshold(minEntries1D);
68  mCalib->set2DFitThreshold(minEntries2D);
69  mCalib->setElectronCut(fitThreshold, fitPasses, fitThresholdLowFactor);
70  mCalib->setMaterialType(mMatType);
71  }

73  void finaliseCCDB(o2::framework::ConcreteDataMatcher& matcher, void* obj) final
74  {
75  if (o2::base::GRPGeomHelper::instance().finaliseCCDB(matcher, obj)) {
76  return;
77  }
78  if (matcher == ConcreteDataMatcher("TPC", "TIMEGAIN", 0)) {
79  mCalib->setCalibrationInput(*(o2::tpc::CalibdEdxCorrection*)obj);
80  const auto meanParamTot = mCalib->getCalibrationInput().getMeanParams(ChargeType::Tot);
81  LOGP(info, "Updating TimeGain with {} dimensions and mean qTot Params {}", mCalib->getCalibrationInput().getDims(), utils::elementsToString(meanParamTot));
82  return;
83  }
84  }

86  void run(ProcessingContext& pc) final
87  {
88  o2::base::GRPGeomHelper::instance().checkUpdates(pc);
89  checkUpdates(pc);
90  const auto tfcounter = o2::header::get<DataProcessingHeader*>(pc.inputs().get("tracks").header)->startTime;
91  const auto tracks = pc.inputs().get<gsl::span<TrackTPC>>("tracks");
92 
93  LOGP(detail, "Processing TF {} with {} tracks", tfcounter, tracks.size());
94  mCalib->fill(tracks);
95 
96  // store run number and CCDB time only once
97  if ((mTimeStampStart == 0) || (pc.services().get<o2::framework::TimingInfo>().timeslice == 0)) {
98  mRunNumber = processing_helpers::getRunNumber(pc);
99  mTimeStampStart = processing_helpers::getTimeStamp(pc, o2::base::GRPGeomHelper::instance().getOrbitResetTimeMS());
100  LOGP(info, "Setting start time stamp for writing to CCDB to {}", mTimeStampStart);
101  }
102  }
103 
104  void endOfStream(EndOfStreamContext& eos) final
105  {
106  LOGP(info, "Finalizing calibration");
107  mCalib->finalize(mMakeGaussianFits);
108  mCalib->print();
109  sendOutput(eos.outputs());
110 
111  if (mDumpToFile) {
112  mCalib->dumpToFile("calibdEdx_Obj.root", "calib");
113  mCalib->getCalib().writeToFile("calibdEdx.root", "ccdb_object");
114  if (mDumpToFile > 1) {
115  mCalib->writeTTree("calibdEdx.histo.tree.root");
116  }
117  }
118  }

120  private:
121  void sendOutput(DataAllocator& output)
122  {
123  const auto& corr = mCalib->getCalib();
124  o2::ccdb::CcdbObjectInfo info(CDBTypeMap.at(CDBType::CalTimeGain), std::string{}, std::string{}, std::map<std::string, std::string>{{"runNumber", std::to_string(mRunNumber)}}, mTimeStampStart, o2::ccdb::CcdbObjectInfo::INFINITE_TIMESTAMP);
125  auto image = o2::ccdb::CcdbApi::createObjectImage(&corr, &info);
126  LOGP(info, "Sending object {} / {} of size {} bytes, valid for {} : {} ", info.getPath(), info.getFileName(), image->size(), info.getStartValidityTimestamp(), info.getEndValidityTimestamp());
127  output.snapshot(Output{o2::calibration::Utils::gDataOriginCDBPayload, "TPC_CalibdEdx", 0}, *image.get());
128  output.snapshot(Output{o2::calibration::Utils::gDataOriginCDBWrapper, "TPC_CalibdEdx", 0}, info);
129  }

149 DataProcessorSpec getCalibdEdxSpec(const o2::base::Propagator::MatCorrType matType)
150 {
151  const bool enableAskMatLUT = matType == o2::base::Propagator::MatCorrType::USEMatCorrLUT;
152  std::vector<OutputSpec> outputs;
153  outputs.emplace_back(ConcreteDataTypeMatcher{o2::calibration::Utils::gDataOriginCDBPayload, "TPC_CalibdEdx"}, Lifetime::Sporadic);
154  outputs.emplace_back(ConcreteDataTypeMatcher{o2::calibration::Utils::gDataOriginCDBWrapper, "TPC_CalibdEdx"}, Lifetime::Sporadic);
155  std::vector<InputSpec> inputs{{"tracks", "TPC", "MIPS", Lifetime::Sporadic}};
156  inputs.emplace_back("tpctimegain", "TPC", "TIMEGAIN", 0, Lifetime::Condition, ccdbParamSpec(o2::tpc::CDBTypeMap.at(o2::tpc::CDBType::CalTimeGain), {}, 1)); // time-dependent
157 
158  auto ccdbRequest = std::make_shared<o2::base::GRPGeomRequest>(true, // orbitResetTime
159  false, // GRPECS=true
160  false, // GRPLHCIF
161  true, // GRPMagField
162  enableAskMatLUT, // askMatLUT
163  o2::base::GRPGeomRequest::None, // geometry
164  inputs,
165  true,
166  true);
167 
168  return DataProcessorSpec{
169  "tpc-calib-dEdx",
170  inputs,
171  outputs,
172  adaptFromTask<CalibdEdxDevice>(ccdbRequest, matType),
173  Options{
174  {"min-entries-sector", VariantType::Int, 1000, {"min entries per GEM stack to enable sector by sector correction. Below this value we only perform one fit per ROC type (IROC, OROC1, ...; no side nor sector information)."}},
175  {"min-entries-1d", VariantType::Int, 10000, {"minimum entries per stack to fit 1D correction"}},
176  {"min-entries-2d", VariantType::Int, 50000, {"minimum entries per stack to fit 2D correction"}},
177  {"fit-passes", VariantType::Int, 3, {"number of fit iterations"}},
178  {"fit-threshold", VariantType::Float, 0.2f, {"dEdx width around the MIP peak used in the fit"}},
179  {"fit-threshold-low-factor", VariantType::Float, 1.5f, {"factor for low dEdx width around the MIP peak used in the fit"}},
180 
181  {"dedxbins", VariantType::Int, 70, {"number of dEdx bins"}},
182  {"min-dedx", VariantType::Float, 10.0f, {"minimum value for dEdx histograms"}},
183  {"max-dedx", VariantType::Float, 90.0f, {"maximum value for dEdx histograms"}},
184  {"angularbins", VariantType::Int, 36, {"number of angular bins: Tgl and Snp"}},
185  {"fit-snp", VariantType::Bool, false, {"enable Snp correction"}},
186  {"disable-gaussian-fits", VariantType::Bool, false, {"disable calibration with gaussian fits and use mean instead"}},
187 
188  {"file-dump", VariantType::Int, 0, {"directly dump calibration to file"}}}};
189 }
