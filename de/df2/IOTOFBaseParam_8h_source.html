23struct ChipSpecifics {
24 int NCols = 0;
25 int NRows = 0;
26 float PitchCol = 0.;
27 float PitchRow = 0.;
28 float PassiveEdgeReadOut = 0.;
29 float PassiveEdgeTop = 0.;
30 float PassiveEdgeSide = 0.;
31 float SensorLayerThicknessEff = 0.;
32 float SensorLayerThickness = 0.;
33
34 int NPixels() const { return NCols * NRows; }
35 float ActiveMatrixSizeCols() const { return PitchCol * NCols; }
36 float ActiveMatrixSizeRows() const { return PitchRow * NRows; }
37 float SensorSizeCols() const { return ActiveMatrixSizeCols() + 2 * PassiveEdgeSide; }
38 float SensorSizeRows() const { return ActiveMatrixSizeRows() + PassiveEdgeTop + PassiveEdgeReadOut; }
39};
