69inline constexpr int findKeyIndex(int key, const int* arr, int n) noexcept
70{
71 // finds index of key in the array
72 int imn = 0;
73 int imx = n - 1;
74 while (imx >= imn) {
75 const int mid = (imx + imn) >> 1;
76 if (arr[mid] == key) {
77 return mid;
78 }
79
80 if (arr[mid] < key) {
81 imn = mid + 1;
82 } else {
83 imx = mid - 1;
84 }
85 }
86 return -1;
87}
