162  if (doReAlign) {

164  }
165

170 
171  LOG(info) << "Loading magnetic field and reference geometry from input files";
172

181  LOG(fatal) << "No GRP file";
182  }
183 
184  auto geoIdealFile = ic.options().get<string>("geo-file-ideal");
185  if (std::filesystem::exists(geoIdealFile)) {
186  base::GeometryManager::loadGeometry(geoIdealFile.c_str());
187  transformation = geo::transformationFromTGeoManager(*gGeoManager);
188  for (int i = 0; i < 156; i++) {
189  int iDEN = GetDetElemId(i);
190  transformIdeal[iDEN] = transformation(iDEN);
191  }
192  } else {
193  LOG(fatal) << "No ideal geometry";
194  }
195 
196  auto geoRefFile = ic.options().get<string>("geo-file-ref");
197  if (std::filesystem::exists(geoRefFile)) {
198  base::GeometryManager::loadGeometry(geoRefFile.c_str());
199  transformation = geo::transformationFromTGeoManager(*gGeoManager);
200  for (int i = 0; i < 156; i++) {
201  int iDEN = GetDetElemId(i);
202  transformRef[iDEN] = transformation(iDEN);

204  } else {
205  LOG(fatal) << "No reference geometry";

275  void processWithMatching(vector<ROFRecord>& mchROFs, vector<TrackMCH>& mchTracks, vector<Cluster>& mchClusters,
276  vector<dataformats::TrackMCHMID>& muonTracks)
277  {
278  // processing for each track
279  for (const auto& mchROF : mchROFs) {
280 
281  for (int iMCHTrack = mchROF.getFirstIdx();
282  iMCHTrack <= mchROF.getLastIdx(); ++iMCHTrack) {
283  // MCH-MID matching
284  if (!FindMuon(iMCHTrack, muonTracks)) {
285  continue;
286  }
287 
288  auto mchTrack = mchTracks.at(iMCHTrack);
289  int id_track = iMCHTrack;
290  int nb_clusters = mchTrack.getNClusters();
291 
292  // Track selection, considering only tracks having at least 10 clusters
293  if (nb_clusters <= 9) {
294  continue;
295  }
296 
297  // Format conversion from TrackMCH to Track(MCH internal use)
298  mch::Track convertedTrack = MCHFormatConvert(mchTrack, mchClusters, doReAlign);
299 
300  // Erase removable track
301  if (RemoveTrack(convertedTrack)) {
302  continue;
303  }
304 
305  // Track processing, saving residuals
306  mAlign.ProcessTrack(convertedTrack, transformation, doAlign, weightRecord);
307  }
308  }
309  }

344  //_________________________________________________________________________________________________
345  void run(framework::ProcessingContext& pc)
346  {
347  auto tStart = std::chrono::high_resolution_clock::now();
348  LOG(info) << "Starting alignment process";
349  if (doMatched) {
350  LOG(info) << "Using MCH-MID matched tracks";
351  }
352  if (mCCDBRequest) {
353 
354  LOG(info) << "Checking CCDB updates with processing context";
355  base::GRPGeomHelper::instance().checkUpdates(pc);
356 
357  auto geoIdeal = pc.inputs().get<TGeoManager*>("geomIdeal");
358  LOG(info) << "Loading ideal geometry from CCDB";
359  transformation = geo::transformationFromTGeoManager(*geoIdeal);
360  for (int i = 0; i < 156; i++) {
361  int iDEN = GetDetElemId(i);
362  transformIdeal[iDEN] = transformation(iDEN);
363  }
364  }
365 
366  // Load new geometry if we need to do re-align
367  if (doReAlign) {
368  if (NewGeoFileName != "") {
369  LOG(info) << "Loading re-alignment geometry";
370  base::GeometryManager::loadGeometry(NewGeoFileName.c_str());
371  transformation = geo::transformationFromTGeoManager(*gGeoManager);
372  for (int i = 0; i < 156; i++) {
373  int iDEN = GetDetElemId(i);
374  transformNew[iDEN] = transformation(iDEN);
375  }
376  } else {
377  LOG(fatal) << "No re-alignment geometry";
378  }
379  }
380 
381  if (!readFromRec) {
382  // Loading input data
383  LOG(info) << "Loading MCH tracks";
384  auto [fMCH, mchReader] = LoadData(mchFileName, "o2sim");
385  TTreeReaderValue<vector<ROFRecord>> mchROFs = {*mchReader, "trackrofs"};
386  TTreeReaderValue<vector<TrackMCH>> mchTracks = {*mchReader, "tracks"};
387  TTreeReaderValue<vector<Cluster>> mchClusters = {*mchReader, "trackclusters"};
388 
389  if (doMatched) {
390  LOG(info) << "Loading MID tracks";
391  auto [fMUON, muonReader] = LoadData(muonFileName.c_str(), "o2sim");
392  TTreeReaderValue<vector<dataformats::TrackMCHMID>> muonTracks = {*muonReader, "tracks"};
393  int nTF = muonReader->GetEntries(false);
394  if (mchReader->GetEntries(false) != nTF) {
395  LOG(error) << mchFileName << " and " << muonFileName << " do not contain the same number of TF";
396  exit(-1);
397  }
398 
399  LOG(info) << "Starting track processing";
400  while (mchReader->Next() && muonReader->Next()) {
401  int id_event = mchReader->GetCurrentEntry();
402  processWithMatching(*mchROFs, *mchTracks, *mchClusters, *muonTracks);
403  }
404  } else {
405  LOG(info) << "Starting track processing";
406  while (mchReader->Next()) {
407  int id_event = mchReader->GetCurrentEntry();
408  processWithOutMatching(*mchROFs, *mchTracks, *mchClusters);
409  }
410  }
411  }
412 
413  // Global fit
414  if (doAlign) {
415  mAlign.GlobalFit(params, errors, pulls);
416  }
417  auto tEnd = std::chrono::high_resolution_clock::now();
418  mElapsedTime = tEnd - tStart;
419 
420  // Generate new geometry w.r.t alignment results
421  if (doAlign) {
422 
423  LOG(info) << "Generating aligned geometry using global parameters";
424  vector<detectors::AlignParam> ParamAligned;
425  mAlign.ReAlign(ParamAligned, params);
426 
427  TFile* FileAlign = TFile::Open("AlignParam.root", "RECREATE");
428  FileAlign->cd();
429  FileAlign->WriteObjectAny(&ParamAligned, "std::vector<o2::detectors::AlignParam>", "alignment");
430  FileAlign->Close();
431 
432  string Geo_file;
433 
434  if (doReAlign) {
435  Geo_file = Form("%s%s", "o2sim_geometry_ReAlign", ".root");
436  } else {
437  Geo_file = Form("%s%s", "o2sim_geometry_Align", ".root");
438  }
439 
440  // Store aligned geometry
441  gGeoManager->Export(Geo_file.c_str());
442 
443  // Store param plots
444  drawHisto(params, errors, pulls, outFileName);
445 
446  // Export align params in ideal frame
447  TransRef(ParamAligned);
448  }
449 
450  mAlign.terminate();
451 
452  pc.services().get<ControlService>().endOfStream();
453  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
454  }

458  void TransRef(vector<detectors::AlignParam>& ParamsTrack)
459  {
460  LOG(info) << "Transforming align params to the frame of ideal geometry";
461  vector<o2::detectors::AlignParam> ParamsRef;
462  o2::detectors::AlignParam param_Ref;
463 
464  for (int hc = 0; hc < 20; hc++) {
465 
466  ParamsRef.emplace_back(ParamsTrack.at(fgSNDetElemHalfCh[hc]));
467 
468  for (int de = 0; de < fgNDetElemHalfCh[hc]; de++) {
469 
470  int iDEN = fgDetElemHalfCh[hc][de];
471  o2::detectors::AlignParam param_Track = ParamsTrack.at(fgSNDetElemHalfCh[hc] + 1 + de);
472 
473  LOG(debug) << Form("%s%s", "Processing DET Elem: ", (param_Track.getSymName()).c_str());
474 
475  TGeoHMatrix delta_track;
476  TGeoRotation r("Rotation/Track", param_Track.getPsi() / pi() * 180.0, param_Track.getTheta() / pi() * 180.0, param_Track.getPhi() / pi() * 180.0);
477  delta_track.SetRotation(r.GetRotationMatrix());
478  delta_track.SetDx(param_Track.getX());
479  delta_track.SetDy(param_Track.getY());
480  delta_track.SetDz(param_Track.getZ());
481 
482  TGeoHMatrix transRef = transformIdeal[iDEN];
483  TGeoHMatrix transTrack = doReAlign ? transformNew[iDEN] : transformRef[iDEN];
484  TGeoHMatrix transRefTrack = transTrack * transRef.Inverse();
485  TGeoHMatrix delta_ref = delta_track * transRefTrack;
486 
487  param_Ref.setSymName((param_Track.getSymName()).c_str());
488  param_Ref.setGlobalParams(delta_ref);
489  param_Ref.applyToGeometry();
490  ParamsRef.emplace_back(param_Ref);
491  }
492  }
493 
494  TFile* fOut = TFile::Open("AlignParam@ideal.root", "RECREATE");
495  fOut->WriteObjectAny(&ParamsRef, "std::vector<o2::detectors::AlignParam>", "alignment");
496  fOut->Close();
497  }

500  Track MCHFormatConvert(TrackMCH& mchTrack, vector<Cluster>& mchClusters, bool doReAlign)
501  {
502 
503  Track convertedTrack = Track();
504 
505  // Get clusters for current track
506  int id_cluster_first = mchTrack.getFirstClusterIdx();
507  int id_cluster_last = mchTrack.getLastClusterIdx();
508 
509  for (int id_cluster = id_cluster_first;
510  id_cluster < id_cluster_last + 1; ++id_cluster) {
511 
512  Cluster* cluster = &(mchClusters.at(id_cluster));
513  const int DEId_cluster = cluster->getDEId();
514  const int CId_cluster = cluster->getChamberId();
515  const int ind_cluster = cluster->getClusterIndex();
516 
517  // Transformations to new geometry from reference geometry
518  if (doReAlign) {
519 
520  math_utils::Point3D<double> local;
521  math_utils::Point3D<double> master;
522 
523  master.SetXYZ(cluster->getX(), cluster->getY(), cluster->getZ());
524 
525  transformRef[cluster->getDEId()].MasterToLocal(master, local);
526  transformNew[cluster->getDEId()].LocalToMaster(local, master);
527 
528  cluster->x = master.x();
529  cluster->y = master.y();
530  cluster->z = master.z();
531  }
532  convertedTrack.createParamAtCluster(*cluster);
533  }
534 
535  return Track(convertedTrack);
536  }

538  //_________________________________________________________________________________________________
539  bool RemoveTrack(Track& track)
540  {
541 
542  bool removeTrack = false;
543 
544  try {
545  trackFitter.fit(track, false);
546  } catch (exception const& e) {
547  removeTrack = true;
548  return removeTrack;
549  }
550 
551  auto itStartingParam = std::prev(track.rend());
552 
553  while (true) {
554 
555  try {
556  trackFitter.fit(track, true, false, (itStartingParam == track.rbegin()) ? nullptr : &itStartingParam);
557  } catch (exception const&) {
558  removeTrack = true;
559  break;
560  }
561 
562  double worstLocalChi2 = -1.0;
563 
564  track.tagRemovableClusters(0x1F, false);
565 
566  auto itWorstParam = track.end();
567 
568  for (auto itParam = track.begin(); itParam != track.end(); ++itParam) {
569  if (itParam->getLocalChi2() > worstLocalChi2) {
570  worstLocalChi2 = itParam->getLocalChi2();
571  itWorstParam = itParam;
572  }
573  }
574 
575  if (worstLocalChi2 < mImproveCutChi2) {
576  break;
577  }
578 
579  if (!itWorstParam->isRemovable()) {
580  removeTrack = true;
581  track.removable();
582  break;
583  }
584 
585  auto itNextParam = track.removeParamAtCluster(itWorstParam);
586  auto itNextToNextParam = (itNextParam == track.end()) ? itNextParam : std::next(itNextParam);
587  itStartingParam = track.rbegin();
588 
589  if (track.getNClusters() < 10) {
590  removeTrack = true;
591  break;
592  } else {
593  while (itNextToNextParam != track.end()) {
594  if (itNextToNextParam->getClusterPtr()->getChamberId() != itNextParam->getClusterPtr()->getChamberId()) {
595  itStartingParam = std::make_reverse_iterator(++itNextParam);
596  break;
597  }
598  ++itNextToNextParam;
599  }
600  }
601  }
602 
603  if (!removeTrack) {
604  for (auto& param : track) {
605  param.setParameters(param.getSmoothParameters());
606  param.setCovariances(param.getSmoothCovariances());
607  }
608  }
609 
610  return removeTrack;
611  }
