64 struct PadCoordinates {
65 std::vector<double> xVals = std::vector<double>(4);
66 std::vector<double> yVals = std::vector<double>(4);
67
68 void rotate(float angDeg)
69 {
70 const auto ang = 0.017453292519943295 * angDeg;
71 const auto cs = std::cos(ang);
72 const auto sn = std::sin(ang);
73 for (size_t i = 0; i < xVals.size(); ++i) {
74 const auto x = xVals[i] * cs - yVals[i] * sn;
75 const auto y = xVals[i] * sn + yVals[i] * cs;
76 xVals[i] = x;
77 yVals[i] = y;
78 }
79 }
80 };
