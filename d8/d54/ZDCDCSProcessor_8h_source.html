41 struct ZDCDCSinfo {
42  std::pair<uint64_t, double> firstValue; // first value seen by the ZDC DCS processor
43  std::pair<uint64_t, double> lastValue; // last value seen by the ZDC DCS processor
44  std::pair<uint64_t, double> midValue; // mid value seen by the ZDC DCS processor
45  std::pair<uint64_t, double> maxChange; // maximum variation seen by the ZDC DCS processor
46 
47  ZDCDCSinfo()
48  {
49  firstValue = std::make_pair(0, -999999999);
50  lastValue = std::make_pair(0, -999999999);
51  midValue = std::make_pair(0, -999999999);
52  maxChange = std::make_pair(0, -999999999);
53  }
54  void makeEmpty()
55  {
56  firstValue.first = lastValue.first = midValue.first = maxChange.first = 0;
57  firstValue.second = lastValue.second = midValue.second = maxChange.second = -999999999;
58  }
59  void print() const;
60 
61  ClassDefNV(ZDCDCSinfo, 1);
62 };
