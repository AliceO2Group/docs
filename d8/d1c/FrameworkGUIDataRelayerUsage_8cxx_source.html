97 void displayDataRelayer(DeviceMetricsInfo const& metrics,
98  DeviceInfo const& info,
99  DataProcessingStates const& states,
100  ImVec2 const& size)
101 {
102  auto getNumRecords = [&states]() -> size_t {
103  auto& view = states.statesViews[(int)ProcessingStateId::DATA_RELAYER_BASE];
104  if (view.size == 0) {
105  return 0;
106  }
107  // The first number is the size of the pipeline
108  int numRecords = strtoul(states.statesBuffer.data() + view.first, nullptr, 10);
109  return numRecords;
110  };
111  auto getRecord = [](size_t i) -> int {
112  return i;
113  };
114  auto getNumItems = [&states](int record) -> int {
115  auto& view = states.statesViews[(int)ProcessingStateId::DATA_RELAYER_BASE + record];
116  if (view.size == 0) {
117  return 0;
118  }
119  char const* beginData = strchr(states.statesBuffer.data() + view.first, ' ') + 1;
120  // The number of elements is given by the size of the state, minus the header
121  int size = view.size - (beginData - (states.statesBuffer.data() + view.first));
122  return size;
123  };
124  auto getItem = [&states](int const& record, size_t i) -> int8_t const* {
125  static int8_t const zero = '0';
126  static int8_t const error = '4';
127  char const *buffer = states.statesBuffer.data();
128  auto& view = states.statesViews[(int)ProcessingStateId::DATA_RELAYER_BASE + record];
129  if (view.size == 0) {
130  return &zero;
131  }
132  char const* const beginData = strchr(buffer + view.first, ' ') + 1;
133  // Protect against buffer overflows
134  if (view.size <= beginData - buffer + i - view.first) {
135  return &error;
136  }
137  return (int8_t const*)beginData + i; };
138  auto getValue = [](int8_t const& item) -> int { return item - '0'; };
139  auto getColor = [](int value) {
140  static const ImU32 SLOT_EMPTY = ImColor(70, 70, 70, 255);
141  static const ImU32 SLOT_FULL = ImColor(PaletteHelpers::RED);
142  static const ImU32 SLOT_DISPATCHED = ImColor(PaletteHelpers::YELLOW);
143  static const ImU32 SLOT_DONE = ImColor(PaletteHelpers::GREEN);
144  static const ImU32 SLOT_ERROR = ImColor(0xfe, 0x43, 0x65, 255);
145  switch (value) {
146  case 0:
147  return SLOT_EMPTY;
148  case 1:
149  return SLOT_FULL;
150  case 2:
151  return SLOT_DISPATCHED;
152  case 3:
153  return SLOT_DONE;
154  }
155  return SLOT_ERROR;
156  };
157  auto describeCell = [&states](int input, int slot) -> void {
158  ImGui::BeginTooltip();
159  ImGui::Text("Input query matched values for slot: %d", slot);
160  auto& view = states.statesViews[(short)ProcessingStateId::CONTEXT_VARIABLES_BASE + (short)slot];
161  auto begin = view.first;
162  for (size_t vi = 0; vi < data_matcher::MAX_MATCHING_VARIABLE; ++vi) {
163  std::string_view state(states.statesBuffer.data() + begin, view.size);
164  // find the semi-colon, which separates entries in the variable list
165  auto pos = state.find(';');
166  std::string_view value = state.substr(0, pos);
167  // Do not display empty values.
168  if (value.empty()) {
169  begin += 1;
170  continue;
171  }
172  switch (vi) {
173  case o2::framework::data_matcher::STARTTIME_POS:
174  ImGui::Text("$%zu (startTime): %.*s", vi, (int)value.size(), value.data());
175  break;
176  case o2::framework::data_matcher::TFCOUNTER_POS:
177  ImGui::Text("$%zu (tfCounter): %.*s", vi, (int)value.size(), value.data());
178  break;
179  case o2::framework::data_matcher::FIRSTTFORBIT_POS:
180  ImGui::Text("$%zu (firstTForbit): %.*s", vi, (int)value.size(), value.data());
181  break;
182  default:
183  ImGui::Text("$%zu: %.*s", vi, (int)value.size(), value.data());
184  }
185  begin += pos + 1;
186  }
187  ImGui::EndTooltip();
188  };
189 
190  if (getNumRecords()) {
191  HeatMapHelper::draw<int, int8_t>("DataRelayer",
192  size,
193  getNumRecords,
194  getRecord,
195  getNumItems,
196  getItem,
197  getValue,
198  getColor,
199  describeCell);
200  }
201 }
