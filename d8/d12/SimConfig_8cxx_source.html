82 void SimConfig::determineActiveModules(std::vector<std::string> const& inputargs, std::vector<std::string> const& skippedModules, std::vector<std::string>& activeModules, bool isUpgrade)
83 {
84  using o2::detectors::DetID;
85 
86  // input args is a vector of module strings as obtained from the -m,--modules options
87  // of SimConfig
88  activeModules = inputargs;
89 #ifdef ENABLE_UPGRADES
90  if (activeModules[0] != "all") {
91  if (isUpgrade) {
92  for (int i = 0; i < activeModules.size(); ++i) {
93  if (activeModules[i] != "A3IP" &&
94  activeModules[i] != "IT3" &&
95  activeModules[i] != "TRK" &&
96  activeModules[i] != "FT3" &&
97  activeModules[i] != "FCT" &&
98  activeModules[i] != "TF3" &&
99  activeModules[i] != "RCH" &&
100  activeModules[i] != "MI3" &&
101  activeModules[i] != "ECL") {
102  LOGP(fatal, "List of active modules contains {}, which is not a module from the upgrades.", activeModules[i]);
103  }
104  }
105  }
106  if (!isUpgrade) {
107  for (int i = 0; i < activeModules.size(); ++i) {
108  if (activeModules[i] == "A3IP" ||
109  activeModules[i] == "TRK" ||
110  activeModules[i] == "FT3" ||
111  activeModules[i] == "FCT" ||
112  activeModules[i] == "TF3" ||
113  activeModules[i] == "RCH" ||
114  activeModules[i] == "MI3" ||
115  activeModules[i] == "ECL") {
116  LOGP(fatal, "List of active modules contains {}, which is not a run 3 module", activeModules[i]);
117  }
118  }
119  }
120  }
121 #endif
122  if (activeModules.size() == 1 && activeModules[0] == "all") {
123  activeModules.clear();
124 #ifdef ENABLE_UPGRADES
125  if (isUpgrade) {
126  for (int d = DetID::First; d <= DetID::Last; ++d) {
127  if (d == DetID::TRK ||
128  d == DetID::FT3 ||
129  d == DetID::FCT ||
130  d == DetID::TF3 ||
131  d == DetID::RCH ||
132  d == DetID::ECL ||
133  d == DetID::MI3) {
134  activeModules.emplace_back(DetID::getName(d));
135  }
136  }
137  activeModules.emplace_back("A3IP");
138  activeModules.emplace_back("A3ABSO");
139  activeModules.emplace_back("A3MAG");
140  } else {
141 #endif
142  // add passive components manually (make a PassiveDetID for them!)
143  activeModules.emplace_back("HALL");
144  activeModules.emplace_back("MAG");
145  activeModules.emplace_back("DIPO");
146  activeModules.emplace_back("COMP");
147  activeModules.emplace_back("PIPE");
148  activeModules.emplace_back("ABSO");
149  activeModules.emplace_back("SHIL");
150  for (int d = DetID::First; d <= DetID::Last; ++d) {
151 #ifdef ENABLE_UPGRADES
152  if (d != DetID::IT3 && d != DetID::TRK && d != DetID::FT3 && d != DetID::FCT && d != DetID::TF3 && d != DetID::RCH && d != DetID::ECL && d != DetID::MI3) {
153  activeModules.emplace_back(DetID::getName(d));
154  }
155  }
156 #else
157  activeModules.emplace_back(DetID::getName(d));
158 #endif
159  }
160  }
161  filterSkippedElements(activeModules, skippedModules);
162 }

164 bool SimConfig::determineActiveModulesList(const std::string& version, std::vector<std::string> const& inputargs, std::vector<std::string> const& skippedModules, std::vector<std::string>& activeModules)
165 {
166  DetectorList_t modules;
167  DetectorMap_t map;
168  if (auto pos = version.find(':'); pos != std::string::npos) {
169  auto pversion = version.substr(0, pos);
170  auto ppath = version.substr(pos + 1);
171  if (!parseDetectorMapfromJSON(ppath, map)) {
172  LOGP(error, "Could not parse {}; check errors above!", ppath);
173  return false;
174  }
175  if (map.find(pversion) == map.end()) {
176  LOGP(error, "List {} is not defined in custom JSON file!", pversion);
177  printDetMap(map);
178  return false;
179  }
180  modules = map[pversion];
181  LOGP(info, "Running with version {} from custom detector list '{}'", pversion, ppath);
182  } else {
183  // Otherwise check 'official' versions which provided in config
184  auto o2env = std::getenv("O2_ROOT");
185  if (!o2env) {
186  LOGP(error, "O2_ROOT environment not defined");
187  return false;
188  }
189  const std::string rootpath(fmt::format("{}/share/config/o2simdefaultdetectorlist.json", o2env));
190  if (!parseDetectorMapfromJSON(rootpath, map)) {
191  LOGP(error, "Could not parse {} -> check errors above!", rootpath);
192  return false;
193  }
194  if (map.find(version) == map.end()) {
195  LOGP(error, "List {} is not defined in 'official' JSON file!", version);
196  printDetMap(map);
197  return false;
198  }
199  modules = map[version];
200  LOGP(info, "Running with official detector version '{}'", version);
201  }
202  // check if specified modules are in list
203  if (inputargs.size() != 1 || inputargs[0] != "all") {
204  std::vector<std::string> diff;
205  for (const auto& in : inputargs) {
206  if (std::find(modules.begin(), modules.end(), in) == std::end(modules)) {
207  diff.emplace_back(in);
208  }
209  }
210  if (!diff.empty()) {
211  LOGP(error, "Modules specified that are not present in detector list {}", version);
212  for (int j{0}; const auto& m : diff) {
213  LOGP(info, " - {: <2}. {}", j++, m);
214  }
215  printDetMap(map, version);
216  return false;
217  }
218  }
219  // Insert into active modules if module is built buy -m or insert all if default for -m is used ("all")
220  std::copy_if(modules.begin(), modules.end(), std::back_inserter(activeModules),
221  [&inputargs](const auto& e) { return (inputargs.size() == 1 && inputargs[0] == "all") || (std::find(inputargs.begin(), inputargs.end(), e) != inputargs.end()); });
222  return filterSkippedElements(activeModules, skippedModules);
223 }

225 void SimConfig::determineReadoutDetectors(std::vector<std::string> const& activeModules, std::vector<std::string> const& enableReadout, std::vector<std::string> const& disableReadout, std::vector<std::string>& readoutDetectors)
226 {
227  using o2::detectors::DetID;
228 
229  readoutDetectors.clear();
230 
231  auto isDet = [](std::string const& s) {
232  return DetID::nameToID(s.c_str()) >= DetID::First;
233  };
234 
235  if (enableReadout.empty()) {
236  // if no readout explicitly given, use all detectors from active modules
237  for (auto& am : activeModules) {
238  if (!isDet(am)) {
239  // either we found a passive module or one with disabled readout ==> skip
240  continue;
241  }
242  readoutDetectors.emplace_back(am);
243  }
244  } else {
245  for (auto& er : enableReadout) {
246  if (!isDet(er)) {
247  // either we found a passive module or one with disabled readout ==> skip
248  LOG(fatal) << "Enabled readout for " << er << " which is not a detector.";
249  }
250  if (std::find(activeModules.begin(), activeModules.end(), er) == activeModules.end()) {
251  // add to active modules if not yet there
252  LOG(fatal) << "Module " << er << " not constructed and cannot be used for readout (make sure it is contained in -m option).";
253  }
254  readoutDetectors.emplace_back(er);
255  }
256  }
257  for (auto& dr : disableReadout) {
258  if (!isDet(dr)) {
259  // either we found a passive module or one with disabled readout ==> skip
260  LOG(fatal) << "Disabled readout for " << dr << " which is not a detector.";
261  }
262  if (std::find(activeModules.begin(), activeModules.end(), dr) == activeModules.end()) {
263  // add to active modules if not yet there
264  LOG(fatal) << "Module " << dr << " not constructed, makes no sense to disable its readout (make sure it is contained in -m option).";
265  }
266  auto iter = std::find(readoutDetectors.begin(), readoutDetectors.end(), dr);
267  if (iter != readoutDetectors.end()) {
268  readoutDetectors.erase(iter);
269  }
270  }
271 }

273 bool SimConfig::resetFromParsedMap(boost::program_options::variables_map const& vm)
274 {
275  using o2::detectors::DetID;
276  mConfigData.mMCEngine = vm["mcEngine"].as<std::string>();
277  mConfigData.mNoGeant = vm["noGeant"].as<bool>();
278 
279  // Reset modules and detectors as they are anyway re-parsed
280  mConfigData.mReadoutDetectors.clear();
281  mConfigData.mActiveModules.clear();
282 
283  // Get final set of active Modules
284  if (!determineActiveModulesList(vm["detectorList"].as<std::string>(), vm["modules"].as<std::vector<std::string>>(), vm["skipModules"].as<std::vector<std::string>>(), mConfigData.mActiveModules)) {
285  return false;
286  }
287 
288  if (mConfigData.mNoGeant) {
289  // CAVE is all that's needed (and that will be built either way), so clear all modules and set the O2TrivialMCEngine
290  mConfigData.mActiveModules.clear();
291  // force usage of O2TrivialMCEngine, no overhead from actual transport engine initialisation
292  mConfigData.mMCEngine = "O2TrivialMCEngine";
293  } else if (mConfigData.mMCEngine.compare("O2TrivialMCEngine") == 0) {
294  LOG(error) << "The O2TrivialMCEngine engine can only be used with --noGeant option";
295  return false;
296  }
297 
298  const auto& activeModules = mConfigData.mActiveModules;
299 
300  // get final set of detectors which are readout
301  determineReadoutDetectors(activeModules, vm["readoutDetectors"].as<std::vector<std::string>>(), vm["skipReadoutDetectors"].as<std::vector<std::string>>(), mConfigData.mReadoutDetectors);
302 
303  mConfigData.mGenerator = vm["generator"].as<std::string>();
304  mConfigData.mTrigger = vm["trigger"].as<std::string>();
305  mConfigData.mNEvents = vm["nEvents"].as<unsigned int>();
306  mConfigData.mExtKinFileName = vm["extKinFile"].as<std::string>();
307  mConfigData.mEmbedIntoFileName = vm["embedIntoFile"].as<std::string>();
308  mConfigData.mStartEvent = vm["startEvent"].as<unsigned int>();
309  mConfigData.mBMax = vm["bMax"].as<float>();
310  mConfigData.mIsMT = vm["isMT"].as<bool>();
311  mConfigData.mOutputPrefix = vm["outPrefix"].as<std::string>();
312  mConfigData.mLogSeverity = vm["logseverity"].as<std::string>();
313  mConfigData.mLogVerbosity = vm["logverbosity"].as<std::string>();
314  mConfigData.mKeyValueTokens = vm["configKeyValues"].as<std::string>();
315  mConfigData.mConfigFile = vm["configFile"].as<std::string>();
316  mConfigData.mPrimaryChunkSize = vm["chunkSize"].as<unsigned int>();
317  mConfigData.mInternalChunkSize = vm["chunkSizeI"].as<int>();
318  mConfigData.mStartSeed = vm["seed"].as<ULong_t>();
319  mConfigData.mSimWorkers = vm["nworkers"].as<int>();
320  if (vm.count("timestamp")) {
321  mConfigData.mTimestamp = vm["timestamp"].as<uint64_t>();
322  mConfigData.mTimestampMode = TimeStampMode::kManual;
323  } else {
324  mConfigData.mTimestamp = std::chrono::time_point_cast<std::chrono::milliseconds>(std::chrono::system_clock::now()).time_since_epoch().count();
325  mConfigData.mTimestampMode = TimeStampMode::kNow;
326  }
327  mConfigData.mRunNumber = vm["run"].as<int>();
328  mConfigData.mCCDBUrl = vm["CCDBUrl"].as<std::string>();
329  mConfigData.mAsService = vm["asservice"].as<bool>();
330  mConfigData.mForwardKine = vm["forwardKine"].as<bool>();
331  mConfigData.mWriteToDisc = !vm["noDiscOutput"].as<bool>();
332  if (vm.count("noemptyevents")) {
333  mConfigData.mFilterNoHitEvents = true;
334  }
335  mConfigData.mFromCollisionContext = vm["fromCollContext"].as<std::string>();
336  // we decompose the argument to fetch
337  // (a) collision contextfilename
338  // (b) sim prefix to use from the context
339  auto pos = mConfigData.mFromCollisionContext.find(':');
340  std::string collcontextfile{mConfigData.mFromCollisionContext};
341  std::string simprefix{mConfigData.mOutputPrefix};
342  if (pos != std::string::npos) {
343  collcontextfile = mConfigData.mFromCollisionContext.substr(0, pos);
344  simprefix = mConfigData.mFromCollisionContext.substr(pos + 1);
345  }
346  adjustFromCollContext(collcontextfile, simprefix);
347 
348  // analyse vertex options
349  if (!parseVertexModeString(vm["vertexMode"].as<std::string>(), mConfigData.mVertexMode)) {
350  return false;
351  }
352 
353  // analyse field options
354  // either: "ccdb" or +-2[U],+-5[U] and 0[U]; +-<intKGaus>U
355  auto& fieldstring = vm["field"].as<std::string>();
356  std::regex re("(ccdb)|([+-]?(0|[2-9]|[12][0-9]|20)U?)");
357  if (!std::regex_match(fieldstring, re)) {
358  LOG(error) << "Invalid field option " << fieldstring;
359  return false;
360  }
361  if (fieldstring == "ccdb") {
362  mConfigData.mFieldMode = SimFieldMode::kCCDB;
363  } else if (fieldstring.find("U") != std::string::npos) {
364  mConfigData.mFieldMode = SimFieldMode::kUniform;
365  }
366  if (fieldstring != "ccdb") {
367  mConfigData.mField = std::stoi((vm["field"].as<std::string>()).substr(0, (vm["field"].as<std::string>()).rfind("U")));
368  }
369  if (!parseFieldString(fieldstring, mConfigData.mField, mConfigData.mFieldMode)) {
370  return false;
371  }
372 
373  return true;
374 }

413 bool SimConfig::filterSkippedElements(std::vector<std::string>& elements, std::vector<std::string> const& skipped)
414 {
415  for (auto& s : skipped) {
416  if (s.empty()) { // nothing to skip here
417  continue;
418  }
419  auto iter = std::find(elements.begin(), elements.end(), s);
420  if (iter != elements.end()) {
421  // take it out
422  elements.erase(iter);
423  } else {
424  LOGP(error, "Skipped modules specified that are not present in built modules!");
425  LOGP(error, "Built modules:");
426  for (int j{0}; const auto& m : elements) {
427  LOGP(error, " + {: <2}. {}", j++, m);
428  }
429  std::vector<std::string> diff;
430  for (const auto& skip : skipped) {
431  if (std::find(elements.begin(), elements.end(), skip) == std::end(elements)) {
432  diff.emplace_back(skip);
433  }
434  }
435  LOGP(error, "Specified skipped modules not present in built modules:");
436  for (int j{0}; const auto& m : diff) {
437  LOGP(error, " - {: <2}. {}", j++, m);
438  }
439  return false;
440  }
441  }
442  return true;
443 }

445 void SimConfig::adjustFromCollContext(std::string const& collcontextfile, std::string const& prefix)
446 {
447  // When we use pregenerated collision contexts, some options
448  // need to be auto-adjusted. Do so and inform about this in the logs.
449  if (collcontextfile == "") {
450  return;
451  }
452 
453  auto context = o2::steer::DigitizationContext::loadFromFile(collcontextfile);
454  if (context) {
455  // find the events belonging to a source that corresponds to a sim prefix
456  LOG(info) << "Looking up simprefixes " << prefix;
457  int sourceid = context->findSimPrefix(prefix);
458  if (sourceid == -1) {
459  LOG(error) << "Could not find collisions with sim prefix " << prefix << " in the collision context. The collision context specifies the following prefixes:";
460  for (auto& sp : context->getSimPrefixes()) {
461  LOG(info) << sp;
462  }
463  LOG(fatal) << "Aborting due to prefix error";
464  } else {
465  auto collisionmap = context->getCollisionIndicesForSource(sourceid);
466  LOG(info) << "Found " << collisionmap.size() << " events in the collisioncontext for prefix " << prefix;
467 
468  // check if collisionmap is dense (otherwise it will get screwed up with order/indexing in ROOT output)
469  bool good = true;
470  for (auto index = 0; index < collisionmap.size(); ++index) {
471  if (collisionmap.find(index) == collisionmap.end()) {
472  good = false;
473  }
474  }
475  if (!good) {
476  LOG(fatal) << "events in collisioncontext are non-compact ";
477  }
478 
479  // do some adjustments based on the number of events to be simulated
