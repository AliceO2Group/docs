82void SimConfig::determineActiveModules(std::vector<std::string> const& inputargs, std::vector<std::string> const& skippedModules, std::vector<std::string>& activeModules, bool isUpgrade)
83{
84 using o2::detectors::DetID;
85
86 // input args is a vector of module strings as obtained from the -m,--modules options
87 // of SimConfig
88 activeModules = inputargs;
89#ifdef ENABLE_UPGRADES
90 if (activeModules[0] != "all") {
91 if (isUpgrade) {
92 for (int i = 0; i < activeModules.size(); ++i) {
93 if (activeModules[i] != "A3IP" &&
94 activeModules[i] != "IT3" &&
95 activeModules[i] != "TRK" &&
96 activeModules[i] != "FT3" &&
97 activeModules[i] != "FCT" &&
98 activeModules[i] != "TF3" &&
99 activeModules[i] != "RCH" &&
100 activeModules[i] != "MI3" &&
101 activeModules[i] != "ECL" &&
102 activeModules[i] != "FD3") {
103 LOGP(fatal, "List of active modules contains {}, which is not a module from the upgrades.", activeModules[i]);
104 }
105 }
106 }
107 if (!isUpgrade) {
108 for (int i = 0; i < activeModules.size(); ++i) {
109 if (activeModules[i] == "A3IP" ||
110 activeModules[i] == "TRK" ||
111 activeModules[i] == "FT3" ||
112 activeModules[i] == "FCT" ||
113 activeModules[i] == "TF3" ||
114 activeModules[i] == "RCH" ||
115 activeModules[i] == "MI3" ||
116 activeModules[i] == "ECL" ||
117 activeModules[i] == "FD3") {
118 LOGP(fatal, "List of active modules contains {}, which is not a run 3 module", activeModules[i]);
119 }
120 }
121 }
122 }
123#endif
124 if (activeModules.size() == 1 && activeModules[0] == "all") {
125 activeModules.clear();
126#ifdef ENABLE_UPGRADES
127 if (isUpgrade) {
128 for (int d = DetID::First; d <= DetID::Last; ++d) {
129 if (d == DetID::TRK ||
130 d == DetID::FT3 ||
131 d == DetID::FCT ||
132 d == DetID::TF3 ||
133 d == DetID::RCH ||
134 d == DetID::ECL ||
135 d == DetID::FD3 ||
136 d == DetID::MI3) {
137 activeModules.emplace_back(DetID::getName(d));
138 }
139 }
140 activeModules.emplace_back("A3IP");
141 activeModules.emplace_back("A3ABSO");
142 activeModules.emplace_back("A3MAG");
143 } else {
144#endif
145 // add passive components manually (make a PassiveDetID for them!)
146 activeModules.emplace_back("HALL");
147 activeModules.emplace_back("MAG");
148 activeModules.emplace_back("DIPO");
149 activeModules.emplace_back("COMP");
150 activeModules.emplace_back("PIPE");
151 activeModules.emplace_back("ABSO");
152 activeModules.emplace_back("SHIL");
153 for (int d = DetID::First; d <= DetID::Last; ++d) {
154#ifdef ENABLE_UPGRADES
155 if (d != DetID::IT3 && d != DetID::TRK && d != DetID::FT3 && d != DetID::FCT && d != DetID::TF3 && d != DetID::RCH && d != DetID::ECL && d != DetID::FD3 && d != DetID::MI3) {
156 activeModules.emplace_back(DetID::getName(d));
157 }
158 }
159#else
160 activeModules.emplace_back(DetID::getName(d));
161#endif
162 }
163 }
164 filterSkippedElements(activeModules, skippedModules);
165}

167bool SimConfig::determineActiveModulesList(const std::string& version, std::vector<std::string> const& inputargs, std::vector<std::string> const& skippedModules, std::vector<std::string>& activeModules)
168{
169 DetectorList_t modules;
170 DetectorMap_t map;
171 if (auto pos = version.find(':'); pos != std::string::npos) {
172 auto pversion = version.substr(0, pos);
173 auto ppath = version.substr(pos + 1);
174 if (!parseDetectorMapfromJSON(ppath, map)) {
175 LOGP(error, "Could not parse {}; check errors above!", ppath);
176 return false;
177 }
178 if (map.find(pversion) == map.end()) {
179 LOGP(error, "List {} is not defined in custom JSON file!", pversion);
180 printDetMap(map);
181 return false;
182 }
183 modules = map[pversion];
184 LOGP(info, "Running with version {} from custom detector list '{}'", pversion, ppath);
185 } else {
186 // Otherwise check 'official' versions which provided in config
187 auto o2env = std::getenv("O2_ROOT");
188 if (!o2env) {
189 LOGP(error, "O2_ROOT environment not defined");
190 return false;
191 }
192 const std::string rootpath(fmt::format("{}/share/config/o2simdefaultdetectorlist.json", o2env));
193 if (!parseDetectorMapfromJSON(rootpath, map)) {
194 LOGP(error, "Could not parse {} -> check errors above!", rootpath);
195 return false;
196 }
197 if (map.find(version) == map.end()) {
198 LOGP(error, "List {} is not defined in 'official' JSON file!", version);
199 printDetMap(map);
200 return false;
201 }
202 modules = map[version];
203 LOGP(info, "Running with official detector version '{}'", version);
204 }
205 // check if specified modules are in list
206 if (inputargs.size() != 1 || inputargs[0] != "all") {
207 std::vector<std::string> diff;
208 for (const auto& in : inputargs) {
209 if (std::find(modules.begin(), modules.end(), in) == std::end(modules)) {
210 diff.emplace_back(in);
211 }
212 }
213 if (!diff.empty()) {
214 LOGP(error, "Modules specified that are not present in detector list {}", version);
215 for (int j{0}; const auto& m : diff) {
216 LOGP(info, " - {: <2}. {}", j++, m);
217 }
218 printDetMap(map, version);
219 return false;
220 }
221 }
222 // Insert into active modules if module is built buy -m or insert all if default for -m is used ("all")
223 std::copy_if(modules.begin(), modules.end(), std::back_inserter(activeModules),
224 [&inputargs](const auto& e) { return (inputargs.size() == 1 && inputargs[0] == "all") || (std::find(inputargs.begin(), inputargs.end(), e) != inputargs.end()); });
225 return filterSkippedElements(activeModules, skippedModules);
226}

228void SimConfig::determineReadoutDetectors(std::vector<std::string> const& activeModules, std::vector<std::string> const& enableReadout, std::vector<std::string> const& disableReadout, std::vector<std::string>& readoutDetectors)
229{
230 using o2::detectors::DetID;
231
232 readoutDetectors.clear();
233
234 auto isDet = [](std::string const& s) {
235 return DetID::nameToID(s.c_str()) >= DetID::First;
236 };
237
238 if (enableReadout.empty()) {
239 // if no readout explicitly given, use all detectors from active modules
240 for (auto& am : activeModules) {
241 if (!isDet(am)) {
242 // either we found a passive module or one with disabled readout ==> skip
243 continue;
244 }
245 readoutDetectors.emplace_back(am);
246 }
247 } else {
248 for (auto& er : enableReadout) {
249 if (!isDet(er)) {
250 // either we found a passive module or one with disabled readout ==> skip
251 LOG(fatal) << "Enabled readout for " << er << " which is not a detector.";
252 }
253 if (std::find(activeModules.begin(), activeModules.end(), er) == activeModules.end()) {
254 // add to active modules if not yet there
255 LOG(fatal) << "Module " << er << " not constructed and cannot be used for readout (make sure it is contained in -m option).";
256 }
257 readoutDetectors.emplace_back(er);
258 }
259 }
260 for (auto& dr : disableReadout) {
261 if (!isDet(dr)) {
262 // either we found a passive module or one with disabled readout ==> skip
263 LOG(fatal) << "Disabled readout for " << dr << " which is not a detector.";
264 }
265 if (std::find(activeModules.begin(), activeModules.end(), dr) == activeModules.end()) {
266 // add to active modules if not yet there
267 LOG(fatal) << "Module " << dr << " not constructed, makes no sense to disable its readout (make sure it is contained in -m option).";
268 }
269 auto iter = std::find(readoutDetectors.begin(), readoutDetectors.end(), dr);
270 if (iter != readoutDetectors.end()) {
271 readoutDetectors.erase(iter);
272 }
273 }
274}

276std::pair<std::string, std::string> SimConfig::getCollContextFilenameAndEventPrefix() const
277{
278 // we decompose the argument to fetch
279 // (a) collision contextfilename
280 // (b) sim prefix to use from the context
281 auto pos = mConfigData.mFromCollisionContext.find(':');
282 std::string collcontextfile{mConfigData.mFromCollisionContext};
283 std::string simprefix{mConfigData.mOutputPrefix};
284 if (pos != std::string::npos) {
285 collcontextfile = mConfigData.mFromCollisionContext.substr(0, pos);
286 simprefix = mConfigData.mFromCollisionContext.substr(pos + 1);
287 }
288 return std::make_pair(collcontextfile, simprefix);
289}

291bool SimConfig::resetFromParsedMap(boost::program_options::variables_map const& vm)
292{
293 using o2::detectors::DetID;
294 mConfigData.mMCEngine = vm["mcEngine"].as<std::string>();
295 mConfigData.mNoGeant = vm["noGeant"].as<bool>();
296
297 // Reset modules and detectors as they are anyway re-parsed
298 mConfigData.mReadoutDetectors.clear();
299 mConfigData.mActiveModules.clear();
300
301 // Get final set of active Modules
302 if (!determineActiveModulesList(vm["detectorList"].as<std::string>(), vm["modules"].as<std::vector<std::string>>(), vm["skipModules"].as<std::vector<std::string>>(), mConfigData.mActiveModules)) {
303 return false;
304 }
305
306 if (mConfigData.mNoGeant) {
307 // CAVE is all that's needed (and that will be built either way), so clear all modules and set the O2TrivialMCEngine
308 mConfigData.mActiveModules.clear();
309 // force usage of O2TrivialMCEngine, no overhead from actual transport engine initialisation
310 mConfigData.mMCEngine = "O2TrivialMCEngine";
311 } else if (mConfigData.mMCEngine.compare("O2TrivialMCEngine") == 0) {
312 LOG(error) << "The O2TrivialMCEngine engine can only be used with --noGeant option";
313 return false;
314 }
315
316 const auto& activeModules = mConfigData.mActiveModules;
317
318 // get final set of detectors which are readout
319 determineReadoutDetectors(activeModules, vm["readoutDetectors"].as<std::vector<std::string>>(), vm["skipReadoutDetectors"].as<std::vector<std::string>>(), mConfigData.mReadoutDetectors);
320
321 mConfigData.mGenerator = vm["generator"].as<std::string>();
322 mConfigData.mTrigger = vm["trigger"].as<std::string>();
323 mConfigData.mNEvents = vm["nEvents"].as<unsigned int>();
324 mConfigData.mExtKinFileName = vm["extKinFile"].as<std::string>();
325 mConfigData.mEmbedIntoFileName = vm["embedIntoFile"].as<std::string>();
326 mConfigData.mStartEvent = vm["startEvent"].as<unsigned int>();
327 mConfigData.mBMax = vm["bMax"].as<float>();
328 mConfigData.mIsMT = vm["isMT"].as<bool>();
329 mConfigData.mOutputPrefix = vm["outPrefix"].as<std::string>();
330 mConfigData.mLogSeverity = vm["logseverity"].as<std::string>();
331 mConfigData.mLogVerbosity = vm["logverbosity"].as<std::string>();
332 mConfigData.mKeyValueTokens = vm["configKeyValues"].as<std::string>();
333 mConfigData.mConfigFile = vm["configFile"].as<std::string>();
334 mConfigData.mPrimaryChunkSize = vm["chunkSize"].as<unsigned int>();
335 mConfigData.mInternalChunkSize = vm["chunkSizeI"].as<int>();
336 mConfigData.mStartSeed = vm["seed"].as<ULong_t>();
337 mConfigData.mSimWorkers = vm["nworkers"].as<int>();
338 if (vm.count("timestamp")) {
339 mConfigData.mTimestamp = vm["timestamp"].as<uint64_t>();
340 mConfigData.mTimestampMode = TimeStampMode::kManual;
341 } else {
342 mConfigData.mTimestamp = std::chrono::time_point_cast<std::chrono::milliseconds>(std::chrono::system_clock::now()).time_since_epoch().count();
343 mConfigData.mTimestampMode = TimeStampMode::kNow;
344 }
345 mConfigData.mRunNumber = vm["run"].as<int>();
346 mConfigData.mCCDBUrl = vm["CCDBUrl"].as<std::string>();
347 mConfigData.mAsService = vm["asservice"].as<bool>();
348 mConfigData.mForwardKine = vm["forwardKine"].as<bool>();
349 mConfigData.mWriteToDisc = !vm["noDiscOutput"].as<bool>();
350 if (vm.count("noemptyevents")) {
351 mConfigData.mFilterNoHitEvents = true;
352 }
353 mConfigData.mFromCollisionContext = vm["fromCollContext"].as<std::string>();
354 auto collcontext_simprefix = getCollContextFilenameAndEventPrefix();
355 adjustFromCollContext(collcontext_simprefix.first, collcontext_simprefix.second);
356
357 // analyse vertex options
358 if (!parseVertexModeString(vm["vertexMode"].as<std::string>(), mConfigData.mVertexMode)) {
359 return false;
360 }
361
362 // analyse field options
363 // either: "ccdb" or +-2[U],+-5[U] and 0[U]; +-<intKGaus>U
364 auto& fieldstring = vm["field"].as<std::string>();
365 std::regex re("(ccdb)|([+-]?(0|[2-9]|[12][0-9]|20)U?)");
366 if (!std::regex_match(fieldstring, re)) {
367 LOG(error) << "Invalid field option " << fieldstring;
368 return false;
369 }
370 if (fieldstring == "ccdb") {
371 mConfigData.mFieldMode = SimFieldMode::kCCDB;
372 } else if (fieldstring.find("U") != std::string::npos) {
373 mConfigData.mFieldMode = SimFieldMode::kUniform;
374 }
375 if (fieldstring != "ccdb") {
376 mConfigData.mField = std::stoi((vm["field"].as<std::string>()).substr(0, (vm["field"].as<std::string>()).rfind("U")));
377 }
378 if (!parseFieldString(fieldstring, mConfigData.mField, mConfigData.mFieldMode)) {
379 return false;
380 }
381
382 return true;
383}

385bool SimConfig::parseVertexModeString(std::string const& vertexstring, VertexMode& mode)
386{
387 // vertexstring must be either kNoVertex, kDiamondParam, kCCDB
388 if (vertexstring == "kNoVertex") {
389 mode = VertexMode::kNoVertex;
390 return true;
391 } else if (vertexstring == "kDiamondParam") {
392 mode = VertexMode::kDiamondParam;
393 return true;
394 } else if (vertexstring == "kCCDB") {
395 mode = VertexMode::kCCDB;
396 return true;
397 } else if (vertexstring == "kCollContext") {
398 mode = VertexMode::kCollCxt;
399 return true;
400 }
401 LOG(error) << "Vertex mode must be one of kNoVertex, kDiamondParam, kCCDB, kCollContext";
402 return false;
403}

406{
407 // analyse field options
408 // either: "ccdb" or +-2[U],+-5[U] and 0[U]; +-<intKGaus>U
409 std::regex re("(ccdb)|([+-]?(0|[2-9]|[12][0-9]|20)U?)");

411 LOG(error) << "Invalid field option " << fieldstring;

418 }
419 if (fieldstring != "ccdb") {
420 fieldvalue = std::stoi(fieldstring.substr(0, fieldstring.rfind("U")));
421 }
422 return true;
423}

425bool SimConfig::filterSkippedElements(std::vector<std::string>& elements, std::vector<std::string> const& skipped)
426{
427 for (auto& s : skipped) {
428 if (s.empty()) { // nothing to skip here
429 continue;
430 }
431 auto iter = std::find(elements.begin(), elements.end(), s);
432 if (iter != elements.end()) {
433 // take it out
434 elements.erase(iter);
435 } else {
436 LOGP(error, "Skipped modules specified that are not present in built modules!");
437 LOGP(error, "Built modules:");
438 for (int j{0}; const auto& m : elements) {
439 LOGP(error, " + {: <2}. {}", j++, m);
440 }
441 std::vector<std::string> diff;
442 for (const auto& skip : skipped) {
443 if (std::find(elements.begin(), elements.end(), skip) == std::end(elements)) {
444 diff.emplace_back(skip);
445 }
446 }
447 LOGP(error, "Specified skipped modules not present in built modules:");
448 for (int j{0}; const auto& m : diff) {
449 LOGP(error, " - {: <2}. {}", j++, m);
450 }
451 return false;
452 }
453 }
454 return true;
455}

457void SimConfig::adjustFromCollContext(std::string const& collcontextfile, std::string const& prefix)
458{
459 // When we use pregenerated collision contexts, some options
460 // need to be auto-adjusted. Do so and inform about this in the logs.
461 if (collcontextfile == "") {
462 return;
463 }
464
465 auto context = o2::steer::DigitizationContext::loadFromFile(collcontextfile);
466 if (context) {
467 // find the events belonging to a source that corresponds to a sim prefix
468 LOG(info) << "Looking up simprefixes " << prefix;
469 int sourceid = context->findSimPrefix(prefix);
470 if (sourceid == -1) {
471 LOG(error) << "Could not find collisions with sim prefix " << prefix << " in the collision context. The collision context specifies the following prefixes:";
472 for (auto& sp : context->getSimPrefixes()) {
473 LOG(info) << sp;
474 }
475 LOG(fatal) << "Aborting due to prefix error";
476 } else {
477 auto collisionmap = context->getCollisionIndicesForSource(sourceid);
478 LOG(info) << "Found " << collisionmap.size() << " events in the collisioncontext for prefix " << prefix;
479
480 // check if collisionmap is dense (otherwise it will get screwed up with order/indexing in ROOT output)
481 bool good = true;
482 for (auto index = 0; index < collisionmap.size(); ++index) {
483 if (collisionmap.find(index) == collisionmap.end()) {
484 good = false;
485 }
486 }
487 if (!good) {
488 LOG(fatal) << "events in collisioncontext are non-compact ";
489 }
490
491 // do some adjustments based on the number of events to be simulated
492 if (mConfigData.mNEvents == 0 || mConfigData.mNEvents == collisionmap.size()) {
493 // we take what is specified in the context
494 mConfigData.mNEvents = collisionmap.size();
495 } else {
496 LOG(warning) << "The number of events on the command line " << mConfigData.mNEvents << " and in the collision context differ. We take the one from collision context " << collisionmap.size();
497 mConfigData.mNEvents = collisionmap.size();
498 }
499 LOG(info) << "Setting number of events to simulate to " << mConfigData.mNEvents;
500 }
501 } else {
502 LOG(fatal) << "Could not open collision context file " << collcontextfile;
503 }
504}

506bool SimConfig::resetFromArguments(int argc, char* argv[])
507{
508 // Arguments parsing
