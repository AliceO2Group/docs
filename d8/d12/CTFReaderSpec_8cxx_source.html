62 template <typename T>
63 bool readFromTree(TTree& tree, const std::string brname, T& dest, int ev = 0)
64 {
65  auto* br = tree.GetBranch(brname.c_str());
66  if (br && br->GetEntries() > ev) {
67  auto* ptr = &dest;
68  br->SetAddress(&ptr);
69  br->GetEntry(ev);
70  br->ResetAddress();
71  return true;
72  }
73  return false;
74 }

135 void CTFReaderSpec::stopReader()
136 {
137  if (!mFileFetcher) {
138  return;
139  }
140  LOGP(info, "CTFReader stops processing, {} files read, {} files failed", mFilesRead - mNFailedFiles, mNFailedFiles);
141  LOGP(info, "CTF reading total timing: Cpu: {:.3f} Real: {:.3f} s for {} TFs ({} accepted) in {} loops, spent {:.2} s in {} data waiting states",
142  mTimer.CpuTime(), mTimer.RealTime(), mCTFCounter, mCTFCounterAcc, mFileFetcher->getNLoops(), 1e-6 * mTotalWaitTime, mNWaits);
143  mRunning = false;
144  mFileFetcher->stop();
145  mFileFetcher.reset();
146  mCTFTree.reset();
147  if (mCTFFile) {
148  mCTFFile->Close();
149  }
150  mCTFFile.reset();

181 void CTFReaderSpec::runTimeRangesToIRFrameSelector(const o2::framework::TimingInfo& timingInfo)
182 {
183  // convert entries in the runTimeRanges to IRFrameSelector, if needed, convert time to orbit
184  mIRFrameSelector.clear();
185  auto ent = mRunTimeRanges.find(timingInfo.runNumber);
186  if (ent == mRunTimeRanges.end()) {
187  LOGP(info, "RunTimeRanges selection was provided but run {} has no entries, all TFs will be processed", timingInfo.runNumber);
188  return;
189  }
190  o2::parameters::AggregatedRunInfo rinfo;
191  auto& ccdb = o2::ccdb::BasicCCDBManager::instance();
192  rinfo = o2::parameters::AggregatedRunInfo::buildAggregatedRunInfo(ccdb, timingInfo.runNumber);
193  if (rinfo.runNumber != timingInfo.runNumber || rinfo.orbitsPerTF < 1) {
194  LOGP(fatal, "failed to extract AggregatedRunInfo for run {}", timingInfo.runNumber);
195  }
196  mTFLength = rinfo.orbitsPerTF;
197  std::vector<o2::dataformats::IRFrame> frames;
198  for (const auto& rng : ent->second) {
199  long orbMin = 0, orbMax = 0;
200  if (mConvRunTimeRangesToOrbits > 0) {
201  orbMin = rinfo.orbitSOR + (rng.first - rinfo.sor) / (o2::constants::lhc::LHCOrbitMUS * 0.001);
202  orbMax = rinfo.orbitSOR + (rng.second - rinfo.sor) / (o2::constants::lhc::LHCOrbitMUS * 0.001);
203  } else {
204  orbMin = rng.first;
205  orbMax = rng.second;
206  }
207  if (orbMin < 0) {
208  orbMin = 0;
209  }
210  if (orbMax < 0) {
211  orbMax = 0;
212  }
213  if (timingInfo.runNumber > 523897) {
214  orbMin = (orbMin / rinfo.orbitsPerTF) * rinfo.orbitsPerTF;
215  orbMax = (orbMax / rinfo.orbitsPerTF + 1) * rinfo.orbitsPerTF - 1;
216  }
217  LOGP(info, "TFs overlapping with orbits {}:{} will be {}", orbMin, orbMax, mInput.invertIRFramesSelection ? "rejected" : "selected");
218  frames.emplace_back(InteractionRecord{0, uint32_t(orbMin)}, InteractionRecord{o2::constants::lhc::LHCMaxBunches, uint32_t(orbMax)});
219  }
220  mIRFrameSelector.setOwnList(frames, true);
221 }

223 void CTFReaderSpec::loadRunTimeSpans(const std::string& flname)
224 {
225  std::ifstream inputFile(flname);
226  if (!inputFile) {
227  LOGP(fatal, "Failed to open selected run/timespans file {}", mInput.fileRunTimeSpans);
228  }
229  std::string line;
230  size_t cntl = 0, cntr = 0;
231  while (std::getline(inputFile, line)) {
232  cntl++;
233  for (char& ch : line) { // Replace semicolons and tabs with spaces for uniform processing
234  if (ch == ';' || ch == '\t' || ch == ',') {
235  ch = ' ';
236  }
237  }
238  o2::utils::Str::trim(line);
239  if (line.size() < 1 || line[0] == '#') {
240  continue;
241  }
242  auto tokens = o2::utils::Str::tokenize(line, ' ');
243  auto logError = [&cntl, &line]() { LOGP(error, "Expected format for selection is tripplet <run> <range_min> <range_max>, failed on line#{}: {}", cntl, line); };
244  if (tokens.size() >= 3) {
245  int run = 0;
246  long rmin, rmax;
247  try {
248  run = std::stoi(tokens[0]);
249  rmin = std::stol(tokens[1]);
250  rmax = std::stol(tokens[2]);
251  } catch (...) {
252  logError();
253  continue;
254  }
255 
256  constexpr long ISTimeStamp = 1514761200000L;
257  int convmn = rmin > ISTimeStamp ? 1 : 0, convmx = rmax > ISTimeStamp ? 1 : 0; // values above ISTimeStamp are timestamps (need to be converted to orbits)
258  if (rmin > rmax) {
259  LOGP(fatal, "Provided range limits are not in increasing order, entry is {}", line);
260  }
261  if (mConvRunTimeRangesToOrbits == -1) {
262  if (convmn != convmx) {
263  LOGP(fatal, "Provided range limits should be both consistent either with orbit number or with unix timestamp in ms, entry is {}", line);
264  }
265  mConvRunTimeRangesToOrbits = convmn; // need to convert to orbit if time
266  LOGP(info, "Interpret selected time-spans input as {}", mConvRunTimeRangesToOrbits == 1 ? "timstamps(ms)" : "orbits");
267  } else {
268  if (mConvRunTimeRangesToOrbits != convmn || mConvRunTimeRangesToOrbits != convmx) {
269  LOGP(fatal, "Provided range limits should are not consistent with previously determined {} input, entry is {}", mConvRunTimeRangesToOrbits == 1 ? "timestamps" : "orbits", line);
270  }
271  }
272 
273  mRunTimeRanges[run].emplace_back(rmin, rmax);
274  cntr++;
275  } else {
276  logError();
277  }
278  }
279  LOGP(info, "Read {} time-spans for {} runs from {}", cntr, mRunTimeRanges.size(), mInput.fileRunTimeSpans);
280  inputFile.close();
281 }

284 void CTFReaderSpec::openCTFFile(const std::string& flname)
285 {
286  try {
287  mFilesRead++;
288  mCTFFile.reset(TFile::Open(flname.c_str()));
289  if (!mCTFFile || !mCTFFile->IsOpen() || mCTFFile->IsZombie()) {
290  throw std::runtime_error(fmt::format("failed to open CTF file {}, skipping", flname));
291  }
292  mCTFTree.reset((TTree*)mCTFFile->Get(std::string(o2::base::NameConf::CTFTREENAME).c_str()));
293  if (!mCTFTree) {
294  throw std::runtime_error(fmt::format("failed to load CTF tree from {}, skipping", flname));
295  }
296  if (mCTFTree->GetEntries() < 1) {
297  throw std::runtime_error(fmt::format("CTF tree in {} has 0 entries, skipping", flname));
298  }
299  } catch (const std::exception& e) {
300  LOG(error) << "Cannot process " << flname << ", reason: " << e.what();
301  mCTFTree.reset();
302  mCTFFile.reset();
303  mNFailedFiles++;
304  if (mFileFetcher) {
305  mFileFetcher->popFromQueue(mInput.maxLoops < 1);
306  }
307  }
308  mCurrTreeEntry = 0;
309 }

312 void CTFReaderSpec::run(ProcessingContext& pc)
313 {
314  if (mInput.tfRateLimit == -999) {
315  mInput.tfRateLimit = std::stoi(pc.services().get<RawDeviceService>().device()->fConfig->GetValue<std::string>("timeframes-rate-limit"));
316  }
317  std::string tfFileName;
318  bool waitAcknowledged = false;
319  long startWait = 0;
320 
321  while (mRunning) {
322  if (mCTFTree) { // there is a tree open with multiple CTF
323  if (mInput.ctfIDs.empty() || mInput.ctfIDs[mSelIDEntry] == mCTFCounter) { // no selection requested or matching CTF ID is found
324  LOG(debug) << "TF " << mCTFCounter << " of " << mInput.maxTFs << " loop " << mFileFetcher->getNLoops();
325  mSelIDEntry++;
326  if (processTF(pc)) {
327  break;
328  }
329  }
330  // explict CTF ID selection list or IRFrame was provided and current entry is not selected
331  LOGP(info, "Skipping CTF#{} ({} of {} in {})", mCTFCounter, mCurrTreeEntry, mCTFTree->GetEntries(), mCTFFile->GetName());
332  checkTreeEntries();
333  mCTFCounter++;
334  continue;
335  }
336  //
337  tfFileName = mFileFetcher->getNextFileInQueue();
338  if (tfFileName.empty()) {
339  if (!mFileFetcher->isRunning()) { // nothing expected in the queue
340  mRunning = false;
341  break;
342  }
343  if (!waitAcknowledged) {
344  startWait = std::chrono::time_point_cast<std::chrono::microseconds>(std::chrono::system_clock::now()).time_since_epoch().count();
345  waitAcknowledged = true;
346  }
347  pc.services().get<RawDeviceService>().waitFor(5);
348  continue;
349  }
350  if (waitAcknowledged) {
351  long waitTime = std::chrono::time_point_cast<std::chrono::microseconds>(std::chrono::system_clock::now()).time_since_epoch().count() - startWait;
352  mTotalWaitTime += waitTime;
353  if (++mNWaits > 1) {
354  LOGP(warn, "Resuming reading after waiting for data {:.2} s (accumulated {:.2} s delay in {} waits)", 1e-6 * waitTime, 1e-6 * mTotalWaitTime, mNWaits);
355  }
356  waitAcknowledged = false;
357  }
358  LOG(info) << "Reading CTF input " << ' ' << tfFileName;
359  openCTFFile(tfFileName);
360  }
361 
362  if (mCTFCounter >= mInput.maxTFs || (!mInput.ctfIDs.empty() && mSelIDEntry >= mInput.ctfIDs.size())) { // done
363  LOGP(info, "All CTFs from selected range were injected, stopping");
364  mRunning = false;
365  } else if (mRunning && !mCTFTree && mFileFetcher->getNextFileInQueue().empty() && !mFileFetcher->isRunning()) { // previous tree was done, can we read more?
366  mRunning = false;
367  }
368 
369  if (!mRunning) {
370  pc.services().get<ControlService>().endOfStream();
371  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
372  stopReader();
373  const std::string dummy{"ctf_read_ntf.txt"};
374  if (mCTFCounterAcc == 0) {
375  LOGP(warn, "No TF passed selection, writing a 0 to file {}", dummy);
376  }
377  try {
378  std::ofstream outfile;
379  outfile.open(dummy, std::ios::out | std::ios::trunc);
380  outfile << mCTFCounterAcc << std::endl;
381  } catch (...) {
382  LOGP(error, "Failed to write {}", dummy);
383  }
384  }
385 }

388 bool CTFReaderSpec::processTF(ProcessingContext& pc)
389 {
390  auto cput = mTimer.CpuTime();
391  mTimer.Start(false);
392 
393  static RateLimiter limiter;
394  CTFHeader ctfHeader;
395  if (!readFromTree(*(mCTFTree.get()), "CTFHeader", ctfHeader, mCurrTreeEntry)) {
396  throw std::runtime_error("did not find CTFHeader");
397  }
398  if (mImposeRunStartMS > 0) {
399  ctfHeader.creationTime = mImposeRunStartMS + ctfHeader.firstTForbit * o2::constants::lhc::LHCOrbitMUS * 1e-3;
400  }
401  if (ctfHeader.creationTime == 0) { // try to repair header with ad hoc data
402  tryToFixCTFHeader(ctfHeader);
403  }
404 
405  if (mUseLocalTFCounter) {
406  ctfHeader.tfCounter = mCTFCounterAcc;
407  }
408 
409  LOG(info) << ctfHeader;
410 
411  auto& timingInfo = pc.services().get<o2::framework::TimingInfo>();
412  timingInfo.firstTForbit = ctfHeader.firstTForbit;
413  timingInfo.creation = ctfHeader.creationTime;
414  timingInfo.tfCounter = ctfHeader.tfCounter;
415  timingInfo.runNumber = ctfHeader.run;
416 
417  if (mRunTimeRanges.size() && timingInfo.runNumber != mRunNumberPrev) {
418  runTimeRangesToIRFrameSelector(timingInfo);
419  }
420  mRunNumberPrev = timingInfo.runNumber;
421 
422  if (mIRFrameSelector.isSet()) {
423  o2::InteractionRecord ir0(0, timingInfo.firstTForbit);
424  o2::InteractionRecord ir1(o2::constants::lhc::LHCMaxBunches - 1, timingInfo.firstTForbit < 0xffffffff - (mTFLength - 1) ? timingInfo.firstTForbit + (mTFLength - 1) : 0xffffffff);
425  auto irSpan = mIRFrameSelector.getMatchingFrames({ir0, ir1});
426  bool acc = true;
427  if (mInput.skipSkimmedOutTF) {
428  acc = (irSpan.size() > 0) ? !mInput.invertIRFramesSelection : mInput.invertIRFramesSelection;
429  LOGP(info, "IRFrame selection contains {} frames for TF [{}] : [{}]: {}use this TF (selection inversion mode is {})",
430  irSpan.size(), ir0.asString(), ir1.asString(), acc ? "" : "do not ", mInput.invertIRFramesSelection ? "ON" : "OFF");
431  }
432  if (!acc) {
433  return false;
434  }
435  if (mInput.checkTFLimitBeforeReading) {
436  limiter.check(pc, mInput.tfRateLimit, mInput.minSHM);
437  }
438  auto outVec = pc.outputs().make<std::vector<o2::dataformats::IRFrame>>(OutputRef{"selIRFrames"}, irSpan.begin(), irSpan.end());
439  } else {
440  if (mInput.checkTFLimitBeforeReading) {
441  limiter.check(pc, mInput.tfRateLimit, mInput.minSHM);
442  }
443  }
444 
445  // send CTF Header
446  pc.outputs().snapshot({"header", mInput.subspec}, ctfHeader);
447 
448  processDetector<o2::itsmft::CTF>(DetID::ITS, ctfHeader, pc);
449  processDetector<o2::itsmft::CTF>(DetID::MFT, ctfHeader, pc);
450  processDetector<o2::emcal::CTF>(DetID::EMC, ctfHeader, pc);
451  processDetector<o2::hmpid::CTF>(DetID::HMP, ctfHeader, pc);
452  processDetector<o2::phos::CTF>(DetID::PHS, ctfHeader, pc);
453  processDetector<o2::tpc::CTF>(DetID::TPC, ctfHeader, pc);
454  processDetector<o2::trd::CTF>(DetID::TRD, ctfHeader, pc);
455  processDetector<o2::ft0::CTF>(DetID::FT0, ctfHeader, pc);
456  processDetector<o2::fv0::CTF>(DetID::FV0, ctfHeader, pc);
457  processDetector<o2::fdd::CTF>(DetID::FDD, ctfHeader, pc);
458  processDetector<o2::tof::CTF>(DetID::TOF, ctfHeader, pc);
459  processDetector<o2::mid::CTF>(DetID::MID, ctfHeader, pc);
460  processDetector<o2::mch::CTF>(DetID::MCH, ctfHeader, pc);
461  processDetector<o2::cpv::CTF>(DetID::CPV, ctfHeader, pc);
462  processDetector<o2::zdc::CTF>(DetID::ZDC, ctfHeader, pc);
463  processDetector<o2::ctp::CTF>(DetID::CTP, ctfHeader, pc);
464  mCTFCounterAcc++;
465 
466  // send sTF acknowledge message
467  if (!mInput.sup0xccdb) {
468  auto& stfDist = pc.outputs().make<o2::header::STFHeader>(OutputRef{"TFDist", 0xccdb});
469  stfDist.id = uint64_t(mCurrTreeEntry);
470  stfDist.firstOrbit = ctfHeader.firstTForbit;
471  stfDist.runNumber = uint32_t(ctfHeader.run);
472  }
473 
474  auto entryStr = fmt::format("({} of {} in {})", mCurrTreeEntry, mCTFTree->GetEntries(), mCTFFile->GetName());
475  checkTreeEntries();
476  mTimer.Stop();
477 
478  // do we need to wait to respect the delay ?
479  long tNow = std::chrono::time_point_cast<std::chrono::microseconds>(std::chrono::system_clock::now()).time_since_epoch().count();
480  if (mCTFCounter) {
481  auto tDiff = tNow - mLastSendTime;
482  if (tDiff < mInput.delay_us) {
483  pc.services().get<RawDeviceService>().waitFor((mInput.delay_us - tDiff) / 1000); // respect requested delay before sending
484  }
485  }
486  if (!mInput.checkTFLimitBeforeReading) {
487  limiter.check(pc, mInput.tfRateLimit, mInput.minSHM);
488  }
489  tNow = std::chrono::time_point_cast<std::chrono::microseconds>(std::chrono::system_clock::now()).time_since_epoch().count();
490  LOGP(info, "Read CTF {} {} in {:.3f} s, {:.4f} s elapsed from previous CTF", mCTFCounter, entryStr, mTimer.CpuTime() - cput, mCTFCounter ? 1e-6 * (tNow - mLastSendTime) : 0.);
491  mLastSendTime = tNow;
492  mCTFCounter++;
493  return true;
494 }

497 void CTFReaderSpec::checkTreeEntries()
498 {

500  if (++mCurrTreeEntry >= mCTFTree->GetEntries() || (mInput.maxTFsPerFile > 0 && mCurrTreeEntry >= mInput.maxTFsPerFile)) { // this file is done, check if there are other files
501  mCTFTree.reset();
502  mCTFFile->Close();
503  mCTFFile.reset();
504  if (mFileFetcher) {
505  mFileFetcher->popFromQueue(mInput.maxLoops < 1);
506  }
507  }

511 void CTFReaderSpec::setMessageHeader(ProcessingContext& pc, const CTFHeader& ctfHeader, const std::string& lbl, unsigned subspec) const

589 DataProcessorSpec getCTFReaderSpec(const CTFReaderInp& inp)
590 {
591  std::vector<InputSpec> inputs;
592  std::vector<OutputSpec> outputs;
593  std::vector<ConfigParamSpec> options;
594 
595  outputs.emplace_back(OutputLabel{"header"}, "CTF", "HEADER", inp.subspec, Lifetime::Timeframe);
596  for (auto id = DetID::First; id <= DetID::Last; id++) {
597  if (inp.detMask[id]) {
598  DetID det(id);
599  outputs.emplace_back(OutputLabel{det.getName()}, det.getDataOrigin(), "CTFDATA", inp.subspec, Lifetime::Timeframe);
600  }
601  }
602  if (!inp.fileIRFrames.empty() || !inp.fileRunTimeSpans.empty()) {
603  outputs.emplace_back(OutputLabel{"selIRFrames"}, "CTF", "SELIRFRAMES", 0, Lifetime::Timeframe);
604  }
605  if (!inp.sup0xccdb) {
606  outputs.emplace_back(OutputSpec{{"TFDist"}, o2::header::gDataOriginFLP, o2::header::gDataDescriptionDISTSTF, 0xccdb});
607  }
608 
609  options.emplace_back(ConfigParamSpec{"select-ctf-ids", VariantType::String, "", {"comma-separated list CTF IDs to inject (from cumulative counter of CTFs seen)"}});
610  options.emplace_back(ConfigParamSpec{"impose-run-start-timstamp", VariantType::Int64, 0L, {"impose run start time stamp (ms), ignored if 0"}});
611  options.emplace_back(ConfigParamSpec{"local-tf-counter", VariantType::Bool, false, {"reassign header.tfCounter from local TF counter"}});
612  options.emplace_back(ConfigParamSpec{"fetch-failure-threshold", VariantType::Float, 0.f, {"Fail if too many failures( >0: fraction, <0: abs number, 0: no threshold)"}});
613  options.emplace_back(ConfigParamSpec{"limit-tf-before-reading", VariantType::Bool, false, {"Check TF limiting before reading new TF, otherwhise before injecting it"}});
614  options.emplace_back(ConfigParamSpec{"max-tf", VariantType::Int, -1, {"max CTFs to process (<= 0 : infinite)"}});
615  options.emplace_back(ConfigParamSpec{"max-tf-per-file", VariantType::Int, -1, {"max TFs to process per ctf file (<= 0 : infinite)"}});
616 
617  if (!inp.metricChannel.empty()) {
618  options.emplace_back(ConfigParamSpec{"channel-config", VariantType::String, inp.metricChannel, {"Out-of-band channel config for TF throttling"}});
619  }
620 
621  return DataProcessorSpec{
622  "ctf-reader",
623  inputs,
624  outputs,
625  AlgorithmSpec{adaptFromTask<CTFReaderSpec>(inp)},
626  options};
627 }
