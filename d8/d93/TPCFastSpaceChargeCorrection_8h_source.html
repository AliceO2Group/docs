317 GPUdi() void TPCFastSpaceChargeCorrection::schrinkUV(int32_t slice, int32_t row, float& u, float& v) const
318 {
320 
321  const TPCFastTransformGeo::RowInfo& rowInfo = mGeo.getRowInfo(row);
322 
323  float uWidth05 = rowInfo.getUwidth() * (0.5f + fInterpolationSafetyMargin);
324  float vWidth = mGeo.getTPCzLength(slice);
325 
326  if (u < -uWidth05) {
327  u = -uWidth05;
328  }
329  if (u > uWidth05) {
330  u = uWidth05;
331  }
332  if (v < -0.1f * vWidth) {
333  v = -0.1f * vWidth;
334  }
335  if (v > 1.1f * vWidth) {
336  v = 1.1f * vWidth;
337  }
338 }

340 GPUdi() void TPCFastSpaceChargeCorrection::schrinkCorrectedUV(int32_t slice, int32_t row, float& corrU, float& corrV) const
341 {
343 
344  const TPCFastTransformGeo::RowInfo& rowInfo = mGeo.getRowInfo(row);
345  const SliceRowInfo& sliceRowInfo = getSliceRowInfo(slice, row);
346 
347  float uMargin = fInterpolationSafetyMargin * rowInfo.getUwidth();
348  float vMargin = fInterpolationSafetyMargin * mGeo.getTPCzLength(slice);
349 
350  if (corrU < sliceRowInfo.activeArea.cuMin - uMargin) {
351  corrU = sliceRowInfo.activeArea.cuMin - uMargin;
352  }
353 
354  if (corrU > sliceRowInfo.activeArea.cuMax + uMargin) {
355  corrU = sliceRowInfo.activeArea.cuMax + uMargin;
356  }
357 
358  if (corrV < 0.f - vMargin) {
359  corrV = 0.f - vMargin;
360  }
361 
362  if (corrV > sliceRowInfo.activeArea.cvMax + vMargin) {
363  corrV = sliceRowInfo.activeArea.cvMax + vMargin;
364  }
365 }

367 GPUdi() void TPCFastSpaceChargeCorrection::convUVtoGrid(int32_t slice, int32_t row, float u, float v, float& gu, float& gv) const
368 {
369  // TODO optimise !!!
370  gu = 0.f;
371  gv = 0.f;
372 
373  schrinkUV(slice, row, u, v);
374 
375  const SliceRowInfo& info = getSliceRowInfo(slice, row);
376  const SplineType& spline = getSpline(slice, row);
377 
378  float su0 = 0.f, sv0 = 0.f;
379  mGeo.convUVtoScaledUV(slice, row, u, info.gridV0, su0, sv0);
380  mGeo.convUVtoScaledUV(slice, row, u, v, gu, gv);
381 
382  gv = (gv - sv0) / (1.f - sv0);
383  gu *= spline.getGridX1().getUmax();
384  gv *= spline.getGridX2().getUmax();
385 }

412  const SplineType& spline = getSpline(slice, row);
413  const float* splineData = getSplineData(slice, row);

415  convUVtoGrid(slice, row, u, v, gridU, gridV);

419  dxuv[0] = dxuv[1] = dxuv[2] = 0;

478 GPUdi() float TPCFastSpaceChargeCorrection::getMaxDriftLength(int32_t slice, int32_t row, float pad) const
479 {
480  const RowActiveArea& area = getSliceRowInfo(slice, row).activeArea;
481  const float* c = area.maxDriftLengthCheb;
482  float x = -1.f + 2.f * pad / mGeo.getRowInfo(row).maxPad;
483  float y = c[0] + c[1] * x;
484  float f0 = 1.f;
485  float f1 = x;
486  x *= 2.f;
487  for (int32_t i = 2; i < 5; i++) {
488  double f = x * f1 - f0;
489  y += c[i] * f;
490  f0 = f1;
491  f1 = f;
492  }
493  return y;
494 }
