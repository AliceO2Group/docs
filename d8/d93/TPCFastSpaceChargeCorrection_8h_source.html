316 GPUdi() void TPCFastSpaceChargeCorrection::schrinkUV(int slice, int row, float& u, float& v) const
317 {
319 
320  const TPCFastTransformGeo::RowInfo& rowInfo = mGeo.getRowInfo(row);
321 
322  float uWidth05 = rowInfo.getUwidth() * (0.5f + fInterpolationSafetyMargin);
323  float vWidth = mGeo.getTPCzLength(slice);
324 
325  if (u < -uWidth05) {
326  u = -uWidth05;
327  }
328  if (u > uWidth05) {
329  u = uWidth05;
330  }
331  if (v < -0.1f * vWidth) {
332  v = -0.1f * vWidth;
333  }
334  if (v > 1.1f * vWidth) {
335  v = 1.1f * vWidth;
336  }
337 }

339 GPUdi() void TPCFastSpaceChargeCorrection::schrinkCorrectedUV(int slice, int row, float& corrU, float& corrV) const
340 {
342 
343  const TPCFastTransformGeo::RowInfo& rowInfo = mGeo.getRowInfo(row);
344  const SliceRowInfo& sliceRowInfo = getSliceRowInfo(slice, row);
345 
346  float uMargin = fInterpolationSafetyMargin * rowInfo.getUwidth();
347  float vMargin = fInterpolationSafetyMargin * mGeo.getTPCzLength(slice);
348 
349  if (corrU < sliceRowInfo.activeArea.cuMin - uMargin) {
350  corrU = sliceRowInfo.activeArea.cuMin - uMargin;
351  }
352 
353  if (corrU > sliceRowInfo.activeArea.cuMax + uMargin) {
354  corrU = sliceRowInfo.activeArea.cuMax + uMargin;
355  }
356 
357  if (corrV < 0.f - vMargin) {
358  corrV = 0.f - vMargin;
359  }
360 
361  if (corrV > sliceRowInfo.activeArea.cvMax + vMargin) {
362  corrV = sliceRowInfo.activeArea.cvMax + vMargin;
363  }
364 }

366 GPUdi() void TPCFastSpaceChargeCorrection::convUVtoGrid(int slice, int row, float u, float v, float& gu, float& gv) const
367 {
368  // TODO optimise !!!
369  gu = 0.f;
370  gv = 0.f;
371 
372  schrinkUV(slice, row, u, v);
373 
374  const SliceRowInfo& info = getSliceRowInfo(slice, row);
375  const SplineType& spline = getSpline(slice, row);
376 
377  float su0 = 0.f, sv0 = 0.f;
378  mGeo.convUVtoScaledUV(slice, row, u, info.gridV0, su0, sv0);
379  mGeo.convUVtoScaledUV(slice, row, u, v, gu, gv);
380 
381  gv = (gv - sv0) / (1.f - sv0);
382  gu *= spline.getGridX1().getUmax();

465 GPUdi() float TPCFastSpaceChargeCorrection::getMaxDriftLength(int slice, int row, float pad) const
466 {
467  const RowActiveArea& area = getSliceRowInfo(slice, row).activeArea;
468  const float* c = area.maxDriftLengthCheb;
469  float x = -1.f + 2.f * pad / mGeo.getRowInfo(row).maxPad;
470  float y = c[0] + c[1] * x;
471  float f0 = 1.f;
472  float f1 = x;
473  x *= 2.f;
474  for (int i = 2; i < 5; i++) {
475  double f = x * f1 - f0;
476  y += c[i] * f;
477  f0 = f1;
478  f1 = f;
479  }
480  return y;
481 }
