32 bool RawParserHelper::checkPrintError(size_t& localCounter)
33 {
34  localCounter++;
35  sErrors++;
36  if (sErrorLimit == 0) {
37  return false;
38  }
39  if (sErrors > sErrorLimit * sErrorScale) {
40  sErrorScale *= 10;
41  }
42  return sErrors % sErrorScale == 0;
43 }

46 {

49  if (++contDeadBeef <= maxWarn) {
