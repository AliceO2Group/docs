51  void run(framework::ProcessingContext& pc)
52  {
53  static bool finished = false;
54  if (finished) {
55  return;
56  }
57  LOG(info) << "Doing HMPID digitization";
58 
59  // read collision context from input
60  auto context = pc.inputs().get<o2::steer::DigitizationContext*>("collisioncontext");
61 
62  if (context->hasTriggerInput()) {
63  auto ctpdigits = context->getCTPDigits();
64  LOG(info) << "Yes, we have CTP digits";
65  LOG(info) << "We have " << ctpdigits->size() << " digits";
66  } else {
67  LOG(info) << "No trigger input available ... selftriggering";
68  }
69 
70  context->initSimChains(o2::detectors::DetID::HMP, mSimChains);
71 
72  auto& irecords = context->getEventRecords();
73  for (auto& record : irecords) {
74  LOG(info) << "HMPID TIME RECEIVED " << record.getTimeNS();
75  }
76 
77  auto& eventParts = context->getEventParts();
78  std::vector<o2::hmpid::Digit> digitsAccum; // accumulator for digits
79  o2::dataformats::MCTruthContainer<o2::MCCompLabel> labelAccum; // timeframe accumulator for labels
80  mIntRecord.clear();
81 
82  auto flushDigitsAndLabels = [this, &digitsAccum, &labelAccum]() {
83  // flush previous buffer
84  mDigits.clear();
85  mLabels.clear();
86  mDigitizer.flush(mDigits);
87  LOG(info) << "HMPID flushed " << mDigits.size() << " digits at this time ";
88  LOG(info) << "NUMBER OF LABEL OBTAINED " << mLabels.getNElements();
89  int32_t first = digitsAccum.size(); // this is the first
90  std::copy(mDigits.begin(), mDigits.end(), std::back_inserter(digitsAccum));
91  labelAccum.mergeAtBack(mLabels);
92 
93  // save info for the triggers accepted
94  LOG(info) << "Trigger Orbit :" << mDigitizer.getOrbit() << " BC:" << mDigitizer.getBc();
95  mIntRecord.push_back(o2::hmpid::Trigger(o2::InteractionRecord(mDigitizer.getBc(), mDigitizer.getOrbit()), first, digitsAccum.size() - first));
96  };
97 
98  // loop over all composite collisions given from context
99  // (aka loop over all the interaction records)
100  for (int collID = 0; collID < irecords.size(); ++collID) {
101  // try to start new readout cycle by setting the trigger time
102  auto triggeraccepted = mDigitizer.setTriggerTime(irecords[collID].getTimeNS());
103  if (triggeraccepted) {
104  auto withinactivetime = mDigitizer.setEventTime(irecords[collID].getTimeNS());
105  if (withinactivetime) {
106  // for each collision, loop over the constituents event and source IDs
107  // (background signal merging is basically taking place here)
108  for (auto& part : eventParts[collID]) {
109  mDigitizer.setEventID(part.entryID);
110  mDigitizer.setSrcID(part.sourceID);
111 
112  // get the hits for this event and this source
113  std::vector<o2::hmpid::HitType> hits;
114  context->retrieveHits(mSimChains, "HMPHit", part.sourceID, part.entryID, &hits);
115  LOG(info) << "For collision " << collID << " eventID " << part.entryID << " found HMP " << hits.size() << " hits ";
116 
117  mDigitizer.setLabelContainer(&mLabels);
118  mLabels.clear();
119  mDigits.clear();
120 
121  mDigitizer.process(hits, mDigits);
122  }
123 
124  flushDigitsAndLabels(); // flush previous readout cycle
125  } else {
126  LOG(info) << "COLLISION " << collID << "FALLS WITHIN A DEAD TIME";
127  }
128  }
129  }
130 
131  // send out to next stage
132  pc.outputs().snapshot(Output{"HMP", "DIGITS", 0}, digitsAccum);
133  pc.outputs().snapshot(Output{"HMP", "INTRECORDS", 0}, mIntRecord);
134  if (pc.outputs().isAllowed({"HMP", "DIGITLBL", 0})) {
135  pc.outputs().snapshot(Output{"HMP", "DIGITLBL", 0}, labelAccum);
136  }
137  LOG(info) << "HMP: Sending ROMode= " << mROMode << " to GRPUpdater";
138  pc.outputs().snapshot(Output{"HMP", "ROMode", 0}, mROMode);
139 
140  // we should be only called once; tell DPL that this process is ready to exit
141  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
142  finished = true;
143  }
144 
145  private:
146  HMPIDDigitizer mDigitizer;
147  std::vector<TChain*> mSimChains;
148  std::vector<o2::hmpid::Digit> mDigits;
149  o2::dataformats::MCTruthContainer<o2::MCCompLabel> mLabels; // labels which get filled
150  std::vector<o2::hmpid::Trigger> mIntRecord;
151 
152  // RS: at the moment using hardcoded flag for continuous readout
153  o2::parameters::GRPObject::ROMode mROMode = o2::parameters::GRPObject::PRESENT; // readout mode
154 };

157 {
158  // create the full data processor spec using

163  std::vector<OutputSpec> outputs;
164  outputs.emplace_back("HMP", "DIGITS", 0, Lifetime::Timeframe);
