250GPUdi() int32_t GPUTPCGMTrackParam::initResetT0()
251{
252 const float absQPt = CAMath::Abs(mP[4]);
253 if (absQPt < (150.f / 40.f)) {

256 return CAMath::Max(10.f, 150.f / mP[4]);

280GPUdi() float GPUTPCGMTrackParam::GetMirroredY(float Bz) const
281{
282 // get Y of the point which has the same X, but located on the other side of trajectory
283 float qptBz = GetQPt() * Bz;
284 float cosPhi2 = 1.f - GetSinPhi() * GetSinPhi();
285 if (CAMath::Abs(qptBz) < 1.e-8f) {
286 qptBz = 1.e-8f;
287 }
288 if (cosPhi2 < 0.f) {
289 cosPhi2 = 0.f;
290 }
291 return GetY() - 2.f * CAMath::Sqrt(cosPhi2) / qptBz;
292}
