75void GPUTPCGMOfflineFitter::RefitTrack(GPUTPCGMMergedTrack& track, const GPUTPCGMPolynomialField* field, GPUTPCGMMergedTrackHit* clusters)
76{
77 // copy of HLT RefitTrack() with calling of the offline fit utilities
78
79 if (!track.OK()) {
80 return;
81 }
82
83 int32_t nTrackHits = track.NClusters();
84 cout << "call FitOffline .. " << endl;
85 bool ok = FitOffline(field, track, clusters + track.FirstClusterRef(), nTrackHits);
86 cout << ".. end of call FitOffline " << endl;
87
88 GPUTPCGMTrackParam t = track.Param();
89 float Alpha = track.Alpha();
90
91 if (fabsf(t.QPt()) < 1.e-4) {
92 t.QPt() = 1.e-4;
93 }
94
95 track.SetOK(ok);
96 track.SetNClustersFitted(nTrackHits);
97 track.Param() = t;
98 track.Alpha() = Alpha;
99
100 {
101 int32_t ind = track.FirstClusterRef();
102 float alphaa = fCAParam.Alpha(clusters[ind].slice);
103 float xx = clusters[ind].fX;
104 float yy = clusters[ind].fY;
105 float zz = clusters[ind].fZ - track.Param().GetZOffset();
106 float sinA = CAMath::Sin(alphaa - track.Alpha());
107 float cosA = CAMath::Cos(alphaa - track.Alpha());
108 track.SetLastX(xx * cosA - yy * sinA);
109 track.SetLastY(xx * sinA + yy * cosA);
110 track.SetLastZ(zz);
111 }
112}

115{

121 // add the information we have

125 c.SetDetector(sector);
126 c.SetRow(row); // ?? is it right row numbering for the TPC tracker ??
127 c.SetX(h.fX);

134bool GPUTPCGMOfflineFitter::FitOffline(const GPUTPCGMPolynomialField* field, GPUTPCGMMergedTrack& gmtrack, GPUTPCGMMergedTrackHit* clusters, int32_t& N)
135{
136 const float maxSinPhi = GPUCA_MAX_SIN_PHI;
137
138 int32_t maxN = N;
139 float covYYUpd = 0.;
140 float lastUpdateX = -1.;
141
142 const bool rejectChi2ThisRound = 0;
143 const bool markNonFittedClusters = 0;
144 const double kDeg2Rad = 3.14159265358979323846 / 180.;
145 const float maxSinForUpdate = CAMath::Sin(70. * kDeg2Rad);
146
147 bool ok = 1;
148
149 AliTPCtracker::SetIteration(2);
150
151 AliTPCseed seed;
152 gmtrack.Param().GetExtParam(seed, gmtrack.Alpha());
153
154 AliTPCtracker::AddCovariance(&seed);
155
156 N = 0;
157 lastUpdateX = -1;
158
159 // find last leg
160 int32_t ihitStart = 0;
161 for (int32_t ihit = 0; ihit < maxN; ihit++) {
162 if (clusters[ihit].leg != clusters[ihitStart].leg) {
163 ihitStart = ihit;
164 }
165 }
166
167 for (int32_t ihit = ihitStart; ihit < maxN; ihit++) {
168 if (clusters[ihit].fState < 0) {
169 continue; // hit is excluded from fit
170 }
171 float xx = clusters[ihit].fX;
172 float yy = clusters[ihit].fY;
173 float zz = clusters[ihit].fZ;
174
175 if (DOUBLE && ihit + 1 >= 0 && ihit + 1 < maxN && clusters[ihit].row == clusters[ihit + 1].row) {
176 float count = 1.;
177 do {
178 if (clusters[ihit].slice != clusters[ihit + 1].slice || clusters[ihit].leg != clusters[ihit + 1].leg || fabsf(clusters[ihit].fY - clusters[ihit + 1].fY) > 4. || fabsf(clusters[ihit].fZ - clusters[ihit + 1].fZ) > 4.) {
179 break;
180 }
181 ihit += 1;
182 xx += clusters[ihit].fX;
183 yy += clusters[ihit].fY;
184 zz += clusters[ihit].fZ;
185 count += 1.;
186 } while (ihit + 1 >= 0 && ihit + 1 < maxN && clusters[ihit].row == clusters[ihit + 1].row);
187 xx /= count;
188 yy /= count;
189 zz /= count;
190 }
191
192 // Create AliTPCclusterMI for the hit
193
194 AliTPCclusterMI cluster;
195 Int_t tpcindex = CreateTPCclusterMI(clusters[ihit], cluster);
196 if (tpcindex < 0) {
197 continue;
198 }
199 Double_t sy2 = 0, sz2 = 0;
200 AliTPCtracker::ErrY2Z2(&seed, &cluster, sy2, sz2);
201 cluster.SetSigmaY2(sy2);
202 cluster.SetSigmaZ2(sz2);
203 cluster.SetQ(10);
204 cluster.SetMax(10);
205
206 Int_t iRow = clusters[ihit].row;
207
208 if (iRow < AliHLTTPCGeometry::GetNRowLow()) {
209 AliTPCtracker::fSectors = AliTPCtracker::fInnerSec;
210 } else {
211 AliTPCtracker::fSectors = AliTPCtracker::fOuterSec;
212 }
213
214 seed.SetClusterIndex2(iRow, tpcindex);
215 seed.SetClusterPointer(iRow, &cluster);
216 seed.SetCurrentClusterIndex1(tpcindex);
217
218 int32_t retVal;
219 float threshold = 3. + (lastUpdateX >= 0 ? (fabsf(seed.GetX() - lastUpdateX) / 2) : 0.);
220 if (N > 2 && (fabsf(yy - seed.GetY()) > threshold || fabsf(zz - seed.GetZ()) > threshold)) {
221 retVal = 2;
222 } else {
223 Int_t err = !(AliTPCtracker::FollowToNext(seed, iRow));
224
225 const int32_t err2 = N > 0 && CAMath::Abs(seed.GetSnp()) >= maxSinForUpdate;
226 if (err || err2) {
227 if (markNonFittedClusters) {
228 if (N > 0 && (fabsf(yy - seed.GetY()) > 3 || fabsf(zz - seed.GetZ()) > 3)) {
229 clusters[ihit].fState = -2;
230 } else if (err && err >= -3) {
231 clusters[ihit].fState = -1;
232 }
233 }
234 continue;
235 }
236
237 // retVal = prop.Update( yy, zz, clusters[ihit].row, param, rejectChi2ThisRound);
238 retVal = 0;
239 }
240
241 if (retVal == 0) // track is updated
242 {
243 lastUpdateX = seed.GetX();
244 covYYUpd = seed.GetCovariance()[0];
245 ihitStart = ihit;
246 N++;
247 } else if (retVal == 2) { // cluster far away form the track
248 if (markNonFittedClusters) {
249 clusters[ihit].fState = -2;
250 }
251 } else {
252 break; // bad chi2 for the whole track, stop the fit
253 }
254 } // end loop over clusters
255
256 GPUTPCGMTrackParam t;
257 t.SetExtParam(seed);
258
259 float Alpha = seed.GetAlpha();
260
261 t.ConstrainSinPhi();
262
263 bool ok1 = N >= GPUCA_TRACKLET_SELECTOR_MIN_HITS_B5(t.GetQPt()) && t.CheckNumericalQuality(covYYUpd);
264 if (!ok1) {
265 return (false);
266 }
267
268 // const float kDeg2Rad = 3.1415926535897 / 180.f;
269 const float kSectAngle = 2 * 3.1415926535897 / 18.f;
270
271 if (fCAParam.GetTrackReferenceX() <= 500) {
272 GPUTPCGMPropagator prop;
273 prop.SetMaterialTPC();
274 prop.SetPolynomialField(field);
275 prop.SetMaxSinPhi(maxSinPhi);
276 prop.SetToyMCEventsFlag(fCAParam.ToyMCEventsFlag());
277
278 for (int32_t k = 0; k < 3; k++) // max 3 attempts
279 {
280 int32_t err = prop.PropagateToXAlpha(fCAParam.GetTrackReferenceX(), Alpha, 0);
281 t.ConstrainSinPhi();
282 if (fabsf(t.GetY()) <= t.GetX() * tan(kSectAngle / 2.f)) {
283 break;
284 }
285 float dAngle = floor(atan2(t.GetY(), t.GetX()) / kDeg2Rad / 20.f + 0.5f) * kSectAngle;
286 Alpha += dAngle;
287 if (err || k == 2) {
288 t.Rotate(dAngle);
289 break;
290 }
291 }
292 } else if (fabsf(t.GetY()) > t.GetX() * tan(kSectAngle / 2.f)) {
293 float dAngle = floor(atan2(t.GetY(), t.GetX()) / kDeg2Rad / 20.f + 0.5f) * kSectAngle;
294 t.Rotate(dAngle);
295 Alpha += dAngle;
296 }
297 if (Alpha > 3.1415926535897) {
298 Alpha -= 2 * 3.1415926535897;
299 } else if (Alpha <= -3.1415926535897) {
300 Alpha += 2 * 3.1415926535897;
301 }
302
303 gmtrack.Param() = t;
304 gmtrack.Alpha() = Alpha;
305
306 return (ok);
307}
