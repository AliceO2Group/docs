38WorkflowSpec defineDataProcessing(ConfigContext const& configcontext)
39{
40 WorkflowSpec specs;
41
42 std::string confDig = configcontext.options().get<std::string>("hbfutils-config");
43 if (!confDig.empty() && confDig != "none") {
44 o2::conf::ConfigurableParam::updateFromFile(confDig, "HBFUtils");
45 }
46 o2::conf::ConfigurableParam::updateFromString(configcontext.options().get<std::string>("configKeyValues"));
47 specs.emplace_back(o2::mid::getDigitReaderSpec(false));
48 specs.emplace_back(o2::mid::getRawWriterSpec());
49
