46 std::string detectors(const std::vector<std::string>& det, unsigned mask)
47 {
48  std::string result;
49  std::string delim;
50  int bit = 1;
51  for (const auto& i : det) {
52  if (mask & bit) {
53  result += delim + i;
54  delim = ",";
55  }
56  bit = bit << 1;
57  }
58  return result;
