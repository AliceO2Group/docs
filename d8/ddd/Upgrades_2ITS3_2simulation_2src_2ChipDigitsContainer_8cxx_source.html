24void ChipDigitsContainer::addNoise(UInt_t rofMin, UInt_t rofMax, const o2::its3::DigiParams* params)
25{
26 UInt_t row = 0;
27 UInt_t col = 0;
28 Int_t nhits = 0;
29 constexpr float ns2sec = 1e-9;
30 float mean = 0.f;
31 int nel = 0;
32
33 if (isIB()) {
34 // Inner barrel: use ITS3-specific noise interface with IB segmentation.
35 mean = params->getIBNoisePerPixel() * SegmentationIB::NPixels;
36 nel = static_cast<int>(params->getIBChargeThreshold() * 1.1);
37 } else {
38 // Outer barrel: use base class noise interface with OB segmentation.
39 mean = params->getNoisePerPixel() * SegmentationOB::NPixels;
40 nel = static_cast<int>(params->getChargeThreshold() * 1.1);
41 }
42
43 for (UInt_t rof = rofMin; rof <= rofMax; ++rof) {
44 nhits = gRandom->Poisson(mean);
45 for (Int_t i = 0; i < nhits; ++i) {
46 row = gRandom->Integer(maxRows);
47 col = gRandom->Integer(maxCols);
48 if (mNoiseMap && mNoiseMap->isNoisy(mChipIndex, row, col)) {
49 continue;
50 }
51 if (mDeadChanMap && mDeadChanMap->isNoisy(mChipIndex, row, col)) {
52 continue;
53 }
54 auto key = getOrderingKey(rof, row, col);
55 if (!findDigit(key)) {
56 addDigit(key, rof, row, col, nel, o2::MCCompLabel(true));
57 }
58 }
59 }
60}
