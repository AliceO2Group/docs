69  void run(o2::framework::ProcessingContext& pc) final
70  {
71  const int lane = pc.inputs().get<int>("lane");
72  if (lane >= mInputLanes) {
73  LOGP(error, "Received data from lane {} which is >= than the specified number of expected lanes of {}!", lane, mInputLanes);
74  return;
75  }
76 
77  mCCDBBuffer[lane] = pc.inputs().get<std::vector<long>>("tsccdb");
78  if (mProcessedTimeStamp > mCCDBBuffer[lane].front()) {
79  LOGP(warning, "Already received data from a later time stamp {} then the currently received time stamp {}! (This might not be an issue)", mProcessedTimeStamp, mCCDBBuffer[lane].front());
80  } else {
81  mProcessedTimeStamp = mCCDBBuffer[lane].front();
82  }
83 
84  if (!mProcessSACs) {
85  mIntervalsBuffer[lane] = pc.inputs().get<std::vector<unsigned int>>("intervals");
86  }
87 
88  for (auto& ref : InputRecordWalker(pc.inputs(), mFilter[mProcessSACs])) {
89  auto const* dataHeader = o2::framework::DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
90  const int side = dataHeader->subSpecification;
91  mIDCOneBuffer[lane][side].mIDCOne = pc.inputs().get<std::vector<float>>(ref);
92  LOGP(info, "Received {} 1D-IDCs for side {}", mIDCOneBuffer[lane][side].mIDCOne.size(), side);
93 
94  if (mProcessSACs && mIntervalsBuffer[lane].empty()) {
95  const auto nValues = mIDCOneBuffer[lane][side].mIDCOne.size();
96  const int nIntervals = nValues / mIntervalsSACs;
97  const int nFirstInterval = nValues % mIntervalsSACs;
98  if (nFirstInterval == 0) {
99  mIntervalsBuffer[lane] = std::vector<unsigned int>(nIntervals, mIntervalsSACs);
100  } else {
101  mIntervalsBuffer[lane] = std::vector<unsigned int>(nIntervals + 1, mIntervalsSACs);
102  mIntervalsBuffer[lane].front() = nFirstInterval;
103  }
104  }
105  }
106 
107  // buffer IDCs for TPC scaler
108  if (!mProcessSACs && !mDisableScaler) {
109  const long startTS = mCCDBBuffer[lane].front();
110  TPCScalerProc& scaler = mTPCScalerCont.idcs[startTS];
111  scaler.timestamp[0] = startTS;
112  scaler.timestamp[1] = mCCDBBuffer[lane].back();
113  for (auto& ref : InputRecordWalker(pc.inputs(), mFilterI0)) {
114  auto const* dataHeader = o2::framework::DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
115  const int side = dataHeader->subSpecification;
116  const float idc0mean = pc.inputs().get<float>(ref);
117  LOGP(info, "Received {} IDC0 mean for side {}", idc0mean, side);
118  scaler.idc1[side] = mIDCOneBuffer[lane][side].mIDCOne;
119  auto& vecIDC = scaler.idc1[side];
120 
121  // normalize 1D-IDCs to unity as it should be
122  if (vecIDC.size() > 0) {
123  const float mean = std::reduce(vecIDC.begin(), vecIDC.end()) / static_cast<float>(vecIDC.size());
124  LOGP(info, "normalizing by {}", mean);
125  if (std::abs(mean) > 0.001) {
126  std::transform(vecIDC.begin(), vecIDC.end(), vecIDC.begin(), [&mean](auto val) { return val / mean; });
127  }
128  }
129 
130  // scale IDC1 with IDC0Mean
131  std::transform(scaler.idc1[side].begin(), scaler.idc1[side].end(), scaler.idc1[side].begin(), [&idc0mean](auto idc) { return idc * idc0mean; });
132  }
133  // check if A- and C-side has the same length!
134  const int lenA = scaler.idc1[0].size();
135  const int lenC = scaler.idc1[1].size();
136  if (lenA != lenC) {
137  // This should never happen
138  LOGP(warning, "Received IDCs have different length! A-side length: {} and C-side length: {}", lenA, lenC);
139  // add dummy to shorter vector
140  const int maxLen = std::max(lenA, lenC);
141  scaler.idc1[0].resize(maxLen);
142  scaler.idc1[1].resize(maxLen);
143  }
144 
145  mRun = processing_helpers::getRunNumber(pc);
146  makeTPCScaler(pc.outputs(), false);
147  }
148 
149  FourierCoeffSAC coeffSAC;
150  if (lane == mExpectedInputLane) {
151  const int nSides = mIDCOneBuffer[lane][Side::A].mIDCOne.empty() + mIDCOneBuffer[lane][Side::C].mIDCOne.empty();
152  // int iProcessLane = lane;
153  for (int iProcessLaneTmp = 0; iProcessLaneTmp < mInputLanes; ++iProcessLaneTmp) {
154  const int nSidesCurrLane = mIDCOneBuffer[mExpectedInputLane][Side::A].mIDCOne.empty() + mIDCOneBuffer[mExpectedInputLane][Side::C].mIDCOne.empty();
155  if (nSidesCurrLane != nSides) {
156  break;
157  }
158 
159  for (int iSide = 0; iSide < SIDES; ++iSide) {
160  const Side side = (iSide == 0) ? A : C;
161  if (mIDCOneBuffer[mExpectedInputLane][side].mIDCOne.empty()) {
162  continue;
163  }
164  LOGP(info, "Processing input lane: {} for Side: {}", mExpectedInputLane, iSide);
165 
166  // perform fourier transform of 1D-IDCs
167  mIDCFourierTransform[side].setIDCs(std::move(mIDCOneBuffer[mExpectedInputLane][side]), mIntervalsBuffer[mExpectedInputLane]);
168  mIDCFourierTransform[side].calcFourierCoefficients(mIntervalsBuffer[mExpectedInputLane].size());
169 
170  if (!mProcessSACs) {
171  o2::ccdb::CcdbObjectInfo ccdbInfo(CDBTypeMap.at(((side == 0) ? CDBType::CalIDCFourierA : CDBType::CalIDCFourierC)), std::string{}, std::string{}, std::map<std::string, std::string>{}, mCCDBBuffer[mExpectedInputLane].front(), mCCDBBuffer[mExpectedInputLane].back());
172  auto imageFFT = o2::ccdb::CcdbApi::createObjectImage(&mIDCFourierTransform[side].getFourierCoefficients(), &ccdbInfo);
173  LOGP(info, "Sending object {} / {} of size {} bytes, valid for {} : {} ", ccdbInfo.getPath(), ccdbInfo.getFileName(), imageFFT->size(), ccdbInfo.getStartValidityTimestamp(), ccdbInfo.getEndValidityTimestamp());
174  pc.outputs().snapshot(Output{o2::calibration::Utils::gDataOriginCDBPayload, getDataDescriptionCCDBFourier(), 0}, *imageFFT.get());
175  pc.outputs().snapshot(Output{o2::calibration::Utils::gDataOriginCDBWrapper, getDataDescriptionCCDBFourier(), 0}, ccdbInfo);
176  } else {
177  coeffSAC.mCoeff[side] = mIDCFourierTransform[side].getFourierCoefficients();
178  }
179 
180  if (mDumpFFT) {
181  LOGP(info, "dumping FT to file");
182  mIDCFourierTransform[side].dumpToFile(fmt::format("FourierAGG_{:02}_side{}.root", processing_helpers::getCurrentTF(pc), (int)side).data());
183  }
184 
185  if (mSendOutDebug) {
186  sendOutput(pc.outputs(), side);
187  }
188  }
189 
190  if (mProcessSACs) {
191  o2::ccdb::CcdbObjectInfo ccdbInfo(CDBTypeMap.at(CDBType::CalSACFourier), std::string{}, std::string{}, std::map<std::string, std::string>{}, mCCDBBuffer[mExpectedInputLane].front(), mCCDBBuffer[mExpectedInputLane].back());
192  auto imageFFT = o2::ccdb::CcdbApi::createObjectImage(&coeffSAC, &ccdbInfo);
193  LOGP(info, "Sending object {} / {} of size {} bytes, valid for {} : {} ", ccdbInfo.getPath(), ccdbInfo.getFileName(), imageFFT->size(), ccdbInfo.getStartValidityTimestamp(), ccdbInfo.getEndValidityTimestamp());
194  pc.outputs().snapshot(Output{o2::calibration::Utils::gDataOriginCDBPayload, getDataDescriptionCCDBFourier(), 0}, *imageFFT.get());
195  pc.outputs().snapshot(Output{o2::calibration::Utils::gDataOriginCDBWrapper, getDataDescriptionCCDBFourier(), 0}, ccdbInfo);
196  }
197  mExpectedInputLane = ++mExpectedInputLane % mInputLanes;
198  }
199  }

248  void makeTPCScaler(DataAllocator& output, const bool eos)
249  {
250  LOGP(info, "Making TPC scalers");
251  // check if IDC scalers can be created - check length of continous received IDCs
252  std::vector<std::pair<long, long>> times;
253  times.reserve(mTPCScalerCont.idcs.size());
254  for (const auto& idc : mTPCScalerCont.idcs) {
255  times.emplace_back(idc.second.timestamp[0], idc.second.timestamp[1]);
256  }
257 
258  // sort received times of the IDCs
259  std::sort(times.begin(), times.end());
260 
261  // loop over times and make checks
262  const int checkGapp = 10;
263  // if the diff between end of data[i] and start of data[i+1] is smaller than this, the data is contigous
264  long timesDuration = (times.front().second - times.front().first);
265 
266  // make check and store lastValid index in case IDC scalers can be created
267  int lastValidIdx = -1;
268  for (int i = 1; i < times.size(); ++i) {
269  // check time diff between start of current IDCs and end time of last IDCs
270  const auto deltaTime = times[i].first - times[i - 1].second;
271  // check if IDCs are contigous
272  if (deltaTime > (timesDuration / checkGapp)) {
273  // check if the gap is very large - in this case the gapp might be lost, so just write out the TPC scaler until the gap
274  if (deltaTime > (checkGapp * timesDuration)) {
275  lastValidIdx = i - 1;
276  }
277  LOGP(info, "Breaking as big gap between IDCs of {} detected", deltaTime);
278  break;
279  }
280 
281  // check if time length is >= than mLengthIDCScalerSeconds
282  if ((times[i].first - times.front().first) / 1000 >= mLengthIDCScalerSeconds) {
283  lastValidIdx = i;
284  }
285  }
286 
287  LOGP(info, "Creating IDC scalers with {} IDC objects", lastValidIdx);
288 
289  if (eos) {
290  // in case of eos write out everything
291  lastValidIdx = times.empty() ? -1 : times.size() - 1;
292  }
293 
294  // create IDC scaler in case index is valid
295  if (lastValidIdx >= 0) {
296  o2::tpc::TPCScaler scaler;
297  scaler.setIonDriftTimeMS(170);
298  scaler.setRun(mRun);
299  scaler.setStartTimeStampMS(times.front().first);
300  const auto idcIntegrationTime = 12 /*12 orbits integration interval per IDC*/ * o2::constants::lhc::LHCOrbitMUS / 1000;
301  scaler.setIntegrationTimeMS(idcIntegrationTime);
302 
303  std::vector<float> idc1A;
304  std::vector<float> idc1C;
305  long idc1ASize = 0;
306  long idc1CSize = 0;
307 
308  // in case already one object is stored add internal overlap
309  if (mIDCSCalerEndTSLast != 0) {
310  const int nOverlap = 500;
311  idc1ASize += nOverlap;
312  idc1CSize += nOverlap;
313  const auto& scalerALast = mScalerLast.getScalers(o2::tpc::Side::A);
314  const auto& scalerCLast = mScalerLast.getScalers(o2::tpc::Side::C);
315  if (scalerALast.size() > nOverlap) {
316  idc1A.insert(idc1A.end(), scalerALast.end() - nOverlap, scalerALast.end());
317  idc1C.insert(idc1C.end(), scalerCLast.end() - nOverlap, scalerCLast.end());
318  // adjust start time
319  scaler.setStartTimeStampMS(scaler.getStartTimeStampMS() - nOverlap * idcIntegrationTime);
320  }
321  } else {
322  // store end timestamp as start time stamp for first object for correct time stamp in CCDB
323  mIDCSCalerEndTSLast = scaler.getStartTimeStampMS();
324  }
325 
326  for (int iter = 0; iter < 2; ++iter) {
327  if (iter == 1) {
328  idc1A.reserve(idc1ASize);
329  idc1C.reserve(idc1CSize);
330  }
331  for (int i = 0; i <= lastValidIdx; ++i) {
332  const auto& time = times[i];
333  const auto& idc = mTPCScalerCont.idcs[time.first];
334  if (iter == 0) {
335  idc1ASize += idc.idc1[0].size();
336  idc1CSize += idc.idc1[1].size();
337  } else {
338  idc1A.insert(idc1A.end(), idc.idc1[0].begin(), idc.idc1[0].end());
339  idc1C.insert(idc1C.end(), idc.idc1[1].begin(), idc.idc1[1].end());
340  // in case of eos check if the IDCs are contigous and add dummy values!
341  if (eos && (i < lastValidIdx)) {
342  const float deltaTime = times[i + 1].first - time.second;
343  // if delta time is too large add dummy values
344  if (deltaTime > (timesDuration / checkGapp)) {
345  const int nDummyValues = deltaTime / idcIntegrationTime + 0.5;
346  // add dummy to A
347  if (idc.idc1[0].size() > 0) {
348  float meanA = std::reduce(idc.idc1[0].begin(), idc.idc1[0].end()) / static_cast<float>(idc.idc1[0].size());
349  idc1A.insert(idc1A.end(), nDummyValues, meanA);
350  }
351 
352  if (idc.idc1[1].size() > 0) {
353  // add dummy to C
354  float meanC = std::reduce(idc.idc1[1].begin(), idc.idc1[1].end()) / static_cast<float>(idc.idc1[1].size());
355  idc1C.insert(idc1C.end(), nDummyValues, meanC);
356  }
357  }
358  }
359  mTPCScalerCont.idcs.erase(time.first);
360  }
361  }
362  }
363  scaler.setScaler(idc1A, o2::tpc::Side::A);
364  scaler.setScaler(idc1C, o2::tpc::Side::C);
365 
366  // store in CCDB
367  TTree tree("ccdb_object", "ccdb_object");
368  tree.Branch("TPCScaler", &scaler);
369  tree.Fill();
370 
371  o2::ccdb::CcdbObjectInfo ccdbInfoIDC(CDBTypeMap.at(CDBType::CalScaler), std::string{}, std::string{}, std::map<std::string, std::string>{}, mIDCSCalerEndTSLast, scaler.getEndTimeStampMS(o2::tpc::Side::A));
372  auto imageIDC = o2::ccdb::CcdbApi::createObjectImage(&tree, &ccdbInfoIDC);
373  LOGP(info, "Sending object {} / {} of size {} bytes, valid for {} : {} ", ccdbInfoIDC.getPath(), ccdbInfoIDC.getFileName(), imageIDC->size(), ccdbInfoIDC.getStartValidityTimestamp(), ccdbInfoIDC.getEndValidityTimestamp());
374  output.snapshot(Output{o2::calibration::Utils::gDataOriginCDBPayload, getDataDescriptionCCDBTPCScaler(), 0}, *imageIDC.get());
375  output.snapshot(Output{o2::calibration::Utils::gDataOriginCDBWrapper, getDataDescriptionCCDBTPCScaler(), 0}, ccdbInfoIDC);
376 
377  // store end timestamp
378  mIDCSCalerEndTSLast = scaler.getEndTimeStampMS(o2::tpc::Side::A);
379 
380  // for debugging
381  if (mDumpFFT) {
382  static int countwrite = 0;
383  scaler.dumpToFile(fmt::format("TPCScaler_snapshot_{}.root", countwrite++).data(), "ccdb_object");
384  }
385 
386  // buffer current scaler object
387  mScalerLast = std::move(scaler);
388  }
389  }

423  inputSpecs,
424  outputSpecs,
425  AlgorithmSpec{adaptFromTask<TPCFourierTransformAggregatorSpec>(nFourierCoefficientsStore, rangeIDC, senddebug, processSACs, inputLanes)},
426  Options{{"intervalsSACs", VariantType::Int, 11, {"Number of integration intervals which will be sampled for the fourier coefficients"}},
427  {"dump-coefficients-agg", VariantType::Bool, false, {"Dump fourier coefficients to file"}},
428  {"tpcScalerLengthS", VariantType::Float, 300.f, {"Length of the TPC scalers in seconds"}},
429  {"disable-scaler", VariantType::Bool, false, {"Disable creation of IDC scaler"}}}};
