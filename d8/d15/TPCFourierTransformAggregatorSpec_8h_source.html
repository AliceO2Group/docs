56  void run(o2::framework::ProcessingContext& pc) final
57  {
58  const int lane = pc.inputs().get<int>("lane");
59  if (lane >= mInputLanes) {
60  LOGP(error, "Received data from lane {} which is >= than the specified number of expected lanes of {}!", lane, mInputLanes);
61  return;
62  }
63 
64  mCCDBBuffer[lane] = pc.inputs().get<std::vector<uint64_t>>("tsccdb");
65  if (mProcessedTimeStamp > mCCDBBuffer[lane].front()) {
66  LOGP(error, "Already processed a later time stamp {} then the received time stamp {}!", mProcessedTimeStamp, mCCDBBuffer[lane].front());
67  } else {
68  mProcessedTimeStamp = mCCDBBuffer[lane].front();
69  }
70 
71  if (!mProcessSACs) {
72  mIntervalsBuffer[lane] = pc.inputs().get<std::vector<unsigned int>>("intervals");
73  }
74 
75  for (auto& ref : InputRecordWalker(pc.inputs(), mFilter[mProcessSACs])) {
76  auto const* dataHeader = o2::framework::DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
77  const int side = dataHeader->subSpecification;
78  mIDCOneBuffer[lane][side].mIDCOne = pc.inputs().get<std::vector<float>>(ref);
79  LOGP(info, "Received {} 1D-IDCs for side {}", mIDCOneBuffer[lane][side].mIDCOne.size(), side);
80 
81  if (mProcessSACs && mIntervalsBuffer[lane].empty()) {
82  const auto nValues = mIDCOneBuffer[lane][side].mIDCOne.size();
83  const int nIntervals = nValues / mIntervalsSACs;
84  const int nFirstInterval = nValues % mIntervalsSACs;
85  if (nFirstInterval == 0) {
86  mIntervalsBuffer[lane] = std::vector<unsigned int>(nIntervals, mIntervalsSACs);
87  } else {
88  mIntervalsBuffer[lane] = std::vector<unsigned int>(nIntervals + 1, mIntervalsSACs);
89  mIntervalsBuffer[lane].front() = nFirstInterval;
90  }
91  }
92  }
93 
94  FourierCoeffSAC coeffSAC;
95  if (lane == mExpectedInputLane) {
96  const int nSides = mIDCOneBuffer[lane][Side::A].mIDCOne.empty() + mIDCOneBuffer[lane][Side::C].mIDCOne.empty();
97  // int iProcessLane = lane;
98  for (int iProcessLaneTmp = 0; iProcessLaneTmp < mInputLanes; ++iProcessLaneTmp) {
99  const int nSidesCurrLane = mIDCOneBuffer[mExpectedInputLane][Side::A].mIDCOne.empty() + mIDCOneBuffer[mExpectedInputLane][Side::C].mIDCOne.empty();
100  if (nSidesCurrLane != nSides) {
101  break;
102  }
103 
104  for (int iSide = 0; iSide < SIDES; ++iSide) {
105  const Side side = (iSide == 0) ? A : C;
106  if (mIDCOneBuffer[mExpectedInputLane][side].mIDCOne.empty()) {
107  continue;
108  }
109  LOGP(info, "Processing input lane: {} for Side: {}", mExpectedInputLane, iSide);
110 
111  // perform fourier transform of 1D-IDCs
112  mIDCFourierTransform[side].setIDCs(std::move(mIDCOneBuffer[mExpectedInputLane][side]), mIntervalsBuffer[mExpectedInputLane]);
113  mIDCFourierTransform[side].calcFourierCoefficients(mIntervalsBuffer[mExpectedInputLane].size());
114 
115  if (!mProcessSACs) {
116  o2::ccdb::CcdbObjectInfo ccdbInfo(CDBTypeMap.at(((side == 0) ? CDBType::CalIDCFourierA : CDBType::CalIDCFourierC)), std::string{}, std::string{}, std::map<std::string, std::string>{}, mCCDBBuffer[mExpectedInputLane].front(), mCCDBBuffer[mExpectedInputLane].back());
117  auto imageFFT = o2::ccdb::CcdbApi::createObjectImage(&mIDCFourierTransform[side].getFourierCoefficients(), &ccdbInfo);
118  LOGP(info, "Sending object {} / {} of size {} bytes, valid for {} : {} ", ccdbInfo.getPath(), ccdbInfo.getFileName(), imageFFT->size(), ccdbInfo.getStartValidityTimestamp(), ccdbInfo.getEndValidityTimestamp());
119  pc.outputs().snapshot(Output{o2::calibration::Utils::gDataOriginCDBPayload, getDataDescriptionCCDBFourier(), 0}, *imageFFT.get());
120  pc.outputs().snapshot(Output{o2::calibration::Utils::gDataOriginCDBWrapper, getDataDescriptionCCDBFourier(), 0}, ccdbInfo);
121  } else {
122  coeffSAC.mCoeff[side] = mIDCFourierTransform[side].getFourierCoefficients();
123  }
124 
125  if (mDebug) {
126  LOGP(info, "dumping FT to file");
127  mIDCFourierTransform[side].dumpToFile(fmt::format("FourierAGG_{:02}_side{}.root", processing_helpers::getCurrentTF(pc), side).data());
128  }
129 
130  if (mSendOutDebug) {
131  sendOutput(pc.outputs(), side);
132  }
133  }
134 
135  if (mProcessSACs) {
136  o2::ccdb::CcdbObjectInfo ccdbInfo(CDBTypeMap.at(CDBType::CalSACFourier), std::string{}, std::string{}, std::map<std::string, std::string>{}, mCCDBBuffer[mExpectedInputLane].front(), mCCDBBuffer[mExpectedInputLane].back());
137  auto imageFFT = o2::ccdb::CcdbApi::createObjectImage(&coeffSAC, &ccdbInfo);
138  LOGP(info, "Sending object {} / {} of size {} bytes, valid for {} : {} ", ccdbInfo.getPath(), ccdbInfo.getFileName(), imageFFT->size(), ccdbInfo.getStartValidityTimestamp(), ccdbInfo.getEndValidityTimestamp());
139  pc.outputs().snapshot(Output{o2::calibration::Utils::gDataOriginCDBPayload, getDataDescriptionCCDBFourier(), 0}, *imageFFT.get());
140  pc.outputs().snapshot(Output{o2::calibration::Utils::gDataOriginCDBWrapper, getDataDescriptionCCDBFourier(), 0}, ccdbInfo);
141  }
142  mExpectedInputLane = ++mExpectedInputLane % mInputLanes;
143  }
144  }

183 DataProcessorSpec getTPCFourierTransformAggregatorSpec(const unsigned int rangeIDC, const unsigned int nFourierCoefficientsStore, const bool debug = false, const bool senddebug = false, const bool processSACs = false, const int inputLanes = 1)
184 {
185  std::vector<OutputSpec> outputSpecs;
186  outputSpecs.emplace_back(ConcreteDataTypeMatcher{o2::calibration::Utils::gDataOriginCDBPayload, TPCFourierTransformAggregatorSpec::getDataDescriptionCCDBFourier()}, Lifetime::Sporadic);
187  outputSpecs.emplace_back(ConcreteDataTypeMatcher{o2::calibration::Utils::gDataOriginCDBWrapper, TPCFourierTransformAggregatorSpec::getDataDescriptionCCDBFourier()}, Lifetime::Sporadic);
188 
189  if (senddebug) {
190  outputSpecs.emplace_back(ConcreteDataTypeMatcher{gDataOriginTPC, TPCFourierTransformAggregatorSpec::getDataDescriptionFourier()}, Lifetime::Sporadic);
191  }
192 
193  std::vector<InputSpec> inputSpecs;
194  if (!processSACs) {
195  inputSpecs.emplace_back(InputSpec{"idcone", ConcreteDataTypeMatcher{gDataOriginTPC, TPCFactorizeIDCSpec<>::getDataDescriptionIDC1()}, Lifetime::Sporadic});
196  inputSpecs.emplace_back(InputSpec{"tsccdb", gDataOriginTPC, TPCFactorizeIDCSpec<>::getDataDescriptionTimeStamp(), Lifetime::Sporadic});
197  inputSpecs.emplace_back(InputSpec{"intervals", gDataOriginTPC, TPCFactorizeIDCSpec<>::getDataDescriptionIntervals(), Lifetime::Sporadic});
198  inputSpecs.emplace_back(InputSpec{"lane", gDataOriginTPC, TPCFactorizeIDCSpec<>::getDataDescriptionLane(), Lifetime::Sporadic});
199  } else {
200  inputSpecs.emplace_back(InputSpec{"sacone", ConcreteDataTypeMatcher{gDataOriginTPC, TPCFactorizeSACSpec::getDataDescriptionSAC1()}, Lifetime::Sporadic});
201  inputSpecs.emplace_back(InputSpec{"tsccdb", gDataOriginTPC, TPCFactorizeSACSpec::getDataDescriptionTimeStamp(), Lifetime::Sporadic});
202  inputSpecs.emplace_back(InputSpec{"lane", gDataOriginTPC, TPCFactorizeSACSpec::getDataDescriptionLane(), Lifetime::Sporadic});
203  }
204 
205  return DataProcessorSpec{
206  "tpc-aggregator-ft",
207  inputSpecs,
208  outputSpecs,
209  AlgorithmSpec{adaptFromTask<TPCFourierTransformAggregatorSpec>(nFourierCoefficientsStore, rangeIDC, debug, senddebug, processSACs, inputLanes)},
210  Options{{"intervalsSACs", VariantType::Int, 11, {"Number of integration intervals which will be sampled for the fourier coefficients"}}}};
211 }
