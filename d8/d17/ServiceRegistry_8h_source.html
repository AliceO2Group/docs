233  // Lookup a given @a typeHash for a given @a threadId at
234  // a unique (per typeHash) location. There might
235  // be other typeHash which sit in the same place, but
236  // the if statement will rule them out. As long as
237  // only one thread writes in a given i + id location
238  // as guaranteed by the atomic, mServicesKey[i + id] will
239  // either be 0 or the final value.
240  // This method should NEVER register a new service, event when requested.

269  template <class I, class C, enum ServiceKind K = ServiceKind::Serial>
270  void registerService(C* service, Salt salt = ServiceRegistry::globalDeviceSalt())
271  {
272  // This only works for concrete implementations of the type T.

278  ServiceRegistry::registerService(typeHash, reinterpret_cast<void*>(service), K, salt, typeid(C).name());

295  template <typename T>
296  std::enable_if_t<std::is_const_v<T> == false, bool> active(Salt salt) const
297  {
298  constexpr ServiceTypeHash typeHash{TypeIdHelpers::uniqueId<T>()};
299  if (this->getPos(typeHash, GLOBAL_CONTEXT_SALT) != -1) {
300  return true;
301  }
302  auto result = this->getPos(typeHash, salt) != -1;
303  return result;
304  }

309  template <typename T>
310  T& get(Salt salt) const
311  {
312  constexpr ServiceTypeHash typeHash{TypeIdHelpers::uniqueId<T>()};
313  auto ptr = this->get(typeHash, salt, ServiceKindExtractor<T>::kind, typeid(T).name());
314  if (O2_BUILTIN_LIKELY(ptr != nullptr)) {
315  if constexpr (std::is_const_v<T>) {
316  return *reinterpret_cast<T const*>(ptr);
317  } else {
318  return *reinterpret_cast<T*>(ptr);
319  }
320  }
321  throwError(typeid(T).name(), typeHash.hash, salt.streamId, salt.dataProcessorId);
322  O2_BUILTIN_UNREACHABLE();
323  }
