70  static constexpr uint32_t ExpectChipHeader = 0x1 << 0;
71  static constexpr uint32_t ExpectChipTrailer = 0x1 << 1;
72  static constexpr uint32_t ExpectChipEmpty = 0x1 << 2;
73  static constexpr uint32_t ExpectRegion = 0x1 << 3;
74  static constexpr uint32_t ExpectData = 0x1 << 4;
75  static constexpr uint32_t ExpectBUSY = 0x1 << 5;
76  static constexpr uint32_t ExpectNextChip = ExpectChipHeader | ExpectChipEmpty;
77  static constexpr int NRows = 512;
78  static constexpr int RowMask = NRows - 1;
79  static constexpr int NCols = 1024;
80  static constexpr int NRegions = 32;
81  static constexpr int NDColInReg = NCols / NRegions / 2;
82  static constexpr int HitMapSize = 7;
83 
84  // masks for records components
85  static constexpr uint32_t MaskEncoder = 0x3c00; // encoder (double column) ID takes 4 bit max (0:15)
86  static constexpr uint32_t MaskPixID = 0x3ff; // pixel ID within encoder (double column) takes 10 bit max (0:1023)
87  static constexpr uint32_t MaskDColID = MaskEncoder | MaskPixID; // mask for encoder + dcolumn combination
88  static constexpr uint32_t MaskRegion = 0x1f; // region ID takes 5 bits max (0:31)
89  static constexpr uint32_t MaskChipID = 0x0f; // chip id in module takes 4 bit max
90  static constexpr uint32_t MaskROFlags = 0x0f; // RO flags in chip trailer takes 4 bit max
91  static constexpr uint8_t MaskErrBusyViolation = 0x1 << 3;
92  static constexpr uint8_t MaskErrDataOverrun = 0x3 << 2;
93  static constexpr uint8_t MaskErrFatal = 0x7 << 1;
94  static constexpr uint8_t MaskErrFlushedIncomplete = 0x1 << 2;
95  static constexpr uint8_t MaskErrStrobeExtended = 0x1 << 1;
96  static constexpr uint32_t MaskTimeStamp = 0xff; // Time stamps as BUNCH_COUNTER[10:3] bits
97  static constexpr uint32_t MaskReserved = 0xff; // mask for reserved byte
98  static constexpr uint32_t MaskHitMap = 0x7f; // mask for hit map: at most 7 hits in bits (0:6)
99  //
100  // flags for data records
101  static constexpr uint32_t REGION = 0xc0; // flag for region
102  static constexpr uint32_t REGION_MASK = 0xe0; // mask for detecting the region
103  static constexpr uint32_t CHIPHEADER = 0xa0; // flag for chip header
104  static constexpr uint32_t CHIPTRAILER = 0xb0; // flag for chip trailer
105  static constexpr uint32_t CHIPEMPTY = 0xe0; // flag for empty chip
106  static constexpr uint32_t DATALONG = 0x0000; // flag for DATALONG
107  static constexpr uint32_t DATASHORT = 0x4000; // flag for DATASHORT
108  static constexpr uint32_t BUSYOFF = 0xf0; // flag for BUSY_OFF
109  static constexpr uint32_t BUSYON = 0xf1; // flag for BUSY_ON
110  static constexpr uint32_t ERROR_MASK = 0xf0; // flag for all error triggers

114  static bool isData(uint8_t v) { return (v & (0x1 << 7)) == 0; }

127  template <class T, typename CG>
128  static int decodeChip(ChipPixelData& chipData, T& buffer, std::vector<uint16_t>& seenChips, CG cidGetter)
129  {
130  // read record for single non-empty chip, updating on change module and cycle.
131  // return number of records filled (>0), EOFFlag or Error
132  //
133  bool needSorting = false; // if DColumns order is wrong, do explicit reordering
134  auto roErrHandler = [&chipData](uint8_t roErr) {
135 #ifdef ALPIDE_DECODING_STAT
136  if (roErr == MaskErrBusyViolation) {
137  chipData.setError(ChipStat::BusyViolation);
138  } else if (roErr == MaskErrDataOverrun) {
139  chipData.setError(ChipStat::DataOverrun);
140  } else if (roErr == MaskErrFatal) {
141  chipData.setError(ChipStat::Fatal);
142  } else if (roErr == MaskErrFlushedIncomplete) {
143  chipData.setError(ChipStat::FlushedIncomplete);
144  } else if (roErr == MaskErrStrobeExtended) {
145  chipData.setError(ChipStat::StrobeExtended);
146  }
147 #endif
148  };
149 
150  uint8_t dataC = 0, timestamp = 0;
151  uint16_t dataS = 0, region = 0;
152 #ifdef ALPIDE_DECODING_STAT
153  uint16_t rowPrev = 0xffff;
154 #endif
155  //
156  int nRightCHits = 0; // counter for the hits in the right column of the current double column
157  std::uint16_t rightColHits[NRows]; // buffer for the accumulation of hits in the right column
158  std::uint16_t colDPrev = 0xffff; // previously processed double column (to dected change of the double column)
159 
160  uint32_t expectInp = ExpectNextChip; // data must always start with chip header or chip empty flag
161 
162  chipData.clear();
163  LOG(debug) << "NewEntry";
164  while (buffer.next(dataC)) {
165  //
166  LOGP(debug, "dataC: {:#x} expect {:#b}", int(dataC), int(expectInp));
167 
168  // Busy ON / OFF can appear at any point of the data stream, checking it with priority
169  if (dataC == BUSYON) {
170 #ifdef ALPIDE_DECODING_STAT
171  chipData.setError(ChipStat::BusyOn);
172 #endif
173  continue;
174  }
175  if (dataC == BUSYOFF) {
176 #ifdef ALPIDE_DECODING_STAT
177  chipData.setError(ChipStat::BusyOff);
178 #endif
179  continue;
180  }
181 
182  if ((expectInp & ExpectChipEmpty) && isChipEmpty(dataC)) { // empty chip was expected
183  uint16_t chipIDGlo = cidGetter(dataC & MaskChipID);
184  if (chipIDGlo == 0xffff) {
185  chipData.setChipID(chipIDGlo);
186 #ifdef ALPIDE_DECODING_STAT
187  chipData.setErrorInfo(dataC & MaskChipID);
188  chipData.setError(ChipStat::WrongAlpideChipID);
189 #endif
190  chipData.getData().clear();
191  return unexpectedEOF("CHIP_EMPTY:WrongChipID"); // abandon cable data
192  }
193  chipData.setChipID(chipIDGlo); // here we set the global chip ID
194  if (!buffer.next(timestamp)) {
195 #ifdef ALPIDE_DECODING_STAT
196  chipData.setError(ChipStat::TruncatedChipEmpty);
197 #endif
198  return unexpectedEOF("CHIP_EMPTY:Timestamp"); // abandon cable data
199  }
200  seenChips.push_back(chipIDGlo);
201  chipData.resetChipID();
202  expectInp = ExpectNextChip;
203  continue;
204  }
205 
206  if ((expectInp & ExpectChipHeader) && isChipHeader(dataC)) { // chip header was expected
207  uint16_t chipIDGlo = cidGetter(dataC & MaskChipID);
208  if (chipIDGlo == 0xffff) {
209  chipData.setChipID(chipIDGlo);
210 #ifdef ALPIDE_DECODING_STAT
211  chipData.setErrorInfo(dataC & MaskChipID);
212  chipData.setError(ChipStat::WrongAlpideChipID);
213 #endif
214  chipData.getData().clear();
215  return unexpectedEOF("CHIP_EMPTY:WrongChipID"); // abandon cable data
216  }
217  chipData.setChipID(chipIDGlo); // here we set the global chip ID
218  if (!buffer.next(timestamp)) {
219 #ifdef ALPIDE_DECODING_STAT
220  chipData.setError(ChipStat::TruncatedChipHeader);
221 #endif
222  return unexpectedEOF("CHIP_HEADER"); // abandon cable data
223  }
224  expectInp = ExpectRegion; // now expect region info
225  continue;
226  }
227 
228  // region info ?
229  if ((expectInp & ExpectRegion) && (dataC & REGION_MASK) == REGION) { // chip header was seen, or hit data read
230  region = dataC & MaskRegion;
231  expectInp = ExpectData;
232  continue;
233  }
234 
235  if ((expectInp & ExpectChipTrailer) && isChipTrailer(dataC)) { // chip trailer was expected
236  expectInp = ExpectNextChip;
237  chipData.setROFlags(dataC & MaskROFlags);
238 #ifdef ALPIDE_DECODING_STAT
239  uint8_t roErr = dataC & MaskROFlags;
240  if (roErr) {
241  roErrHandler(roErr);
242  }
243 #endif
244  // in case there are entries in the "right" columns buffer, add them to the container
245  if (nRightCHits) {
246  colDPrev++;
247  for (int ihr = 0; ihr < nRightCHits; ihr++) {
248  addHit(chipData, rightColHits[ihr], colDPrev);
249  }
250  }
251  break;
252  }
253 
254  // hit info ?
255  if ((expectInp & ExpectData)) {
256  if (isData(dataC)) { // region header was seen, expect data
257  // note that here we are checking on the byte rather than the short, need complete to ushort
258  dataS = dataC << 8;
259  if (!buffer.next(dataC)) {
260 #ifdef ALPIDE_DECODING_STAT
261  chipData.setError(ChipStat::TruncatedRegion);
262 #endif
263  return unexpectedEOF("CHIPDATA"); // abandon cable data
264  }
265  dataS |= dataC;
266  LOGP(debug, "dataC: {:#x} dataS: {:#x} expect {:#b} in ExpectData", int(dataC), int(dataS), int(expectInp));
267 
268  // we are decoding the pixel addres, if this is a DATALONG, we will fetch the mask later
269  uint16_t dColID = (dataS & MaskEncoder) >> 10;
270  uint16_t pixID = dataS & MaskPixID;
271 
272  // convert data to usual row/pixel format
273  uint16_t row = pixID >> 1;
274  // abs id of left column in double column
275  uint16_t colD = (region * NDColInReg + dColID) << 1; // TODO consider <<4 instead of *NDColInReg?
276  bool rightC = (row & 0x1) ? !(pixID & 0x1) : (pixID & 0x1); // true for right column / lalse for left
277 
278  if (row == rowPrev && colD == colDPrev) {
279  // this is a special test to exclude repeated data of the same pixel fired
280 #ifdef ALPIDE_DECODING_STAT
281  chipData.setError(ChipStat::RepeatingPixel);
282  chipData.addErrorInfo((uint64_t(colD + rightC) << 16) | uint64_t(row));
283 #endif
284  if ((dataS & (~MaskDColID)) == DATALONG) { // skip pattern w/o decoding
285  uint8_t hitsPattern = 0;
286  if (!buffer.next(hitsPattern)) {
287 #ifdef ALPIDE_DECODING_STAT
288  chipData.setError(ChipStat::TruncatedLondData);
289 #endif
290  return unexpectedEOF("CHIP_DATA_LONG:Pattern"); // abandon cable data
291  }
292  if (hitsPattern & (~MaskHitMap)) {
293 #ifdef ALPIDE_DECODING_STAT
294  chipData.setError(ChipStat::WrongDataLongPattern);
295 #endif
296  return unexpectedEOF("CHIP_DATA_LONG:Pattern"); // abandon cable data
297  }
298  LOGP(debug, "hitsPattern: {:#b} expect {:#b}", int(hitsPattern), int(expectInp));
299  }
300  expectInp = ExpectChipTrailer | ExpectData | ExpectRegion;
301  continue; // end of DATA(SHORT or LONG) processing
302  } else if (colD != colDPrev) {
303  // if we start new double column, transfer the hits accumulated in the right column buffer of prev. double column
304  if (colD < colDPrev && colDPrev != 0xffff) {
305 #ifdef ALPIDE_DECODING_STAT
306  chipData.setError(ChipStat::WrongDColOrder); // abandon cable data
307 #endif
308  return unexpectedEOF("Wrong column order"); // abandon cable data
309  needSorting = true; // effectively disabled
310  }
311  colDPrev++;
312  for (int ihr = 0; ihr < nRightCHits; ihr++) {
313  addHit(chipData, rightColHits[ihr], colDPrev);
314  }
315  nRightCHits = 0; // reset the buffer
316  }
317  rowPrev = row;
318  colDPrev = colD;
319 
320  // we want to have hits sorted in column/row, so the hits in right column of given double column
321  // are first collected in the temporary buffer
322  // real columnt id is col = colD + 1;
323  if (rightC) {
324  rightColHits[nRightCHits++] = row; // col = colD+1
325  } else {
326  addHit(chipData, row, colD); // col = colD, left column hits are added directly to the container
327  }
328 
329  if ((dataS & (~MaskDColID)) == DATALONG) { // multiple hits ?
330  uint8_t hitsPattern = 0;
331  if (!buffer.next(hitsPattern)) {
332 #ifdef ALPIDE_DECODING_STAT
333  chipData.setError(ChipStat::TruncatedLondData);
334 #endif
335  return unexpectedEOF("CHIP_DATA_LONG:Pattern"); // abandon cable data
336  }
337  LOGP(debug, "hitsPattern: {:#b} expect {:#b}", int(hitsPattern), int(expectInp));
338  if (hitsPattern & (~MaskHitMap)) {
339 #ifdef ALPIDE_DECODING_STAT
340  chipData.setError(ChipStat::WrongDataLongPattern);
341 #endif
342  return unexpectedEOF("CHIP_DATA_LONG:Pattern"); // abandon cable data
343  }
344  for (int ip = 0; ip < HitMapSize; ip++) {
345  if (hitsPattern & (0x1 << ip)) {
346  uint16_t addr = pixID + ip + 1, rowE = addr >> 1;
347  if (addr & ~MaskPixID) {
348 #ifdef ALPIDE_DECODING_STAT
349  chipData.setError(ChipStat::WrongRow);
350 #endif
351  return unexpectedEOF(fmt::format("Non-existing encoder {} decoded, DataLong was {:x}", pixID, dataS)); // abandon cable data
352  }
353  rightC = ((rowE & 0x1) ? !(addr & 0x1) : (addr & 0x1)); // true for right column / lalse for left
354  // the real columnt is int colE = colD + rightC;
355  if (rightC) { // same as above
356  rightColHits[nRightCHits++] = rowE;
357  } else {
358  addHit(chipData, rowE, colD + rightC); // left column hits are added directly to the container
359  }
360  }
361  }
362  }
363  } else if (ChipStat::getAPENonCritical(dataC) >= 0) { // check for recoverable APE, if on: continue with ExpectChipTrailer | ExpectData | ExpectRegion expectation
364 #ifdef ALPIDE_DECODING_STAT
365  chipData.setError(ChipStat::DecErrors(ChipStat::getAPENonCritical(dataC)));
366 #endif
367  } else {
368 #ifdef ALPIDE_DECODING_STAT
369  chipData.setError(ChipStat::NoDataFound);
370 #endif
371  return unexpectedEOF(fmt::format("Expected DataShort or DataLong mask, got {:x}", dataS)); // abandon cable data
372  }
373  expectInp = ExpectChipTrailer | ExpectData | ExpectRegion;
374  continue; // end of DATA(SHORT or LONG) processing
375  }
376 
377  if (!dataC) {
378  if (expectInp == ExpectNextChip) {
379  continue;
380  }
381  chipData.setError(ChipStat::TruncatedBuffer);
382  return unexpectedEOF("Abandon on 0-padding"); // abandon cable data
383  }
384 
385  if ((expectInp & ExpectRegion) && isChipTrailer(dataC)) {
386  if (dataC & MaskROFlags) {
387  // in case of BUSY VIOLATION the Trailer may come directly after the Header
388  expectInp = ExpectNextChip;
389  chipData.setROFlags(dataC & MaskROFlags);
390  roErrHandler(dataC & MaskROFlags);
391  break;
392  } else {
393 #ifdef ALPIDE_DECODING_STAT
394  chipData.setError(ChipStat::TrailerAfterHeader);
395 #endif
396  return unexpectedEOF("Trailer after header"); // abandon cable data
397  }
398  }
399 
400  // check for APE errors, see https://alice.its.cern.ch/jira/browse/O2-1717?focusedCommentId=274714&page=com.atlassian.jira.plugin.system.issuetabpanels:comment-tabpanel#comment-274714
401  bool fatalAPE = false;
402  auto codeAPE = ChipStat::getAPECode(dataC, fatalAPE);
403  if (codeAPE >= 0) {
404 #ifdef ALPIDE_DECODING_STAT
405  chipData.setError(ChipStat::DecErrors(codeAPE));
406 #endif
407  if (fatalAPE) {
408  return unexpectedEOF(fmt::format("APE error {:#02x} [expectation = {:#02x}]", int(dataC), int(expectInp))); // abandon cable data
409  } else {
410  LOGP(error, "Code should not have entered here, APE: {:#02x}, expectation: {:#02x}", codeAPE, int(expectInp));
411  return unexpectedEOF(fmt::format("APE error {:#02x} [expectation = {:#02x}]", int(dataC), int(expectInp))); // abandon cable data
412  }
413  }
414 #ifdef ALPIDE_DECODING_STAT
415  chipData.setError(ChipStat::UnknownWord);
416  // fill the error buffer with a few bytes of wrong data
417  const uint8_t* begPtr = buffer.data();
418  const uint8_t* endPtr = buffer.getEnd();
419  const uint8_t* curPtr = buffer.getPtr();
420  size_t offsBack = std::min(ChipPixelData::MAXDATAERRBYTES - ChipPixelData::MAXDATAERRBYTES_AFTER, size_t(curPtr - begPtr));
421  size_t offsAfter = std::min(ChipPixelData::MAXDATAERRBYTES_AFTER, size_t(endPtr - curPtr));
422  std::memcpy(chipData.getRawErrBuff().data(), curPtr - offsBack, offsBack + offsAfter);
423  chipData.setNBytesInRawBuff(offsBack + offsAfter);
424 #endif
425  return unexpectedEOF(fmt::format("Unknown word 0x{:x} [expectation = 0x{:x}]", int(dataC), int(expectInp))); // abandon cable data
426  }
427 
428  if (!(expectInp & ExpectNextChip)) {
429 #ifdef ALPIDE_DECODING_STAT
430  chipData.setError(ChipStat::TruncatedRegion);
431 #endif
432  return unexpectedEOF("Missing CHIP_TRAILER"); // abandon cable data
433  }
434 
435  if (needSorting && chipData.getData().size()) { // d.columns were in a wrong order, need to sort the data, RS: effectively disabled
436  LOGP(error, "This code path should have been disabled");
437  auto& pixData = chipData.getData();
438  std::sort(pixData.begin(), pixData.end(),
439  [](PixelData& a, PixelData& b) { return a.getCol() < b.getCol() || (a.getCol() == b.getCol() && a.getRowDirect() < b.getRowDirect()); });
440  // if the columns ordering was wrong, detection of same pixel fired twice might have failed, make sure there are no duplicates
441  auto currPix = pixData.begin(), prevPix = currPix++;
442  while (currPix != pixData.end()) {
443  if (prevPix->getCol() == currPix->getCol() && prevPix->getRowDirect() == currPix->getRowDirect()) {
444  currPix = pixData.erase(prevPix);
445  }
446  prevPix = currPix++;
447  }
448  }
449  if (chipData.getData().size()) {
450  seenChips.push_back(chipData.getChipID());
451  }
452  return chipData.getData().size();
453  }

457  template <typename LG, typename CG>
458  static bool verifyDecodedCable(
459  std::map<int, ChipPixelData*>& seenChips, PayLoadCont& buffer,
460  std::vector<uint16_t>& seenChipIDs, LG lidGetter, CG cidGetter)
461  {
462  PayLoadCont reconstructedData;
463 
464  // Ensure the length of the reconstructed buffer.
465  int bufferLength = 0;
466  for (auto it = seenChips.begin(); it != seenChips.end(); ++it) {
467  bufferLength += it->second->getData().size();
468  }
469  bufferLength += seenChipIDs.size() * 2;
470  reconstructedData.ensureFreeCapacity(40 * bufferLength);
471 
472  // Encode the seen chips in the order they were decoded.
473  for (int ID : seenChipIDs) {
474  ChipPixelData currentChip;
475  int localID = lidGetter(ID);
476  if (seenChips.count(ID)) {
477  currentChip = *seenChips[ID];
478  }
479  AlpideCoder encoder;
480  encoder.encodeChip(reconstructedData, currentChip, localID,
481  /*dummy bc*/ 0, currentChip.getROFlags());
482  }
483 
484  // Pad the end of the reconstructed buffer with the zero bytes.
485  if (buffer.getSize() > reconstructedData.getSize())
486  reconstructedData.fill(0x00,
487  buffer.getSize() - reconstructedData.getSize());
488 
489  auto hexToString = [](uint8_t v) {
490  std::stringstream ss;
491  ss << "0x" << std::setfill('0') << std::setw(2) << std::hex
492  << std::uppercase << (0xFF & v);
493  return ss.str();
494  };
495 
496  auto reportError = [&](std::string message) {
497  LOG(error) << "Error during decoder verification: " << message;
498  LOG(debug) << "Raw Data:";
499  buffer.rewind();
500  uint8_t dataC = 0;
501  int index = 1;
502  while (buffer.next(dataC)) {
503  LOG(debug) << index++ << ". " << hexToString(dataC);
504  }
505  LOG(debug) << "Reconstructed Data:";
506  reconstructedData.rewind();
507  index = 1;
508  while (reconstructedData.next(dataC)) {
509  LOG(debug) << index++ << ". " << hexToString(dataC);
510  }
511  };
512 
513  // The reconstructed buffer is very similar to the original data flow
514  // with the exception to several pieces of information that get lost
515  // during the decoding:
516  // 1. Error trigger words: these are absent in the reconstructed buffer.
517  // In case of BUSYON/BUSYOFF words, the verification is allowed to
518  // continue.
519  // 2. Bunch counter for frame: the reconstructed buffer does contain the
520  // corresponding words, but has a dummy value.
521 
522  ChipPixelData* currentChip = nullptr;
523  if (seenChipIDs.size())
524  currentChip = seenChips[seenChipIDs[0]];
525  buffer.rewind();
526  while (true) {
527  uint8_t dataRec = 0;
528  uint8_t dataRaw = 0;
529 
530  if (reconstructedData.isEmpty() || buffer.isEmpty()) {
531  // If either buffer is empty, verify that both buffers reached the end.
532  // If one of the streams is non-empty, then verify that the remaining
533  // bytes are zeroes.
534  if (reconstructedData.isEmpty() && buffer.isEmpty()) {
535  break;
536  }
537  PayLoadCont& nonEmptyBuffer =
538  !buffer.isEmpty() ? buffer : reconstructedData;
539  uint8_t dataC = 0;
540  while (nonEmptyBuffer.next(dataC)) {
541  if (dataC != 0x00) {
542  reportError("Buffer sizes mismatch.");
543  return false;
544  }
545  }
546  break;
547  }
548 
549  reconstructedData.current(dataRec);
550  buffer.current(dataRaw);
551  if (dataRaw == dataRec) {
552  if (isChipHeaderOrEmpty(dataRaw)) {
553  uint16_t ID = cidGetter(dataRaw & MaskChipID);
554  if (seenChips.count(ID)) {
555  currentChip = seenChips[ID];
556  } else {
557  currentChip = nullptr;
558  }
559  // If the data correspond to the CHIPHEADER or CHIPEMPTY data words,
560  // skip the next byte that represent bunch counters.
561  buffer.next(dataRaw);
562  reconstructedData.next(dataRec);
563  }
564  buffer.next(dataRaw);
565  reconstructedData.next(dataRec);
566  continue;
567  }
568 
569  if (dataRaw == BUSYON || dataRaw == BUSYOFF) {
570  // Placement of BUSYON and BUSYOFF triggers is arbitrary, just ignore
571  // the byte in the raw stream and move forward.
572  buffer.next(dataRaw);
573  continue;
574  }
575 
576  VerifierMismatchResult res =
577  handleVerifierMismatch(buffer, reconstructedData, currentChip);
578  switch (res) {
579  case VerifierMismatchResult::RESOLVED:
580  LOG(debug) << "Mismatch " << hexToString(dataRaw) << " / "
581  << hexToString(dataRec)
582  << " was resolved, able to continue verification";
583  continue;
584  case VerifierMismatchResult::EXPECTED_MISMATCH:
585  LOG(debug) << "Mismatch " << hexToString(dataRaw) << " / "
586  << hexToString(dataRec)
587  << " was expected, aborting the verification";
588  return true;
589  case VerifierMismatchResult::UNEXPECTED_MISMATCH: {
590  // If the read bytes is not related to the special cases, report
591  // error.
592  std::stringstream errorStream;
593  errorStream
594  << "Unexpected byte mismatch during decoder verification. "
595  "Expected: "
596  << hexToString(dataRaw)
597  << ", Reconstructed: " << hexToString(dataRec);
598  reportError(errorStream.str());
599  }
600  }
601  return false;
602  }
603  return true;
604  }

606  enum VerifierMismatchResult {
607  UNEXPECTED_MISMATCH, // Genuine mismatch, stop verification
608  EXPECTED_MISMATCH, // Mismatch expected, need to abort verification
609  RESOLVED // Mismatch resolved, can continue
610  };

612  static VerifierMismatchResult handleVerifierMismatch(
613  PayLoadCont& buffer, PayLoadCont& reconstructedData,
614  ChipPixelData* currentChip)
615  {
616  VerifierMismatchResult res = VerifierMismatchResult::UNEXPECTED_MISMATCH;
617  uint8_t dataRec = 0;
618  uint8_t dataRaw = 0;
619  reconstructedData.current(dataRec);
620  buffer.current(dataRaw);
621  auto inner = [&](int errIdx) {
622  if (res != VerifierMismatchResult::UNEXPECTED_MISMATCH ||
623  dataRaw == dataRec) {
624  // The mismatch was resolved, no need to check the rest of the errors
625  return;
626  }
627  switch (errIdx) {
628  case ChipStat::BusyViolation:
629  case ChipStat::DataOverrun:
630  case ChipStat::Fatal:
631  break;
632  case ChipStat::BusyOn:
633  case ChipStat::BusyOff:
634  // We don't need to do anything with these errors.
635  break;
636  case ChipStat::TruncatedChipEmpty:
637  // This error cannot cause mismatch since it can be reconstructed
638  // via the encoder.
639  break;
640  case ChipStat::TruncatedChipHeader:
641  case ChipStat::TruncatedRegion:
642  if (isChipHeader(dataRaw) && isChipEmpty(dataRec)) {
643  // In case of TruncatedChipHeader, the raw data must have a chip
644  // header while the reconstructed chip is empty. The verifier
645  // cannot continue the verification further.
646  res = VerifierMismatchResult::EXPECTED_MISMATCH;
647  }
648  break;
649  case ChipStat::TruncatedLondData:
650  case ChipStat::WrongDataLongPattern:
651  if (isData(dataRaw) && isChipTrailer(dataRec)) {
652  // If the decoder encountered an issue with DATALONG, the verifier
653  // must have a mismatch between data on the raw stream and trailer
654  // on the reconstructed stream
655  res = VerifierMismatchResult::EXPECTED_MISMATCH;
656  }
657  break;
658  case ChipStat::NoDataFound:
659  case ChipStat::UnknownWord:
660  case ChipStat::RepeatingPixel:
661  case ChipStat::WrongRow:
662  break;
663  case ChipStat::APE_STRIP_START:
664  case ChipStat::APE_ILLEGAL_CHIPID:
665  case ChipStat::APE_DET_TIMEOUT:
666  case ChipStat::APE_OOT:
667  case ChipStat::APE_PROTOCOL_ERROR:
668  case ChipStat::APE_LANE_FIFO_OVERFLOW_ERROR:
669  case ChipStat::APE_FSM_ERROR:
670  case ChipStat::APE_PENDING_DETECTOR_EVENT_LIMIT:
671  case ChipStat::APE_PENDING_LANE_EVENT_LIMIT:
672  case ChipStat::APE_O2N_ERROR:
673  case ChipStat::APE_RATE_MISSING_TRG_ERROR: {
674  uint8_t errorByte = ChipStat::getAPEByte((ChipStat::DecErrors)errIdx);
675  if (dataRaw == errorByte) {
676  buffer.next(dataRaw); // Skipping error byte
677  // If we encountered the byte corresponding to the APE error,
678  // check that the rest of the raw stream consists of only
679  // padding.
680  while (buffer.next(dataRaw)) {
681  if (dataRaw != 0x00) {
682  break;
683  }
684  }
685  if (buffer.isEmpty()) {
686  res = VerifierMismatchResult::EXPECTED_MISMATCH;
687  }
688  }
689  break;
690  }
691  case ChipStat::APE_PE_DATA_MISSING:
692  case ChipStat::APE_OOT_DATA_MISSING:
693  case ChipStat::WrongDColOrder:
694  case ChipStat::InterleavedChipData:
695  case ChipStat::TruncatedBuffer:
696  break;
697  case ChipStat::TrailerAfterHeader:
698  if (isChipHeader(dataRaw) && isChipEmpty(dataRec)) {
699  // This error can be verified by skipping a bunch counter byte and
700  // checking that the following byte corresponds to the chip trailer
701  buffer.next(dataRaw); // Skipping chip header
702  buffer.next(dataRaw); // Skipping bunch counter
703  buffer.current(dataRaw);
704  if (isChipTrailer(dataRaw)) {
705  res = VerifierMismatchResult::EXPECTED_MISMATCH;
706  }
707  }
708  break;
709  case ChipStat::FlushedIncomplete:
710  case ChipStat::StrobeExtended:
711  break;
712  case ChipStat::WrongAlpideChipID:
713  // If the chip doesn't have a valid ID, we must stop the verification
714  res = VerifierMismatchResult::EXPECTED_MISMATCH;
715  break;
716  default:
717  LOG(error) << "Unknown error set by chip during verifier mismatch";
718  }
719  };
720  if (currentChip) {
721  currentChip->forEachSetError(inner);
722  }
723  return res;
724  }
