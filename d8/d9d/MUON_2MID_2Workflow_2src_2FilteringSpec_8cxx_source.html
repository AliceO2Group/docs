43 FilteringDeviceDPL(bool useMC, std::vector<of::OutputSpec> outputSpecs) : mUseMC(useMC)
44 {
45 mOutputs = specs::buildOutputs(outputSpecs);
46
47 if (useMC) {
48 mFillLabels = [](size_t inIdx, size_t outIdx, const o2::dataformats::MCTruthContainer<MCLabel>* inMCContainer, o2::dataformats::MCTruthContainer<MCLabel>& outMCContainer, const ColumnData& col) {
49 auto labels = inMCContainer->getLabels(inIdx);
50 for (auto& label : labels) {
51 if (label.getDEId() == col.deId && label.getColumnId() == col.columnId) {
52 bool hasLabel = false;
53 for (int istrip = label.getFirstStrip(), last = label.getLastStrip(); istrip <= last; ++istrip) {
54 if ((label.getCathode() == 1 && col.isNBPStripFired(istrip)) || (label.getCathode() == 0 && col.isBPStripFired(label.getStripInLine(istrip), label.getLine(istrip)))) {
55 hasLabel = true;
56 break;
57 }
58 }
59 if (hasLabel) {
60 outMCContainer.addElement(outIdx, label);
61 }
62 }
63 }
64 };

91 void run(of::ProcessingContext& pc)
92 {
93 // Triggers finalizeCCDB
94 pc.inputs().get<std::vector<ColumnData>*>("mid_bad_channels");
95 pc.inputs().get<std::vector<ColumnData>*>("mid_rejectlist");
96
97 auto data = specs::getData(pc, "mid_filter_in", EventType::Standard);
98 auto inROFRecords = specs::getRofs(pc, "mid_filter_in", EventType::Standard);
99
100 std::unique_ptr<const o2::dataformats::MCTruthContainer<MCLabel>> inMCContainer = mUseMC ? specs::getLabels(pc, "mid_filter_in") : nullptr;
101
102 auto& maskedData = pc.outputs().make<std::vector<ColumnData>>(of::OutputRef{"mid_filter_out_0"});
103 auto& maskedRofs = pc.outputs().make<std::vector<ROFRecord>>(of::OutputRef{"mid_filter_out_rof_0"});
104
105 maskedData.reserve(data.size());
106 maskedRofs.reserve(inROFRecords.size());
107
108 o2::dataformats::MCTruthContainer<MCLabel> outMCContainer;
109
110 for (auto& rof : inROFRecords) {
111 auto firstEntry = maskedData.size();
112 for (auto dataIt = data.begin() + rof.firstEntry, end = data.begin() + rof.getEndIndex(); dataIt != end; ++dataIt) {
113 auto col = *dataIt;
114 if (mMasksHandler.applyMask(col)) {
115 // Data are not fully masked
116 maskedData.emplace_back(col);
117 auto inIdx = std::distance(data.begin(), dataIt);
118 mFillLabels(inIdx, maskedData.size() - 1, inMCContainer.get(), outMCContainer, col);
119 }
120 }
121 auto nEntries = maskedData.size() - firstEntry;
122 if (nEntries > 0) {
123 maskedRofs.emplace_back(rof);
124 maskedRofs.back().firstEntry = firstEntry;
125 maskedRofs.back().nEntries = nEntries;
126 }
127 }
128
129 if (mUseMC) {
130 pc.outputs().snapshot(mOutputs[2], outMCContainer);
131 }
132 }

136 bool mUseMC{false};

138 std::function<void(size_t, size_t, const o2::dataformats::MCTruthContainer<MCLabel>*, o2::dataformats::MCTruthContainer<MCLabel>&, const ColumnData& col)> mFillLabels{[](size_t, size_t, const o2::dataformats::MCTruthContainer<MCLabel>*, o2::dataformats::MCTruthContainer<MCLabel>&, const ColumnData&) {}};
139 std::vector<ColumnData> mBadChannels{};
