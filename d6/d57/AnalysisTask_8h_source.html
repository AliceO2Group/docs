170  }() &&
171  ...);

201  static auto extractFilteredFromRecord(InputRecord& record, ExpressionInfo& info, pack<Os...> const&)
202  {
203  auto table = o2::soa::ArrowHelpers::joinTables(std::vector<std::shared_ptr<arrow::Table>>{extractTableFromRecord<Os>(record)...});
204  expressions::updateFilterInfo(info, table);
205  if constexpr (!o2::soa::is_smallgroups_v<std::decay_t<T>>) {
206  if (info.selection == nullptr) {
207  soa::missingFilterDeclaration(info.processHash, info.argumentIndex);
208  }
209  }
210  if constexpr (soa::is_soa_iterator_v<T>) {
211  return typename T::parent_t({table}, info.selection);
212  } else {
213  return T({table}, info.selection);
214  }

264  template <typename Task, typename R, typename C, typename Grouping, typename... Associated>
265  static void invokeProcess(Task& task, InputRecord& inputs, R (C::*processingFunction)(Grouping, Associated...), std::vector<ExpressionInfo>& infos, ArrowTableSlicingCache& slices)
266  {
267  using G = std::decay_t<Grouping>;
268  auto groupingTable = AnalysisDataProcessorBuilder::bindGroupingTable(inputs, processingFunction, infos);
269 
270  // set filtered tables for partitions with grouping
271  homogeneous_apply_refs([&groupingTable](auto& x) {
272  PartitionManager<std::decay_t<decltype(x)>>::setPartition(x, groupingTable);
273  PartitionManager<std::decay_t<decltype(x)>>::bindInternalIndices(x, &groupingTable);
274  return true;
275  },
276  task);
277 
278  if constexpr (sizeof...(Associated) == 0) {
279  // single argument to process
280  homogeneous_apply_refs([&groupingTable](auto& x) {
281  PartitionManager<std::decay_t<decltype(x)>>::bindExternalIndices(x, &groupingTable);
282  GroupedCombinationManager<std::decay_t<decltype(x)>>::setGroupedCombination(x, groupingTable);
283  return true;
284  },
285  task);
286  if constexpr (soa::is_soa_iterator_v<G>) {
287  for (auto& element : groupingTable) {
288  std::invoke(processingFunction, task, *element);
289  }
290  } else {
291  static_assert(soa::is_soa_table_like_v<G> || is_enumeration_v<G>,
292  "Single argument of process() should be a table-like or an iterator");
293  std::invoke(processingFunction, task, groupingTable);
294  }
295  } else {
296  // multiple arguments to process
297  static_assert(((soa::is_soa_iterator_v<std::decay_t<Associated>> == false) && ...),
298  "Associated arguments of process() should not be iterators");
299  auto associatedTables = AnalysisDataProcessorBuilder::bindAssociatedTables(inputs, processingFunction, infos);
300  // pre-bind self indices
301  std::apply(
302  [&task](auto&... t) mutable {
303  (homogeneous_apply_refs(
304  [&t](auto& p) {
305  PartitionManager<std::decay_t<decltype(p)>>::bindInternalIndices(p, &t);
306  return true;
307  },
308  task),
309  ...);
310  },
311  associatedTables);
312 
313  auto binder = [&task, &groupingTable, &associatedTables](auto& x) mutable {
314  x.bindExternalIndices(&groupingTable, &std::get<std::decay_t<Associated>>(associatedTables)...);
315  homogeneous_apply_refs([&x](auto& t) mutable {
316  PartitionManager<std::decay_t<decltype(t)>>::setPartition(t, x);
317  PartitionManager<std::decay_t<decltype(t)>>::bindExternalIndices(t, &x);
318  return true;
319  },
320  task);
321  };
322  groupingTable.bindExternalIndices(&std::get<std::decay_t<Associated>>(associatedTables)...);
323 
324  // always pre-bind full tables to support index hierarchy
325  std::apply(
326  [&binder](auto&... x) mutable {
327  (binder(x), ...);
328  },
329  associatedTables);
330 
331  // GroupedCombinations bound separately, as they should be set once for all associated tables
332  homogeneous_apply_refs([&groupingTable, &associatedTables](auto& t) {
333  GroupedCombinationManager<std::decay_t<decltype(t)>>::setGroupedCombination(t, groupingTable, associatedTables);
334  return true;
335  },
336  task);
337  overwriteInternalIndices(associatedTables, associatedTables);
338  if constexpr (soa::is_soa_iterator_v<std::decay_t<G>>) {
339  auto slicer = GroupSlicer(groupingTable, associatedTables, slices);
340  for (auto& slice : slicer) {
341  auto associatedSlices = slice.associatedTables();
342  overwriteInternalIndices(associatedSlices, associatedTables);
343  std::apply(
344  [&binder](auto&... x) mutable {
345  (binder(x), ...);
346  },
347  associatedSlices);
348 
349  // bind partitions and grouping table
350  homogeneous_apply_refs([&groupingTable](auto& x) {
351  PartitionManager<std::decay_t<decltype(x)>>::bindExternalIndices(x, &groupingTable);
352  return true;
353  },
354  task);
355 
356  invokeProcessWithArgs(task, processingFunction, slice.groupingElement(), associatedSlices);
357  }
358  } else {
359  // bind partitions and grouping table
360  homogeneous_apply_refs([&groupingTable](auto& x) {
361  PartitionManager<std::decay_t<decltype(x)>>::bindExternalIndices(x, &groupingTable);
362  return true;
363  },
364  task);
365 
366  invokeProcessWithArgs(task, processingFunction, groupingTable, associatedTables);
367  }
368  }
369  }

467 DataProcessorSpec adaptAnalysisTask(ConfigContext const& ctx, Args&&... args)
468 {

471  auto [name_str, task] = getTaskNameSetProcesses<T>(args...);
472 
473  auto suffix = ctx.options().get<std::string>("workflow-suffix");
474  if (!suffix.empty()) {
475  name_str += suffix;
476  }
477  const char* name = name_str.c_str();
478 
479  auto hash = runtime_hash(name);

481  std::vector<OutputSpec> outputs;
482  std::vector<InputSpec> inputs;
483  std::vector<ConfigParamSpec> options;
484  std::vector<ExpressionInfo> expressionInfos;
485  std::vector<StringPair> bindingsKeys;
486  std::vector<StringPair> bindingsKeysUnsorted;
487 
489  homogeneous_apply_refs([&options, &hash](auto& x) { return OptionManager<std::decay_t<decltype(x)>>::appendOption(options, x); }, *task.get());
491  homogeneous_apply_refs([&inputs](auto& x) { return ConditionManager<std::decay_t<decltype(x)>>::appendCondition(inputs, x); }, *task.get());
492 
494  if constexpr (requires { AnalysisDataProcessorBuilder::inputsFromArgs(&T::process, "default", true, inputs, expressionInfos, bindingsKeys, bindingsKeysUnsorted); }) {
495  AnalysisDataProcessorBuilder::inputsFromArgs(&T::process, "default", true, inputs, expressionInfos, bindingsKeys, bindingsKeysUnsorted);
496  }
497  homogeneous_apply_refs(
498  [name = name_str, &expressionInfos, &inputs, &bindingsKeys, &bindingsKeysUnsorted](auto& x) {
499  using D = std::decay_t<decltype(x)>;
500  if constexpr (is_base_of_template_v<ProcessConfigurable, D>) {
501  // this pushes (argumentIndex,processHash,schemaPtr,nullptr) into expressionInfos for arguments that are Filtered/filtered_iterators
502  AnalysisDataProcessorBuilder::inputsFromArgs(x.process, (name + "/" + x.name).c_str(), x.value, inputs, expressionInfos, bindingsKeys, bindingsKeysUnsorted);
503  return true;
504  }
505  return false;
506  },
507  *task.get());
508 
509  // add preslice declarations to slicing cache definition
510  homogeneous_apply_refs([&bindingsKeys, &bindingsKeysUnsorted](auto& x) { return PresliceManager<std::decay_t<decltype(x)>>::registerCache(x, bindingsKeys, bindingsKeysUnsorted); }, *task.get());
511 
512  // request base tables for spawnable extended tables
513  // this checks for duplications
514  homogeneous_apply_refs([&inputs](auto& x) {
515  return SpawnManager<std::decay_t<decltype(x)>>::requestInputs(inputs, x);
516  },
517  *task.get());
518 
519  // request base tables for indices to be built
520  homogeneous_apply_refs([&inputs](auto& x) {
521  return IndexManager<std::decay_t<decltype(x)>>::requestInputs(inputs, x);
522  },
523  *task.get());
524 
525  // no static way to check if the task defines any processing, we can only make sure it subscribes to at least something
526  if (inputs.empty() == true) {
527  LOG(warn) << "Task " << name_str << " has no inputs";
528  }
529 
530  homogeneous_apply_refs([&outputs, &hash](auto& x) { return OutputManager<std::decay_t<decltype(x)>>::appendOutput(outputs, x, hash); }, *task.get());
531 
532  auto requiredServices = CommonServices::defaultServices();
533  auto arrowServices = CommonServices::arrowServices();
534  requiredServices.insert(requiredServices.end(), arrowServices.begin(), arrowServices.end());
535  homogeneous_apply_refs([&requiredServices](auto& x) { return ServiceManager<std::decay_t<decltype(x)>>::add(requiredServices, x); }, *task.get());
