186 GPUdi() CONSTEXPR T GPUCommonMath::nextMultipleOf(T val)
187 {
188  if CONSTEXPR (I & (I - 1)) {
189  T tmp = val % I;
190  if (tmp) {
191  val += I - tmp;
192  }
193  return val;
194  } else {
195  return (val + I - 1) & ~(T)(I - 1);
196  }
197  return 0; // BUG: Cuda complains about missing return value with constexpr if

379 GPUdi() float GPUCommonMath::InvSqrt(float _x)
380 {
381 #ifdef GPUCA_NO_FAST_MATH
382  return 1.f / Sqrt(_x);
383 #elif defined(__CUDACC__) || defined(__HIPCC__)
384  return __frsqrt_rn(_x);
385 #elif defined(__FAST_MATH__)
386  return 1.f / sqrtf(_x);
387 #else
388  union {
389  float f;
390  int i;
391  } x = {_x};
392  const float xhalf = 0.5f * x.f;
393  x.i = 0x5f3759df - (x.i >> 1);
394  x.f = x.f * (1.5f - xhalf * x.f * x.f);
395  return x.f;
396 #endif
397 }

521  val = 0.f;
522  }
523  if (val >= 0) {
