119 struct ClResTPCCont {
120  // contributor to TPC Cluster
121  std::array<float, 3> xyz{};
122  std::array<float, 3> below{};
123  std::array<float, 3> above{};
124  float snp = 0.;
125  float tgl = 0.;
126  float q2pt = 0.;
127  bool corrAttach = false;
128 
129  int getNExt() const { return (below[0] > 1.) + (above[0] > 1.); }
130 
131  float getClX() const { return xyz[0]; }
132  float getClY() const { return xyz[1]; }
133  float getClZ() const { return xyz[2]; }
134 
135  float getDY() const { return xyz[1] - getYRef(); }
136  float getDZ() const { return xyz[2] - getZRef(); }
137 
138  float getYRef() const
139  {
140  float y = 0;
141  int n = 0;
142  if (below[0] > 1.) {
143  y += below[1];
144  n++;
145  }
146  if (above[0] > 1.) {
147  y += above[1];
148  n++;
149  }
150  return n == 1 ? y : 0.5 * y;
151  }
152 
153  float getZRef() const
154  {
155  float z = 0;
156  int n = 0;
157  if (below[0] > 1.) {
158  z += below[2];
159  n++;
160  }
161  if (above[0] > 1.) {
162  z += above[2];
163  n++;
164  }
165  return n == 1 ? z : 0.5 * z;
166  }
167 
168  float getDXMin() const
169  {
170  float adxA = 1e9, adxB = 1e9;
171  if (above[0] > 1.) {
172  adxA = xyz[0] - above[0];
173  }
174  if (below[0] > 1.) {
175  adxB = xyz[1] - below[0];
176  }
177  return std::abs(adxA) < std::abs(adxB) ? adxA : adxB;
178  }
179 
180  float getDXMax() const
181  {
182  float adxA = 0, adxB = 0;
183  if (above[0] > 1.) {
184  adxA = xyz[0] - above[0];
185  }
186  if (below[0] > 1.) {
187  adxB = xyz[0] - below[0];
188  }
189  return std::abs(adxA) > std::abs(adxB) ? adxA : adxB;
190  }
191 
192  float getEY() const { return getNExt() > 1 ? below[1] - above[1] : -999; }
193  float getEZ() const { return getNExt() > 1 ? below[2] - above[2] : -999; }
194 
195  ClassDefNV(ClResTPCCont, 1);
196 };
