114 void AODProducerWorkflowDPL::createCTPReadout(const o2::globaltracking::RecoContainer& recoData, std::vector<o2::ctp::CTPDigit>& ctpDigits, ProcessingContext& pc)
115 {
116  // Extraxt CTP Config from CCDB
117  const auto ctpcfg = pc.inputs().get<o2::ctp::CTPConfiguration*>("ctpconfig");
118  ctpcfg->printStream(std::cout);
119  // o2::ctp::CTPConfiguration ctpcfg = o2::ctp::CTPRunManager::getConfigFromCCDB(-1, std::to_string(runNumber)); // how to get run
120  // Extract inputs from recoData
121  uint64_t classMaskEMCAL = 0, classMaskTRD = 0, classMaskPHOSCPV = 0;
122  for (const auto& trgclass : ctpcfg->getCTPClasses()) {
123  if (trgclass.cluster->getClusterDetNames().find("EMC") != std::string::npos) {
124  classMaskEMCAL = trgclass.classMask;
125  }
126  if (trgclass.cluster->getClusterDetNames().find("PHS") != std::string::npos) {
127  classMaskPHOSCPV = trgclass.classMask;
128  }
129  if (trgclass.cluster->getClusterDetNames().find("TRD") != std::string::npos) {
130  classMaskTRD = trgclass.classMask;
131  }
132  }
133  LOG(info) << "createCTPReadout: Class Mask EMCAL -> " << classMaskEMCAL;
134  LOG(info) << "createCTPReadout: Class Mask PHOS/CPV -> " << classMaskPHOSCPV;
135  LOG(info) << "createCTPReadout: Class Mask TRD -> " << classMaskTRD;
136 
137  // const auto& fddRecPoints = recoData.getFDDRecPoints();
138  // const auto& fv0RecPoints = recoData.getFV0RecPoints();
139  const auto& triggerrecordEMCAL = recoData.getEMCALTriggers();
140  const auto& triggerrecordPHOSCPV = recoData.getPHOSTriggers();
141  const auto& triggerrecordTRD = recoData.getTRDTriggerRecords();
142  // For EMCAL filter remove calibration triggers
143  std::vector<o2::emcal::TriggerRecord> triggerRecordEMCALPhys;
144  for (const auto& trg : triggerrecordEMCAL) {
145  if (trg.getTriggerBits() & o2::trigger::Cal) {
146  continue;
147  }
148  triggerRecordEMCALPhys.push_back(trg);
149  }
150  // const auto& triggerrecordTRD =recoData.getITSTPCTRDTriggers()
151  //
152 
153  // Find TVX triggers, only TRD/EMCAL/PHOS/CPV triggers in coincidence will be accepted
154  std::set<uint64_t> bcsMapT0triggers;
155  const auto& ft0RecPoints = recoData.getFT0RecPoints();
156  for (auto& ft0RecPoint : ft0RecPoints) {
157  auto t0triggers = ft0RecPoint.getTrigger();
158  if (t0triggers.getVertex()) {
159  uint64_t globalBC = ft0RecPoint.getInteractionRecord().toLong();
160  bcsMapT0triggers.insert(globalBC);
161  }
162  }
163 
164  auto genericCTPDigitizer = [&bcsMapT0triggers, &ctpDigits](auto triggerrecords, uint64_t classmask) -> int {
165  // Strategy:
166  // find detector trigger based on trigger record from readout and add CTPDigit if trigger there
167  int cntwarnings = 0;
168  uint32_t orbitPrev = 0;
169  uint16_t bcPrev = 0;
170  for (auto& trigger : triggerrecords) {
171  auto orbitPrevT = orbitPrev;
172  auto bcPrevT = bcPrev;
173  bcPrev = trigger.getBCData().bc;
174  orbitPrev = trigger.getBCData().orbit;
175  // dedicated for TRD: remove bogus triggers
176  if (orbitPrev < orbitPrevT || bcPrev >= o2::constants::lhc::LHCMaxBunches || (orbitPrev == orbitPrevT && bcPrev < bcPrevT)) {
177  cntwarnings++;
178  // LOGP(warning, "Bogus TRD trigger at bc:{}/orbit:{} (previous was {}/{}), with {} tracklets and {} digits",bcPrev, orbitPrev, bcPrevT, orbitPrevT, trig.getNumberOfTracklets(), trig.getNumberOfDigits());
179  } else {
180  uint64_t globalBC = trigger.getBCData().toLong();
181  auto t0entry = bcsMapT0triggers.find(globalBC);
182  if (t0entry != bcsMapT0triggers.end()) {
183  auto ctpdig = std::find_if(ctpDigits.begin(), ctpDigits.end(), [globalBC](const o2::ctp::CTPDigit& dig) { return static_cast<uint64_t>(dig.intRecord.toLong()) == globalBC; });
184  if (ctpdig != ctpDigits.end()) {
185  // CTP digit existing from other trigger, merge detector class mask
186  ctpdig->CTPClassMask |= std::bitset<64>(classmask);
187  LOG(debug) << "createCTPReadout: Merging " << classmask << " CTP digits with existing digit, CTP mask " << ctpdig->CTPClassMask;
188  } else {
189  // New CTP digit needed
190  LOG(debug) << "createCTPReadout: New CTP digit needed for class " << classmask << std::endl;
191  auto& ctpdigNew = ctpDigits.emplace_back();
192  ctpdigNew.intRecord.setFromLong(globalBC);
193  ctpdigNew.CTPClassMask = classmask;
194  }
195  } else {
196  LOG(warning) << "createCTPReadout: Found " << classmask << " and no MTVX:" << globalBC;
197  }
198  }
199  }
200  return cntwarnings;
201  };
202 
203  auto warningsTRD = genericCTPDigitizer(triggerrecordTRD, classMaskTRD);
204  auto warningsEMCAL = genericCTPDigitizer(triggerRecordEMCALPhys, classMaskEMCAL);
205  auto warningsPHOSCPV = genericCTPDigitizer(triggerrecordPHOSCPV, classMaskPHOSCPV);
206 
207  LOG(info) << "createCTPReadout:# of TRD bogus triggers:" << warningsTRD;
208  LOG(info) << "createCTPReadout:# of EMCAL bogus triggers:" << warningsEMCAL;
209  LOG(info) << "createCTPReadout:# of PHOS/CPV bogus triggers:" << warningsPHOSCPV;
210 }

212 void AODProducerWorkflowDPL::collectBCs(const o2::globaltracking::RecoContainer& data,
213  const std::vector<o2::InteractionTimeRecord>& mcRecords,
214  std::map<uint64_t, int>& bcsMap)
215 {
216  const auto& primVertices = data.getPrimaryVertices();
217  const auto& fddRecPoints = data.getFDDRecPoints();
218  const auto& ft0RecPoints = data.getFT0RecPoints();
219  const auto& fv0RecPoints = data.getFV0RecPoints();
220  const auto& caloEMCCellsTRGR = data.getEMCALTriggers();
221  const auto& caloPHOSCellsTRGR = data.getPHOSTriggers();
222  const auto& cpvTRGR = data.getCPVTriggers();
223  const auto& ctpDigits = data.getCTPDigits();
224  const auto& zdcBCRecData = data.getZDCBCRecData();
225 
226  bcsMap[mStartIR.toLong()] = 1; // store the start of TF
227 
228  // collecting non-empty BCs and enumerating them
229  for (auto& rec : mcRecords) {
230  uint64_t globalBC = rec.toLong();
231  bcsMap[globalBC] = 1;
232  }
233 
234  for (auto& fddRecPoint : fddRecPoints) {
235  uint64_t globalBC = fddRecPoint.getInteractionRecord().toLong();
236  bcsMap[globalBC] = 1;
237  }
238 
239  for (auto& ft0RecPoint : ft0RecPoints) {
240  uint64_t globalBC = ft0RecPoint.getInteractionRecord().toLong();
241  bcsMap[globalBC] = 1;
242  }
243 
244  for (auto& fv0RecPoint : fv0RecPoints) {
245  uint64_t globalBC = fv0RecPoint.getInteractionRecord().toLong();
246  bcsMap[globalBC] = 1;
247  }
248 
249  for (auto& zdcRecData : zdcBCRecData) {
250  uint64_t globalBC = zdcRecData.ir.toLong();
251  bcsMap[globalBC] = 1;
252  }
253 
254  for (auto& vertex : primVertices) {
255  auto& timeStamp = vertex.getTimeStamp();
256  double tsTimeStamp = timeStamp.getTimeStamp() * 1E3; // mus to ns
257  uint64_t globalBC = relativeTime_to_GlobalBC(tsTimeStamp);
258  bcsMap[globalBC] = 1;
259  }
260 
261  for (auto& emcaltrg : caloEMCCellsTRGR) {
262  uint64_t globalBC = emcaltrg.getBCData().toLong();
263  bcsMap[globalBC] = 1;
264  }
265 
266  for (auto& phostrg : caloPHOSCellsTRGR) {
267  uint64_t globalBC = phostrg.getBCData().toLong();
268  bcsMap[globalBC] = 1;
269  }
270 
271  for (auto& cpvtrg : cpvTRGR) {
272  uint64_t globalBC = cpvtrg.getBCData().toLong();
273  bcsMap[globalBC] = 1;
274  }
275 
276  for (auto& ctpDigit : ctpDigits) {
277  uint64_t globalBC = ctpDigit.intRecord.toLong();
278  bcsMap[globalBC] = 1;
279  }
280 
281  int bcID = 0;
282  for (auto& item : bcsMap) {
283  item.second = bcID;
284  bcID++;
285  }
286 }

378 template <typename mftTracksCursorType, typename AmbigMFTTracksCursorType>
379 void AODProducerWorkflowDPL::addToMFTTracksTable(mftTracksCursorType& mftTracksCursor, AmbigMFTTracksCursorType& ambigMFTTracksCursor,
380  GIndex trackID, const o2::globaltracking::RecoContainer& data, int collisionID,
381  std::uint64_t collisionBC, const std::map<uint64_t, int>& bcsMap)
382 {
383  // mft tracks
384  int bcSlice[2] = {-1, -1};
385  const auto& track = data.getMFTTrack(trackID);
386  const auto& rof = data.getMFTTracksROFRecords()[mMFTROFs[trackID.getIndex()]];
387  float trackTime = rof.getBCData().differenceInBC(mStartIR) * o2::constants::lhc::LHCBunchSpacingNS + mMFTROFrameHalfLengthNS + mMFTROFBiasNS;
388  float trackTimeRes = mMFTROFrameHalfLengthNS;
389  bool needBCSlice = collisionID < 0;
390  std::uint64_t bcOfTimeRef;
391  if (needBCSlice) {
392  double error = mTimeMarginTrackTime + trackTimeRes;
393  bcOfTimeRef = fillBCSlice(bcSlice, trackTime - error, trackTime + error, bcsMap);
394  } else {
395  bcOfTimeRef = collisionBC - mStartIR.toLong(); // by default (unambiguous) track time is wrt collision BC
396  }
397  trackTime -= bcOfTimeRef * o2::constants::lhc::LHCBunchSpacingNS;
398 
399  // the Cellular Automaton track-finding algorithm flag is stored in first of the 4 bits not used for the cluster size
400  uint64_t mftClusterSizesAndTrackFlags = track.getClusterSizes();
401  mftClusterSizesAndTrackFlags |= (track.isCA()) ? (1ULL << (60)) : 0;
402 
403  mftTracksCursor(collisionID,
404  track.getX(),
405  track.getY(),
406  truncateFloatFraction(track.getZ(), mTrackX), // for the forward tracks Z has the same role as X in barrel
407  truncateFloatFraction(track.getPhi(), mTrackAlpha),
408  truncateFloatFraction(track.getTanl(), mTrackTgl),
409  truncateFloatFraction(track.getInvQPt(), mTrack1Pt),
410  mftClusterSizesAndTrackFlags,
411  truncateFloatFraction(track.getTrackChi2(), mTrackChi2),
412  truncateFloatFraction(trackTime, mTrackTime),
413  truncateFloatFraction(trackTimeRes, mTrackTimeError));
414  if (needBCSlice) {
415  ambigMFTTracksCursor(mTableTrMFTID, bcSlice);
416  }
417 }

419 template <typename TracksCursorType, typename TracksCovCursorType, typename TracksExtraCursorType, typename TracksQACursorType, typename AmbigTracksCursorType,
420  typename MFTTracksCursorType, typename AmbigMFTTracksCursorType,
421  typename FwdTracksCursorType, typename FwdTracksCovCursorType, typename AmbigFwdTracksCursorType, typename FwdTrkClsCursorType>
422 void AODProducerWorkflowDPL::fillTrackTablesPerCollision(int collisionID,
423  std::uint64_t collisionBC,
424  const o2::dataformats::VtxTrackRef& trackRef,
425  const gsl::span<const GIndex>& GIndices,
426  const o2::globaltracking::RecoContainer& data,
427  TracksCursorType& tracksCursor,
428  TracksCovCursorType& tracksCovCursor,
429  TracksExtraCursorType& tracksExtraCursor,
430  TracksQACursorType& tracksQACursor,
431  AmbigTracksCursorType& ambigTracksCursor,
432  MFTTracksCursorType& mftTracksCursor,
433  AmbigMFTTracksCursorType& ambigMFTTracksCursor,
434  FwdTracksCursorType& fwdTracksCursor,
435  FwdTracksCovCursorType& fwdTracksCovCursor,
436  AmbigFwdTracksCursorType& ambigFwdTracksCursor,
437  FwdTrkClsCursorType& fwdTrkClsCursor,
438  const std::map<uint64_t, int>& bcsMap)
439 {
440  for (int src = GIndex::NSources; src--;) {
441  if (!GIndex::isTrackSource(src)) {
442  continue;
443  }
444  int start = trackRef.getFirstEntryOfSource(src);
445  int end = start + trackRef.getEntriesOfSource(src);
446  int nToReserve = end - start; // + last index for a given table
447  if (src == GIndex::Source::MFT) {
448  mftTracksCursor.reserve(nToReserve + mftTracksCursor.lastIndex());
449  } else if (src == GIndex::Source::MCH || src == GIndex::Source::MFTMCH || src == GIndex::Source::MCHMID) {
450  fwdTracksCursor.reserve(nToReserve + fwdTracksCursor.lastIndex());
451  fwdTracksCovCursor.reserve(nToReserve + fwdTracksCovCursor.lastIndex());
452  } else {
453  tracksCursor.reserve(nToReserve + tracksCursor.lastIndex());
454  tracksCovCursor.reserve(nToReserve + tracksCovCursor.lastIndex());
455  tracksExtraCursor.reserve(nToReserve + tracksExtraCursor.lastIndex());
456  }
457  for (int ti = start; ti < end; ti++) {
458  const auto& trackIndex = GIndices[ti];
459  if (GIndex::includesSource(src, mInputSources)) {
460  if (src == GIndex::Source::MFT) { // MFT tracks are treated separately since they are stored in a different table
461  if (trackIndex.isAmbiguous() && mGIDToTableMFTID.find(trackIndex) != mGIDToTableMFTID.end()) { // was it already stored ?
462  continue;
463  }
464  addToMFTTracksTable(mftTracksCursor, ambigMFTTracksCursor, trackIndex, data, collisionID, collisionBC, bcsMap);
465  mGIDToTableMFTID.emplace(trackIndex, mTableTrMFTID);
466  mTableTrMFTID++;

550  addToTracksExtraTable(tracksExtraCursor, extraInfo);
551  mStrTrkIndices[collStrTrk.second] = mTableTrID;
552  mTableTrID++;
553  }

592 template <typename FwdTracksCursorType, typename FwdTracksCovCursorType, typename AmbigFwdTracksCursorType>
593 void AODProducerWorkflowDPL::addToFwdTracksTable(FwdTracksCursorType& fwdTracksCursor, FwdTracksCovCursorType& fwdTracksCovCursor,
594  AmbigFwdTracksCursorType& ambigFwdTracksCursor, GIndex trackID,
595  const o2::globaltracking::RecoContainer& data, int collisionID, std::uint64_t collisionBC,
596  const std::map<uint64_t, int>& bcsMap)
597 {
598  const auto& mchTracks = data.getMCHTracks();
599  const auto& midTracks = data.getMIDTracks();
600  const auto& mchmidMatches = data.getMCHMIDMatches();
601  const auto& mchClusters = data.getMCHTrackClusters();
602 
603  FwdTrackInfo fwdInfo;
604  FwdTrackCovInfo fwdCovInfo;
605  int bcSlice[2] = {-1, -1};
606 
607  // helper lambda for mch bitmap -- common for global and standalone tracks
608  auto getMCHBitMap = [&](int mchTrackID) {
609  if (mchTrackID != -1) { // check matching just in case
610  const auto& mchTrack = mchTracks[mchTrackID];
611  int first = mchTrack.getFirstClusterIdx();
612  int last = mchTrack.getLastClusterIdx();
613  for (int i = first; i <= last; i++) { // check chamberIds of all clusters
614  const auto& cluster = mchClusters[i];
615  int chamberId = cluster.getChamberId();
616  fwdInfo.mchBitMap |= 1 << chamberId;
617  }
618  }
619  };
620 
621  auto getMIDBitMapBoards = [&](int midTrackID) {
622  if (midTrackID != -1) { // check matching just in case
623  const auto& midTrack = midTracks[midTrackID];
624  fwdInfo.midBitMap = midTrack.getHitMap();
625  fwdInfo.midBoards = midTrack.getEfficiencyWord();
626  }
627  };
628 
629  auto extrapMCHTrack = [&](int mchTrackID) {
630  const auto& track = mchTracks[mchTrackID];
631 
632  // mch standalone tracks extrapolated to vertex
633  // compute 3 sets of tracks parameters :
634  // - at vertex
635  // - at DCA
636  // - at the end of the absorber
637  // extrapolate to vertex
638  float vx = 0, vy = 0, vz = 0;
639  if (collisionID >= 0) {
640  const auto& v = data.getPrimaryVertex(collisionID);
641  vx = v.getX();
642  vy = v.getY();
643  vz = v.getZ();
644  }
645 
646  o2::mch::TrackParam trackParamAtVertex(track.getZ(), track.getParameters(), track.getCovariances());
647  if (mPropMuons) {
648  double errVtx{0.0}; // FIXME: get errors associated with vertex if available
649  double errVty{0.0};
650  if (!o2::mch::TrackExtrap::extrapToVertex(trackParamAtVertex, vx, vy, vz, errVtx, errVty)) {
651  return false;
652  }
653  }
654 
655  // extrapolate to DCA
656  o2::mch::TrackParam trackParamAtDCA(track.getZ(), track.getParameters());
657  if (!o2::mch::TrackExtrap::extrapToVertexWithoutBranson(trackParamAtDCA, vz)) {
658  return false;
659  }
660 
661  // extrapolate to the end of the absorber
662  o2::mch::TrackParam trackParamAtRAbs(track.getZ(), track.getParameters());
663  if (!o2::mch::TrackExtrap::extrapToZ(trackParamAtRAbs, -505.)) { // FIXME: replace hardcoded 505
664  return false;
665  }
666 
667  double dcaX = trackParamAtDCA.getNonBendingCoor() - vx;
668  double dcaY = trackParamAtDCA.getBendingCoor() - vy;
669  double dca = std::sqrt(dcaX * dcaX + dcaY * dcaY);
670 
671  double xAbs = trackParamAtRAbs.getNonBendingCoor();
672  double yAbs = trackParamAtRAbs.getBendingCoor();
673 
674  double dpdca = track.getP() * dca;
675  double dchi2 = track.getChi2OverNDF();
676 
677  auto fwdmuon = mMatching.MCHtoFwd(trackParamAtVertex);
678 
679  fwdInfo.x = fwdmuon.getX();
680  fwdInfo.y = fwdmuon.getY();
681  fwdInfo.z = fwdmuon.getZ();
682  fwdInfo.phi = fwdmuon.getPhi();
683  fwdInfo.tanl = fwdmuon.getTgl();
684  fwdInfo.invqpt = fwdmuon.getInvQPt();

686  fwdInfo.chi2 = dchi2;
687  fwdInfo.pdca = dpdca;
688  fwdInfo.nClusters = track.getNClusters();
689 
690  fwdCovInfo.sigX = TMath::Sqrt(fwdmuon.getCovariances()(0, 0));
691  fwdCovInfo.sigY = TMath::Sqrt(fwdmuon.getCovariances()(1, 1));
692  fwdCovInfo.sigPhi = TMath::Sqrt(fwdmuon.getCovariances()(2, 2));
693  fwdCovInfo.sigTgl = TMath::Sqrt(fwdmuon.getCovariances()(3, 3));
694  fwdCovInfo.sig1Pt = TMath::Sqrt(fwdmuon.getCovariances()(4, 4));
695  fwdCovInfo.rhoXY = (Char_t)(128. * fwdmuon.getCovariances()(0, 1) / (fwdCovInfo.sigX * fwdCovInfo.sigY));
696  fwdCovInfo.rhoPhiX = (Char_t)(128. * fwdmuon.getCovariances()(0, 2) / (fwdCovInfo.sigPhi * fwdCovInfo.sigX));
697  fwdCovInfo.rhoPhiY = (Char_t)(128. * fwdmuon.getCovariances()(1, 2) / (fwdCovInfo.sigPhi * fwdCovInfo.sigY));
698  fwdCovInfo.rhoTglX = (Char_t)(128. * fwdmuon.getCovariances()(0, 3) / (fwdCovInfo.sigTgl * fwdCovInfo.sigX));
699  fwdCovInfo.rhoTglY = (Char_t)(128. * fwdmuon.getCovariances()(1, 3) / (fwdCovInfo.sigTgl * fwdCovInfo.sigY));
700  fwdCovInfo.rhoTglPhi = (Char_t)(128. * fwdmuon.getCovariances()(2, 3) / (fwdCovInfo.sigTgl * fwdCovInfo.sigPhi));
701  fwdCovInfo.rho1PtX = (Char_t)(128. * fwdmuon.getCovariances()(0, 4) / (fwdCovInfo.sig1Pt * fwdCovInfo.sigX));
702  fwdCovInfo.rho1PtY = (Char_t)(128. * fwdmuon.getCovariances()(1, 4) / (fwdCovInfo.sig1Pt * fwdCovInfo.sigY));
703  fwdCovInfo.rho1PtPhi = (Char_t)(128. * fwdmuon.getCovariances()(2, 4) / (fwdCovInfo.sig1Pt * fwdCovInfo.sigPhi));
704  fwdCovInfo.rho1PtTgl = (Char_t)(128. * fwdmuon.getCovariances()(3, 4) / (fwdCovInfo.sig1Pt * fwdCovInfo.sigTgl));
705 
706  return true;
707  };
708

710  int mchTrackID = trackID.getIndex();

714  }

721  fwdInfo.trackTypeId = o2::aod::fwdtrack::MuonStandaloneTrack;

723  auto mchTrackID = mchmidMatch.getMCHRef().getIndex();
724  if (!extrapMCHTrack(mchTrackID)) {
725  LOGF(warn, "Unable to extrapolate MCH track with ID %d! Dummy parameters will be used", mchTrackID);
726  }

740  fwdInfo.y = track.getY();

742  fwdInfo.phi = track.getPhi();
743  fwdInfo.tanl = track.getTanl();

751  fwdInfo.matchmchtrackid = mIndexTableFwd[track.getMCHTrackID()];
752  fwdInfo.trackTime = track.getTimeMUS().getTimeStamp() * 1.e3;

754 
755  getMCHBitMap(track.getMCHTrackID());
756  getMIDBitMapBoards(track.getMIDTrackID());
757

773

775  }

784  }

786

2586 {

2588  constexpr float ETAEMCAL = 0.75; // eta of EMCAL/DCAL with margin
2589  constexpr float ZEMCALFastCheck = 460.; // Max Z (with margin to check with straightline extrapolarion)
2590  constexpr float ETADCALINNER = 0.22; // eta of the DCAL PHOS Hole (at XEMCAL)
2591  constexpr float ETAPHOS = 0.13653194; // nominal eta of the PHOS acceptance (at XPHOS): -log(tan((TMath::Pi()/2 - atan2(63, 460))/2))
2592  constexpr float ETAPHOSMARGIN = 0.17946979; // etat of the PHOS acceptance with 20 cm margin (at XPHOS): -log(tan((TMath::Pi()/2 + atan2(63+20., 460))/2)), not used, for the ref only
2593  constexpr float ETADCALPHOSSWITCH = (ETADCALINNER + ETAPHOS) / 2; // switch to DCAL to PHOS check if eta < this value
2594  constexpr short SNONE = 0, SEMCAL = 0x1, SPHOS = 0x2;
2595  constexpr short SECTORTYPE[18] = {
2596  SNONE, SNONE, SNONE, SNONE, // 0:3
2597  SEMCAL, SEMCAL, SEMCAL, SEMCAL, SEMCAL, SEMCAL, // 3:9 EMCAL only
2598  SNONE, SNONE, // 10:11
2599  SPHOS, // 12 PHOS only
2600  SPHOS | SEMCAL, SPHOS | SEMCAL, SPHOS | SEMCAL, // 13:15 PHOS & DCAL
2601  SEMCAL, // 16 DCAL only
2602  SNONE // 17
2603  };
2604

2606  auto prop = o2::base::Propagator::Instance();
2607  // 1st propagate to EMCAL nominal radius
2608  float xtrg = 0;
2609  // quick check with straight line propagtion
2610  if (!outTr.getXatLabR(XEMCAL, xtrg, prop->getNominalBz(), o2::track::DirType::DirOutward) ||
2611  (std::abs(outTr.getZAt(xtrg, 0)) > ZEMCALFastCheck) ||
2612  !prop->PropagateToXBxByBz(outTr, xtrg, 0.95, 10, o2::base::Propagator::MatCorrType::USEMatCorrLUT)) {
2613  LOGP(debug, "preliminary step: does not reach R={} {}", XEMCAL, outTr.asString());
2614  return;
2615  }
2616  // we do not necessarilly reach wanted radius in a single propagation
2617  if ((outTr.getX() * outTr.getX() + outTr.getY() * outTr.getY() < XEMCAL2) &&
2618  (!outTr.rotateParam(outTr.getPhi()) ||
2619  !outTr.getXatLabR(XEMCAL, xtrg, prop->getNominalBz(), o2::track::DirType::DirOutward) ||
2620  !prop->PropagateToXBxByBz(outTr, xtrg, 0.95, 10, o2::base::Propagator::MatCorrType::USEMatCorrLUT))) {
2621  LOGP(debug, "does not reach R={} {}", XEMCAL, outTr.asString());
2622  return;
2623  }

2626

2628  int ntri = 0;
2629  while (ntri < 2) {
2630  auto outTrTmp = outTr;
2631  float alpha = o2::math_utils::sector2Angle(sector);
2632  if ((std::abs(outTr.getZ()) > ZEMCALFastCheck) || !outTrTmp.rotateParam(alpha) ||
2633  !prop->PropagateToXBxByBz(outTrTmp, xprop, 0.95, 10, o2::base::Propagator::MatCorrType::USEMatCorrLUT)) {
2634  LOGP(debug, "failed on rotation to {} (sector {}) or propagation to X={} {}", alpha, sector, xprop, outTrTmp.asString());
2635  return false;
2636  }
2637  // make sure we are still in the target sector
2638  int sectorTmp = o2::math_utils::angle2Sector(outTrTmp.getPhiPos());
2639  if (sectorTmp == sector) {
2640  outTr = outTrTmp;
2641  break;
2642  }
2643  sector = sectorTmp;
2644  ntri++;
2645  }
2646  if (ntri == 2) {
2647  LOGP(debug, "failed to rotate to sector, {}", outTr.asString());
2648  return false;
2649  }
2650  return true;
2651  };
2652 
2653  // we are at the EMCAL X, check if we are in the good sector
2654  if (!propExactSector(XEMCAL) || SECTORTYPE[sector] == SNONE) { // propagation failed or neither EMCAL not DCAL/PHOS
2655  return;
2656  }
2657

2659  float r = std::sqrt(outTr.getX() * outTr.getX() + outTr.getY() * outTr.getY()), tg = std::atan2(r, outTr.getZ());

2661  if (etaAbs > ETAEMCAL) {
2662  LOGP(debug, "eta = {} is off at EMCAL radius", eta, outTr.asString());
2663  return;
2664  }
2665  // are we in the PHOS hole (with margin)?
2666  if ((SECTORTYPE[sector] & SPHOS) && etaAbs < ETADCALPHOSSWITCH) { // propagate to PHOS radius
2667  if (!propExactSector(XPHOS)) {
2668  return;
2669  }
2670  r = std::sqrt(outTr.getX() * outTr.getX() + outTr.getY() * outTr.getY());
2671  tg = std::atan2(r, outTr.getZ());
2672  eta = -std::log(std::tan(0.5f * tg));
2673  } else if (!(SECTORTYPE[sector] & SEMCAL)) { // are in the sector with PHOS only
2674  return;
2675  }
2676  extraInfoHolder.trackPhiEMCAL = outTr.getPhiPos();

2775 void AODProducerWorkflowDPL::addRefGlobalBCsForTOF(const o2::dataformats::VtxTrackRef& trackRef, const gsl::span<const GIndex>& GIndices,
2776  const o2::globaltracking::RecoContainer& data, std::map<uint64_t, int>& bcsMap)
2777 {
2778  // Orphan tracks need to refer to some globalBC and for tracks with TOF this BC should be whithin an orbit
2779  // from the track abs time (to guarantee time precision). Therefore, we may need to insert some dummy globalBCs
2780  // to guarantee proper reference.
2781  // complete globalBCs by dummy entries necessary to provide BC references for TOF tracks with requested precision
2782  // to provide a reference for the time of the orphan tracks we should make sure that there are no gaps longer
2783  // than needed to store the time with sufficient precision
2784 
2785  // estimate max distance in BCs between TOF time and eventual reference BC
2786  int nbitsFrac = 24 - (32 - o2::math_utils::popcount(mTrackTime)); // number of bits used to encode the fractional part of float truncated by the mask
2787  int nbitsLoss = std::max(0, int(std::log2(TOFTimePrecPS))); // allowed bit loss guaranteeing needed precision in PS
2788  assert(nbitsFrac > 1);
2789  std::uint64_t maxRangePS = std::uint64_t(0x1) << (nbitsFrac + nbitsLoss);
2790  int maxGapBC = maxRangePS / (o2::constants::lhc::LHCBunchSpacingNS * 1e3); // max gap in BCs allowing to store time with required precision
2791  LOG(info) << "Max gap of " << maxGapBC << " BCs to closest globalBC reference is needed for TOF tracks to provide precision of "
2792  << TOFTimePrecPS << " ps";
2793 
2794  // check if there are tracks orphan tracks at all
2795  if (!trackRef.getEntries()) {
2796  return;
2797  }
2798  // the bscMap has at least TF start BC
2799  std::uint64_t maxBC = mStartIR.toLong();
2800  const auto& tofClus = data.getTOFClusters();
2801  for (int src = GIndex::NSources; src--;) {
2802  if (!GIndex::getSourceDetectorsMask(src)[o2::detectors::DetID::TOF]) { // check only tracks with TOF contribution
2803  continue;
2804  }
2805  int start = trackRef.getFirstEntryOfSource(src);
2806  int end = start + trackRef.getEntriesOfSource(src);
2807  for (int ti = start; ti < end; ti++) {
2808  auto& trackIndex = GIndices[ti];
2809  const auto& tofMatch = data.getTOFMatch(trackIndex);
2810  const auto& tofInt = tofMatch.getLTIntegralOut();
2811  float intLen = tofInt.getL();
2812  float tofExpMom = 0.;
2813  if (tofInt.getTOF(o2::track::PID::Pion) > 0.f) {
2814  float expBeta = (intLen / (tofInt.getTOF(o2::track::PID::Pion) * cSpeed));
2815  if (expBeta > o2::constants::math::Almost1) {
2816  expBeta = o2::constants::math::Almost1;
2817  }
2818  tofExpMom = o2::constants::physics::MassPionCharged * expBeta / std::sqrt(1.f - expBeta * expBeta);
2819  } else {
2820  continue;
2821  }
2822  double massZ = o2::track::PID::getMass2Z(data.getTrackParam(trackIndex).getPID());
2823  double energy = sqrt((massZ * massZ) + (tofExpMom * tofExpMom));
2824  double exp = intLen * energy / (cSpeed * tofExpMom);
2825  auto tofSignal = (tofMatch.getSignal() - exp) * 1e-3; // time in ns wrt TF start
2826  auto bc = relativeTime_to_GlobalBC(tofSignal);
2827 
2828  auto it = bcsMap.lower_bound(bc);
2829  if (it == bcsMap.end() || it->first > bc + maxGapBC) {
2830  bcsMap.emplace_hint(it, bc, 1);
2831  LOG(debug) << "adding dummy BC " << bc;
2832  }
2833  if (bc > maxBC) {
2834  maxBC = bc;
2835  }
2836  }
2837  }
2838  // make sure there is a globalBC exceeding the max encountered bc
2839  if ((--bcsMap.end())->first <= maxBC) {
2840  bcsMap.emplace_hint(bcsMap.end(), maxBC + 1, 1);
2841  }
2842  // renumber BCs
2843  int bcID = 0;
2844  for (auto& item : bcsMap) {
2845  item.second = bcID;
2846  bcID++;
2847  }

2850 std::uint64_t AODProducerWorkflowDPL::fillBCSlice(int (&slice)[2], double tmin, double tmax, const std::map<uint64_t, int>& bcsMap) const
2851 {
2852  // for ambiguous tracks (no or multiple vertices) we store the BC slice corresponding to track time window used for track-vertex matching,
2853  // see VertexTrackMatcher::extractTracks creator method, i.e. central time estimated +- uncertainty defined as:
2854  // 1) for tracks having a gaussian time error: PVertexerParams.nSigmaTimeTrack * trackSigma + PVertexerParams.timeMarginTrackTime
2855  // 2) for tracks having time uncertainty in a fixed time interval (TPC,ITS,MFT..): half of the interval + PVertexerParams.timeMarginTrackTime
2856  // The track time in the TrackExtraInfo is stored in ns wrt the collision BC for unambigous tracks and wrt bcSlice[0] for ambiguous ones,
2857  // with convention for errors: trackSigma in case (1) and half of the time interval for case (2) above.
2858 
2859  // find indices of widest slice of global BCs in the map compatible with provided BC range. bcsMap is guaranteed to be non-empty.
2860  // We also assume that tmax >= tmin.
2861 
2862  uint64_t bcMin = relativeTime_to_GlobalBC(tmin), bcMax = relativeTime_to_GlobalBC(tmax);
2863 
2864  /*
2865  // brute force way of searching bcs via direct binary search in the map
2866  auto lower = bcsMap.lower_bound(bcMin), upper = bcsMap.upper_bound(bcMax);
2867 
2868  if (lower == bcsMap.end()) {
2869  --lower;
2870  }
2871  if (upper != lower) {
2872  --upper;
2873  }
2874  slice[0] = std::distance(bcsMap.begin(), lower);
2875  slice[1] = std::distance(bcsMap.begin(), upper);
2876  */
2877 
2878  // faster way to search in bunch crossing via the accelerated bunch crossing lookup structure
2879  auto p = mBCLookup.lower_bound(bcMin);
2880  // assuming that bcMax will be >= bcMin and close to bcMin; we can find
2881  // the upper bound quickly by lineary iterating from p.first to the point where
2882  // the time becomes larger than bcMax.
2883  // (if this is not the case we could determine it with a similar call to mBCLookup)
2884  auto& bcvector = mBCLookup.getBCTimeVector();
2885  auto upperindex = p.first;
2886  while (upperindex < bcvector.size() && bcvector[upperindex] <= bcMax) {
2887  upperindex++;
2888  }
2889  if (upperindex != p.first) {
2890  upperindex--;
2891  }
2892  slice[0] = p.first;
2893  slice[1] = upperindex;
2894 
2895  auto bcOfTimeRef = p.second - this->mStartIR.toLong();
2896  LOG(debug) << "BC slice t:" << tmin << " " << slice[0]
2897  << " t: " << tmax << " " << slice[1]
2898  << " bcref: " << bcOfTimeRef;
