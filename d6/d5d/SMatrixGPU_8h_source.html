212 template <unsigned int I>
213 struct meta_matrix_dot {
214 
215  template <class MatrixA, class MatrixB>
216  static GPUdi() typename MatrixA::value_type f(const MatrixA& lhs,
217  const MatrixB& rhs,
218  const unsigned int offset)
219  {
220  return lhs.apply(offset / MatrixB::kCols * MatrixA::kCols + I) *
221  rhs.apply(MatrixB::kCols * I + offset % MatrixB::kCols) +
222  meta_matrix_dot<I - 1>::f(lhs, rhs, offset);
223  }
224 
225  template <class MatrixA, class MatrixB>
226  static GPUdi() typename MatrixA::value_type g(const MatrixA& lhs,
227  const MatrixB& rhs,
228  unsigned int i,
229  unsigned int j)
230  {
231  return lhs(i, I) * rhs(I, j) +
232  meta_matrix_dot<I - 1>::g(lhs, rhs, i, j);
233  }
234 };

552  GPUd() static void Evaluate(SMatrixGPU<T, D1, D2, R1>& lhs, const Expr<A, T, D1, D2, R2>& rhs)
553  {
554  if (!rhs.IsInUse(lhs.begin())) {
555  unsigned int l = 0;
556  for (unsigned int i = 0; i < D1; ++i) {
557  for (unsigned int j = 0; j < D2; ++j) {
558  lhs.mRep[l] = rhs(i, j);
559  l++;
560  }
561  }
562  } else {
563  T tmp[D1 * D2];
564  unsigned int l = 0;
565  for (unsigned int i = 0; i < D1; ++i) {
566  for (unsigned int j = 0; j < D2; ++j) {
567  tmp[l] = rhs(i, j);
568  l++;
569  }
570  }
571 
572  for (unsigned int i = 0; i < D1 * D2; ++i) {
573  lhs.mRep[i] = tmp[i];
574  }
575  }
576  }

776 template <unsigned int D, unsigned int N>
777 template <class T>
778 GPUdi() void Inverter<D, N>::InvertBunchKaufman(MatRepSymGPU<T, D>& rhs, int& ifail)
779 {
780  typedef T value_type;
781  int i, j, k, s;
782  int pivrow;
783  const int nrow = MatRepSymGPU<T, D>::kRows;
784 
785  SVectorGPU<T, MatRepSymGPU<T, D>::kRows> xvec;
786  SVectorGPU<int, MatRepSymGPU<T, D>::kRows> pivv;
787 
788  typedef int* pivIter;
789  typedef T* mIter;
790 
791  mIter x = xvec.begin();
792  // x[i] is used as helper storage, needs to have at least size nrow.
793  pivIter piv = pivv.begin();
794  // piv[i] is used to store details of exchanges
795 
796  value_type temp1, temp2;
797  mIter ip, mjj, iq;
798  value_type lambda, sigma;
799  const value_type alpha = .6404; // = (1+sqrt(17))/8
800  // LM (04/2009) remove this useless check (it is not in LAPACK) which fails inversion of
801  // a matrix with values < epsilon in the diagonal
802  //
803  // const double epsilon = 32*std::numeric_limits<T>::epsilon();
804  // whenever a sum of two doubles is below or equal to epsilon
805  // it is set to zero.
806  // this constant could be set to zero but then the algorithm
807  // doesn't neccessarily detect that a matrix is singular
808 
809  for (i = 0; i < nrow; i++) {
810  piv[i] = i + 1;
811  }
812 
813  ifail = 0;
814 
815  // compute the factorization P*A*P^T = L * D * L^T
816  // L is unit lower triangular, D is direct sum of 1x1 and 2x2 matrices
817  // L and D^-1 are stored in A = *this, P is stored in piv[]
818 
819  for (j = 1; j < nrow; j += s) // main loop over columns
820  {
821  mjj = rhs.Array() + j * (j - 1) / 2 + j - 1;
822  lambda = 0; // compute lambda = max of A(j+1:n,j)
823  pivrow = j + 1;
824  ip = rhs.Array() + (j + 1) * j / 2 + j - 1;
825  for (i = j + 1; i <= nrow; ip += i++) {
826  if (o2::gpu::GPUCommonMath::Abs(*ip) > lambda) {
827  lambda = o2::gpu::GPUCommonMath::Abs(*ip);
828  pivrow = i;
829  }
830  }
831 
832  if (lambda == 0) {
833  if (*mjj == 0) {
834  ifail = 1;
835  return;
836  }
837  s = 1;
838  *mjj = 1.0f / *mjj;
839  } else {
840  if (o2::gpu::GPUCommonMath::Abs(*mjj) >= lambda * alpha) {
841  s = 1;
842  pivrow = j;
843  } else {
844  sigma = 0; // compute sigma = max A(pivrow, j:pivrow-1)
845  ip = rhs.Array() + pivrow * (pivrow - 1) / 2 + j - 1;
846  for (k = j; k < pivrow; k++) {
847  if (o2::gpu::GPUCommonMath::Abs(*ip) > sigma) {
848  sigma = o2::gpu::GPUCommonMath::Abs(*ip);
849  }
850  ip++;
851  }
852  // sigma cannot be zero because it is at least lambda which is not zero
853  if (o2::gpu::GPUCommonMath::Abs(*mjj) >= alpha * lambda * (lambda / sigma)) {
854  s = 1;
855  pivrow = j;
856  } else if (o2::gpu::GPUCommonMath::Abs(*(rhs.Array() + pivrow * (pivrow - 1) / 2 + pivrow - 1)) >= alpha * sigma) {
857  s = 1;
858  } else {
859  s = 2;
860  }
861  }
862  if (pivrow == j) // no permutation neccessary
863  {
864  piv[j - 1] = pivrow;
865  if (*mjj == 0) {
866  ifail = 1;
867  return;
868  }
869  temp2 = *mjj = 1.0f / *mjj; // invert D(j,j)
870 
871  // update A(j+1:n, j+1,n)
872  for (i = j + 1; i <= nrow; i++) {
873  temp1 = *(rhs.Array() + i * (i - 1) / 2 + j - 1) * temp2;
874  ip = rhs.Array() + i * (i - 1) / 2 + j;
875  for (k = j + 1; k <= i; k++) {
876  *ip -= static_cast<T>(temp1 * *(rhs.Array() + k * (k - 1) / 2 + j - 1));
877  // if (o2::gpu::GPUCommonMath::Abs(*ip) <= epsilon)
878  // *ip=0;
879  ip++;
880  }
881  }
882  // update L
883  ip = rhs.Array() + (j + 1) * j / 2 + j - 1;
884  for (i = j + 1; i <= nrow; ip += i++) {
885  *ip *= static_cast<T>(temp2);
886  }
887  } else if (s == 1) // 1x1 pivot
888  {
889  piv[j - 1] = pivrow;
890 
891  // interchange rows and columns j and pivrow in
892  // submatrix (j:n,j:n)
893  ip = rhs.Array() + pivrow * (pivrow - 1) / 2 + j;
894  for (i = j + 1; i < pivrow; i++, ip++) {
895  temp1 = *(rhs.Array() + i * (i - 1) / 2 + j - 1);
896  *(rhs.Array() + i * (i - 1) / 2 + j - 1) = *ip;
897  *ip = static_cast<T>(temp1);
898  }
899  temp1 = *mjj;
900  *mjj = *(rhs.Array() + pivrow * (pivrow - 1) / 2 + pivrow - 1);
901  *(rhs.Array() + pivrow * (pivrow - 1) / 2 + pivrow - 1) = static_cast<T>(temp1);
902  ip = rhs.Array() + (pivrow + 1) * pivrow / 2 + j - 1;
903  iq = ip + pivrow - j;
904  for (i = pivrow + 1; i <= nrow; ip += i, iq += i++) {
905  temp1 = *iq;
906  *iq = *ip;
907  *ip = static_cast<T>(temp1);
908  }
909 
910  if (*mjj == 0) {
911  ifail = 1;
912  return;
913  }
914  temp2 = *mjj = 1.0f / *mjj; // invert D(j,j)
915 
916  // update A(j+1:n, j+1:n)
917  for (i = j + 1; i <= nrow; i++) {
918  temp1 = *(rhs.Array() + i * (i - 1) / 2 + j - 1) * temp2;
919  ip = rhs.Array() + i * (i - 1) / 2 + j;
920  for (k = j + 1; k <= i; k++) {
921  *ip -= static_cast<T>(temp1 * *(rhs.Array() + k * (k - 1) / 2 + j - 1));
922  // if (o2::gpu::GPUCommonMath::Abs(*ip) <= epsilon)
923  // *ip=0;
924  ip++;
925  }
926  }
927  // update L
928  ip = rhs.Array() + (j + 1) * j / 2 + j - 1;
929  for (i = j + 1; i <= nrow; ip += i++) {
930  *ip *= static_cast<T>(temp2);
931  }
932  } else // s=2, ie use a 2x2 pivot
933  {
934  piv[j - 1] = -pivrow;
935  piv[j] = 0; // that means this is the second row of a 2x2 pivot
936 
937  if (j + 1 != pivrow) {
938  // interchange rows and columns j+1 and pivrow in
939  // submatrix (j:n,j:n)
940  ip = rhs.Array() + pivrow * (pivrow - 1) / 2 + j + 1;
941  for (i = j + 2; i < pivrow; i++, ip++) {
942  temp1 = *(rhs.Array() + i * (i - 1) / 2 + j);
943  *(rhs.Array() + i * (i - 1) / 2 + j) = *ip;
944  *ip = static_cast<T>(temp1);
945  }
946  temp1 = *(mjj + j + 1);
947  *(mjj + j + 1) =
948  *(rhs.Array() + pivrow * (pivrow - 1) / 2 + pivrow - 1);
949  *(rhs.Array() + pivrow * (pivrow - 1) / 2 + pivrow - 1) = static_cast<T>(temp1);
950  temp1 = *(mjj + j);
951  *(mjj + j) = *(rhs.Array() + pivrow * (pivrow - 1) / 2 + j - 1);
952  *(rhs.Array() + pivrow * (pivrow - 1) / 2 + j - 1) = static_cast<T>(temp1);
953  ip = rhs.Array() + (pivrow + 1) * pivrow / 2 + j;
954  iq = ip + pivrow - (j + 1);
955  for (i = pivrow + 1; i <= nrow; ip += i, iq += i++) {
956  temp1 = *iq;
957  *iq = *ip;
958  *ip = static_cast<T>(temp1);
959  }
960  }
961  // invert D(j:j+1,j:j+1)
962  temp2 = *mjj * *(mjj + j + 1) - *(mjj + j) * *(mjj + j);
963  if (temp2 == 0) {
964  LOGF(error, "SymMatrix::bunch_invert: error in pivot choice");
965  }
966  temp2 = 1. / temp2;
967  // this quotient is guaranteed to exist by the choice
968  // of the pivot
969  temp1 = *mjj;
970  *mjj = static_cast<T>(*(mjj + j + 1) * temp2);
971  *(mjj + j + 1) = static_cast<T>(temp1 * temp2);
972  *(mjj + j) = static_cast<T>(-*(mjj + j) * temp2);
973 
974  if (j < nrow - 1) // otherwise do nothing
975  {
976  // update A(j+2:n, j+2:n)
977  for (i = j + 2; i <= nrow; i++) {
978  ip = rhs.Array() + i * (i - 1) / 2 + j - 1;
979  temp1 = *ip * *mjj + *(ip + 1) * *(mjj + j);
980  // if (o2::gpu::GPUCommonMath::Abs(temp1 ) <= epsilon)
981  // temp1 = 0;
982  temp2 = *ip * *(mjj + j) + *(ip + 1) * *(mjj + j + 1);
983  // if (o2::gpu::GPUCommonMath::Abs(temp2 ) <= epsilon)
984  // temp2 = 0;
985  for (k = j + 2; k <= i; k++) {
986  ip = rhs.Array() + i * (i - 1) / 2 + k - 1;
987  iq = rhs.Array() + k * (k - 1) / 2 + j - 1;
988  *ip -= static_cast<T>(temp1 * *iq + temp2 * *(iq + 1));
989  // if (o2::gpu::GPUCommonMath::Abs(*ip) <= epsilon)
990  // *ip = 0;
991  }
992  }
993  // update L
994  for (i = j + 2; i <= nrow; i++) {
995  ip = rhs.Array() + i * (i - 1) / 2 + j - 1;
996  temp1 = *ip * *mjj + *(ip + 1) * *(mjj + j);
997  // if (o2::gpu::GPUCommonMath::Abs(temp1) <= epsilon)
998  // temp1 = 0;
999  *(ip + 1) = *ip * *(mjj + j) + *(ip + 1) * *(mjj + j + 1);
1000  // if (o2::gpu::GPUCommonMath::Abs(*(ip+1)) <= epsilon)
1001  // *(ip+1) = 0;
1002  *ip = static_cast<T>(temp1);
1003  }
1004  }
1005  }
1006  }
1007  } // end of main loop over columns
1008 
1009  if (j == nrow) // the the last pivot is 1x1
1010  {
1011  mjj = rhs.Array() + j * (j - 1) / 2 + j - 1;
1012  if (*mjj == 0) {
1013  ifail = 1;
1014  return;
1015  } else {
1016  *mjj = 1.0f / *mjj;
1017  }
1018  } // end of last pivot code
1019 
1020  // computing the inverse from the factorization
1021 
1022  for (j = nrow; j >= 1; j -= s) // loop over columns
1023  {
1024  mjj = rhs.Array() + j * (j - 1) / 2 + j - 1;
1025  if (piv[j - 1] > 0) // 1x1 pivot, compute column j of inverse
1026  {
1027  s = 1;
1028  if (j < nrow) {
1029  ip = rhs.Array() + (j + 1) * j / 2 + j - 1;
1030  for (i = 0; i < nrow - j; ip += 1 + j + i++) {
1031  x[i] = *ip;
1032  }
1033  for (i = j + 1; i <= nrow; i++) {
1034  temp2 = 0;
1035  ip = rhs.Array() + i * (i - 1) / 2 + j;
1036  for (k = 0; k <= i - j - 1; k++) {
1037  temp2 += *ip++ * x[k];
1038  }
1039  for (ip += i - 1; k < nrow - j; ip += 1 + j + k++) {
1040  temp2 += *ip * x[k];
1041  }
1042  *(rhs.Array() + i * (i - 1) / 2 + j - 1) = static_cast<T>(-temp2);
1043  }
1044  temp2 = 0;
1045  ip = rhs.Array() + (j + 1) * j / 2 + j - 1;
1046  for (k = 0; k < nrow - j; ip += 1 + j + k++) {
1047  temp2 += x[k] * *ip;
1048  }
1049  *mjj -= static_cast<T>(temp2);
1050  }
1051  } else // 2x2 pivot, compute columns j and j-1 of the inverse
1052  {
1053  if (piv[j - 1] != 0) {
1054  LOGF(error, "error in piv %lf \n", static_cast<T>(piv[j - 1]));
1055  }
1056  s = 2;
1057  if (j < nrow) {
1058  ip = rhs.Array() + (j + 1) * j / 2 + j - 1;
1059  for (i = 0; i < nrow - j; ip += 1 + j + i++) {
1060  x[i] = *ip;
1061  }
1062  for (i = j + 1; i <= nrow; i++) {
1063  temp2 = 0;
1064  ip = rhs.Array() + i * (i - 1) / 2 + j;
1065  for (k = 0; k <= i - j - 1; k++) {
1066  temp2 += *ip++ * x[k];
1067  }
1068  for (ip += i - 1; k < nrow - j; ip += 1 + j + k++) {
1069  temp2 += *ip * x[k];
1070  }
1071  *(rhs.Array() + i * (i - 1) / 2 + j - 1) = static_cast<T>(-temp2);
1072  }
1073  temp2 = 0;
1074  ip = rhs.Array() + (j + 1) * j / 2 + j - 1;
1075  for (k = 0; k < nrow - j; ip += 1 + j + k++) {
1076  temp2 += x[k] * *ip;
1077  }
1078  *mjj -= static_cast<T>(temp2);
1079  temp2 = 0;
1080  ip = rhs.Array() + (j + 1) * j / 2 + j - 2;

1082  temp2 += *ip * *(ip + 1);

1104  }

1158  value_type s11, s12;

1162  // double epsilon = 8*std::numeric_limits<T>::epsilon();
1163  // could be set to zero (like it was before)
1164  // but then the algorithm often doesn't detect
1165  // that a matrix is singular

1168  int jrange = 0, jover = 1, junder = -1;

1170  jfail = jrange;

1174  mIter mjj = mj;
1175  for (unsigned int j = 1; j <= n; j++) {
1176  unsigned int k = j;
1177  p = (o2::gpu::GPUCommonMath::Abs(*mjj));
1178  if (j != n) {
1179  mIter mij = mj + n + j - 1;
1180  for (unsigned int i = j + 1; i <= n; i++) {
1181  q = (o2::gpu::GPUCommonMath::Abs(*(mij)));
1182  if (q > p) {
1183  k = i;
1184  p = q;
1185  }
1186  mij += n;
1187  }
1188  if (k == j) {
1189  if (p <= epsilon) {
1190  det = 0;
1191  ifail = imposs;
1192  jfail = jrange;
1193  return ifail;
1194  }

1196  // must not change. So I change it twice.
1197  }

1199  mIter mkl = rhs.Array() + (k - 1) * n;

1203  *(mkl++) = static_cast<T>(tf);
1204  }

1206  ir[nxch] = (((j) << 12) + (k));
1207  } else {

1209  det = 0.0;
1210  ifail = imposs;
1211  jfail = jrange;
1212  return ifail;
1213  }
1214  }
