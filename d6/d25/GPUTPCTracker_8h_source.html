211 GPUdi() static int32_t CalculateHitWeight(int32_t NHits, float chi2)
212 {
213 const float chi2_suppress = 6.f;
214 float weight = (((float)NHits * (chi2_suppress - chi2 / 500.f)) * (1e9f / chi2_suppress / 160.f));
215 if (weight < 0.f || weight > 2e9f) {
216 return 0;
217 }
218 return ((int32_t)weight);
219 // return( (NHits << 16) + num);
220 }
