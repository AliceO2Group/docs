185 GPUdi() static int32_t CalculateHitWeight(int32_t NHits, float chi2)
186 {
187 const float chi2_suppress = 6.f;
188 float weight = (((float)NHits * (chi2_suppress - chi2 / 500.f)) * (1e9f / chi2_suppress / 160.f));
189 if (weight < 0.f || weight > 2e9f) {
190 return 0;
191 }
192 return ((int32_t)weight);
193 // return( (NHits << 16) + num);
194 }
