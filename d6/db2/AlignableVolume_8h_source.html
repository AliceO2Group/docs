258inline void AlignableVolume::getMatrixT2G(TGeoHMatrix& m) const
259{
260 // compute tracking to global matrix, i.e. glo = T2G*tra = L2G*loc = L2G*T2L*tra
261 m = getMatrixL2GIdeal();
262 m *= getMatrixT2L();
