27int32_t GPUChainTracking::ExtrapolationTracking(uint32_t iSlice, int32_t threadId, bool synchronizeOutput)
28{
29 runKernel<GPUTPCExtrapolationTracking>({GetGridBlk(256, iSlice % mRec->NStreams()), {iSlice}});
30 TransferMemoryResourceLinkToHost(RecoStep::TPCSliceTracking, processors()->tpcTrackers[iSlice].MemoryResCommon(), iSlice % mRec->NStreams());
31 if (synchronizeOutput) {
32 SynchronizeStream(iSlice % mRec->NStreams());
33 }
34 return (0);
35}

37int32_t GPUChainTracking::RunTPCTrackingSlices()
38{
39 if (mRec->GPUStuck()) {
40 GPUWarning("This GPU is stuck, processing of tracking for this event is skipped!");
41 return (1);
42 }
43
44 const auto& threadContext = GetThreadContext();
45
46 int32_t retVal = RunTPCTrackingSlices_internal();
47 if (retVal) {
48 SynchronizeGPU();
49 }
50 return (retVal != 0);

389 }
390 }
391 if (!(GetRecoStepsOutputs() & GPUDataTypes::InOutType::TPCSectorTracks) && param().rec.tpc.extrapolationTracking) {
392 std::vector<bool> blocking(NSLICES * mRec->NStreams());
393 for (int32_t i = 0; i < NSLICES; i++) {
394 for (int32_t j = 0; j < mRec->NStreams(); j++) {
395 blocking[i * mRec->NStreams() + j] = i % mRec->NStreams() == j;
396 }
397 }
398 for (uint32_t iSlice = 0; iSlice < NSLICES; iSlice++) {
399 uint32_t tmpSlice = GPUTPCExtrapolationTracking::ExtrapolationTrackingSliceOrder(iSlice);
400 if (!((GetRecoStepsOutputs() & GPUDataTypes::InOutType::TPCSectorTracks) || (doGPU && !(GetRecoStepsGPU() & RecoStep::TPCMerging)))) {
401 uint32_t sliceLeft, sliceRight;
402 GPUTPCExtrapolationTracking::ExtrapolationTrackingSliceLeftRight(tmpSlice, sliceLeft, sliceRight);
403 if (doGPU && !blocking[tmpSlice * mRec->NStreams() + sliceLeft % mRec->NStreams()]) {
404 StreamWaitForEvents(tmpSlice % mRec->NStreams(), &mEvents->slice[sliceLeft]);
405 blocking[tmpSlice * mRec->NStreams() + sliceLeft % mRec->NStreams()] = true;
406 }
407 if (doGPU && !blocking[tmpSlice * mRec->NStreams() + sliceRight % mRec->NStreams()]) {
408 StreamWaitForEvents(tmpSlice % mRec->NStreams(), &mEvents->slice[sliceRight]);
409 blocking[tmpSlice * mRec->NStreams() + sliceRight % mRec->NStreams()] = true;
410 }
411 }
412 ExtrapolationTracking(tmpSlice, 0, false);
413 }
414 }
415 for (uint32_t iSlice = 0; iSlice < NSLICES; iSlice++) {
416 if (doGPU && transferRunning[iSlice]) {
417 ReleaseEvent(mEvents->slice[iSlice]);
418 }
419 }
420 } else {
421 mSliceSelectorReady = NSLICES;
422 GPUCA_OPENMP(parallel for if(!doGPU && GetProcessingSettings().ompKernels != 1) num_threads(mRec->SetAndGetNestedLoopOmpFactor(!doGPU, NSLICES)))
423 for (uint32_t iSlice = 0; iSlice < NSLICES; iSlice++) {
424 if (param().rec.tpc.extrapolationTracking) {
425 ExtrapolationTracking(iSlice, 0);
426 }
427 if (GetRecoStepsOutputs() & GPUDataTypes::InOutType::TPCSectorTracks) {
428 WriteOutput(iSlice, 0);
429 }
430 }
431 mRec->SetNestedLoopOmpFactor(1);
432 }
433
434 if (param().rec.tpc.extrapolationTracking && GetProcessingSettings().debugLevel >= 3) {
435 for (uint32_t iSlice = 0; iSlice < NSLICES; iSlice++) {
436 GPUInfo("Slice %d - Tracks: Local %d Extrapolated %d - Hits: Local %d Extrapolated %d", iSlice,
437 processors()->tpcTrackers[iSlice].CommonMemory()->nLocalTracks, processors()->tpcTrackers[iSlice].CommonMemory()->nTracks, processors()->tpcTrackers[iSlice].CommonMemory()->nLocalTrackHits, processors()->tpcTrackers[iSlice].CommonMemory()->nTrackHits);
438 }
439 }
440
441 if (GetProcessingSettings().debugMask & 1024 && !GetProcessingSettings().deterministicGPUReconstruction) {
442 for (uint32_t i = 0; i < NSLICES; i++) {
443 processors()->tpcTrackers[i].DumpOutput(*mDebugFile);
444 }
445 }
446
447 if (DoProfile()) {
448 return (1);
449 }
450 for (uint32_t i = 0; i < NSLICES; i++) {
451 mIOPtrs.nSliceTracks[i] = *processors()->tpcTrackers[i].NTracks();
452 mIOPtrs.sliceTracks[i] = processors()->tpcTrackers[i].Tracks();
453 mIOPtrs.nSliceClusters[i] = *processors()->tpcTrackers[i].NTrackHits();
454 mIOPtrs.sliceClusters[i] = processors()->tpcTrackers[i].TrackHits();
455 if (GetProcessingSettings().keepDisplayMemory && !GetProcessingSettings().keepAllMemory) {
456 TransferMemoryResourcesToHost(RecoStep::TPCSliceTracking, &processors()->tpcTrackers[i], -1, true);
457 }
458 }
459 if (GetProcessingSettings().debugLevel >= 2) {
460 GPUInfo("TPC Slice Tracker finished");
461 }
462 mRec->PopNonPersistentMemory(RecoStep::TPCSliceTracking, qStr2Tag("TPCSLTRK"));
463 return 0;
464}
