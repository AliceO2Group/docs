29template <int32_t I>
30int32_t GPUChainTracking::RunTRDTracking()
31{
32#ifndef GPUCA_ALIROOT_LIB
33 auto& Tracker = processors()->getTRDTracker<I>();
34 if (!Tracker.IsInitialized()) {
35 return 1;
36 }
37
38 Tracker.Reset();
39 if (mIOPtrs.nTRDTracklets == 0) {
40 return 0;
41 }
42
43 bool isTriggeredEvent = (param().continuousMaxTimeBin == 0);
44
45 if (!isTriggeredEvent) {
46 Tracker.SetProcessPerTimeFrame(true);
47 }
48
49 Tracker.SetGenerateSpacePoints(mIOPtrs.trdSpacePoints == nullptr);
50
51 mRec->PushNonPersistentMemory(qStr2Tag("TRDTRACK"));
52 SetupGPUProcessor(&Tracker, true);
53
54 if constexpr (I == GPUTRDTrackerKernels::gpuVersion) {
55 for (uint32_t i = 0; i < mIOPtrs.nMergedTracks; i++) {
56 const GPUTPCGMMergedTrack& trk = mIOPtrs.mergedTracks[i];
57 if (!Tracker.PreCheckTrackTRDCandidate(trk)) {
58 continue;
59 }
60 const GPUTRDTrackGPU& trktrd = param().rec.tpc.nWaysOuter ? (GPUTRDTrackGPU)trk.OuterParam() : (GPUTRDTrackGPU)trk;
61 if (!Tracker.CheckTrackTRDCandidate(trktrd)) {
62 continue;
63 }
64 GPUTRDTrackerGPU::HelperTrackAttributes trkAttribs, *trkAttribsPtr{nullptr};
65 if (!isTriggeredEvent) {
66 const float tpcTBinMUS = 0.199606f;
67 trkAttribs.mTime = trk.GetParam().GetTZOffset() * tpcTBinMUS;
68 trkAttribs.mTimeAddMax = 50.f; // half of a TPC drift time in us
69 trkAttribs.mTimeSubMax = 50.f; // half of a TPC drift time in us
70 if (!trk.CCE()) {
71 if (trk.CSide()) {
72 // track has only C-side clusters
73 trkAttribs.mSide = 1;
74 } else {
75 // track has only A-side clusters
76 trkAttribs.mSide = -1;
77 }
78 }
79 trkAttribsPtr = &trkAttribs;
80 }
81 if (Tracker.LoadTrack(trktrd, i, false, trkAttribsPtr)) {
82 return 1;
83 }
84 }
85 } else {
86#ifdef GPUCA_HAVE_O2HEADERS
87 for (uint32_t i = 0; i < mIOPtrs.nOutputTracksTPCO2; i++) {
88 const auto& trk = mIOPtrs.outputTracksTPCO2[i];
89
90 if (!Tracker.PreCheckTrackTRDCandidate(trk)) {
91 continue;
92 }
93 const GPUTRDTrack& trktrd = (GPUTRDTrack)trk;
94 if (!Tracker.CheckTrackTRDCandidate(trktrd)) {
95 continue;
96 }
97
98 GPUTRDTracker::HelperTrackAttributes trkAttribs, *trkAttribsPtr{nullptr};
99 if (!isTriggeredEvent) {
100 const float tpcTBinMUS = 0.199606f;
101 trkAttribs.mTime = trk.getTime0() * tpcTBinMUS;
102 trkAttribs.mTimeAddMax = trk.getDeltaTFwd() * tpcTBinMUS;
103 trkAttribs.mTimeSubMax = trk.getDeltaTBwd() * tpcTBinMUS;
104 if (trk.hasASideClustersOnly()) {
105 trkAttribs.mSide = -1;
106 } else if (trk.hasCSideClustersOnly()) {
107 trkAttribs.mSide = 1;
108 }
109 trkAttribsPtr = &trkAttribs;
110 }
111 if (Tracker.LoadTrack(trktrd, i, false, trkAttribsPtr)) {
112 return 1;
113 }
114 }
115#endif
116 }
117
118 DoTRDGPUTracking<I>();
119
120 mIOPtrs.nTRDTracks = Tracker.NTracks();
121 if constexpr (I == GPUTRDTrackerKernels::gpuVersion) {
122 mIOPtrs.trdTracks = Tracker.Tracks();
123 mIOPtrs.trdTracksO2 = nullptr;
124 } else {
125#ifdef GPUCA_HAVE_O2HEADERS
126 mIOPtrs.trdTracks = nullptr;
127 mIOPtrs.trdTracksO2 = Tracker.Tracks();
128#endif
129 }
130 mRec->PopNonPersistentMemory(RecoStep::TRDTracking, qStr2Tag("TRDTRACK"));
131
132#endif // GPUCA_ALIROOT_LIB
133 return 0;
134}

136template <int32_t I, class T>
137int32_t GPUChainTracking::DoTRDGPUTracking(T* externalInstance)
138{
139#ifdef GPUCA_HAVE_O2HEADERS
140 bool doGPU = GetRecoStepsGPU() & RecoStep::TRDTracking;
141 auto* Tracker = &processors()->getTRDTracker<I>();
142 auto* TrackerShadow = doGPU ? &processorsShadow()->getTRDTracker<I>() : Tracker;
143 if (externalInstance) {
144 if constexpr (std::is_same_v<decltype(Tracker), decltype(externalInstance)>) {
145 Tracker = externalInstance;
146 } else {
147 throw std::runtime_error("Must not provide external instance that does not match template type");
148 }
149 }
150 Tracker->PrepareTracking(this);
151
152 int32_t useStream = 0;
153
154 const auto& threadContext = GetThreadContext();
155 SetupGPUProcessor(Tracker, false);
156 if (doGPU) {
157 TrackerShadow->OverrideGPUGeometry(reinterpret_cast<GPUTRDGeometry*>(mFlatObjectsDevice.mCalibObjects.trdGeometry));
158 mInputsHost->mNTRDTracklets = mInputsShadow->mNTRDTracklets = processorsShadow()->ioPtrs.nTRDTracklets = mIOPtrs.nTRDTracklets;
159 mInputsHost->mNTRDTriggerRecords = mInputsShadow->mNTRDTriggerRecords = processorsShadow()->ioPtrs.nTRDTriggerRecords = mIOPtrs.nTRDTriggerRecords;
160 mInputsHost->mDoSpacepoints = mInputsShadow->mDoSpacepoints = !Tracker->GenerateSpacepoints();
161 AllocateRegisteredMemory(mInputsHost->mResourceTRD);
162 processorsShadow()->ioPtrs.trdTracklets = mInputsShadow->mTRDTracklets;
163 processorsShadow()->ioPtrs.trdSpacePoints = Tracker->GenerateSpacepoints() ? Tracker->SpacePoints() : mInputsShadow->mTRDSpacePoints;
164 if constexpr (std::is_same_v<decltype(processorsShadow()->ioPtrs.trdTracks), decltype(TrackerShadow->Tracks())>) {
165 processorsShadow()->ioPtrs.trdTracks = TrackerShadow->Tracks();
166 } else {
167 processorsShadow()->ioPtrs.trdTracks = nullptr;
168 }
169 processorsShadow()->ioPtrs.nTRDTracks = mIOPtrs.nTRDTracks;
170 processorsShadow()->ioPtrs.trdTriggerTimes = mInputsShadow->mTRDTriggerTimes;
171 processorsShadow()->ioPtrs.trdTrackletIdxFirst = mInputsShadow->mTRDTrackletIdxFirst;
172 GPUMemCpy(RecoStep::TRDTracking, mInputsShadow->mTRDTracklets, mIOPtrs.trdTracklets, sizeof(*mIOPtrs.trdTracklets) * mIOPtrs.nTRDTracklets, useStream, true);
173 if (!Tracker->GenerateSpacepoints()) {
174 GPUMemCpy(RecoStep::TRDTracking, mInputsShadow->mTRDSpacePoints, mIOPtrs.trdSpacePoints, sizeof(*mIOPtrs.trdSpacePoints) * mIOPtrs.nTRDTracklets, useStream, true);
175 }
176 GPUMemCpy(RecoStep::TRDTracking, mInputsShadow->mTRDTriggerTimes, mIOPtrs.trdTriggerTimes, sizeof(*mIOPtrs.trdTriggerTimes) * mIOPtrs.nTRDTriggerRecords, useStream, true);
177 GPUMemCpy(RecoStep::TRDTracking, mInputsShadow->mTRDTrackletIdxFirst, mIOPtrs.trdTrackletIdxFirst, sizeof(*mIOPtrs.trdTrackletIdxFirst) * mIOPtrs.nTRDTriggerRecords, useStream, true);
178 if (mIOPtrs.trdTrigRecMask) {
179 processorsShadow()->ioPtrs.trdTrigRecMask = mInputsShadow->mTRDTrigRecMask;
180 GPUMemCpy(RecoStep::TRDTracking, mInputsShadow->mTRDTrigRecMask, mIOPtrs.trdTrigRecMask, sizeof(*mIOPtrs.trdTrigRecMask) * mIOPtrs.nTRDTriggerRecords, useStream, true);
181 } else {
182 processorsShadow()->ioPtrs.trdTrigRecMask = nullptr;
183 }
184 WriteToConstantMemory(RecoStep::TRDTracking, (char*)&processors()->ioPtrs - (char*)processors(), &processorsShadow()->ioPtrs, sizeof(processorsShadow()->ioPtrs), useStream);
185 WriteToConstantMemory(RecoStep::TRDTracking, (char*)&processors()->getTRDTracker<I>() - (char*)processors(), TrackerShadow, sizeof(*TrackerShadow), useStream);
186 }
187
188 TransferMemoryResourcesToGPU(RecoStep::TRDTracking, Tracker, useStream);
189 runKernel<GPUTRDTrackerKernels, I>(GetGridAuto(useStream), externalInstance ? Tracker : nullptr);
190 TransferMemoryResourcesToHost(RecoStep::TRDTracking, Tracker, useStream);
191 SynchronizeStream(useStream);
192
193 if (GetProcessingSettings().debugLevel >= 2) {
194 GPUInfo("GPU TRD tracker Finished");
195 }
196#endif
197 return (0);
198}
