113 inline Int_t SymBDMatrix::GetIndex(Int_t row, Int_t col) const
114 {
115  // lower triangle band is actually filled
116  if (row < col) {
117  Swap(row, col);
118  }
119  col -= row;
120  if (col < -GetBandHWidth()) {
121  return -1;
122  }
123  return GetIndex(row) + col;
124 }
