68 void strips2json(const std::vector<o2::mid::ExtendedMappingInfo>& infos, const char* outDir)
69 {
70  rapidjson::StringBuffer buffer;
71  rapidjson::Writer<rapidjson::StringBuffer> writer(buffer);
72 
73  auto strip2json = [&](const o2::mid::ExtendedMappingInfo& info) {
74  writer.StartObject();
75  writer.Key("id");
76  writer.Int(info.id);
77  writer.Key("vertices");
78  writer.StartArray();
79  // (x1,y1)
80  writer.StartObject();
81  writer.Key("x");
82  writer.Int(info.xpos);
83  writer.Key("y");
84  writer.Int(info.ypos);
85  writer.EndObject();
86  // (x2,y1)
87  writer.StartObject();
88  writer.Key("x");
89  writer.Int(info.xpos + info.xwidth);
90  writer.Key("y");
91  writer.Int(info.ypos);
92  writer.EndObject();
93  // (x2,y2)
94  writer.StartObject();
95  writer.Key("x");
96  writer.Int(info.xpos + info.xwidth);
97  writer.Key("y");
98  writer.Int(info.ypos + info.ywidth);
99  writer.EndObject();
100  // (x1,y2)
101  writer.StartObject();
102  writer.Key("x");
103  writer.Int(info.xpos);
104  writer.Key("y");
105  writer.Int(info.ypos + info.ywidth);
106  writer.EndObject();
107  // (x1,y1)
108  writer.StartObject();
109  writer.Key("x");
110  writer.Int(info.xpos);
111  writer.Key("y");
112  writer.Int(info.ypos);
113  writer.EndObject();
114  writer.EndArray();
115  writer.EndObject();
116  };
117 
118  std::array<std::string, 2> planes{"bending", "non-bending"};
119 
120  for (int ide = 0; ide < o2::mid::detparams::NDetectionElements; ++ide) {
121  for (int icath = 0; icath < 2; ++icath) {
122  std::string fname = fmt::format("{}/strips.{}.{}.json", outDir, ide, planes[icath]);
123  writer.StartArray();
124  for (auto& info : infos) {
125  if (info.deId == ide && info.cathode == icath) {
126  strip2json(info);
127  }
128  }
129  writer.EndArray();
130  std::ofstream outFile(fname);
131  outFile << buffer.GetString() << std::endl;
132  buffer.Clear();
133  }
134  }
135 }

137 void de2json(const std::map<int, std::vector<std::pair<int, int>>>& deMap, const char* outDir)
138 {
139  rapidjson::StringBuffer buffer;
140  rapidjson::Writer<rapidjson::StringBuffer> writer(buffer);
141  for (auto& item : deMap) {
142  buffer.Clear();
143  writer.StartObject();
144  writer.Key("deId");
145  writer.Int(item.first);
146  writer.Key("vertices");
147  writer.StartArray();
148  for (auto& vtx : item.second) {
149  writer.StartObject();
150  writer.Key("x");
151  writer.Int(vtx.first);
152  writer.Key("y");
153  writer.Int(vtx.second);
154  writer.EndObject();
155  }
156  writer.EndArray();
157  writer.EndObject();
158 
159  std::string fname = fmt::format("{}/de.{}.json", outDir, item.first);
160  std::ofstream outFile(fname);
161  outFile << buffer.GetString() << std::endl;
