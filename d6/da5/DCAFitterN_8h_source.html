31 struct TrackCovI {
32  float sxx, syy, syz, szz;
33 
34  GPUd() TrackCovI(const o2::track::TrackParCov& trc, float xerrFactor = 1.) { set(trc, xerrFactor); }
35 
36  GPUdDefault() TrackCovI() = default;
37 
38  GPUd() void set(const o2::track::TrackParCov& trc, float xerrFactor = 1)
39  {
40  // we assign Y error to X for DCA calculation
41  // (otherwise for quazi-collinear tracks the X will not be constrained)
42  float cyy = trc.getSigmaY2(), czz = trc.getSigmaZ2(), cyz = trc.getSigmaZY(), cxx = cyy * xerrFactor;
43  float detYZ = cyy * czz - cyz * cyz;
44  if (detYZ > 0.) {
45  auto detYZI = 1. / detYZ;
46  sxx = 1. / cxx;
47  syy = czz * detYZI;
48  syz = -cyz * detYZI;
49  szz = cyy * detYZI;
50  } else {
51 #ifndef GPUCA_GPUCODE
52  throw std::runtime_error("invalid track covariance");
53 #else
54  printf("invalid track covariance\n");
55 #endif
56  }
57  }
58 };

62 struct TrackDeriv {
63  float dydx, dzdx, d2ydx2, d2zdx2;
64  GPUdDefault() TrackDeriv() = default;
65  GPUd() TrackDeriv(const o2::track::TrackPar& trc, float bz) { set(trc, bz); }
66  GPUd() void set(const o2::track::TrackPar& trc, float bz)
67  {
68  float snp = trc.getSnp(), csp = o2::gpu::GPUCommonMath::Sqrt((1. - snp) * (1. + snp)), cspI = 1. / csp, crv2c = trc.getCurvature(bz) * cspI;
69  dydx = snp * cspI; // = snp/csp
70  dzdx = trc.getTgl() * cspI; // = tgl/csp
71  d2ydx2 = crv2c * cspI * cspI; // = crv/csp^3
72  d2zdx2 = crv2c * dzdx * dydx; // = crv*tgl*snp/csp^3
73  }
74 };

358 template <int N, typename... Args>
359 template <class... Tr>
360 GPUd() int DCAFitterN<N, Args...>::process(const Tr&... args)
361 {
362  // This is a main entry point: fit PCA of N tracks
363  mCallID++;
364  static_assert(sizeof...(args) == N, "incorrect number of input tracks");
365  assign(0, args...);
366  clear();
367  for (int i = 0; i < N; i++) {
368  mTrAux[i].set(*mOrigTrPtr[i], mBz);
369  }
370  if (!mCrossings.set(mTrAux[0], *mOrigTrPtr[0], mTrAux[1], *mOrigTrPtr[1], mMaxDXYIni, mIsCollinear)) { // even for N>2 it should be enough to test just 1 loop
371  return 0; // no crossing
372  }
373  for (int ih = 0; ih < MAXHYP; ih++) {
374  mPropFailed[ih] = false;
375  }
376  if (mUseAbsDCA) {
377  calcRMatrices(); // needed for fast residuals derivatives calculation in case of abs. distance minimization
378  }
379  if (mCrossings.nDCA == MAXHYP) { // if there are 2 candidates and they are too close, chose their mean as a starting point
380  auto dst2 = (mCrossings.xDCA[0] - mCrossings.xDCA[1]) * (mCrossings.xDCA[0] - mCrossings.xDCA[1]) +
381  (mCrossings.yDCA[0] - mCrossings.yDCA[1]) * (mCrossings.yDCA[0] - mCrossings.yDCA[1]);
382  if (dst2 < mMaxDist2ToMergeSeeds) {
383  mCrossings.nDCA = 1;
384  mCrossings.xDCA[0] = 0.5 * (mCrossings.xDCA[0] + mCrossings.xDCA[1]);
385  mCrossings.yDCA[0] = 0.5 * (mCrossings.yDCA[0] + mCrossings.yDCA[1]);
386  }
387  }
388  // check all crossings
389  for (int ic = 0; ic < mCrossings.nDCA; ic++) {
390  // check if radius is acceptable
391  if (mCrossings.xDCA[ic] * mCrossings.xDCA[ic] + mCrossings.yDCA[ic] * mCrossings.yDCA[ic] > mMaxR2) {
392  continue;
393  }
394  mCrossIDCur = ic;
395  mCrossIDAlt = (mCrossings.nDCA == 2 && mAllowAltPreference) ? 1 - ic : -1; // works for max 2 crossings
396  mNIters[mCurHyp] = 0;
397  mTrPropDone[mCurHyp] = false;
398  mChi2[mCurHyp] = -1.;
399  mPCA[mCurHyp][0] = mCrossings.xDCA[ic];
400  mPCA[mCurHyp][1] = mCrossings.yDCA[ic];
401 
402  if (mUseAbsDCA ? minimizeChi2NoErr() : minimizeChi2()) {
403  mOrder[mCurHyp] = mCurHyp;
404  if (mPropagateToPCA && !propagateTracksToVertex(mCurHyp)) {
405  continue; // discard candidate if failed to propagate to it
406  }
407  mCurHyp++;
408  }
409  }
410 
411  for (int i = mCurHyp; i--;) { // order in quality
412  for (int j = i; j--;) {
413  if (mChi2[mOrder[i]] < mChi2[mOrder[j]]) {
414  o2::gpu::GPUCommonMath::Swap(mOrder[i], mOrder[j]);
415  }
416  }
417  }
418  if (mUseAbsDCA && mWeightedFinalPCA) {
419  for (int i = mCurHyp; i--;) {
420  recalculatePCAWithErrors(i);
421  }
422  }
423  return mCurHyp;
424 }

427 template <int N, typename... Args>
428 GPUd() bool DCAFitterN<N, Args...>::calcPCACoefs()
429 {
430  //< calculate Ti matrices for global vertex decomposition to V = sum_{0<i<N} Ti pi, see EQ.T in the ref
431  if (!calcInverseWeight()) {
432  return false;
433  }
434  for (int i = N; i--;) { // build Mi*Ei matrix
435  const auto& taux = mTrAux[i];
436  const auto& tcov = mTrcEInv[mCurHyp][i];
437  MatStd3D miei;
438  miei[0][0] = taux.c * tcov.sxx;
439  miei[0][1] = -taux.s * tcov.syy;
440  miei[0][2] = -taux.s * tcov.syz;
441  miei[1][0] = taux.s * tcov.sxx;
442  miei[1][1] = taux.c * tcov.syy;
443  miei[1][2] = taux.c * tcov.syz;
444  miei[2][0] = 0;
445  miei[2][1] = tcov.syz;
446  miei[2][2] = tcov.szz;
447  mTrCFVT[mCurHyp][i] = mWeightInv * miei;
448  }
449  return true;
450 }

453 template <int N, typename... Args>
454 GPUd() bool DCAFitterN<N, Args...>::calcInverseWeight()
455 {
456  //< calculate [sum_{0<j<N} M_j*E_j*M_j^T]^-1 used for Ti matrices, see EQ.T
457  auto* arrmat = mWeightInv.Array();
458  memset(arrmat, 0, sizeof(mWeightInv));
459  enum { XX,
460  XY,
461  YY,
462  XZ,
463  YZ,
464  ZZ };
465  for (int i = N; i--;) {
466  const auto& taux = mTrAux[i];
467  const auto& tcov = mTrcEInv[mCurHyp][i];
468  arrmat[XX] += taux.cc * tcov.sxx + taux.ss * tcov.syy;
469  arrmat[XY] += taux.cs * (tcov.sxx - tcov.syy);
470  arrmat[XZ] += -taux.s * tcov.syz;
471  arrmat[YY] += taux.cc * tcov.syy + taux.ss * tcov.sxx;
472  arrmat[YZ] += taux.c * tcov.syz;
473  arrmat[ZZ] += tcov.szz;
474  }
475  // invert 3x3 symmetrix matrix
476  return mWeightInv.Invert();
477 }

480 template <int N, typename... Args>
481 GPUd() void DCAFitterN<N, Args...>::calcResidDerivatives()
482 {
483  //< calculate matrix of derivatives for weighted chi2: residual i vs parameter X of track j
484  MatStd3D matMT;
485  for (int i = N; i--;) { // residual being differentiated
486  const auto& taux = mTrAux[i];
487  for (int j = N; j--;) { // track over which we differentiate
488  const auto& matT = mTrCFVT[mCurHyp][j]; // coefficient matrix for track J
489  const auto& trDx = mTrDer[mCurHyp][j]; // track point derivs over track X param
490  auto& dr1 = mDResidDx[i][j];
491  auto& dr2 = mD2ResidDx2[i][j];
492  // calculate M_i^tr * T_j
493  matMT[0][0] = taux.c * matT[0][0] + taux.s * matT[1][0];
494  matMT[0][1] = taux.c * matT[0][1] + taux.s * matT[1][1];
495  matMT[0][2] = taux.c * matT[0][2] + taux.s * matT[1][2];
496  matMT[1][0] = -taux.s * matT[0][0] + taux.c * matT[1][0];
497  matMT[1][1] = -taux.s * matT[0][1] + taux.c * matT[1][1];
498  matMT[1][2] = -taux.s * matT[0][2] + taux.c * matT[1][2];
499  matMT[2][0] = matT[2][0];
500  matMT[2][1] = matT[2][1];
501  matMT[2][2] = matT[2][2];
502 
503  // calculate DResid_i/Dx_j = (delta_ij - M_i^tr * T_j) * DTrack_k/Dx_k
504  dr1[0] = -(matMT[0][0] + matMT[0][1] * trDx.dydx + matMT[0][2] * trDx.dzdx);
505  dr1[1] = -(matMT[1][0] + matMT[1][1] * trDx.dydx + matMT[1][2] * trDx.dzdx);
506  dr1[2] = -(matMT[2][0] + matMT[2][1] * trDx.dydx + matMT[2][2] * trDx.dzdx);
507 
508  // calculate D2Resid_I/(Dx_J Dx_K) = (delta_ijk - M_i^tr * T_j * delta_jk) * D2Track_k/dx_k^2
509  dr2[0] = -(matMT[0][1] * trDx.d2ydx2 + matMT[0][2] * trDx.d2zdx2);
510  dr2[1] = -(matMT[1][1] * trDx.d2ydx2 + matMT[1][2] * trDx.d2zdx2);
511  dr2[2] = -(matMT[2][1] * trDx.d2ydx2 + matMT[2][2] * trDx.d2zdx2);
512 
513  if (i == j) {
514  dr1[0] += 1.;
515  dr1[1] += trDx.dydx;
516  dr1[2] += trDx.dzdx;
517 
518  dr2[1] += trDx.d2ydx2;
519  dr2[2] += trDx.d2zdx2;
520  }
521  } // track over which we differentiate
522  } // residual being differentiated
523 }

526 template <int N, typename... Args>
527 GPUd() void DCAFitterN<N, Args...>::calcResidDerivativesNoErr()
528 {
529  //< calculate matrix of derivatives for absolute distance chi2: residual i vs parameter X of track j
530  constexpr double NInv1 = 1. - NInv; // profit from Rii = I/Ninv
531  for (int i = N; i--;) { // residual being differentiated
532  const auto& trDxi = mTrDer[mCurHyp][i]; // track point derivs over track X param
533  auto& dr1ii = mDResidDx[i][i];
534  auto& dr2ii = mD2ResidDx2[i][i];
535  dr1ii[0] = NInv1;
536  dr1ii[1] = NInv1 * trDxi.dydx;
537  dr1ii[2] = NInv1 * trDxi.dzdx;
538 
539  dr2ii[0] = 0;
540  dr2ii[1] = NInv1 * trDxi.d2ydx2;
541  dr2ii[2] = NInv1 * trDxi.d2zdx2;
542 
543  for (int j = i; j--;) { // track over which we differentiate
544  auto& dr1ij = mDResidDx[i][j];
545  auto& dr1ji = mDResidDx[j][i];
546  const auto& trDxj = mTrDer[mCurHyp][j]; // track point derivs over track X param
547  auto cij = mCosDif[i][j], sij = mSinDif[i][j]; // M_i^T*M_j / N matrices non-trivial elements = {ci*cj+si*sj , si*cj-ci*sj }, see 5 in ref.
548 
549  // calculate DResid_i/Dx_j = (delta_ij - R_ij) * DTrack_j/Dx_j for j<i
550  dr1ij[0] = -(cij + sij * trDxj.dydx);
551  dr1ij[1] = -(-sij + cij * trDxj.dydx);
552  dr1ij[2] = -trDxj.dzdx * NInv;
553 
554  // calculate DResid_j/Dx_i = (delta_ij - R_ji) * DTrack_i/Dx_i for j<i
555  dr1ji[0] = -(cij - sij * trDxi.dydx);
556  dr1ji[1] = -(sij + cij * trDxi.dydx);
557  dr1ji[2] = -trDxi.dzdx * NInv;
558 
559  auto& dr2ij = mD2ResidDx2[i][j];
560  auto& dr2ji = mD2ResidDx2[j][i];
561  // calculate D2Resid_I/(Dx_J Dx_K) = (delta_ij - Rij) * D2Track_j/dx_j^2 * delta_jk for j<i
562  dr2ij[0] = -sij * trDxj.d2ydx2;
563  dr2ij[1] = -cij * trDxj.d2ydx2;
564  dr2ij[2] = -trDxj.d2zdx2 * NInv;
565 
566  // calculate D2Resid_j/(Dx_i Dx_k) = (delta_ij - Rji) * D2Track_i/dx_i^2 * delta_ik for j<i
567  dr2ji[0] = sij * trDxi.d2ydx2;
568  dr2ji[1] = -cij * trDxi.d2ydx2;
569  dr2ji[2] = -trDxi.d2zdx2 * NInv;
570 
571  } // track over which we differentiate
572  } // residual being differentiated
573 }

591 template <int N, typename... Args>
592 GPUd() void DCAFitterN<N, Args...>::calcChi2Derivatives()
593 {
594  //< calculate 1st and 2nd derivatives of wighted DCA (chi2) over track parameters X, see EQ.Chi2 in the ref
595  o2::gpu::gpustd::array<o2::gpu::gpustd::array<Vec3D, N>, N> covIDrDx; // tempory vectors of covI_j * dres_j/dx_i
596 
597  // chi2 1st derivative
598  for (int i = N; i--;) {
599  auto& dchi1 = mDChi2Dx[i]; // DChi2/Dx_i = sum_j { res_j * covI_j * Dres_j/Dx_i }
600  dchi1 = 0;
601  for (int j = N; j--;) {
602  const auto& res = mTrRes[mCurHyp][j]; // vector of residuals of track j
603  const auto& covI = mTrcEInv[mCurHyp][j]; // inverse cov matrix of track j
604  const auto& dr1 = mDResidDx[j][i]; // vector of j-th residuals 1st derivative over X param of track i
605  auto& cidr = covIDrDx[i][j]; // vector covI_j * dres_j/dx_i, save for 2nd derivative calculation
606  cidr[0] = covI.sxx * dr1[0];
607  cidr[1] = covI.syy * dr1[1] + covI.syz * dr1[2];
608  cidr[2] = covI.syz * dr1[1] + covI.szz * dr1[2];
609  // calculate res_i * covI_j * dres_j/dx_i
610  dchi1 += o2::math_utils::Dot(res, cidr);
611  }
612  }
613  // chi2 2nd derivative
614  for (int i = N; i--;) {
615  for (int j = i + 1; j--;) { // symmetric matrix
616  auto& dchi2 = mD2Chi2Dx2[i][j]; // D2Chi2/Dx_i/Dx_j = sum_k { Dres_k/Dx_j * covI_k * Dres_k/Dx_i + res_k * covI_k * D2res_k/Dx_i/Dx_j }
617  dchi2 = 0;
618  for (int k = N; k--;) {
619  const auto& dr1j = mDResidDx[k][j]; // vector of k-th residuals 1st derivative over X param of track j
620  const auto& cidrkj = covIDrDx[i][k]; // vector covI_k * dres_k/dx_i
621  dchi2 += o2::math_utils::Dot(dr1j, cidrkj);
622  if (k == j) {
623  const auto& res = mTrRes[mCurHyp][k]; // vector of residuals of track k
624  const auto& covI = mTrcEInv[mCurHyp][k]; // inverse cov matrix of track k
625  const auto& dr2ij = mD2ResidDx2[k][j]; // vector of k-th residuals 2nd derivative over X params of track j
626  dchi2 += res[0] * covI.sxx * dr2ij[0] + res[1] * (covI.syy * dr2ij[1] + covI.syz * dr2ij[2]) + res[2] * (covI.syz * dr2ij[1] + covI.szz * dr2ij[2]);
627  }
628  }
629  }
630  }
631 }

875 //___________________________________________________________________
876 template <int N, typename... Args>
877 GPUd() bool DCAFitterN<N, Args...>::minimizeChi2()
878 {
879  // find best chi2 (weighted DCA) of N tracks in the vicinity of the seed PCA
880  for (int i = N; i--;) {
881  mCandTr[mCurHyp][i] = *mOrigTrPtr[i];
882  auto x = mTrAux[i].c * mPCA[mCurHyp][0] + mTrAux[i].s * mPCA[mCurHyp][1]; // X of PCA in the track frame
883  if (x < mMinXSeed || !propagateToX(mCandTr[mCurHyp][i], x)) {
884  return false;
885  }
886  setTrackPos(mTrPos[mCurHyp][i], mCandTr[mCurHyp][i]); // prepare positions
887  mTrcEInv[mCurHyp][i].set(mCandTr[mCurHyp][i], XerrFactor); // prepare inverse cov.matrices at starting point
888  }
889 
890  if (mMaxDZIni > 0 && !roughDZCut()) { // apply rough cut on tracks Z difference
891  return false;
892  }
893 
894  if (!calcPCACoefs()) { // prepare tracks contribution matrices to the global PCA
895  return false;
896  }
897  calcPCA(); // current PCA
898  calcTrackResiduals(); // current track residuals
899  float chi2Upd, chi2 = calcChi2();
900  do {
901  calcTrackDerivatives(); // current track derivatives (1st and 2nd)
902  calcResidDerivatives(); // current residals derivatives (1st and 2nd)
903  calcChi2Derivatives(); // current chi2 derivatives (1st and 2nd)
904 
905  // do Newton-Rapson iteration with corrections = - dchi2/d{x0..xN} * [ d^2chi2/d{x0..xN}^2 ]^-1
906  if (!mD2Chi2Dx2.Invert()) {
907 #ifndef GPUCA_GPUCODE_DEVICE
908  LOG(error) << "InversionFailed";
909 #else
910  printf("InversionFailed\n");
911 #endif
912  return false;
913  }
914  VecND dx = mD2Chi2Dx2 * mDChi2Dx;
915  if (!correctTracks(dx)) {

938 GPUd() bool DCAFitterN<N, Args...>::minimizeChi2NoErr()

940  // find best chi2 (absolute DCA) of N tracks in the vicinity of the PCA seed
941

943  mCandTr[mCurHyp][i] = *mOrigTrPtr[i];
944  auto x = mTrAux[i].c * mPCA[mCurHyp][0] + mTrAux[i].s * mPCA[mCurHyp][1]; // X of PCA in the track frame
945  if (x < mMinXSeed || !propagateParamToX(mCandTr[mCurHyp][i], x)) {
946  return false;
947  }
948  setTrackPos(mTrPos[mCurHyp][i], mCandTr[mCurHyp][i]); // prepare positions
949  }
950  if (mMaxDZIni > 0 && !roughDZCut()) { // apply rough cut on tracks Z difference
951  return false;
952  }
953 
954  calcPCANoErr(); // current PCA
955  calcTrackResiduals(); // current track residuals
956  float chi2Upd, chi2 = calcChi2NoErr();
957  do {
958  calcTrackDerivatives(); // current track derivatives (1st and 2nd)
959  calcResidDerivativesNoErr(); // current residals derivatives (1st and 2nd)
960  calcChi2DerivativesNoErr(); // current chi2 derivatives (1st and 2nd)
961 
962  // do Newton-Rapson iteration with corrections = - dchi2/d{x0..xN} * [ d^2chi2/d{x0..xN}^2 ]^-1
963  if (!mD2Chi2Dx2.Invert()) {
964 #ifndef GPUCA_GPUCODE_DEVICE
965  LOG(error) << "InversionFailed";
966 #else
967  printf("InversionFailed\n");
968 #endif
969  return false;
970  }
971  VecND dx = mD2Chi2Dx2 * mDChi2Dx;
972  if (!correctTracks(dx)) {
973  return false;
974  }
975  calcPCANoErr(); // updated PCA
976  if (mCrossIDAlt >= 0 && closerToAlternative()) {
977  mAllowAltPreference = false;
978  return false;
979  }
980  calcTrackResiduals(); // updated residuals
981  chi2Upd = calcChi2NoErr(); // updated chi2
982  if (getAbsMax(dx) < mMinParamChange || chi2Upd > chi2 * mMinRelChi2Change) {
983  chi2 = chi2Upd;
984  break; // converged
985  }
986  chi2 = chi2Upd;
987  } while (++mNIters[mCurHyp] < mMaxIter);
988  //
989  mChi2[mCurHyp] = chi2 * NInv;
