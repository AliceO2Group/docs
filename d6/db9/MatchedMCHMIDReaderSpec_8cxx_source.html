54void MatchMCHMIDReader::init(InitContext& ic)
55{
56 mFileName = o2::utils::Str::concat_string(o2::utils::Str::rectifyDirectory(ic.options().get<std::string>("input-dir")),

61void MatchMCHMIDReader::run(ProcessingContext& pc)
62{
63 auto ent = mTree->GetReadEntry() + 1;
64 assert(ent < mTree->GetEntries()); // this should not happen
65 mTree->GetEntry(ent);
66 LOG(info) << "Pushing " << mTracks.size() << " MCHMID matches at entry " << ent;
67
68 pc.outputs().snapshot(OutputRef{"muontracks"}, mTracks);
69 if (mUseMC) {
70 pc.outputs().snapshot(OutputRef{"muontracklabels"}, mLabels);
71 }
72
73 if (mTree->GetReadEntry() + 1 >= mTree->GetEntries()) {
74 pc.services().get<ControlService>().endOfStream();
75 pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
76 }
77}

79void MatchMCHMIDReader::connectTree(const std::string& filename)
80{
81 mTree.reset(nullptr); // in case it was already loaded
82 mFile.reset(TFile::Open(filename.c_str()));
83 assert(mFile && !mFile->IsZombie());
84 mTree.reset((TTree*)mFile->Get("o2sim"));
85 assert(mTree);
86 mTree->SetBranchAddress("tracks", &mTracksPtr);
87 if (mUseMC) {
88 mTree->SetBranchAddress("tracklabels", &mLabelsPtr);
89 }
90 LOG(info) << "Loaded tree from " << filename << " with " << mTree->GetEntries() << " entries";
91}

93DataProcessorSpec getMCHMIDMatchedReaderSpec(bool useMC)
94{
95 std::vector<OutputSpec> outputs;
96 outputs.emplace_back(OutputSpec{{"muontracks"}, "GLO", "MTC_MCHMID", 0, Lifetime::Timeframe});
97 if (useMC) {
98 outputs.emplace_back(OutputSpec{{"muontracklabels"}, "GLO", "MCMTC_MCHMID", 0, Lifetime::Timeframe});
99 }
100
101 return DataProcessorSpec{
102 "mchmid-matches-reader",
103 Inputs{},
104 outputs,
105 AlgorithmSpec{adaptFromTask<MatchMCHMIDReader>(useMC)},
106 Options{
107 {"matchmchmid-track-infile", VariantType::String, "muontracks.root", {"Name of the input file"}},
108 {"input-dir", VariantType::String, "none", {"Input directory"}}}};
109}
