33DsDetId decodeDsDetId(uint32_t x)
34{
35 uint16_t deId = static_cast<uint16_t>((x & 0xFFFF0000) >> 16);
36 uint16_t dsId = static_cast<uint16_t>(x & 0xFFFF);
37 return DsDetId(deId, dsId);
38}
