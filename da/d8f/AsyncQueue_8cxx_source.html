38 auto AsyncQueueHelpers::run(AsyncQueue& queue, TimesliceId oldestPossible) -> void
39 {
40  if (queue.tasks.empty()) {
41  return;
42  }
43  O2_SIGNPOST_ID_GENERATE(opid, async_queue);
44  O2_SIGNPOST_START(async_queue, opid, "run", "Attempting at running %zu tasks with oldestPossible timeframe %zu", queue.tasks.size(), oldestPossible.value);
45  std::vector<int> order;
46  order.resize(queue.tasks.size());
47  std::iota(order.begin(), order.end(), 0);
48  // Decide wether or not they can run as a first thing
49  for (auto& task : queue.tasks) {
50  if (task.timeslice.value <= oldestPossible.value) {
51  task.runnable = true;
52  }
53  O2_SIGNPOST_EVENT_EMIT(async_queue, opid, "run",
54  "Task %d (timeslice %zu), score %d, debounce %d is %{public}s when oldestPossible timeframe is %zu",
55  task.id.value, task.timeslice.value, queue.prototypes[task.id.value].score, task.debounce,
56  task.runnable ? "runnable" : "not runnable", oldestPossible.value);
57  }
58 
59  // Sort by runnable, timeslice, then priority and finally debounce
60  std::sort(order.begin(), order.end(), [&queue](int a, int b) {
61  if (queue.tasks[a].runnable && !queue.tasks[b].runnable) {
62  return true;
63  }
64  if (!queue.tasks[a].runnable && queue.tasks[b].runnable) {
65  return false;
66  }
67  if (queue.tasks[a].timeslice.value == queue.tasks[b].timeslice.value) {
68  if (queue.tasks[a].id.value == -1 || queue.tasks[b].id.value == -1) {
69  return false;
70  }
71  if (queue.tasks[a].id.value == queue.tasks[b].id.value) {
72  return queue.tasks[a].debounce > queue.tasks[b].debounce;
73  }
74  return queue.prototypes[queue.tasks[a].id.value].score > queue.prototypes[queue.tasks[b].id.value].score;
75  } else {
76  return queue.tasks[a].timeslice.value > queue.tasks[b].timeslice.value;
77  }
78  });
79 
80  for (auto i : order) {
81  if (queue.tasks[i].runnable) {
82  O2_SIGNPOST_EVENT_EMIT(async_queue, opid, "run", "Running task %d (%d), (timeslice %zu), score %d, debounce %d", queue.tasks[i].id.value, i, queue.tasks[i].timeslice.value, queue.prototypes[queue.tasks[i].id.value].score, queue.tasks[i].debounce);
83  } else {
84  O2_SIGNPOST_EVENT_EMIT(async_queue, opid, "run", "Skipping task %d (%d) (timeslice %zu), score %d, debounce %d", queue.tasks[i].id.value, i, queue.tasks[i].timeslice.value, queue.prototypes[queue.tasks[i].id.value].score, queue.tasks[i].debounce);
85  }
86  }
87  // Keep only the tasks with the highest debounce value for a given id
88  auto newEnd = std::unique(order.begin(), order.end(), [&queue](int a, int b) {
89  return queue.tasks[a].runnable == queue.tasks[b].runnable && queue.tasks[a].id.value == queue.tasks[b].id.value && queue.tasks[a].debounce >= 0 && queue.tasks[b].debounce >= 0;
90  });
91  for (auto ii = newEnd; ii != order.end(); ii++) {
92  O2_SIGNPOST_EVENT_EMIT(async_queue, opid, "dropping", "Dropping task %d for timeslice %zu", queue.tasks[*ii].id.value, queue.tasks[*ii].timeslice.value);
93  }
94  order.erase(newEnd, order.end());
95 
96  if (order.empty() && queue.tasks.size() > 0) {
97  O2_SIGNPOST_END(async_queue, opid, "run", "Not running iteration %zu pending %zu.",
98  queue.iteration, queue.tasks.size());
99  return;
100  } else if (order.empty()) {
101  O2_SIGNPOST_END(async_queue, opid, "run", "Not running iteration %zu. No tasks.", queue.iteration);
102  return;
103  }
104  O2_SIGNPOST_EVENT_EMIT(async_queue, opid, "run", "Running %zu tasks in iteration %zu", order.size(), queue.iteration);
105 
106  int runCount = 0;
107  for (auto i : order) {
108  if (queue.tasks[i].runnable) {
109  runCount++;
110  // If a task is runable, we can run the task and remove it from the queue
111  O2_SIGNPOST_EVENT_EMIT(async_queue, opid, "run", "Running task %{public}s (%d) for timeslice %zu",
112  queue.prototypes[queue.tasks[i].id.value].name.c_str(), i,
113  queue.tasks[i].timeslice.value);
114  queue.tasks[i].task(opid.value);
115  O2_SIGNPOST_EVENT_EMIT(async_queue, opid, "run", "Done running %d", i);
116  }
117  }
118  // Remove all runnable tasks regardless they actually
119  // ran or they were skipped due to debouncing.
120  queue.tasks.erase(std::remove_if(queue.tasks.begin(), queue.tasks.end(), [&queue](AsyncTask const& task) {
121  return task.runnable;
122  }),
123  queue.tasks.end());
124  O2_SIGNPOST_END(async_queue, opid, "run", "Done running %d/%zu tasks", runCount, order.size());
125 }
