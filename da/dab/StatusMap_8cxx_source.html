62 std::map<int, std::vector<int>> applyMask(const o2::mch::StatusMap& statusMap,
63  uint32_t mask)
64 {
65  std::map<int, std::vector<int>> rejectList;
66 
67  if (mask == 0) {
68  return rejectList;
69  }
70 
71  for (const auto& status : statusMap) {
72  auto channel = status.first;
73  if (!channel.isValid()) {
74  continue;
75  }
76  if ((mask & status.second) != 0) {
77  rejectList[channel.getDeId()].emplace_back(channel.getDePadIndex());
78  }
79  }
80  return rejectList;
81 }
