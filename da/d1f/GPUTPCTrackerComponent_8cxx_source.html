134int32_t GPUTPCTrackerComponent::ReadConfigurationString(const char* arguments)
135{
136 // Set configuration parameters for the CA tracker component from the string
137
138 int32_t iResult = 0;
139 if (!arguments) {
140 return iResult;
141 }
142
143 TString allArgs = arguments;
144 TString argument;
145 int32_t bMissingParam = 0;
146
147 TObjArray* pTokens = allArgs.Tokenize(" ");
148
149 int32_t nArgs = pTokens ? pTokens->GetEntries() : 0;
150
151 for (int32_t i = 0; i < nArgs; i++) {
152 argument = ((TObjString*)pTokens->At(i))->GetString();
153 if (argument.IsNull()) {
154 continue;
155 }
156
157 if (argument.CompareTo("-solenoidBz") == 0) {
158 if ((bMissingParam = (++i >= pTokens->GetEntries()))) {
159 break;
160 }
161 HLTWarning("argument -solenoidBz is deprecated, magnetic field set up globally (%f)", GetBz());
162 continue;
163 }
164
165 if (argument.CompareTo("-minNClustersOnTrack") == 0) {
166 if ((bMissingParam = (++i >= pTokens->GetEntries()))) {
167 break;
168 }
169 fMinNTrackClusters = ((TObjString*)pTokens->At(i))->GetString().Atoi();
170 HLTInfo("minNClustersOnTrack set to: %d", fMinNTrackClusters);
171 continue;
172 }
173
174 if (argument.CompareTo("-minTrackPt") == 0) {
175 if ((bMissingParam = (++i >= pTokens->GetEntries()))) {
176 break;
177 }
178 fMinTrackPt = ((TObjString*)pTokens->At(i))->GetString().Atof();
179 HLTInfo("minTrackPt set to: %f", fMinTrackPt);
180 continue;
181 }
182
183 if (argument.CompareTo("-clusterZCut") == 0) {
184 if ((bMissingParam = (++i >= pTokens->GetEntries()))) {
185 break;
186 }
187 fClusterZCut = TMath::Abs(((TObjString*)pTokens->At(i))->GetString().Atof());
188 HLTInfo("ClusterZCut set to: %f", fClusterZCut);
189 continue;
190 }
191
192 if (argument.CompareTo("-neighboursSearchArea") == 0) {
193 if ((bMissingParam = (++i >= pTokens->GetEntries()))) {
194 break;
195 }
196 mNeighboursSearchArea = TMath::Abs(((TObjString*)pTokens->At(i))->GetString().Atof());
197 HLTInfo("NeighboursSearchArea set to: %f", mNeighboursSearchArea);
198 continue;
199 }
200
201 if (argument.CompareTo("-errorCorrectionY") == 0) {
202 if ((bMissingParam = (++i >= pTokens->GetEntries()))) {
203 break;
204 }
205 fClusterErrorCorrectionY = ((TObjString*)pTokens->At(i))->GetString().Atof();
206 HLTInfo("Cluster Y error correction factor set to: %f", fClusterErrorCorrectionY);
207 continue;
208 }
209
210 if (argument.CompareTo("-errorCorrectionZ") == 0) {
211 if ((bMissingParam = (++i >= pTokens->GetEntries()))) {
212 break;
213 }
214 fClusterErrorCorrectionZ = ((TObjString*)pTokens->At(i))->GetString().Atof();
215 HLTInfo("Cluster Z error correction factor set to: %f", fClusterErrorCorrectionZ);
216 continue;
217 }
218
219 if (argument.CompareTo("-allowGPU") == 0) {
220 fAllowGPU = 1;
221 HLTInfo("Will try to run tracker on GPU");
222 continue;
223 }
224
225 if (argument.CompareTo("-GlobalTracking") == 0) {
226 fGlobalTracking = 1;
227 HLTInfo("Global Tracking Activated");
228 continue;
229 }
230
231 if (argument.CompareTo("-GPUHelperThreads") == 0) {
232 if ((bMissingParam = (++i >= pTokens->GetEntries()))) {
233 break;
234 }
235 fGPUHelperThreads = ((TObjString*)pTokens->At(i))->GetString().Atoi();
236 HLTInfo("Number of GPU Helper Threads set to: %d", fGPUHelperThreads);
237 continue;
238 }
239
240 if (argument.CompareTo("-CPUTrackers") == 0) {
241 if ((bMissingParam = (++i >= pTokens->GetEntries()))) {
242 break;
243 }
244 fCPUTrackers = ((TObjString*)pTokens->At(i))->GetString().Atoi();
245 HLTInfo("Number of CPU Trackers set to: %d", fCPUTrackers);
246 continue;
247 }
248
249 if (argument.CompareTo("-SearchWindowDZDR") == 0) {
250 if ((bMissingParam = (++i >= pTokens->GetEntries()))) {
251 break;
252 }
253 fSearchWindowDZDR = ((TObjString*)pTokens->At(i))->GetString().Atof();
254 HLTInfo("Search Window DZDR set to: %f", fSearchWindowDZDR);
255 continue;
256 }
257
258 if (argument.CompareTo("-GPUDeviceNum") == 0) {
259 if ((bMissingParam = (++i >= pTokens->GetEntries()))) {
260 break;
261 }
262 fGPUDeviceNum = ((TObjString*)pTokens->At(i))->GetString().Atoi();
263 HLTInfo("Using GPU Device Number %d", fGPUDeviceNum);
264 continue;
265 }
266
267 if (argument.CompareTo("-GPUType") == 0) {
268 if ((bMissingParam = (++i >= pTokens->GetEntries()))) {
269 break;
270 }
271 fGPUType = ((TObjString*)pTokens->At(i))->GetString();
272 continue;
273 }
274
275 if (argument.CompareTo("-GPUStuckProtection") == 0) {
276 if ((bMissingParam = (++i >= pTokens->GetEntries()))) {
277 break;
278 }
279 fGPUStuckProtection = ((TObjString*)pTokens->At(i))->GetString().Atoi();
280 continue;
281 }
282
283 if (argument.CompareTo("-AsyncGPUStuckProtection") == 0) {
284 if ((bMissingParam = (++i >= pTokens->GetEntries()))) {
285 break;
286 }
287 fAsync = ((TObjString*)pTokens->At(i))->GetString().Atoi();
288 continue;
289 }
290
291 HLTError("Unknown option \"%s\"", argument.Data());
292 iResult = -EINVAL;
293 }
294 delete pTokens;
295
296 if (bMissingParam) {
297 HLTError("Specifier missed for parameter \"%s\"", argument.Data());
298 iResult = -EINVAL;
299 }
300
301 return iResult;
302}

304int32_t GPUTPCTrackerComponent::ReadCDBEntry(const char* cdbEntry, const char* chainId)
305{
306 // see header file for class documentation
307
308 const char* defaultNotify = "";
309
310 if (!cdbEntry) {
311 cdbEntry = "HLT/ConfigTPC/TPCCATracker";
312 defaultNotify = " (default)";
313 chainId = 0;
314 }
315
316 HLTInfo("configure from entry \"%s\"%s, chain id %s", cdbEntry, defaultNotify, (chainId != nullptr && chainId[0] != 0) ? chainId : "<none>");
317 AliCDBEntry* pEntry = AliCDBManager::Instance()->Get(cdbEntry); //,GetRunNo());
318
319 if (!pEntry) {
320 HLTError("cannot fetch object \"%s\" from CDB", cdbEntry);
321 return -EINVAL;
322 }
323
324 TObjString* pString = dynamic_cast<TObjString*>(pEntry->GetObject());
325
326 if (!pString) {
327 HLTError("configuration object \"%s\" has wrong type, required TObjString", cdbEntry);
328 return -EINVAL;
329 }
330
331 HLTInfo("received configuration object string: \"%s\"", pString->GetString().Data());
332
333 return ReadConfigurationString(pString->GetString().Data());
334}

336int32_t GPUTPCTrackerComponent::Configure(const char* cdbEntry, const char* chainId, const char* commandLine)
337{
338 // Configure the component
339 // There are few levels of configuration,
340 // parameters which are set on one step can be overwritten on the next step
341
342 //* read hard-coded values
343 SetDefaultConfiguration();
344
345 //* read the default CDB entry
346 int32_t iResult1 = ReadCDBEntry(nullptr, chainId);
347
348 //* read magnetic field
349 fSolenoidBz = GetBz();
350
351 //* read the actual CDB entry if required
352 int32_t iResult2 = (cdbEntry) ? ReadCDBEntry(cdbEntry, chainId) : 0;
353
354 //* read extra parameters from input (if they are)
355 int32_t iResult3 = 0;
356
357 if (commandLine && commandLine[0] != '\0') {
358 HLTInfo("received configuration string from HLT framework: \"%s\"", commandLine);
359 iResult3 = ReadConfigurationString(commandLine);
360 }
361
362 if (fRec) {
363 ConfigureSlices();
364 }
365
366 return iResult1 ? iResult1 : (iResult2 ? iResult2 : iResult3);
367}

369int32_t GPUTPCTrackerComponent::ConfigureSlices()
370{
371 // Initialize the tracker slices
372 GPUSettingsRec rec;
373 GPUSettingsGRP grp;
374 GPUSettingsProcessing devProc;
375
376 grp.solenoidBzNominalGPU = fSolenoidBz;
377 grp.continuousMaxTimeBin = 0; // triggered events
378 if (mNeighboursSearchArea > 0) {
379 rec.tpc.neighboursSearchArea = mNeighboursSearchArea;
380 }
381 if (fClusterErrorCorrectionY > 1.e-4) {
382 rec.tpc.clusterError2CorrectionY = fClusterErrorCorrectionY * fClusterErrorCorrectionY;
383 }
384 if (fClusterErrorCorrectionZ > 1.e-4) {
385 rec.tpc.clusterError2CorrectionZ = fClusterErrorCorrectionZ * fClusterErrorCorrectionZ;
386 }
387 rec.tpc.minNClustersTrackSeed = fMinNTrackClusters;
388 rec.SetMinTrackPtB5(fMinTrackPt * fabsf(fSolenoidBz / 5));
389 rec.tpc.searchWindowDZDR = fSearchWindowDZDR;
390 devProc.nDeviceHelperThreads = fGPUHelperThreads;
391 rec.tpc.globalTracking = fGlobalTracking;
392 devProc.stuckProtection = fGPUStuckProtection;
393 rec.nonConsecutiveIDs = true;
394 rec.tpc.mergerReadFromTrackerDirectly = false;
395 devProc.ompThreads = 1;
396 devProc.ompKernels = false;
397
398 GPURecoStepConfiguration steps;
399 steps.steps.set(GPUDataTypes::RecoStep::TPCSliceTracking);
400 steps.inputs.set(GPUDataTypes::InOutType::TPCClusters);
401 steps.outputs.set(GPUDataTypes::InOutType::TPCSectorTracks);
402
403 fRec->SetSettings(&grp, &rec, &devProc, &steps);
404 fChain->LoadClusterErrors();
405 return fRec->Init();
406}

423int32_t GPUTPCTrackerComponent::DoInit(int argc, const char** argv)
424{
425 if (fRec) {
426 return EINPROGRESS;
427 }
428
429 // Configure the CA tracker component
430 TString arguments = "";
431 for (int32_t i = 0; i < argc; i++) {
432 if (!arguments.IsNull()) {
433 arguments += " ";
434 }
435 arguments += argv[i];
436 }
437
438 int32_t retVal = Configure(nullptr, nullptr, arguments.Data());
439 if (retVal == 0) {
440 if (fAsync) {
441 if (fAsyncProcessor.Initialize(1)) {
442 return (-ENODEV);
443 }
444 void* initRetVal;
445 if (fAsyncProcessor.InitializeAsyncMemberTask(this, &GPUTPCTrackerComponent::TrackerInit, nullptr, &initRetVal) != 0) {
446 return (-ENODEV);
447 }
448 if (initRetVal) {
449 return (-ENODEV);
450 }
451 } else {
452 if (TrackerInit(nullptr) != nullptr) {
453 return (-ENODEV);
454 }
455 }
456 }
457
458 return (retVal);
459}

489int32_t GPUTPCTrackerComponent::DoEvent(const AliHLTComponentEventData& evtData, const AliHLTComponentBlockData* blocks, AliHLTComponentTriggerData& /*trigData*/, AliHLTUInt8_t* outputPtr, AliHLTUInt32_t& size, vector<AliHLTComponentBlockData>& outputBlocks)
490{
491 //* process event
492 if (!fRec) {
493 HLTError("CATracker not initialized properly");
494 return -ENOENT;
495 }
496
497 AliHLTTPCTrackerWrapperData tmpPar;
498 tmpPar.fEvtData = &evtData;
499 tmpPar.fBlocks = blocks;
500 tmpPar.mOutputPtr = outputPtr;
501 tmpPar.fSize = &size;
502 tmpPar.mOutputBlocks = &outputBlocks;
503
504 static int32_t trackerTimeout = 0;
505 if (trackerTimeout) {
506 size = 0;
507 return (0);
508 }
509
510 int32_t retVal;
511 if (fAsync) {
512 void* asyncRetVal = nullptr;
513 if (fAsyncProcessor.InitializeAsyncMemberTask(this, &GPUTPCTrackerComponent::TrackerDoEvent, &tmpPar, &asyncRetVal, fAsync) != 0) {
514 HLTError("Tracking timed out, disabling this tracker instance");
515 trackerTimeout = 1;
516 size = 0;
517 return (-ENODEV);
518 } else {
519 retVal = (int32_t)(size_t)asyncRetVal;
520 }
521 } else {
522 retVal = (int32_t)(size_t)TrackerDoEvent(&tmpPar);
523 }
524 return (retVal);
525}

527void* GPUTPCTrackerComponent::TrackerDoEvent(void* par)
528{
529 AliHLTTPCTrackerWrapperData* tmpPar = reinterpret_cast<AliHLTTPCTrackerWrapperData*>(par);
530
531 const AliHLTComponentEventData& evtData = *(tmpPar->fEvtData);
532 const AliHLTComponentBlockData* blocks = tmpPar->fBlocks;
533 AliHLTUInt8_t* outputPtr = tmpPar->mOutputPtr;
534 AliHLTUInt32_t& size = *(tmpPar->fSize);
535 vector<AliHLTComponentBlockData>& outputBlocks = *(tmpPar->mOutputBlocks);
536
537 AliHLTUInt32_t maxBufferSize = size;
538 size = 0; // output size
539
540 if (GetFirstInputBlock(kAliHLTDataTypeSOR) || GetFirstInputBlock(kAliHLTDataTypeEOR)) {
541 return 0;
542 }
543
544 fBenchmark.StartNewEvent();
545 fBenchmark.Start(0);
546
547 // Logging( kHLTLogWarning, "HLT::TPCCATracker::DoEvent", "DoEvent", "CA::DoEvent()" );
548 if (evtData.fBlockCnt <= 0) {
549 HLTWarning("no blocks in event");
550 return 0;
551 }
552
553 // Prepare everything for all slices
554 const AliHLTTPCClusterXYZData* clustersXYZ[NSLICES][fgkNPatches] = {nullptr};
555 const AliHLTTPCRawClusterData* clustersRaw[NSLICES][fgkNPatches] = {nullptr};
556
557 for (uint64_t ndx = 0; ndx < evtData.fBlockCnt; ndx++) {
558 const AliHLTComponentBlockData& pBlock = blocks[ndx];
559 int32_t slice = AliHLTTPCDefinitions::GetMinSliceNr(pBlock);
560 int32_t patch = AliHLTTPCDefinitions::GetMinPatchNr(pBlock);
561 if (pBlock.fDataType == AliHLTTPCDefinitions::RawClustersDataType()) {
562 clustersRaw[slice][patch] = (const AliHLTTPCRawClusterData*)pBlock.fPtr;
563 } else if (pBlock.fDataType == AliHLTTPCDefinitions::ClustersXYZDataType()) {
564 clustersXYZ[slice][patch] = (const AliHLTTPCClusterXYZData*)pBlock.fPtr;
565 }
566 }
567
568 GPUTPCClusterData* clusterData[NSLICES] = {nullptr};
569 int32_t nClusters[NSLICES] = {0};
570
571 int32_t nClustersTotal = 0;
572 for (int32_t slice = 0; slice < NSLICES; slice++) {
573 int32_t nClustersSliceTotal = 0;
574 for (int32_t patch = 0; patch < 6; patch++) {
575 if (clustersXYZ[slice][patch]) {
576 nClustersSliceTotal += clustersXYZ[slice][patch]->fCount;
577 }
578 }
579 if (nClustersSliceTotal > 500000) {
580 HLTWarning("Too many clusters in tracker input: Slice %d, Number of Clusters %d, slice not included in tracking", slice, nClustersSliceTotal);
581 nClusters[slice] = nClustersSliceTotal;
582 } else if (nClustersSliceTotal == 0) {
583 nClusters[slice] = nClustersSliceTotal;
584 } else {
585 clusterData[slice] = new GPUTPCClusterData[nClustersSliceTotal];
586 nClusters[slice] = nClustersSliceTotal;
587 GPUTPCClusterData* pCluster = clusterData[slice];
588 for (int32_t patch = 0; patch < 6; patch++) {
589 if (clustersXYZ[slice][patch] != nullptr && clustersRaw[slice][patch] != nullptr) {
590 const AliHLTTPCClusterXYZData& clXYZ = *clustersXYZ[slice][patch];
591 const AliHLTTPCRawClusterData& clRaw = *clustersRaw[slice][patch];
592
593 if (clXYZ.fCount != clRaw.fCount) {
594 HLTError("Number of entries in raw and xyz clusters are not mached %d vs %d", clXYZ.fCount, clRaw.fCount);
595 continue;
596 }
597
598 const int32_t firstRow = AliHLTTPCGeometry::GetFirstRow(patch);
599 for (int32_t ic = 0; ic < clXYZ.fCount; ic++) {
600 const AliHLTTPCClusterXYZ& c = clXYZ.fClusters[ic];
601 const AliHLTTPCRawCluster& cRaw = clRaw.fClusters[ic];
602 if (c.GetZ() > fClusterZCut || c.GetZ() < -fClusterZCut) {
603 continue;
604 }
605 if (c.GetX() < 1.f) {
606 continue; // cluster xyz position was not calculated for whatever reason
607 }
608 pCluster->id = AliHLTTPCGeometry::CreateClusterID(slice, patch, ic);
609 pCluster->x = c.GetX();
610 pCluster->y = c.GetY();
611 pCluster->z = c.GetZ();
612 pCluster->row = firstRow + cRaw.GetPadRow();
613 pCluster->flags = cRaw.GetFlags();
614 if (cRaw.GetSigmaPad2() < kAlmost0 || cRaw.GetSigmaTime2() < kAlmost0) {
615 pCluster->flags |= GPUTPCGMMergedTrackHit::flagSingle;
616 }
617 pCluster->amp = cRaw.GetCharge();
618#ifdef GPUCA_FULL_CLUSTERDATA
619 pCluster->pad = cRaw.GetPad();
620 pCluster->time = cRaw.GetTime();
621 pCluster->ampMax = cRaw.GetQMax();
622 pCluster->sigmaPad2 = cRaw.GetSigmaPad2();
623 pCluster->sigmaTime2 = cRaw.GetSigmaTime2();
624#endif
625 pCluster++;
626 }
627 }
628 }
629 nClusters[slice] = pCluster - clusterData[slice];
630 nClustersTotal += nClusters[slice];
631 HLTDebug("Read %d->%d hits for slice %d", nClustersSliceTotal, nClusters[slice], slice);
632 }
633 }
634
635 if (nClustersTotal == 0) {
636 // No input, skip processing
637 fBenchmark.Stop(0);
638 return (0);
639 }
640
641 fChain->ClearIOPointers();
642 for (int32_t i = 0; i < NSLICES; i++) {
643 fChain->mIOPtrs.clusterData[i] = clusterData[i];
644 fChain->mIOPtrs.nClusterData[i] = nClusters[i];
645 }
646
647 // Prepare Output
648 fRec->SetOutputControl(outputPtr, maxBufferSize);
649
650 // reconstruct the event
651 fBenchmark.Start(1);
652 try {
653 fRec->PrepareEvent();
