33template <int NLayers>
34float Vertexer<NLayers>::clustersToVertices(LogFunc logger)
35{
36 LogFunc evalLog = [](const std::string&) {};
37
38 if (mTimeFrame->hasMCinformation() && mVertParams[0].useTruthSeeding) {
39 float t = evaluateTask(&Vertexer::addTruthSeeds, StateNames[mCurStep = TruthSeeding], 0, evalLog);
40 sortVertices();
41 ++mTimeFrameCounter;
42 return t;
43 }
44
45 TrackingParameters trkPars;
46 mTraits->updateVertexingParameters(mVertParams);
47
48 auto handleException = [&](const auto& err) {
49 LOGP(error, "Encountered critical error in step {}, stopping further processing of this TF: {}", StateNames[mCurStep], err.what());
50 if (!mVertParams[0].DropTFUponFailure) {
51 throw err;
52 } else {
53 LOGP(error, "Dropping this TF!");
54 }
55 };
56
57 float timeTracklet{0.f}, timeSelection{0.f}, timeVertexing{0.f}, timeInit{0.f};
58 bool completed = false;
59 try {
60 for (int iteration = 0; iteration < (int)mVertParams.size(); ++iteration) {
61 mMemoryPool->setMaxMemory(mVertParams[iteration].MaxMemory);
62 unsigned int nTracklets01{0}, nTracklets12{0};
63 logger(fmt::format("=== ITS {} Seeding vertexer iteration {} summary:", mTraits->getName(), iteration));
64 trkPars.PhiBins = mTraits->getVertexingParameters()[0].PhiBins;
65 trkPars.ZBins = mTraits->getVertexingParameters()[0].ZBins;
66 auto timeInitIteration = evaluateTask(&Vertexer::initialiseVertexer, StateNames[mCurStep = Init], iteration, evalLog, trkPars, iteration);
67 auto timeTrackletIteration = evaluateTask(&Vertexer::findTracklets, StateNames[mCurStep = Trackleting], iteration, evalLog, iteration);
68 nTracklets01 = mTimeFrame->getTotalTrackletsTF(0);
69 nTracklets12 = mTimeFrame->getTotalTrackletsTF(1);
70 auto timeSelectionIteration = evaluateTask(&Vertexer::validateTracklets, StateNames[mCurStep = Selection], iteration, evalLog, iteration);
71 const auto nVerticesBefore = mTimeFrame->getPrimaryVertices().size();
72 auto timeVertexingIteration = evaluateTask(&Vertexer::findVertices, StateNames[mCurStep = Finding], iteration, evalLog, iteration);
73 const auto nVerticesAfter = mTimeFrame->getPrimaryVertices().size();
74 printEpilog(logger, nTracklets01, nTracklets12, mTimeFrame->getNLinesTotal(), nVerticesAfter - nVerticesBefore, nVerticesAfter, timeInitIteration, timeTrackletIteration, timeSelectionIteration, timeVertexingIteration);
75 timeInit += timeInitIteration;
76 timeTracklet += timeTrackletIteration;
77 timeSelection += timeSelectionIteration;
78 timeVertexing += timeVertexingIteration;
79
80 // update LUT with all currently found vertices so in second iteration we can check vertPerROFThreshold
81 sortVertices();
82 mTimeFrame->updateROFVertexLookupTable();
83 }
84 completed = true;
85 } catch (const BoundedMemoryResource::MemoryLimitExceeded& err) {
86 handleException(err);
87 } catch (const std::bad_alloc& err) {
88 handleException(err);
89 } catch (...) {
90 LOGP(fatal, "Uncaught exception!");
91 }
92
93 if (completed) {
94 ++mTimeFrameCounter;
95 }
96
97 return timeInit + timeTracklet + timeSelection + timeVertexing;
98}

100template <int NLayers>
101void Vertexer<NLayers>::sortVertices()
102{
103 auto& pvs = mTimeFrame->getPrimaryVertices();
104 bounded_vector<size_t> indices(pvs.size(), mMemoryPool.get());
105 std::iota(indices.begin(), indices.end(), 0);
106 // provide vertices sorted by lower-bound
107 std::sort(indices.begin(), indices.end(), [&pvs](size_t i, size_t j) {
108 const auto& a = pvs[i].getTimeStamp();
109 const auto& b = pvs[j].getTimeStamp();
110 const auto aLower = a.lower();
111 const auto bLower = b.lower();
112 if (aLower != bLower) {
113 return aLower < bLower;
114 }
115 return pvs[i].getNContributors() > pvs[j].getNContributors();
116 });
117 bounded_vector<Vertex> sortedVtx(mMemoryPool.get());
118 sortedVtx.reserve(pvs.size());
119 for (const size_t idx : indices) {
120 sortedVtx.push_back(pvs[idx]);
121 }
122 pvs.swap(sortedVtx);
123 if (mTimeFrame->hasMCinformation()) {
124 auto& mc = mTimeFrame->getPrimaryVerticesLabels();
125 bounded_vector<VertexLabel> sortedMC(mMemoryPool.get());
126 for (const size_t idx : indices) {
127 sortedMC.push_back(mc[idx]);
128 }
129 mc.swap(sortedMC);
130 }
131}

152template <int NLayers>
153void Vertexer<NLayers>::printSummary() const
154{
155 LOGP(info, "Vertexer summary: Processed {} TFs", mTimeFrameCounter);
156 for (size_t iteration = 0; iteration < mTimingStats.size(); ++iteration) {
157 for (size_t state = 0; state < NSteps; ++state) {
158 const auto& stats = mTimingStats[iteration][state];
159 if (!stats.calls) {
160 continue;
161 }
162 LOGP(info, " - iter {} {}: calls={} total={:.2f} ms avg={:.2f} ms", iteration, StateNames[state], stats.calls, stats.totalTimeMs, stats.averageTimeMs());
163 }
164 }
