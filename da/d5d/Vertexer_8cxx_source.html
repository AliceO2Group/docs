39 float Vertexer::clustersToVertices(std::function<void(std::string s)> logger)
40 {
41  float total{0.f};
42  TrackingParameters trkPars;
43  trkPars.PhiBins = mTraits->getVertexingParameters().PhiBins;
44  trkPars.ZBins = mTraits->getVertexingParameters().ZBins;
45  total += evaluateTask(&Vertexer::initialiseVertexer, "Vertexer initialisation", logger, trkPars);
46  total += evaluateTask(&Vertexer::findTracklets, "Vertexer tracklet finding", logger);
47  total += evaluateTask(&Vertexer::validateTracklets, "Vertexer adjacent tracklets validation", logger);
48  total += evaluateTask(&Vertexer::findVertices, "Vertexer vertex finding", logger);
49  printEpilog(logger, total);
50  return total;
51 }
