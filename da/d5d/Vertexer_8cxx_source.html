38template <int nLayers>
39float Vertexer<nLayers>::clustersToVertices(LogFunc logger)
40{
41 LogFunc evalLog = [](const std::string&) {};
42
43 if (mTimeFrame->hasMCinformation() && mVertParams[0].useTruthSeeding) {
44 return evaluateTask(&Vertexer::addTruthSeeds, StateNames[mCurState = TruthSeeding], 0, evalLog);
45 }
46
47 TrackingParameters trkPars;
48 TimeFrameGPUParameters tfGPUpar;
49 mTraits->updateVertexingParameters(mVertParams, tfGPUpar);
50
51 auto handleException = [&](const auto& err) {
52 LOGP(error, "Encountered critical error in step {}, stopping further processing of this TF: {}", StateNames[mCurState], err.what());
53 if (!mVertParams[0].DropTFUponFailure) {
54 throw err;
55 } else {
56 LOGP(error, "Dropping this TF!");
57 }
58 };
59
60 float timeTracklet{0.f}, timeSelection{0.f}, timeVertexing{0.f}, timeInit{0.f};
61 try {
62 for (int iteration = 0; iteration < std::min(mVertParams[0].nIterations, (int)mVertParams.size()); ++iteration) {
63 mMemoryPool->setMaxMemory(mVertParams[iteration].MaxMemory);
64 unsigned int nTracklets01{0}, nTracklets12{0};
65 logger(fmt::format("=== ITS {} Seeding vertexer iteration {} summary:", mTraits->getName(), iteration));
66 trkPars.PhiBins = mTraits->getVertexingParameters()[0].PhiBins;
67 trkPars.ZBins = mTraits->getVertexingParameters()[0].ZBins;
68 auto timeInitIteration = evaluateTask(&Vertexer::initialiseVertexer, StateNames[mCurState = Init], iteration, evalLog, trkPars, iteration);
69 auto timeTrackletIteration = evaluateTask(&Vertexer::findTracklets, StateNames[mCurState = Trackleting], iteration, evalLog, iteration);
70 nTracklets01 = mTimeFrame->getTotalTrackletsTF(0);
71 nTracklets12 = mTimeFrame->getTotalTrackletsTF(1);
72 auto timeSelectionIteration = evaluateTask(&Vertexer::validateTracklets, StateNames[mCurState = Validating], iteration, evalLog, iteration);
73 auto timeVertexingIteration = evaluateTask(&Vertexer::findVertices, StateNames[mCurState = Finding], iteration, evalLog, iteration);
74 printEpilog(logger, nTracklets01, nTracklets12, mTimeFrame->getNLinesTotal(), mTimeFrame->getTotVertIteration()[iteration], timeInitIteration, timeTrackletIteration, timeSelectionIteration, timeVertexingIteration);
75 timeInit += timeInitIteration;
76 timeTracklet += timeTrackletIteration;
77 timeSelection += timeSelectionIteration;
78 timeVertexing += timeVertexingIteration;
79 }
80 } catch (const BoundedMemoryResource::MemoryLimitExceeded& err) {
81 handleException(err);
82 } catch (const std::bad_alloc& err) {
83 handleException(err);
84 } catch (...) {
85 LOGP(fatal, "Uncaught exception!");
86 }
87
88 return timeInit + timeTracklet + timeSelection + timeVertexing;
89}
