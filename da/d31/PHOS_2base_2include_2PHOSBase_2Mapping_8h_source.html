87 // channel 2 -> human name M3-2, DDL 9
88 // channel 3 -> human name M3-3, DDL 10
89 if (iddl < 6) {
90 flp = 164;
91 if (iddl < 2) {
92 crorc = 304;
93 link = iddl;
94 } else {
95 crorc = 243;
96 link = iddl - 2;
97 }
98 } else {
99 flp = 165;
100 if (iddl < 10) {
101 crorc = 75;
102 } else {
103 crorc = 106;
104 }
105 link = (iddl - 6) % 4;
106 }
107 }
