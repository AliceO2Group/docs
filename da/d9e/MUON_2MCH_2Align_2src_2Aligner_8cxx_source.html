195 void Aligner::init(TString DataRecFName, TString ConsRecFName)
196 {
197 
199 
204  if (fInitialized) {
205  LOG(fatal) << "Millepede already initialized";
206  }
207 
208  if (!mRead) {
209  if (!fDisableRecordWriter) {
210  mRecordWriter->setCyclicAutoSave(mNEntriesAutoSave);
211  mRecordWriter->setDataFileName(DataRecFName);
212  fMillepede->SetRecordWriter(mRecordWriter);
213 
214  if (mWithConstraintsRecWriter) {
215  mConstraintsRecWriter->setCyclicAutoSave(mNEntriesAutoSave);
216  mConstraintsRecWriter->setDataFileName(ConsRecFName);
217  fMillepede->SetConstraintsRecWriter(mConstraintsRecWriter);
218  }
219  } else {
220  fMillepede->SetRecord(&fTrackRecord);
221  }
222 
223  } else {
224 
225  TChain* ch = new TChain(mRecordReader->getDataTreeName());
226  if (DataRecFName.EndsWith(".root")) {
227  ch->AddFile(DataRecFName);
228  }
229  int nent = ch->GetEntries();
230 
231  if (nent < 1) {
232  LOG(fatal) << "Obtained chain is empty, please check your record ROOT file.";
233  }
234 
235  mRecordReader->connectToChain(ch);
236  fMillepede->SetRecordReader(mRecordReader);
237 
238  if (mConstraintsRecReader) {
239 
240  TChain* ch_cons = new TChain(mConstraintsRecReader->getDataTreeName());
241  if (ConsRecFName.EndsWith(".root")) {
242  ch_cons->AddFile(ConsRecFName);
243  }
244  int nent_cons = ch_cons->GetEntries();
245 
246  if (nent_cons < 1) {
247  LOG(fatal) << "Obtained chain is empty, please check your record ROOT file.";
248  }
249 
250  mConstraintsRecReader->connectToChain(ch_cons);
251  fMillepede->SetConstraintsRecReader(mConstraintsRecReader);
252  }
253  }
254 
255  // assign proper groupID to free parameters
256  int nGlobal = 0;
257  for (int iPar = 0; iPar < fNGlobal; ++iPar) {
258 
259  if (fGlobalParameterStatus[iPar] == kFixedParId) {
260  // fixed parameters are left unchanged
261  continue;
262 
263  } else if (fGlobalParameterStatus[iPar] == kFreeParId || fGlobalParameterStatus[iPar] == kGroupBaseId) {
264 
265  // free parameters or first element of group are assigned a new group id
266  fGlobalParameterStatus[iPar] = nGlobal++;
267  continue;
268 
269  } else if (fGlobalParameterStatus[iPar] < kGroupBaseId) {
270 
271  // get detector element id from status, get chamber parameter id
272  const int iDeBase(kGroupBaseId - 1 - fGlobalParameterStatus[iPar]);
273  const int iParBase = iPar % fgNParCh;
274 
275  // check
276  if (iDeBase < 0 || iDeBase >= iPar / fgNParCh) {
277  LOG(fatal) << "Group for parameter index " << iPar << " has wrong base detector element: " << iDeBase;
278  }
279 
280  // assign identical group id to current
281  fGlobalParameterStatus[iPar] = fGlobalParameterStatus[iDeBase * fgNParCh + iParBase];
282  LOG(info) << "Parameter " << iPar << " grouped to detector " << iDeBase << " (" << GetParameterMaskString(1 << iParBase).Data() << ")";
283 
284  } else {
285  LOG(fatal) << "Unrecognized parameter status for index " << iPar << ": " << fGlobalParameterStatus[iPar];
286  }
287  }
288 
289  LOG(info) << "Free Parameters: " << nGlobal << " out of " << fNGlobal;
290 
291  // initialize millepedes
292  fMillepede->InitMille(fNGlobal, fNLocal, fNStdDev, fResCut, fResCutInitial, fGlobalParameterStatus);
293 
294  if (!mRead) {
295  if (!fDisableRecordWriter) {
296  mRecordWriter->init();
297  } else {
298  fMillepede->DisableRecordWriter();
299  }
300  }
301 
302  fInitialized = true;
303 
304  // some debug output
305  for (int iPar = 0; iPar < fgNParCh; ++iPar) {
306  LOG(info) << "fAllowVar[" << iPar << "]= " << fAllowVar[iPar];
307  }
308 
309  // set allowed variations for all parameters
310  for (int iDet = 0; iDet < fgNDetElem; ++iDet) {
311  for (int iPar = 0; iPar < fgNParCh; ++iPar) {
312  fMillepede->SetParSigma(iDet * fgNParCh + iPar, fAllowVar[iPar]);
313  }
314  }
315 
316  // Set iterations
317  if (fStartFac > 1) {
318  fMillepede->SetIterations(fStartFac);
319  }
320  // setup monitoring TFile
321  if (fDoEvaluation) {
322  // if (fDoEvaluation && fRefitStraightTracks) {
323  string Path_file = Form("%s%s", "Residual", ".root");
324 
325  fTFile = new TFile(Path_file.c_str(), "RECREATE");
326  fTTree = new TTree("TreeE", "Evaluation");
327 
328  const int kSplitlevel = 98;
329  const int kBufsize = 32000;
330 
331  fTrkClRes = new o2::mch::LocalTrackClusterResidual();
332  fTTree->Branch("fClDetElem", &(fTrkClRes->fClDetElem), "fClDetElem/I");
333  fTTree->Branch("fClDetElemNumber", &(fTrkClRes->fClDetElemNumber), "fClDetElemNumber/I");
334  fTTree->Branch("fClusterX", &(fTrkClRes->fClusterX), "fClusterX/F");
335  fTTree->Branch("fClusterY", &(fTrkClRes->fClusterY), "fClusterY/F");
336  fTTree->Branch("fTrackX", &(fTrkClRes->fTrackX), "fTrackX/F");
337  fTTree->Branch("fTrackY", &(fTrkClRes->fTrackY), "fTrackY/F");
338  fTTree->Branch("fClusterXloc", &(fTrkClRes->fClusterXloc), "fClusterXloc/F");
339  fTTree->Branch("fClusterYloc", &(fTrkClRes->fClusterYloc), "fClusterYloc/F");
340  fTTree->Branch("fTrackXloc", &(fTrkClRes->fTrackXloc), "fTrackXloc/F");
341  fTTree->Branch("fTrackYloc", &(fTrkClRes->fTrackYloc), "fTrackYloc/F");
342  fTTree->Branch("fTrackSlopeX", &(fTrkClRes->fTrackSlopeX), "fTrackSlopeX/F");
343  fTTree->Branch("fTrackSlopeY", &(fTrkClRes->fTrackSlopeY), "fTrackSlopeY/F");
344  fTTree->Branch("fBendingMomentum", &(fTrkClRes->fBendingMomentum), "fBendingMomentum/F");

346  fTTree->Branch("fResiduYGlobal", &(fTrkClRes->fResiduYGlobal), "fResiduYGlobal/F");

373 void Aligner::ProcessTrack(Track& track, const o2::mch::geo::TransformationCreator& transformation, bool doAlignment, double weight)
374 {
375 
377  // reset track records
378 
379  if (fMillepede->GetRecord()) {
380  fMillepede->GetRecord()->Reset();
381  }
382 
383  // loop over clusters to get starting values
384  bool first(true);
385 
386  auto itTrackParam = track.begin();
387  for (; itTrackParam != track.end(); ++itTrackParam) {
388 
389  // get cluster
390  const Cluster* cluster = itTrackParam->getClusterPtr();
391  if (!cluster) {
392  continue;
393  }
394  // for first valid cluster, save track position as "starting" values
395  if (first) {
396 
397  first = false;
398  FillTrackParamData(&*itTrackParam);
399  fTrackPos0[0] = fTrackPos[0];
400  fTrackPos0[1] = fTrackPos[1];
401  fTrackPos0[2] = fTrackPos[2];
402  fTrackSlope0[0] = fTrackSlope[0];
403  fTrackSlope0[1] = fTrackSlope[1];
404 
405  break;
406  }
407  }
408 
409  // redo straight track fit
410  if (fRefitStraightTracks) {
411 
412  // refit straight track
413  const LocalTrackParam trackParam(RefitStraightTrack(track, fTrackPos0[2]));
414 
415  /*
416  copy new parameters to stored ones for derivatives calculation
417  this is done only if BFieldOn is false, for which these parameters are used
418  */
419  if (!fBFieldOn) {
420  fTrackPos0[0] = trackParam.fTrackX;
421  fTrackPos0[1] = trackParam.fTrackY;
422  fTrackPos0[2] = trackParam.fTrackZ;
423  fTrackSlope0[0] = trackParam.fTrackSlopeX;
424  fTrackSlope0[1] = trackParam.fTrackSlopeY;
425  }
426  }
427 
428  // second loop to perform alignment
429  itTrackParam = track.begin();
430  for (; itTrackParam != track.end(); ++itTrackParam) {
431 
432  // get cluster
433  const Cluster* cluster = itTrackParam->getClusterPtr();
434  if (!cluster) {
435  continue;
436  }
437 
438  // fill local variables for this position --> one measurement
439 
440  FillDetElemData(cluster); // function to get the transformation matrix
441  FillRecPointData(cluster);
442  FillTrackParamData(&*itTrackParam);
443 
444  // 'inverse' (GlobalToLocal) rotation matrix
445  // const double* r(fGeoCombiTransInverse.GetRotationMatrix());
446  o2::math_utils::Transform3D trans = transformation(cluster->getDEId());
447  // LOG(info) << Form("cluster ID: %i", cluster->getDEId());
448  TMatrixD transMat(3, 4);
449  trans.GetTransformMatrix(transMat);
450  // transMat.Print();
451  double r[12];
452  r[0] = transMat(0, 0);
453  r[1] = transMat(0, 1);
454  r[2] = transMat(0, 2);
455  r[3] = transMat(1, 0);
456  r[4] = transMat(1, 1);
457  r[5] = transMat(1, 2);
458  r[6] = transMat(2, 0);
459  r[7] = transMat(2, 1);
460  r[8] = transMat(2, 2);
461  r[9] = transMat(0, 3);
462  r[10] = transMat(1, 3);
463  r[11] = transMat(2, 3);
464  // calculate measurements
465  if (fBFieldOn) {
466 
467  // use residuals (cluster - track) for measurement
468  fMeas[0] = r[0] * (fClustPos[0] - fTrackPos[0]) + r[1] * (fClustPos[1] - fTrackPos[1]);
469  fMeas[1] = r[3] * (fClustPos[0] - fTrackPos[0]) + r[4] * (fClustPos[1] - fTrackPos[1]);
470  } else {
471 
472  // use cluster position for measurement
473  fMeas[0] = (r[0] * fClustPos[0] + r[1] * fClustPos[1]);
474  fMeas[1] = (r[3] * fClustPos[0] + r[4] * fClustPos[1]);
475  }
476  // printf("DE %d, X: %f %f ; Y: %f %f ; Z: %f\n", cluster->getDEId(), fClustPos[0], fTrackPos[0], fClustPos[1], fTrackPos[1], fClustPos[2]);
477 
478  if (fDoEvaluation) {
479 
480  const float InvBendingMom = itTrackParam->getInverseBendingMomentum();
481  const float TrackCharge = itTrackParam->getCharge();
482 
483  double B[3] = {0.0, 0.0, 0.0};
484  double x[3] = {fTrackPos[0], fTrackPos[1], fTrackPos[2]};
485  TGeoGlobalMagField::Instance()->Field(x, B);
486  const float Bx = B[0];
487  const float By = B[1];
488  const float Bz = B[2];
489 
490  fTrkClRes->fClDetElem = cluster->getDEId();
491  fTrkClRes->fClDetElemNumber = GetDetElemNumber(cluster->getDEId());
492  fTrkClRes->fClusterX = fClustPos[0];
493  fTrkClRes->fClusterY = fClustPos[1];
494  fTrkClRes->fClusterZ = fClustPos[2];
495 
496  // fTrkClRes->fTrackX = fTrackPos0[0] + fTrackSlope0[0] * (fTrackPos[2] - fTrackPos0[2]); // fTrackPos[0];
497  // fTrkClRes->fTrackY = fTrackPos0[1] + fTrackSlope0[1] * (fTrackPos[2] - fTrackPos0[2]); // fTrackPos[1];
498  // fTrkClRes->fTrackSlopeX = fTrackSlope0[0];
499  // fTrkClRes->fTrackSlopeY = fTrackSlope0[1];
500 
501  fTrkClRes->fTrackX = fTrackPos[0];
502  fTrkClRes->fTrackY = fTrackPos[1];
503  fTrkClRes->fTrackZ = fTrackPos[2];
504 
505  fTrkClRes->fClusterXloc = r[0] * fClustPos[0] + r[1] * fClustPos[1];
506  fTrkClRes->fClusterYloc = r[3] * fClustPos[0] + r[4] * fClustPos[1];
507 
508  fTrkClRes->fTrackXloc = r[0] * fTrackPos[0] + r[1] * fTrackPos[1];
509  fTrkClRes->fTrackYloc = r[3] * fTrackPos[0] + r[4] * fTrackPos[1];
510 
511  fTrkClRes->fTrackSlopeX = fTrackSlope[0];
512  fTrkClRes->fTrackSlopeY = fTrackSlope[1];
513 
514  fTrkClRes->fBendingMomentum = TrackCharge / InvBendingMom;
515 
516  fTrkClRes->fResiduXGlobal = fClustPos[0] - fTrackPos[0];
517  fTrkClRes->fResiduYGlobal = fClustPos[1] - fTrackPos[1];
518  fTrkClRes->fResiduXLocal = r[0] * (fClustPos[0] - fTrackPos[0]) + r[1] * (fClustPos[1] - fTrackPos[1]);
519  fTrkClRes->fResiduYLocal = r[3] * (fClustPos[0] - fTrackPos[0]) + r[4] * (fClustPos[1] - fTrackPos[1]);
520 
521  fTrkClRes->fCharge = TrackCharge;
522 
523  fTrkClRes->fBx = Bx;
524  fTrkClRes->fBy = By;
525  fTrkClRes->fBz = Bz;
526 
527  if (fTTree) {
528  fTTree->Fill();
529  }
530  }
531  // Set local equations
532  LocalEquationX(r);
533  LocalEquationY(r);
534  }
535 
536  // copy track record
537  if (!fDisableRecordWriter) {
538  mRecordWriter->setRecordRun(fRunNumber);
539  mRecordWriter->setRecordWeight(weight);
540  }
541 
542  // save record data
543  if (doAlignment) {
544  if (!fDisableRecordWriter) {
545  mRecordWriter->fillRecordTree();
546  }
547  }
548 }

574 void Aligner::FixChamber(int iCh, unsigned int mask)
575 {
577 
578  // check boundaries
579  if (iCh < 1 || iCh > 10) {
580  LOG(fatal) << "Invalid chamber index " << iCh;
581  }
582 
583  // get first and last element
584  const int iDetElemFirst = fgSNDetElemCh[iCh - 1];
585  const int iDetElemLast = fgSNDetElemCh[iCh];
586  for (int i = iDetElemFirst; i < iDetElemLast; ++i) {
587 
588  LOG(info) << "Fixing " << GetParameterMaskString(mask).Data() << " for detector element " << i;
