533 //_____________________________________________________
534 inline float TrackResiduals::getDXI(int ix) const
535 {
536  if (mUniformBins[VoxX]) {
537  return mDXI;
538  } else {
539  if (ix < param::NRowsPerROC[0]) {
540  // we are in the IROC
541  return 1.f / param::RowDX[0];
542  } else if (ix > param::NRowsAccumulated[param::NROCTypes - 2]) {
543  // we are in the last OROC
544  return 1.f / param::RowDX[param::NROCTypes - 1];
545  } else if (ix < param::NRowsAccumulated[1]) {
546  // OROC1
547  return 1.f / param::RowDX[1];
548  } else {
549  // OROC2
550  return 1.f / param::RowDX[2];
551  }
552  }
553 }

631 //_____________________________________________________
632 inline int TrackResiduals::getY2XBinExact(float y2x, int ix) const
633 {
634  if (y2x < -mMaxY2X[ix]) {
635  return -1;
636  }
637  if (y2x > mMaxY2X[ix]) {
638  return mNY2XBins;
639  }
640  if (mUniformBins[VoxF]) {
641  return static_cast<int>((y2x + mMaxY2X[ix]) * getDY2XI(ix));
642  }
643  y2x /= mMaxY2X[ix];
644  for (int iBin = 0; iBin < mNY2XBins; ++iBin) {
645  if (y2x < mY2XBinsCenter[iBin] + mY2XBinsDH[iBin]) {
646  return iBin;
647  }
648  }
649  return mNY2XBins;
650 }

652 //_____________________________________________________
653 inline int TrackResiduals::getY2XBin(float y2x, int ix) const
654 {
655  int bp = getY2XBinExact(y2x, ix);
656  return bp > -1 ? (bp < mNY2XBins ? bp : mNY2XBins - 1) : 0;
657 }

659 //_____________________________________________________
660 inline int TrackResiduals::getZ2XBinExact(float z2x) const
661 {
662  if (mUniformBins[VoxZ]) {
663  float bz = z2x * getDZ2XI();
664  if (bz >= mNZ2XBins) {
665  return -1;
666  }
667  if (bz < 0) {
668  // accounting for clusters which were moved to the wrong side
669  // TODO: how can this happen?
670  bz = 0;
671  }
672  return static_cast<int>(bz);
673  }
674  for (int iBin = 0; iBin < mNZ2XBins; ++iBin) {
675  if (z2x < mZ2XBinsCenter[iBin] + mZ2XBinsDH[iBin]) {
676  return iBin;
677  }
678  }
679  return -1;
680 }

682 //_____________________________________________________

684 {
