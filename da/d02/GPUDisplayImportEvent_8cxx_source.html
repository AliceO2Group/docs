40void GPUDisplay::DrawGLScene_updateEventData()
41{
42 mTimerDraw.ResetStart();
43 if (mIOPtrs->clustersNative) {
44 mCurrentClusters = mIOPtrs->clustersNative->nClustersTotal;
45 } else {
46 mCurrentClusters = 0;
47 for (int32_t iSlice = 0; iSlice < NSLICES; iSlice++) {
48 mCurrentClusters += mIOPtrs->nClusterData[iSlice];
49 }
50 }
51 if (mNMaxClusters < mCurrentClusters) {
52 mNMaxClusters = mCurrentClusters;
53 mGlobalPosPtr.reset(new float4[mNMaxClusters]);
54 mGlobalPos = mGlobalPosPtr.get();
55 }
56
57 mCurrentSpacePointsTRD = mIOPtrs->nTRDTracklets;
58 if (mCurrentSpacePointsTRD > mNMaxSpacePointsTRD) {
59 mNMaxSpacePointsTRD = mCurrentSpacePointsTRD;
60 mGlobalPosPtrTRD.reset(new float4[mNMaxSpacePointsTRD]);
61 mGlobalPosPtrTRD2.reset(new float4[mNMaxSpacePointsTRD]);
62 mGlobalPosTRD = mGlobalPosPtrTRD.get();
63 mGlobalPosTRD2 = mGlobalPosPtrTRD2.get();
64 }
65
66 mCurrentClustersITS = mIOPtrs->itsClusters ? mIOPtrs->nItsClusters : 0;
67 if (mNMaxClustersITS < mCurrentClustersITS) {
68 mNMaxClustersITS = mCurrentClustersITS;
69 mGlobalPosPtrITS.reset(new float4[mNMaxClustersITS]);
70 mGlobalPosITS = mGlobalPosPtrITS.get();
71 }
72
73 mCurrentClustersTOF = mIOPtrs->nTOFClusters;
74 if (mNMaxClustersTOF < mCurrentClustersTOF) {
75 mNMaxClustersTOF = mCurrentClustersTOF;
76 mGlobalPosPtrTOF.reset(new float4[mNMaxClustersTOF]);
77 mGlobalPosTOF = mGlobalPosPtrTOF.get();
78 }
79
80 uint32_t nTpcMergedTracks = mConfig.showTPCTracksFromO2Format ? mIOPtrs->nOutputTracksTPCO2 : mIOPtrs->nMergedTracks;
81 if ((size_t)nTpcMergedTracks > mTRDTrackIds.size()) {
82 mTRDTrackIds.resize(nTpcMergedTracks);
83 }
84 if (mIOPtrs->nItsTracks > mITSStandaloneTracks.size()) {
85 mITSStandaloneTracks.resize(mIOPtrs->nItsTracks);
86 }
87 for (uint32_t i = 0; i < nTpcMergedTracks; i++) {
88 mTRDTrackIds[i] = -1;
89 }
90 auto tmpDoTRDTracklets = [&](auto* trdTracks) {
91 for (uint32_t i = 0; i < mIOPtrs->nTRDTracks; i++) {
92 if (trdTracks[i].getNtracklets()) {
93 mTRDTrackIds[trdTracks[i].getRefGlobalTrackIdRaw()] = i;
94 }
95 }
96 };
97 if (mIOPtrs->trdTracksO2) {
98#ifdef GPUCA_HAVE_O2HEADERS
99 tmpDoTRDTracklets(mIOPtrs->trdTracksO2);
100#endif
101 } else {
102 tmpDoTRDTracklets(mIOPtrs->trdTracks);
103 }
104 if (mIOPtrs->nItsTracks) {
105 std::fill(mITSStandaloneTracks.begin(), mITSStandaloneTracks.end(), true);
106 if (mIOPtrs->tpcLinkITS) {
107 for (uint32_t i = 0; i < nTpcMergedTracks; i++) {
108 if (mIOPtrs->tpcLinkITS[i] != -1) {
109 mITSStandaloneTracks[mIOPtrs->tpcLinkITS[i]] = false;
110 }
111 }
112 }
113 }
114
115 if (mCfgH.trackFilter) {
116 uint32_t nTracks = mConfig.showTPCTracksFromO2Format ? mIOPtrs->nOutputTracksTPCO2 : mIOPtrs->nMergedTracks;
117 mTrackFilter.resize(nTracks);
118 std::fill(mTrackFilter.begin(), mTrackFilter.end(), true);
119 if (buildTrackFilter()) {
120 SetInfo("Error running track filter from %s", mConfig.filterMacros[mCfgH.trackFilter - 1].c_str());
121 } else {
122 uint32_t nFiltered = 0;
123 for (uint32_t i = 0; i < mTrackFilter.size(); i++) {
124 nFiltered += !mTrackFilter[i];
125 }
126 if (mUpdateTrackFilter) {
127 SetInfo("Applied track filter %s - filtered %u / %u", mConfig.filterMacros[mCfgH.trackFilter - 1].c_str(), nFiltered, (uint32_t)mTrackFilter.size());
128 }
129 }
130 }
131 mUpdateTrackFilter = false;
132
133 mMaxClusterZ = 0;
134 GPUCA_OPENMP(parallel for num_threads(getNumThreads()) reduction(max : mMaxClusterZ))
135 for (int32_t iSlice = 0; iSlice < NSLICES; iSlice++) {
136 int32_t row = 0;
137 uint32_t nCls = mParam->par.earlyTpcTransform ? mIOPtrs->nClusterData[iSlice] : mIOPtrs->clustersNative ? mIOPtrs->clustersNative->nClustersSector[iSlice]
138 : 0;
139 for (uint32_t i = 0; i < nCls; i++) {
140 int32_t cid;
141 if (mParam->par.earlyTpcTransform) {
142 const auto& cl = mIOPtrs->clusterData[iSlice][i];
143 cid = cl.id;
144 row = cl.row;
145 } else {
146 cid = mIOPtrs->clustersNative->clusterOffset[iSlice][0] + i;
147 while (row < GPUCA_ROW_COUNT - 1 && mIOPtrs->clustersNative->clusterOffset[iSlice][row + 1] <= (uint32_t)cid) {
148 row++;
149 }
150 }
151 if (cid >= mNMaxClusters) {
152 throw std::runtime_error("Cluster Buffer Size exceeded");
153 }
154 float4* ptr = &mGlobalPos[cid];
155 if (mParam->par.earlyTpcTransform) {
156 const auto& cl = mIOPtrs->clusterData[iSlice][i];
157 mParam->Slice2Global(iSlice, (mCfgH.clustersOnNominalRow ? mParam->tpcGeometry.Row2X(row) : cl.x) + mCfgH.xAdd, cl.y, cl.z, &ptr->x, &ptr->y, &ptr->z);
158 } else {
159 float x, y, z;
160 const auto& cln = mIOPtrs->clustersNative->clusters[iSlice][0][i];
161 GPUTPCConvertImpl::convert(*mCalib->fastTransform, *mParam, iSlice, row, cln.getPad(), cln.getTime(), x, y, z);
162 if (mCfgH.clustersOnNominalRow) {
163 x = mParam->tpcGeometry.Row2X(row);
164 }
165 mParam->Slice2Global(iSlice, x + mCfgH.xAdd, y, z, &ptr->x, &ptr->y, &ptr->z);
166 }
167
168 if (fabsf(ptr->z) > mMaxClusterZ) {
169 mMaxClusterZ = fabsf(ptr->z);
170 }
171 ptr->z += iSlice < 18 ? mCfgH.zAdd : -mCfgH.zAdd;
172 ptr->x *= GL_SCALE_FACTOR;
173 ptr->y *= GL_SCALE_FACTOR;
174 ptr->z *= GL_SCALE_FACTOR;
175 ptr->w = tCLUSTER;
176 }
177 }
178
179 int32_t trdTriggerRecord = -1;
180 float trdZoffset = 0;
181 GPUCA_OPENMP(parallel for num_threads(getNumThreads()) reduction(max : mMaxClusterZ) firstprivate(trdTriggerRecord, trdZoffset))
182 for (int32_t i = 0; i < mCurrentSpacePointsTRD; i++) {
183 while (mParam->par.continuousTracking && trdTriggerRecord < (int32_t)mIOPtrs->nTRDTriggerRecords - 1 && mIOPtrs->trdTrackletIdxFirst[trdTriggerRecord + 1] <= i) {
184 trdTriggerRecord++;
185#ifdef GPUCA_HAVE_O2HEADERS
186 float trdTime = mIOPtrs->trdTriggerTimes[trdTriggerRecord] * 1e3 / o2::constants::lhc::LHCBunchSpacingNS / o2::tpc::constants::LHCBCPERTIMEBIN;
187 trdZoffset = fabsf(mCalib->fastTransformHelper->getCorrMap()->convVertexTimeToZOffset(0, trdTime, mParam->continuousMaxTimeBin));
188#endif
189 }
190 const auto& sp = mIOPtrs->trdSpacePoints[i];
191 int32_t iSec = trdGeometry()->GetSector(mIOPtrs->trdTracklets[i].GetDetector());
192 float4* ptr = &mGlobalPosTRD[i];
193 mParam->Slice2Global(iSec, sp.getX() + mCfgH.xAdd, sp.getY(), sp.getZ(), &ptr->x, &ptr->y, &ptr->z);
194 ptr->z += ptr->z > 0 ? trdZoffset : -trdZoffset;
195 if (fabsf(ptr->z) > mMaxClusterZ) {
196 mMaxClusterZ = fabsf(ptr->z);
197 }
198 ptr->x *= GL_SCALE_FACTOR;
199 ptr->y *= GL_SCALE_FACTOR;
200 ptr->z *= GL_SCALE_FACTOR;
201 ptr->w = tTRDCLUSTER;
202 ptr = &mGlobalPosTRD2[i];
203 mParam->Slice2Global(iSec, sp.getX() + mCfgH.xAdd + 4.5f, sp.getY() + 1.5f * sp.getDy(), sp.getZ(), &ptr->x, &ptr->y, &ptr->z);
204 ptr->z += ptr->z > 0 ? trdZoffset : -trdZoffset;
205 if (fabsf(ptr->z) > mMaxClusterZ) {
206 mMaxClusterZ = fabsf(ptr->z);
207 }
208 ptr->x *= GL_SCALE_FACTOR;
209 ptr->y *= GL_SCALE_FACTOR;
210 ptr->z *= GL_SCALE_FACTOR;
211 ptr->w = tTRDCLUSTER;
212 }
213
214 GPUCA_OPENMP(parallel for num_threads(getNumThreads()) reduction(max : mMaxClusterZ))
215 for (int32_t i = 0; i < mCurrentClustersTOF; i++) {
216#ifdef GPUCA_HAVE_O2HEADERS
217 float4* ptr = &mGlobalPosTOF[i];
218 mParam->Slice2Global(mIOPtrs->tofClusters[i].getSector(), mIOPtrs->tofClusters[i].getX() + mCfgH.xAdd, mIOPtrs->tofClusters[i].getY(), mIOPtrs->tofClusters[i].getZ(), &ptr->x, &ptr->y, &ptr->z);
219 float ZOffset = 0;
220 if (mParam->par.continuousTracking) {
221 float tofTime = mIOPtrs->tofClusters[i].getTime() * 1e-3 / o2::constants::lhc::LHCBunchSpacingNS / o2::tpc::constants::LHCBCPERTIMEBIN;
222 ZOffset = fabsf(mCalib->fastTransformHelper->getCorrMap()->convVertexTimeToZOffset(0, tofTime, mParam->continuousMaxTimeBin));
223 ptr->z += ptr->z > 0 ? ZOffset : -ZOffset;
224 }
225 if (fabsf(ptr->z) > mMaxClusterZ) {
226 mMaxClusterZ = fabsf(ptr->z);
227 }
228 ptr->x *= GL_SCALE_FACTOR;
229 ptr->y *= GL_SCALE_FACTOR;
230 ptr->z *= GL_SCALE_FACTOR;
231 ptr->w = tTOFCLUSTER;
232#endif
233 }
234
235 if (mCurrentClustersITS) {
236#ifdef GPUCA_HAVE_O2HEADERS
237 float itsROFhalfLen = 0;
238#ifdef GPUCA_O2_LIB // Not available in standalone benchmark
239 if (mParam->par.continuousTracking) {
240 const auto& alpParams = o2::itsmft::DPLAlpideParam<o2::detectors::DetID::ITS>::Instance();
241 itsROFhalfLen = alpParams.roFrameLengthInBC / (float)o2::tpc::constants::LHCBCPERTIMEBIN / 2;
242 }
243#endif
244 int32_t i = 0;
245 for (uint32_t j = 0; j < mIOPtrs->nItsClusterROF; j++) {
246 float ZOffset = 0;
247 if (mParam->par.continuousTracking) {
248 o2::InteractionRecord startIR = o2::InteractionRecord(0, mIOPtrs->settingsTF && mIOPtrs->settingsTF->hasTfStartOrbit ? mIOPtrs->settingsTF->tfStartOrbit : 0);
249 float itsROFtime = mIOPtrs->itsClusterROF[j].getBCData().differenceInBC(startIR) / (float)o2::tpc::constants::LHCBCPERTIMEBIN;
250 ZOffset = fabsf(mCalib->fastTransformHelper->getCorrMap()->convVertexTimeToZOffset(0, itsROFtime + itsROFhalfLen, mParam->continuousMaxTimeBin));
251 }
252 if (i != mIOPtrs->itsClusterROF[j].getFirstEntry()) {
253 throw std::runtime_error("Inconsistent ITS data, number of clusters does not match ROF content");
254 }
255 for (int32_t k = 0; k < mIOPtrs->itsClusterROF[j].getNEntries(); k++) {
256 float4* ptr = &mGlobalPosITS[i];
257 const auto& cl = mIOPtrs->itsClusters[i];
258 auto* itsGeo = o2::its::GeometryTGeo::Instance();
259 auto p = cl.getXYZGlo(*itsGeo);
260 ptr->x = p.X();
261 ptr->y = p.Y();
262 ptr->z = p.Z();
263 ptr->z += ptr->z > 0 ? ZOffset : -ZOffset;
264 if (fabsf(ptr->z) > mMaxClusterZ) {
265 mMaxClusterZ = fabsf(ptr->z);
266 }
267 ptr->x *= GL_SCALE_FACTOR;
268 ptr->y *= GL_SCALE_FACTOR;
269 ptr->z *= GL_SCALE_FACTOR;
270 ptr->w = tITSCLUSTER;
271 i++;
272 }
273 }
274#endif
275 }
