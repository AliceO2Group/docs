370int32_t GPUReconstructionOCL::ExitDevice_Runtime()
371{
372 // Uninitialize OPENCL
373 SynchronizeGPU();
374
375 if (mMaster == nullptr) {
376 if (mDeviceMemoryBase) {
377 clReleaseMemObject(mInternals->mem_gpu);
378 clReleaseMemObject(mInternals->mem_constant);
379 for (uint32_t i = 0; i < mInternals->kernels.size(); i++) {
380 clReleaseKernel(mInternals->kernels[i].first);
381 }
382 mInternals->kernels.clear();
383 }
384 if (mHostMemoryBase) {
385 clEnqueueUnmapMemObject(mInternals->command_queue[0], mInternals->mem_host, mHostMemoryBase, 0, nullptr, nullptr);
386 for (int32_t i = 0; i < mNStreams; i++) {
387 clReleaseCommandQueue(mInternals->command_queue[i]);
388 }
389 clReleaseMemObject(mInternals->mem_host);
390 }
391
392 clReleaseProgram(mInternals->program);
393 clReleaseContext(mInternals->context);
394 GPUInfo("OPENCL Uninitialized");
395 }
396 mDeviceMemoryBase = nullptr;
397 mHostMemoryBase = nullptr;
398
399 return (0);
400}

442int32_t GPUReconstructionOCL::DoStuckProtection(int32_t stream, deviceEvent event)
443{
444 if (mProcessingSettings.stuckProtection) {
445 cl_int tmp = 0;
446 for (int32_t i = 0; i <= mProcessingSettings.stuckProtection / 50; i++) {
447 usleep(50);
448 clGetEventInfo(event.get<cl_event>(), CL_EVENT_COMMAND_EXECUTION_STATUS, sizeof(tmp), &tmp, nullptr);
449 if (tmp == CL_COMPLETE) {
450 break;
451 }
452 }
453 if (tmp != CL_COMPLETE) {
454 mGPUStuck = 1;
455 quit("GPU Stuck, future processing in this component is disabled, skipping event (GPU Event State %d)", (int32_t)tmp);
456 }
457 } else {
458 clFinish(mInternals->command_queue[stream]);
459 }
460 return 0;
461}
