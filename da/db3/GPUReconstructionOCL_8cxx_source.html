91int32_t GPUReconstructionOCLBackend::InitDevice_Runtime()
92{
93 if (mMaster == nullptr) {
94 cl_int ocl_error;
95 cl_uint num_platforms;
96 if (GPUFailedMsgI(clGetPlatformIDs(0, nullptr, &num_platforms))) {
97 quit("Error getting OpenCL Platform Count");
98 }
99 if (num_platforms == 0) {
100 quit("No OpenCL Platform found");
101 }
102 if (mProcessingSettings.debugLevel >= 2) {
103 GPUInfo("%d OpenCL Platforms found", num_platforms);
104 }
105
106 // Query platforms
107 mInternals->platforms.reset(new cl_platform_id[num_platforms]);
108 if (GPUFailedMsgI(clGetPlatformIDs(num_platforms, mInternals->platforms.get(), nullptr))) {
109 quit("Error getting OpenCL Platforms");
110 }
111
112 bool found = false;
113 if (mProcessingSettings.platformNum >= 0) {
114 if (mProcessingSettings.platformNum >= (int32_t)num_platforms) {
115 quit("Invalid platform specified");
116 }
117 mInternals->platform = mInternals->platforms[mProcessingSettings.platformNum];
118 found = true;
119 if (mProcessingSettings.debugLevel >= 2) {
120 char platform_profile[256] = {}, platform_version[256] = {}, platform_name[256] = {}, platform_vendor[256] = {};
121 clGetPlatformInfo(mInternals->platform, CL_PLATFORM_PROFILE, sizeof(platform_profile), platform_profile, nullptr);
122 clGetPlatformInfo(mInternals->platform, CL_PLATFORM_VERSION, sizeof(platform_version), platform_version, nullptr);
123 clGetPlatformInfo(mInternals->platform, CL_PLATFORM_NAME, sizeof(platform_name), platform_name, nullptr);
124 clGetPlatformInfo(mInternals->platform, CL_PLATFORM_VENDOR, sizeof(platform_vendor), platform_vendor, nullptr);
125 GPUInfo("Selected Platform %d: (%s %s) %s %s", mProcessingSettings.platformNum, platform_profile, platform_version, platform_vendor, platform_name);
126 }
127 } else {
128 for (uint32_t i_platform = 0; i_platform < num_platforms; i_platform++) {
129 char platform_profile[256] = {}, platform_version[256] = {}, platform_name[256] = {}, platform_vendor[256] = {};
130 clGetPlatformInfo(mInternals->platforms[i_platform], CL_PLATFORM_PROFILE, sizeof(platform_profile), platform_profile, nullptr);
131 clGetPlatformInfo(mInternals->platforms[i_platform], CL_PLATFORM_VERSION, sizeof(platform_version), platform_version, nullptr);
132 clGetPlatformInfo(mInternals->platforms[i_platform], CL_PLATFORM_NAME, sizeof(platform_name), platform_name, nullptr);
133 clGetPlatformInfo(mInternals->platforms[i_platform], CL_PLATFORM_VENDOR, sizeof(platform_vendor), platform_vendor, nullptr);
134 const char* platformUsageInfo = "";
135 if (!found && CheckPlatform(i_platform)) {
136 found = true;
137 mInternals->platform = mInternals->platforms[i_platform];
138 if (mProcessingSettings.debugLevel >= 2) {
139 platformUsageInfo = " !!! Using this platform !!!";
140 }
141 }
142 if (mProcessingSettings.debugLevel >= 2) {
143 GPUInfo("Available Platform %d: (%s %s) %s %s%s", i_platform, platform_profile, platform_version, platform_vendor, platform_name, platformUsageInfo);
144 }
145 }
146 }
147
148 if (found == false) {
149 quit("Did not find compatible OpenCL Platform");
150 }
151
152 cl_uint count, bestDevice = (cl_uint)-1;
153 if (GPUFailedMsgI(clGetDeviceIDs(mInternals->platform, CL_DEVICE_TYPE_ALL, 0, nullptr, &count))) {
154 quit("Error getting OPENCL Device Count");
155 }
156
157 // Query devices
158 mInternals->devices.reset(new cl_device_id[count]);
159 if (GPUFailedMsgI(clGetDeviceIDs(mInternals->platform, CL_DEVICE_TYPE_ALL, count, mInternals->devices.get(), nullptr))) {
160 quit("Error getting OpenCL devices");
161 }
162
163 char device_vendor[64], device_name[64];
164 cl_device_type device_type;
165 cl_uint freq, shaders;
166
167 if (mProcessingSettings.debugLevel >= 2) {
168 GPUInfo("Available OPENCL devices:");
169 }
170 std::vector<bool> devicesOK(count, false);
171 for (uint32_t i = 0; i < count; i++) {
172 if (mProcessingSettings.debugLevel >= 3) {
173 GPUInfo("Examining device %d", i);
174 }
175 cl_uint nbits;
176 cl_bool endian;
177
178 clGetDeviceInfo(mInternals->devices[i], CL_DEVICE_NAME, 64, device_name, nullptr);
179 clGetDeviceInfo(mInternals->devices[i], CL_DEVICE_VENDOR, 64, device_vendor, nullptr);
180 clGetDeviceInfo(mInternals->devices[i], CL_DEVICE_TYPE, sizeof(cl_device_type), &device_type, nullptr);
181 clGetDeviceInfo(mInternals->devices[i], CL_DEVICE_MAX_CLOCK_FREQUENCY, sizeof(freq), &freq, nullptr);
182 clGetDeviceInfo(mInternals->devices[i], CL_DEVICE_MAX_COMPUTE_UNITS, sizeof(shaders), &shaders, nullptr);
183 clGetDeviceInfo(mInternals->devices[i], CL_DEVICE_ADDRESS_BITS, sizeof(nbits), &nbits, nullptr);
184 clGetDeviceInfo(mInternals->devices[i], CL_DEVICE_ENDIAN_LITTLE, sizeof(endian), &endian, nullptr);
185 int32_t deviceOK = true;
186 const char* deviceFailure = "";
187 if (mProcessingSettings.gpuDeviceOnly && ((device_type & CL_DEVICE_TYPE_CPU) || !(device_type & CL_DEVICE_TYPE_GPU))) {
188 deviceOK = false;
189 deviceFailure = "No GPU device";
190 }
191 if (nbits / 8 != sizeof(void*)) {
192 deviceOK = false;
193 deviceFailure = "No 64 bit device";
194 }
195 if (!endian) {
196 deviceOK = false;
197 deviceFailure = "No Little Endian Mode";
198 }
199
200 double bestDeviceSpeed = -1, deviceSpeed = (double)freq * (double)shaders;
201 if (mProcessingSettings.debugLevel >= 2) {
202 GPUImportant("Device %s%2d: %s %s (Frequency %d, Shaders %d, %d bit) (Speed Value: %ld)%s %s", deviceOK ? " " : "[", i, device_vendor, device_name, (int32_t)freq, (int32_t)shaders, (int32_t)nbits, (int64_t)deviceSpeed, deviceOK ? " " : " ]", deviceOK ? "" : deviceFailure);
203 }
204 if (!deviceOK) {
205 continue;
206 }
207 devicesOK[i] = true;
208 if (deviceSpeed > bestDeviceSpeed) {
209 bestDevice = i;
210 bestDeviceSpeed = deviceSpeed;
211 } else {
212 if (mProcessingSettings.debugLevel >= 2) {
213 GPUInfo("Skipping: Speed %f < %f", deviceSpeed, bestDeviceSpeed);
214 }
215 }
216 }
217 if (bestDevice == (cl_uint)-1) {
218 quit("No %sOPENCL Device available, aborting OPENCL Initialisation", count ? "appropriate " : "");
219 }
220
221 if (mProcessingSettings.deviceNum > -1) {
222 if (mProcessingSettings.deviceNum >= (signed)count) {
223 quit("Requested device ID %d does not exist", mProcessingSettings.deviceNum);
224 } else if (!devicesOK[mProcessingSettings.deviceNum]) {
225 quit("Unsupported device requested (%d)", mProcessingSettings.deviceNum);
226 } else {
227 bestDevice = mProcessingSettings.deviceNum;
228 }
229 }
230 mInternals->device = mInternals->devices[bestDevice];
231
232 cl_ulong constantBuffer, globalMem, localMem;
233 char deviceVersion[64];
234 size_t maxWorkGroup, maxWorkItems[3];
235 clGetDeviceInfo(mInternals->device, CL_DEVICE_NAME, 64, device_name, nullptr);
236 clGetDeviceInfo(mInternals->device, CL_DEVICE_VENDOR, 64, device_vendor, nullptr);
237 clGetDeviceInfo(mInternals->device, CL_DEVICE_TYPE, sizeof(cl_device_type), &device_type, nullptr);
238 clGetDeviceInfo(mInternals->device, CL_DEVICE_MAX_CLOCK_FREQUENCY, sizeof(freq), &freq, nullptr);
239 clGetDeviceInfo(mInternals->device, CL_DEVICE_MAX_COMPUTE_UNITS, sizeof(shaders), &shaders, nullptr);
240 clGetDeviceInfo(mInternals->device, CL_DEVICE_GLOBAL_MEM_SIZE, sizeof(globalMem), &globalMem, nullptr);
241 clGetDeviceInfo(mInternals->device, CL_DEVICE_MAX_CONSTANT_BUFFER_SIZE, sizeof(constantBuffer), &constantBuffer, nullptr);
242 clGetDeviceInfo(mInternals->device, CL_DEVICE_LOCAL_MEM_SIZE, sizeof(localMem), &localMem, nullptr);
243 clGetDeviceInfo(mInternals->device, CL_DEVICE_VERSION, sizeof(deviceVersion) - 1, deviceVersion, nullptr);
244 clGetDeviceInfo(mInternals->device, CL_DEVICE_MAX_WORK_GROUP_SIZE, sizeof(maxWorkGroup), &maxWorkGroup, nullptr);
245 clGetDeviceInfo(mInternals->device, CL_DEVICE_MAX_WORK_ITEM_SIZES, sizeof(maxWorkItems), maxWorkItems, nullptr);
246 int versionMajor, versionMinor;
247 sscanf(deviceVersion, "OpenCL %d.%d", &versionMajor, &versionMinor);
248 if (mProcessingSettings.debugLevel >= 2) {
249 GPUInfo("Using OpenCL device %d: %s %s with properties:", bestDevice, device_vendor, device_name);
250 GPUInfo("\tVersion = %s", deviceVersion);
251 GPUInfo("\tFrequency = %d", (int32_t)freq);
252 GPUInfo("\tShaders = %d", (int32_t)shaders);
253 GPUInfo("\tGLobalMemory = %ld", (int64_t)globalMem);
254 GPUInfo("\tContantMemoryBuffer = %ld", (int64_t)constantBuffer);
255 GPUInfo("\tLocalMemory = %ld", (int64_t)localMem);
256 GPUInfo("\tmaxThreadsPerBlock = %ld", (int64_t)maxWorkGroup);
257 GPUInfo("\tmaxThreadsDim = %ld %ld %ld", (int64_t)maxWorkItems[0], (int64_t)maxWorkItems[1], (int64_t)maxWorkItems[2]);
258 GPUInfo(" ");
259 }
260#ifndef GPUCA_NO_CONSTANT_MEMORY
261 if (gGPUConstantMemBufferSize > constantBuffer) {
262 quit("Insufficient constant memory available on GPU %d < %d!", (int32_t)constantBuffer, (int32_t)gGPUConstantMemBufferSize);
263 }
264#endif
265
266 mDeviceName = device_name;
267 mDeviceName += " (OpenCL)";
268 mBlockCount = shaders;
269 mWarpSize = 32;
270 mMaxThreads = std::max<int32_t>(mMaxThreads, maxWorkGroup * mBlockCount);
271
272 mInternals->context = clCreateContext(nullptr, ContextForAllPlatforms() ? count : 1, ContextForAllPlatforms() ? mInternals->devices.get() : &mInternals->device, nullptr, nullptr, &ocl_error);
273 if (GPUFailedMsgI(ocl_error)) {
274 quit("Could not create OPENCL Device Context!");
275 }
276
277 if (GetOCLPrograms()) {
278 return 1;
279 }
280
281 if (mProcessingSettings.debugLevel >= 2) {
282 GPUInfo("OpenCL program and kernels loaded successfully");
283 }
284
285 mInternals->mem_gpu = clCreateBuffer(mInternals->context, CL_MEM_READ_WRITE, mDeviceMemorySize, nullptr, &ocl_error);
286 if (GPUFailedMsgI(ocl_error)) {
287 clReleaseContext(mInternals->context);
288 quit("OPENCL Memory Allocation Error");
289 }
290
291 mInternals->mem_constant = clCreateBuffer(mInternals->context, CL_MEM_READ_ONLY, gGPUConstantMemBufferSize, nullptr, &ocl_error);
292 if (GPUFailedMsgI(ocl_error)) {
293 clReleaseMemObject(mInternals->mem_gpu);
294 clReleaseContext(mInternals->context);
295 quit("OPENCL Constant Memory Allocation Error");
296 }
297
298 if (device_type & CL_DEVICE_TYPE_CPU) {
299 if (mProcessingSettings.deviceTimers && mProcessingSettings.debugLevel >= 2) {
300 GPUInfo("Disabling device timers for CPU device");
301 }
302 mProcessingSettings.deviceTimers = 0;
303 }
304 for (int32_t i = 0; i < mNStreams; i++) {
305#ifdef CL_VERSION_2_0
306 cl_queue_properties prop = 0;
307 if (versionMajor >= 2 && IsGPU() && mProcessingSettings.deviceTimers) {
308 prop |= CL_QUEUE_PROFILING_ENABLE;
309 }
310 mInternals->command_queue[i] = clCreateCommandQueueWithProperties(mInternals->context, mInternals->device, &prop, &ocl_error);
311 if (mProcessingSettings.deviceTimers && ocl_error == CL_INVALID_QUEUE_PROPERTIES) {
312 GPUError("GPU device timers not supported by OpenCL platform, disabling");
313 mProcessingSettings.deviceTimers = 0;
314 prop = 0;
315 mInternals->command_queue[i] = clCreateCommandQueueWithProperties(mInternals->context, mInternals->device, &prop, &ocl_error);
316 }
317#else
318 mInternals->command_queue[i] = clCreateCommandQueue(mInternals->context, mInternals->device, 0, &ocl_error);
319#endif
320 if (GPUFailedMsgI(ocl_error)) {
321 quit("Error creating OpenCL command queue");
322 }
323 }
324 if (GPUFailedMsgI(clEnqueueMigrateMemObjects(mInternals->command_queue[0], 1, &mInternals->mem_gpu, 0, 0, nullptr, nullptr))) {
325 quit("Error migrating buffer");
326 }
327 if (GPUFailedMsgI(clEnqueueMigrateMemObjects(mInternals->command_queue[0], 1, &mInternals->mem_constant, 0, 0, nullptr, nullptr))) {
328 quit("Error migrating buffer");
329 }
330
331 mInternals->mem_host = clCreateBuffer(mInternals->context, CL_MEM_READ_WRITE | CL_MEM_ALLOC_HOST_PTR, mHostMemorySize, nullptr, &ocl_error);
332 if (GPUFailedMsgI(ocl_error)) {
333 quit("Error allocating pinned host memory");
334 }
335
336 const char* krnlGetPtr = "__kernel void krnlGetPtr(__global char* gpu_mem, __global char* constant_mem, __global size_t* host_mem) {if (get_global_id(0) == 0) {host_mem[0] = (size_t) gpu_mem; host_mem[1] = (size_t) constant_mem;}}";
337 cl_program program = clCreateProgramWithSource(mInternals->context, 1, (const char**)&krnlGetPtr, nullptr, &ocl_error);
338 if (GPUFailedMsgI(ocl_error)) {
339 quit("Error creating program object");
340 }
341 ocl_error = clBuildProgram(program, 1, &mInternals->device, "", nullptr, nullptr);
342 if (GPUFailedMsgI(ocl_error)) {
343 char build_log[16384];
344 clGetProgramBuildInfo(program, mInternals->device, CL_PROGRAM_BUILD_LOG, 16384, build_log, nullptr);
345 GPUImportant("Build Log:\n\n%s\n\n", build_log);
346 quit("Error compiling program");
347 }
348 cl_kernel kernel = clCreateKernel(program, "krnlGetPtr", &ocl_error);
349 if (GPUFailedMsgI(ocl_error)) {
350 quit("Error creating kernel");
351 }
352
353 if (GPUFailedMsgI(OCLsetKernelParameters(kernel, mInternals->mem_gpu, mInternals->mem_constant, mInternals->mem_host)) ||
354 GPUFailedMsgI(clExecuteKernelA(mInternals->command_queue[0], kernel, 16, 16, nullptr)) ||
355 GPUFailedMsgI(clFinish(mInternals->command_queue[0])) ||
356 GPUFailedMsgI(clReleaseKernel(kernel)) ||
357 GPUFailedMsgI(clReleaseProgram(program))) {
358 quit("Error obtaining device memory ptr");
359 }
360
361 if (mProcessingSettings.debugLevel >= 2) {
362 GPUInfo("Mapping hostmemory");
363 }
364 mHostMemoryBase = clEnqueueMapBuffer(mInternals->command_queue[0], mInternals->mem_host, CL_TRUE, CL_MAP_READ | CL_MAP_WRITE, 0, mHostMemorySize, 0, nullptr, nullptr, &ocl_error);
365 if (GPUFailedMsgI(ocl_error)) {
366 quit("Error allocating Page Locked Host Memory");
367 }
368
369 mDeviceMemoryBase = ((void**)mHostMemoryBase)[0];
370 mDeviceConstantMem = (GPUConstantMem*)((void**)mHostMemoryBase)[1];
371
372 if (mProcessingSettings.debugLevel >= 1) {
373 GPUInfo("Memory ptrs: GPU (%ld bytes): %p - Host (%ld bytes): %p", (int64_t)mDeviceMemorySize, mDeviceMemoryBase, (int64_t)mHostMemorySize, mHostMemoryBase);
374 memset(mHostMemoryBase, 0xDD, mHostMemorySize);
375 }
376
377 GPUInfo("OPENCL Initialisation successfull (%d: %s %s (Frequency %d, Shaders %d), %ld / %ld bytes host / global memory, Stack frame %d, Constant memory %ld)", bestDevice, device_vendor, device_name, (int32_t)freq, (int32_t)shaders, (int64_t)mDeviceMemorySize,
378 (int64_t)mHostMemorySize, -1, (int64_t)gGPUConstantMemBufferSize);
379 } else {
380 GPUReconstructionOCL* master = dynamic_cast<GPUReconstructionOCL*>(mMaster);
381 mBlockCount = master->mBlockCount;
382 mWarpSize = master->mWarpSize;
383 mMaxThreads = master->mMaxThreads;
384 mDeviceName = master->mDeviceName;
385 mDeviceConstantMem = master->mDeviceConstantMem;
386 mInternals = master->mInternals;
387 }
388
389 for (uint32_t i = 0; i < mEvents.size(); i++) {
390 cl_event* events = (cl_event*)mEvents[i].data();
391 new (events) cl_event[mEvents[i].size()];
392 }
393
394 return (0);
395}

397int32_t GPUReconstructionOCLBackend::ExitDevice_Runtime()
398{
399 // Uninitialize OPENCL
400 SynchronizeGPU();
401
402 if (mMaster == nullptr) {
403 if (mDeviceMemoryBase) {
404 clReleaseMemObject(mInternals->mem_gpu);
405 clReleaseMemObject(mInternals->mem_constant);
406 for (uint32_t i = 0; i < mInternals->kernels.size(); i++) {
407 clReleaseKernel(mInternals->kernels[i].first);
408 }
409 mInternals->kernels.clear();
410 }
411 if (mHostMemoryBase) {
412 clEnqueueUnmapMemObject(mInternals->command_queue[0], mInternals->mem_host, mHostMemoryBase, 0, nullptr, nullptr);
413 for (int32_t i = 0; i < mNStreams; i++) {
414 clReleaseCommandQueue(mInternals->command_queue[i]);
415 }
416 clReleaseMemObject(mInternals->mem_host);
417 }
418
419 clReleaseProgram(mInternals->program);
420 clReleaseContext(mInternals->context);
421 GPUInfo("OPENCL Uninitialized");
422 }
423 mDeviceMemoryBase = nullptr;
424 mHostMemoryBase = nullptr;
425
426 return (0);
427}

429size_t GPUReconstructionOCLBackend::GPUMemCpy(void* dst, const void* src, size_t size, int32_t stream, int32_t toGPU, deviceEvent* ev, deviceEvent* evList, int32_t nEvents)
430{
431 if (evList == nullptr) {
432 nEvents = 0;
433 }
434 if (mProcessingSettings.debugLevel >= 3) {
435 stream = -1;
436 }
437 if (stream == -1) {
438 SynchronizeGPU();
439 }
440 if (toGPU == -2) {
441 GPUFailedMsg(clEnqueueCopyBuffer(mInternals->command_queue[stream == -1 ? 0 : stream], mInternals->mem_gpu, mInternals->mem_gpu, (char*)src - (char*)mDeviceMemoryBase, (char*)dst - (char*)mDeviceMemoryBase, size, nEvents, evList->getEventList<cl_event>(), ev->getEventList<cl_event>()));
442 } else if (toGPU) {
443 GPUFailedMsg(clEnqueueWriteBuffer(mInternals->command_queue[stream == -1 ? 0 : stream], mInternals->mem_gpu, stream == -1, (char*)dst - (char*)mDeviceMemoryBase, size, src, nEvents, evList->getEventList<cl_event>(), ev->getEventList<cl_event>()));
444 } else {
445 GPUFailedMsg(clEnqueueReadBuffer(mInternals->command_queue[stream == -1 ? 0 : stream], mInternals->mem_gpu, stream == -1, (char*)src - (char*)mDeviceMemoryBase, size, dst, nEvents, evList->getEventList<cl_event>(), ev->getEventList<cl_event>()));
446 }
447 if (mProcessingSettings.serializeGPU & 2) {
448 GPUDebug(("GPUMemCpy " + std::to_string(toGPU)).c_str(), stream, true);
449 }
450 return size;
451}

469int32_t GPUReconstructionOCLBackend::DoStuckProtection(int32_t stream, deviceEvent event)
470{
471 if (mProcessingSettings.stuckProtection) {
472 cl_int tmp = 0;
473 for (int32_t i = 0; i <= mProcessingSettings.stuckProtection / 50; i++) {
474 usleep(50);
475 clGetEventInfo(event.get<cl_event>(), CL_EVENT_COMMAND_EXECUTION_STATUS, sizeof(tmp), &tmp, nullptr);
476 if (tmp == CL_COMPLETE) {
477 break;
478 }
479 }
480 if (tmp != CL_COMPLETE) {
481 mGPUStuck = 1;
482 quit("GPU Stuck, future processing in this component is disabled, skipping event (GPU Event State %d)", (int32_t)tmp);
483 }
484 } else {
485 clFinish(mInternals->command_queue[stream]);
486 }
487 return 0;
488}

520int32_t GPUReconstructionOCLBackend::GPUDebug(const char* state, int32_t stream, bool force)
521{
522 // Wait for OPENCL-Kernel to finish and check for OPENCL errors afterwards, in case of debugmode
523 if (!force && mProcessingSettings.debugLevel <= 0) {
