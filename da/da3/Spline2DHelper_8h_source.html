151 template <typename DataT>
152 int32_t Spline2DHelper<DataT>::setSpline(
153  const Spline2DContainer<DataT>& spline, int32_t nAuxiliaryPointsU, int32_t nAuxiliaryPointsV)
154 {
155  // Prepare creation of 2D irregular spline
156  // The should be at least one (better, two) Auxiliary measurements on each segnment between two knots and at least 2*nKnots measurements in total
157  // Returns 0 when the spline can not be constructed with the given nAuxiliaryPoints
158 
159  int32_t ret = 0;
160  mFdimensions = spline.getYdimensions();
161  if (mHelperU1.setSpline(spline.getGridX1(), mFdimensions, nAuxiliaryPointsU) != 0) {
162  ret = storeError(-2, "Spline2DHelper::setSpline2D: error by setting U axis");
163  }
164  if (mHelperU2.setSpline(spline.getGridX2(), mFdimensions, nAuxiliaryPointsV) != 0) {
165  ret = storeError(-3, "Spline2DHelper::setSpline2D: error by setting V axis");
166  }
167  return ret;
168 }
