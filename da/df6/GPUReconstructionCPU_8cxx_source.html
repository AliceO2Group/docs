55template <class T, int32_t I, typename... Args>
56inline void GPUReconstructionCPU::runKernelBackend(const krnlSetupTime& _xyz, const Args&... args)
57{
58 auto& x = _xyz.x;
59 auto& y = _xyz.y;
60 if (x.device == krnlDeviceType::Device) {
61 throw std::runtime_error("Cannot run device kernel on host");
62 }
63 if (x.nThreads != 1) {
64 throw std::runtime_error("Cannot run device kernel on host with nThreads != 1");
65 }
66 int32_t nThreads = getNKernelHostThreads(false);
67 if (nThreads > 1) {
68 if (GetProcessingSettings().debugLevel >= 5) {
69 GPUInfo("Running %d Threads", mThreading->activeThreads->max_concurrency());
70 }
71 tbb::this_task_arena::isolate([&] {
72 mThreading->activeThreads->execute([&] {
73 tbb::parallel_for(tbb::blocked_range<uint32_t>(0, x.nBlocks, 1), [&](const tbb::blocked_range<uint32_t>& r) {
74 typename T::GPUSharedMemory smem;
75 for (uint32_t iB = r.begin(); iB < r.end(); iB++) {
76 T::template Thread<I>(x.nBlocks, 1, iB, 0, smem, T::Processor(*mHostConstantMem)[y.index], args...);
77 }
78 });
79 });
80 });
81 } else {
82 for (uint32_t iB = 0; iB < x.nBlocks; iB++) {
83 typename T::GPUSharedMemory smem;
84 T::template Thread<I>(x.nBlocks, 1, iB, 0, smem, T::Processor(*mHostConstantMem)[y.index], args...);
85 }
86 }
87}

89template <>
90inline void GPUReconstructionCPU::runKernelBackend<GPUMemClean16, 0>(const krnlSetupTime& _xyz, void* const& ptr, uint64_t const& size)
91{
92 int32_t nThreads = std::max<int32_t>(1, std::min<int32_t>(size / (16 * 1024 * 1024), getNKernelHostThreads(true)));
93 if (nThreads > 1) {
94 tbb::parallel_for(0, nThreads, [&](int iThread) {
95 size_t threadSize = size / nThreads;
96 if (threadSize % 4096) {
97 threadSize += 4096 - threadSize % 4096;
98 }
99 size_t offset = threadSize * iThread;
100 size_t mySize = std::min<size_t>(threadSize, size - offset);
101 if (mySize) {
102 memset((char*)ptr + offset, 0, mySize);
103 } // clang-format off
104 }, tbb::static_partitioner()); // clang-format on
105 } else {
106 memset(ptr, 0, size);
107 }
108}

142size_t GPUReconstructionCPU::TransferMemoryResourcesHelper(GPUProcessor* proc, int32_t stream, bool all, bool toGPU)
143{
144 int32_t inc = toGPU ? GPUMemoryResource::MEMORY_INPUT_FLAG : GPUMemoryResource::MEMORY_OUTPUT_FLAG;
145 int32_t exc = toGPU ? GPUMemoryResource::MEMORY_OUTPUT_FLAG : GPUMemoryResource::MEMORY_INPUT_FLAG;
146 size_t n = 0;
147 for (uint32_t i = 0; i < mMemoryResources.size(); i++) {
148 GPUMemoryResource& res = mMemoryResources[i];
149 if (res.mPtr == nullptr) {
150 continue;
151 }
152 if (proc && res.mProcessor != proc) {
153 continue;
154 }
155 if (!(res.mType & GPUMemoryResource::MEMORY_GPU) || (res.mType & GPUMemoryResource::MEMORY_CUSTOM_TRANSFER)) {
156 continue;
157 }
158 if (!GetProcessingSettings().keepAllMemory && !all && (res.mType & exc) && !(res.mType & inc)) {
159 continue;
160 }
161 if (toGPU) {
162 n += TransferMemoryResourceToGPU(&mMemoryResources[i], stream);
163 } else {
164 n += TransferMemoryResourceToHost(&mMemoryResources[i], stream);
165 }
166 }
167 return n;
168}

183{

215int32_t GPUReconstructionCPU::RunChains()
216{
217 mMemoryScalers->temporaryFactor = 1.;
218 if (GetProcessingSettings().memoryScalingFuzz) {
219 static std::mt19937 rng;
220 static std::uniform_int_distribution<uint64_t> dist(0, 1000000);
221 uint64_t fuzzFactor = GetProcessingSettings().memoryScalingFuzz == 1 ? dist(rng) : GetProcessingSettings().memoryScalingFuzz;
222 GPUInfo("Fuzzing memory scaling factor with %lu", fuzzFactor);
223 mMemoryScalers->fuzzScalingFactor(fuzzFactor);
224 }
225
226 mStatNEvents++;
227 mNEventsProcessed++;
228
229 if (GetProcessingSettings().debugLevel >= 3 || GetProcessingSettings().allocDebugLevel) {
230 GPUInfo("Allocated memory when starting processing %34s", "");
231 PrintMemoryOverview();
232 }
233 mTimerTotal.Start();
234 const std::clock_t cpuTimerStart = std::clock();
235 int32_t retVal = 0;
236 if (GetProcessingSettings().doublePipeline) {
237 retVal = EnqueuePipeline();
238 } else {
239 if (mSlaves.size() || mMaster) {
240 WriteConstantParams(); // Reinitialize // TODO: Get this in sync with GPUChainTracking::DoQueuedUpdates, and consider the doublePipeline
241 }
242 for (uint32_t i = 0; i < mChains.size(); i++) {
243 retVal = mChains[i]->RunChain();
244 }
245 }
246 if (retVal != 0 && retVal != 2) {
247 return retVal;
248 }
249 mTimerTotal.Stop();
250 if (GetProcessingSettings().tpcFreeAllocatedMemoryAfterProcessing) {
251 ClearAllocatedMemory();
252 }
253 mStatCPUTime += (double)(std::clock() - cpuTimerStart) / CLOCKS_PER_SEC;
254 if (GetProcessingSettings().debugLevel >= 3 || GetProcessingSettings().allocDebugLevel) {
255 GPUInfo("Allocated memory when ending processing %36s", "");
256 PrintMemoryOverview();
257 }
258
259 mStatWallTime = (mTimerTotal.GetElapsedTime() * 1000000. / mStatNEvents);
260 std::string nEventReport;
261 if (GetProcessingSettings().debugLevel >= 0 && mStatNEvents > 1) {
262 nEventReport += " (avergage of " + std::to_string(mStatNEvents) + " runs)";
263 }
264 double kernelTotal = 0;
265 std::vector<double> kernelStepTimes(gpudatatypes::N_RECO_STEPS, 0.);
266
267 if (GetProcessingSettings().debugLevel >= 1) {
268 for (uint32_t i = 0; i < mTimers.size(); i++) {
269 double time = 0;
270 if (mTimers[i] == nullptr) {
271 continue;
272 }
273 for (int32_t j = 0; j < mTimers[i]->num; j++) {
274 HighResTimer& timer = mTimers[i]->timer[j];
275 time += timer.GetElapsedTime();
276 if (GetProcessingSettings().resetTimers) {
277 timer.Reset();
278 }
279 }
280
281 uint32_t type = mTimers[i]->type;
282 if (type == 0) {
283 kernelTotal += time;
284 int32_t stepNum = getRecoStepNum(mTimers[i]->step);
285 kernelStepTimes[stepNum] += time;
286 }
287 char bandwidth[256] = "";
288 if (mTimers[i]->memSize && mStatNEvents && time != 0.) {
289 snprintf(bandwidth, 256, " (%8.3f GB/s - %'14zu bytes - %'14zu per call)", mTimers[i]->memSize / time * 1e-9, mTimers[i]->memSize / mStatNEvents, mTimers[i]->memSize / mStatNEvents / mTimers[i]->count);
290 }
291 printf("Execution Time: Task (%c %8ux): %50s Time: %'10.0f us%s\n", type == 0 ? 'K' : 'C', mTimers[i]->count, mTimers[i]->name.c_str(), time * 1000000 / mStatNEvents, bandwidth);
292 if (GetProcessingSettings().resetTimers) {
293 mTimers[i]->count = 0;
294 mTimers[i]->memSize = 0;
295 }
296 }
297 }
298 if (GetProcessingSettings().recoTaskTiming) {
299 for (int32_t i = 0; i < gpudatatypes::N_RECO_STEPS; i++) {
300 if (kernelStepTimes[i] != 0. || mTimersRecoSteps[i].timerTotal.GetElapsedTime() != 0.) {
301 printf("Execution Time: Step : %11s %38s Time: %'10.0f us %64s ( Total Time : %'14.0f us, CPU Time : %'14.0f us, %'7.2fx )\n", "Tasks",
302 gpudatatypes::RECO_STEP_NAMES[i], kernelStepTimes[i] * 1000000 / mStatNEvents, "", mTimersRecoSteps[i].timerTotal.GetElapsedTime() * 1000000 / mStatNEvents, mTimersRecoSteps[i].timerCPU * 1000000 / mStatNEvents, mTimersRecoSteps[i].timerCPU / mTimersRecoSteps[i].timerTotal.GetElapsedTime());
303 }
304 if (mTimersRecoSteps[i].bytesToGPU) {
305 printf("Execution Time: Step (D %8ux): %11s %38s Time: %'10.0f us (%8.3f GB/s - %'14zu bytes - %'14zu per call)\n", mTimersRecoSteps[i].countToGPU, "DMA to GPU", gpudatatypes::RECO_STEP_NAMES[i], mTimersRecoSteps[i].timerToGPU.GetElapsedTime() * 1000000 / mStatNEvents,
306 mTimersRecoSteps[i].bytesToGPU / mTimersRecoSteps[i].timerToGPU.GetElapsedTime() * 1e-9, mTimersRecoSteps[i].bytesToGPU / mStatNEvents, mTimersRecoSteps[i].bytesToGPU / mTimersRecoSteps[i].countToGPU);
307 }
308 if (mTimersRecoSteps[i].bytesToHost) {
309 printf("Execution Time: Step (D %8ux): %11s %38s Time: %'10.0f us (%8.3f GB/s - %'14zu bytes - %'14zu per call)\n", mTimersRecoSteps[i].countToHost, "DMA to Host", gpudatatypes::RECO_STEP_NAMES[i], mTimersRecoSteps[i].timerToHost.GetElapsedTime() * 1000000 / mStatNEvents,
310 mTimersRecoSteps[i].bytesToHost / mTimersRecoSteps[i].timerToHost.GetElapsedTime() * 1e-9, mTimersRecoSteps[i].bytesToHost / mStatNEvents, mTimersRecoSteps[i].bytesToHost / mTimersRecoSteps[i].countToHost);
311 }
312 if (GetProcessingSettings().resetTimers) {
313 mTimersRecoSteps[i].bytesToGPU = mTimersRecoSteps[i].bytesToHost = 0;
314 mTimersRecoSteps[i].timerToGPU.Reset();
315 mTimersRecoSteps[i].timerToHost.Reset();
316 mTimersRecoSteps[i].timerTotal.Reset();
317 mTimersRecoSteps[i].timerCPU = 0;
318 mTimersRecoSteps[i].countToGPU = 0;
319 mTimersRecoSteps[i].countToHost = 0;
320 }
321 }
322 for (int32_t i = 0; i < gpudatatypes::N_GENERAL_STEPS; i++) {
323 if (mTimersGeneralSteps[i].GetElapsedTime() != 0.) {
324 printf("Execution Time: General Step : %50s Time: %'10.0f us\n", gpudatatypes::GENERAL_STEP_NAMES[i], mTimersGeneralSteps[i].GetElapsedTime() * 1000000 / mStatNEvents);
325 }
326 }
327 if (GetProcessingSettings().debugLevel >= 1) {
328 mStatKernelTime = kernelTotal * 1000000 / mStatNEvents;
329 printf("Execution Time: Total : %50s Time: %'10.0f us%s\n", "Total Kernel", mStatKernelTime, nEventReport.c_str());
330 }
331 printf("Execution Time: Total : %50s Time: %'10.0f us ( CPU Time : %'10.0f us, %7.2fx ) %s\n", "Total Wall", mStatWallTime, mStatCPUTime * 1000000 / mStatNEvents, mStatCPUTime / mTimerTotal.GetElapsedTime(), nEventReport.c_str());
332 } else if (GetProcessingSettings().debugLevel >= 0) {
333 GPUInfo("Total Wall Time: %10.0f us%s", mStatWallTime, nEventReport.c_str());
334 }
335 if (GetProcessingSettings().resetTimers) {
336 mStatNEvents = 0;
337 mStatCPUTime = 0;
338 mTimerTotal.Reset();
339 }
340
341 if (GetProcessingSettings().memoryStat) {
342 PrintMemoryStatistics();
343 } else if (GetProcessingSettings().debugLevel >= 2) {
344 PrintMemoryOverview();
345 }
346
347 return retVal;
348}
