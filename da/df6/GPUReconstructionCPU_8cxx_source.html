63 template <class T, int32_t I, typename... Args>
64 inline int32_t GPUReconstructionCPUBackend::runKernelBackendInternal(const krnlSetupTime& _xyz, const Args&... args)
65 {
66  auto& x = _xyz.x;
67  auto& y = _xyz.y;
68  if (x.device == krnlDeviceType::Device) {
69  throw std::runtime_error("Cannot run device kernel on host");
70  }
71  if (x.nThreads != 1) {
72  throw std::runtime_error("Cannot run device kernel on host with nThreads != 1");
73  }
74  uint32_t num = y.num == 0 || y.num == -1 ? 1 : y.num;
75  for (uint32_t k = 0; k < num; k++) {
76  int32_t ompThreads = 0;
77  if (mProcessingSettings.ompKernels == 2) {
78  ompThreads = mProcessingSettings.ompThreads / mNestedLoopOmpFactor;
79  if ((uint32_t)getOMPThreadNum() < mProcessingSettings.ompThreads % mNestedLoopOmpFactor) {
80  ompThreads++;
81  }
82  ompThreads = std::max(1, ompThreads);
83  } else {
84  ompThreads = mProcessingSettings.ompKernels ? mProcessingSettings.ompThreads : 1;
85  }
86  if (ompThreads > 1) {
87  if (mProcessingSettings.debugLevel >= 5) {
88  printf("Running %d ompThreads\n", ompThreads);
89  }
90  GPUCA_OPENMP(parallel for num_threads(ompThreads))
91  for (uint32_t iB = 0; iB < x.nBlocks; iB++) {
92  typename T::GPUSharedMemory smem;
93  T::template Thread<I>(x.nBlocks, 1, iB, 0, smem, T::Processor(*mHostConstantMem)[y.start + k], args...);
94  }
95  } else {
96  for (uint32_t iB = 0; iB < x.nBlocks; iB++) {
97  typename T::GPUSharedMemory smem;
98  T::template Thread<I>(x.nBlocks, 1, iB, 0, smem, T::Processor(*mHostConstantMem)[y.start + k], args...);
99  }
100  }
101  }
102  return 0;
103 }

139 size_t GPUReconstructionCPU::TransferMemoryResourcesHelper(GPUProcessor* proc, int32_t stream, bool all, bool toGPU)
140 {
141  int32_t inc = toGPU ? GPUMemoryResource::MEMORY_INPUT_FLAG : GPUMemoryResource::MEMORY_OUTPUT_FLAG;
142  int32_t exc = toGPU ? GPUMemoryResource::MEMORY_OUTPUT_FLAG : GPUMemoryResource::MEMORY_INPUT_FLAG;
143  size_t n = 0;
144  for (uint32_t i = 0; i < mMemoryResources.size(); i++) {
145  GPUMemoryResource& res = mMemoryResources[i];
146  if (res.mPtr == nullptr) {
147  continue;
148  }
149  if (proc && res.mProcessor != proc) {
150  continue;
151  }
152  if (!(res.mType & GPUMemoryResource::MEMORY_GPU) || (res.mType & GPUMemoryResource::MEMORY_CUSTOM_TRANSFER)) {
153  continue;
154  }
155  if (!mProcessingSettings.keepAllMemory && !all && (res.mType & exc) && !(res.mType & inc)) {
156  continue;
157  }
158  if (toGPU) {
159  n += TransferMemoryResourceToGPU(&mMemoryResources[i], stream);
160  } else {
161  n += TransferMemoryResourceToHost(&mMemoryResources[i], stream);
162  }
163  }
164  return n;
165 }

211 int32_t GPUReconstructionCPU::RunChains()
212 {
213  mMemoryScalers->temporaryFactor = 1.;
214  mStatNEvents++;
215  mNEventsProcessed++;
216 
217  timerTotal.Start();
218  if (mProcessingSettings.doublePipeline) {
219  int32_t retVal = EnqueuePipeline();
220  if (retVal) {
221  return retVal;
222  }
223  } else {
224  if (mThreadId != GetThread()) {
225  if (mProcessingSettings.debugLevel >= 2) {
226  GPUInfo("Thread changed, migrating context, Previous Thread: %d, New Thread: %d", mThreadId, GetThread());
227  }
228  mThreadId = GetThread();
229  }
230  if (mSlaves.size() || mMaster) {
231  WriteConstantParams(); // Reinitialize // TODO: Get this in sync with GPUChainTracking::DoQueuedUpdates, and consider the doublePipeline
232  }
233  for (uint32_t i = 0; i < mChains.size(); i++) {
234  int32_t retVal = mChains[i]->RunChain();
235  if (retVal) {
236  return retVal;
237  }
238  }
239  }
240  timerTotal.Stop();
241 
242  mStatWallTime = (timerTotal.GetElapsedTime() * 1000000. / mStatNEvents);
243  std::string nEventReport;
244  if (GetProcessingSettings().debugLevel >= 0 && mStatNEvents > 1) {
245  nEventReport += " (avergage of " + std::to_string(mStatNEvents) + " runs)";
246  }
247  if (GetProcessingSettings().debugLevel >= 1) {
248  double kernelTotal = 0;
249  std::vector<double> kernelStepTimes(GPUDataTypes::N_RECO_STEPS);
250 
251  for (uint32_t i = 0; i < mTimers.size(); i++) {
252  double time = 0;
253  if (mTimers[i] == nullptr) {
254  continue;
255  }
256  for (int32_t j = 0; j < mTimers[i]->num; j++) {
257  HighResTimer& timer = mTimers[i]->timer[j];
258  time += timer.GetElapsedTime();
259  if (mProcessingSettings.resetTimers) {
260  timer.Reset();
261  }
262  }
263 
264  uint32_t type = mTimers[i]->type;
265  if (type == 0) {
266  kernelTotal += time;
267  int32_t stepNum = getRecoStepNum(mTimers[i]->step);
268  kernelStepTimes[stepNum] += time;
269  }
270  char bandwidth[256] = "";
271  if (mTimers[i]->memSize && mStatNEvents && time != 0.) {
272  snprintf(bandwidth, 256, " (%8.3f GB/s - %'14zu bytes - %'14zu per call)", mTimers[i]->memSize / time * 1e-9, mTimers[i]->memSize / mStatNEvents, mTimers[i]->memSize / mStatNEvents / mTimers[i]->count);
273  }
274  printf("Execution Time: Task (%c %8ux): %50s Time: %'10.0f us%s\n", type == 0 ? 'K' : 'C', mTimers[i]->count, mTimers[i]->name.c_str(), time * 1000000 / mStatNEvents, bandwidth);
275  if (mProcessingSettings.resetTimers) {
276  mTimers[i]->count = 0;
277  mTimers[i]->memSize = 0;
278  }
279  }
280  for (int32_t i = 0; i < GPUDataTypes::N_RECO_STEPS; i++) {
281  if (kernelStepTimes[i] != 0. || mTimersRecoSteps[i].timerTotal.GetElapsedTime() != 0.) {
282  printf("Execution Time: Step : %11s %38s Time: %'10.0f us %64s ( Total Time : %'14.0f us)\n", "Tasks", GPUDataTypes::RECO_STEP_NAMES[i], kernelStepTimes[i] * 1000000 / mStatNEvents, "", mTimersRecoSteps[i].timerTotal.GetElapsedTime() * 1000000 / mStatNEvents);
283  }
284  if (mTimersRecoSteps[i].bytesToGPU) {
285  printf("Execution Time: Step (D %8ux): %11s %38s Time: %'10.0f us (%8.3f GB/s - %'14zu bytes - %'14zu per call)\n", mTimersRecoSteps[i].countToGPU, "DMA to GPU", GPUDataTypes::RECO_STEP_NAMES[i], mTimersRecoSteps[i].timerToGPU.GetElapsedTime() * 1000000 / mStatNEvents,
286  mTimersRecoSteps[i].bytesToGPU / mTimersRecoSteps[i].timerToGPU.GetElapsedTime() * 1e-9, mTimersRecoSteps[i].bytesToGPU / mStatNEvents, mTimersRecoSteps[i].bytesToGPU / mTimersRecoSteps[i].countToGPU);
287  }
288  if (mTimersRecoSteps[i].bytesToHost) {
289  printf("Execution Time: Step (D %8ux): %11s %38s Time: %'10.0f us (%8.3f GB/s - %'14zu bytes - %'14zu per call)\n", mTimersRecoSteps[i].countToHost, "DMA to Host", GPUDataTypes::RECO_STEP_NAMES[i], mTimersRecoSteps[i].timerToHost.GetElapsedTime() * 1000000 / mStatNEvents,
290  mTimersRecoSteps[i].bytesToHost / mTimersRecoSteps[i].timerToHost.GetElapsedTime() * 1e-9, mTimersRecoSteps[i].bytesToHost / mStatNEvents, mTimersRecoSteps[i].bytesToHost / mTimersRecoSteps[i].countToHost);
291  }
292  if (mProcessingSettings.resetTimers) {
293  mTimersRecoSteps[i].bytesToGPU = mTimersRecoSteps[i].bytesToHost = 0;
294  mTimersRecoSteps[i].timerToGPU.Reset();
295  mTimersRecoSteps[i].timerToHost.Reset();
296  mTimersRecoSteps[i].timerTotal.Reset();
297  mTimersRecoSteps[i].countToGPU = 0;
298  mTimersRecoSteps[i].countToHost = 0;
299  }
300  }
301  for (int32_t i = 0; i < GPUDataTypes::N_GENERAL_STEPS; i++) {
302  if (mTimersGeneralSteps[i].GetElapsedTime() != 0.) {
303  printf("Execution Time: General Step : %50s Time: %'10.0f us\n", GPUDataTypes::GENERAL_STEP_NAMES[i], mTimersGeneralSteps[i].GetElapsedTime() * 1000000 / mStatNEvents);
304  }
305  }
306  mStatKernelTime = kernelTotal * 1000000 / mStatNEvents;
307  printf("Execution Time: Total : %50s Time: %'10.0f us%s\n", "Total Kernel", mStatKernelTime, nEventReport.c_str());
308  printf("Execution Time: Total : %50s Time: %'10.0f us%s\n", "Total Wall", mStatWallTime, nEventReport.c_str());
309  } else if (GetProcessingSettings().debugLevel >= 0) {
310  GPUInfo("Total Wall Time: %lu us%s", (uint64_t)mStatWallTime, nEventReport.c_str());
311  }
312  if (mProcessingSettings.resetTimers) {
313  mStatNEvents = 0;
314  timerTotal.Reset();
315  }
316 
317  return 0;
318 }
