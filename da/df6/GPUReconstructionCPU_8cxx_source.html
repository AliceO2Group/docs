54template <class T, int32_t I, typename... Args>
55inline void GPUReconstructionCPU::runKernelBackend(const krnlSetupTime& _xyz, const Args&... args)
56{
57 auto& x = _xyz.x;
58 auto& y = _xyz.y;
59 if (x.device == krnlDeviceType::Device) {
60 throw std::runtime_error("Cannot run device kernel on host");
61 }
62 if (x.nThreads != 1) {
63 throw std::runtime_error("Cannot run device kernel on host with nThreads != 1");
64 }
65 int32_t nThreads = getNKernelHostThreads(false);
66 if (nThreads > 1) {
67 if (GetProcessingSettings().debugLevel >= 5) {
68 GPUInfo("Running %d Threads", mThreading->activeThreads->max_concurrency());
69 }
70 tbb::this_task_arena::isolate([&] {
71 mThreading->activeThreads->execute([&] {
72 tbb::parallel_for(tbb::blocked_range<uint32_t>(0, x.nBlocks, 1), [&](const tbb::blocked_range<uint32_t>& r) {
73 typename T::GPUSharedMemory smem;
74 for (uint32_t iB = r.begin(); iB < r.end(); iB++) {
75 T::template Thread<I>(x.nBlocks, 1, iB, 0, smem, T::Processor(*mHostConstantMem)[y.index], args...);
76 }
77 });
78 });
79 });
80 } else {
81 for (uint32_t iB = 0; iB < x.nBlocks; iB++) {
82 typename T::GPUSharedMemory smem;
83 T::template Thread<I>(x.nBlocks, 1, iB, 0, smem, T::Processor(*mHostConstantMem)[y.index], args...);
84 }
85 }
86}

88template <>
89inline void GPUReconstructionCPU::runKernelBackend<GPUMemClean16, 0>(const krnlSetupTime& _xyz, void* const& ptr, uint64_t const& size)
90{
91 int32_t nThreads = std::max<int32_t>(1, std::min<int32_t>(size / (16 * 1024 * 1024), getNKernelHostThreads(true)));
92 if (nThreads > 1) {
93 tbb::parallel_for(0, nThreads, [&](int iThread) {
94 size_t threadSize = size / nThreads;
95 if (threadSize % 4096) {
96 threadSize += 4096 - threadSize % 4096;
97 }
98 size_t offset = threadSize * iThread;
99 size_t mySize = std::min<size_t>(threadSize, size - offset);
100 if (mySize) {
101 memset((char*)ptr + offset, 0, mySize);
102 } // clang-format off
103 }, tbb::static_partitioner()); // clang-format on
104 } else {
105 memset(ptr, 0, size);
106 }
107}

141size_t GPUReconstructionCPU::TransferMemoryResourcesHelper(GPUProcessor* proc, int32_t stream, bool all, bool toGPU)
142{
143 int32_t inc = toGPU ? GPUMemoryResource::MEMORY_INPUT_FLAG : GPUMemoryResource::MEMORY_OUTPUT_FLAG;
144 int32_t exc = toGPU ? GPUMemoryResource::MEMORY_OUTPUT_FLAG : GPUMemoryResource::MEMORY_INPUT_FLAG;
145 size_t n = 0;
146 for (uint32_t i = 0; i < mMemoryResources.size(); i++) {
147 GPUMemoryResource& res = mMemoryResources[i];
148 if (res.mPtr == nullptr) {
149 continue;
150 }
151 if (proc && res.mProcessor != proc) {
152 continue;
153 }
154 if (!(res.mType & GPUMemoryResource::MEMORY_GPU) || (res.mType & GPUMemoryResource::MEMORY_CUSTOM_TRANSFER)) {
155 continue;
156 }
157 if (!GetProcessingSettings().keepAllMemory && !all && (res.mType & exc) && !(res.mType & inc)) {
158 continue;
159 }
160 if (toGPU) {
161 n += TransferMemoryResourceToGPU(&mMemoryResources[i], stream);
162 } else {
163 n += TransferMemoryResourceToHost(&mMemoryResources[i], stream);
164 }
165 }
166 return n;
167}

214int32_t GPUReconstructionCPU::RunChains()
215{
216 mMemoryScalers->temporaryFactor = 1.;
217 if (GetProcessingSettings().memoryScalingFuzz) {
218 static std::mt19937 rng;
219 static std::uniform_int_distribution<uint64_t> dist(0, 1000000);
220 uint64_t fuzzFactor = GetProcessingSettings().memoryScalingFuzz == 1 ? dist(rng) : GetProcessingSettings().memoryScalingFuzz;
221 GPUInfo("Fuzzing memory scaling factor with %lu", fuzzFactor);
222 mMemoryScalers->fuzzScalingFactor(fuzzFactor);
223 }
224
225 mStatNEvents++;
226 mNEventsProcessed++;
227
228 if (GetProcessingSettings().debugLevel >= 3 || GetProcessingSettings().allocDebugLevel) {
229 GPUInfo("Allocated memory when starting processing %34s", "");
230 PrintMemoryOverview();
231 }
232 mTimerTotal.Start();
233 const std::clock_t cpuTimerStart = std::clock();
234 int32_t retVal = 0;
235 if (GetProcessingSettings().doublePipeline) {
236 retVal = EnqueuePipeline();
237 } else {
238 if (mSlaves.size() || mMaster) {
239 WriteConstantParams(); // Reinitialize // TODO: Get this in sync with GPUChainTracking::DoQueuedUpdates, and consider the doublePipeline
240 }
241 for (uint32_t i = 0; i < mChains.size(); i++) {
242 retVal = mChains[i]->RunChain();
243 }
244 }
245 if (retVal != 0 && retVal != 2) {
246 return retVal;
247 }
248 mTimerTotal.Stop();
249 if (GetProcessingSettings().tpcFreeAllocatedMemoryAfterProcessing) {
250 ClearAllocatedMemory();
251 }
252 mStatCPUTime += (double)(std::clock() - cpuTimerStart) / CLOCKS_PER_SEC;
253 if (GetProcessingSettings().debugLevel >= 3 || GetProcessingSettings().allocDebugLevel) {
254 GPUInfo("Allocated memory when ending processing %36s", "");
255 PrintMemoryOverview();
256 }
257
258 mStatWallTime = (mTimerTotal.GetElapsedTime() * 1000000. / mStatNEvents);
259 std::string nEventReport;
260 if (GetProcessingSettings().debugLevel >= 0 && mStatNEvents > 1) {
261 nEventReport += " (avergage of " + std::to_string(mStatNEvents) + " runs)";
262 }
263 double kernelTotal = 0;
264 std::vector<double> kernelStepTimes(GPUDataTypes::N_RECO_STEPS, 0.);
265
266 if (GetProcessingSettings().debugLevel >= 1) {
267 for (uint32_t i = 0; i < mTimers.size(); i++) {
268 double time = 0;
269 if (mTimers[i] == nullptr) {
270 continue;
271 }
272 for (int32_t j = 0; j < mTimers[i]->num; j++) {
273 HighResTimer& timer = mTimers[i]->timer[j];
274 time += timer.GetElapsedTime();
275 if (GetProcessingSettings().resetTimers) {
276 timer.Reset();
277 }
278 }
279
280 uint32_t type = mTimers[i]->type;
281 if (type == 0) {
282 kernelTotal += time;
283 int32_t stepNum = getRecoStepNum(mTimers[i]->step);
284 kernelStepTimes[stepNum] += time;
285 }
286 char bandwidth[256] = "";
287 if (mTimers[i]->memSize && mStatNEvents && time != 0.) {
288 snprintf(bandwidth, 256, " (%8.3f GB/s - %'14zu bytes - %'14zu per call)", mTimers[i]->memSize / time * 1e-9, mTimers[i]->memSize / mStatNEvents, mTimers[i]->memSize / mStatNEvents / mTimers[i]->count);
289 }
290 printf("Execution Time: Task (%c %8ux): %50s Time: %'10.0f us%s\n", type == 0 ? 'K' : 'C', mTimers[i]->count, mTimers[i]->name.c_str(), time * 1000000 / mStatNEvents, bandwidth);
291 if (GetProcessingSettings().resetTimers) {
292 mTimers[i]->count = 0;
293 mTimers[i]->memSize = 0;
294 }
295 }
296 }
297 if (GetProcessingSettings().recoTaskTiming) {
298 for (int32_t i = 0; i < GPUDataTypes::N_RECO_STEPS; i++) {
299 if (kernelStepTimes[i] != 0. || mTimersRecoSteps[i].timerTotal.GetElapsedTime() != 0.) {
300 printf("Execution Time: Step : %11s %38s Time: %'10.0f us %64s ( Total Time : %'14.0f us, CPU Time : %'14.0f us, %'7.2fx )\n", "Tasks",
301 GPUDataTypes::RECO_STEP_NAMES[i], kernelStepTimes[i] * 1000000 / mStatNEvents, "", mTimersRecoSteps[i].timerTotal.GetElapsedTime() * 1000000 / mStatNEvents, mTimersRecoSteps[i].timerCPU * 1000000 / mStatNEvents, mTimersRecoSteps[i].timerCPU / mTimersRecoSteps[i].timerTotal.GetElapsedTime());
302 }
303 if (mTimersRecoSteps[i].bytesToGPU) {
304 printf("Execution Time: Step (D %8ux): %11s %38s Time: %'10.0f us (%8.3f GB/s - %'14zu bytes - %'14zu per call)\n", mTimersRecoSteps[i].countToGPU, "DMA to GPU", GPUDataTypes::RECO_STEP_NAMES[i], mTimersRecoSteps[i].timerToGPU.GetElapsedTime() * 1000000 / mStatNEvents,
305 mTimersRecoSteps[i].bytesToGPU / mTimersRecoSteps[i].timerToGPU.GetElapsedTime() * 1e-9, mTimersRecoSteps[i].bytesToGPU / mStatNEvents, mTimersRecoSteps[i].bytesToGPU / mTimersRecoSteps[i].countToGPU);
306 }
307 if (mTimersRecoSteps[i].bytesToHost) {
308 printf("Execution Time: Step (D %8ux): %11s %38s Time: %'10.0f us (%8.3f GB/s - %'14zu bytes - %'14zu per call)\n", mTimersRecoSteps[i].countToHost, "DMA to Host", GPUDataTypes::RECO_STEP_NAMES[i], mTimersRecoSteps[i].timerToHost.GetElapsedTime() * 1000000 / mStatNEvents,
309 mTimersRecoSteps[i].bytesToHost / mTimersRecoSteps[i].timerToHost.GetElapsedTime() * 1e-9, mTimersRecoSteps[i].bytesToHost / mStatNEvents, mTimersRecoSteps[i].bytesToHost / mTimersRecoSteps[i].countToHost);
310 }
311 if (GetProcessingSettings().resetTimers) {
312 mTimersRecoSteps[i].bytesToGPU = mTimersRecoSteps[i].bytesToHost = 0;
313 mTimersRecoSteps[i].timerToGPU.Reset();
314 mTimersRecoSteps[i].timerToHost.Reset();
315 mTimersRecoSteps[i].timerTotal.Reset();
316 mTimersRecoSteps[i].timerCPU = 0;
317 mTimersRecoSteps[i].countToGPU = 0;
318 mTimersRecoSteps[i].countToHost = 0;
319 }
320 }
321 for (int32_t i = 0; i < GPUDataTypes::N_GENERAL_STEPS; i++) {
322 if (mTimersGeneralSteps[i].GetElapsedTime() != 0.) {
323 printf("Execution Time: General Step : %50s Time: %'10.0f us\n", GPUDataTypes::GENERAL_STEP_NAMES[i], mTimersGeneralSteps[i].GetElapsedTime() * 1000000 / mStatNEvents);
324 }
325 }
326 if (GetProcessingSettings().debugLevel >= 1) {
327 mStatKernelTime = kernelTotal * 1000000 / mStatNEvents;
328 printf("Execution Time: Total : %50s Time: %'10.0f us%s\n", "Total Kernel", mStatKernelTime, nEventReport.c_str());
329 }
330 printf("Execution Time: Total : %50s Time: %'10.0f us ( CPU Time : %'10.0f us, %7.2fx ) %s\n", "Total Wall", mStatWallTime, mStatCPUTime * 1000000 / mStatNEvents, mStatCPUTime / mTimerTotal.GetElapsedTime(), nEventReport.c_str());
331 } else if (GetProcessingSettings().debugLevel >= 0) {
332 GPUInfo("Total Wall Time: %10.0f us%s", mStatWallTime, nEventReport.c_str());
333 }
334 if (GetProcessingSettings().resetTimers) {
335 mStatNEvents = 0;
336 mStatCPUTime = 0;
337 mTimerTotal.Reset();
338 }
339
340 if (GetProcessingSettings().memoryStat) {
341 PrintMemoryStatistics();
342 } else if (GetProcessingSettings().debugLevel >= 2) {
343 PrintMemoryOverview();
344 }
345
346 return retVal;
347}
