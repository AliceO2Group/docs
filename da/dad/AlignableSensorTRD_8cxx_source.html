56 void AlignableSensorTRD::dPosTraDParCalib(const AlignmentPoint* pnt, double* deriv, int calibID, const AlignableVolume* parent) const
57 {
58  // calculate point position X,Y,Z derivatives wrt calibration parameter calibID of given parent
59  // parent=0 means top detector object calibration
60  //
61  deriv[0] = deriv[1] = deriv[2] = 0;
62  //
63  if (!parent) { // TRD detector global calibration
64  //
65  switch (calibID) {
66  case AlignableDetectorTRD::CalibNRCCorrDzDtgl: // correction for Non-Crossing tracklets Z,Y shift: Z -> Z + calib*tgl, Y -> Y + calib*tgl*tilt*sign(tilt);
67  {
68  double sgYZ = pnt->getYZErrTracking()[1]; // makes sense only for nonRC tracklets
69  if (std::abs(sgYZ) > 0.01) {
70  const double kTilt = 2. * TMath::DegToRad();
71  deriv[2] = pnt->getTrParamWSA()[AlignmentPoint::kParTgl];
72  deriv[1] = deriv[2] * Sign(kTilt, sgYZ);
73  }
74  break;
75  }
76  case AlignableDetectorTRD::CalibDVT: // correction for bias in VdriftT
77  {
78  // error in VdriftT equivalent to shift in X at which Y measurement is evaluated
79  // Y -> Y + dVdriftT * tg_phi, where tg_phi is the slope of the track in YX plane
80  double snp = pnt->getTrParamWSA(AlignmentPoint::kParSnp), slpY = snp / std::sqrt((1.f - snp) * (1.f + snp));
81  deriv[1] = slpY;
82  break;
83  }
84  default:
85  break;
86  }
87  }
88 }
